וְהַעֲבַרְתִּ֥י אֶתְכֶ֖ם תַּ֣חַת הַשָּׁ֑בֶט וְהֵבֵאתִ֥י אֶתְכֶ֖ם בְּמָסֹ֥רֶת הַבְּרִֽית׃
וּבָרוֹתִ֣י מִכֶּ֗ם הַמֹּרְדִ֤ים וְהַפּֽוֹשְׁעִים֙ בִּ֔י מֵאֶ֤רֶץ מְגֽוּרֵיהֶם֙ אוֹצִ֣יא אוֹתָ֔ם וְאֶל־אַדְמַ֥ת יִשְׂרָאֵ֖ל לֹ֣א יָב֑וֹא וִידַעְתֶּ֖ם כִּי־אֲנִ֥י יְהוָֽה׃
וְאַתֶּ֨ם בֵּֽית־יִשְׂרָאֵ֜ל כֹּה־אָמַ֣ר׀ אֲדנָ֣י יְהֹוִ֗ה אִ֤ישׁ גִּלּוּלָיו֙ לְכ֣וּ עֲבֹ֔דוּ וְאַחַ֕ר אִם־אֵינְכֶ֖ם שֹׁמְעִ֣ים אֵלָ֑י וְאֶת־שֵׁ֤ם קָדְשִׁי֙ לֹ֣א תְחַלְּלוּ־ע֔וֹד בְּמַתְּנוֹתֵיכֶ֖ם וּבְגִלּוּלֵיכֶֽם׃
כִּ֣י בְהַר־קָדְשִׁ֞י בְּהַ֣ר׀ מְר֣וֹם יִשְׂרָאֵ֗ל נְאֻם֙ אֲדנָ֣י יְהֹוִ֔ה שָׁ֣ם יַעַבְדֻ֜נִי כָּל־בֵּ֧ית יִשְׂרָאֵ֛ל [כֻּלֹּ֖ה] כֻּלּ֖וֹ בָּאָ֑רֶץ שָׁ֣ם אֶרְצֵ֔ם וְשָׁ֞ם אֶדְר֣וֹשׁ אֶת־תְּרוּמֹֽתֵיכֶ֗ם וְאֶת־רֵאשִׁ֛ית מַשְׂאוֹתֵיכֶ֖ם בְּכָל־קָדְשֵׁיכֶֽם׃
בְּרֵ֣יחַ נִיחֹחַ֘ אֶרְצֶ֣ה אֶתְכֶם֒ בְּהוֹצִיאִ֤י אֶתְכֶם֙ מִן־הָ֣עַמִּ֔ים וְקִבַּצְתִּ֣י אֶתְכֶ֔ם מִן־הָ֣אֲרָצ֔וֹת אֲשֶׁ֥ר נְפֹצֹתֶ֖ם בָּ֑ם וְנִקְדַּשְׁתִּ֥י בָכֶ֖ם לְעֵינֵ֥י הַגּוֹיִֽם׃
וִֽידַעְתֶּם֙ כִּֽי־אֲנִ֣י יְהוָ֔ה בַּהֲבִיאִ֥י אֶתְכֶ֖ם אֶל־אַדְמַ֣ת יִשְׂרָאֵ֑ל אֶל־הָאָ֗רֶץ אֲשֶׁ֤ר נָשָׂ֙אתִי֙ אֶת־יָדִ֔י לָתֵ֥ת אוֹתָ֖הּ לַאֲבוֹתֵיכֶֽם׃
וּזְכַרְתֶּם־שָׁ֗ם אֶת־דַּרְכֵיכֶם֙ וְאֵת֙ כָּל־עֲלִיל֣וֹתֵיכֶ֔ם אֲשֶׁ֥ר נִטְמֵאתֶ֖ם בָּ֑ם וּנְקֹֽטֹתֶם֙ בִּפְנֵיכֶ֔ם בְּכָל־רָעוֹתֵיכֶ֖ם אֲשֶׁ֥ר עֲשִׂיתֶֽם׃
וִֽידַעְתֶּם֙ כִּֽי־אֲנִ֣י יְהוָ֔ה בַּעֲשׂוֹתִ֥י אִתְּכֶ֖ם לְמַ֣עַן שְׁמִ֑י לֹא֩ כְדַרְכֵיכֶ֨ם הָרָעִ֜ים וְכַעֲלִילוֹתֵיכֶ֤ם הַנִּשְׁחָתוֹת֙ בֵּ֣ית יִשְׂרָאֵ֔ל נְאֻ֖ם אֲדנָ֥י יְהֹוִֽה׃{פ}
