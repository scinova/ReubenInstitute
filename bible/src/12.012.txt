וְאָֽמַרְתָּ֙ בַּיּ֣וֹם הַה֔וּא אוֹדְךָ֣ יְהוָ֔ה כִּ֥י אָנַ֖פְתָּ בִּ֑י יָשֹׁ֥ב אַפְּךָ֖ וּֽתְנַחֲמֵֽנִי׃
הִנֵּ֨ה אֵ֧ל יְשׁוּעָתִ֛י אֶבְטַ֖ח וְלֹ֣א אֶפְחָ֑ד כִּֽי־עָזִּ֤י וְזִמְרָת֙ יָ֣הּ יְהוָ֔ה וַֽיְהִי־לִ֖י לִישׁוּעָֽה׃
וּשְׁאַבְתֶּם־מַ֖יִם בְּשָׂשׂ֑וֹן מִמַּעַיְנֵ֖י הַיְשׁוּעָֽה׃
וַאֲמַרְתֶּ֞ם בַּיּ֣וֹם הַה֗וּא הוֹד֤וּ לַֽיהוָה֙ קִרְא֣וּ בִשְׁמ֔וֹ הוֹדִ֥יעוּ בָעַמִּ֖ים עֲלִילֹתָ֑יו הַזְכִּ֕ירוּ כִּ֥י נִשְׂגָּ֖ב שְׁמֽוֹ׃
זַמְּר֣וּ יְהוָ֔ה כִּ֥י גֵא֖וּת עָשָׂ֑ה [מיּדַ֥עַת] מוּדַ֥עַת זֹ֖את בְּכָל־הָאָֽרֶץ׃
צַהֲלִ֥י וָרֹ֖נִּי יוֹשֶׁ֣בֶת צִיּ֑וֹן כִּי־גָד֥וֹל בְּקִרְבֵּ֖ךְ קְד֥וֹשׁ יִשְׂרָאֵֽל׃{ס}
