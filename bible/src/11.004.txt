וַיֵּצֵ֨א אֶחָ֣ד אֶל־הַשָּׂדֶה֘ לְלַקֵּ֣ט אֹרֹת֒ וַיִּמְצָא֙ גֶּ֣פֶן שָׂדֶ֔ה וַיְלַקֵּ֥ט מִמֶּ֛נּוּ פַּקֻּעֹ֥ת שָׂדֶ֖ה מְלֹ֣א בִגְד֑וֹ וַיָּבֹ֗א וַיְפַלַּ֛ח אֶל־סִ֥יר הַנָּזִ֖יד כִּי־לֹ֥א יָדָֽעוּ׃
וַיִּֽצְק֥וּ לַאֲנָשִׁ֖ים לֶאֱכ֑וֹל וַ֠יְהִי כְּאָכְלָ֨ם מֵהַנָּזִ֜יד וְהֵ֣מָּה צָעָ֗קוּ וַיֹּֽאמְרוּ֙ מָ֤וֶת בַּסִּיר֙ אִ֣ישׁ הָאֱלֹהִ֔ים וְלֹ֥א יָכְל֖וּ לֶאֱכֹֽל׃
וַיֹּ֙אמֶר֙ וּקְחוּ־קֶ֔מַח וַיַּשְׁלֵ֖ךְ אֶל־הַסִּ֑יר וַיֹּ֗אמֶר צַ֤ק לָעָם֙ וְיֹאכֵ֔לוּ וְלֹ֥א הָיָ֛ה דָּבָ֥ר רָ֖ע בַּסִּֽיר׃{ס} 
וְאִ֨ישׁ בָּ֜א מִבַּ֣עַל שָׁלִ֗שָׁה וַיָּבֵא֩ לְאִ֨ישׁ הָאֱלֹהִ֜ים לֶ֤חֶם בִּכּוּרִים֙ עֶשְׂרִֽים־לֶ֣חֶם שְׂעֹרִ֔ים וְכַרְמֶ֖ל בְּצִקְלֹנ֑וֹ וַיֹּ֕אמֶר תֵּ֥ן לָעָ֖ם וְיֹאכֵֽלוּ׃
וַיֹּ֙אמֶר֙ מְשָׁ֣רְת֔וֹ מָ֚ה אֶתֵּ֣ן זֶ֔ה לִפְנֵ֖י מֵ֣אָה אִ֑ישׁ וַיֹּ֗אמֶר תֵּ֤ן לָעָם֙ וְיֹאכֵ֔לוּ כִּ֣י כֹ֥ה אָמַ֛ר יְהוָ֖ה אָכֹ֥ל וְהוֹתֵֽר׃
וַיִּתֵּ֧ן לִפְנֵיהֶ֛ם וַיֹּאכְל֥וּ וַיּוֹתִ֖רוּ כִּדְבַ֥ר יְהוָֽה׃{פ}
