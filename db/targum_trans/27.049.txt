לְשַׁבָּחָא עַל יְדֵיהוֹן דִבְנֵי קֹרַח תּוּשְׁבַּחְתָּא:
שְׁמַעוּ אַחְוְיָתָא דָא כָל עַמְמַיָא אַצִיתוּ כָּל דְיָרֵי אַרְעָא:
אוּף בְּנֵי אָדָם קַדְמָאָה אוּף בְּנוֹי דְיַעֲקֹב כַּחֲדָא זַכָּאָה וְחַיָבָא:
פּוּמֵי יְמַלֵל חוּכְמְתָא וְרִנְגַת לִבָּא בִינָא:
אַצְלֵי לְמִתְלָא אוּדְנִי אַשְׁרֵי לְמִפְתַּח בְּכִנָרָא אוּחַדְתִּי:
מְטוּל אִדְחַל בְּיוֹם אַסְעָרוּת בִּישָׁא אֱלָהֵן דְחוֹבַת סוּרְחָנִי בְּסוֹפִי יַחְזְרִנַנִי:
לְחַיָבַיָא דִי מִתְרַחֲצִין עַל נִכְסְהוֹן וּבִסְגִיאוּת עָתְרֵיהוֹן מִשְׁתַּבְּחִין:
אֲחוּהִי דְאִשְׁתְּבָא לָא מַפְרֵק יִפְרוֹק גְבַר מִנִכְסוֹי וְלָא יְהַב לֵאלָהָא פוּרְקָנֵיהּ:
וִיהֵי יַקִיר פּוּרְקָנֵיהּ וְיִפְסוֹק בִּישׁוּתֵיהּ וּפוּרְעֲנוּתָא לַעֲלָם:
וִיחֵי תוּב לְחַיֵי עַלְמָא לָא יֶחֱמֵי דִין גֵהִנָם:
אֲרוּם יֶחֱמוּן חַכִּימַיָא רְשִׁיעַיָא בִגְהִנָם יִתְדָנוּן כַּחֲדָא כְּסִילַיָא וְשַׁטְיָא יֵיבְדוּן וְיִשְׁבְּקוּן לְצַדִיקַיָא מָמוֹנְהוֹן:
בְּבֵית קְבוּרְתְּהוֹן יִשְׁרוֹן לַעֲלָם וְלָא יְקוּמוּן מִמַשְׁכְּנֵיהוֹן לְדָרֵי דָרַיָא מְטוּל דְאִתְרַבְרְבוּ וּקְנוֹ שׁוּם בִּישׁ עֲלַוֵי אַרְעָא:
וְגַבְרָא חַיָבָא בִיקָרָא לָא יְבִית עִם צַדִיקַיָא אַמְתִיל הֵיךְ בְּעִירָא אִשְׁתְּוָא לִלְמָא:
דֵין אָרְחַתְהוֹן גְרַם שְׁטוּתָא לְהוֹן וּבְסוֹפְהוֹן בְּפוּמְהוֹן יִתְנוּן סוּרְחָנוּתְהוֹן לְעַלְמָא דְאָתֵי:
הֵיךְ עָנָא צַדִיקַיָא שַׁווּ לְמוֹתָא וּקְטַלוּן וְתַבַּרוּ צַדִיקַיָא וּפָלְחֵי אוֹרַיְתָא וּתְרִיצַיָא אַלְקוּ מְטוּל הֲכִי גוּפֵיהוֹן יְהוֹן בַּלְיָן בִּגְהִנָם מִן בִּגְלַל דְאוֹשִׁיטוּ יְדֵיהוֹן וְחַבִּילוּ בֵּית מְדוֹר שְׁכִנְתֵּיהּ דִילֵיהּ:
אֲמַר דָוִד בְּרוּחַ נְבוּאָה בְּרַם אֱלָהָא יִפְרוֹק נַפְשִׁי מִן דִין גֵהִנָם אֲרוּם יַלְפִנַנִי אוֹרַיְתֵיהּ לְעָלְמִין:
(עַל קֹרַח וְסִיעֲתֵיהּ אִתְנַבֵּי וַאֲמַר) לָא תִדְחַל (מֹשֶׁה) אֲרוּם אִתְעַתַּר (קֹרַח) גַבְרָא (דְמַצוּתָא) אֲרוּם יִסְגֵי אַיְקָר בֵּיתֵיהּ:
אֲרוּם לָא בְמוֹתֵיהּ יִסַב כוֹלָא לָא יֵחוֹת בַּתְרוֹי אַיְקָרֵיהּ:
אֲרוּם נַפְשֵׁיהּ (דְמֹשֶׁה) בְּחַיוֹהִי יְבָרֵךְ לָךְ וְיוֹדוּנָךְ צַדִיקַיָא אֲרוּם תֵּיטִיב לְפַלְחִין קֳדָמָךְ:
תֵּעוּל דוּכְרָנְהוֹן דְצַדִיקַיָא עַד דָר אַבְהָתְהוֹן וְרַשִׁיעַיָא לְעָלְמֵי עָלְמַיָא לָא יֶחֱמוּן נְהוֹרָא:
גַבְרָא חַיָבָא בִּזְמַן דְאִיתוֹהִי בִּיקָרָא לָא יִתְבַּיֵן וּבְאִסְתַּלָקוּת יְקָרֵיהּ מִנֵיהּ אִמְתִיל לִבְעִירָא וְאִשְׁתְּוֵי לִלְמָא:
