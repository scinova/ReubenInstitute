עַל יַד דָוִד תּוּשְׁבַּחְתָּא אִין חִסְדָא אַתְּ גָמֵל עִמִי אִין דִינָא אַתְּ עָבֵד עִמִי עַל כּוֹלָא אֲשַׁבַּח קֳדָמָךְ יְיָ אֲזַמֵר:
אַשְׂכְּלִנָךְ בְּאוֹרַח שְׁלִים אֵימָתַי תֵּעוּל לְוָתִי אֶתְהַלֵךְ בִּשְׁלֵימוּת לִבְּבֵי בְּגוֹ בֵית אוּלְפָנִי:
לָא אֲשַׁוֵי לְלִבִּי מִלְתָא דְרַשִׁיעָא עָבְדֵי בִישִׁין וְשָׂטִין מִן פִּקוּדַיָא סְנֵיתִי לָא יִדְבְּקוּן בִּי:
לִבָּא עוּקְמָנָא יֶעְדֵי מִנִי יִצְרָא בִישָׁא לָא אֵדָע:
דְמִשְׁתָּעֵי לִישָׁן תְּלִיתַי עַל חַבְרֵיהּ יָתֵיהּ אַמַגֵר (וְיִלְקֵי בְּצָרַעְתָּא) דְאָזֵיל בְּגַסוּת עַיְנִין עִמֵיהּ לָא אֲדוּר בְּעַלְמָא:
עֵינַי בְּקַשִׁיטֵי אַרְעָא לְמִתַּב בִּמְחִיצַת צַדִיקֵי דְאָזֵל בְּאוּרְחָא שְׁלִים הוּא יְקוּם עִם שַׁמָשַׁי:
לָא יִתֵב בִּמְצַע בֵּית מוּקְדָשִׁי עָבֵד נְכִילוּ דִי מְמַלֵל שִׁקְרִין לֵית לֵיהּ רְשׁוּ לְאִתְכַּוָנָא קָבֵל עַיְנָי:
לְעָלְמָא דְאָתִי דְאִמְתִיל לִנְהוֹר צַפְרָא אֲמַגֵר כָּל רַשִׁיעֵי אַרְעָא לְשֵׁיצָאָה מִן יְרוּשְׁלֵם קַרְתָּא דַייָ כָּל עָבְדֵי שְׁקָר:
