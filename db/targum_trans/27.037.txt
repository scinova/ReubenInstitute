לְדָוִד לָא תְגָרֵג בְּמִבְאֲשִׁין לְמֶהֱוֵי דִכְוָתְהוֹן וְלָא תַטְנֵין בְּעָבְדֵי טְלוּמָא לְמִזְדַוֵג עִמְהוֹן:
מְטוּל דְסוֹפֵיהוֹן הֵיךְ עִסְבָּא בְּסִרְהוֹבְיָא יִתְמוֹלְלוּ וְהֵיךְ יְרוֹק דִתְאָה יְנַתְּרוּן:
רְחֵיץ בְּמֵימְרָא דַייָ וַעֲבַד טָב שְׁרֵי אַרְעָא וַחֲסֵן בְּהֵימְנוּתָא:
וְתִתְפְּנֵק עַל יְיָ וְיִתֶּן לָךְ מִשְׁיָלֵי לִבָּךְ:
גַלֵי עַל יְיָ אָרְחָתָךְ וּסְבוֹר עַל מֵימְרֵיהּ וְהוּא יַעְבֵּד:
וְיִפּוֹק הֵיךְ נְהוֹרָא צִדְקָךְ וְדִינָךְ הֵיךְ טִיהֲרָא:
שְׁתוֹק קֳדָם יְיָ וְאוֹרִיךְ לֵיהּ לָא תְגָרֵג בְּרַשִׁיעָא דְמַצְלָח אוֹרְחֵיהּ בְּגַבְרָא דְעָבֵיד עֵיצַת חֶטְאִין:
אוֹרֵיךְ מֵרְגֵז וּשְׁבוֹק רִתְחָא לָא תְגָרֵג בְּרַם לְאַבְאָשָׁא:
מְטוּל דְמִבְאֲשִׁין יִשְׁתֵּיצוּן וּדְסָבְרִין בְמֵימְרָא דַייָ הִנוּן יִירְתוּן אַרְעָא:
וְתוּב בִּזְעֵיר וְלֵית רַשִׁיעָא וְתִתְבַּיֵן עַל אַתְרֵיהּ וְלֵיתוֹי:
וְעֶנְוְתָנִין יִרְתוּן אַרְעָא וְיִתְפַּנְקוּן עַל סְגִיעוּת שְׁלָמָא:
חָשֵׁב רַשִׁיעָא בִּישׁ עַל צַדִיקָא וּמְעַסֵי עֲלוֹי כַּכוֹי:
יְיָ יִגְחַךְ עֲלוֹי אֲרוּם חָמָא אֲרוּם עָל יוֹמָא דִתְבִירֵיהּ:
חַרְבָּא שְׁלִיפוּ רַשִׁיעַיָא וּמְתַחוּ קַשְׁוָתְהוֹן לְמִקְטוֹל עֲנִיֵי וַחֲשִׁיכֵי לְמִכּוֹס תְּרִיצֵי אוּרְחָא:
סַיָפְהוֹן תֵּעוֹל בְּלִבְּהוֹן וְקַשְׁוָתְהוֹן יִתַּבְּרוּן:
טָב קֳדָם יְיָ זְעֵרוּת צַדִיקַיָא מֵרִכְפַּת רַשִׁיעִין סַגִיעִין:
אֲרוּם דְרוֹעֵי דְרַשִׁיעַיָא יִתַּבְּרוּן וְתָמִיךְ צַדִיקַיָא מֵימְרָא דַיְיָ:
יְדִיעִין קֳדָם יְיָ יוֹמֵיהוֹן דִשְׁלֵמִין וְאַחְסַנְתֵּהוֹן לַעֲלַם תְּהֵי:
לָא יִבְהֲתוּן בְּעִדָן בִּישְׁתָּא וּבְיוֹמֵי אוּלְצָנָא סָבְעִין:
אֲרוּם רַשִׁיעַיָא יֵיבְדוּן וּבַעֲלֵי דְבָבַיָא דַייָ הֵיךְ יְקַר פַּלְגִיסִין דְמִין אַוְלָא מִתְפַּטְמִין וְסוֹפֵיהוֹן מִתְנַכְסִין הֵיכְנָא רַשִׁיעֵי יְסוּפוּן וּבִתְנָן גֵהִנָם יִגְמְרוּן:
מוֹזִיף רַשִׁיעָא וְלָא פְרַע וְצַדִיק מְחַיֵס וִיהַב:
מְטוּל דְמִתְבָּרְכִין בְּמֵימְרָא יִרְתוּן אַרְעָא וּדְמִתְלָטְטֵין בְּמוֹתָא יִשְׁתְּצוּן:
מִן קֳדָם יְיָ פְּסִיעָתוֹ דְגַבְרָא מְתַקְנָן וְאוֹרְחָתֵיהּ יִתְרְעֵי:
אֲרוּם יִפּוֹל מְרַע לָא יְמוּת מְטוּל דַייָ סָעִיד בִּידֵיהּ:
טָלֵי הֲוֵית בְּרַם סִיבִית וְלָא חֲמֵית צַדִיקָא שְׁבִיקָא וּבְנוֹי תָּבְעִין לַחְמָא מֵחוּסְרָנָא:
מִן בִּגְלָל דְכָל יוֹמָא חָיֵס וּמוֹזִיף וְדַרְעֵיהּ לְבִרְכְתָא:
סְטֵי מִבִּישׁ וַעֲבַד טִיבוּ וּשְׁרֵי לְחַיֵי עָלְמָא:
אֲרוּם יְיָ רְחֵם דִינָא וְלָא יִשְׁבּוֹק יַת חֲסִידוֹי לַעֲלַם מְנַטְרִין וּבְנֵיהוֹן דְרַשִׁיעַיָא יִשְׁתֵּצוּן:
צַדִיקַיָא יַחְסְנוּן אַרְעָא וְיִשְׁרוּן לְעַלְמִין עֲלָהּ:
פּוּמָא דְצַדִיקַיָא רָנֵן חוּכְמְתָא וְלִישָׁנֵיהּ מְמַלְלָא דִינָא:
נִימוּסָא דֶאֱלָהֵיהּ בְּלִבֵּיהּ לָא מִזְדַעְזְעִין אִסְתַּוְרוֹי:
מִסְתְּכֵי רַשִׁיעָא לְצַדִיקָא וּבָעֵי לְמִקְטְלֵיהּ:
יְיָ לָא יִשְׁבְּקִנֵיהּ בִּידֵיהּ וְלָא יְחַיְבִנֵיהּ בְּמִדַיְנֵיהּ:
סְבַר בְּמֵימְרֵיהּ דַייָ וְטַר אָרְחֵיהּ וִירוֹמְמִנָךְ לְמֶחְסַן אַרְעָא בְּאִשְׁתְּצָיוּת רַשִׁיעַיָא תֶחֱמֵי:
חֲמִיתִי רַשִׁיעָא תַקִיפָא וְחָסִין הֵיךְ אִילָן יַצִיב וְעַבּוּף:
וּפְסַק מִן עַלְמָא וְהָא לֵיתוֹי וּבְעֵית יָתֵיהּ וְלָא אִשְׁתְּכָח:
נְטַר שְׁלֵימוּתָא וַחֲמֵי תְרִיצוּתָא אֲרוּם דְסוֹף בַּר נְשָׁא שְׁלָמָא:
וּמְרוֹדִין יִשְׁתֵּיצוּן כַּחֲדָא סוֹפֵיהוֹן דְרַשִׁיעַיָא שֵׁיצְיָא:
וּפוּרְקָנָא דְצַדִיקַיָא מִן קֳדָם יְיָ עוּשְׁנְהוֹן בְּעִדָן עָקְתָא:
וְסַיְעִינוּן יְיָ וְשֵׁזִיב יַתְהוֹן שֵׁזֵיב יַתְהוֹן מִן חַיָבַיָא וְיִפְרְקִנוּן מְטוּל דְאִתְרְחִיצוּ בְּמֵימְרֵיהּ:
