צְלוֹתָא עַל מִסְכְּנָא אֲרוּם מִשְׁתַּלְהֵי וּקְדָם יְיָ יֵימַר צְלוֹתֵיהּ:
יְיָ קַבֵּל צְלוֹתִי וּפְגִינוּתִי לְקָדָמָךְ תֵּעוּל:
לָא תְסַלֵק שְׁכִנְתָּךְ מִנִי בְּיוֹם עָקְתִי אַצְלֵי לְוָתִי אוּדְנָךְ בְּיוֹמָא דְאֶקְרֵי סַרְהֵיב עֲנֵי יָתִי:
אֲרוּם אִשְׁתֵּצִיאוּ הֵיךְ תְּנָנָא יוֹמַי וְאֵבָרַי הֵיךְ תַּפְיָא חַרְרָן:
לְקָא הֵיךְ עִסְבָּא לִבִּי אֲרוּם אִתְנְשֵׁיתִי אוֹרָיַת אוּלְפָנִי:
מִן קָל תִּנְחֲתִי אִדְבַּק גַרְמִי לְבִסְרִי:
אִדְמֵית לִקְאָתָא דְמַדְבְּרָא הֲוֵית הֵיךְ קְפוּפָא דִי בְאַרְעָא נְגוּבְתָּא:
שִׁקְדֵית כָּל לֵילְיָא וַהֲוֵיתִי הֵיךְ צִפַּר דְפָרֵחַ וְנָיֵד בִּלְחוֹדֵיהּ עַל אִגְרָא:
כָּל יוֹמָא יְחַסְדוּן לִי בַעֲלֵי דְבָבַי מְתַלְעֲבֵי בְמֵימְרִי קַיְמוּ מַגָן:
אֲרוּם אַפְרָא הֵיךְ מְזוֹנָא סְעַדֵית וְשִׁקְיוּתִי בִּבְכוּתָא מִזְגֵית:
מִן קֳדָם רוּגְזָךְ וְקִצְפָּךְ אֲרוּם נְטַלְתַּנִי וּטְלַקְתָּנִי:
יוֹמַי הֵיךְ טוּלָא דְיִצְלֵי וַאֲנָא הֵיךְ עִסְבָּא אִיבָשׁ:
וְאַתְּ יְיָ מְדוֹרֵךְ לַעֲלַם בִּשְׁמַיָא תֵּתֵב וְדָכְרָנָךְ לְדָר וְדָר:
אַתְּ תְּקוּם תְּרַחֵם צִיוֹן אֲרוּם עִדָן לְמֵחוּס עֲלָהּ אֲרוּם אֲתָא זִמְנָא:
אֲרוּם רָעוּ עַבְדָךְ יַת אֲבָנָהָא וְיַת עַפְרָהָא יְחַיְסוּן:
וְיִדַחֲלוּן עַמַיָא יַת שְׁמָא דַייָ וְכָל מַלְכַיָא דְאַרְעָא יַת יְקָרָךְ:
אֲרוּם אִתְבְּנֵי בְּמֵימְרָא דַייָ קַרְתָּא דְצִיוֹן אִתְגְלֵי בִּיקָרֵיהּ:
אִתְפְּנֵי לְוַת צְלוֹתֵיהּ דְמִצְטַדְיָן וְלָא בַּסֵר יַת צְלוֹתְהוֹן:
תִּתְכְּתֵב צְלוֹתָא דָא לְדָר בַּתְרָי וְעַמָא דַעֲתִיד לְאִתְבְרָאָה יְשַׁבַּח יָהּ:
אֲרוּם אוֹדִיק מִשְׁמֵי מְרוֹמָא דְקוּדְשֵׁיהּ יְיָ מִן שְׁמַיָא לְאַרְעָא אִסְתַּכָּל:
לְמִשְׁמַע אַנִיקוּתָא דַאֲסִירַיָא לְמִשְׁרֵי בְנַיָא דִמְסִירִין לְמִיתוּתָא:
לְמִשְׁתָּעֵי בְצִיוֹן שְׁמָא דַייָ וְתוּשְׁבְּחֵיהּ בִּירוּשְׁלֵם:
בְּאִתְכַּנְשׁוּת עַמִין כַּחֲדָא וּמַלְכְּוָתָא לְמִפְלַח קֳדָם יְיָ:
אִסְתְּגֵיף מִן טוֹרַח דְאוֹרַח גָלוּתָא חֵילִי אִתְקַצְרוּן יוֹמָי:
אֵימַר קֳדָם אֱלָהִי לָא תְסַלְקִנַנִי מִן עָלְמָא בְּפַלְגוּת יוֹמָי אַיְתֵי יָתִי לְעָלְמָא דְאָתֵי מְטוּל דִי בְדָרֵי דָרַיָא שְׁנָתָךְ:
מִן שֵׁרוּיָא דְאִתְבְּרִי כָּל בִּרְיְתָא אַרְעָא יִסַדְתָּא וְעוֹבְדֵי יְדָךְ שְׁמַיָא:
הִנוּן יְהוֹבְדוּן וְאַתְּ תְּקוּם וְכוּלְהוֹן הֵיךְ לְבוּשָׁא יִתְבַּלְיָן הֵיךְ אִצְטְלֵי תַחְלְפִינוּן וְיִתְחַלְפוּן:
וְאַתְּ הוּא דִי בְרֵיתִנוּן וּשְׁנָתָךְ לָא מִשְׁתַּלְמָן:
בְּנֵי עַבְדָךְ יִשְׁרוֹן בְּאַרְעָא וְזַרְעֲיָתְהוֹן קָדָמָךְ יִתַּקְנוּן:
