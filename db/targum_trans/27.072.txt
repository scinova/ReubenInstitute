עַל יְדוֹי דִשְׁלֹמֹה אִתְאֲמַר בִּנְבוּאָה אֱלָהָא הִילְכוֹת דִינָךְ לְמַלְכָּא מְשִׁיחָא הַב וְצִדְקָתָךְ לִבְרֵיהּ דְדָוִד מַלְכָּא:
יָדִין עַמָךְ בְּצִדְקָא וַעֲנִיָךְ בְּהִלְכָת דִינָא:
יִטְלוּן יָתְבֵי טוּרַיָא שְׁלָמָא לְבֵית יִשְׂרָאֵל וְגַלְמָתָא בִּזְכוּתָא:
יְדִין עֲנִיֵי עַמָא יִפְרוֹק לִבְנֵי חֲשִׁיכָא וְיִשׁוֹף גַבְרָא טְלוֹמָא:
יִדְחַלוּן מִנָךְ עִם מִסַק שִׁמְשָׁא וִיצָלוּן קֳדָמָךְ קֳדָם מִנְהַר סִיהֲרָא דָרֵי דָרַיָא:
יֵחוֹת הֵיךְ מִטְרָא דְרַעֲוָא עַל עִסְבָּא דִגְזִיז מִן גוֹבַאי הֵיךְ רְסִיסֵי מַלְקוֹשָׁא דְנָטְפִין עַל עִסְבָּא דְאַרְעָא:
יִסְגוּן בְּיוֹמוֹהִי צַדִיקַיָא וּסְגִיעֵי שְׁלָמָא עַד דִי יִשְׁתֵּיצוּן פָּלְחֵי סִהֲרָא:
וְיִשְׁלוֹט מִן סְטַר יַמָא רַבָּא לִסְטַר יַמָא רַבָּא וּמִן פְּרָת עַד סְיָפֵי אַרְעָא:
קֳדָמוֹי יִגְחֲנוּן אֵיפַרְכַיָא וּבַעֲלֵי דְבָבוֹי עַפְרָא יְלַחֲכוּן:
מַלְכַיָא דְטַרְסִיס וְגִיסֵי יַמָא דְאוֹקְיָנוּס תִּקְרֻבְתָּא יְתִיבוּן מַלְכַיָא דִשְׁבָא וּסְבָא דוֹרוֹן יִקְרְבוּן:
וְיִסְגְדוּן לֵיהּ כָּל מַלְכַיָא כָּל עַמְמַיָא יִשְׁתַּעְבְּדוּן לֵיהּ:
אֲרוּם יִפְצֵי חֲשִׁיכָא דְבָעֵי מָטוּ וְעַנְיָא וְלֵית סָעֵיד לֵיהּ:
יֵחוֹס עַל מִסְכָּנָא וַחֲשׁוּכָא וְנַפְשָׁת חֲשִׁיכַיָא יִפְרוֹק:
מֵאוּנְסָא וּמֵחֲטוֹפִין יִפְרוֹק נַפְשֵׁיהוֹן וִיהֵי יַקִיר אַדְמְהוֹן קֳדָמוֹי:
וִיחֵי וְיִתֵּן לֵיהּ מִזְהַב דְאַיְתִיאוּ לֵיהּ מִשְׁבָא וִיצַלֵי מְטוּלְתֵּיהּ תְּדִירָא כָּל יוֹמָא יְבָרְכִנֵיהּ:
יְהִי סָעִיד לַחְמָא בְּאַרְעָא בְּרֵישׁ טוּרַיָא יַרְגִישׁ הֵיךְ כְּלִבְנָן אִבֵּיהּ וִינַצְצוּן מִן קַרְתָּא דִירוּשְׁלֶם הֵיךְ עִסְבָּא דְאַרְעָא:
יְהִי שְׁמֵיהּ מַדְכַּר לַעֲלָם וּקְדָם מֶהֱוֵי שִׁמְשָׁא מְזַמָן הֲוָה שְׁמֵיהּ וְיִתְבָּרְכוּן בִּזְכוּתֵיהּ כָּל עַמַיָא וְיֵמְרוּן טָב לֵיהּ:
בְּרִיךְ יְיָ אֱלֹהִים אֱלָהָא דְיִשְׂרָאֵל דַעֲבִיד פְּרִישָׁן רַבְרְבָן בִּלְחוֹדוֹהִי:
וּבְרִיךְ שׁוּם יְקָרֵיהּ לְעָלְמָא וְיִתְמַלֵי מִזִיו יְקָרֵיהּ כָּל אַרְעָא אָמֵן וְאָמֵן:
גְמִירִין צְלוֹתַיָא דְדָוִד בַּר יִשָׁי:
