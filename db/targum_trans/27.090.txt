צְלוֹתָא דְצַלֵי מֹשֶׁה נְבִיָא דַייָ כַּד חָבוּ עַמָא בֵית יִשְׂרָאֵל בְּמַדְבְּרָא עֲנֵי וְכֵן אָמַר יְיָ דְמַדוֹר בֵּית שְׁכִנְתָּךְ בִּשְׁמַיָא אַתְּ הֲוֵיתָא לָנָא סָעֵיד בְּכָל דָר וָדָר:
כַּד אִתְגְלֵי קֳדָמָךְ דְעַמָךְ עֲתִידִין לְמֵחוֹב אַתְקֵנְתָּא תְּתוּבָא עַד דְלָא טוּרַיָא אִתְנַטַלוּ וְאִתְבְּרִיאַת אַרְעָא וְיָתְבֵי תֵבֵל מִן עַלְמָא הָדֵין עַד עַלְמָא דְאָתֵי אַתְּ הוּא אֱלָהָא:
תְּתֵיב בַּר נַשׁ עַל חוֹבֵיהּ עַד מוֹתָא וַאֲמַרֵת תּוּבוּ בְנֵי נָשָׁא:
אֲרוּם אֲלַף שְׁנִין בְּעֵינָךְ מִתְחַשְׁבִין קֳדָמָךְ הֵיךְ יוֹמָא דְאִתְמָלֵי אֲרוּם יֶעְבַּר וְהֵיךְ מַטְרַת לֵילְיָא:
וְאִין לָא תָיְבִין תֵּיתֵי עֲלֵיהוֹן מוֹתָא הֵיךְ דְמָכִין יְהוֹן וּלְעָלְמָא דְאָתִי הֵיךְ עִסְבָּא דְמִתְפָּרְכָא יִתְחַלְפוּן:
דַמְיָן עוֹבָדֵיהוֹן הֵיךְ עִסְבָּא דִי בְצַפְרָא יְנִיץ וְיַסְגֵי לְרַמְשָׁא מִתְמוֹלֵל וּמִתְיַבֵּשׁ מִן חוּרְבָּא:
אֲרוּם אִשְׁתֵּיצָנָא מִן תּוּקְפָךְ וּמִן רוּגְזָךְ אִתְבַּהֵלְנָא:
שַׁוִיתָא חוֹבָנָא לְקִבְלָךְ עַוְיַת טַלְיוּתָנָא קָבֵל נְהוֹר אַפָּךְ:
אֲרוּם כָּל יוֹמָנָא אִתְפְּנִיאוּ קֳדָמָךְ בְּרוּגְזָךְ שֵׁצֵינָא יוֹמֵי חַיָנָא הֵיךְ הֲבֵל פּוּמָא דְסִתְוָא:
יוֹמֵי שְׁנוֹתָנָא בְּעָלְמָא הָדֵין שׁוּבְעִין שְׁנִין מִתְאַלְמִין וְאִין בִּגְבוּרְתָּא תַּמְנָן שְׁנִין וְסוּגְעֵיהוֹן לֵיאוּת וּשְׁקַר לְחַיָבַיָא אֲרוּם עֲדוּ בְּסַרְהוֹבְיָא וְטָיְסִין לְצַפְרָא:
מַן הוּא דְיָדַע לְאַתְבָא עוּשְׁנָא דְתוּקְפָךְ אֱלָהֵין צַדִיקַיָא דְדָחֲלִין מִנָךְ מְשַׁדְכִין רוּגְזָךְ:
לְמִמְנֵי יוֹמָנָא כֵּן יִתְכַּוֵן לְהוֹדַע בְּרַם נְבִיָא דְלִבֵּיהּ מַבַּע חוּכְמָתָא:
תּוּב יְיָ עַד אֵימָתַי תְסַגְפִנָנָא וְתוּב מִן בִּישְׁתָּא דִי מַלֵלְתָּא לְמֶעְבַּד לְעַבְדָךְ:
שַׂבַּע יָתָנָא לְעַלְמָא דִמְתִיל לְצַפְרָא טוּבָךְ וּנְרַנֵן וְנֶחְדֵי בְּכוּלְהוֹן יוֹמָנָא:
שַׂמַח יָתָנָא הֵיךְ יוֹמִין דְסַגִפְתָּנָא הֵיךְ שְׁנַיָא דַחֲמֵינָא בִּישְׁתָּא:
יִתְחֲמֵי לְעַבְדָךְ עוֹבָדֵי נִסָךְ וְשִׁבְהוֹרָךְ עַל בְּנֵיהוֹן:
וִיהֵי בְסִימוּתָא דְגַן עֵדֶן מִן קֳדָם יְיָ עֲלָנָא וְעוֹבָדֵי יְדֵינָא מִנֵיהּ יִתַּקְנוּן:
