לְשַׁבָּחָא עַל נִסַיָא דְאִתְרְחִישׁוּ לְעַבְדָא דַייָ לְדָוִד דְשַׁבָּח בִּנְבוּאָה קֳדָם יְיָ יַת פִּתְגָמֵי שִׁירְתָא הָדָא עַל כָּל יוֹמַיָא דְשֵׁזְבֵיהּ יְיָ מִן יְדָא דְכָל בַּעֲלֵי דְבָבוֹי וּמֵחַרְבָּא דְשָׁאוּל:
וַאֲמַר אֲחַבְּבִנָךְ יְיָ תּוּקְפִי:
יְיָ תּוּקְפִי וְרֻחְצָנִי וּמְשֵׁזִיב יָתִי אֱלָהָא דְאִתְרְעֵי בִּי קָרְבַנִי לִדְחַלְתֵּיהּ תְּרֵיסִי דְמִן קֳדָמוֹי אִתְיְהַב לִי תְקוֹף וּפוּרְקָן עַל בַּעֲלֵי דְבָבֵי רָחְצָנִי:
אֲמַר דָוִד בְתוּשְׁבַּחְתָּא אֲנָא מְצַלֵי קֳדָם יְיָ וּמִן בַּעֲלֵי דְבָבִי פָּרִיק יָתִי:
אַקְפַתְנִי עָקָא כְאִתָּא דְיָתְבָא עַל מַתְבְּרָא וְחֵיל לֵית לָהּ לְמֵילַד וְהִיא מְסַכְּנָא לְמֵמָת וְסִיעַת טָלוֹמַיָא בַּעֲתַת יָתִי:
מַשִׁרְיַת חַיָבִין אַקְפוּנִי אַקְדוּמִי יָתִי דִמְזַיְנִין בְּמָנֵי קְטוֹל:
בִּדְעָיְקָא לִי אֲנָא מְצַלֵי קֳדָם יְיָ וּקְדָם אֱלָהִי אֲנָא מִתְחַנֵן וּמְקַבֵּל צְלוֹתִי מִן הֵיכְלֵיהּ וּבָעוּתִי קֳדָמוֹי מִתְקַבְּלָא בְּאוּדְנוֹי וּמִתְעַבְדָא:
וְאִתְרְגִיפַת וְאִתְרְגִישַׁת אַרְעָא וְאֻשְׁיָת טוּרַיָא זְעוּ וְאִתְרְטִישׁוּ מְטוּל דִתְקֵיף לֵיהּ:
סְלִיק זְדוֹנֵיהּ דְפַרְעֹה הֵיךְ קוּטְרָא דִבְכֵן שְׁלַח רוּגְזֵיהּ כְּאֶשָׁא בָּעֲרָא דְמִן קֳדָמוֹי מִשְׁתֵּצְיָא מְזוֹפִיתֵיהּ כְּגוּמְרִין דִי נוּר דָלְקָא מִמֵימְרֵיהּ:
וְאַרְכֵן שְׁמַיָא וְאִתְגְלֵי יְקָרֵיהּ וַעֲנַן אֲמִטְתָא כַּבֵּשׁ קֳדָמוֹי:
וְאִתְגְלֵי בִּגְבוּרְתֵּיהּ עַל כְּרוּבִין קַלִילִין וּדְבַר בִּתְקוֹף עַל גַדְפֵי זַעֲפָא:
וְאַשְׁרֵי שְׁכִינְתֵּיהּ בְּעַרְפְּלָא וְאִתְחְזַר בַּעֲנָנֵי יְקָרֵיהּ הֵיךְ מְטַלַלְתָּא וְאַחֵית מִטְרִין דְרַעֲוָא עַל עַמֵיהּ וּמַיִן תַּקִיפִין מִן רִכְפַּת עֲנָנִין דְקִבְּלָא עַל רַשִׁיעַיָא מִן רוּמֵיהּ דְעָלְמָא:
מִן זְהוֹר יְקָרֵיהּ עֲנָנֵי שְׁמַיָא עֲבַרוּ בִּמְזוֹפִיתָא בְּגוּמְרִין דְנוּר בַּרְדָא דָלְקָא מִן מֵימְרֵיהּ:
וְאַכְלֵי מִן שְׁמַיָא יְיָ וְעִלָאָה אֲרִים מֵימְרֵיהּ רָמָא בַרְדָא וְגוּמְרִין דִי נוּר:
וְשַׁדַר מֵימְרֵיהּ הֵיךְ גִירִין וּבַרְדִין וּבַרְקִין סַגִיאִין יְשַׂגְשְׂגוּן:
וְאִתְחַמִיאוּ עִמְקֵי מַיָא וְאִתְגְלִיאוּ שִׁתְאֲסַיָא דְתֵבֵל מִמְזוֹפִיתָא מִן קֳדָם יְיָ מִמֵימַר תְּקוֹף רוּגְזָךְ:
שְׁלַח נְבִיוֹהִי מֶלֶךְ תַּקִיף דְיָתֵיב בִּתְקוֹף רוּמָא דַבְּרַנִי שֵׁזְבַנִי מֵעַמְמִין סַגִיאִין:
שֵׁזְבַנִי מִן סָנְאַי אֲרוּם עֲשִׁינִין הִינוּן וּמִן בַּעֲלֵי דְבָבִי אֲרוּם אִתְגַבַּרוּ עֲלָי:
אַקְדִימוּ לִי בְיוֹם טִלְטוּלִי וַהֲוָה מֵימְרָא דַייָ סָמִיךְ לִי:
וְאַפֵּק יָתִי לִרְוָחָא שֵׁזְבַנִי מְטוּל דְאִתְרְעֵי בִי:
יִגְמְלִנַנִי יְיָ כִזְכוּתִי כִּבְרִירוּת יְדֵי יְתִיב לִי:
מְטוּל דִנְטָרֵית אוֹרְחָן דְתַקְנָן קֳדָם יְיָ וְלָא הַלְכֵית בִּרְשַׁע קֳדָם יְיָ:
מְטוּל דְכָל דִינוֹ גְלוֹי לְקִבְלִי לְמֶעְבַּדְהוֹן וּקְיָמוֹי לָא אַעְדֵי מֶנִי:
וַהֲוֵיתִי שְׁלִים בִּדְחַלְתֵּיהּ וַהֲוֵיתִי נָטִיר נַפְשִׁי מֵחוֹבִין:
וּפְרַע יְיָ לִי הֵיךְ זְכוּתִי כִּבְרִירוּת אַיְדִי קֳדָם מֵימְרֵיהּ:
עִם אַבְרָהָם דְאִשְׁתַּכַּח חֲסִיד קֳדָמָךְ אַסְגֵיתָא לְמֶעְבַּד חִסְדָא עִם זַרְעֵיהּ דְיִצְחָק דַהֲוָה שְׁלִים בִּדְחַלְתָּךְ אַשְׁלֶמְתָּא מֵימַר רְעוּתָךְ עִמֵיהּ:
עִם יַעֲקֹב דַהֲוָה בְּרִיר קֳדָמָךְ בְּחַרְתָּא בְנוֹי מִן כָּל עַמְמַיָא וְאַפְרֵשְׁתָּא זַרְעֵיהּ מִן כָּל פְּסִילָא וְעִם פַּרְעֹה וְזַרְעֵיהּ וּמִצְרָאֵי דַחֲשִׁיבוּ מַחְשְׁבָן בִּישָׁן עַל עַמָךְ בַּלְבֵּלְתִּנוּן בְּמַחְשְׁבָתְהוֹן:
מְטוּל דְאַנְתְּ יַת עַמָא בֵּית יִשְׂרָאֵל דַחֲשִׁיכִין בֵּינֵי עַמְמַיָא בְּגָלוּתָא אַנְתְּ עֲתִיד לְמִפְרַק וּבְמֵימְרָךְ אוּמַיָא תַקִיפַיָא דְמִתְגַבְּרִין עֲלֵיהוֹן תַּמְאִיךְ:
מְטוּל דְאַנְתְּ תַּנְהַר שְׁרָגָא דְיִשְׂרָאֵל דְמִטַפְיָא בְגָלוּתָא דְאַנְתְּ הוּא מָרֵיהּ דִנְהוֹרָא דְיִשְׂרָאֵל יְיָ אֱלָהִי יַפְקִנַנִי מֵחֶשְׁכָּא לִנְהוֹר יַחְמִנַנִי בְנִיחָמוּתִי דְעָלְמָא דַעֲתִיד לְמֵיתֵי לְצַדִיקַיָא:
מְטוּל דִבְמֵימְרָךְ אַסְגֵי מַשִׁרְיָן וּבְמֵימַר אֱלָהִי אֶכְבּוֹשׁ כְּרַכִּין תַּקִיפִין:
אֱלָהָא דְכִוְנָן אָרְחָתֵיהּ אוֹרַיְיתָא דַייָ בְּחִירָא תְּרֵיס הוּא לְכָל דְמִתְרַחֲצִין עֲלוֹי:
אֲרוּם נִסָא וּפוּרְקָנָא דְתַעְבֵּיד לִמְשִׁיחָךְ וּלְשִׁיוּרֵי עַמָךְ דְיִשְׁתָּאֲרוּן יוֹדוּן כָּל עַמְמַיָא אוּמַיָא וְלִשָׁנַיָא וְיֵימְרוּן לֵית אֱלָהָא אֶלָא יְיָ אֲרוּם לֵית בַּר מִנָךְ וְעַמֵךְ יֵימְרוּן לֵית דְתַקִיף אֶלָא אֱלָהָנָא:
אֱלָהָא דִמְזָרֵז לִי קָמוּר בְּחֵילָא וּמְתַקֵן שְׁלִים אָרְחִי:
דִמְשַׁוֵי רִגְלַי קַלִילִין הֵיךְ אַיָלָתָא וְעַל בֵּית תּוּקְפִי יְקַימִנָנִי:
מַלֵיף אַיְדֵי לְמֶעְבַּד קְרָבָא וּמַתְקֵיף הֵיךְ קֶשֶׁת כַּרְכּוֹמָא דַרְעָי:
וִיהַבְתְּ לִי תְקוֹף וּפוּרְקַן וִימִינָךְ תְּסֵיְיעֵנִי וּבְמֵימְרָךְ אַסְגַעְתַּנִי:
אַפְתֵּיתָ פְּסִיְעַתִי בְּאַתְרִי וְלָא אִזְדַעְזַא רְכוּבְתִּי:
אֶרְדוֹף סָנְאַי וְשֵׁצִיתִנוּן וְלָא שְׁבוּת עַד דִגְמַרְתִּנוּן:
אֲגַמְרִנוּן וְלָא יָכִילוּ לְמֵקָם וּנְפַלוּ קְטִילִין תְּחוֹת פַּרְסָת רִגְלָי:
וְזֵרַזְתָּא לִי חֵילָא הֵיךְ קָמוֹר לְמֶעְבַּד קְרָבָא תַּבַּרְתָּא עַמְמִין דְקַיְמִין לְאַבְאָשָׁא לִי תְחוֹתָי:
וּבַעֲלֵי דְבָבִי תַּבַּרְתָּ קֳדָמַי יְהַבְתִּנוּן מְחַזְרֵי קְדַל וְסָנְאַי אֲשֵׁיצִנוּן:
בָּעָן סָעִיד וְלֵית לְהוֹן פָּרִיק מַצְלָן קֳדָם יְיָ וְלָא מְקַבֵּל צְלוֹתְהוֹן:
וְדוֹשַׁשְׁתִּינוּן הֵיךְ גַרְגִשְׁתָּא דְאַרְעָא עַל אַנְפֵּי זַעְפָּא וְהֵיךְ סְיַן אַשְׁקָקֵי בַעֲטֵית אִנוּן:
תְּשֵׁזְבִנַנִי מִפְּלֻגַת עַמְמַיָא תְּנַטְרִינַנִי בְמֵיזְרֵי מַזְיָן בְּרֵישׁ עַמְמַיָא עַמָא דְלָא חַכִּימַת יִפְלְחוּנַנִי:
לְשֵׁמַע אִידִין יִשְׁתַּמְעוּן לִי בְּנֵי עַמְמַיָא יְכַדְבוּן קֳדָמַי:
בְּנֵי עַמְמַיָא יְסוּפוּן וִיטַלְטְלוּן מִבִּרְנְיָתְהוֹן:
קַיָם הוּא יְיָ וּבְרִיךְ תַּקִיפָא דְמִן קֳדָמוֹהִי מִתְיְהַב לִי תְקוֹף וּפוּרְקַן וּמְרוֹמֵם אֱלָהָא תְּקוֹף פּוּרְקָנִי:
אֱלָהָא דְעָבֵד פוּרְעֲנוּתָא לִי וּמְתַבַּר עַמְמַיָא דְקַיְמִין לְאַבְאָשָׁא לִי תְחוֹתִי:
מְשֵׁזְבִי מִבַּעֲלֵי דְבָבִי לְחוֹד עַל דְקָיְימִין לְאַבְאָשָׁא לִי תִּגְבְּרִנַנִי מִן גוֹג וּמַשִׁרְיָת עַמְמִין דַחֲטוֹפִין עִמְהוֹן תְּשֵׁזְבִנַנִי:
מְטוּל הֵיכְנָא אוֹדֶה קֳדָמָךְ בֵּינֵי עַמְמַיָא יְיָ וְלִשְׁמָךְ אֲזַמֵר תּוּשְׁבַּחְתָּא:
מַסְגֵי לְמֶעְבַּד פֻּרְקָן עִם מַלְכֵּיהּ וַעֲבַד טִיבוּ לִמְשִׁיחֵיהּ לְדָוִד וּלְזַרְעֵיהּ עַד עָלְמָא:
