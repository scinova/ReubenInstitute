בְּמֵימְרָךְ יְיָ רְחִיצִית לָא אֶבְהַת לַעֲלָם:
בְּצִדְקָתָךְ פְּצֵי יָתִי וְשֵׁזְבִנַנִי אַצְלֵי לְוָתִי אוּדְנָךְ וּפְרוֹק יָתִי:
הֱוֵי לִי לְטִינָר תַּקִיף עָשִׁין לְמֵעַל תְּדִירָא פְּקַדְתָּ לְמִפְרַק יָתִי אֲרֵי תּוּקְפִי וּכְרַךְ חוּסְנִי אַתְּ:
אֱלָהִי שֵׁזִיב יָתִי מִן יַד רַשִׁיעָא מִן יְדָא דְמָרֵי עִילָא וַחֲטוֹף:
אֲרוּ אַתְּ הוּא סוֹבָרִי יְיָ אֱלָהִי רוּחֲצָנִי מִן טַלְיוּתִי:
עֲלָיִךְ סַמְכֵית מִן כְּרֵיסָא מִן מְעַיָא דְאִמִי אַנְתְּ מַפְקִי בְּמֵימְרָךְ תּוּשְׁבָּחְתִּי תְּדִירָא:
הֵיךְ תָּמְהָא הֲוֵיתִי לְסַגִיעִין וְאַנְתְּ רוּחֲצָנִי וְעוּשְׁנָא דִילִי:
יִתְמְלֵי פּוּמִי תּוּשְׁבַּחְתָּךְ כָּל יוֹמָא שִׁבְהוֹרָךְ:
לָא תִטְלוֹק יָתִי לְעִדָן סֵיבוּ כַּד יִפְסוֹק חֵילִי לָא תִשְׁבְּקִנַנִי:
אֲרוּם אֲמַרוּ בַּעֲלֵי דְבָבִי עֲלַי בִּישׁ וְנַטְרֵי נַפְשִׁי יִתְמְלִיכוּ כַּחֲדָא:
לְמֵימַר אֱלָהָא שַׁבְקֵיהּ רְדוּפוּ וַאֲחִידוּ יָתֵיהּ אֲרוּם לֵית דְפָצֵי:
אֱלָהָא לָא תַרְחִיק מִנִי אֱלָהִי לְסַעְדִי זְרֵיז:
יִבְהֲתוּן יִשְׁתֵּיצוּן סָטְנֵי נַפְשִׁי יִתְעַטְפוּן קְלָנָא וְכִיסוּפָא תָּבְעֵי בִּישׁוּתִי:
וַאֲנָא תְדִירָא אוֹרִיךְ וְאוֹסִיף עַל כָּל תּוּשְׁבַּחְתָּךְ:
פּוּמִי יִשְׁתָּעֵי צִדְקוּתָךְ כָּל יוֹמָא פּוּרְקָנָךְ אֲרוּם לָא יְדַעִית מִנְיָנֵהוֹן:
אֵעוּל בִּגְבוּרְתָּא דַייָ אֱלֹהִים אִדְכַּר צִדְקוּתָךְ בִּלְחוֹדָךְ:
אֱלָהָא אַלֵיפְתַּנִי בְּנִסַיָא מִן טַלְיוּתִי וְעַד הַעִדָנָא אַשְׁנֵי פְּרִישָׁתָךְ:
וּלְחוֹד עַד זִקְנָא וְסֵיבוּתָא אֱלָהָא לָא תִשְׁבּוֹק יָתִי עַד דַאֲתַנֵי תְּקוֹף אֶדְרְעָךְ לְדָר דָרֵי לְכָל דִי יֵיתִי כֹּחַ גְבוּרְתָּךְ:
וְצִדְקָתָךְ אֱלָהָא עַד שְׁמֵי מְרוֹמָא דִי עֲבַדְתָּ רַבְרְבָן אֱלָהָא מַן כְּוָתָךְ:
דִי אֲחֵמְתַּנִי עָקָן סַגִיעִין וּבִישִׁין תְּתוּב תְּחַיֵי יָתָנָא וּמִתְּהוֹמָה אַרְעִיתָא תְּתִיב תַּסֵק יָתָנָא:
תַּסְגֵי רַבְיוּתִי וְתַחְזוֹר תַּנְחַמִנַנִי:
לְחוֹד אֲנָא אוֹדֵי קֳדָמָךְ בִּזְיָנֵי זְמַר וְנִבְלָא אֲתַנֵי קוּשְׁטָךְ אֱלָהִי אֲשַׁבַּח קֳדָמָךְ בְּכִנָרָא קַדִישָׁא יִשְׂרָאֵל:
תְּבוּעָן סִפְוָתִי אֲרוּם אֲשַׁבַּח קֳדָמָךְ וְנַפְשִׁי דִי פְרַקְתָּא:
לְחוֹד לִישָׁנִי כָּל יוֹמָא תְּרַנֵן צִדְקָתָךְ אֲרוּם בְּהִיתוּ אֲרוּם אִיתְחַסַדוּ תָּבְעֵי בִישׁוּתִי:
