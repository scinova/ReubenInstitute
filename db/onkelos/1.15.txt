בָּתַר פִּתְגָמַיָא הָאִלֵין הֲוָה פִּתְגָמָא דַיְיָ עִם אַבְרָם בִּנְבוּאָה לְמֵימָר לָא תִדְחַל אַבְרָם מֵימְרִי תְּקוֹף לָךְ אַגְרָךְ סַגִי לַחֲדָא
וַאֲמַר אַבְרָם יְיָ אֱלֹהִים מַה תִּתֶּן לִי וַאֲנָא אָזֵיל בְּלָא וְלָד וּבַר פַּרְנָסָא הָדֵין דִבְּבֵיתִי הוּא דַמַשְׂקָאָה אֱלִיעֶזֶר
וַאֲמַר אַבְרָם הָא לִי לָא יְהַבְתְּ וְלָד וְהָא בַר בֵיתִי יָרִיתּ יָתִי
וְהָא פִּתְגָמָא דַיְיָ עִמֵהּ לְמֵימַר לָא יִרְתִינָךְ דֵין אֱלָהֵין בַּר דְתוֹלִיד הוּא יִרְתִינָךְ
וְאַפֵּיק יָתֵהּ לְבָרָא וַאֲמַר אִסְתְּכִי כְעַן לִשְׁמַיָא וּמְנֵי כּוֹכְבַיָא אִם תִּכּוֹל לְמִמְנֵי יָתְהוֹן וַאֲמַר לֵיהּ כְּדֵין יְהוֹן בְּנָךְ
וְהֵימִין בְּמֵימְרָא דַיְיָ וְחָשְׁבַהּ לֵיהּ לְזָכוּ
וַאֲמַר לֵהּ אֲנָא יְיָ דְאַפֵּקְתָּךְ מֵאוּרָא דְכַשְׂדָאֵי לְמִתַּן לָךְ יָת אַרְעָא הָדָא לְמֵירְתַהּ
וַאֲמָר יְיָ אֱלֹהִים בְּמָה אִידַע אֲרֵי אֵרְתִנַהּ
וַאֲמַר לֵהּ קָרֵב קֳדָמַי עֶגְלִין תְּלָתָא וְעִזִין תְּלָתָא וְדִכְרִין תְּלָת וְשַׁפְנִינָא וּבַר יוֹנָה
וְקָרֵב קֳדָמוֹהִי יָת כָּל אִלֵין וּפַלִיג יָתְהוֹן בְּשָׁוֶה וִיהַב פַּלְגַיָא פִּלוּג לָקֳבֵיל חַבְרֵיהּ וְיָת עוֹפָא לָא פַּלִיג
וּנְחַת עוֹפָא עַל פַּגְלַיָא וְאַפְרַח יָתְהוֹן אַבְרָם
וַהֲוָה שִׁמְשָׁא לְמֵיעַל וְשִׁינְתָא נְפָלַת עַל אַבְרָם וְהָא אֵימָתָא קְבֵל סַגִי נְפָלַת עֲלוֹהִי
וַאֲמַר לְאַבְרָם מִדַע תִּדַע אֲרֵי דַיָרִין יְהוֹן בְּנָיךְ בְּאַרְעָא דְלָא דִילְהוֹן וְיִפְלְחוּן בְּהוֹן וִיעַנוּן יָתְהוֹן אַרְבַּע מְאָה שְׁנִין
וְאַף יָת עַמָא דְיִפְלְחוּן בְּהוֹן דָיִין אֲנָא וּבָתַר כֵּן יִפְקוּן בְּקִנְיָנָא סַגִי
וְאַתְּ תֵעוֹל לְוָת אַבָהָתָךְ בִּשְׁלָם תִּתְקְבַר בְּסֵיבוּ טָבָא
וְדָרָא רְבִיעָאָה יְתוּבוּן הָכָא אֲרֵי לָא שְׁלִים חוֹבָא דֶאֱמוֹרָאָה עַד כְּעָן
וַהֲוָה שִׁמְשָׁא עַלַת וְקִבְלָא הֲוָה וְהָא תַּנוּר דִתְנַן וּבָעוּר דְאֶשָׁתָא דִעֲדָא בֵּין פַּלְגַיָא הָאִלֵין
בְּיוֹמָא הַהוּא גְזַר יְיָ עִם אַבְרָם קְיָם לְמֵימָר לִבְנָךְ יְהָבִית יָת אַרְעָא הָדָא מִן נַהֲרָא דְמִצְרַיִם וְעַד נַהֲרָא רַבָּא נַהֲרָא פְרָת
יָת שַׁלְמָאֵי וְיָת קְנִזָאֵי וְיָת קַדְמוֹנָאֵי
וְיָת חִתָּאֵי וְיָת פְּרִזָאֵי וְיָת גִבָּרָאֵי
וְיָת אֱמוֹרָאֵי וְיָת כְנַעֲנָאֵי וְיָת גִרְגָשָׁאֵי וְיָת יְבוּסָאֵי
