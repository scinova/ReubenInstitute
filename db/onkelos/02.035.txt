וְאַכְנֵשׁ משֶׁה יָת כָּל כְּנִשְׁתָּא דִבְנֵי יִשְרָאֵל וַאֲמַר לְהוֹן אִלֵין פִּתְגָמַיָא דִי פַקֵד יְיָ לְמֶעְבַּד יָתְהוֹן:
שִׁתָּא יוֹמִין תִּתְעֲבֵד עִבִדְתָּא וּבְיוֹמָא שְׁבִיעָאָה יְהֵי לְכוֹן קוּדְשָׁא שַׁבָּא שַׁבְּתָא קֳדָם יְיָ כָּל דְיַעֲבֵד בֵּה עִבִדְתָּא יִתְקְטֵל:
לָא תְבַעֲרוּן אֶשָׁתָא בְּכֹל מוֹתְבָנֵיכוֹן בְּיוֹמָא דְשַׁבְּתָא:
וַאֲמַר משֶׁה לְכָל כְּנִשְׁתָּא דִבְנֵי יִשְׂרָאֵל לְמֵימָר דֵין פִּתְגָּמָא דִי פַקֵד יְיָ לְמֵימָר:
סָבוּ מִנְכוֹן אַפְרָשׁוּתָא קֳדָם יְיָ כֹּל דְיִתְרְעֵי לִבֵּהּ יַיְתֵי יָת אַפְרָשׁוּתָא קֳדָם יְיָ דַהֲבָא וְכַסְפָא וּנְחָשָׁא:
וְתִכְלָא וְאַרְגְוָנָא וּצְבַע זְהוֹרִי וּבוּץ וּמֵעַזֵי:
וּמַשְׁכֵי דְדִכְרֵי מְסַמְקֵי וּמַשְׁכֵי דְסַסְגוֹנָא וְאָעֵי שִׁטִין:
וּמִשְׁחָא לְאַנְהָרוּתָא וּבוּסְמַיָא לִמְשַׁח רְבוּתָא וְלִקְטֹרֶת בּוּסְמַיָא:
וְאַבְנֵי בוּרְלָא וְאַבְנֵי אַשְׁלָמוּתָא לְשַׁקָעָא בְּאֵפוֹדָא וּבְחוּשְׁנָא:
וְכָל חַכִּימֵי לִבָּא בְּכוֹן יֵיתוּן וְיַעְבְדוּן יָת כָּל דִי פַקֵד יְיָ:
יָת מַשְׁכְּנָא יָת פְּרָסֵהּ וְיָת חוֹפָאֵהּ יָת פּוּרְפוֹהִי וְיָת דַפּוֹהִי יָת עַבְרוֹהִי יָת עַמוּדוֹהִי וְיָת סַמְכוֹהִי:
יָת אֲרוֹנָא וְיָת אֲרִיחוֹהִי יָת כַּפֻּרְתָּא וְיָת פָּרֻכְתָּא דִפְרָסָא:
יָת פָּתוֹרָא וְיָת אֲרִיחוֹהִי וְיָת כָּל מָנוֹהִי וְיָת לְחֵם אַפַּיָא:
וְיָת מְנַרְתָּא דְאַנְהוֹרֵי וְיָת מָנָהָא וְיָת בּוֹצִינָהָא וְיָת מִשְׁחָא דְאַנְהָרוּתָא:
וְיָת מַדְבְּחָא דִקְטוֹרֶת בּוּסְמַיָא וְיָת אֲרִיחוֹהִי וְיָת מִשְׁחָא דִרְבוּתָא וְיָת קְטֹרֶת בּוּסְמַיָא וְיָת פְּרָסָא דְתַרְעָא לִתְרַע מַשְׁכְּנָא:
יָת מַדְבְּחָא דַעֲלָתָא וְיָת סְרָדָא דִנְחָשָׁא דִילֵהּ יָת אָרִיחוֹהִי וְיָת כָּל מָנוֹהִי יָת כִּיוֹרָא וְיָת בְּסִיסֵהּ:
יָת סְרָדֵי דַרְתָּא יָת עַמוּדוֹהִי וְיָת סַמְכָהָא וְיָת פְּרָסָא דִתְרַע דַרְתָּא:
יָת סִכֵּי מַשְׁכְּנָא וְיָת סִכֵּי דְדַרְתָּא וְיָת אַטוּנֵיהוֹן:
יָת לְבוּשֵׁי שִׁמוּשָׁא לְשַׁמָשָׁא בְקוּדְשָׁא יָת לְבוּשֵׁי קוּדְשָׁא לְאַהֲרֹן כַּהֲנָא וְיָת לְבוּשֵׁי בְנוֹהִי לְשַׁמָשָׁא:
וּנְפָקוּ כָּל כְּנִשְׁתָּא דִבְנֵי יִשְׂרָאֵל מִן קֳדָם משֶׁה:
וַאֲתוֹ כָּל גְבַר דְאִתְרְעֵי לִבֵּהּ וְכֹל דִי אַשְׁלֵמַת רוּחֵהּ יָתֵהּ אַיְתִיאוּ יָת אַפְרָשׁוּתָא קֳדָם יְיָ לַעֲבִידַת מַשְׁכַּן זִמְנָא וּלְכָל פֻּלְחָנֵהּ וְלִלְבוּשֵׁי קוּדְשָׁא:
וּמַיְתַן גֻבְרַיָא עַל נְשַׁיָא כֹּל דְאִתְרְעֵי לִבֵּהּ אַיְתִיאוּ שֵׁירִין וְשִׁבְכִין וְעִזְקַן וּמָחוּךְ כָּל מָן דִדְהַב וְכָל גְּבַר דִי אֲרֵים אֲרָמוּת דַהֲבָא קֳדָם יְיָ:
וְכָל גְבַר דְאִשְׁתְּכַח עִמֵהּ תִּכְלָא וְאַרְגְוָנָא וּצְבַע זְהוֹרִי וּבוּץ וּמֵעַזֵי וּמַשְׁכֵי דְדִכְרֵי מְסַמְקֵי וּמַשְׁכֵי סַסְגוֹנָא אַיְתִיאוּ:
כָּל דַאֲרֵים אֲרָמוּת כְּסַף וּנְחַשׁ אַיְתִיאוּ יָת אַפְרָשׁוּתָא קֳדָם יְיָ וְכֹל דִי אִשְׁתְּכַח עִמֵהּ אָעֵי שִׁטִין לְכָל עִבִידַת פֻּלְחָנָא אַיְתִיאוּ:
וְכָל אִתְּתָא חַכִּימַת לִבָּא בִּידָהָא מְעַזְלָא וּמַיְתַן כַּד עָזֵל יָת תִּכְלָא וְיָת אַרְגְוָנָא יָת צְבַע זְהוֹרִי וְיָת בּוּצָא:
וְכָל נְשַׁיָא דְאִתְרְעֵי לִבְּהֶן עִמְהֶן בְּחָכְמָא עַזְלַן יָת מֵעַזְיָא:
וְרַבְרְבַיָא אַיְתִיאוּ יָת אַבְנֵי בוּרְלָא וְיָת אַבְנֵי אַשְׁלָמוּתָא לְשַׁקָעָא בְאֵפוֹדָא וּבְחוּשְׁנָא:
וְיָת בּוּסְמָא וְיָת מִשְׁחָא לְאַנְהָרוּתָא וּלְמִשְׁחָא דִרְבוּתָא וְלִקְטוֹרֶת בּוּסְמַיָא:
כָּל גְבַר וְאִתְּתָא דְאִתְרְעֵי לִבְּהוֹן עִמְהוֹן לְאַיְתָאָה לְכָל עִבִדְתָּא דִי פַקֵד יְיָ לְמֶעְבַּד בִּידָא דְמשֶׁה אַיְתִיאוּ בְנֵי יִשְׂרָאֵל נְדַבְתָּא קֳדָם יְיָ:
וַאֲמַר משֶׁה לִבְנֵי יִשְׂרָאֵל חֲזוֹ דְרַבִּי יְיָ בְּשׁוּם בְּצַלְאֵל בַּר אוּרִי בַר חוּר לְשִׁבְטָא דִיהוּדָה:
וְאַשְׁלֵם עִמֵהּ רוּחַ נְבוּאָה מִן קֳדָם יְיָ בְּחָכְמְתָא בְּסוּכְלְתָנוּ וּבְמַדַע וּבְכָל עִבִידָא:
וּלְאַלָפָא אוּמַנְוָן לְמֶעְבַּד בְּדַהֲבָא וּבְכַסְפָא וּבִנְחָשָׁא:
וּבְאוּמָנוּת אֶבֶן טָבָא לְאַשְׁלָמָא וּבְנַגָרוּת אָעָא לְמֶעְבַּד בְּכָל עִבִידַת אוּמַנְוָן:
וּלְאַלָפָא יְהַב בְּלִבֵּהּ הוּא וְאָהֳלִיאָב בַּר אֲחִיסָמָךְ לְשִׁבְטָא דְדָן:
אַשְׁלֵם עִמְהוֹן חַכִּימוּת לִבָּא לְמֶעְבַּד כָּל עִבִידַת נַגָר וְאָמָן וְצַיָר בְּתִכְלָא וּבְאַרְגְוָנָא בִּצְבַע זְהוֹרִי וּבְבוּצָא וּמַחֵי עָבְדֵי כָּל עֲבִידָא וּמַלְפֵי אוּמַנְוָן:
