וּמַלִיל יְיָ עִם משֶׁה לְמֵימָר
מַלֵל עִם בְּנֵי יִשְׂרָאֵל וְתֵימַר לְהוֹן גְבַר אוֹ אִתְּתָא אֲרֵי יְפָרֵשׁ לְמִדַר נְדַר נְזִירוּ לְמֵיזַר קֳדָם יְיָ
מֵחֲמַר חֲדַת וְעַתִּיק יַנְזִיר חַל דַחֲמַר חֲדַת וְחַל דַחֲמַר עַתִּיק לָא יִשְׁתֵּי וְכָל מַתְרוּת עִנְבִין לָא יִשְׁתֵּי וְעִנְבִין רַטִיבִין וִיבֵשִׁין לָא יֵיכוּל
כֹּל יוֹמֵי נִזְרֵהּ מִכֹּל דְיִתְעֲבֵד מִגֻפְנָא דְחַמְרָא מִפּוּרְצְנִין וְעַד עִצוּרִין לָא יֵיכוּל
כָּל יוֹמֵי נְדַר נִזְרֵהּ מַסְפַּר לָא יֶעְבַּר עַל רֵישֵׁהּ עַד מִשְׁלַם יוֹמַיָא דְיַנְזֵר קֳדָם יְיָ קַדִישׁ יְהֵי מַרְבֵּי פֵרוּעַ שְׂעַר רֵישֵׁהּ
כָּל יוֹמִין דְיַנְזֵר קֳדָם יְיָ עַל נַפְשָׁא דְמֵתָא לָא יֵיעוּלּ
לְאָבוּהִי וּלְאִמֵהּ לְאַחוֹהִי וּלְאַחָתֵהּ לָא יִסְתָּאַב לְהוֹן בְּמוֹתְהּוֹן אֲרֵי כְּלִילָא דֶאֱלָהֵהּ עַל רֵישֵׁהּ
כֹּל יוֹמֵי נִזְרֵהּ קַדִישׁ הוּא קֳדָם יְיָ
וַאֲרֵי יְמוּת מֵתָא עֲלוֹהִי בִּתְכֵּף שָׁלוּ וִיסָאֵב רֵישׁ נִזְרֵהּ וִיגַלַח רֵישֵׁהּ בְּיוֹמָא דִדְכוּתֵהּ בְּיוֹמָא שְׁבִיעָאָה יְגַלְחִנֵהּ
וּבְיוֹמָא תְמִינָאָה יַיְתֵי תַּרְתֵּין שַׁפְנִינִין אוֹ תְרֵין בְּנֵי יוֹנָה לְוָת כַּהֲנָא לִתְרַע מַשְׁכַּן זִמְנָא
וְיַעְבֵּד כַּהֲנָא חַד לְחַטָאתָא וְחַד לַעֲלָתָא וִיכַפֵּר עֲלוֹהִי מִדְחָב עַל מִתָא וִיקַדַשׁ יָת רֵישֵׁהּ בְּיוֹמָא הַּהוּא
וְיַזֵר קֳדָם יְיָ יָת יוֹמֵי נִזְרֵהּ וְיַיְתִי אִמַר בַּר שַׁתֵּהּ לַאֲשָׁמָא וְיוֹמַיָא קַדְמָאֵי יִבְטְלוּן אֲרֵי אִסְתָּאַב נִזְרֵהּ
וְדָא אוֹרַיְתָא דִנְזִירָא בְּיוֹם מִשְׁלַם יוֹמֵי נִזְרֵהּ יַיְתִי יָתֵהּ לִתְרַע מַשְׁכַּן זִמְנָא
וִיקָרֵב יָת קֻרְבָּנֵהּ קֳדָם יְיָ אִמַר בַּר שַׁתֵּהּ שְׁלִים חַד לַעֲלָתָא וְאִמַרְתָּא חֲדָא בַּת שַׁתָּהּ שְׁלֶמְתָּא לְחַטָאתָא וּדְכַר חַד שְׁלִים לְנִכְסַת קוּדְשַׁיָא
וְסַלָא דְפַּטִרִיָא סֻלְּתָּא גְרִיצַן דְפִילָן בִּמְשַׁח וְאִסְפּוֹגִין פַּטִירִין דִמְשִׁיחִין בִּמְשָׁח וּמִנְחַתְהוֹן וְנִסְכֵּיהוֹן
וִיקָרֵב כַּהֲנָא קֳדָם יְיָ וְיַעְבֵּד יָת חַטָאתֵהּ וְיָת עֲלָתֵהּ
וְיָת דִכְרָא יַעְבֵּד נִכְסַת קוּדְשַׁיָא קֳדָם יְיָ עַל סַלָא דְפַטִּירַיָא וְיַעְבֵּד כַּהֲנָא יָת מִנְחָתֵהּ וְיָת נִסְכֵּהּ
וִיגַלַח נְזִירָא בִּתְרַע מַשְׁכַּן זִמְנָא יָת רֵישׁ נִזְרֵהּ וְיִסַב יָת שְׂעַר רֵישׁ נִזְרֵהּ וְיִתֵּן עַל אֶשָׁתָא דִי תְחוֹת דוּדָא דְנִכְסַת קוּדְשַׁיָא
וְיִסַב כַּהֲנָא יָת דְרָעָא בְּשֵׁלָא מִן דִכְרָא וּגְרִצְתָּא פַטִּירְתָּא חֲדָא מִן סַלָא וְאִסְפּוֹג פַּטִיר חָד וְיִתֵּן עַל יְדֵי נְזִירָא בָּתַר דְגַלַח יָת נִזְרֵהּ
וִירִים יָתְהוֹן כַּהֲנָא אֲרָמָא קֳדָם יְיָ קוּדְשָׁא הוּא לְכַהֲנָא עַל חַדְיָא דַאֲרָמוּתָא וְעַל שַׁקָא דְאַפְרָשׁוּתָא וּבָתַר כֵּן יִשְׁתֵּי נְזִירָא חַמְרָא
דָא אוֹרַיְתָא דִנְזִירָא דִי יִדַר קֻרְבָּנֵהּ קֳדָם יְיָ עַל נִזְרֵהּ בַּר מִדְתַדְבֵּק יְדֵהּ כְּפוּם נִדְרֵהּ דִי יִדַר כֵּן יַעְבֵּד עַל אוֹרַיְתָא דְנִזְרֵהּ
וּמַלִיל יְיָ עִם משֶׁה לְמֵימָר
מַלֵל עִם אַהֲרֹן וְעִם בְּנוֹהִי לְמֵימַר כְּדֵין תְּבָרְכוּן יָת בְּנֵי יִשְׂרָאֵל תֵּימְרוּן לְהוֹן
יְבָרְכִינָךְ יְיָ וְיִטְרִינָךְ
יַנְהַר יְיָ שְׁכִנְתֵּהּ לְוָתָךְ וִירַחֵם יָתָךְ
יִסַב יְיָ אַפֵּהּ לְוָתָךְ וִישַׁוֵי לָךְ שְׁלָם
וִישַׁווּן יָת בִּרְכַּת שְׁמִי עַל בְּנֵי יִשְׂרָאֵל וַאֲנָא אֲבָרֵכִנוּן
