וְיָת מַשְׁכְּנָא תַּעְבֵּד עֲשַׂר יְרִיעָן דְבוּץ שְׁזִיר וְתִכְלָא וְאַרְגְּוָנָא וּצְבַע זְהוֹרִי צוּרַת כְּרוּבִין עוֹבַד אֳמָן תַּעְבֵּד יָתְהוֹן:
אֻרְכָּא דִירִיעֲתָא חֲדָא עַשְׂרִין וְתַמְנֵי אַמִין וּפוּתְיָא אַרְבַּע אַמִין דִירִיעֲתָא חֲדָא מְשַׁחְתָּא חֲדָא לְכָל יְרִיעֲתָא:
חֲמֵשׁ יְרִיעָן יְהֶוְיָן מְלַפְּפָן חֲדָא עִם חֲדָא וַחֲמֵשׁ יְרִיעָן מְלַפְּפָן חֲדָא עִם חֲדָא:
וְתַעְבֵּד עֲנוּבִין דְתִכְלָא עַל שִׂפְתָא דִירִיעֲתָא חֲדָא מִסִטְרָא בֵּית לוֹפֵי וְכֵן תַּעְבֵּד בְּשִׂפְתָא דִירִיעֲתָא בְּסִטְרָא דְבֵית לוֹפֵי תִּנְיֵתָא:
חַמְשִׁין עֲנוּבִין תַּעְבֵּד בִּירִיעֲתָא חֲדָא וְחַמְשִׁין עֲנוּבִין תַּעְבֵּד בְּסִטְרָא דִירִיעֲתָא דְבֵית לוֹפֵי תִּנְיָנָא מְכַוְנָן עֲנוּבַיָא חֲדָא לָקֳבֵל חֲדָא:
וְתַעְבֵּד חַמְשִׁין פּוּרְפִין דִדְהָב וּתְלָפֵף יָת יְרִיעֲתָא חֲדָא עִם חֲדָא בְּפּוּרְפַיָא וִיהֵי מַשְׁכְּנָא חָד:
וְתַעְבֵּד יְרִיעָן מְעַזֵי לִפְרָסָא עַל מַשְׁכְּנָא חֲדָא עֶסְרֵי יְרִיעָן תַּעְבֵּד יָתְהוֹן:
אֻרְכָּא דִירִיעֲתָא חֲדָא תְּלָתִין אַמִין וּפוּתְיָא אַרְבַּע אַמִין דִירִיעֲתָא חֲדָא מְשַׁחְתָּא חֲדָא לַחֲדָא עֶסְרֵי יְרִיעָן:
וּתְלָפֵף יָת חֲמֵשׁ יְרִיעָן לְחוֹד וְיָת שִׁית יְרִיעָן לְחוֹד וְתֵיעוּף יָת יְרִיעֲתָא שְׁתִיתֵיתָא לָקֳבֵל אַפֵּי מַשְׁכְּנָא:
וְתַעְבֵּד חַמְשִׁין עֲנוּבִין עַל שִׂפְתָא דִירִיעֲתָא חֲדָא בְּסִטְּרָא בֵּית לוֹפֵי וְחַמְשִׁין עֲנוּבִין עַל שִׂפְתָא דִירִיעֲתָא דְבֵית לוֹפֵי תִּנְיֵתָא:
וְתַעְבֵּד פּוּרְפִין דִנְחָשָׁא חַמְשִׁין וְתָעֵיל יָת פּוּרְפַיָא בַּעֲנוּבַיָא וּתְלָפֵף יָת מַשְׁכְּנָא וִיהֵי חָד:
וְסִרְחָא דְיַתִּירָא בִּירִיעַת מַשְׁכְּנָא פַּלְגוּת יְרִיעֲתָא דְיַתִּירָא תִּסְרַח עַל אֲחוֹרֵי מַשְׁכְּנָא:
וְאַמְתָא מִכָּא וְאַמְתָא מִכָּא בִּדְיַתִּירָא בְּאֻרְכָּא יְרִיעֲתָא דְמַשְׁכְּנָא יְהֵי סְרִיחַ עַל סִטְרֵי מַשְׁכְּנָא מִכָּא וּמִכָּא לְכַסָיוּתֵהּ:
וְתַעְבֵּד חוֹפָאָה לִפְרָסָא מַשְׁכֵי דְדִכְרֵי מְסַמְקֵי וְחוֹפָאָה מַשְׁכֵי דְסַסְגוֹנָא מִלְעֵלָא:
וְתַעְבֵּד יָת דַפַּיָא לְמַשְׁכְּנָא דְאָעֵי שִטִין קָיְמִין:
עֲשַׂר אַמִין אֻרְכָּא דְדַפָּא וְאַמְתָא וּפַלְגוּת אַמְתָא פּוּתְיָא דְדַפָּא חָד:
תַּרְתֵּין צִירִין לְדַפָּא חַד מְשַׁלְבִין חֲדָא לָקֳבֵל חֲדָא כֵּן תַּעְבֵּד לְכֹל דַפֵּי מַשְׁכְּנָא:
וְתַעְבֵּד יָת דַפַּיָא לְמַשְׁכְּנָא עַסְרִין דַפִּין לְרוּחַ עֵבַר דְרוֹמָא:
וְאַרְבְּעִין סַמְכִין דִכְסַף תַּעְבֵּד תְּחוֹת עַסְרִין דַפִּין תְּרֵין סַמְכִין תְּחוֹת דַפָּא חַד לִתְרֵין צִירוֹהִי וּתְרֵין סַמְכִין תְּחוֹת דַפָּא חַד לִתְרֵין צִירוֹהִי:
וְלִסְטַר מַשְׁכְּנָא תִּנְיֵתָא לְרוּחַ צִפּוּנָא עֶסְרִין דַפִּין:
וְאַרְבְּעִין סַמְכֵיהוֹן דִכְסַף תְּרֵין סַמְכִין תְּחוֹת דַפָּא חַד וּתְרֵין סַמְכִין תְּחוֹת דַפָּא חָד:
וְלִסְיָפֵי מַשְׁכְּנָא מַעַרְבָא תַּעְבֵּד שִׁתָּא דַפִּין:
וּתְרֵין דַפִּין תַּעְבֵּד לְזָוְיַת מַשְׁכְּנָא בְּסוֹפֵיהוֹן:
וִיהוֹן מְכַוְנִין מִלְרַע וְכַחֲדָא יְהוֹן מְכַוְנִין עַל רֵישֵׁהּ לְעִזְקְתָא חֲדָא כֵּן יְהֵי לְתַרְוֵיהוֹן לִתְרֵין זָוְיָן יְהוֹן:
וִיהוֹן תְּמַנְיָא דַפִּין וְסַמְכֵיהוֹן דִכְסַף שִׁתָּא עֲשַׂר סַמְכִין תְּרֵין סַמְכִין תְּחוֹת דַפָּא חַד וּתְרֵין סַמְכִין תְּחוֹת דַפָּא חָד:
וְתַעְבֵּד עַבְּרֵי דְאָעֵי שִׁטִין חַמְשָׁא לְדַפֵּי סְטַּר מַשְׁכְּנָא חָד:
וְחַמְשָׁא עַבְּרִין לְדַפֵּי סְטַר מַשְׁכְּנָא תִּנְיֵתָא וְחַמְשָׁא עַבְּרִין לְדַפֵּי סְטַּר מַשְׁכְּנָא לְסוֹפֵיהוֹן מַעַרְבָא:
וְעַבְּרָא מְצִיעָאָה בְּגוֹ דַפַּיָא מַעְבַר מִן סְיָפֵי לִסְיָפֵי:
וְיָת דַפַּיָא תַּחְפֵּי דַהֲבָא וְיָת עִזְקָתְהוֹן תַּעְבֵּד דַהֲבָא אַתְרָא לְעַבְרַיָא וְתַחְפֵּי יָת עַבְּרַיָא דַהֲבָא:
וּתְקִים יָת מַשְׁכְּנָא כְּהִלְכְתֵהּ דִי אִתְחֲזֵיתָא בְּטוּרָא:
וְתַעְבֵּד פָּרֻכְתָּא תִּכְלָא וְאַרְגְוָנָא וּצְבַע זְהוֹרִי וּבוּץ שְׁזִיר עוֹבַד אֳמָן יַעְבֵּד יָתַהּ צוּרַת כְּרוּבִין:
וְתִתֵּן יָתַהּ עַל אַרְבְּעָא עַמוּדֵי שִׁטִין מְחַפְּיָן דַהֲבָא וָוֵיהוֹן דַהֲבָא עַל אַרְבְּעָא סַמְכֵי דִכְסָף:
וְתִתֵּן יָת פָּרֻכְתָּא תְּחוֹת פּוּרְפַיָא וְתָעֵיל תַּמָן מִגָּו לְפָרֻכְתָּא יָת אֲרוֹנָא דְסַהֲדוּתָא וְתַפְרֵשׁ פָּרֻכְתָּא לְכוֹן בֵּין קוּדְשָׁא וּבֵין קֹדֶשׁ קוּדְשַׁיָא:
וְתִתֵּן יָת כַּפֻּרְתָּא עַל אֲרוֹנָא דְסַהֲדוּתָא בְּקֹדֶשׁ קוּדְשַׁיָא:
וּתְשַׁוִי יָת פָּתוֹרָא מִבָּרָא לְפָרֻכְתָּא וְיָת מְנַרְתָּא לָקֳבֵל פָּתוֹרָא עַל צִדָּא דְמַשְּׁכְּנָא דָרוֹמָא וּפָתוֹרָא תִתֵּן עַל סְטַר צִפּוּנָא:
וְתַעְבֵּד פְּרָסָא לִתְרַע מַשְׁכְּנָא תִּכְלָא וְאַרְגְוָנָא וּצְבַע זְהוֹרִי וּבוּץ שְׁזִיר עוֹבַד צַיָר:
וְתַעְבֵּד לִפְרָסָא חַמְשָׁא עַמוּדֵי שִׁטִין וְתַחֲפֵי יָתְהוֹן דַהֲבָא וָוֵיהוֹן דַהֲבָא וְתַתִּיךְ לְהוֹן חַמְשָׁא סַמְכֵי נְחָשָׁא:
