וּמַלִיל יְיָ עִם משֶׁה לְמֵימָר
עֲבֵיד לָךְ תַּרְתֵּין חֲצוֹצְרָן דִכְסַף נְגִיד תַּעְבֵּד יַתְהוֹן וִיהֶוְיָן לָךְ לְעַרְעָא כְנִשְׁתָּא וּלְאַטָלָא יָת מַשִׁרְיָתָא
וְיִתְקְעוּן בְּהוֹן וְיִזְדַמְנוּן לְוָתָךְ כָּל כְּנִשְׁתָּא לִתְרַע מַשְׁכַּן זִמְנָא
וְאִם בַּחֲדָא יִתְקְעוּן וְיִזְדַמְנוּן לְוָתָךְ רַבְרְבַיָא רֵישֵׁי אַלְפַיָא דְיִשְׂרָאֵל
וְתִתְקְעוּן יַבֶּבְתָּא וְיִטְלוּן מַשִׁרְיָתָא דְשָׁרָן קִדוּמָא
וְתִתְקְעוּן יַבֶּבְתָּא תִּנְיָנוּת וְיִטְלוּן מַשִׁרְיָתָא דְשָׁרָן דָרוֹמָא יַבָּבָא יִתְקְעוּן לְמַטְלָנֵיהוֹן
וּבְמִכְנַשׁ יָת קְהָלָא תִּתְקְעוּן וְלָא תְיַבְּבוּן
וּבְנֵי אַהֲרֹן כַּהֲנַיָא יִתְקְעוּן בַּחֲצוֹצְרָתָא וִיהֶוְיָן לְכוֹן לִקְיַם עֲלָם לְדָרֵיכוֹן
וַאֲרֵי תֵעֲלוּן לְאַגָחָא קְרָבָא בְּאַרְעַכוֹן עַל מְעִיקֵי דִמְעִיקִין לְכוֹן וּתְיַבְּבוּן בַּחֲצוֹצְרָתָא וְיֵיעוֹל דוּכְרָנֵיכוֹן לְטָּבָא קֳדָם יְיָ אֱלָהָכוֹן וְתִתְפָּרְקוּן מִסַנְאֵיכוֹן
וּבְיוֹם חֶדְוַתְכוֹן וּבְמוֹעֲדֵיכוֹן וּבְרֵישֵׁי יַרְחֵיכוֹן וְתִתְקְעוּן בַּחֲצוֹצְרָתָא עַל עֲלָוַתְכוֹן וְעַל נִכְסַת קוּדְשֵׁיכוֹן וִיהֶוְיָן לְכוֹן לְדוּכְרָנָא קֳדָם אֱלָהָכוֹן אֲנָא יְיָ אֱלָהָכוֹן
וַהֲוָה בְּשַׁתָּא תִנְיֵתָא בְּיַרְחָא תִנְיָנָא בְּעַשְׂרִין לְיַרְחָא אִסְתַּלֵק עֲנָנָא מֵעִלַוֵי מַשְׁכְּנָא דְסַהֲדוּתָא
וּנְטָלוּ בְנֵי יִשְׂרָאֵל לְמַטְלָנֵיהוֹן מִמַדְבְּרָא דְסִינַי וּשְׁרָא עֲנָנָא בְּמַדְבְּרָא דְפָארָן
וּנְטָלוּ בְקַדְמֵיתָא עַל מֵימְרָא דַיְיָ בִּידָא דְמשֶׁה
וּנְטַל טֵקַס מַשְׁרִית בְּנֵי יְהוּדָה בְּקַדְמֵיתָא לְחֵילֵיהוֹן וְעַל חֵילֵהּ נַחְשׁוֹן בַּר עַמִינָדָב
וְעַל חֵילָא דְשִׁבְטָא דִבְנֵי יִשָׂשׂכָר נְתַנְאֵל בַּר צוּעָר
וְעַל חֵילָא דְשִׁבְטָא דִבְנֵי זְבוּלֻן אֱלִיאָב בַּר חֵלוֹן
וּמִתְפְּרַק מַשְׁכְּנָא וְנָטְלִין בְּנֵי גֵרְשׁוֹן וּבְנֵי מְרָרִי נָטְלֵי מַשְׁכְּנָא
וּנְטַל טֵקַס מַשְׁרִית רְאוּבֵן לְחֵילֵיהוֹן וְעַל חֵילֵהּ אֱלִיצוּר בַּר שְׁדֵיאוּר
וְעַל חֵילָא דְשִׁבְטָּא דִבְנֵי שִׁמְעוֹן שְׁלֻמִיאֵל בַּר צוּרִישַׁדָי
וְעַל חֵילָא דְשִׁבְטָא דִבְנֵי גָד אֶלְיָסָף בַּר דְעוּאֵל
וְנָטְלִין בְּנֵי קְהָת נָטְלֵי מַקְדְשָׁא וּמְקִימִין יָת מַשְׁכְּנָא עַד מֵיתֵיהוֹן
וּנְטַּל טֵקַס מַשְׁרִית בְּנֵי אֶפְרַיִם לְחֵילֵיהוֹן וְעַל חֵילֵהּ אֱלִישָׁמָע בַּר עַמִהוּד
וְעַל חֵילָא דְשִׁבְטָא דִבְנֵי מְנַשֶׁה גַמְלִיאֵל בַּר פְּדָהצוּר
וְעַל חֵילָא דְשִׁבְטָא דִבְנֵי בִנְיָמִן אֲבִידָן בַּר גִדְעוֹנִי
וּנְטַל טֵקַס מַשְׁרִית בְּנֵי דָן מַכְנִישׁ לְכָל מַשִׁרְיָתָא לְחֵילֵיהוֹן וְעַל חֵילֵהּ אֲחִיעֶזֶר בַּר עַמִישַׁדָי
וְעַל חֵילָא דְשִׁבְטָא דִבְנֵי אָשֵׁר פַּגְעִיאֵל בַּר עָכְרָן
וְעַל חֵילָא דְשִׁבְטָא דִבְנֵי נַפְתָּלִי אֲחִירַע בַּר עֵינָן
אִלֵין מַטְלָנֵי בְּנֵי יִשְׂרָאֵל לְחֵילֵיהוֹן וּנְטָלוּ
וַאֲמַר משֶׁה לְחוֹבָב בַּר רְעוּאֵל מִדְיָנָאָה חֲמוֹהִי דְמשֶׁה נָטְלִין אֲנַחְנָא לְאַתְרָא דִי אֲמַר יְיָ יָתֵהּ אֶתֵּן לְכוֹן אִיתָא עִמָנָא וְנוֹטֵיב לָךְ אֲרֵי יְיָ מַלֵיל לְאַיְתָאָה טָּבָא עַל יִשְׂרָאֵל
וַאֲמַר לֵהּ לָא אֵזִל אֶלָהֵן לְאַרְעִי וּלְיַלָדוּתִי אֵזִל
וַאֲמַר לָא כְעַן תִּשְׁבּוֹק יָתָנָא אֲרֵי עַל כֵּן יְדַעַתְּ כַּד הֲוֵינָא שָׁרַן בְּמַדְבְּרָא וּגְבוּרָן דְאִתְעֲבִידָן לָנָא חֲזֵיתָא בְעֵינָיִךְ
וִיהֵי אֲרֵי תֵזִל עִמָנָא וִיהֵי טָבָא הַהוּא דִי יוֹטִיב יְיָ עִמָנָא וְנוֹטִיב לָךְ
וּנְטָּלוּ מִטוּרָא דְאִתְגְלֵי עֲלוֹהִי יְקָרָא דַיְיָ מַהֲלַךְ תְּלָתָא יוֹמִין וַאֲרוֹן קְיָמָא דַיְיָ נְטַּל קֳדָמֵיהוֹן מַהֲלַךְ תְּלָתָא יוֹמִין לְאַתְקְנָאָה לְהוֹן אֲתַר בֵּית מֵישְׁרֵי
וַעֲנַן יְקָרָא דַיְיָ מַטֵיל עֲלֵיהוֹן בִּימָמָא בְּמֵטַלְהוֹן מִן מַשְׁרִיתָא
וַהֲוָה בְּמֵטַל אֲרוֹנָא וַאֲמַר משֶׁה אִתְגְלֵי יְיָ וְיִתְבַּדְרוּן סַנְאָיךְ וִיעַרְקוּן בַּעֲלֵּי דְבָבָךְ מִן קֳדָמָךְ
וּבְמַשְׁרוֹהִי אֲמַר תּוּב יְיָ שְׁרֵי בִיקָרָךְ בְּגוֹ רִבְבָן אַלְפַיָא דְיִשְׂרָאֵל
