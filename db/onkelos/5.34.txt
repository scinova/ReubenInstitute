וּסְלֵיק מֹשֶׁה מִמֵישְׁרָא דְמוֹאָב לְטוּרָא דִנְבוֹ רֵישׁ רָמָתָא דִי עַל אַפֵּי יְרֵחוֹ וְאַחְזְיֵהּ יְיָ יָת כָּל אַרְעָא יָת גִלְעָד עַד דָן
וְיָת כָּל נַפְתָּלִי וְיָת אַרְעָא דְאֶפְרַיִם וּמְנַשֶׁה וְיָת כָּל אַרְעָא דִיהוּדָה עַד יַמָא מַעַרְבָא
וְיָת דָרוֹמָא וְיָת מֵישְׁרָא בִּקְעֲתָא דִירֵחוֹ קַרְתָּא דְדִקְלַיָא עַד צוֹעַר
וַאֲמַר יְיָ לֵהּ דָא אַרְעָא דִי קַיֵמִית לְאַבְרָהָם לְיִצְחָק וּלְיַעֲקֹב לְמֵימָר לִבְנָיךְ אֶתְּנִנַהּ אַחֲזִיתָךְ בְּעֵינָיךְ וּלְתַמָן לָא תְעִבַּר
וּמִית תַּמָן מֹשֶׁה עַבְדָא דַיְיָ בְּאַרְעָא דְמוֹאָב עַל מֵימְרָא דַיְיָ
וּקְבַר יָתֵהּ בְּחֵילָתָא בְּאַרְעָא דְמוֹאָב לְקֳבֵל בֵּית פְּעוֹר וְלָא יְדַע אֱנַשׁ יָת קְבֻרְתֵּה עַד יוֹמָא הָדֵין
וּמֹשֶׁה בַּר מְאָה וְעַשְׂרִין שְׁנִין כַּד מִית לָא כְהַת עֵינוֹהִי וְלָא שְׁנָא זִיו יְקָרָא דְאַפּוֹהִי
וּבְכוֹ בְנֵי יִשְׂרָאֵל יָת מֹשֶׁה בְּמֵישְׁרַיָא דְמוֹאָב תְּלָתִין יוֹמִין וּשְׁלִימוּ יוֹמֵי בָכִיתָא אֶבְלָא דְמֹשֶׁה
וִיהוֹשֻׁעַ בַּר נוּן מְלֵי רוּחַ חָכְמְתָא אֲרֵי סְמַךְ מֹשֶׁה יָת יְדוֹהִי עֲלוֹהִי וְקַבִּילוּ מִנֵהּ בְּנֵי יִשְׂרָאֵל וַעֲבָדוּ כְּמָא דִי פַּקִיד יְיָ יָת מֹשֶׁה
וְלָא קָם נְבִיָא עוֹד בְּיִשְׂרָאֵל כְּמֹשֶׁה דִי אִתְגְלִי לֵהּ יְיָ אַפִּין בְּאַפִּין
לְכָל אָתַיָא וּמוֹפְתַיָא דִי שָׁלְחֵהּ יְיָ לְמֶעְבַּד בְּאַרְעָא דְמִצְרָיִם לְפַרְעֹה וּלְכָל עַבְדוֹהִי וּלְכָל אַרְעֵהּ
וּלְכֹל יְדָא תַקֶפְתָּא וּלְכֹל חֵזְוָנָא רַבָּא דִי עֲבַד מֹשֶׁה לְעֵינֵי כָּל יִשְׂרָאֵל
