וּלְמשֶׁה אֲמַר סַק לִקֳדָם יְיָ אַתְּ וְאַהֲרֹן נָדָב וַאֲבִיהוּא וְשִׁבְעִין מִסָבֵי יִשְׂרָאֵל וְתִסְגְדוּן מֵרָחִיק:
וְיִתְקְרֵב משֶׁה בִלְחוֹדוֹהִי לִקֳדָם יְיָ וְאִנוּן לָא יִתְקָרְבוּן וְעַמָא לָא יִסְקוּן עִמֵהּ:
וַאֲתָא משֶׁה וְאִשְׁתָּעִי לְעַמָא יָת כָּל פִּתְגָמַיָא דַיְיָ וְיָת כָּל דִינַיָא וַאֲתֵיב כָּל עַמָא קָלָא חַד וַאֲמָרוּ כָּל פִּתְגָמַיָא דִי מַלִיל יְיָ נַעְבֵּד:
וּכְתַב משֶׁה יָת כָּל פִּתְגָמַיָא דַיְיָ וְאַקְדֵם בְּצַפְרָא וּבְנָא מַדְבְּחָא בְּשִׁפּוֹלֵי טוּרָא וְתַרְתָּא עֶשְׂרֵי קָמָן לִתְרֵי עֲסַר שִׁבְטַיָא דְיִשְׂרָאֵל:
וּשְׁלַח יָת בּוּכְרֵי בְּנֵי יִשְׂרָאֵל וְאַסִיקוּ עֲלָוָן וְקָרִיבוּ נִכְסַת קוּדְשִׁין קֳדָם יְיָ תּוֹרִין:
וּנְסִיב משֶׁה פַּלְגוּת דְמָא וְשַׁוִי בְּמִזְרְקַיָא וּפַלְגוּת דְמָא זְרַק עַל מַדְבְּחָא:
וּנְסִיב סִפְרָא דִקְיָמָא וּקְרָא קֳדָם עַמָא וַאֲמָרוּ כֹּל דִי מַלִיל יְיָ נַעְבֵּד וּנְקַבֵּל:
וּנְסִיב משֶׁה יָת דְמָא וּזְרַק עַל מַדְבְּחָא לְכַפָּרָא עַל עַמָא וַאֲמַר הָא דֵין דָם קְיָמָא דִגְזַר יְיָ עִמְכוֹן עַל כָּל פִּתְגָמַיָא הָאִלֵין:
וּסְלֵיק משֶׁה וְאַהֲרֹן נָדָב וַאֲבִיהוּא וְשִׁבְעִין מִסָבֵי יִשְׂרָאֵל:
וַחֲזוֹ יָת יְקַר אֱלָהָא דְיִשְׂרָאֵל וּתְחוֹת כָּרְסָא יְקָרֵהּ כְּעוֹבַד אֶבֶן טָבָא וּכְמֶחֱזֵי שְׁמַיָא לְבָרִירוּ:
וּלְרַבְרְבֵי בְּנֵי יִשְׂרָאֵל לָא הֲוָה נִזְקָא וַחֲזוֹ יָת יְקָרָא דַיְיָ וַהֲווֹ חָדָן בְּקָרְבָּנֵיהוֹן דְאִתְקַבָּלוּ בְּרַעֲוָא כְּאִלוּ אָכְלִין וְשָׁתָן:
וַאֲמַר יְיָ לְמשֶׁה סַק לִקֳדָמַי לְטוּרָא וֶהֱוֵי תַמָן וְאֶתֵּן לָךְ יָת לוּחֵי אַבְנָא וְאוֹרַיְתָא וְתַפְקֶדְתָּא דִכְתָבִית לְאַלוּפֵיהוֹן:
וְקָם משֶׁה וִיהוֹשֻׁעַ מְשֻׁמְשָׁנֵהּ וּסְלֵיק משֶׁה לְטוּרָא דְאִתְגְלִי עֲלוֹהִי יְקָרָא דַיְיָ:
וּלְסָבַיָא אֲמַר אוֹרִיכוּ לָנָא הָכָא עַד דִנְתוּב לְוָתְכוֹן וְהָא אַהֲרֹן וְחוּר עִמְכוֹן מָן דְאִית לֵהּ דִינָא יִתְקְרֵב לִקֳדָמֵיהוֹן:
וּסְלֵק משֶׁה לְטוּרָא וַחֲפָא עֲנָנָא יָת טוּרָא:
וּשְׁרָא יְקָרָא דַיְיָ עַל טוּרָא דְסִינַי וַחֲפָהִי עֲנָנָא שִׁתָּא יוֹמִין וּקְרָא לְמשֶׁה בְּיוֹמָא שְׁבִיעָאָה מִגוֹ עֲנָנָא:
וְחֵיזוּ יְקָרָא דַיְיָ כְּחֵיזוּ אֶשָׁא אָכְלָא בְּרֵישׁ טוּרָא לְעֵינֵי בְּנֵי יִשְׂרָאֵל:
וְעַל משֶׁה בְּגוֹ עֲנָנָא וּסְלֵיק לְטוּרָא וַהֲוָה משֶׁה בְּטּוּרָא אַרְבְּעִין יְמָמִין וְאַרְבְּעִין לֵילָוָן:
