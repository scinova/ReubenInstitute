וַהֲוָה כָל אַרְעָא לִישָׁן חָד וּמַמְלָל חָד
וַהֲוָה בְּמִטַּלְהוֹן בְּקַדְמֵיתָא וְאַשְׁכָּחוּ בִקְעֲתָא בְּאַרְעָא דְבָבֶל וִיתִיבוּ תַמָּן
וַאֲמָרוּ גְּבַר לְחַבְרֵהּ הָבוּ נִרְמֵי לִבְנִין וְנוֹקְדִינוּן בְּנוּרָא יוֹקַדְתָּא (נ''י בְּנוּרָא) וַהֲוַת לְהוֹן לְבִנְתָּא לְאַבְנָא וְחֵימָרָא הֲוָה לְהוֹן לְשִׁיעַ
וַאֲמָרוּ הָבוּ נִבְנֵי לָנָא קַרְתָּא וּמַגְדְּלָא וְרֵישֵׁהּ מָטֵי עַד צֵית שְׁמַיָּא וְנַעְבֵּיד לָנָא שׁוּם דִּילְמָא נִתְבַּדַּר עַל אַפֵּי כָל אַרְעָא
וְאִתְגְּלֵי יְיָ לְאִתְפְּרָעָא עַל עוֹבָדֵי קַרְתָּא וּמַגְדְּלָא דִּי בְנוֹ בְּנֵי אֱנָשָׁא
וַאֲמַר יְיָ הָא עַמָּא חַד וְלִישָׁן חַד לְכֻלְּהוֹן וְדֵין דְּשָׁרִיוּ לְמֶעְבָּד וּכְעַן לָא יִתִּמְנַע מִנְּהוֹן כֹּל דִחַשִּׁיבוּ לְמֶעְבָּד
הָבוּ נִתְגְּלִי וּנְבַלְבֵּל תַּמָּן לִישָׁנְהוֹן דְּלָא יִשְׁמְעוּן (גְּבַר) אֱנַשׁ לִישָׁן חַבְרֵהּ
וּבַדַּר יְיָ יָתְהוֹן מִתַּמָּן עַל אַפֵּי כָל אַרְעָא וּמְנָעוּ לְמִבְנֵי (נ''י וְאִתְמְנָע מִלְּמִבְנֵי) קַרְתָּא
עַל כֵּן קְרָא שְׁמַהּ בָּבֶל אֲרֵי תַמָּן בַּלְבֵּל יְיָ לִישַׁן כָּל אַרְעָא וּמִתַּמָּן בַּדָּרִנּוּן יְיָ עַל אַפֵּי כָּל אַרְעָא
אִלֵּין תּוּלְדַת שֵׁם שֵׁם בַּר מְאָה שְׁנִין וְאוֹלִיד יָת אַרְפַּכְשָׁד תַּרְתֵּין שְׁנִין בָּתַר טּוֹפָנָא
וַחֲיָא שֵׁם בָּתַר דְּאוֹלִיד יָת אַרְפַּכְשָׁד חֲמֵשׁ מְאָה שְׁנִין וְאוֹלִיד בְּנִין וּבְנָן
וְאַרְפַּכְשַׁד חֲיָא תְּלָתִין וַחֲמֵשׁ שְׁנִין וְאוֹלִיד יָת שָׁלַח
וַחֲיָא אַרְפַּכְשַׁד בָּתַר דְּאוֹלִיד יָת שֶׁלַח אַרְבַּע מְאָה וּתְלָת שְׁנִין וְאוֹלִיד בְּנִין וּבְנָן
וְשֶׁלַח חֲיָא תְּלָתִין שְׁנִין וְאוֹלִיד יָת עֵבֶר
וַחֲיָא שֶׁלַח בָּתַר דְּאוֹלִיד יָת עֵבֶר אַרְבַּע מְאָה וּתְלָת שְׁנִין וְאוֹלִיד בְּנִין וּבְנָן
וַחֲיָא עֵבֶר תְּלָתִין וְאַרְבַּע שְׁנִין וְאוֹלִיד יָת פָּלֶג
וַחֲיָא עֵבֶר בָּתַר דְּאוֹלִיד יָת פֶּלֶג אַרְבַּע מְאָה וּתְּלָתִין שְׁנִין וְאוֹלִיד בְּנִין וּבְנָן
וַחֲיָא פֶלֶג תְּלָתִין שְׁנִין וְאוֹלִיד יָת רְעוּ
וַחֲיָא פֶלֶג בָּתַר דְּאוֹלִיד יָת רְעוּ מָאתָן וּתְשַׁע שְׁנִין וְאוֹלִיד בְּנִין וּבְנָן
וַחֲיָא רְעוּ תְּלָתִין וְתַרְתֵּין שְׁנִין וְאוֹלִיד יָת שְׂרוּג
וַחֲיָא רְעוּ בָּתַר דְּאוֹלִיד יָת שְׂרוּג מָאתָן וּשְׁבַע שְׁנִין וְאוֹלִיד בְּנִין וּבְנָן
וַחֲיָא שְׂרוּג תְּלָתִין שְׁנִין וְאוֹלִיד יָת נָחוֹר
וַחֲיָא שְׂרוּג בָּתַר דְּאוֹלִיד יָת נָחוֹר מָאתָן שְׁנִין וְאוֹלִיד בְּנִין וּבְנָן
וַחֲיָא נָחוֹר עֶשְׂרִין וּתְשַׁע שְׁנִין וְאוֹלִיד יָת תָּרַח
וַחֲיָא נָחוֹר בָּתַר דְּאוֹלִיד יָת תֶּרַח מְאָה וּתְשַׁע עֲשַׂר שְׁנִין וְאוֹלִיד בְּנִין וּבְנָן
וַחֲיָא תֶרַח שִׁבְעִין שְׁנִין וְאוֹלִיד יָת אַבְרָם יָת נָחוֹר וְיָת הָרָן
וְאִלֵּין תּוּלְדַת תֶּרַח תֶּרַח אוֹלִיד יָת אַבְרָם יָת נָחוֹר וְיָת הָרָן וְהָרָן אוֹלִיד יָת לוֹטּ
וּמִית הָרָן עַל אַפֵּי תֶּרַח אֲבוּהִי בְּאַרְעָא דְּיַלָּדוּתֵהּ בְּאוּרָא דְכַסְדָּאֵי
וּנְסִיב אַבְרָם וְנָחוֹר לְהוֹן נְשִׁין שׁוּם אִתַּת אַבְרָם שָׂרָי וְשׁוּם אִתַּת נָחוֹר מִלְכָּה בַּת הָרָן אֲבוּהָא דְּמִלְכָּה וַאֲבוּהָא דְּיִסְכָּה
וַהֲוַת שָׂרַי עֲקָרָה לֵית לָהּ וְלָד
וּדְבַר תֶּרַח יָת אַבְרָם בְּרֵהּ וְיָת לוֹט בַּר הָרָן בַּר בְּרֵהּ וְיָת שָׂרַי כַּלָּתֵהּ אִתַּת אַבְרָם בְּרֵהּ וּנְפָקוּ עִמְּהוֹן מֵאוּרָא דְּכַסְדָּאֵי לְמֵיזַל לְאַרְעָא דִּכְנָעַן וַאֲתוֹ עַד חָרָן וִיתִיבוּ תַמָּן
וַהֲווֹ יוֹמֵי תֶרַח מָאָתן וַחֲמֵשׁ שְׁנִין וּמִית תֶּרַח בְּחָרָן
