וַאֲמַר יְיָ לְנֹחַ: _עוּל_ אַתְּ וְכָל +אֱנַשׁ+ בֵּיתָךְ לְתֵבוֹתָא, אֲרֵי יָתָךְ חֲזֵיתִי זַכַּאי קֳדָמַי בְּדָרָא הָדֵין.
מִכֹּל בְּעִירָא דַכְיָא תִּסַּב לָךְ שַׁבְעָא-שַׁבְעָא, _דְּכַר וְנוּקְבָא_, וּמִן בְּעִירָא _דְּלָא אִיתָהָא_ דַכְיָא הִיא - תְּרֵין, _דְּכַר וְנוּקְבָא_.
אַף מֵעוֹפָא דִשְׁמַיָּא שַׁבְעָא-שַׁבְעָא, דְּכַר וְנוּקְבָא, _לְקַיָּמָא_ זַרְעָא עַל אַפֵּי כָל אַרְעָא.
אֲרֵי לִ+זְמַן+ יוֹמִין עוֹד שַׁבְעָא, אֲנָא _מָחֵת מִטְרָא_ עַל אַרְעָא, אַרְבְּעִין _יְמָמִין_ וְאַרְבְּעִין לֵילָוָן; וְאֶמְחֵי יָת כָּל יְקוּמָא דִּי עַבְדֵת מֵעַל אַפֵּי אַרְעָא.
וַעֲבַד נֹחַ כְּכֹל דִּי פַקְּדֵהּ יְיָ.
וְנֹחַ בַּר שִׁתְּ מְאָה שְׁנִין, וְטּוֹפָנָא הֲוָה, מַיָּא עַל אַרְעָא.
_וְעָאל_ נֹחַ, וּבְנוֹהִי, וְאִתְּתֵהּ, וּנְשֵׁי בְנוֹהִי עִמֵּהּ, לְתֵבוֹתָא, מִן קֳדָם מֵי _טוֹפָנָא_.
מִן בְּעִירָא דַּכְיָא, וּמִן בְּעִירָא דִּי לֵיתָהָא דַּכְיָא, וּמִן עוֹפָא, וְכֹל דִּי רָחֵשׁ עַל אַרְעָא -
תְּרֵין-תְּרֵין _עַלּוּ_ לְוַת נֹחַ לְתֵבוֹתָא, דְּכַר וְנוּקְבָא, כְּמָא דִּי פַּקֵּיד יְיָ יָת נֹחַ.
וַהֲוָה לִ+זְמַן+ שִׁבְעַת יוֹמִין, וּמֵי טוֹפָנָא הֲווֹ עַל אַרְעָא.
בִּשְׁנַת שִׁתְּ מְאָה שְׁנִין לְחַיֵּי נֹחַ, בְּיַרְחָא תִּנְיָנָא, בְּשִׁבְעַת עַשְׂרָא יוֹמָא לְיַרְחָא - בְּיוֹמָא הָדֵין אִתְבְּזָעוּ כָּל מַבּוּעֵי תְּהוֹמָא רַבָּא, _וְכַוֵּי_ שְׁמַיָּא אִתְפַּתָּחוּ.
וַהֲוָה מִטְרָא +נָחֵת+ עַל אַרְעָא, אַרְבְּעִין _יְמָמִין_, וְאַרְבְּעִין לֵילָוָן.
_בִּכְרַן_ יוֹמָא הָדֵין _עָאל_ נֹחַ, וְשֵׁם וְחָם וָיֶפֶת בְּנֵי נֹחַ, וְאִתַּת נֹחַ, וּתְלָתָא נְשֵׁי בְנוֹהִי עִמְהוֹן, לְתֵבוֹתָא.
אִנּוּן, וְכָל חַיְתָא לִזְנַהּ, וְכָל בְּעִירָא לְזְנַהּ, וְכָל רִחֲשָׁא דְּרָחֵשׁ עַל אַרְעָא לִזְנוֹהִי, וְכָל עוֹפָא לִזְנוֹהִי, כֹּל צִפַּר, כָּל _דְּפָרָח_.
_וְעַלּוּ_ _עִם_ נֹחַ לְתֵבוֹתָא, תְּרֵין-תְּרֵין מִכָּל בִּשְׂרָא, דִּי בֵהּ רוּחָא דְּחַיֵּי.
_וְעָלַיָּא_ - דְּכַר וְנוּקְבָא - מִכָּל בִּשְׂרָא _עַלּוּ_, כְּמָא דִּי פַּקֵּיד יָתֵהּ יְיָ, _וַאֲגֵן_ יְיָ +(בְּמֵימְרֵהּ)+ עֲלוֹהִי.
וַהֲוָה _טוֹפָנָא_ אַרְבְּעִין _יְמָמִין_ עַל אַרְעָא, וּסְגִיאוּ מַיָּא, וּנְטָלוּ יָת תֵּבוֹתָא, וְאִתָּרָמַת מֵעַל אַרְעָא.
_וּתְקִיפוּ_ מַיָּא, וּסְגִיאוּ לַחֲדָא עַל אַרְעָא; וּמְהַלְּכָא תֵבוֹתָא עַל אַפֵּי מַיָּא.
וּמַיָּא _תְּקִיפוּ_ לַחֲדָא-לַחֲדָא עַל אַרְעָא, _וְאִתְחֲפִיאוּ_ כָּל טוּרַיָּא רָמַיָּא, דִּי תְחוֹת כָּל שְׁמַיָּא.
חֲמֵשׁ עַשְׂרֵי אַמִּין מִלְעֵלָּא, _תְּקִיפוּ_ מַיָּא, _וְאִתְחֲפִיאוּ_ טוּרַיָּא.
_וּמִית_ כָּל בִּשְׂרָא דְּרָחֵשׁ עַל אַרְעָא: בְּעוֹפָא, וּבִבְעִירָא, וּבְחַיְתָא, וּבְכָל רִחֲשָׁא דְּרָחֵשׁ עַל אַרְעָא, וְכֹל אֱנָשָׁא.
כֹּל דִּי נִשְׁמְתָא רוּחָא דְּחַיִּין בְּאַפּוֹהִי, מִכֹּל דִּי _בְּיַבֶּשְׁתָּא_ - מִיתוּ.
וּמְחָא יָת כָּל יְקוּמָא דִּי עַל אַפֵּי אַרְעָא, מֵאֱנָשָׁא עַד בְּעִירָא, עַד רִחֲשָׁא וְעַד עוֹפָא דִּשְׁמַיָּא, וְאִתְמְחִיאוּ מִן אַרְעָא; וְאִשְׁתָּאַר בְּרַם נֹחַ, וְדִי עִמֵּהּ בְּתֵבוֹתָא.
_וּתְקִיפוּ_ מַיָּא עַל אַרְעָא, מְאָה וְחַמְשִׁין יוֹמִין.
