וַאֲמַר יְיָ לְנֹחַ עוּל אַתְּ וְכָל אֱנַשׁ בֵּיתָךְ לְתֵבוֹתָא אֲרֵי יָתָךְ חֲזֵיתִי זַכַּאי קֳדָמַי בְּדָרָא הָדֵין
מִכֹּל בְּעִירָא דַכְיָא תִּסַּב לָךְ שַׁבְעָא שַׁבְעָא דְּכַר וְנוּקְבָא וּמִן בְּעִירָא דְּלָא אִיתָהָא דַכְיָא הִיא תְּרֵין דְּכַר וְנוּקְבָא
אַף מֵעוֹפָא דִשְׁמַיָּא שַׁבְעָא שַׁבְעָא דְּכַר וְנוּקְבָא לְקַיָּמָא זַרְעָא עַל אַפֵּי כָל אַרְעָא
אֲרֵי לִזְמַן יוֹמִין עוֹד שַׁבְעָא אֲנָא מָחֵת מִטְרָא עַל אַרְעָא אַרְבְּעִין יְמָמִין וְאַרְבְּעִין לֵילָוָן וְאֶמְחֵי יָת כָּל יְקוּמָא דִּי עַבְדֵת מֵעַל אַפֵּי אַרְעָא
וַעֲבַד נֹחַ כְּכֹל דִּי פַקְּדֵהּ יְיָ
וְנֹחַ בַּר שִׁתְּ מְאָה שְׁנִין וְטּוֹפָנָא הֲוָה מַיָּא עַל אַרְעָא
וְעָאל נֹחַ וּבְנוֹהִי וְאִתְּתֵהּ וּנְשֵׁי בְנוֹהִי עִמֵּהּ לְתֵבוֹתָא מִן קֳדָם מֵי טוֹפָנָא
מִן בְּעִירָא דַּכְיָא וּמִן בְּעִירָא דִּי לֵיתָהָא דַּכְיָא וּמִן עוֹפָא וְכֹל דִּי רָחֵשׁ עַל אַרְעָא
תְּרֵין תְּרֵין עַלּוּ לְוַת נֹחַ לְתֵבוֹתָא דְּכַר וְנוּקְבָא כְּמָא דִּי פַּקֵּיד יְיָ יָת נֹחַ
וַהֲוָה לִזְמַן שִׁבְעַת יוֹמִין וּמֵי טוֹפָנָא הֲווֹ עַל אַרְעָא
בִּשְׁנַת שִׁתְּ מְאָה שְׁנִין לְחַיֵּי נֹחַ בְּיַרְחָא תִּנְיָנָא בְּשִׁבְעַת עַשְׂרָא יוֹמָא לְיַרְחָא בְּיוֹמָא הָדֵין אִתְבְּזָעוּ כָּל מַבּוּעֵי תְּהוֹמָא רַבָּא וְכַוֵּי שְׁמַיָּא אִתְפַּתָּחוּ
וַהֲוָה מִטְרָא נָחֵת עַל אַרְעָא אַרְבְּעִין יְמָמִין וְאַרְבְּעִין לֵילָוָן
בִּכְרַן יוֹמָא הָדֵין עָאל נֹחַ וְשֵׁם וְחָם וָיֶפֶת בְּנֵי נֹחַ וְאִתַּת נֹחַ וּתְלָתָא נְשֵׁי בְנוֹהִי עִמְהוֹן לְתֵבוֹתָא
אִנּוּן וְכָל חַיְתָא לִזְנַהּ וְכָל בְּעִירָא לְזְנַהּ וְכָל רִחֲשָׁא דְּרָחֵשׁ עַל אַרְעָא לִזְנוֹהִי וְכָל עוֹפָא לִזְנוֹהִי כֹּל צִפַּר כָּל דְּפָרָח
וְעַלּוּ עִם נֹחַ לְתֵבוֹתָא תְּרֵין תְּרֵין מִכָּל בִּשְׂרָא דִּי בֵהּ רוּחָא דְּחַיֵּי
וְעָלַיָּא דְּכַר וְנוּקְבָא מִכָּל בִּשְׂרָא עַלּוּ כְּמָא דִּי פַּקֵּיד יָתֵהּ יְיָ וַאֲגֵן יְיָ (בְּמֵימְרֵהּ) עֲלוֹהִי
וַהֲוָה טוֹפָנָא אַרְבְּעִין יְמָמִין עַל אַרְעָא וּסְגִיאוּ מַיָּא וּנְטָלוּ יָת תֵּבוֹתָא וְאִתָּרָמַת מֵעַל אַרְעָא
וּתְקִיפוּ מַיָּא וּסְגִיאוּ לַחֲדָא עַל אַרְעָא וּמְהַלְּכָא תֵבוֹתָא עַל אַפֵּי מַיָּא
וּמַיָּא תְּקִיפוּ לַחֲדָא לַחֲדָא עַל אַרְעָא וְאִתְחֲפִיאוּ כָּל טוּרַיָּא רָמַיָּא דִּי תְחוֹת כָּל שְׁמַיָּא
חֲמֵשׁ עַשְׂרֵי אַמִּין מִלְעֵלָּא תְּקִיפוּ מַיָּא וְאִתְחֲפִיאוּ טוּרַיָּא
וּמִית כָּל בִּשְׂרָא דְּרָחֵשׁ עַל אַרְעָא בְּעוֹפָא וּבִבְעִירָא וּבְחַיְתָא וּבְכָל רִחֲשָׁא דְּרָחֵשׁ עַל אַרְעָא וְכֹל אֱנָשָׁא
כֹּל דִּי נִשְׁמְתָא רוּחָא דְּחַיִּין בְּאַפּוֹהִי מִכֹּל דִּי בְּיַבֶּשְׁתָּא מִיתוּ
וּמְחָא יָת כָּל יְקוּמָא דִּי עַל אַפֵּי אַרְעָא מֵאֱנָשָׁא עַד בְּעִירָא עַד רִחֲשָׁא וְעַד עוֹפָא דִּשְׁמַיָּא וְאִתְמְחִיאוּ מִן אַרְעָא וְאִשְׁתָּאַר בְּרַם נֹחַ וְדִי עִמֵּהּ בְּתֵבוֹתָא
וּתְקִיפוּ מַיָּא עַל אַרְעָא מְאָה וְחַמְשִׁין יוֹמִין
