וַאֲמַר יְיָ לְמשֶׁה כְּעַן תֶּחֱזֵי דְאֶעְבֵּד לְפַרְעֹה אֲרֵי בִידָא תַקִיפָא יְשַׁלְחִנוּן וּבִידָא תַקִיפָא יְתָרֵכִנוּן מֵאַרְעֵהּ: ססס:
וּמַלִיל יְיָ עִם משֶׁה וַאֲמַר לֵהּ אֲנָא יְיָ:
וְאִתְגְלֵיתִי לְאַבְרָהָם לְיִצְחָק וּלְיַעֲקֹב בְּאֵל שַׁדָי וּשְׁמִי יְיָ לָא הוֹדָעִית לְהוֹן:
וְאַף אֲקֵימִית יָת קְיָמִי עִמְהוֹן לְמִתַּן לְהוֹן יָת אַרְעָא דִכְנָעַן יָת אַרְעָא תּוֹתָבוּתְהוֹן דְאִתּוֹתָבוּ בָהּ:
וְאַף קֳדָמַי שְׁמִיעַ יָת קְבֵילַת בְּנֵי יִשְׂרָאֵל דִי מִצְרָאֵי מַפְלְחִין בְּהוֹן וְדָכִירְנָא יָת קְיָמִי:
בְּכֵן אֱמַר לִבְנֵי יִשְׂרָאֵל אֲנָא יְיָ וְאַפֵּיק יָתְכוֹן מִגוֹ דְחוֹק פָּלְחַן מִצְרָאֵי וֶאֱשֵׁיזֵב יָתְכוֹן מִפָּלְחָנְהוֹן וְאֶפְרוֹק יָתְכוֹן בִּדְרָעָה מְרָמָא וּבְדִינִין רַבְרְבִין:
וֶאֱקָרֵיב יָתְכוֹן קדָמַי לְעַמָא וְאֶהֱוֵי לְכוֹן לֶאֱלָהּ וְתִדְעוּן אֲרֵי אֲנָא יְיָ אֱלָהָכוֹן דְאַפֵּיק יָתְכוֹן מִגוֹ דְחוֹק פָּלְחַן מִצְרָיִם:
וְאָעֵיל יָתְכוֹן לְאַרְעָא דִי קַיֵמִית בְּמֵימְרִי לְמִתַּן יָתַהּ לְאַבְרָהָם לְיִצְחָק וּלְיַעֲקֹב וְאֶתֵּן יָתַהּ לְכוֹן יְרֻתָּא אֲנָא יְיָ:
וּמַלֵיל משֶׁה כֵּן לִבְנֵי יִשְׂרָאֵל וְלָא קַבִּילוּ מִן משֶׁה מִמַעְיַק רוּחָא וּמִפָּלְחָנָא דַהֲוָה קָשְׁיָא עֲלֵיהוֹן:
וּמַלִיל יְיָ עִם משֶׁה לְמֵימָר:
עוּל מַלֵיל לְוָת פַּרְעֹה מַלְכָּא דְמִצְרָיִם וִישַׁלַח יָת בְּנֵי יִשְׂרָאֵל מֵאַרְעֵהּ:
וּמַלִיל משֶׁה קֳדָם יְיָ לְמֵימָר הָא בְנֵי יִשְׂרָאֵל לָא קַבִּילוּ מִנִי וְאֵכְדֵין יְקַבֵּל מִנִי פַרְעֹה וַאֲנָא יַקִיר מַמְלָל:
וּמַלִיל יְיָ עִם משֶׁה וְעִם אַהֲרֹן וּפַקֵדִנוּן לְוָת בְּנֵי יִשְׂרָאֵל וּלְוָת פַּרְעֹה מַלְכָּא דְמִצְרָיִם לְאַפָּקָא יָת בְּנֵי יִשְׂרָאֵל מֵאַרְעָא דְמִצְרָיִם:
אִלֵין רֵישֵׁי בֵית אֲבָהָתְהוֹן בְּנֵי רְאוּבֵן בּוּכְרָא דְיִשְׂרָאֵל חֲנוֹךְ וּפַלוּא חֶצְרוֹן וְכַרְמִי אִלֵין זַרְעֲיַת רְאוּבֵן:
וּבְנֵי שִׁמְעוֹן יְמוּאֵל וְיָמִין וְאֹהַד וְיָכִין וְצֹחַר וְשָׁאוּל בַּר כְּנַעֲנֵיתָא אִלֵין זַרְעֲיַת שִׁמְעוֹן:
וְאִלֵין שְׁמָהַת בְּנֵי לֵוִי לְתוּלְדָתְהוֹן גֵרְשׁוֹן וּקְהָת וּמְרָרִי וּשְׁנֵי חַיֵי לֵוִי מְאָה וּתְלָתִין וּשְׁבַע שְׁנִין:
בְּנֵי גֵרְשׁוֹן לִבְנִי וְשִׁמְעִי לְזַרְעֲיָתְהוֹן:
וּבְנֵי קְהָת עַמְרָם וְיִצְהָר וְחֶבְרוֹן וְעֻזִיאֵל וּשְׁנֵי חַיֵי קְהָת מְאָה וּתְלָתִין וּתְלַת שְׁנִין:
וּבְנֵי מְרָרִי מַחְלִי וּמוּשִׁי אִלֵין זַרְעֲיַת לֵוִי לְתוּלְדָתְהוֹן:
וּנְסִיב עַמְרָם יָת יוֹכֶבֶד אֲחַת אֲבוּהִי לֵהּ לְאִנְתּוּ וִילִידַת לֵהּ יָת אַהֲרֹן וְיָת משֶׁה וּשְׁנֵי חַיֵי עַמְרָם מְאָה וּתְלָתִין וּשְׁבַע שְׁנִין:
וּבְנֵי יִצְהָר קֹרַח וָנֶפֶג וְזִכְרִי:
וּבְנֵי עֻזִיאֵל מִישָׁאֵל וְאֶלְצָפָן וְסִתְרִי:
וּנְסִיב אַהֲרֹן יָת אֱלִישֶׁבַע בַּת עַמִינָדָב אַחָתֵהּ דְנַחְשׁוֹן לֵהּ לְאִנְתּוּ וִילִידַת לֵהּ יָת נָדָב וְיָת אֲבִיהוּא יָת אֶלְעָזָר וְיָת אִיתָמָר:
וּבְנֵי קֹרַח אַסִיר וְאֶלְקָנָה וַאֲבִיאָסָף אִלֵין זַרְעֲיַת קֹרַח:
וְאֶלְעָזָר בַּר אַהֲרֹן נְסִיב לֵהּ מִבְּנַת פּוּטִיאֵל לֵהּ לְאִנְתּוּ וִילִידַת לֵהּ יָת פִּינְחָס אִלֵין רֵישֵׁי אֲבָהַת לֵיוָאֵי לְזַרְעֲיָתְהוֹן:
הוּא אַהֲרֹן וּמשֶׁה דִי אֲמַר יְיָ לְהוֹן אַפִּיקוּ יָת בְּנֵי יִשְׂרָאֵל מֵאַרְעָא דְמִצְרַיִם עַל חֵילֵיהוֹן:
אִנוּן דְמַלְלִין עִם פַּרְעֹה מַלְכָּא דְמִצְרַיִם לְאַפָּקָא יָת בְּנֵי יִשְׂרָאֵל מִמִצְרָיִם הוּא משֶׁה וְאַהֲרֹן:
וַהֲוָה בְּיוֹמָא דְמַלִיל יְיָ עִם משֶׁה בְּאַרְעָא דְמִצְרָיִם:
וּמַלִיל יְיָ עִם משֶׁה לְמֵימַר אֲנָא יְיָ מַלֵל עִם פַּרְעֹה מַלְכָּא דְמִצְרַיִם יָת כָּל דִי אֲנָא מְמַלֵל עִמָךְ:
וַאֲמַר משֶׁה קֳדָם יְיָ הָא אֲנָא יַקִיר מַמְלָל וְאֶכְדֵין יְקַבֵּל מִנִי פַּרְעֹה:
