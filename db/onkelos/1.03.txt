וְחִוְיָא הֲוָה _חַכִּים_ מִכֹּל חֵוַת בְּרָא, דִּי עֲבַד יְיָ אֱלֹהִים, וַאֲמַר לְאִתְּתָא: "_בְּקוּשְׁטָא_ אֲרֵי אֲמַר יְיָ 'לָא תֵיכְלוּן מִכֹּל אִילַן גִּינְתָא!'".
וַאֲמָרַת אִתְּתָא לְחִוְיָא: “מִפֵּירֵי אִילַן גִּינְתָא נֵיכוּל,
וּמִפֵּירֵי אִילָנָא דִּי בִ_מְצִיעוּת_ גִּינְתָא, אֲמַר יְיָ: 'לָא תֵיכְלוּן מִנֵּהּ, וְלָא תְקַרְבוּן בֵּהּ - דִּילְמָא תְּמוּתוּן!'”.
וַאֲמַר חִוְיָא לְאִיתְּתָא: “לָא מוּת תְּמוּתוּן!
אֲרֵי _גַּלֵּי_ +קֳדָם+ יְיָ, אֲרֵי בְּיוֹמָא דְּתֵיכְלוּן מִנֵּהּ, וְיִתְפַּתְּחַן עֵינֵיכוֹן; וּתְהוֹן כְּ_רַבְרְבִין חַכִּימִין_ +בֵּין+ טַב לְבִישׁ.
וַחֲזַת אִתְּתָא אֲרֵי טַב אִילַן לְמֵיכָל, וַאֲרֵי _אַסֵּי_ הוּא לְעַיְנִין, וּמְרַגֵּג אִילָנָא _לְאִסְתַּכָּלָא בֵהּ_; וּנְסֵיבַת מֵאִבֵּהּ - וַאֲכָלַת. וִיהָבַת אַף לְבַעְלַהּ עִמַּהּ - וַאֲכָל.
וְאִתְפַּתָּחָא עֵינֵי תַרְוֵיהוֹן, וִידָעוּ, אֲרֵי עַרְטִילָאִין אִינוּן; וְחַטִּיטוּ לְהוֹן טַרְפֵי תְאֵנִין, וַעֲבָדוּ לְהוֹן זְרָזִין.
וּשְׁמָעוּ יָת קַל +מֵימְרָא+ דַּיְיָ אֱלֹהִים מְהַלֵּךְ בְּגִינְתָא, _לִמְנַח_ יוֹמָא; וְאִיטַמַּר אָדָם וְאִתְּתֵהּ מִן +קֳדָם+ יְיָ אֱלֹהִים בְּגוֹ אִילַן גִּינְתָא.
וּקְרָא יְיָ אֱלֹהִים לְאָדָם, וַאֲמַר לֵהּ: "אָן אָתְּ?"
וַאֲמַר: "יָת קַל +מֵימְרָךְ+ שַׁמְעִית בְּגִינְתָא; וּדְחֵילִית אֲרֵי עַרְטִילַאי אֲנָא - וְאִיטַמָּרִית".
וַאֲמַר: "מָן חַוֵי לָךְ, אֲרֵי עַרְטִילַאי אָתְּ? הֲמִן אִילָנָא, דִּפַקֶּדְתָּךְ בְּדִיל דְּלָא לְמֵיכַל מִנֵּהּ - אֲכָלְתְּ?!"
וַאֲמַר הָאָדָם: "אִתְּתָא דִּיהַבְתָּ עִמִּי - הִיא יַהֲבַת לִי מִן אִילָנָא, וַאֲכָלִית".
וַאֲמַר יְיָ אֱלֹהִים לְאִתְּתָא: "מַה דָּא עַבְדָת??" וַאֲמָרַת אִתְּתָא: "חִוְיָא אַטְעַיָנִי - וַאֲכָלִית".
וַאֲמַר יְיָ אֱלֹהִים לְחִוְיָא: "אֲרֵי עֲבַדְתָּ דָּא - לִיט אַתְּ מִכָּל בְּעִירָא, וּמִכֹּל חֵוַת בָּרָא!! עַל מְעָךְ תֵּיזִילּ, וְעַפְרָא תֵיכוּל כָּל יוֹמֵי חַיָּיךְ.
וּדְבָבוּ אֱשַׁוֵּי בֵּינָךְ וּבֵין אִתְּתָא, וּבֵין בְּנָךְ וּבֵין בְּנָהַהּ; הוּא +יְהֵי דְּכִיר לָךְ מַה דִּעֲבַדְתָּ לֵהּ+ מִלְּקַדְמִין וְאַתְּ +תְּהֵא נָטִיר לֵהּ לְ+_סוֹפָא_.
לְאִתְּתָא אֲמַר: "אַסְגָּאָה אַסְגֵּי צַעֲרַיְכִי וְעִדּוּיַיְכִי, בְּצְעַר תֵּלְדִי בְנִין; וּלְוַת בַּעְלִךְ תְּהֵא תִּאוּבְתִּיךְ, וְהוּא יִשְׁלַט בִּךְ".
וּלְאָדָם אֲמַר: "אֲרֵי קַבֵּילְתָּ לְמֵימַר אִתְּתָךְ, וַאֲכַלְתָּ מִן אִילָנָא דִּי פַקֵּידְתָּךְ לְמֵימַר: 'לָא תֵיכוּל מִנֵּהּ' - לִיטָא אַרְעָא בְּדִילָךְ; _בַּעֲמַל_ תִּיכְלִינַהּ, כָּל יוֹמֵי חַיָּיךְ.
וְכוּבִין וְאַטְדִּין תַּצְמַח לָךְ, וְתֵיכוּל יָת עִסְבָּא דְחַקְלָא.
בְּזִעֲתָא דְאַפָּךְ תֵּיכוּל לַחְמָא, [עַד דְּתִיתוּב לְאַרְעָא|עַד דְּתִיתָּתַב לְאַרְעָא], דְּמִנַּהּ _אִתְבְּרֵיתָא_; אֲרֵי עַפְרָא אַתְּ, וּלְעַפְרָא תְּתוּב!"
וּקְרָא אָדָם שׁוּם אִתְּתֵהּ 'חַוָּה' - אֲרֵי הִיא הֲוַת אִמָּא דְכָל _בְּנֵי אֱנָשָׁא_.
וַעֲבַד יְיָ אֱלֹהִים לְאָדָם וּלְאִתְּתֵהּ +לְבוּשִׁין דִּיקָר עַל מְשַׁךְ בִּשְׂרֵיהוֹן+, וְאַלְבִּישִׁינוּן.
וַאֲמַר יְיָ אֱלֹהִים: "הָא אָדָם הֲוָה _יְחִידַי_ +בְּעַלְמָא+ מִנֵּהּ לְמִידַע טַב וּבִישׁ, וּכְעַן, דִּילְמָא יוֹשִׁיט יְדֵהּ, וְיִסַב אַף מֵאִילָן חַיָּיא, וְיֵכוּל - וִיחֵי לְעָלָם!"
וְשַׁלְּחֵהּ יְיָ אֱלֹהִים מִגִּינְתָא דְעֵדֶן, לְמִפְלַח יַת אַדְמְתָא דְּ_אִתְבְּרִי_ מִתַּמָּן.
וְתָרִךְ יָת אָדָם. וְאַשְׁרֵי מִלְּקַדְמִין לְגִינְתָא דְעֵדֶן יָת כְּרוּבַיָּא, וְיָת _שְׁנַן_ חַרְבָּא דְּמִתְהַפְּכָא, לְמִיטַר יָת אוֹרַח אִילַן חַיָּיא.
