וּמִן תִּכְלָא וְאַרְגְוָנָא וּצְבַע זְהוֹרִי עֲבָדוּ לְבוּשֵׁי שִׁמוּשָׁא לְשַׁמָשָׁא בְּקוּדְשָׁא וַעֲבָדוּ יָת לְבוּשֵׁי קוּדְשָׁא דִי לְאַהֲרֹן כְּמָא דִי פַקִיד יְיָ יָת משֶׁה:
וַעֲבַד יַת אֵפוֹדָא דַהֲבָא תִּכְלָא וְאַרְגְוָנָא וּצְבַע זְהוֹרִי וּבוּץ שְׁזִיר:
וְרַדִידוּ יָת טַסֵי דַהֲבָא וְקַצִיצוּ חוּטִין לְמֶעְבַּד בְּגוֹ תִּכְלָא וּבְגוֹ אַרְגְוָנָא וּבְגוֹ צְבַע זְהוֹרִי וּבְגוֹ בוּצָא עוֹבַד אֳמָן:
כַּתְפָּן עֲבָדוּ לֵהּ מְלַפְּפָן עַל תְּרֵין סִטְרוֹהִי יִתְלָפֵף:
וְהֶמְיַן תִּקוּנֵהּ דִי עֲלוֹהִי מִנֵהּ הוּא כְּעוֹבָדוֹהִי דַהֲבָא תִּכְלָא וְאַרְגְוָנָא וּצְבַע זְהוֹרִי וּבוּץ שְׁזִיר כְּמָא דִי פַקִיד יְיָ יָת משֶׁה:
וַעֲבָדוּ יָת אַבְנֵי בוּרְלָא מְשַׁקְעָן מְרַמְצָן דִדְהָב גְלִיפָן כְּתַב מְפָרַשׁ עַל שְׁמָהַת בְּנֵי יִשְׂרָאֵל:
וְשַׁוִי יָתְהוֹן עַל כִּתְפָא דְאֵפוֹדָא אַבְנֵי דוּכְרָנָא לִבְנֵי יִשְׂרָאֵל כְּמָא דִי פַקִיד יְיָ יָת משֶׁה:
וַעֲבַד יָת חוּשְׁנָא עוֹבַד אֳמָן כְּעוֹבְדֵי אֵפוֹדָא דַהֲבָּא תִּכְלָא וְאַרְגְוָנָא וּצְבַע זְהוֹרִי וּבוּץ שְׁזִיר:
מְרַבַּע הֲוָה עֵיף עֲבָדוּ יָת חוּשְׁנָא זֶרְתָּא אֻרְכֵּהּ וְזֶרְתָּא פְתָיֵהּ עֵיף:
וְאַשְׁלִימוּ בֵהּ אַרְבְּעָא סִדְרֵי אֶבֶן טָבָא סִדְרָא קַדְמָאָה סַמְקָן יַרְקָן וּבַרְקָן סִדְרָא חָד:
וְסִדְרָא תִּנְיָנָא אִזְמָרַגְדִין שַׁבְזִיז וְסַבְהֲלוֹם:
וְסִדְרָא תְּלִיתָאָה קַנְכֵּרֵי טַרְקְיָא וְעֵין עֶגְלָא:
וְסִדְרָא רְבִיעָאָה כְּרוּם יַמָּא וּבוּרְלָא וּפַנְטֵּרֵי מְשַׁקְעָן מְרַמְצָן דִדְהַב בְּאַשְׁלָמוּתְהוֹן:
וְאַבְנַיָא עַל שְׁמָהַת בְּנֵי יִשְׂרָאֵל אִנוּן תַּרְתֵּי עֶסְרֵי עַל שְׁמָהַתְהוֹן כְּתַב מְפָרַשׁ כִּגְלוֹף דְעִזְקָא גְבַר עַל שְׁמֵהּ לִתְרֵין עֲסַר שִׁבְטִין:
וַעֲבָדוּ עַל חוּשְׁנָא תִּכִּין מְתַחְמָן עוֹבַד גְדִילוּ דְהַב דְכֵי:
וַעֲבָדוּ תַּרְתֵּין מְרַמְצָן דִדְהַב וְתַרְתֵּין עִזְקָן דִדְהָב וִיהָבוּ יָת תַּרְתֵּין עִזְקָתָא עַל תְּרֵין סִטְרֵי חוּשְׁנָא:
וִיהָבוּ תַּרְתֵּין גְדִילָתָא דִדְהַב עַל תַּרְתֵּין עִזְקָתָא עַל סִטְרֵי חוּשְׁנָא:
וְיָת תַּרְתֵּין גְדִילָן דְעַל תַּרְתֵּין סִטְרוֹהִי יְהָבוּ עַל תַּרְתֵּין מְרַמְצָתָא וִיהָבֻנוּן עַל כַּתְפֵּי אֵפוֹדָא לָקֳבֵל אַפּוֹהִי:
וַעֲבָדוּ תַּרְתֵּין עִזְקַן דִדְהַב וְשַׁוִיאוּ עַל תְּרֵין סִטְרֵי חוּשְׁנָא עַל סִפְתֵהּ דִי לְעִבְרָא דְאֵפוֹדָא מִלְגָיו:
וַעֲבָדוּ תַּרְתֵּין עִזְקַן דִדְהַב וִיהָבֻנוּן עַל תַּרְתֵּין כִּתְפֵי אֵפוֹדָא מִלְרַע מִלָקֳבֵל אַפּוֹהִי לָקֳבֵל בֵּית לוֹפֵי מֵעִלָוֵי לְהֶמְיַן אֵפוֹדָא:
וְיַחֲדוּן יָת חוּשְׁנָא מֵעִזְקְתֵהּ לְעִזְקָתָא דְאֵפוֹדָא בְּחוּטָּא דִתְכֶלְתָּא לְמֶהֱוֵי עַל הֶמְיַן אֵפוֹדָא וְלָא יִתְפָּרֵק חוּשְׁנָא מֵעִלָוֵי אֵפוֹדָא כְּמָא דִי פַקִיד יְיָ יָת משֶׁה:
וַעֲבַד יָת מְעִילָא דְאֵפוֹדָא עוֹבַד מָחֵי גְמִיר תִּכְלָא:
וּפוּם מְעִילָא כָּפִיל לְגַוֵהּ כְּפוּם שִׁרְיָן תּוֹרָא מַקַף לְפוּמֵהּ סְחוֹר סְחוֹר לָא יִתְבְּזָע:
וַעֲבָדוּ עַל שִׁפּוֹלֵי מְעִילָא רִמוֹנֵי תִּכְלָא וְאַרְגְוָנָא וּצְבַע זְהוֹרִי שְׁזִיר:
וַעֲבָדוּ זַגַיָא דְהַב דְכֵי וִיהָבוּ יָת זַגַיָא בְּגוֹ רִמוֹנַיָא עַל שִׁפּוֹלֵי מְעִילָא סְחוֹר סְחוֹר בְּגוֹ רִמוֹנַיָא:
זַגָא וְרִמוֹנָא זַגָא וְרִמוֹנָא עַל שִׁפּוֹלֵי מְעִילָא סְחוֹר סְחוֹר לְשַׁמָשָׁא כְּמָא דִי פַקִיד יְיָ יָת משֶׁה:
וַעֲבָדוּ יָת כִּתּוּנִין דְבוּצָא עוֹבַד מָחֵי לְאַהֲרֹן וִלִבְנוֹהִי:
וְיָת מִצְנֶפְתָּא דְבוּצָא וְיָת שְׁבַח כּוֹבָעַיָא דְבוּצָא וְיָת מִכְנְסֵי בוּצָא דְבוּץ שְׁזִיר:
וְיָת הֶמְיָנָא דְבוּץ שְּׁזִיר וְתִכְלָא וְאַרְגְוָנָא וּצְבַע זְהוֹרִי עוֹבַד צַיָר כְּמָא דִי פַקִיד יְיָ יָת משֶׁה:
וַעֲבָדוּ יָת צִיצָא כְלִילָא דְקוּדְשָׁא דְהַב דְכֵי וּכְתָבוּ עֲלוֹהִי גְלִיפָן כְּתַב מְפָרַשׁ קֹדֶשׁ לַיְיָ:
וִיהָבוּ עֲלוֹהִי חוּטָא תְכֶלְתָּא לְמִתַּן עַל מִצְנֶפְתָּא מִלְעֵלָא כְּמָא דִי פַקִיד יְיָ יָת משֶׁה:
וּשְׁלִים כָּל פָּלְחַן מַשְׁכְּנָא מַשְׁכַּן זִמְנָא וַעֲבָדוּ בְּנֵי יִשְׂרָאֵל כְּכֹל דִי פַקִיד יְיָ יָת משֶׁה כֵּן עֲבָדוּ:
וְאַיְתִיוּ יָת מַשְׁכְּנָא לְמשֶׁה יָת מַשְׁכְּנָא וְיָת כָּל מָנוֹהִי פוּרְפוֹהִי דַפּוֹהִי עַבְּרוֹהִי וְעַמוּדוֹהִי וְסַמְכוֹהִי:
וְיָת חוֹפָאָה מַשְׁכֵי דְדִכְרֵי מְסַמְקֵי וְיָת חוֹפָאָה מַשְׁכֵי דְסַסְגוֹנָא וְיָת פָּרֻכְתָּא דִפְרָסָא:
יָת אֲרוֹנָא דְסַהֲדוּתָא וְיָת אֲרִיחוֹהִי וְיָת כַּפֻּרְתָּא:
יָת פָּתוֹרָא יָת כָּל מָנוֹהִי וְיָת לְחֵם אַפַּיָא:
יָת מְנַרְתָּא דָכִיתָא יָת בּוֹצִינָהָא בּוֹצִנֵי סִדְרָא וְיָת כָּל מָנָהָא וְיָת מִשְׁחָא דְאַנְהָרוּתָא:
וְיָת מַדְבְּחָא דְדַהֲבָא וְיָת מִשְׁחָא דִרְבוּתָא וְיָת קְטֹרֶת בּוּסְמַיָא וְיָת פְּרָסָא דִתְרַע מַשְׁכְּנָא:
יָת מַדְבְּחָא דִנְחָשָׁא וְיָת סְרָדָא דִנְחָשָׁא דִילֵהּ יָת אֲרִיחוֹהִי וְיָת כָּל מָנוֹהִי יָת כִּיוֹרָא וְיָת בְּסִיסֵהּ:
יָת סְרָדֵי דְדַרְתָּא יָת עַמוּדָהָא וְיָת סַמְכָהָא וְיָת פְּרָסָא לִתְרַע דַרְתָּא יָת אֲטוּנוֹהִי וְסִכָּהָא וְיָת כָּל מָנֵי פָּלְחַן מַשְׁכְּנָא לְמַשְׁכַּן זִמְנָא:
יָת לְבוּשֵׁי שִׁמוּשָׁא לְשַׁמָשָׁא בְקוּדְשָׁא יָת לְבוּשֵׁי קוּדְשָׁא לְאַהֲרֹן כַּהֲנָא וְיָת לְבוּשֵׁי בְנוֹהִי לְשַׁמָשָׁא:
כְּכֹל דִי פַקִיד יְיָ יָת משֶׁה כֵּן עֲבָדוּ בְּנֵי יִשְׂרָאֵל יָת כָּל פָּלְחָנָא:
וַחֲזָא משֶׁה יָת כָּל עִבִדְתָּא וְהָא עֲבָדוּ יָתַהּ כְּמָא דִי פַקִיד יְיָ כֵּן עֲבָדוּ וּבָרִיךְ יָתְהוֹן משֶׁה:
