וּמַלִיל יְיָ עִם מֹשֶׁה לְמֵימָר
מַלֵל עִם בְּנֵי יִשְׂרָאֵל וְתֵימַר לְהוֹן אֲנָא יְיָ אֱלָהָכוֹן
כְּעוֹבָדֵי עַמָא דְאַרְעָא דְמִצְרַיִם דִי יְתֶבְתּוּן בַּהּ לָא תַעְבְּדוּן וּכְעוֹבָדֵי עַמָא דְאַרְעָא דִכְנַעַן דִי אֲנָא מָעֵל יָתְכוֹן תַּמָן לָא תַעְבְּדוּן וּבְנִימוֹסֵיהוֹן לָא תְהָכוּן
יָת דִינַי תַּעְבְּדוּן וְיָת קְיָמַי תִּטְרוּן לְהַלָכָא בְּהוֹן אֲנָא יְיָ אֱלָהָכוֹן
וְתִטְרוּן יָת קְיָמַי וְיָת דִינַי דִי יַעְבֵּד יָתְהוֹן אֲנָשָׁא וְיֵחֵי בְהוֹן לְחַיֵי עָלְמָא אֲנָא יְיָ
גְבַר גְבַר לְכָל קָרִיב בִּסְרֵהּ לָא תִקְרְבוּן לְגַלָאָה עֶרְיָא אֲנָא יְיָ
עֶרְיַת אֲבוּךְ וְעֶרְיַת אִמָךְ לָא תְגַלֵי אִמָךְ הִיא לָא תְגַלֵי עֶרְיָתָהּ
עֶרְיַת אִתַּת אָבוּךְ לָא תְגַלֵי עֶרְיַת אָבוּךְ הִיא
עֶרְיַת אֲחָתָךְ בַּת אֲבוּךְ אוֹ בַת אִמָךְ דִילִידָא מִן אֲבוּךְ מִן אִתַּת אָחֳרִי אוֹ מִן אִמָךְ לִגְבַר אָחֳרָן לָא תְגַלֵי עֶרְיַתְהֶן
עֶרְיַת בַּת בְּרָךְ אוֹ בַת בְּרַתָּךְ לָא תְגַלֵי עֶרְיַתְהֶן אֲרֵי עֶרְיְתָךְ אִנוּן
עֶרְיַת בַּת אִתַּת אֲבוּךְ דִילִידָא מִן אֲבוּךְ אֲחָתָךְ הִיא לָא תְגַלֵי עֶרְיָתָהּ
עֶרְיַת אֲחַת אֲבוּךְ לָא תְגַלֵי קָרִיבַת אֲבוּךְ הִיא
עֶרְיַת אֲחַת אִמָךְ לָא תְגַלֵי אֲרֵי קָרִיבַת אִמָךְ הִיא
עֶרְיַת אַח אֲבוּךְ לָא תְגַלֵי לְאִתְּתֵהּ לָא תִקְרַב אִתַּת אַח אֲבוּךְ הִיא
עֶרְיַת כַּלָתָךְ לָא תְגַלֵי אִתַּת בְּרָךְ הִיא לָא תְגַלֵי עֶרְיָתָהּ
עֶרְיַת אִתַּת אֲחוּךְ לָא תְגַלֵי עֶרְיַת אֲחוּךְ הִיא
עֶרְיַת אִתְּתָא וּבְרַתָּהּ לָא תְגַלֵי יָת בַּת בְּרָהּ וְיָת בַּת בְּרַתָּהּ לָא תִסַב לְגַלָאָה עֶרְיָתָהּ קָרִיבָן אִנוּן עֵצַת חִטְאִין הִיא
וְאִתְּתָא עִם אֲחָתָהּ לָא תִסָב לְעָקָאָה לְגַלָאָה עֶרְיְתָהּ עֲלָהּ בְּחַיָיהָא
וּלְאִתְּתָא בְּרִחוּק סוֹבְתַהּ לָא תִקְרַב לְגַלָאָה עֶרְיְתָהּ
וּבְאִתַּת חַבְרָךְ לָא תִתֵּן שְׁכֻבְתָּךְ לְזַרְעָא לְאִסְתָּאָבָא בָהּ
וּמִבְּנָיִךְ לָא תִתֵּן לְאַעְבָּרָא לְמוֹלֶךְ וְלָא תַחֵל יָת שְׁמָא דֶאֱלָהָךְ אֲנָא יְיָ
וְעִם דְכוּרָא לָא תִשְׁכּוּב מִשְׁכְּבֵי אִתְּתָא תּוֹעֶבְתָּא הִיא
וּבְכָל בְּעִירָא לָא תִתֵּן שְׁכֻבְתָּךְ לְאִסְתָּאָבָא בָהּ וְאִתְּתָא לָא תְקוּם קֳדָם בְּעִירָא לְמִשְׁלַט בָּהּ תִּבְלָא הוּא
לָא תִסְתַּאֲבוּן בְּכָל אִלֵין אֲרֵי בְּכָל אִלֵין אִסְתַּאָבוּ עַמְמַיָא דִי אֲנָא מַגְלֵי מִן קֳדָמֵיכוֹן
וְאִסְתָּאָבַת אַרְעָא וְאַסְעָרִית חוֹבָהּ עֲלָהּ וְרוֹקֵינַת אַרְעָא יָת יָתְבָהָא
וְתִטְרוּן אַתּוּן יָת קְיָמַי וְיָת דִינַי וְלָא תַעְבְּדוּן מִכֹּל תּוֹעֶבְתָּא הָאִלֵין יַצִיבָא וְגִיוֹרָא דְיִתְגַיְרוּן בֵּינֵיכוֹן
אֲרֵי יָת כָּל תּוֹעֶבְתָּא הָאִלֵין עֲבָדוּ אֱנָשֵׁי אַרְעָא דִי קֳדָמֵיכוֹן וְאִסְתָּאָבַת אַרְעָא
וְלָא תְרוֹקֵן אַרְעָא יָתְכוֹן בְּסָאֳבֵיכוֹן יָתָהּ כְּמָא דְרוֹקֵינַת יָת עַמְמַיָא דִי קֳדָמֵיכוֹן
אֲרֵי כָּל דִי יַעְבֵּד מִכֹּל תּוֹעֶבְתָּא הָאִלֵין וְיִשְׁתֵּיצוּן נַפְשָׁתָא דְיַעְבְּדוּן מִגוֹ עַמְהוֹן
וְתִטְרוּן יָת מַטְרַת מֵימְרִי בְּדִיל דְלָא לְמֶעְבַּד מִנִימוֹסֵי תוֹעֶבְתָּא דְאִתְעֲבָדָא קֳדָמֵיכוֹן וְלָא תִסְתַּאֲבוּן בְּהוֹן אֲנָא יְיָ אֱלָהָכוֹן
