וַאֲמַר יְיָ לְמשֶׁה עוֹד מַכְתַּשׁ חַד אַיְתִי עַל פַּרְעֹה וְעַל מִצְרַיִם בָּתַר כֵּן יְשַׁלַח יָתְכוֹן מִכָּא כְּשַׁלָחוּתֵהּ גְמֵירָא תָּרָכָא יְתָרֵךְ יָתְכוֹן מִכָּא:
מַלֵיל כְּעַן קֳדָם עַמָא וּשְׁאִילוּ גְבַר מִן חַבְרֵהּ וְאִתְּתָא מִן חַבְרָתַהּ מָנִין דִכְסַף וּמָנִין דִדְהָב:
וִיהַב יְיָ יָת עַמָא לְרַחֲמִין בְּעֵינֵי מִצְרָיִם אַף גַבְרָא משֶׁה רַב לַחֲדָא בְּאַרְעָא דְמִצְרַיִם בְּעֵינֵי עַבְדֵי פַרְעֹה וּבְעֵינֵי עַמָא:
וַאֲמַר משֶׁה כִּדְנַן אֲמַר יְיָ כְּפַלְגוּת לֵילְיָא אֲנָא מִתְגְלִי בְּגוֹ מִצְרָיִם:
וִימוּת כָּל בּוּכְרָא בְּאַרְעָא דְמִצְרַיִם מִבּוּכְרָא דְפַרְעֹה דְעָתִיד לְמִתַּב עַל כָּרְסֵי מַלְכוּתֵהּ עַד בּוּכְרָא דְאַמְתָא דִי בָּתַר רֵיחְיָא וְכֹל בּוּכְרָא דִבְעִירָא:
וּתְהֵי צְוַחְתָּא רַבָּא בְּכָל אַרְעָא דְמִצְרָיִם דִכְוָתֵהּ לָא הֲוַת וְדִכְוָתֵהּ לָא תוֹסֵף:
וּלְכֹל בְּנֵי יִשְׂרָאֵל לָא יַנְזֵק כַּלְבָּא בְּלִישְׁנֵהּ לְמִנְבַּח לְמֵאֲנָשָׁא וְעַד בְּעִירָא בְּדִיל דְתִדְעוּן דְיַפְרֵישׁ יְיָ בֵּין מִצְרַיִם וּבֵין יִשְׂרָאֵל:
וְיֶחֱתוּן כָּל עַבְדָיךְ אִלֵין לְוָתִי וְיִבְעוֹן מִנִי לְמֵימָר פּוּק אַתְּ וְכָל עַמָא דִי עִמָךְ וּבָתַר כֵּן אֶפּוֹק וּנְפַק מִלְוָת פַּרְעֹה בִּתְקוֹף רְגָז:
וַאֲמַר יְיָ לְמשֶׁה לָא יְקַבֵּל מִנְכוֹן פַּרְעֹה בְּדִיל לְאַסְגָאָה מוֹפְתַי בְּאַרְעָא דְמִצְרָיִם:
וּמשֶׁה וְאַהֲרֹן עֲבָדוּ יָת כָּל מוֹפְתַיָא הָאִלֵין קֳדָם פַּרְעֹה וְתַקַף יְיָ יָת לִבָּא דְפַרְעֹה וְלָא שַׁלַח יָת בְּנֵי יִשְׂרָאֵל מֵאַרְעֵהּ:
