וּמַלֵיל יְיָ עִם משֶׁה לְמֵימָר
אֱמַר לְאֶלְעָזָר בַּר אַהֲרֹן כַּהֲנָא וְיַפְרֵשׁ יָת מַחְתְּיָתָא מִבֵּין יְקִידַיָא וְיָת אֶשָׁתָא יַרְחִיק לְהָלָא אֲרֵי אִתְקַדָשׁוּ
יָת מַחְתְּיַת חַיָיבַיָא הָאִלֵין דְאִתְחַיְיבוּ בְּנַפְשָׁתֵיהוֹן וְיַעְבְּדוּן יַתְהוֹן טַסִין רְדִידִין חוֹפָאָה לְמַדְבְּחָא אֲרֵי קָרֵבֻנוּן קֳדָם יְיָ וְאִתְקַדָשׁוּ וִיהֶוְיָן לְאָת לִבְּנֵי יִשְׂרָאֵל
וּנְסִיב אֶלְעָזָר כַּהֲנָא יָת מַחְתְּיָתָא דִנְחָשָׁא דְקָרִיבוּ יוֹקִידַיָא וְרַדִידֻנוּן חוֹפָאָה לְמַדְבְּחָא
דָכְרָנָא לִבְנֵי יִשְׂרָאֵל בְּדִיל דִי לָא יִקְרַב גְבַר חִלוֹנַי דִי לָא מִזַרְעָא דְאַהֲרֹן הוּא לְאַסָקָא קְטוֹרֶת בּוּסְמִין קֳדָם יְיָ וְלָא יְהֵי כְקֹרַח וְכִכְנִשְׁתֵּיהּ כְּמָא דִי מַלֵיל יְיָ בִּידָא דְמשֶׁה לֵיהּ
וְאִתְרַעֲמוּ כָּל כְּנִשְׁתָּא דִבְנֵי יִשְׂרָאֵל מִיוֹמָא דְבַתְרוֹהִי עַל משֶׁה וְעַל אַהֲרֹן לְמֵימָר אַתּוּן גְרַמְתּוּן דְמִית עַמָא דַיְיָ
וַהֲוָה בְּאִתְכַּנָשׁוּת כְּנִשְׁתָּא עַל משֶׁה וְעַל אַהֲרֹן וְאִתְפְּנִיאוּ לְמַשְׁכַּן זִמְנָא וְהָא חֲפָהִי עֲנָנָא וְאִתְגְלִי יְקָרָא דַיְיָ
וְעָאל משֶׁה וְאַהֲרֹן לָקֳדָם מַשְׁכַּן זִמְנָא
וּמַלֵיל יְיָ עִם משֶׁה לְמֵימָר
אִתְפְרָשׁוּ מִגוֹ כְּנִשְׁתָּא הָדָא וְאֱשֵׁיצֵי יָתְהוֹן כְּשָׁעָה וּנְפָלוּ עַל אַפֵּיהוֹן
וַאֲמַר משֶׁה לְאַהֲרֹן סַב יָת מַחְתִּיתָא וְהַב עֲלַהּ אֶשָׁתָא מֵעִלָוֵי מַדְבְּחָא וְשַׁו קְטוֹרֶת בּוּסְמַיָא וְאוֹבִיל בִּפְרִיעַ לִכְנִשְׁתָּא וְכַפֵּר עֲלֵיהוֹן אֲרֵי נְפַק רָגְזָא מִן קֳדָם יְיָ שָׁרֵי מוֹתָנָא
וּנְסֵיב אַהֲרֹן כְּמָא דְמַלֵיל משֶׁה וּרְהַט לְגוֹ קְהָלָא וְהָא שָׁרֵי מוֹתָנָא בְּעַמָא וִיהַב יָת קְטּוֹרֶת בּוּסְמַיָא וְכַפֵּר עַל עַמָא
וְקָם בֵּין מֵתַיָא וּבֵין חַיַיָא וְאִתְכְּלִי מוֹתָנָא
וַהֲווֹ דְמִיתוּ בְּמוֹתָנָא אַרְבַּעַת עֲשַׂר אַלְפִין וּשְׁבַע מְאָה בַּר מִדְמִיתוּ עַל פְּלֻגְתָּא דְקֹרַח
וְתָב אַהֲרֹן לְמשֶׁה לִתְרַע מַשְׁכַּן זִמְנָא וּמוֹתָנָא אִתְכְּלִי
וּמַלֵיל יְיָ עִם משֶׁה לְמֵימָר
מַלֵיל עִם בְּנֵי יִשְׂרָאֵל וְסַב מִנְהוֹן חוּטְרָא חוּטְרָא לְבֵית אַבָּא מִן כָּל רַבְרְבָנֵיהוֹן לְבֵית אֲבָהַתְהוֹן תְּרֵי עֲשַׂר חוּטְרִין גְבַר יָת שְׁמֵיהּ תִּכְתּוֹב עַל חוּטְרֵיהּ
וְיָת שְׁמָא דְאַהֲרֹן תִּכְתּוֹב עַל חוּטְרָא דְלֵוִי אֲרֵי חוּטְרָא חָד לְרֵישׁ בֵּית אֲבָהַתְהוֹן
וְתַצְנְעִינוּן בְּמַשְׁכַּן זִמְנָא קֳדָם סַהֲדוּתָא דִי אֲזַמִין מֵימְרִי לְכוֹן תַּמָן
וִיהֵי גַבְרָא דִי אִתְרְעֵי בֵּהּ חוּטְרֵיהּ יִנְעֵי וַאֲנִיחַ מִן קֳדָמַי יָת תּוּרְעֲמַת בְּנֵי יִשְׂרָאֵל דִי אִנוּן מִתְרַעֲמִין עֲלֵיכוֹן
וּמַלֵיל משֶׁה עִם בְּנֵי יִשְׂרָאֵל וִיהָבוּ לֵיהּ כָּל רַבְרְבָנֵיהוֹן חוּטְרָא לְרַבָּא חָד חוּטְרָא לְרַבָּא חָד לְבֵית אֲבָהַתְהוֹן תְּרֵי עֲשַׂר חוּטְרִין וְחוּטְרָא דְאַהֲרֹן בְּגוֹ חוּטְרֵיהוֹן
וְאַצְנַע משֶׁה יָת חוּטְּרַיָא קֳדָם יְיָ בְּמַשְׁכְּנָא דְסַהֲדוּתָא
וַהֲוָה בְּיוֹמָא דְבַתְרוֹהִי וְעַאל משֶׁה לְמַשְׁכְּנָא דְסַהֲדוּתָא וְהָא נְעָא חוּטְרָא דְאַהֲרֹן לְבֵית לְוֵי וְאַפֵּק לַבְלְבִין וְאָנֵץ נֵץ וְכָפֵית שִׁיגְדִין
וְאַפֵיק משֶׁה יָת כָּל חוּטְרַיָא מִן קֳדָם יְיָ לְוָת כָּל בְּנֵי יִשְׂרָאֵל וְאִשְׁתְּמוֹדָעוּ וּנְסִיבוּ גְבַר חוּטְרֵיהּ
וַאֲמַר יְיָ לְמשֶׁה אָתֵיב יַת חוּטְרָא דְאַהֲרֹן לְקֳדָם סַהֲדוּתָא לְמַטָרָא לְאָת לְעַמָא סַרְבָנָא וִיסוּפַן תּוּרְעֲמַתְהוֹן מִן קֳדָמַי וְלָא יְמוּתוּן
וַעֲבַד משֶׁה כְּמָא דִי פַקֵיד יְיָ יָתֵיה כֵּן עֲבָד
וַאֲמָרוּ בְּנֵי יִשְׂרָאֵל לְמשֶׁה לְמֵימָר הָא מִנָנָא קְטָלַת חַרְבָּא הָא מִנָנָא בְּלָעַת אַרְעָא וְהָא מִנָנָא דְמִיתוּ בְמוֹתָנָא
כֹּל דְקָרֵב מִקְרַב לְמַשְׁכְּנָא דַיְיָ מָאִית הָא אַנַחְנָא סַיְפִין לִמְמָת
