וּמַלִיל יְיָ עִם מֹשֶׁה לְמֵימָר
פַּקֵד יָת אַהֲרֹן וְיָת בְּנוֹהִי לְמֵימָר דָא אוֹרַיְתָא דַעֲלָתָא הִיא עֲלָתָא דְמִתּוֹקְדָא עַל מַדְבְּחָא כָל לֵילְיָא עַד צַפְרָא וְאֶשָׁתָא דְמַדְבְּחָא תְּהֵא יָקְדָא בֵהּ
וְיַלְבֵּשׁ כַּהֲנָא לְבוּשִׁין דְבוּץ וּמִכְנְסִין דְבוּץ יַלְבֵּשׁ עַל בִּסְרֵהּ וְיַפְרֵשׁ יָת קִטְמָא דִי תֵיכוּל אֶשָׁתָא יָת עֲלָתָא עַל מַדְבְּחָא וִישַׁוִנֵהּ בִּסְטַּר מַדְבְּחָא
וְיַשְׁלַח יָת לְבוּשׁוֹהִי וְיִלְבַּשׁ לְבוּשִׁין אָחֳרָנִין וְיַפֵּק יָת קִטְמָא לְמִבָּרָא לְמַשְׁרִיתָא לַאֲתַר דְכֵי
וְאֶשָׁתָא עַל מַדְבְּחָא תְהֵי יָקְדָא בֵהּ לָא תִטְּפֵי וְיַבְעַר עֲלַהּ כַּהֲנָא אָעַיָא בִּצְפַר בִּצְפָר וְיַסְדַר עֲלַהּ עֲלָתָא וְיַסֵק עֲלַהּ תַּרְבֵּי נִכְסַת קוּדְשַׁיָא
אֶשָׁתָא תְּדִירָא תְּהֵי יָקְדָא עַל מַדְבְּחָא לָא תִטְפֵי
וְדָא אוֹרַיְתָא דְמִנְחָתָא דִיקָרְבוּן יָתַהּ בְּנֵי אַהֲרֹן קֳדָם יְיָ לָקֳדָם מַדְבְּחָא
וְיַפְרֵשׁ מִנֵהּ בְּקֻמְצֵהּ מִסֻלְתָּא דְמִנְחָתָא וּמִמִשְׁחַהּ וְיָת כָּל לְבוּנְתָּא דִי עַל מִנְחָתָא וְיַסֵק לְמַדְבְּחָא לְאִתְקַבָּלָא בְרַעֲוָא אַדְכַּרְתָּא קֳדָם יְיָ
וּדְאִשְׁתָּאַר מִנַהּ יֵכְלוּן אַהֲרֹן וּבְנוֹהִי פַּטִיר תִּתְאַכֵל בַּאֲתַר קַדִּישׁ בְּדָרַת מַשְׁכַּן זִמְנָא יֵכְלֻנַהּ
לָא תִתְאֲפֵי חֲמִיעַ חָלָקְהוֹן יְהָבִית יָתַהּ מִקֻרְבָּנָי קֹדֶשׁ קוּדְשִׁין הִיא כְּחַטָאתָא וְכַאֲשָׁמָא
כָּל דְכוּרָא בִּבְנֵי אַהֲרֹן יֵכְלֻנַהּ קְיָם עָלָם לְדָרָתֵיכוֹן מִקֻרָבָּנַיָא דַיְיָ כֹּל דְיִקְרַב בְּהוֹן יִתְקַדָּשׁ
וּמַלִיל יְיָ עִם מֹשֶׁה לְמֵימָר
דֵין קֻרְבַּן אַהֲרֹן וּבְנוֹהִי דִי יְקָרְבוּן קֳדָם יְיָ בְּיוֹמָא דְרַבִּיאוּ יָתֵהּ חַד מִן עַסְרָא בִתְלַת סְאִין סֻלְתָּא מִנְחָתָא תְּדִירָא פַּלְגוּתַהּ בְּצַפְרָא וּפַלְגוּתַהּ בְּרַמְשָׁא
עַל מַסְרֵיתָא בִּמְשַׁח תִּתְעֲבֵד רְבִיכָא תַיְתִנַהּ תּוּפִינֵי מִנְחַת בִּצוּעִין תְּקָרֵב לְאִתְקַבָּלָא בְרַעֲוָא קֳדָם יְיָ
וְכַהֲנָא דִמְרַבָּא תְחוֹתוֹהִי מִבְּנוֹהִי יַעְבֵּד יָתַהּ קְיָם עָלָם קֳדָם יְיָ גְמִיר תִּתַּסָק
וְכָל מִנְחָתָא דְכַהֲנָא גְמִיר תְּהֵי לָא תִתְאֲכֵל
וּמַלִיל יְיָ עִם מֹשֶׁה לְמֵימָר
מַלֵל עִם אַהֲרֹן וְעִם בְּנוֹהִי לְמֵימָר דָא אוֹרַיְתָא דְחַטָאתָא בַּאֲתַר דִי תִתְנְכֵס עֲלָתָא תִּתְנְכֵס חַטָאתָא קֳדָם יְיָ קֹדֶשׁ קוּדְשִׁין הִיא
כַּהֲנָא דִמְכַפֵּר בִּדְמַהּ יֵיכְלִנַהּ בַּאֲתַר קַדִישׁ תִּתְאֲכֵל בְּדָרַת מַשְׁכַּן זִמְנָא
כֹּל דִי יִקְרַב בְּבִסְרַהּ יִתְקַדַשׁ וְדִי יַדִי מִדְמַהּ עַל לְבוּשׁ דִי יַדִי עֲלַהּ תִּתְחַוָר בַּאֲתַר קַדִישׁ
וּמָן דַחֲסַף דִי תִתְבַּשֵׁל בֵּהּ יִתָּבָר וְאִם בְּמָנָא דִנְחָשָׁא תִּתְבַּשֵׁל וְיִתְמְרֵק וְיִשְׁתַּטֵף בְּמַיָא
כָּל דְכוּרָא בְּכַהֲנַיָא יֵכוּל יָתַהּ קֹדֶשׁ קוּדְשִׁין הִיא
וְכָל חַטָאתָא דִי יִתָּעַלּ מִדְמַהּ לְמַשְׁכַּן זִמְנָא לְכַפָּרָא בְקוּדְשָׁא לָא תִתְאֲכֵל בְּנוּרָא תִּתּוֹקָד
