וְקָרִיבוּ רֵישֵׁי אֲבָהָתָא לְּזַרְעִית בְּנֵי גִלְעָד בַּר מָכִיר בַּר מְנַשֶׁה מִזַרְעֲיַת בְּנֵי יוֹסֵף וּמַלִילוּ קֳדָם משֶׁה וְקָדָם רַבְרְבַיָא רֵישֵׁי אֲבָהָתָא לִבְנֵי יִשְׂרָאֵל
וַאֲמָרוּ יָת רִבּוֹנִי פַּקִיד יְיָ לְמִתַּן יָת אַרְעָא בְאַחֲסָנָא בְּעַדְבָא לִבְנֵי יִשְׂרָאֵל וְרִבּוֹנִי אִתְפַּקֵד בְּמֵימְרָא דַיְיָ לְמִתַּן יָת אַחֲסָנַת צְלָפְחָד אָחוּנָא לִבְנָתֵהּ
וִיהֶוְיַן לְחַד מִבְּנֵי שִׁבְטַיָא דִבְנֵי יִשְׂרָאֵל לִנְשִׁין וְתִתְמְנַע אַחֲסַנְתְּהוֹן מֵאַחֲסָנַת אֲבְהָתָנָא וְתִתּוֹסַף עַל אַחֲסָנַת שִׁבְטַיָא דִיֶהֶוְיָן לְהוֹן וּמֵעֲדַב אַחֲסַנְתָּנָא יִתְמְנָע
וְאִם יְהֵי יוֹבֵלָא לִבְנֵי יִשְׂרָאֵל וְתִתּוֹסַף אַחֲסַנְתְּהוֹן עַל אַחֲסָנַת שִׁבְטָא דִי תְהֶוְיָן לְהוֹן וּמֵאַחֲסָנַת שִׁבְטָא דְאֲבְהָתָנָא יִתְמְנַע אַחֲסַנְתְּהֶן
וּפַקִיד משֶׁה יָת בְּנֵי יִשְׂרָאֵל עַל מֵימְרָא דַיְיָ לְמֵימָר יָאוּת שִׁבְטָא דִבְנֵי יוֹסֵף מְמַלְלִין
דֵין פִּתְגָמָא דִי פַקִיד יְיָ לִבְנַת צְלָפְחָד לְמֵימַר לִדְתַקִין בְּעֵינֵיהוֹן תְּהֶוְיַן לִנְשִׁין בְּרַם לְזַרְעִית שִׁבְטָא דַאֲבוּהֶן תְּהֶוְיַן לִנְשִׁין
וְלָא תַסְחַר אַחֲסָנָא לִבְנֵי יִשְׂרָאֵל מִשִׁבְטָא לְשִׁבְטָא אֲרֵי גְבַר בְּאַחֲסָנַת שִׁבְטָא דַאֲבָהָתוֹהִי יִדְבְּקוּן בְּנֵי יִשְׂרָאֵל
וְכָל בְּרַתָּא יַרְתַת אַחֲסָנָא מִשִׁבְטַּיָא דִבְנֵי יִשְׂרָאֵל לְחַד מִזַרְעִית שִׁבְטָא דַאֲבוּהָא תְּהֵי לְאִנְתּוּ בְּדִיל דְיֵרְתוּן בְּנֵי יִשְׂרָאֵל גְבַר אַחֲסָנַת אֲבָהָתוֹהִי
וְלָא תַסְחַר אַחֲסָנָא מִשִׁבְטָא לְשִׁבְטָא אוֹחֲרָנָא אֲרֵי גְבַר בְּאַחֲסַנְתֵּהּ יִדְבְּקוּן שִׁבְטַיָא דִבְנֵי יִשְׂרָאֵל
כְּמָא דִי פַקִיד יְיָ יָת משֶׁה כֵּן עֲבָדוּ בְּנַת צְלָפְחָד
וַהֲוָאָה מַחְלָה תִרְצָה וְחָגְלָה וּמִלְכָּה וְנֹעָה בְּנַת צְלָפְחָד לִבְנֵי אֲחֵי אֲבוּהֶן לִנְשִׁין
מִזַרְעֲיַת בְּנֵי מְנַשֶׁה בַר יוֹסֵף הֲוָאָה לִנְשִׁין וַהֲוַת אַחֲסַנְתְּהֶן עַל שִׁבְטָא זַרְעִית אֲבוּהֶן
אִלֵין פִּקוּדַיָא וְדִינַיָא דִי פַקִיד יְיָ בִּידָא דְמשֶׁה לְוַת בְּנֵי יִשְׂרָאֵל בְּמֵישְׁרַיָא דְמוֹאָב עַל יַרְדְנָא דִירֵיחוֹ
