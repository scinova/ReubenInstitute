וְתַעְבֵּד יָת מַדְבְּחָא דְאָעֵי שִׁטִין חֲמֵשׁ אַמִין אֻרְכָּא וַחֲמֵשׁ אַמִין פּוּתְיָא מְרַבַּע יְהֵי מַדְבְּחָא וּתְלַת אַמִין רוּמֵהּ:
וְתַעְבֵּד קַרְנוֹהִי עַל אַרְבַּע זִוְיָתֵהּ מִנֵהּ תְּהֶוְיָן קַרְנוֹהִי וְתַחֲפֵי יָתֵהּ נְחָשָׁא:
וְתַעְבֵּד פְּסַכְתְּרָוָתֵהּ לְמִסְפֵי קִטְּמֵיהּ וּמַגְרוֹפְיָתֵהּ וּמִזְרְקָתֵהּ וְצִנוּרְיָתֵהּ וּמַחְתְּיָתֵהּ לְכָל מָנוֹהִי תַּעְבֵּד נְחָשָׁא:
וְתַעְבֵּד לֵהּ סְרָדָא עוֹבַד מְצַדְתָּא דִנְחָשָׁא וְתַעְבֵּד עַל מְצַדְתָּא אַרְבַּע עִזְקָן דִנְחָשָׁא עַל אַרְבַּע סִטְּרוֹהִי:
וְתִתֵּן יָתַהּ תְּחוֹת סוֹבָבֵי מַדְבְּחָא מִלְרָע וּתְהֵי מְצַדְתָּא עַד פַּלְגוּת מַדְבְּחָא:
וְתַעְבֵּד אֲרִיחַיָא לְמַדְבְּחָא אֲרִיחֵי דְאָעֵי שִׁטִין וְתַחֲפֵי יָתְהוֹן נְחָשָׁא:
וְיָעֵל יָת אֲרִיחוֹהִי בְּעִזְקָתָא וִיהוֹן אֲרִיחַיָא עַל תְּרֵין סִטְרֵי מַדְבְּחָא בְּמִטַל יָתֵהּ:
חֲלִיל לוּחֵי תַּעְבֵּד יָתֵהּ כְּמָא דִי אַחֲזִי יָתָךְ בְּטוּרָא כֵּן יַעְבְּדוּן:
וְתַעְבֵּד יָת דָרַת מַשְׁכְּנָא לְרוּחַ עֵבַר דָרוֹמָא סְרָדִין לְדָרְתָּא דְבוּץ שְׁזִיר מְאָה בְאַמִין אֻרְכָּא לְעֵבֶר חָד:
וְעַמוּדוֹהִי עֶשְׂרִין וְסַמְכֵיהוֹן עֶשְׂרִין נְחָשָׁא וָוֵי עַמוּדַיָא וְכִבּוּשֵׁיהוֹן כְּסָף:
וְכֵן לְרוּחַ צִפּוּנָא בְּאֻרְכָּא סְרָדִין מְאָה אֻרְכָּא וְעַמוּדוֹהִי עֶשְׂרִין וְסַמְכֵיהוֹן עֶשְׂרִין נְחָשָׁא וָוֵי עַמוּדַיָא וְכִבּוּשֵׁהוֹן דִכְסָף:
וּפוּתְיָא דְדַרְתָּא לְרוּחַ מַעַרְבָא סְרָדִין חַמְשִׁין אַמִין עַמוּדֵיהוֹן עַשְׂרָא וְסַמְכֵיהוֹן עַשְׂרָא:
וּפוּתְיָא דְדַרְתָּא לְרוּחַ קִדוּמָא מַדִינְחָא חַמְשִׁין אַמִין:
וַחֲמֵשׁ עֶשְׂרֵי אַמִין סְרָדִין לְעִבְרָא עַמוּדֵיהוֹן תְּלָתָא וְסַמְכֵיהוֹן תְּלָתָא:
וּלְעִבְרָא תִּנְיָנָא חֲמֵשׁ עֶשְׂרֵי סְרָדִין עֲמוּדֵיהוֹן תְּלָתָא וְסַמְכֵיהוֹן תְּלָתָא:
וְלִתְרַע דַרְתָּא פְּרָסָא עֶשְׂרִין אַמִין תִּכְלָא וְאַרְגְוָנָא וּצְבַע זְהוֹרִי וּבוּץ שְׁזִיר עוֹבַד צַיָר עַמוּדֵיהוֹן אַרְבְּעָא וְסַמְכֵיהוֹן אַרְבְּעָא:
כָּל עַמוּדֵי דַרְתָּא סְחוֹר סְחוֹר מְכַבְּשִׁין כְּסַף וָוֵיהוֹן כְּסַף וְסַמְכֵיהוֹן נְחָשָׁא:
אֻרְכָּא דְדַרְתָּא מְאָהּ בְאַמִין וּפוּתְיָא חַמְשִׁין בְּחַמְשִׁין וְרוּמָא חֲמֵשׁ אַמִין דְבוּץ שְׁזִיר וְסַמְכֵיהוֹן נְחָשָׁא:
לְכֹל מָאנֵי מַשְׁכְּנָא בְּכֹל פָּלְחָנֵהּ וְכָל סִכּוֹהִי וְכָל סִכֵּי דְדַרְתָּא נְחָשָׁא:ססס:
וְאַתְּ תְּפַקֵד יָת בְּנֵי יִשְׂרָאֵל וְיִסְבוּן לָךְ מִשְׁחָא זֵיתָא דַכְיָא כָּתִישָׁא לְאַנְהָרָא לְאַדְלָקָא בוֹצִינַיָא תְּדִירָא:
בְּמַשְׁכַּן זִמְנָא מִבָּרָא לְפָרֻכְתָּא דְעַל סַהֲדוּתָא יְסַדַר יָתֵהּ אַהֲרֹן וּבְנוֹהִי מֵרַמְשָׁא עַד צַפְרָא קָדָם יְיָ קְיָם עֲלָם לְדָרֵיהוֹן מִן בְּנֵי יִשְׂרָאֵל:
