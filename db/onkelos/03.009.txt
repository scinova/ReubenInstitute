וַהֲוָה בְּיוֹמָא תְמִינָאָה קְרָא מֹשֶׁה לְאַהֲרֹן וְלִבְנוֹהִי וּלְסָבֵי יִשְׂרָאֵל
וַאֲמַר לְאַהֲרֹן סַב לָךְ עֵגֶל בַּר תּוֹרֵי לְחַטָּאתָא וּדְכַר לַעֲלָתָא שַׁלְמִין וְקָרֵב קֳדָם יְיָ
וְעִם בְּנֵי יִשְׂרָאֵל תְּמַלֵל לְמֵימָר סִבוּ צְפִיר בַּר עִזִין לְחַטָאתָא וְעֵגֶל וְאִמָר בְּנֵי שְׁנָא שַׁלְמִין לַעֲלָתָא
וְתוֹר וּדְכַר לְנִכְסַת קוּדְשַׁיָא לְדַבָּחָא קֳדָם יְיָ וּמִנְחָתָא דְפִילָא בִמְשָׁח אֲרֵי יוֹמָא דֵין יְקָרָא דַיְיָ מִתְגְלִי לְכוֹן
וּנְסִיבוּ יָת דִי פַקִיד מֹשֶׁה לָקֳדָם מַשְׁכַּן זִמְנָא וּקְרִיבוּ כָּל כְּנִישְׁתָּא וְקָמוּ קֳדָם יְיָ
וַאֲמַר מֹשֶׁה דֵין פִּתְגָמָא דִי פַקִיד יְיָ תַּעְבְּדוּן וְיִתְגְלִי לְכוֹן יְקָרָא דַיְיָ
וַאֲמַר מֹשֶׁה לְאַהֲרֹן קְרַב לְמַדְבְּחָא וְעִבֵד יָת חַטָאתָךְ וְיָת עֲלָתָךְ וְכַפֵּר עֲלָךְ וְעַל עַמָא וְעִבֵד יָת קֻרְבַּן עַמָא וְכַפֵּר עֲלֵיהוֹן כְּמָא דִי פַקִיד יְיָ
וּקְרַב אַהֲרֹן לְמַדְבְּחָא וּנְכֵס יָת עֶגְלָא דְחַטָאתָא דִי לֵהּ
וְקָרִיבוּ בְּנֵי אַהֲרֹן יָת דְמָא לֵהּ וּטְבַל אֶצְבְּעֵהּ בִּדְמָא וִיהַב עַל קַרְנַת מַדְבְּחָא וְיָת דְמָא אֲרִיק לִיסוֹדָא דְמַדְבְּחָא
וְיָת תַּרְבָּא וְיָת כָּלְיֵתָא וְיָת חִצְרָא מִן כַּבְדָא מִן חַטָאתָא אַסֵק לְמַדְבְּחָא כְּמָא דִי פַקִיד יְיָ יָת מֹשֶׁה
וְיָת בִּשְׂרָא וְיָת מַשְׁכָא אוֹקִיד בְּנוּרָא מִבָּרָא לְמַשְׁרִיתָא
וּנְכֵס יָת עֲלָתָא וְאַמְטִיאוּ בְּנֵי אַהֲרֹן לֵּהּ יָת דְמָא וּזְרָקֵהּ עַל מַדְבְּחָא סְחוֹר סְחוֹר
וְיָת עֲלָתָא אַמְטִיאוּ לֵהּ לְאֵבָרָהָא וְיָת רֵישָׁא וְאַסֵק עַל מַדְבְּחָא
וְחַלִיל יָת גַוָא וְיָת כְּרָעַיָא וְאַסֵק עַל עֲלָתָא לְמַדְבְּחָא
וְקָרֵב יָת קֻרְבַּן עַמָא וּנְסִב יָת צְפִירָא דְחַטָאתָא דִי לְעַמָא וְנָכְסֵהּ וְכַפֵּר בִּדְמֵהּ כְּקַדְמָאָה
וְקָרֵב יָת עֲלָתָא וְעָבְדַהּ כִּדְחָזֵי
וְקָרִיב יָת מִנְחָתָא וּמְלָא יְדֵהּ מִנַהּ וְאַסֵק עַל מַדְבְּחָא בַּר מֵעֲלַת צַפְרָא
וּנְכֵס יָת תּוֹרָא וְיָת דִכְרָא נִכְסַת קוּדְשַׁיָא דִי לְעַמָא וְאַמְטִיאוּ בְּנֵי אַהֲרֹן יָת דְמָא לֵהּ וּזְרָקֵהּ עַל מַדְבְּחָא סְחוֹר סְחוֹר
וְיָת תַּרְבַּיָא מִן תּוֹרָא וּמִן דִכְרָא אֲלִיתָא וְחָפֵי גַוָא וְכָלְיֵתָא וַחֲצַר כַּבְדָא
וְשַׁוִיאוּ יָת תַּרְבַּיָא עַל חֶדְוָתָא וְאַסֵק תַּרְבַּיָא לְמַדְבְּחָא
וְיָת חֶדְוָתָא וְיָת שׁוֹקָא דְיַמִינָא אֲרֵים אַהֲרֹן אֲרָמָא קֳדָם יְיָ כְּמָא דִי פַקִיד מֹשֶׁה
וַאֲרֵם אַהֲרֹן יָת יְדוֹהִי עַל עַמָא וּבָרֵכִנוּן וּנְחַת מִלְמֶעְבַּד חַטָאתָא וַעֲלָתָא וְנִכְסַת קוּדְשַׁיָא
וְעַל מֹשֶׁה וְאַהֲרֹן לְמַשְׁכַּן זִמְנָא וּנְפָקוּ וּבָרִיכוּ יָת עַמָא וְאִתְגְלִי יְקָרָא דַיְיָ לְכָל עַמָא
וּנְפָקַת אֶשָׁתָא מִן קֳדָם יְיָ וַאֲכָלַת עַל מַדְבְּחָא יָת עֲלָתָא וְיָת תַּרְבַּיָא וַחֲזָא כָּל עַמָא וְשַׁבָּחוּ וּנְפָלוּ עַל אַפֵּיהוֹן
