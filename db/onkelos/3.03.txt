וְאִם נִכְסַת קוּדְשַׁיָא קוּרְבָּנֵיהּ אִם מִן תּוֹרֵי הוּא מְקָרֵב אִם דְכַר אִם נוּקְבָא שְׁלִים יְקָרְבִינֵיהּ קֳדָם יְיָ
וְיִסְמוֹךְ יְדֵיהּ עַל רֵישׁ קוּרְבָּנֵיהּ וְיִכְסִינֵיהּ בִּתְרַע מַשְׁכַּן זִמְנָא וְיִזְרְקוּן בְּנֵי אַהֲרֹן כַּהֲנַיָא יַת דְמָא עַל מַדְבְּחָא סְחוֹר סְחוֹר
וִיקָרֵב מִנִכְסַת קוּדְשַׁיָא קוּרְבָּנָא קֳדָם יְיָ יַת תַּרְבָּא דְחָפֵי יַת גַוָא וְיַת כָּל תַּרְבָּא דִי עַל גַוָא
וְיַת תַּרְתֵּין כֻּלְיָן וְיַת תַּרְבָּא דִי עֲלֵיהֶן דִי עַל גִסְסַיָא וְיָת חַצְרָא דְעַל כַּבְדָא עַל כּוּלְיָתָא יַעֲדִינָהּ
וְיִסְקוּן יָתֵיהּ בְּנֵי אַהֲרֹן לְמַדְבְּחָא עַל עֲלָתָא דִי עַל אָעַיָא דִי עַל אֶשָׁתָא קוּרְבַּן דְמִתְקַבֵּל בְּרַעֲוָא קֳדָם יְיָ
וְאִם מִן עָנָא קוּרְבָּנֵיהּ לְנִכְסַת קוּדְשַׁיָא קֳדָם יְיָ דְכַר אוֹ נוּקְבָא שְׁלִים יְקָרְבִינֵיהּ
אִם אִימַר הוּא מְקָרֵב יַת קוּרְבָּנֵיהּ וִיקָרֵיב יָתֵי קֳדָם יְיָ
וְיִסְמוֹךְ יַת יְדֵיהּ עַל רֵישׁ קוּרְבָּנֵיהּ וְיִכּוֹס יָתֵיהּ קֳדָם מַשְׁכַּן זִמְנָא וְיִזְרְקוּן בְּנֵי אַהֲרֹן יַת דְמֵיהּ עַל מַדְבְּחָא סְחוֹר סְחוֹר
וִיקָרֵב מִנִכְסַת קוּדְשַׁיָא קֻרְבָּנָא קֳדָם יְיָ תַּרְבֵּיהּ אֲלִיתָא שְׁלִמְתָּא לָקֳבֵל שֵׁזַרְתָּא יֶעְדִנַהּ וְיָת תַּרְבָּא דְחָפֵי יָת גַוָא וְיָת כָּל תַּרְבָּא דִי עַל גַוָא
וְיָת תַּרְתֵּין כָּלְיָן וְיָת תַּרְבָּא דִי עֲלֵיהֶן דִי עַל גִסְסַיָא וְיָת חַצְרָא דְעַל כַּבְדָא עַל כָּלְיֵתָא יֶעְדִנַהּ
וְיַסְקִנֵהּ כַּהֲנָא לְמַדְבְּחָא לְחֵם קֻרְבָּנָא קֳדָם יְיָ
וְאִם מִן בְּנֵי עִזַיָא קֻרְבָּנֵהּ וִיקָרְבִנֵהּ קֳדָם יְיָ
וְיִסְמוֹךְ יָת יְדֵהּ עַל רֵישֵׁיהּ וְיִכּוֹס יָתֵהּ קֳדָם מַשְׁכַּן זִמְנָא וְיִזְרְקוּן בְּנֵי אַהֲרֹן יָת דְמֵהּ עַל מַדְבְּחָא סְחוֹר סְחוֹר
וִיקָרֵב מִנֵהּ קֻרְבָּנֵהּ קֻרְבָּן קֳדָם יְיָ יָת תַּרְבָּא דְחָפֵי יָת גַוָא וְיָת כָּל תַּרְבָּא דִי עַל גַוָא
וְיָת תַּרְתֵּין כָּלְיָן וְיָת תַּרְבָּא דִי עֲלֵיהֶן דִי עַל גִסְסַיָא וְיָת חַצְרָא דִי עַל כַּבְדָא עַל כָּלְיֵתָא יֶעְדִנַהּ
וְיַסְקִנוּן כַּהֲנָא לְמַדְבְּחָא לְחֵם קֻרְבָּנָא לְאִתְקַבָּלָא בְּרַעֲוָא כָּל תַּרְבָּא קֳדָם יְיָ
קְיָם עָלָם לְדָרָתֵיכוֹן בְּכָל מוֹתְבָנֵיכוֹן כָּל תַּרְבָּא וְכָל דְמָא לָא תֵיכְלוּן
