וְאַתְּ קָרֵב לְוָתָךְ יָת אַהֲרֹן אָחוּךְ וְיָת בְּנוֹהִי עִמֵהּ מִגוֹ בְּנֵי יִשְׂרָאֵל לְשַׁמָשָׁא קֳדָמַי אַהֲרֹן נָדָב וַאֲבִיהוּא אֶלְעָזָר וְאִיתָמָר בְּנֵי אַהֲרֹן:
וְתַעְבֵּד לְבוּשֵׁי קוּדְשָׁא לְאַהֲרֹן אָחוּךְ לִיקָר וּלְתֻשְׁבְּחָן:
וְאַתְּ תְּמַלֵל עִם כָּל חַכִּימֵי לִבָּא דְאַשְׁלֵמִית עִמֵהּ רוּחַ חָכְמְתָא וְיַעְבְּדוּן יָת לְבוּשֵׁי אַהֲרֹן לְקַדָשׁוּתֵהּ לְשַׁמָשָׁא קֳדָמַי:
וְאִלֵין לְבוּשַׁיָא דִי יַעְבְּדוּן חוּשְׁנָא וְאֵפוֹדָא וּמְעִילָא וְכִתּוּנָא מְרַמְצָא מִצְנֶפְתָּא וְהֶמְיָנָא וְיַעְבְּדוּן לְבוּשֵׁי קוּדְשָׁא לְאַהֲרֹן אָחוּךְ וְלִבְנוֹהִי לְשַׁמָשָׁא קֳדָמַי:
וְאִנוּן יִסְבוּן יָת דַהֲבָא וְיָת תִּכְלָא וְיָת אַרְגְוָנָא וְיָת צְבַע זְהוֹרִי וְיָת בּוּצָא:
וְיַעְבְּדוּן יָת אֵפוֹדָא דַהֲבָא תִּכְלָא וְאַרְגְוָנָא צְבַע זְהוֹרִי וּבוּץ שְׁזִיר עוֹבַד אֳמָן:
תַּרְתֵּין כַּתְפִּין מְלַפְּפָן יְהוֹן לֵהּ עַל תְּרֵין סִטְרוֹהִי וְיִתְלָפַף:
וְהֶמְיַן תִּקוּנֵי דִי עֲלוֹהִי כְּעוֹבָדוֹהִי מִנֵהּ יְהֵי דַהֲבָא תִּכְלָא וְאַרְגְּוָנָא וּצְבַע זְהוֹרִי וּבוּץ שְׁזִיר:
וְתִסַב יָת תַּרְתֵּין אַבְנֵי בוּרְלָא וְתִגְלוֹף עֲלֵיהֶן שְׁמָהַת בְּנֵי יִשְׂרָאֵל:
שִׁתָּא מִשְׁמָהָתְהוֹן עַל אַבְנָא חֲדָא וְיָת שְׁמָהַת שִׁתָּא דְאִשְׁתָּאָרוּ עַל אַבְנָא תִנְיֵתָא כְּתוּלְדָתְהוֹן:
עוֹבַד אֳמָן אֶבֶן טָבָא כְּתַב מְפָרַשׁ כִּגְלֹף דְעִזְקָא תִגְלוֹף יָת תַּרְתֵּין אַבְנַיָא עַל שְׁמָהַת בְּנֵי יִשְׂרָאֵל מְשַׁקְעָן מְרַמְצָן דִדְהַב תַּעְבֵּד יָתְהוֹן:
וּתְשַׁוִּי יָת תַּרְתֵּין אַבְנַיָא עַל כִּתְפֵּי אֵפוֹדָא אַבְנֵי דָכְרָנָא לִבְנֵי יִשְׂרָאֵל וְיִטּוֹל אַהֲרֹן יָת שְׁמָהָתְהוֹן קֳדָם יְיָ עַל תְּרֵין כִּתְפּוֹהִי לְדָכְרָנָא:
וְתַעְבֵּד מְרַמְצָן דִדְהַב:
וְתַרְתֵּין תִּכִּין דְהַב דְכֵי מְתַחְמָן תַּעְבֵד יָתְהוֹן עוֹבַד גְדִילוּ וְתִתֵּן יָת תִּכַּיָא גְדִילָתָא עַל מְרַמְצָתָא:
וְתַעְבֵּד חשֶׁן דִינָא עוֹבַד אֳמָן כְּעוֹבַד אֵפוֹדָא תַּעְבְּדִנֵהּ דַהֲבָא תִּכְלָא וְאַרְגְוָנָא וּצְבַע זְהוֹרִי וּבוּץ שְׁזִיר תַּעְבֵּד יָתֵהּ:
מְרַבַּע יְהֵי עֵיף זֶרְתָּא אֻרְכֵּה וְזֶרְתָּא פְתָיֵהּ:
וְתַשְׁלֵם בֵּה אַשְׁלָמוּת אַבְנָא אַרְבְּעָא סִדְרִין אֶבֶן טָּבָא סִדְרָא קַדְמָאָה סָמְקָן יַרְקָן וּבַרְקָן סִדְרָא חָד:
וְסִדְרָא תִּנְיָנָא אִזְמָרַגְדִין וְשַׁבְזִיז וְסִבְהֲלוֹם:
וְסִדְרָא תְּלִיתָאָה קַנְכִּירֵי טַרְקְיָא וְעֵין עֶגְלָא:
וְסִדְרֵא רְבִיעָאָה כְּרוּם יַמָא וּבוּרְלָא וּפַנְטֵרֵי מְרַמְצָן דִדְהַב יְהוֹן בְּאַשְׁלָמוּתְהוֹן:
וְאַבְנַיָא יְהֶוְיַן עַל שְׁמָהַת בְּנֵי יִשְׂרָאֵל תַּרְתֵּי עֶסְרֵי עַל שְׁמָהַתְהוֹן כְּתַב מְפָרַשׁ כִּגְלוֹף דְעִזְקָא גְבַר עַל שְׁמֵהּ תְּהֶוְיָן לִתְרֵין עֲסַר שִׁבְטִין:
וְתַעְבֵּד עַל חוּשְׁנָא תִּכִּין מְתַחְמָן עוֹבַד גְדִילוּ דְהַב דְכֵי:
וְתַעְבֵּד עַל חוּשְׁנָא תַּרְתֵּין עִזְקָן דִדְהַב וְתִתֵּן יָת תַּרְתֵּין עִזְקָתָא עַל תְּרֵין סִטְרֵי חוּשְׁנָא:
וְתִתֵּן יָת תַּרְתֵּין גְדִילָן דִדְהַב עַל תַּרְתֵּין עִזְקָתָא לְסִטְרֵי חוּשְׁנָא:
וְיָת תַּרְתֵּין גְדִילָן דְעַל תַּרְתֵּין סִטְרוֹהִי תִּתֵּן עַל תַּרְתֵּין מְרַמְצָתָא וְתִתֵּן עַל כִּתְפֵּי אֵפוֹדָא לָקֳבֵל אַפּוֹהִי:
וְתַעְבֵּד תַּרְתֵּין עִזְקָן דִדְהַב וּתְשַׁוִּי יָתְהוֹן עַל תְּרֵין סִטְרֵי חוּשְׁנָא עַל סִפְתֵהּ דִי לְעִבְרָא דְאֵפוֹדָא מִלְגָו:
וְתַעְבֵּד תַּרְתֵּין עִזְקָן דִדְהַב וְתִתֵּן יָתְהוֹן עַל תַּרְתֵּין כִּתְפֵּי אֵפוֹדָא מִלְרַע מִלְקֳבֵל אַפּוֹהִי לָקֳבֵל בֵּית לוֹפֵי מֵעִלָוֵי לְהֶמְיַן אֵפוֹדָא:
וְיֶחְדוּן יָת חוּשְׁנָא מֵעִזְקְתֵהּ לְעִזְקְתָא דְאֵפוֹדָא בְּחוּטָא דִתְכֶלְתָּא לְמֶהֱוֵי עַל הֶמְיַן אֵפוֹדָא וְלָא יִתְפָּרֵק חוּשְׁנָא מֵעִלָוֵי אֵפוֹדָא:
וְיִטוֹל אַהֲרֹן יָת שְׁמָהַת בְּנֵי יִשְׂרָאֵל בְּחשֶׁן דִינָא עַל לִבֵּהּ בְּמֵעֲלֵהּ לְקוּדְשָׁא לְדוּכְרָנָא קֳדָם יְיָ תְּדִירָא:
וְתִתֵּן בְּחשֶׁן דִינָא יָת אוּרַיָא וְיָת תֻּמַיָא וִיהוֹן עַל לִבָּא דְאַהֲרֹן בְּמֵעֲלֵהּ קֳדָם יְיָ וְיִטוֹל אַהֲרֹן יָת דִינָא דִבְנֵי יִשְׂרָאֵל עַל לִבֵּהּ קֳדָם יְיָ תְּדִירָא:
וְתַעְבֵּד יָת מְעִילָא דְאֵפוֹדָא גְמִיר תִּכְלָא:
וִיהֵי פוּם רֵישֵׁהּ כָּפִיל לְגַוֵּהּ תּוֹרָא יְהֵא מַקַף לְפוּמֵהּ סְחוֹר סְחוֹר עוֹבַד מָחֵי כְּפוּם שִׂרְיָן יְהֵי לֵהּ לָא יִתְבְּזָע:
וְתַעְבֵּד עַל שִׁיפוֹלוֹהִי רִמוֹנֵי תִּכְלָא וְאַרְגְוָנָא וּצְבַע זְהוֹרִי עַל שִׁיפוֹלוֹהִי סְחוֹר סְחוֹר וְזַגִין דְדַהֲבָא בֵּינֵיהוֹן סְחוֹר סְחוֹר:
זַגָא דְדַהֲבָא וְרִמוֹנָא זַגָא דְדַהֲבָא וְרִמוֹנָא עַל שִׁיפוֹלֵי מְעִילָא סְחוֹר סְחוֹר:
וִיהֵי עַל אַהֲרֹן לְשַׁמָשָׁא וְיִשְׁתְּמַע קָלֵהּ בְּמֵעֲלֵהּ לְקוּדְשָׁא קֳדָם יְיָ וּבְמִפְקֵהּ וְלָא יְמוּת:
וְתַעְבֵּד צִיצָא דִדְהַב דְכֵי וְתִגְלוֹף עֲלוֹהִי כְּתַב מִפָרַשׁ קֹדֶשׁ לַיָי:
וּתְשַׁוִי יָתֵהּ עַל חוּטָא דִתְכֶלְתָּא וִיהֵי עַל מִצְנֶפְתָּא לָקֳבֵל אַפֵּי מִצְנֶפְתָּא יְהֵי:
וִיהֵי עַל בֵּית עֵינוֹהִי דְאַהֲרֹן וְיִטּוֹל אַהֲרֹן יָת עֲוָיַת קוּדְשַׁיָא דִי יְקַדְשׁוּן בְּנֵי יִשְׂרָאֵל לְכָל מַתְּנַּת קוּדְשֵׁיהוֹן וִיהֵי עַל בֵּית עֵינוֹהִי תְּדִירָא לְרַעֲוָא לְהוֹן קֳדָם יְיָ:
וּתְרַמֵץ כִּתּוּנָא בוּצָא וְתַעְבֵּד מִצְנֶפְתָּא דְבוּצָא וְהֶמְיָנָא תַּעְבֵּד עוֹבַד צַיָר:
וְלִבְנֵי אַהֲרֹן תַּעְבֵּד כִּתּוּנִין וְתַעְבֵּד לְהוֹן הֶמְיָנִין וְכוֹבָעִין תַּעְבֵּד לְהוֹן לִיקָר וּלְתֻשְׁבְּחָן:
וְתַלְבֵּש יָתְהוֹן יָת אַהֲרֹן אָחוּךְ וְיָת בְּנוֹהִי עִמֵהּ וּתְרַבֵּי יָתְהוֹן וּתְקָרֵב יָת קֻרְבָּנְהוֹן וּתְקַדֵשׁ יָתְהוֹן וִישַׁמְשׁוּן קֳדָמַי:
וְעִבֵד לְהוֹן מִכְנְסִין דְבוּץ לְכַסָאָה בְשַׂר עֶרְיְתָא מֵחַרְצָן וְעַד יַרְכָּן יְהוֹן:
וִיהוֹן עַל אַהֲרֹן וְעַל בְּנוֹהִי בְּמֵעַלְהוֹן לְּמַשְׁכַּן זִמְנָא אוֹ בְמִקְרָבְהוֹן לְמַדְבְּחָא לְשַׁמָשָׁא בְקוּדְשָׁא וְלָא יְקַבְּלוּן חוֹבָא וִימוּתוּן קְיָם עָלָם לֵהּ וְלִבְנוֹהִי בַתְרוֹהִי:
