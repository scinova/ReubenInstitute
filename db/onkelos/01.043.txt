וְכַפְנָא תַּקִּיף בְּאַרְעָא
וַהֲוָה כַּד שֵׁיצִיאוּ לְמֵיכַל יָת עִיבוּרָא דְּאַיְתִיאוּ מִמִּצְרָיִם וַאֲמַר לְהוֹן אֲבוּהוֹן תּוּבוּ זְבוּנוּ לָנָא זְעֵיר עִיבוּרָא
וַאֲמַר לֵהּ יְהוּדָה לְמֵימָר אַסְהָדָא אַסְהֵד בָּנָא גַּבְרָא לְמֵימָר לָא תֶחֱזוּן אַפַּי אֶלָּהֵין כַּד אָחוּכוֹן עִמְּכוֹן
אִם אִיתָךְ מְשַׁלַּח יָת אָחוּנָא עִמָּנָא נֵיחוֹת וְנִזְבֵּן לָךְ עִיבוּרָא
וְאִם לֵיתָךְ מְשַׁלַּח לָא נֵיחוֹת אֲרֵי גַּבְרָא אֲמַר לָנָא לָא תֶחֱזוּן אַפַּי אֶלָּהֵין כַּד אֲחוּכוֹן עִמְּכוֹן
וַאֲמַר יִשְׂרָאֵל לְמָה אַבְאֶשְׁתּוּן לִי לְחַוָּאָה לְגַבְרָא הַעַד כְּעַן לְכוֹן אָחָא
וַאֲמָרוּ מִשְׁאַל שָׁאֵל גַּבְרָא לָנָא וּלְיַלָּדוּתָנָא לְמֵימַר הַעַד כְּעַן אֲבוּכוֹן קַיָּם הַאִית לְכוֹן אָחָא וְחַוֵּינָא לֵהּ עַל מֵימַר פִּתְגָּמַיָּא הָאִלֵּין הֲמִדַּע הֲוֵינָא יָדְעִין אֲרֵי יֵימַר אוֹחִיתוּ יָת אֲחוּכוֹן
וַאֲמַר יְהוּדָה לְיִשְׂרָאֵל אֲבוּהִי שְׁלַח עוּלֵימָא עִמִּי וְנֵקוּם וְנֵזֵיל וְנֵיחֵי וְלָא נְמוּת אַף אֲנַחְנָא אַף אַתְּ אַף טַפְלָנָא
אֲנָא מְעָרַבְנָא בֵהּ מִן יְדִי תִּבְעִנֵּהּ אִם לָא אַיְתִנֵּהּ לָךְ וַאֲקֵימִנֵּהּ קֳדָמָךְ וְאֵהֵי חָטֵי לָךְ כָּל יוֹמַיָּא
אֲרֵי אִלּוּלָפוֹן בְּדָא אִתְעַכָּבְנָא אֲרֵי כְעַן תַּבְנָא דְּנַן תַּרְתֵּין זִמְנִין
וַאֲמַר לְהוֹן יִשְׂרָאֵל אֲבוּהוֹן אִם כֵּן הָכָא דָּא עִיבִידוּ סִיבוּ מִדִּמְשַׁבַּח בְּאַרְעָא בְּמָנֵיכוֹן וְאוֹחִיתוּ לְגַבְרָא תִּקְרֻבְתָּא זְעֵיר קְטַף וּזְעֵיר דְּבַשׁ שְׁעַף וּלְטוֹם בָּטְנִין וְשִׁגְדִּין
וְכַסְפָּא עַל חַד תְּרֵין סִיבוּ בְיֶדְכוֹן וְיָת כַּסְפָּא דְּאִתּוֹתַב בְּפוּם טוֹעֲנֵיכוֹן תְּתִיבוּן בְּיֶדְכוֹן מָאִים שָׁלוּ הוּא
וְיָת אֲחוּכוֹן דְּבָרוּ וְקוּמוּ תּוּבוּ לְוַת גַּבְרָא
וְאֵל שַׁדַּי יִתֵּן לְכוֹן רַחֲמִין קֳדָם גַּבְרָא וְיִפְטַר לְכוֹן יָת אֲחוּכוֹן אָחֳרָנָא וְיָת בִּנְיָמִין וַאֲנָא כְּמָא דְּאִתְכֵּלִית תְּכֵלִית
וּנְסִיבוּ גֻּבְרַיָּא יָת תִּקְרֻבְתָּא הָדָא וְעַל חַד תְּרֵין כַּסְפָּא נְסִיבוּ בִידֵיהוֹן וּדְבָרוּ יָת בִּנְיָמִן וְקָמוּ וּנְחָתוּ לְמִצְרַיִם וְקָמוּ קֳדָם יוֹסֵף
וַחֲזָא יוֹסֵף עִמְּהוֹן יָת בִּנְיָמִין וַאֲמַר לְדִי מְמַנָּא עַל בֵּיתֵהּ אָעֵיל יָת גֻּבְרַיָּא לְבֵיתָא וּנְכוֹס נִכְסָתָא וְאַתְקֵין אֲרֵי עִמִּי יֵיכְלוּן גֻּבְרַיָּא בְּשֵׁירוּתָא
וַעֲבַד גַּבְרָא כְּמָא דִּי אֲמַר יוֹסֵף וְאָעֵיל גַּבְרָא יָת גֻּבְרַיָּא לְבֵית יוֹסֵף
וּדְחִילוּ גֻּבְרַיָּא אֲרֵי אִתַּעֲלוּ לְבֵית יוֹסֵף וַאֲמָרוּ עַל עֵסַק כַּסְפָּא דְּאִתּוֹתַב בְּטוֹעֲנָנָא בְּקַדְמֵיתָא אֲנַחְנָא מִתַּעֲלִין לְאִתְרַבְרָבָא עֲלָנָא וּלְאִסְתַּקָּפָא עֲלָנָא וּלְמִקְנֵי יָתָנָא לְעַבְדִּין וּלְמִדְבַּר יָת חֲמָרָנָא
וּקְרִיבוּ לְוַת גַּבְרָא דִּי מְמַנָּא עַל בֵּית יוֹסֵף וּמַלִּילוּ עִמֵּהּ בִּתְרַע בֵּיתָא
וַאֲמָרוּ בְּבָעוּ רִבּוֹנִי מֵיחַת נִחֵתְנָא בְּקַדְמֵיתָא לְמִזְבַּן עִיבוּרָא
וַהֲוָה כַּד אָתֵינָא לְבֵית מְבָתָא וּפְתַחְנָא יָת טוֹעֲנָנָא וְהָא כְסַף גְּבַר בְּפוּם טוֹעֲנֵהּ כַּסְפָּנָא בְּמַתְקְלֵהּ וַאֲתֵיבְנָא יָתֵהּ בִּידָנָא
וְכַסְפָּא אָחֳרָנָא אוֹחִיתְנָא בִידָנָא לְמִזְבַּן עִיבוּרָא לָא יְדַעְנָא מָן שַׁוִּי כַסְפָּנָא בְּטוֹעֲנָנָא
וַאֲמַר שְׁלַם לְכוֹן לָא תִדְחֲלוּן אֱלָהֲכוֹן וֵאֱלָהָא דַּאֲבוּכוֹן יְהַב לְכוֹן סִימָּא בְּטוֹעֲנֵיכוֹן כַּסְפְּכוֹן אֲתָא לְוָתִי וְאַפֵּיק לְוָתְהוֹן יָת שִׁמְעוֹן
וְאָעֵיל גַּבְרָא יָת גֻּבְרַיָּא לְבֵית יוֹסֵף וִיהַב מַיָּא וְאַסְחוֹ רִגְלֵיהוֹן וִיהַב כִּסְּתָא לַחֲמָרֵיהוֹן
וְאַתְקִינוּ יָת תִּקְרֻבְתָּא עַד דְּעַל יוֹסֵף בְּשֵׁירוּתָא אֲרֵי שְׁמָעוּ אֲרֵי תַמָּן יֵיכְלוּן לַחֲמָא
וְעַל יוֹסֵף לְבֵיתָא וְאַיְתִיאוּ לֵהּ יָת תִּקְרֻבְתָּא דִּי בִידֵיהוֹן לְבֵיתָא וּסְגִידוּ לֵהּ לְאַרְעָא
וּשְׁאֵיל לְהוֹן לִשְׁלַם וַאֲמַר הַשְׁלַם אֲבוּכוֹן סָבָא דִּי אֲמַרְתּוּן הַעַד כְּעַן קַיָּם
וַאֲמָרוּ שְׁלַם לְעַבְדָךְ לְאָבוּנָא עַד כְּעַן קַיָּם וּכְרָעוּ וּסְגִידוּ
וּזְקַף עֵינוֹהִי וַחֲזָא יָת בִּנְיָמִין אֲחוּהִי בַּר אִמֵּהּ וַאֲמַר הָדֵין אֲחוּכוֹן זְעֵירָא דִּי אֲמַרְתּוּן לִי וַאֲמַר מִן קֳדָם יְיָ יִתְרַחַם עֲלָךְ בְּרִי
