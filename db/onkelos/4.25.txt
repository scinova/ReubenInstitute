וִיתֵיב יִשְׂרָאֵל בְּשִׁיטִין וּשְׁרֵי עַמָא לְמִטְּעֵי בָּתַר בְּנַת מוֹאָב
וּקְרָאָן לְעַמָא לְדִבְחֵי טַעֲוַתְהוֹן וַאֲכַל עַמָא וּסְגִידוּ לְטַעֲוַתְהוֹן
וְאִתְחַבַּר יִשְׂרָאֵל לְבַעֲלָא פְעוֹר וּתְקֵף רוּגְזָא דַיְיָ בְּיִשְׂרָאֵל
וַאֲמַר יְיָ לְמשֶׁה דְבַר יַת כָּל רֵישֵׁי עַמָא וְדוּן וּקְטוֹל דְחַיָב קְטוֹל קֳדָם יְיָ לָקֳבֵל שִׁמְשָׁא וִיתוּב תְּקוֹף רוּגְזָא דַיְיָ מִיִשְׂרָאֵל
וַאֲמַר משֶׁה לְדַיָנֵי יִשְׂרָאֵל קְטִילוּ גְבַר גַבְרוֹהִי דְאִתְחַבָּרוּ לְבַעֲלָא פְעוֹר
וְהָא גַבְרָא מִבְּנֵי יִשְׂרָאֵל אֲתָא וְקָרֵב לְוָת אֲחוֹהִי יָת מִדְיָנֵיתָא לְעֵינֵי משֶׁה וּלְעֵינֵי כָּל כְּנִשְׁתָּא דִבְנֵי יִשְׂרָאֵל וְאִינוּן בָּכָן בִּתְרַע מַשְׁכַּן זִמְנָא
וַחֲזָא פִּינְחָס בַּר אֶלְעָזָר בַּר אַהֲרֹן כַּהֲנָא וְקָם מִגוֹ כְנִשְׁתָּא וּנְסֵיב רוֹמְחָא בִידֵיהּ
וְעַל בָּתַר גַבְרָא בַּר יִשְׂרָאֵל לְקוּבָּתָא וּבְזַע יַת תַּרְוֵיהוֹן יַת גַבְרָא בַּר יִשְׂרָאֵל וְיַת אִתְּתָא לִמְעָהָא וְאִתְכְּלִי מוֹתָנָא מֵעַל בְּנֵי יִשְׂרָאֵל
וַהֲווֹ דְמִיתוּ בְּמוֹתָנָא עַשְׂרִין וְאַרְבְּעָה אַלְפִין
וּמַלִיל יְיָ עִם משֶׁה לְמֵימָר
פִּינְחָס בַּר אֶלְעָזָר בַּר אַהֲרֹן כַּהֲנָא אֲתֵיב יַת חֵמְתִי מֵעַל בְּנֵי יִשְׂרָאֵל בִּדְקַנֵי יַת קִנְאָתִי בֵּינֵיהוֹן וְלָא שֵׁיצֵיתִי יַת בְּנֵי יִשְׂרָאֵל בְּקִנְאָתִי
בְּכֵן אֵימַר הָא אֲנָא גְזַר לֵיהּ יַת קְיָמִי שְׁלָם
וּתְהֵי לֵיהּ וְלִבְנוֹהִי בַּתְרוֹהִי קְיַם כְּהֻנַת עֲלָם חֲלַף דִי קַנֵי קֳדָם אֱלָהֵהּ וְכַפֵּר עַל בְּנֵי יִשְׂרָאֵל
וְשׁוּם גַבְרָא בַּר יִשְרָאֵל קְטִילָא דִי אִתְקְטֵל עִם מִדְיָנֵיתָא זִמְרִי בַּר סָלוּא רַב בֵּית אַבָּא לְבֵית שִׁמְעוֹן
וְשׁוּם אִתְּתָא דְאִתְקְטִילַת מִדְיָנֵיתָא כָּזְבִּי בַּת צוּר רֵישׁ אוּמֵי בֵּית אַבָּא בְּמִדְיָן הוּא
וּמַלִיל יְיָ עִם משֶׁה לְמֵימָר
אָעֵק יַת מִדְיָנָאֵי וְתִקְטְלוּן יָתְהוֹן
אֲרֵי מְעִיקִין אִינוּן לְכוֹן בְּנִכְלֵיהוֹן דִי נְכִילוּ לְכוֹן עַל עֵיסַק פְּעוֹר וְעַל עֵיסַק כָּזְבִּי בַּת רַבָּא דְמִדְיָן אֲחַתְהוֹן דְאִתְקְטִילַת בְּיוֹמָא דְמוֹתָנָא עַל עֵיסַק פְּעוֹר
