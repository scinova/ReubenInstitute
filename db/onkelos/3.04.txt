וּמַלִיל יְיָ עִם מֹשֶׁה לְמֵימָר
מַלֵל עִם בְּנֵי יִשְׂרָאֵל לְמֵימָר אֱנַשׁ אֲרֵי יֵחוֹב בְּשָׁלוּ מִכָּל פִּקוּדַיָא דַיְיָ דִי לָא כַשְׁרִין לְאִתְעֲבָדָא וְיַעְבֵּד מִן חָד מִ נְה וֹן
אִם כַּהֲנָא רַבָּא יֵיחוֹב לְחוֹבַת עַמָא וִיקָרֵב עַל חוֹבְתֵהּ דִי חָב תּוֹר בַּר תּוֹרֵי שְׁלִים קֳדָם יְיָ לְחַטָאתָא
וְיַיְתִי יָת תּוֹרָא לִתְרַע מַשְׁכַּן זִמְנָא לָקֳדָם יְיָ וְיִסְמוֹךְ יָת יְדֵיהּ עַל רֵישׁ תּוֹרָא וְיִכּוֹס יָת תּוֹרָא קֳדָם יְיָ
וְיִסַב כַּהֲנָא רַבָּא מִדְמָא דְתוֹרָא וְיָעֵל יָתֵהּ לְמַשְׁכַּן זִמְנָא
וְיִטְבּוֹל כַּהֲנָא יָת אֶצְבְּעֵהּ בִּדְמָא וְיַדִי מִן דְמָא שְׁבַע זִמְנִין קֳדָם יְיָ קֳדָם פָּרוּכְתָּא דְקוּדְשָׁא
וְיִתֵּן כַּהֲנָא מִן דְמָא עַל קַרְנַת מַדְבַּח קְטוֹרֶת בּוּסְמַיָא קֳדָם יְיָ דִי בְּמַשְׁכַּן זִמְנָא וְיָת כָּל דְמָא דְתוֹרָא יֵישׁוּד לִיסוֹדָא דְּמַדְבְּחָא דַעֲלָתָא דִי בִתְרַע מַשְׁכַּן זִ מְנָא
וְיָת כָּל תְּרַב תּוֹרָא דְחַטָאתָא יַפְרֵשׁ מִנֵהּ יָת תַּרְבָּא דְחָפֵי עַל גַוָא וְיָת כָּל תַּרְבָּא דִי עַל גַוָא
וְיָת תַּרְתֵּין כָּלְיָן וְיָת תַּרְבָּא דִי עֲלֵיהֶן דִי עַל גִסְסַיָא וְיָת חִצְרָא דְעַל כַּבְדָא עַל כָּלְיֵתָא יֶעְדִנַהּ
כְּמָא דִי מִתַּפְרֵשׁ מִתּוֹר נִכְסַּת קוּדְשַׁיָא וְיַסֵקִנוּן כַּהֲנָא עַל מַדְבְּחָא דַעֲלָתָא
וְיָת מְשַׁךְ תּוֹרָא וְיָת כָּל בִּסְרֵהּ עַל רֵישֵׁיהּ וְעַל כְּרָעוֹהִי וְגַוֵהּ וְאוּכְלֵהּ
וְיַפֵּק יָת כָּל תּוֹרָא מִבָּרָא לְמַשְׁרִיתָא לַאֲתַר דְכֵי לַאֲתַר בֵּית מֵישַׁד קִטְמָא וְיוֹקִיד יָתֵהּ עַל אָעַיָא בְּאֶשָׁתָא עַל אֲתַר בֵּית מֵישַׁד קִטְמָא יִתּוֹקָד
וְאִם כָּל כְּנִשְׁתָּא דְיִשְׂרָאֵל יִשְׁתְּלוּן וִיהֵי מְכַסָא פִּתְגָמָא מֵעֵינֵי קְהָלָא וְיַעְבְדוּן חָד מִכָּל פִּקוּדַיָא דַיְיָ דִי לָא כָשְׁרִין לְאִתְעֲבָדָא וִיחוֹבוּן
וְתִתְיְדַע חוֹבְתָא דִי חָבוּ עֲלַהּ וִיקַרְבוּן קְהָלָא תּוֹר בַּר תּוֹרֵי לְחַטָּאתָא וְיַיְתוּן יָתֵהּ לָקֳדָם מַשְׁכַּן זִמְנָא
וְיִסְמְכוּן סָבֵי כְנִשְׁתָּא יָת יְדֵיהוֹן עַל רֵישׁ תּוֹרָא קֳדָם יְיָ וְיִכּוֹס יָת תּוֹרָא קֳדָם יְיָ
וְיָעֵל כַּהֲנָא רַבָּא מִדְמָא דְתוֹרָא לְמַשְׁכַּן זִמְנָא
וְיִטְבּוֹל כַּהֲנָא אִצְבְּעֵהּ מִן דְמָא וְיַדִי שְׁבַע זִמְנִין קֳדָם יְיָ יָת קֳדָם פָּרֻכְתָּא
וּמִן דְמָא יִתֵּן עַל קַרְנַת מַדְבְּחָא דִי קֳדָם יְיָ דִי בְּמַשְׁכַּן זִמְנָא וְיָת כָּל דְמָא יֵשׁוֹד לִיסוֹדָא דְמַדְבְּחָא דַעֲלָתָא דִי בִתְרַע מַשְׁכַּן זִמְנָא
וְיָת כָּל תַּרְבֵּהּ יַפְרֵשׁ מִנֵהּ וְיַסֵק לְמַדְבְּחָא
וְיַעְבֵּד לְתוֹרָא כְּמָא דִי עֲבַד לְתוֹרָא דְחַטָאתָא כֵּן יַעְבֵּד לֵהּ וִיכַפֵּר עֲלֵיהוֹן כַּהֲנָא וְיִשְׁתְּבֵק לְהוֹן
וְיַפֵּק יָת תּוֹרָא לְמִבָּרָא לְמַשְׁרִיתָא וְאוֹקִיד יָתֵהּ כְּמָא דִי אוֹקִיד יָת תּוֹרָא קַדְמָאָה חַטַאת קְהָלָא הוּא
אִם רַבָּא יֵחוֹב וְיַעְבֵּד חַד מִכָּל פִּקוּדַיָא דַיְיָ אֱלָהֵיהּ דְלָא כַשְׁרִין לְאִתְעֲבָדָא בְּשָׁלוּ וְיֵחוֹב
אוֹ אִתְיְדַע לֵהּ חוֹבְתֵהּ דִי חַב בַּהּ וְיַיְתִי יָת קֻרְבָּנֵהּ צְפִיר בַּר עִזִין דְכַר שְׁלִים
וְיִסְמוֹךְ יְדֵהּ עַל רֵישָׁא דִצְפִירָא וְיִכּוֹס יָתֵהּ בַּאֲתַר דְיִכּוֹס יָת עֲלָתָא לָקֳדָם יְיָ חַטָאתָא הוּא
וְיִסַב כַּהֲנָא מִדְמָא דְחַטָּאתָא בְּאֶצְבְּעֵהּ וְיִתֵּן עַל קַרְנַת מַדְבְּחָא דַעֲלָתָא וְיָת דְמֵהּ יֵשׁוֹד לִיסוֹדָא לְמַדְבְּחָא דַעֲלָתָא
וְיָת כָּל תַּרְבֵּה יַסֵק לְמַדְבְּחָא כִּתְרַב נִכְסַת קוּדְשַׁיָּא וִיכַפֵּר עֲלוֹהִי כַּהֲנָא מֵחוֹבְתֵהּ וְיִשְׁתְּבֵק לֵהּ
וְאִם אֱנַשׁ חַד יֵחוֹב בְּשָׁלוּ מֵעַמָא דְאַרְעָא בְּמֶעְבְּדַהּ חַד מִפִּקוּדַיָא דַיְיָ דְלָא כַשְׁרִין לְאִתְעְבָדָא וְיֵחוֹב
אוֹ אִתְיְדַע לֵהּ חוֹבְתֵהּ דִי חָב וְיַיְתִי קֻרְבָּנֵהּ צְפִירַת עִזִין שְׁלֶמְתָּא נְקוּבְתָא עַל חוֹבְתֵהּ דִי חָב
וְיִסְמוֹךְ יָת יְדֵהּ עַל רֵישׁ חַטָאתָא וְיִכּוֹס יָת חַטָאתָא בַּאֲתַר דַעֲלָתָא
וְיִסַב כַּהֲנָא מִדְמַהּ בְּאֶצְבְּעֵהּ וְיִתֵּן עַל קַרְנַת מַדְבְּחָא דַעֲלָתָא וְיָת כָּל דְמַהּ יֵשׁוֹד לִיסוֹדָא דְמַדְבְּחָא
וְיָת כָּל תַּרְבַּהּ יֶעְדֵי כְּמָא דִי מִתַּעְדָא תְרַב מֵעַל נִכְסַת קוּדְשַׁיָא וְיַסֵק כַּהֲנָא לְמַדְבְּחָא לְאִתְקַבָּלָא בְּרַעֲוָא קֳדָם יְיָ וִיכַפֵּר עֲלוֹהִי כַּהֲנָא וְיִשְׁתְּבֵק לֵהּ
וְאִם אִימַר יַיְתִי קֻרְבָּנֵהּ לְחַטָאתָא נֻקְבָא שְׁלֶמְתָּא יַיְתִנַהּ
וְיִסְמוֹךְ יַת יְדֵהּ עַל רֵישׁ חַטָאתָא וְיִכּוֹס יָתַהּ לְחַטָאתָא בְּאַתְרָא דִי יִכּוֹס יַת עֲלָתָא
וְיִסַב כַּהֲנָא מִדְמָא דְחַטָאתָא בְּאֶצְבְּעֵהּ וְיִתֵּן עַל קַרְנַת מַדְבְּחָא דַעֲלָתָא וְיָת כָּל דְמַהּ יֵשׁוֹד לִיסוֹדָא דְ מַ דְבְּחָא
וְיָת כָּל תַּרְבַּהּ יֶעְדֵי כְּמָא דִי מִתַּעְדָא תְרַב אִמְרָא מִנִכְסַת קוּדְשַׁיָא וְיַסֵק כַּהֲנָא יַתְהוֹן לְמַדְבְּחָא עַל קֻרְבָּנַיָא דַיְיָ וִיכַפֵּר עֲלוֹהִי כַהֲנָא עַל חוֹבְתֵהּ דִי חָב וְיִשְׁתְּבֵק לֵהּ
