‘וּמַלִיל יְיָ יָת כָּל פִּתְגָמַיָא הָאִלֵין לְמֵימָר:
“אֲנָא יְיָ אֱלָהָךְ, דִי אַפֵּיקְתָּךְ מֵאַרְעָא דְמִצְרַיִם מִבֵּית _עַבְדוּתָא_.
לָא יְהֵי לָךְ אֱלָהּ אָחֳרָן _בַּר מִנִי_.
לָא תַעְבֵּד לָךְ צֶלֶם וְכָל דְמוּת דִי בִשְׁמַיָא מִלְעֵלָא וְדִי בְאַרְעָא מִלְרָע, וְדִי בְמַיָא מִלְרַע לְאַרְעָא.
לָא תִסְגֻד לְהוֹן וְלָא תִפְלְחִנוּן, אֲרֵי אֲנָא יְיָ אֱלָהָךְ, אֵל קַנָא, מַסְעַר חוֹבֵי אֲבָהָן עַל בְּנִין +מָרָדִין+, עַל דַר תְּלִיתַי, וְעַל דַר רְבִיעַי לְשָׂנְאָי, +כַּד מְשַׁלְמִין בְּנַיָא לְמֶחֱטֵי בָּתַר אֲבָהָתְהוֹן,+
וְעָבֵד טִּיבוּ לְאַלְפֵי +דָרִין+, לְרַחֲמַי וּלְנָטְרֵי פִקוֹדָי.
לָא _תֵימֵי_ יָת שְׁמָא דַיְיָ אֱלָהָךְ לְמַגָנָא, אֲרֵי לָא יִזַכֵּי יְיָ יָת _דִיֵימֵי_ בִשְׁמֵהּ _לְשִׁקְרָא_
הֲוֵי דְכִיר יָת יוֹמָא דְשַׁבְּתָא לְקַדָשׁוּתֵהּ.
שִׁתָּא יוֹמִין תִּפְלָח וְתַעְבֵּד כָּל עִבִדְתָּךְ,
וְיוֹמָא שְׁבִיעָאָה - שַׁבְּתָא _קֳדָם_ יְיָ אֱלָהָךְ; לָא תַעְבֵּד כָּל עִבִדְתָּא, אַתְּ וּבְרָךְ וּבְרַתָּךְ, עַבְדָךְ וְאַמְתָךְ וּבְעִירָךְ, וְגִיוֹרָךְ דִי בְקִרְוָךְ.
אֲרֵי שִׁתָּא יוֹמִין עֲבַד יְיָ יָת שְׁמַיָא וְיָת אַרְעָא, יָת יַמָא וְיָת כָּל דִי בְהוֹן, וְנָח בְּיוֹמָא שְׁבִיעָאָה. עַל כֵּן, בָּרִיךְ יְיָ יָת יוֹמָא דְשַׁבְּתָא וְקַדְשֵׁהּ.
יַקַר יָת אָבוּךְ וְיָת אִמָךְ, בְּדִיל דְיוֹרְכוּן יוֹמָיךְ, עַל אַרְעָא דַיְיָ אֱלָהָךְ יָהֵב לָךְ.
לָא תִקְטוֹל +נְפָשׁ+. לָא תְגוּף. לָא תִגְנוּב. לָא _תַסְהֵד_ בְּחַבְרָךְ סַהֲדוּתָא דְשִׁקְרָא.
לָא תַחְמֵד בֵּית חַבְרָךְ. לָא תַחְמֵד אִתַּת חַבְרָךְ, וְעַבְדֵהּ וְאַמְתֵהּ וְתוֹרֵהּ וַחֲמָרֵהּ, וְכֹל דִי לְחַבְרָךְ”.
וְכָל עַמָא חָזָן יָת קָלַיָא וְיָת בָּעוֹרַיָא, וְיָת קַל שׁוֹפָרָא, וְיָת טוּרָא תָּנֵן. וַחֲזָא עַמָא וְזָעוּ, וְקָמוּ מֵרָחִיק.
וַאֲמָרוּ לְמשֶׁה: 'מַלֵל אַתְּ עִמָנָא _וּנְקַבֵּל_, וְלָא יִתְמַלֵל עִמָנָא _מִן קֳדָם_ יְיָ דִילְּמָא נְמוּת'.
וַאֲמַר משֶׁה לְעַמָא: 'לָא תִדְחֲלוּן! אֲרֵי בְּדִיל לְנַסָאָה יָתְכוֹן _אִתְגְלִי לְכוֹן יְקָרָא דַ_יְיָ, וּבְדִיל דִתְהֵי דַחַלְתֵּהּ עַל אַפֵּיכוֹן בְּדִיל דְלָא תְחוּבוּן'.
וְקָם עַמָא מֵרָחִיק, וּמשֶׁה _קָרֵב לְצַד_ אַמִיטְּתָא, דִתַמָן +יְקָרָא+ דַיְיָ.
וַאֲמַר יְיָ לְמשֶׁה: “כִּדְנַן תֵּימַר לִבְנֵי יִשְׂרָאֵל: ‘אַתּוּן חֲזִיתּוּן אֲרֵי מִן שְׁמַיָא מַלֵילִית עִמְכוֹן.
לָא תַעְבְּדוּן _קֳדָמָי_ _דַחֲלָן_ דִכְסַף, _וְדַחֲלָן_ דִדְהַב לָא תַעְבְּדוּן לְכוֹן.
מַדְבַּח אַדְמְתָא תַּעְבֵּד _קֳדָמַי_, וּתְהֵי דַבָּח עֲלוֹהִי יָת עֲלָוָתָךְ וְיָת נְכַס קוּדְשָׁךְ, _מִן_ עָנָךְ _וּמִן_ תּוֹרָךְ, בְּכָל אַתְרָא דִי _אַשְׁרֵי יָת שְׁכִנְתִּי_, _לְתַמָן אֶשְׁלַח בִּרְכְּתִי לָךְ_ וֶאֱבָרֵכִנָךְ.
וְאִם מַדְבַּח אַבְנִין תַּעְבֵּד _קֳדָמַי_ - לָא תִבְנֵי יָתְהוֹן פְּסִילָן, [דִּלְמָא|לָא] תְרִים חַרְבָּךְ עֲלַהּ וּתְחַלְלִנַהּ.
וְלָא תִסַק בְּדַרְגִין עַל מַדְבְּחִי, דְלָא תִתְגְלֵי עֶרְיְתָךְ עֲלוֹהִי”.
