וּמַלִיל יְיָ עִם מֹשֶׁה לְמֵימָר
מַלֵל עִם בְּנֵי יִשְׂרָאֵל וְתֵימַר לְהוֹן גְבַר אֲרֵי יַפְרֵשׁ נְדַר בְּפֻרְסַן נַפְשָׁתָא קֳדָם יְיָ
וִיהֵי פֻרְסָנֵהּ דְכוּרָא מִבַּר עֶסְרִין שְׁנִין וְעַד בַּר שִׁתִּין שְׁנִין וִיהֵי פֻרְסָנֵהּ חַמְשִׁין סִלְעִין דִכְסַף בְּסִלְעֵי קוּדְשָׁא
וְאִם נְקוּבְתָא הִיא וִיהֵי פֻרְסָנֵהּ תְּלָתִין סִלְעִין
וְאִם מִבַּר חֲמֵשׁ שְׁנִין וְעַד בַּר עֶסְרִין שְׁנִין וִיהֵי פֻרְסָנֵהּ דְכוּרָא עֶסְרִין סִלְעִין וְלִנְקוּבְתָא עֲסַר סִלְעִין
וְאִם מִבַּר יַרְחָא וְעַד בַּר חֲמֵשׁ שְׁנִין וִיהֵי פֻרְסָנֵהּ דְכוּרָא חֲמֵשׁ סִלְעִין דִכְסָף וְלִנְקוּבְתָא פֻרְסָנֵהּ תְּלַת סִלְעִין דִכְסָף
וְאִם מִבַּר שִׁתִּין שְׁנִין וּלְעֵלָא אִם דְכוּרָא וִיהֵי פֻרְסָנֵהּ חֲמֵשׁ עֶסְרֵי סִלְעִין וְלִנְקוּבְתָא עֲסַר סִלְעִין
וְאִם מִסְכֵּן הוּא מִפֻּרְסָנֵהּ וִיקִימִנֵהּ קֳדָם כַּהֲנָא וְיִפְרוֹס יָתֵהּ כַּהֲנָא עַל מֵימַר דִי תַדְבֵּק יְדָא דְנוֹדֵרָא יִפְרְסִנֵהּ כַּהֲנָא
וְאִם בְּעִירָא דִי יְקָרְבוּן מִנַהּ קֻרְבָּנָא קֳדָם יְיָ כָּל דִי יִתֵּן מִנֵהּ קֳדָם יְיָ יְהֵי קוּדְשָׁא
לָא יְחַלְפִנֵהּ וְלָא יְעִבַּר יָתֵהּ טַּב בְּבִישׁ אוֹ בִישׁ בְּטָב וְאִם חַלָפָא יְחַלֵף בְּעִירָא בִּבְעִירָא וִיהֵי הוּא וְחִלוּפֵי יְהֵי קַדִישׁ
וְאִם כָּל בְּעִירָא מְסָאָבָא דִי לָא יְקָרְבוּן מִנַהּ קֻרְבָּנָא קֳדָם יְיָ וִיקִים יָת בְּעִירָא קֳדָם כַּהֲנָא
וְיִפְרוֹס כַּהֲנָא יָתַהּ בֵּין טַב וּבֵין בִּישׁ כְּפֻרְסָנָא דְכַהֲנָא כֵּן יְהֵי
וְאִם מִפְרַק יִפְרְקֻנֵהּ וְיוֹסֵף חֻמְשֵׁה עַל פֻּרְסָנֵהּ
וּגְבַר אֲרֵי יַקְדֵשׁ יָת בֵּיתֵהּ קוּדְשָׁא קֳדָם יְיָ וְיִפְרְסִנֵהּ כַּהֲנָא בֵּין טַב וּבֵין בִּישׁ כְּמָא דִי יִפְרוֹס יָתֵהּ כַּהֲנָא כֵּן יְקוּם
וְאִם דְאַקְדֵשׁ יִפְרוֹק יָת בֵּיתֵהּ וְיוֹסֵף חוֹמֶשׁ כְּסַף פֻּרְסָנֵהּ עֲלוֹהִי וִיהֵי לֵהּ
וְאִם מֵחֲקַל אַחֲסַנְתֵּהּ יַקְדִישׁ גְבַר קֳדָם יְיָ וִיהֵי פֻרְסָנֵהּ לְפוּם זַרְעֵהּ בַּר זְרַע כּוֹר שְׂעוֹרִין בְּחַמְשִׁין סִלְעִין דִכְסָף
אִם מִשַּׁתָּא דְיוֹבֵלָא יַקְדֵשׁ חַקְלֵהּ כְּפֻרְסָנֵהּ יְקוּם
וְאִם בָּתַר יוֹבֵלָא יַקְדֵשׁ חַקְלֵהּ וִיחַשֵׁב לֵהּ כַּהֲנָא יָת כַּסְפָּא עַל מֵימַר שְׁנַיָא דְאִשְׁתָּאָרָן עַד שַׁתָּא דְיוֹבֵלָא וְיִתִּמְנַע מִפֻּרְסָנֵהּ
וְאִם מִפְרַק יִפְרוֹק יָת חַקְלָא דְאַקְדֵשׁ יָתֵהּ וְיוֹסֵף חוֹמֶשׁ כְּסַף פֻּרְסָנֵהּ עֲלוֹהִי וִיקוּם לֵהּ
וְאִם לָא יִפְרוֹק יָת חַקְלָא וְאִם זַבִּין יָת חַקְלָא לִגְבַר אָחֳרָן לָּא יִתְפָּרַק עוֹד
וִיהֵי חַקְלָא בְּמִפְּקֵהּ בְּיוֹבֵלָא קוּדְשָׁא קֳדָם יְיָ כַּחֲקַל חֶרְמָא לְכַהֲנָא תְּהֵי אַחֲסַנְתֵּהּ
וְאִם יָת חֲקַל זְבִינוֹהִי דִי לָא מֵחֲקַל אַחֲסַנְתֵּהּ יַקְדֵשׁ קֳדָם יְיָ
וִיחַשֶׁב לֵהּ כַּהֲנָא יָת מִנְיַן פֻּרְסָנֵהּ עַד שַׁתָּא דְיוֹבֵלָא וְיִתֵּן יָת פֻרְסָנֵהּ בְּיוֹמָא הַהוּא קוּדְשָׁא קֳדָם יְיָ
בְּשַׁתָּא דְיוֹבֵלָא יְתוּב חַקְלָא לִדְזַבְנֵהּ מִנֵהּ לִדְדִילֵהּ אַחֲסָנַת אַרְעָא
וְכָל פֻּרְסָנֵּה יְהֵי בְּסִלְעֵי קוּדְשָׁא עֶסְרִין מָעִין יְהֵי סִלְעָא
בְּרַם בּוּכְרָא דִי יִתְבַּכַּר קֳדָם יְיָ בִּבְעִירָא לָא יַקְדֵשׁ גְבַר יָתֵהּ אִם תּוֹר אִם אִמַר דַיְיָ הוּא
וְאִם בִּבְעִירָא מְסָאָבָא וְיִפְרוֹק בְּפֻרְסָנֵהּ וְיוֹסֵף חֻמְשֵׁהּ עֲלוֹהִי וְאִם לָא יִתְפָּרַק וְיִזְדַבַּן בְּפֻרְסָנֵהּ
בְּרַם כָּל חֶרְמָא דִי יַחְרֵם גְבַר קֳדָם יְיָ מִכָּל דִי לֵהּ מֵאֲנָשָׁא וּבְעִירָא וּמֵחֲקַל אַחֲסַנְתֵּהּ לָא יִזְדַבַּן וְלָא יִתְפָּרָק כָּל חֶרְמָא קֹדֶשׁ קוּדְשִׁין הוּא קֳדָם יְיָ
כָּל חֶרְמָא דִי יִתַּחְרַם מִן אֱנָשָׁא לָא יִתְפָּרָק אִתְקְטָלָא יִתְקְטֵל
וְכָל מַעְשְׂרָא דְאַרְעָא מִזַרְעָא דְאַרְעָא מִפֵּירֵי אִילָנָא דַיְיָ הוּא קוּדְשָׁא קֳדָם יְיָ
וְאִם מִפְרַק יִפְרוֹק גְבַר מִמַעַשְׂרֵהּ חֻמְשֵׁהּ יוֹסֵף עֲלוֹהִי
וְכָל מַעֲשַׂר תּוֹרִין וְעָן כֹּל דְיֵעִבַּר תְּחוֹת חֻטְרָא עֲשִׂירָאָה יְהֵי קַדִישׁ קֳדָם יְיָ
לָא יְבַקַר בֵּין טַב לְבִישׁ וְלָא יְחַלְפִנֵהּ וְאִם חַלָפָא יְחַלְפִנֵהּ וִיהֵי הוּא וְחִלוּפֵהּ יְהֵי קוּדְשָׁא לָא יִתְפָּרָק
