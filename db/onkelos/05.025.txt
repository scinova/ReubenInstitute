אֲרֵי יְהֵי דִין בֵּין גֻבְרַיָא וְיִתְקָרְבוּן לְדִינָא וִידוּנֻנוּן וִיזַכּוּן יָת זַכָּאָה וִיחַיְבוּן יָת חַיָבָא
וִיהֵי אִם בַּר חַיָבָא לְאַלְקָאָה חַיָבָא וְיִרְמִנֵהּ דַיָנָא וְיַלְקִנֵהּ קֳדָמוֹהִי כְּמִסַת חוֹבְתֵהּ בְּמִנְיָן
אַרְבְּעִין יַלְקִנֵהּ לָא יוֹסֵף דִילְמָא יוֹסֵף לְאַלְקָיוּתֵהּ עַל אִלֵין מָחָא רַבָּא וְיֵקַל אֲחוּךְ לְעֵינָיךְ
לָא תֵחוֹד פּוּם תּוֹרָא בִּדְיָשֵׁהּ
אֲרֵי יֵתְבוּן אַחִין כַּחֲדָא וִימוּת חַד מִנְהוֹן וּבַר לֵית לֵהּ לָא תְהֵי אִתַּת מֵתָנָא לְבָרָא לִגְבַר אוֹחֲרָן יְבָמַהּ יֵעוּל עֲלַהּ וְיִסְבַהּ לֵהּ לְאִנְתּוּ וְיִבְּמַהּ
וִיהֵי בּוּכְרָא דִי תְלִיד יְקוּם עַל שְׁמָא דַאֲחוּהִי מֵתָנָא וְלָא יִתִּמְחֵי שְׁמֵהּ מִיִשְׂרָאֵל
וְאִם לָא יִצְבֵי גַבְרָא לְמִסַב יָת יְבִמְתֵּהּ וְתִסַק יְבִמְתֵהּ לִתְרַע בֵּית דִינָא לָקֳדָם סָבַיָא וְתֵימַר לָא צָבֵי יְבָמִי לַאֲקָמָא לַאֲחוּהִי שְׁמָא בְּיִשְׂרָאֵל לָא אָבֵי לְיַבָּמוּתִי
וִיקְרוּן לֵהּ סָבֵי קַרְתֵּהּ וִימַלְלוּן עִמֵהּ וִיקוּם וְיֵימַר לָא רְעֵנָא לְמִסְבַהּ
וְתִתְקְרַב יְבִמְתֵּהּ לְוָתֵה לָקֳדָם סָבַיָא וְתַשְׁרֵי סֵינֵהּ מֵעַל רַגְלֵהּ וְתֵרוֹק בְּאַנְפּוֹהִי וְתָתֵב וְתֵימַר כְּדֵין יִתְעֲבֵד לִגְבַר דִי לָא יִבְנֵי יָת בֵּיתָא דַאֲחוּהִי
וְיִתְקְרֵי שְׁמֵהּ בְּיִשְׂרָאֵל בֵּית שְׁרִי סֵינָא
אֲרֵי יִנְצוּן גֻבְרִין כַּחֲדָא גְבַר וַאֲחוּהִי וְתִתְקְרֵב אִתַּת חַד לְשֵׁזָבָא יָת בַּעֲלַהּ מִיַד מָחוּהִי וְתוֹשֵׁט יְדַהּ וְתִתְקֵף בְּבֵית בַּהֲתְּתֵהּ
וּתְקוּץ יָת יְדַהּ לָא תְחוּס עֵינָךְ
לָא יְהֵי לָךְ בְּכִיסָךְ מַתְקַל וּמַתְקָל רַבְּתָא וּזְעֶרְתָּא
לָא יְהֵי לָךְ בְּבֵיתָךְ מְכִילָא וּמְכִילָא רַבְּתָא וּזְעֶרְתָּא
מַתְקִלִין שָׁלְמִין דִקְשׁוֹט יְהוֹן לָךְ מְכִילָן שָׁלְמִין דִקְשׁוֹט יְהוֹן לָךְ בְּדִיל דְיוֹרְכוּן יוֹמָיךְ עַל אַרְעָא דַיְיָ אֱלָהָךְ יָהֵב לָךְ
אֲרֵי מְרָחָק קֳדָם יְיָ אֱלהָךְ כָּל עָבֵד אִלֵין כֹּל עָבֵד שְׁקַר
הֲוֵי דְכִיר יָת דִי עֲבַד לָךְ עֲמָלֵק בְּאָרְחָא בְּמִפָּקְכוֹן מִמִצְרָיִם
דִי עָרְעָךְ בְּאָרְחָא וְקָטֵל בָּךְ כָּל דַהֲווֹ מִתְאַחֲרִין בַּתְרָךְ וְאַתְּ מְשַׁלְהֵי וְלָאֵי וְלָא דָחֵיל מִן קֳדָם יְיָ
וִיהֵי כַּד יְנִיחַ יְיָ אֱלָהָךְ לָךְ מִכָּל בַּעֲלֵי דְבָבָךְ מִסְחוֹר סְחוֹר בְּאַרְעָא דִי יְיָ אֱלָהָךְ יָהֵב לָךְ אַחֲסָנָא לְמֵירְתַהּ תִּמְחֵי יָת דוּכְרָנָא דַעֲמָלֵק מִתְּחוֹת שְׁמַיָא לָא תִּתִּנְשֵׁי
