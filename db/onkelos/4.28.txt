וּמַלֵיל יְיָ עִם משֶׁה לְמֵימָר
פַקֵיד יָת בְּנֵי יִשְׂרָאֵל וְתֵימַר לְהוֹן יָת קוּרְבָּנִי לְחֵם סִדוּר לְקוּרְבָּנַי לְאִתְקַבָּלָא בְּרַעֲוָא תִּטְרוּן לְקָרָבָא קֳדָמַי בְּזִמְנֵיהּ
וְתֵימַר לְהוֹן דֵין קוּרְבָּנָא דִי תְקַרְבוּן קֳדָם יְיָ אִימְּרִין בְּנֵי שְׁנָא שַׁלְמִין תְּרֵין לְּיוֹמָא עֲלָתָא תְּדִירָא
יָת אִמְרָא חָד תַּעֲבֵד בְּצַפְרָא וְיָת אִמְרָא תִנְיָנָא תַּעֲבֵד בֵּין שִׁמְשַׁיָא
וְחָד מִן עַשְׂרָא בִּתְלַת סָאִין סוּלְתָּא לְמִנְחָתָא דְפִילָא בִּמְשַׁח כְּתִישָׁא רַבְעוּת הִינָא
עֲלָתָא תְּדִירָא דְאִתְעֲבִידָא בְּטוּרָא דְסִינַי לְאִתְקַבָּלָא בְרַעֲוָא קוּרְבָּנָא קֳדָם יְיָ
וְנִסְכֵּיהּ רַבְעוּת הִינָא לְאִמְרָא חָד בְּקוּדְשָׁא יִתְנַסַךְ נִסוּךְ דַחֲמַר עַתִּיק קֳדָם יְיָ
וְיָת אִמְרָא תִנְיָנָא תַּעֲבֵיד בֵּין שִׁמְשַׁיָא כְּמִנְחַת צַפְרָא וּכְנִסְכֵּיהּ תַּעְבֵיד קוּרְבַּן דְמִתְקַבֵּל בְּרַעֲוָא קֳדָם יְיָ
וּבְיוֹמָא דְשַׁבַּתָּא תְּרֵין אִימְרִין בְּנֵי שְׁנָא שַׁלְמִין וּתְרֵין עִשְׂרוֹנִין סוּלְתָּא דְמִנְחָתָא דְפִילָא בִּמְשַׁח וְנִסְכֵּיהּ
עֲלַת שַׁבְּתָא דְתִּתְעֲבֵיד בְּשַׁבַּתָּא עַל עֲלָתָא תְּדִירָא וְנִסְכַּהּ
וּבְרֵישֵׁי יַרְחֵיכוֹן תְּקַרְבוּן עֲלָתָא קֳדָם יְיָ תּוֹרִין בְּנֵי תּוֹרֵי תְּרֵין וּדְכַר חָד אִימְרִין בְּנֵי שְׁנָא שַׁבְעָא שַׁלְמִין
וּתְלָתָא עִשְׂרוֹנִין סוּלְתָּא מִנְחָתָא דְפִילָא בִמְשַׁח לְתוֹרָא חָד וּתְרֵין עִשְׂרוֹנִין סֻלְתָּא מִנְחָתָא דְפִילָא בִמְשַׁח לְדִכְרָא חָד
וְעִשְׂרוֹנָא עִשְׂרוֹנָא סֻלְתָּא מִנְחָתָא דְפִילָא בִמְשַׁח לְאִמְרָא חָד עֲלָתָא לְאִתְקַבָּלָא בְּרַעֲוָא קוּרְבָּנָא קֳדָם יְיָ
וְנִסְכֵּיהוֹן פַּלְגוּת הִינָא יְהֵי לְתוֹרָא וְתַלְתּוּת הִינָא לְדִכְרָא וְרַבְעוּת הִינָא לְאִמְרָא חַמְרָא דָא עֲלַת רֵישׁ יַרְחָא בְּאִתְחַדָתוּתֵיהּ כֵּן לְכָל רֵישֵׁי יַרְחֵי שַׁתָּא
וּצְפִיר בַּר עִזִין חָד לְחַטָאתָא קֳדָם יְיָ עַל עֲלָתָא תְּדִירָא יִתְעֲבֵד וְנִסְכֵּיהּ
וּבְיַרְחָא קַדְמָאָה בְּאַרְבַּעַת עַשְׂרָא יוֹמָא לְיַרְחָא פִּסְחָא קֳדָם יְיָ
וּבְחַמְשַׁת עַשְׂרָא יוֹמָא לְיַרְחָא הָדֵין חַגָא שַׁבְעָא יוֹמִין פַּטִיר יִתְאָכֵל
בְּיוֹמָא קַדְמָאָה מְעָרַע קַדִישׁ כָּל עֲבִידַת פּוּלְחַן לָא תַעְבְדוּן
וּתְקַרְבוּן קוּרְבָּנָא עֲלָתָא קֳדָם יְיָ תּוֹרִין בְּנֵי תוֹרֵי תְּרֵין וּדְכַר חָד וְשַׁבְעָא אִימְרִין בְּנֵי שְׁנָא שַׁלְמִין יְהוֹן לְכוֹן
וּמִנְחַתְהוֹן סוּלְתָּא דְפִילָא בִּמְשַׁח תְּלָתָא עִשְׂרוֹנִין לְתוֹרָא וּתְרֵין עִשְׂרוֹנִין לְדִכְרָא תַּעְבְּדוּן
עִשְׂרוֹנָא עִשְׂרוֹנָא תַּעֲבֵיד לְאִמְרָא חָד כֵּן לְשַׁבְעָא אִימְרִין
וּצְפִירָא דְחַטָאתָא חָד לְכַפָּרָא עֲלֵיכוֹן
בַּר מֵעֲלַת צַפְרָא דְהִּיא עֲלַת תְּדִירָא תַּעְבְּדוּן יָת אִלֵין
כְּאִלֵין תַּעַבְדוּן לְיוֹמָא שַׁבְעָא יוֹמִין לְחֵם קוּרְבַּן דְמִתְקַבֵּל בְּרַעֲוָא קֳדָם יְיָ עַל עֲלָתָא תְדִירָא יִתְעֲבֵיד וִנִסִכֵּיהּ
וּבְיוֹמָא שְׁבִיעָאָה מְעָרַע קַדִישׁ יְהֵי לְכוֹן כָּל עִיבִידַת פּוּלְחַן לָא תַעַבְדוּן
וּבְיוֹמָא דְבִכּוּרַיָא בְּקָרוֹבֵיכוֹן מִנְחֲתָא חֲדַתָּא קֳדָם יְיָ בְּעַצְרָתֵיכוֹן מְעָרַע קַדִישׁ יְהֵי לְכוֹן כָּל עִבִידַת פּוּלְחַן לָא תַעַבְדוּן
וּתְקַרְבוּן עֲלָתָא לְאִתְקַבָּלָא בְרַעֲוָא קֳדָם יְיָ תּוֹרִין בְּנֵי תוֹרֵי תְּרֵין וּדְכַר חָד שַׁבְעָא אִימְרִין בְּנֵי שְׁנָא
וּמִנְחַתְהוֹן סֻלְתָּא דְפִילָא בִמְשָׁח תְּלָתָא עִשְׂרוֹנִין לְתוֹרָא חָד תְּרֵין עִשְׂרוֹנִין לְדִכְרָא חָד
עִשְׂרוֹנָא עִשְׂרוֹנָא לְאִמְרָא חָד לְשַׁבְעָא אִימְרִין
צְפִיר בַּר עִזִין חָד לְכַפָּרָא עֲלֵיכוֹן
בַּר מֵעֲלַת תְּדִירָא וּמִנְחָתֵהּ תַּעְבְּדוּן שַׁלְמִין יְהוֹן לְכוֹן וְנִסְכֵּיהוֹן
