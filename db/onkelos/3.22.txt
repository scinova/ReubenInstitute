וּמַלִיל יְיָ עִם מֹשֶׁה לְמֵימָר
מַלֵיל עִם אַהֲרֹן וְעִם בְּנוֹהִי וְיִפָּרְשׁוּן מִקוּדְשַׁיָא דִבְנֵי יִשְׂרָאֵל וְלָא יְחַלוּן יָת שְׁמָא דְקוּדְשִׁי דִי אִנוּן מְקַדְשִׁין קֳדָמַי אֲנָא יְיָ
אֱמַר לְהוֹן לְדָרֵיכוֹן כָּל גְבַר דִי יִקְרַב מִכָּל בְּנֵיכוֹן לְקוּדְשַׁיָא דִי יְקַדְשׁוּן בְּנֵי יִשְׂרָאֵל קֳדָם יְיָ וְסֹאבְתֵיהּ עֲלוֹהִי וְיִשְׁתֵּיצֵי אֲנָשָׁא הַהִיא מִן קֳדָמַי אֲנָא יְיָ
גְבַר גְבַר מִזַרְעָא דְאַהֲרֹן וְהוּא סְגִיר אוֹ דָאִיב בְּקוּדְשַׁיָא לָא יֵיכוּל עַד דִי יִדְכֵּי וּדְיִקְרַב בְּכָל טְמֵא נַפְשָׁא אוֹ גְבַר דִי תִפּוֹק מִנֵהּ שִׁכְבַת זַרְעָא
אוֹ גְבַר דִי יִקְרַב בְּכָל רִחֲשָׁא דִי יִסְתָּאַב לֵהּ אוֹ בַאֲנָשָׁא דִי יִסְתָּאַב לֵהּ לְכָל סֹאבְתֵיהּ
אֱנַשׁ דִי יִקְרַב בֵּהּ וִיהֵי מְסָאָב עַד רַמְשָׁא וְלָא יֵיכוּל מִן קוּדְשַׁיָא אֱלָהֵן אַסְחֵי בִשׂרֵהּ בְּמַיָא
וּבְמֵעַל שִׁמְשָׁא וְיִדְכֵּי וּבָתַר כֵּן יֵיכוּל מִן קוּדְשַׁיָא אֲרֵי לַחְמֵהּ הוּא
נְבִילָא וּתְבִירָא לָא יֵיכוּל לְאִסְתָּאָבָא בָהּ אֲנָא יְיָ
וְיִטְרוּן יָת מַטְּרַת מֵימְרִי וְלָא יְקַבְּלוּן עֲלוֹהִי חוֹבָא וִימוּתוּן בֵּהּ אֲרֵי יְחַלֻנֵהּ אֲנָא יְיָ מְקַדִשְׁהוֹן
וְכָל חִלוֹנַי לָא יֵיכוּל קוּדְשָׁא תּוֹתָבָא דְכַהֲנָא וַאֲגִירָא לָא יֵיכוּל קוּדְשָׁא
וְכָהֵן אֲרֵי יִקְנֵי נְפַשׁ קִנְיַן כַּסְפֵּהּ הוּא יֵיכוּל בֵּהּ וִילִיד בֵּיתֵהּ אִנוּן יֵיכְלוּן בְּלַחְמֵהּ
וּבַת כָּהֵן אֲרֵי תְהִי לִגְבַר חִלוֹנָי הִיא בְּאַפְרָשׁוּת קוּדְשַׁיָא לָא תֵיכוּל
וּבַת כָּהֵן אֲרֵי תְהֵי אַרְמְלָא וּמְתָרְכָא וּבַר לֵית לַהּ וּתְתוּב לְבֵית אֲבוּהָא כְּרַבְיוּתָהָא מִלַחְמָא דַאֲבוּהָא תֵּיכוּל וְכָל חִלוֹנַי לָא יֵיכוּל בֵּהּ
וּגְבַר אֲרֵי יֵיכוּל קוּדְשָׁא בְּשָׁלוּ וְיוֹסֵף חוּמְשֵׁהּ עֲלוֹהִי וְיִתֵּן לְכַהֲנָא יָת קוּדְשָׁא
וְלָא יְחַלוּן יָת קוּדְשַׁיָא דִבְנֵי יִשְׂרָאֵל יָת דִי יַפְרְשׁוּן קֳדָם יְיָ
וִיקַבְּלוּן עֲלֵיהוֹן עֲוָיָן וְחוֹבִין בְּמֵיכָלְהוֹן בְּסוֹאֲבָא יָת קוּדְשֵׁיהוֹן אֲרֵי אֲנָא יְיָ מְקַדִשְׁהוֹן
וּמַלִיל יְיָ עִם מֹשֶׁה לְמֵימָר
מַלֵל עִם אַהֲרֹן וְעִם בְּנוֹהִי וְעִם כָּל בְּנֵי יִשְׂרָאֵל וְתֵימַר לְהוֹן גְבַר גְבַר מִבֵּית יִשְׂרָאֵל וּמִן גִיוֹרָא בְּיִשְׂרָאֵל דִי יְקָרֵב קֻרְבָּנֵהּ לְכָל נִדְרֵיהוֹן וּלְכָל נִדְבָתְהוֹן דִי יְקָרְבוּן קֳדָם יְיָ לַעֲלָתָא
לְרַעֲוָא לְכוֹן שְׁלִים דְכוּרָא בְּתוֹרֵי בְּאִמְרַיָא וּבְעִזַיָא
כֹּל דִי בֵהּ מוּמָא לָא תְקָרְבוּן אֲרֵי לָא לְרַעֲוָא יְהֵי לְכוֹן
וּגְבַר אֲרֵי יְקָרֵב נִכְסַת קוּדְשַׁיָא קֳדָם יְיָ לְפָרָשָׁא נִדְרָא אוֹ לִנְדַבְתָּא בְּתוֹרֵי אוֹ בְעָנָא שְׁלִים יְהֵי לְרַעֲוָא כָּל מוּמָא לָא יְהֵי בֵהּ
עַוִיר אוֹ תְבִיר אוֹ פְסִיק אוֹ יַבְּלָן אוֹ גַרְבָן אוֹ חֲזָזָן לָא תְקָרְבוּן אִלֵין קֳדָם יְיָ וְקֻרְבָּנָא לָא תִתְּנוּן מִנְהוֹן עַל מַדְבְּחָא קֳדָם יְיָ
וְתוֹר וְאִמַר יַתִּיר וְחַסִיר נְדַבְתָּא תַּעְבֵּד יָתֵהּ וּלְנִדְרָא לָא יְהֵי לְרַעֲוָא
וְדִי מְרִיס וְדִי רְסִיס וְדִי שְׁלִיף וְדִי גְזִיר לָא תְקָרְבוּן קֳדָם יְיָ וּבְאַרְעֲכוֹן לָא תַעְבְּדוּן
וּמִן בַּר עַמְמִין לָא תְקַרְבוּן יָת קֻרְבַּן אֱלָהֲכוֹן מִכָּל אִלֵין אֲרֵי חִבּוּלְהוֹן בְּהוֹן מוּמָא בְהוֹן לָא לְרַעֲוָא יְהוֹן לְכוֹן
וּמַלִיל יְיָ עִם מֹשֶׁה לְמֵימָר
תּוֹר אוֹ אִמָר אוֹ עִזָא אֲרֵי יִתְיְלִיד וִיהֵי שִׁבְעָא יוֹמִין בָּתַר אִמֵהּ וּמִיוֹמָא תְמִינָאָה וּלְהַלָא יִתִּרְעִי לְקָרָבָא קֻרְבָּנָא קֳדָם יְיָ
וְתוֹרְתָּא אוֹ שֵׂיתָא לֵהּ וְלִבְרֵהּ לָא תִכְּסוּן בְּיוֹמָא חָד
וַאֲרֵי תִּכְּסוּן נִכְסַת תּוֹדְתָא קֳדָם יְיָ לְרַעֲוָא לְכוֹן תִּכְּסֻנֵהּ
בְּיוֹמָא הַהוּא יִתְאֲכֵל לָא תַשְׁאֲרוּן מִנֵהּ עַד צַפְרָא אֲנָא יְיָ
וְתִטְרוּן פִּקוּדַי וְתַעֲבְּדוּן יָתְהוֹן אֲנָא יְיָ
וְלָא תְחַלְלוּן יָת שְׁמָא דְקוּדְשַׁי וְאִתְקַדַשׁ בְּגוֹ בְּנֵי יִשְׂרָאֵל אֲנָא יְיָ מְקַדִשְּׁכוֹן
דְאַפִּיק יָתְכוֹן מֵאַרְעָא דְמִצְרַיִם לְמֶהֱוֵי לְכוֹן לֵאלָהָא אֲנָא יְיָ
