וּדְכִיר יְיָ יָת נֹחַ, וְיָת כָּל חַיְתָא וְיָת כָּל בְּעִירָא, דִּי עִמֵּהּ בְּתֵבוֹתָא. וְאַעְבַּר יְיָ רוּחָא עַל אַרְעָא וְנָחוּ מַיָּא.
וְאִסְתְּכָרוּ _מַבּוּעֵי_ תְהוֹמָא, _וְכַוֵּי_ שְׁמַיָּא. וְאִתְכְּלִי מִטְרָא מִן שְׁמַיָּא.
וְתָבוּ מַיָּא מֵעַל אַרְעָא, אָזְלִין וְתָיְבִין; וַחֲסָרוּ מַיָּא, מִסּוֹף מְאָה וְחַמְשִׁין יוֹמִין.
_וְנָחַת_ תֵּבוֹתָא _בְּיַרְחָא_ שְׁבִיעָאָה, בְּשִׁבְעַת עֲשַׂר יוֹמָא לְיַרְחָא, עַל טוּרֵי _'קַרְדּוּ'_.
וּמַיָּא הֲווֹ _אָזְלִין_ וְ_חָסְרִין_, עַד _יַרְחָא_ עֲשִׂירָאָה; בַּעֲשִׂירָאָה, בְּחַד _לְיַרְחָא_, אִתְחֲזִיאוּ רֵישֵׁי טוּרַיָּא.
וַהֲוָה _מִסּוֹף_ אַרְבְּעִין יוֹמִין, וּפְתַח נֹחַ יָת כַּוַּת תֵּבוֹתָא דִּי עֲבָד,
וְשַׁלַּח יָת עוֹרְבָא, וּנְפַק מִפַּק וְתַיִּב, עַד _דְּיַבִּישׁוּ_ מַיָּא מֵעַל אַרְעָא.
וְשַׁלַּח יָת יוֹנָה _מִלְּוָתֵהּ_, לְמֵחֲזֵי הֲקַלִּיאוּ מַיָּא, מֵעַל אַפֵּי אַרְעָא.
וְלָא אַשְׁכַּחַת יוֹנָה מְנַח לְפַרְסַת רַגְלַהּ, וְתָבַת לְוָתֵהּ לְתֵבוֹתָא, אֲרֵי מַיָּא עַל אַפֵּי כָל אַרְעָא; _וְאוֹשִׁיט_ יְדֵהּ וְנָסְבַהּ, _וְאָעֵיל_ יָתַהּ לְוָתֵהּ לְתֵבוֹתָא.
_וְאוֹרִיךְ_ עוֹד שַׁבְעָא יוֹמִין אָחֳרָנִין, וְאוֹסֵיף שַׁלַּח יָת יוֹנָה מִן תֵּבוֹתָא.
וַאֲתַת לְוָתֵהּ יוֹנָה לְעִדַּן רַמְשָׁא, וְהָא טְּרַף זֵיתָא +תְּבִיר+ _נָחִית_ בְּפוּמַהּ; וִידַע נֹחַ, אֲרֵי קַלִּיאוּ מַיָּא מֵעַל אַרְעָא.
וְ_אוֹרִיךְ_ עוֹד, שַׁבְעָא יוֹמִין אָחֳרָנִין; וְשַׁלַּח יָת יוֹנָה, וְלָא אוֹסֵיפַת לְמִתּוּב לְוָתֵהּ עוֹד.
וַהֲוָה בְּשִׁתְּ מְאָה וְחַד שְׁנִין, בְּקַדְמָאָה, בְּחַד _לְיַרְחָא_, _נְגוּבוּ_ מַיָּא מֵעַל אַרְעָא; וְאַעְדִּי נֹחַ יָת _חוֹפָאָה_ דְּתֵבוֹתָא, וַחֲזָא, וְהָא _נְגוּבוּ_ אַפֵּי אַרְעָא.
_וּבְיַרְחָא_ תִּנְיָנָא, בְּעַשְׂרִין וְשַׁבְעָא יוֹמָא _לְיַרְחָא_ - יַבִּישַׁת אַרְעָא.
וּמַלִּיל יְיָ _עִם_ נֹחַ לְמֵימָר:
פּוּק מִן תֵּבוֹתָא - אַתְּ, וְאִתְּתָךְ ,וּבְנָךְ, וּנְשֵׁי בְנָיךְ עִמָךְ.
כָּל חַיְתָא דְּעִמָּךְ, מִכָּל בִּשְׂרָא - בְּעוֹפָא, וּבִבְעִירָא, וּבְכָל רִחֲשָׁא דְּרָחֵשׁ עַל אַרְעָא - אַפֵּיק עִמָּךְ; וְ_יִתְיַלְּדוּן_ בְּאַרְעָא, וְיִפְּשׁוּן וְיִסְגּוּן עַל אַרְעָא.
וּנְפַק נֹחַ, וּבְנוֹהִי, וְאִתְּתֵהּ, וּנְשֵׁי בְנוֹהִי עִמֵּהּ.
כָּל חַיְתָא - כָּל רִחֲשָׁא, וְכָל עוֹפָא, כֹּל דְּרָחֵשׁ עַל אַרְעָא - לְזַרְעֲיָתְהוֹן, נְפָקוּ מִן תֵּבוֹתָא.
וּבְנָא נֹחַ מַדְבְּחָא +קֳדָם+ יְיָ; וּנְסִיב מִכֹּל בְּעִירָא דַּכְיָא, וּמִכֹּל עוֹפָא דְכֵי, וְ_אַסֵּיק_ עֲלָוָן בְּמִדְבְּחָא.
וְ_קַבִּיל_ יְיָ _בְּרַעֲוָא יָת קוּרְבָּנֵהּ_, וַאֲמַר יְיָ _בְּמֵימְרֵהּ_: לָא אוֹסֵף לְמֵילַט עוֹד יָת אַרְעָא בְּדִיל +חוֹבֵי+ אֱנָשָׁא, אֲרֵי יִצְרָא לִבָּא דֶּאֱנָשָׁא בִּישׁ מִ_זְעִירֵהּ_, וְלָא אוֹסֵף עוֹד _לְמִמְחֵי_ יָת כָּל דְּחַי כְּמָא דִעַבָדִית.
עוֹד כָּל יוֹמֵי אַרְעָא, זְרוֹעָא וַחֲצָדָא, וְקוֹרָא וְחוֹמָא, וְקַיְטָא וְ_סִתְוָא_, וִ_ימָם_ וְלֵילְיָא - לָא _יִבְטְּלוּן_.
