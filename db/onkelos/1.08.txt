וּדְכִיר יְיָ יָת נֹחַ וְיָת כָּל חַיְתָא וְיָת כָּל בְּעִירָא דִּי עִמֵּהּ בְּתֵבוֹתָא וְאַעְבַּר יְיָ רוּחָא עַל אַרְעָא וְנָחוּ מַיָּא
וְאִסְתְּכָרוּ מַבּוּעֵי תְהוֹמָא וְכַוֵּי שְׁמַיָּא וְאִתְכְּלִי מִטְרָא מִן שְׁמַיָּא
וְתָבוּ מַיָּא מֵעַל אַרְעָא אָזְלִין וְתָיְבִין וַחֲסָרוּ מַיָּא מִסּוֹף מְאָה וְחַמְשִׁין יוֹמִין
וְנָחַת תֵּבוֹתָא בְּיַרְחָא שְׁבִיעָאָה בְּשִׁבְעַת עֲשַׂר יוֹמָא לְיַרְחָא עַל טוּרֵי קַרְדּוּ
וּמַיָּא הֲווֹ אָזְלִין וְחָסְרִין עַד יַרְחָא עֲשִׂירָאָה בַּעֲשִׂירָאָה בְּחַד לְיַרְחָא אִתְחֲזִיאוּ רֵישֵׁי טוּרַיָּא
וַהֲוָה מִסּוֹף אַרְבְּעִין יוֹמִין וּפְתַח נֹחַ יָת כַּוַּת תֵּבוֹתָא דִּי עֲבָד
וְשַׁלַּח יָת עוֹרְבָא וּנְפַק מִפַּק וְתַיִּב עַד דְּיַבִּישׁוּ מַיָּא מֵעַל אַרְעָא
וְשַׁלַּח יָת יוֹנָה מִלְּוָתֵהּ לְמֵחֲזֵי הֲקַלִּיאוּ מַיָּא מֵעַל אַפֵּי אַרְעָא
וְלָא אַשְׁכַּחַת יוֹנָה מְנַח לְפַרְסַת רַגְלַהּ וְתָבַת לְוָתֵהּ לְתֵבוֹתָא אֲרֵי מַיָּא עַל אַפֵּי כָל אַרְעָא וְאוֹשִׁיט יְדֵהּ וְנָסְבַהּ וְאָעֵיל יָתַהּ לְוָתֵהּ לְתֵבוֹתָא
וְאוֹרִיךְ עוֹד שַׁבְעָא יוֹמִין אָחֳרָנִין וְאוֹסֵיף שַׁלַּח יָת יוֹנָה מִן תֵּבוֹתָא
וַאֲתַת לְוָתֵהּ יוֹנָה לְעִדַּן רַמְשָׁא וְהָא טְּרַף זֵיתָא תְּבִיר נָחִית בְּפוּמַהּ וִידַע נֹחַ אֲרֵי קַלִּיאוּ מַיָּא מֵעַל אַרְעָא
וְאוֹרִיךְ עוֹד שַׁבְעָא יוֹמִין אָחֳרָנִין וְשַׁלַּח יָת יוֹנָה וְלָא אוֹסֵיפַת לְמִתּוּב לְוָתֵהּ עוֹד
וַהֲוָה בְּשִׁתְּ מְאָה וְחַד שְׁנִין בְּקַדְמָאָה בְּחַד לְיַרְחָא נְגוּבוּ מַיָּא מֵעַל אַרְעָא וְאַעְדִּי נֹחַ יָת חוֹפָאָה דְּתֵבוֹתָא וַחֲזָא וְהָא נְגוּבוּ אַפֵּי אַרְעָא
וּבְיַרְחָא תִּנְיָנָא בְּעַשְׂרִין וְשַׁבְעָא יוֹמָא לְיַרְחָא יַבִּישַׁת אַרְעָא
וּמַלִּיל יְיָ עִם נֹחַ לְמֵימָר
פּוּק מִן תֵּבוֹתָא אַתְּ וְאִתְּתָךְ וּבְנָךְ וּנְשֵׁי בְנָיךְ עִמָךְ
כָּל חַיְתָא דְּעִמָּךְ מִכָּל בִּשְׂרָא בְּעוֹפָא וּבִבְעִירָא וּבְכָל רִחֲשָׁא דְּרָחֵשׁ עַל אַרְעָא אַפֵּיק עִמָּךְ וְיִתְיַלְּדוּן בְּאַרְעָא וְיִפְּשׁוּן וְיִסְגּוּן עַל אַרְעָא
וּנְפַק נֹחַ וּבְנוֹהִי וְאִתְּתֵהּ וּנְשֵׁי בְנוֹהִי עִמֵּהּ
כָּל חַיְתָא כָּל רִחֲשָׁא וְכָל עוֹפָא כֹּל דְּרָחֵשׁ עַל אַרְעָא לְזַרְעֲיָתְהוֹן נְפָקוּ מִן תֵּבוֹתָא
וּבְנָא נֹחַ מַדְבְּחָא קֳדָם יְיָ וּנְסִיב מִכֹּל בְּעִירָא דַּכְיָא וּמִכֹּל עוֹפָא דְכֵי וְאַסֵּיק עֲלָוָן בְּמִדְבְּחָא
וְקַבִּיל יְיָ בְּרַעֲוָא יָת קוּרְבָּנֵהּ וַאֲמַר יְיָ בְּמֵימְרֵהּ לָא אוֹסֵף לְמֵילַט עוֹד יָת אַרְעָא בְּדִיל חוֹבֵי אֱנָשָׁא אֲרֵי יִצְרָא לִבָּא דֶּאֱנָשָׁא בִּישׁ מִזְעִירֵהּ וְלָא אוֹסֵף עוֹד לְמִמְחֵי יָת כָּל דְּחַי כְּמָא דִעַבָדִית
עוֹד כָּל יוֹמֵי אַרְעָא זְרוֹעָא וַחֲצָדָא וְקוֹרָא וְחוֹמָא וְקַיְטָא וְסִתְוָא וִימָם וְלֵילְיָא לָא יִבְטְּלוּן
