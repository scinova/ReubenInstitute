אַחְזְיַנִי יְיָ וְהָא תְּרֵין סַלֵי תֵינִין מַחֲתִין מִקָדָם הֵיכְלָא דַייָ בָּתַר דְאַגְלֵי נְבוּכַדְרֶצַר מַלְכָּא דְבָבֶל יַת יְכָנְיָה בַר יְהוֹיָקִים מְלַךְ שִׁבְטָא דְבֵית יְהוּדָה וְיַת רַבְרְבֵי יְהוּדָה וְיַת אוּמָנַיָא וְיַת תַּרְעֲיָא מִירוּשְׁלֵם וְאוֹבֵילִינוּן לְבָבֶל:
סִלָא הֲדָא תֵינִין טָבָא לַחֲדָא כְּתֵינִין בַּכִּירִין וְסַלָא חַד תֵּינִין בִּישִׁין לַחֲדָא (וּדְבִישִׁין בִּישִׁין לַחֲדָא) דְלָא מִתְאַכְלִין מִבִּישׁוּ:
וַאֲמַר יְיָ לִי מָה אַתְּ חָזֵי יִרְמְיָה וַאֲמָרִית תֵּינִין תֵּינַיָא טָבְתָא טָבָן לַחֲדָא וּדְבִישִׁין בִּישִׁין לַחֲדָא דְלָא מִתְאַכְלִין מִבִּישׁוּ:
וַהֲוָה פִּתְגַם נְבוּאָה מִקָדָם יְיָ עִמִי לְמֵימָר:
כִּדְנַן אֲמַר יְיָ אֱלָהָא דְיִשְׂרָאֵל כְּתֵינַיָא טָבָתָא הָאִלֵין כֵּן אִשְׁתְּמוֹדַע יַת גַלְוַת יְהוּדָה דְאַגְלֵיתִי מִן אַתְרָא הָדֵין לַאֲרַע כַּסְדָאֵי לְטָבְתָא:
וַאֲשַׁוֵי מֵימְרֵי עֲלֵיהוֹן לְטָבָא וַאֲתֵיבִינוּן לַאֲרַע הָדָא וְאֶבְנִינוּן וְלָא אֲפַגֵר וַאֲקִימִינוּן וְלָא אֶטַלְטֵל:
וְאֶתֵּן לְהוֹן לֵב לְמִידַע דְחֵלְתִּי אֲרֵי אֲנָא יְיָ וִיהוֹן קֳדָמַי לְעַם וַאֲנָא אֶהֱוֵי לְהוֹן לֶאֱלָהּ אֲרֵי יְתוּבוּן לְפוּלְחָנִי בְּכָל לִבְּהוֹן:
וּכְתֵינַיָא בִּישְׁתָא דְלָא מִתְאַכְלָא מִבִּישׁוּ אֲרֵי כִדְנַן אֲמַר יְיָ כֵּן אֶתֵּן יַת צִדְקִיָהוּ מְלַךְ שִׁבְטָא דְבֵית יְהוּדָה וְיַת רַבְרְבוֹהִי וְיַת שְׁאָרָא דִירוּשְׁלֵם דְאִשְׁתְּאָרוּ בְּאַרְעָא הָדָא וּדְיַתְבִין בְּאַרְעָא דְמִצְרָיִם:
וְאֶתְּנִינוּן לְזִיעַ וּלְבִישָׁא לְכָל מַלְכְּוַת אַרְעָא לְחִיסוּדִין וְלִמְתַל וּלְשׁוֹעִי וְלִלְוָטָא בְּכָל אַתְרַיָא דְאַגְלִינוּן לְתַמָן:
וֶאֱגָרֵי בְהוֹן יַת דְקַטְלִין בְּחַרְבָּא בְּכַפְנָא וּבְמוֹתָא עַד דִיסוּפוּן מֵעַל אַרְעָא דִיהָבֵית לְהוֹן וְלַאֲבָהַתְהוֹן:
