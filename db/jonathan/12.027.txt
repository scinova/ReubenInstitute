בְּעִדָנָא הַהִיא יַסְעַר יְיָ בְּחַרְבֵּהּ רַבְּתָא תַקִפְתָּא וַחֲסַנְתָּא עַל מַלְכָּא דְאִתְרַבְרַב כְּפַרְעֹה מַלְכָּא קַדְמָאָה וְעַל מַלְכָּא דְאִתְגָאֵי כְּסַנְחֶרִיב מַלְכָּא תִנְיָנָא וְיִקְטוֹל יַת מַלְכָּא דְתַקִיף כְּתַנִינָא דִי בְיַמָא:
בְּעִדָנָא הַהִיא כְּנִשְׁתָּא דְיִשְׂרָאֵל דְהִיא כְּכֶרֶם נְצִיב בְאַרְעָא טָבָא שַׁבַּחוּ לָהּ:
אֲנָא יְיָ נְטַר לְהוֹן קְיָם אֲבָהָתְהוֹן וְלָא אֱשֵׁיצִינוּן אֶלָא בִזְמַן דְאִינוּן מַרְגְזִין קֳדָמַי אֲנָא מַשְׁקֵי לְהוֹן כָּס פּוּרְעֲנוּתְהוֹן אִלוּלֵי חוֹבֵיהוֹן גָרְמִין לְאִתְפְּרַע מִנְהוֹן כְּבַר לֵילֵי וֵימָם מֵימְרִי מֵגִין עֲלֵיהוֹן:
הָא גְבוּרָן סַגִי קֳדָמַי הֲלָא אִם יְשַׁווּן בֵּית יִשְׂרָאֵל אֲפֵּיהוֹן לְמֶעְבַּד אוֹרַיְתָא אֶשְׁלַח רוּגְזִי וַחֲמָתִי בְּעַמְמַיָא דְמִתְגְרוּן בְּהוֹן וָאֱשֵׁיצִינוּן כְּמָא דִמְשֵׁיצֵי אֶשְׁתָּא הוֹבַאִי וּבוּר כַּחֲדָא:
אִם יִתְקְפוּן בְּפִתְגָמֵי אוֹרַיְתִי אִתְעֲבֵיד שְׁלָמָא לְהוֹן מִכְּעַן שְׁלָמָא יִתְעֲבֵיד לְהוֹן:
יִתְכַּנְשׁוּן מִבֵּינֵי גַלְוַתְהוֹן וִיתוּבוּן לְאַרְעֲהוֹן תַּמָן יִתְיְלִידוּן דְבֵית יַעֲקֹב יִפְשׁוּן וְיִסְגוּן דְבֵית יִשְׂרָאֵל וְיִמְלוּן אַפֵי תֵבֵל בְּנֵי בְנִין:
הַכְּמַחְתָּא דַהֲוָה מָחֵי מָחוֹהִי אִם כִּקְטוֹל קְטִילוֹהִי יִתְקְטֵיל:
בְּסָאתָא דַהֲוֵיתָא כְאֵיל בָּהּ יְכִילוּן לָךְ דַהֲוֵיתָא שָׁלַח וּמֵעִיק לְהוֹן הֲגֵי עֲלֵיהוֹן בְמִלָא מַתְקֵיף עֲלֵיהוֹן בְּיוֹם לְוָט:
בְּכֵן בְּדָא יִשְׁתְּבֵיקוּן חוֹבַי בֵּית יַעֲקֹב וְדֵין כָּל עוֹבָדֵי אַסְטִיוּת חוֹבוֹהִי בְּשַׁוְיוּתֵהּ כָּל אַבְנֵי אֱגוֹרָא כְּאַבְנָא גִיר מְנַפְצָן לָא יִתְקַיְמוּן אֲשֵׁרַיָא וְחַנִסְנְסַיָא:
אֲרֵי קַרְתָּא דַהֲוָת כְּרִיכָא בִּלְחוֹדָא תִּתַב תְּהֵא מְטַלְטְלָא וּשְׁבִיקָא כְּמַדְבְּרָא בָּהּ יְגִיחוּן צַדִיקַיָא וְיִבְזוּן יַת נִכְסָהָא וִיסוּפוּן מַשִׁרְיָתָהָא מִלְמִפָּק:
יִתְקַצַר חֵילֵיהוֹן וְיִבְהֲתוּן בְּעוֹבְדֵיהוֹן יִתַּבְּרוּן נְשַׁיָא עָלָן לְבֵית דְחַלְתְּהוֹן וּמַלְפָן יַתְהוֹן אֲרֵי לָא עַם סוּכְלְתָן אִינוּן עַל כֵּן לָא יְרַחֵם עֲלֵיהוֹן דְעַבְדִינוּן וּדְבָרָנוּן לָא יָחוּס עֲלֵיהוֹן:
וִיהֵי בְּעִדָנָא הַהִיא יִתְרְמוּן קְטִילִין קֳדָם יְיָ מִכֵּיף נַהֲרָא פְּרָת עַד נַחֲלָא דְמִצְרָיִם וְאַתּוּן תִּתְקָרְבוּן חַד לִסְטַר חַד בְּנֵי יִשְׂרָאֵל:
וִיהֵי בְּעִדָנָא הַהִיא יִתְקַע בְּשׁוֹפָרָא רַבָּא וְיֵיתוּן דִי גְלוֹ לְאַרְעָא דְאַתּוּר וּדְאִטַלְטָלוּ לְאַרְעָא דְמִצְרַיִם וְיִסְגְדוּן קֳדָם יְיָ בְּטוּרָא דְקוּדְשָׁא בִּירוּשְׁלֵם:
