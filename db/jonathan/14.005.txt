וְאַתְּ בַּר אָדָם סַב לָךְ חַרְבָּא חֲרִיפָא מַסְפַּר גַלְבַיָא תִּסַב לָךְ וְתַעְבֵּד עַל רֵישֵׁךְ וְעַל דִקְנָךְ וְתִסֵב לָךְ מוֹזְנָיָן דְמַתְקַל וְתַפְלְגִינוּן:
תִּלְתָּא בְּנוּרָא תַדְלֵיק בְּגוֹ קַרְתָּא כְּמִשְׁלַם יְמֵי צַיְרָא וְתִסַב יַת תִּלְתָּא תִּקְטֵיל בְּחַרְבָּא סָחֲרָנָהָא וְתִלְתָּא יִתְבַּדְרוּן וּדְקַטְלִין בְּחַרְבָּא אֱגָרֵי בַּתְרֵיהוֹן:
וְתִסֵב מִתַּמָן זְעֵיר בְּמִנְיַן וּתְצוּר יַתְהוֹן בִכְנָפָךְ:
וּמִנְהוֹן עוֹד תִּסַב וְתַרְמֵי יַתְהוֹן לְגוֹ נוּרָא וְתוֹקִיד יַתְהוֹן בְּנוּרָא בְּכֵן תִּפּוֹק אֶשְׁתָּא עַל כָּל בֵּית יִשְׂרָאֵל:
כִּדְנַן אֲמַר יְיָ אֱלֹהִים דָא יְרוּשְׁלֵם בְּגוֹ עַמְמַיָא שַׁוִיתָהָא וּסְחוֹר סְחוֹר לָהּ מְדִינָתָא:
וְאַשְׁנִיאַת יַת דִינִי לִמְחַבְּבֵיהוֹן מִן עַמְמַיָא וְיַת קְיָמִי מִן מְדִינָתָא דִי בְּסֶחֱרָנָהָא אֲרֵי בְּדִינַי קָצוּ וּקְיָמַי לָא הֲלִיכוּ בְהוֹן:
בְּכֵן כִּדְנַן אֲמַר יְיָ אֱלֹהִים חֲלַף דְחַבְתּוּן מִן עַמְמַיָא דִי בְסַחֲרָנֵיכוֹן בִּקְיָמַי לָא הֲלֵכְתּוּן וְיַת דִינַי לָא עֲבַדְתּוּן וּכְנִימוּסֵי עַמְמַיָא דִי בְסַחְרָנֵיכוֹן לָא עֲבַדְתּוּן:
בְּכֵן כִּדְנַן אֲמַר יְיָ אֱלֹהִים הָא אֲנָא שָׁלַח רוּגְזִי עֲלָךְ אַף אֲנָא וְאַעֲבֵּיד בְּגַוִיךְ פּוּרְעֲנוּת דִינִין לְעֵינֵי עַמְמַיָא:
וְאַעְבֵּיד בִּיךְ יַת דְלָא עֲבַדֵית וְיַת דְלָא אַעְבֵּיד דִכְוָתֵיהּ עוֹד חֲלַף כָּל תּוֹעֲבָתָיִךְ:
בְּכֵן אֲבָהָן יֵיכְלוּן בְּסַר בְּנֵיהוֹן בְּגַוִיךְ וּבְנַיָא יֵיכְלוּן בְּסַר אֲבָהָתְהוֹן וְאַעְבֵּיד בִּיךְ פּוּרְעֲנוּת דִינִין וַאֲבַדַר יַת כָּל שְׁאָר עַמִיךְ לְכָל רוּחַ:
בְּכֵן קַיָים אֲנָא אֲמַר יְיָ אֱלֹהִים אִם לָא חֲלַף דְמָקְדְשֵׁי סָאֵבְתְּ בְּכָל שִׁקוּצָיִךְ וּבְכָל תּוֹעֲבָתָיִךְ וְאַף אֲנָא אֶקְטוֹף תְּקוֹף דַרְעִיךְ וְלָא יָחוּס מֵימְרִי וְאַף אֲנָא לָא אֲרַחֵם:
תִּלְתִּיךְ בְּמוֹתָא יְמוּתוּן וּבְכַפְנָא יִשְׁתֵּיצוּן בְּגַוִיךְ וְתִלְתָּא בְּחַרְבָּא יִתְקַטְלוּן סַחֲרָנַיִךְ וְתִלְתָּא לְכָל רוּחַ אֲבַדַר וּדְקָטְלִין בְּחַרְבָּא אֱגָרֵי בַּתְרֵיהוֹן:
וְיָחוּל רוּגְזִי וַאֲנִיחַ חֵימְתִי בְהוֹן וַאֲסִיף וְיֵדְעוּן אֲרֵי אֲנָא יְיָ גְזַרֵית בְּמֵימְרִי וַאֲקַיֵים כַּד יָחוּל רוּגְזִי בְּהוֹן:
וְאֶתְּנִנָךְ לְחַרְבָּא וּלְחִסוּדִין בֵּינֵי עַמְמַיָא דִי בְסַחְרָנָךְ לְעֵינֵי כָּל דְעָדֵי:
וּתְהֵין לְחִסוּדִין וּלְאִתְרַבְרְבָא עֲלָךְ צָדוּ וְאִשְׁתְּמָמוּ לְעַמְמַיָא דִבְסָחֳרָנִיךְ כַּד אַעְבֵיד בִּיךְ פּוּרְעֲנוּת דִינִין בִּרְגַז וּבְחֵמָה וּבִתְקוֹף רְגַז אֲנָא יְיָ גְזַרֵית בְּמֵימְרִי:
בְּגָרָיוּתִי יַת מְחַת כַּפְנָא מַכְתָּשִׁין בִּישִׁין דִמְתַקְנִין לְחַבָּלָא דַאֲשַׁלַח יַתְהוֹן לְחַבָּלוּתְכוֹן וְכַפְנָא אוֹסִיף עֲלֵיכוֹן וְאִתְבַּר לְכוֹן סְעַד מֵיכְלָא:
וַאֲשַׁלַח עֲלֵיכוֹן כַּפְנָא וְחַיְתָא בִישְׁתָּא וְיִתְכְּלוּנַיִךְ וּמוֹתָא וּקְטָלָא יַעְדוּן בִּיךְ וּדְקָטְלִין בְּחַרְבָּא אַיְתֵי עֲלָךְ אֲנָא יְיָ גְזָרֵית בְּמֵימְרִי:
