וּנְפַק עַדְבָא תִנְיָנָא לְשִׁמְעוֹן לְשִׁבְטָא דִבְנֵי שִׁמְעוֹן לְזַרְעֲיַתְהוֹן וַהֲוָה אַחֲסַנְתְּהוֹן בְּגוֹ אַחֲסָנַת בְּנֵי יְהוּדָה:
וַהֲוָה לְהוֹן בְּאַחֲסַנְתְּהוֹן בְּאֵר שֶׁבַע וְשֶׁבַע וּמוֹלָדָה:
וַחֲצַר שׁוּעָל וּבָלָה וְעֶצֶם:
וְאֶלְתּוֹלַד וּבְתוּל וְחָרְמָה:
וְצִקְלַג וּבֵית הַמַרְכָּבֹת וַחֲצַר סוּסָה:
וּבֵית לְבָאוֹת וְשָׁרוּחֶן קִרְוִין תְּלַת עַסְרֵי וּפַצְחֵיהֶן:
עַיִן רִמוֹן וְעֶתֶר וְעָשָׁן קִרְוִין אַרְבַּע וּפַצְחֵיהֶן:
וְכָל פַּצְחַיָא דִי בְּסַחְרָנוּת קִרְוַיָא הָאִלֵין עַד בַּעֲלַת בְּאֵר רָאמַת דָרוֹמָא דָא אַחְסָנַת שִׁבְטָא דִבְנֵי שִׁמְעוֹן לְזַרְעֲיַתְהוֹן:
מֵעֲדַב בְּנֵי יְהוּדָה אַחְסָנַת בְּנֵי שִׁמְעוֹן אֲרֵי הֲוָה חִילַק בְּנֵי יְהוּדָה סַגִי מִנְהוֹן וְאַחְסִינוּ בְּנֵי שִׁמְעוֹן בְּגוֹ אַחְסַנְתְּהוֹן:
וּסְלֵיק עַדְבָא תְלִיתָאָה לִבְנֵי זְבוּלֻן לְזַרְעֲיַתְהוֹן וַהֲוָה תְּחוּם אַחְסַנְתְּהוֹן עַד שָׂרִיד:
וּסְלִיק תְּחוּמְהוֹן לְיַמָא וּמַרְעֲלָא וּמְעָרַע בְּדַבָּשֶׁת וּמְעָרַע בְּנַחֲלָא דִי עַל אַפֵי יָקְנְעָם:
וְתַב מִשָׂרִיד לְקִידוּמָא מַדְנַח שִׁמְשָׁא עַל תְּחוּם כִּסְלוֹת תָּבוֹר וְנָפֵיק לְדֻבְּרַת וּסְלִיק לְיָפִיעַ:
וּמִתַּמָן עֲבַר לִקְדוּמָא מַדִינְחָא גִתָּה חֵפֶר עִתָּה קָצִין וְנָפֵיק לְרִמוֹן דִמְתוֹאַר וּמִתַּמָן מִסְתְּחַר לְנֵעָה:
וּמִסְתְּחַר לֵיהּ תְּחוּמָא מִצִפּוּנָא לְחַנָתוֹן וַהֲווֹן מַפְקָנוֹהִי לְחֵילַת יִפְתַּח אֵל:
וְקַטָת וְנַהֲלָל וְשִׁמְרוֹן וְיִדְאֲלָה וּבֵית לָחֶם קִרְוִין תַּרְתָּא עַסְרֵי וּפַצְחֵיהֶן:
דָא אַחְסָנַת בְּנֵי זְבוּלֻן לְזַרְעֲיַתְהוֹן קִרְוַיָא הָאִלֵין וּפַצְחֵיהֶן:
לְיִשָׂשכָר נְפַק עַדְבָא רְבִיעָאָה לִבְנֵי יִשָׂשכָר לְזַרְעֲיָתְהוֹן:
וַהֲוָה תְּחוּמְהוֹן יִזְרְעֶאל וּכְסֻלוֹת וְשׁוּנֵם:
וַחַפָרַיִם וְשִׁיאוֹן וַאֲנָחֲרָת:
וְהָרַבִּית וְקִשְׁיוֹן וָאָבֶץ:
וְרֶמֶת וְעֵין גַנִים וְעֵין חַדָה וּבֵית פַּצֵץ:
וּמְעָרַע תְּחוּמָא בְּתָבוֹר וְשַׁחֲצִימָה וּבֵית שֶׁמֶשׁ וַהֲווֹן מַפְקָנוֹהִי דִתְחוּמְהוֹן לְיַרְדְנָא קִרְוִין שִׁית עַסְרֵי וּפַצְחֵיהֶן:
דָא אַחְסָנַת שִׁבְטָא דִבְנֵי יִשָׂשכָר לְזַרְעֲיַתְהוֹן קִרְוַיָא וּפַצְחֵיהֶן:
וּנְפַק עַדְבָא חֲמִישָׁאָה לְשִׁבְטָא דִבְנֵי אָשֵׁר לְזַרְעֲיַתְהוֹן:
וַהֲוָה תְּחוּמְהוֹן חֶלְקַת וַחֲלִי וָבֶטֶן וְאַכְשָׁף:
וְאַלַמֶלֶךְ וְעַמְעָד וּמִשְׁאָל וּמְעָרַע בְּכַרְמְלָא לְיַמָא וּבְשִׁיחוֹר לִבְנָת:
וְתַב מִמַדְנַח שִׁמְשָׁא בֵּית דָגוֹן וּמְעָרַע בִּזְבוּלֻן וּבְחֵילַת יִפְתַּח אֵל לְצִפּוּנָא בֵּית מֵישְׁרָא וּנְעִיאֵל וְנָפֵיק לְכָבוּל מִשְׂמָאלָא:
וְעֶבְרוֹן וּרְחֹב וְחַמוֹן וְקָנָה עַד צִידוֹן רַבָּה:
וְתַב תְּחוּמָא לְרָמָתָא וְעַד קִרְוִין כְּרִיכָן קִרְוֵי תוּקְפָא וְתַב תְּחוּמָא לְחוֹסָה וִיהוֹן מַפְקָנוֹהִי לְיַמָא מֵעֲדַב אַכְזִיב:
וְעֻמָה וַאֲפֵק וּרְחוֹב קִרְוִין תַּרְתֵּין וְעֶשְׂרִין וּפַצְחֵיהֶן:
דָא אַחֲסָנַת שִׁבְטָא דִבְנֵי אָשֵׁר לְזַרְעֲיַתְהוֹן קִרְוַיָא הָאִלֵין וּפַצְחֵיהֶן:
לִבְנֵי נַפְתָּלִי נְפַק עַדְבָא שְׁתִיתָאָה לִבְנֵי נַפְתָּלִי לְזַרְעֲיַתְהוֹן:
וַהֲוָה תְּחוּמְהוֹן מֵחֵלֶף מֵאַלוֹן בְּצַעֲנַנִים וַאֲדָמִי הַנֶקֶב וְיַבְנְאֵל עַד לַקוּם וַהֲווֹן מַפְקָנוֹהִי לְיַרְדְנָא:
וְתַב תְּחוּמָא לְיַמָא לְאַזְנוֹת תָּבוֹר וּנְפַק מִתַּמָן לְחֻקֹקָה וּמְעָרַע בִּזְבוּלֻן מִדָרוֹמָא וּבְאָשֵׁר יְעָרַע מִמַעֲרָבָא וּבִיהוּדָה יַרְדְנָא מַדְנַח שִׁמְשָׁא:
וְקִרְוִין כְרִיכָן הַצִדִים צֵר וְחַמַת רַקַת וְכִנָרֶת:
וַאֲדָמָה וְהָרָמָה וְחָצוֹר:
וְקֶדֶשׁ וְאֶדְרֶעִי וְעֵין חָצוֹר:
וְיִרְאוֹן וּמִגְדַל אֵל חָרֵם וּבֵית עֲנָת וּבֵית שֶׁמֶשׁ קִרְוִין תְּשַׁע עַשְׂרֵי וּפַצְחֵיהֶן:
דָא אַחֲסָנַת שִׁבְטָא דִבְנֵי נַפְתָּלִי לְזַרְעֲיַתְהוֹן קִרְוַיָא וּפַצְחֵיהֶן:
לְשִׁבְטָא דִבְנֵי דָן לְזַרְעֲיַתְהוֹן נַפְקָא עַדְבָא שְׁבִיעָאָה:
וַהֲוָה תְּחוּם אַחְסַנְתְּהוֹן צָרְעָה וְאֶשְׁתָּאוֹל וְקִרְיַת שִׁמְשָׁא:
וְשַׁעֲלַבִּין וְאַיָלוֹן וְיִתְלָה:
וְאֵילוֹן וְתִמְנָתָה וְעֶקְרוֹן:
וְאֶלְתְּקֵה וְגִבְּתוֹן וּבַעֲלָת:
וִיהֻד וּבְנֵי בְרַק וְגַת רִמוֹן:
וּמֵי הַיַרְקוֹן וְרַקוֹן עִם תְּחוּמָא דִי לָקֳבֵיל יָפוֹ:
וּנְפַק תְּחוּם בְּנֵי דָן מִנְהוֹן וּסְלִיקוּ בְּנֵי דָן וְאַגִיחוּ קְרָבָא עִם לֶשֶׁם וּכְבַשׁוּ יָתָהּ וּמְחוֹ יָתָה לְפִתְגַם דְחֶרֶב וִירִיתוּ יָתָהּ וִיתִיבוּ בָהּ וּקְרוֹ לְלֶשֶׁם דָם כְּשׁוּם דָן אֲבוּהוֹן:
דָא אַחֲסָנַת שַׁבְטָא דִבְנֵי דָן לְזַרְעֲיַתְהוֹן קִרְוַיָא הָאִלֵין וּפַצְחֵיהֶן:
וְשֵׁיצִיאוּ לְאַחְסָנָא יַת אַרְעָא לִתְחוּמָהָא וִיהָבוּ בְנֵי יִשְׂרָאֵל אַחֲסָנָא לִיהוֹשֻׁעַ בַּר נוּן בֵּינֵיהוֹן:
עַל מֵימְרָא דַייָ יְהֲבוּ לֵיהּ יַת קַרְתָּא דִי שְׁאֵל יַת תִּמְנַת סָרַח בְּטוּרָא דְאֶפְרָיִם וּבְנָא יַת קַרְתָּא וִיתִיב בָּהּ:
אִילֵן אַחֲסַנְתַּיָא דִי אַחֲסִין אֶלְעָזָר כַּהֲנָא וִיהוֹשֻׁעַ בַּר נוּן וְרֵישֵׁי אַבְהָתָא לְשִׁבְטַיָא דִבְנֵי יִשְׂרָאֵל בְּעַדְבָא בְּשִׁלֹה קֳדָם יְיָ בִּתְרַע מַשְׁכַּן זִמְנָא וְשֵׁיצִיאוּ מִלְפַלָגָא יַת אַרְעָא:
