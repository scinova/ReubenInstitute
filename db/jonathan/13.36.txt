וַהֲוָה בְּשַׁתָּא רְבִיעֵיתָא לִיהוֹיָקִים בַּר יֹאשִׁיָה מְלַךְ שִׁבְטָא דְבֵית יְהוּדָה הֲוָה פִּתְגָמָא הָדֵין עִם יִרְמְיָה מִן קֳדָם יְיָ לְמֵימָר:
סַב לָךְ מְגִלַת סַפְרָא וְתִכְתּוֹב עֲלָהּ יַת כָּל פִּתְגָמַיָא דְמַלְלֵית עִמָךְ עַל יִשְׂרָאֵל וְעַל יְהוּדָה וְעַל כָּל עַמְמַיָא מִיוֹמָא דְמַלְלֵית עִמָךְ מִימֵי יֹאשִׁיָה וְעַד יוֹמָא הָדֵין:
מָאִים יִשְׁמְעוּן דְבֵית יְהוּדָה יַת כָּל בִּישְׁתָא דַאֲנָא מְחַשֵׁב לְמֶעְבַּד לְהוֹן בְּדִיל דִיתוּבוּן גְבַר מֵאוֹרְחֵיהּ בִּישְׁתָא וְאֶשְׁבּוֹק לְחוֹבֵיהוֹן וּלְחַטָאֵיהוֹן:
וּקְרָא יִרְמְיָה יַת בָּרוּךְ בַּר נֵרִיָה וּכְתַב בָּרוּךְ מִפּוּם דְיִרְמְיָה יַת כָּל פִּתְגָמַיָא דַייָ דְמַלֵיל עִמֵיהּ עַל מְגִלַת סַפְרָא:
וּפַקֵיד יִרְמְיָה יַת בָּרוּךְ לְמֵימָר אֲנָא כְלֵי לֵית אֲנָא יָכִיל לְמֵיעַל לְבֵית מַקְדְשָׁא דַיָי:
וְתֵיתֵי אַתְּ וְתִקְרֵי בִּמְגִלָתָא דִכְתַבְתָּא מִפּוּמִי יַת פִּתְגָמַיָא דַייָ קֳדָם עַמָא בְּבֵית מַקְדְשָׁא דַייָ בְּיוֹמָא דְצוֹמָא וְאַף קֳדָם כָּל בֵּית יְהוּדָה דְאָתָן מִקִרְוֵיהוֹן תִּקְרִינוּן:
מָאִים תִּתְקַבֵּיל בְעוּתְהוֹן קֳדָם יְיָ וִיתוּבוּן גְבַר מֵאוֹרְחָתֵיהּ בִּישְׁתָא אֲרֵי רַב רוּגְזָא וְחֵימְתָא דִגְזַר יְיָ עַל עַמָא הָדֵין:
וַעֲבַד בָּרוּךְ בַּר נֵרִיָה כְּכֹל דִי פַקְדֵיהּ יִרְמְיָה נְבִיָא לְמִקְרֵי בְּסִפְרָא פִּתְגָמָא דַייָ בֵּית מַקְדְשָׁא דַיָי:
וַהֲוָה בְּשַׁתָּא חֲמִשְׁתָּא לִיהוֹיָקִים בַּר יֹאשִׁיָה מְלַךְ שִׁבְטָא דְבֵית יְהוּדָה בְּיַרְחָא תְשִׁיעָאָה גְזָרוּ צוֹמָא קֳדָם יְיָ כָּל עַמָא בִּירוּשְׁלֵם וְכָל עַמָא דְאָתָן מִקִרְוַיָא דְבֵית יְהוּדָה בִּירוּשְׁלֵם:
וּקְרָא בָּרוּךְ בְּסִפְרָא יַת פִּתְגָמֵי יִרְמְיָה נְבִיָא בֵּית מַקְדְשָׁא דַייָ בְּלִשְׁכַּת גְמַרְיָה בַר שְׁפָן סַפְרָא בְּדַרְתָּא עִלִיתָא בְּמַעֲלָנָא דִתְרַע בֵּית מַקְדְשָׁא דַייָ מַדִינְחָאָה קֳדָם כָּל עַמָא:
וּשְׁמַע מִיכָיְהוּ בַּר גְמַרְיָה בַּר שָׁפָן יַת כָּל פִּתְגָמָא דַייָ מֵעַל סִפְרָא:
וּנְחֵית לְבֵית מַלְכָּא לְלִשְׁכַּת סַפְרָא וְהָא תַמָן כָּל רַבְרְבַיָא יַתְבִין אֱלִישָׁמָע סַפְרָא וּדְלָיָה בַר שְׁמַעְיָה וְאֶלְנָתָן בַּר עַכְבּוֹר וּגְמַרְיָה בַר שָׁפָן וְצִדְקִיָה בַר חֲנַנְיָה וְכָל רַבְרְבַיָא:
וְחַוֵי לְהוֹן מִיכָיְהוּ יַת כָּל פִּתְגָמַיָא דִשְׁמַע כַּד קְרָא בָּרוּךְ בְּסַפְרָא קֳדָם עַמָא:
וּשְׁלָחוּ כָּל רַבְרְבַיָא לְוַת בָּרוּךְ יַת יְהוּדִי בַּר נְתַנְיָה בַּר שְׁלֶמְיָה בַּר כּוּשִׁי לְמֵימַר מְגִלָתָא דְקָרֵיתָא בָּהּ קֳדָם עַמָא סַבְהִי בִּידָךְ וֶאֱתָא וּנְסִיב בָּרוּךְ בַּר נֵרִיָה יַת מְגִלָתָא בְיָדֵיהּ וַאֲתָא לְוַתְהוֹן:
וַאֲמָרוּ לֵיהּ תּוּב כְּעַן וּקְרָהָא קֳדָמָנָא וּקְרָא בָּרוּךְ קֳדָמֵיהוֹן:
וַהֲוָה כְּמִשְׁמַעְהוֹן יַת כָּל פִּתְגָמַיָא תְּוָהוּ גְבַר לְחַבְרֵיהּ וַאֲמַרוּ לְבָרוּךְ חַוָאָה נְחַוֵי לְמַלְכָּא יַת כָּל פִּתְגָמַיָא הָאִלֵין:
וְיַת בָּרוּךְ שְׁאִילוּ לְמֵימָר חַוֵי כְעַן לָנָא אֵיכְדֵין כָּתַבְתָּ יַת כָּל פִּתְגָמַיָא הָאִלֵין מִפּוּמֵיהּ:
וַאֲמַר לְהוֹן בָּרוּךְ מִפּוּמֵיהּ קָרֵי לִי יַת כָּל פִּתְגָמַיָא הָאִלֵין וַאֲנָא כָּתֵיב עַל סִפְרָא בִּדְיוּתָא:
וַאֲמַרוּ רַבְרְבַיָא לְבָרוּךְ אֱזֵיל אִטַמַר אַתְּ וְיִרְמְיָה וֶאֱנַשׁ לָא יְדַע אֵיכָא אַתּוּן:
וֶאֱתוֹ לְוַת מַלְכָּא לְדַרְתָּא וְיַת מְגִילָתָא פַּקִידוּ בְּלִשְׁכַּת אֱלִישָׁמָע סַפְרָא וְחַוִיאוּ קֳדָם מַלְכָּא יַת כָּל פִתְגָמַיָא:
וּשְׁלַח מַלְכָּא יַת יְהוּדִי לְמֵיסַב יַת מְגִילָתָא וְנַסְבָהּ מִלִשְׁכַּת אֱלִישָׁמָע סַפְרָא וּקְרָהָא יְהוּדִי קֳדָם מַלְכָּא וְקָדָם כָּל רַבְרְבַיָא דְקַיְמִין מֵעִלַוֵי מַלְכָּא:
וּמַלְכָּא יָתֵיב בֵּית סִתְוָא בְּיַרְחָא תְּשִׁיעַיְתָא וְיַת נוּרָא קֳדָמוֹהִי מַדְלְקִין:
וַהֲוָה כַּד קָרֵי יְהוּדִי תְּלַת פַּצִין וְאַרְבַּע בְּזַע בְּאִזְמֵיל סַפְרָא וּרְמֵי לְנוּרָא דִלְגוּמְרֵיהּ עַד דִשְׁלִימַת כָּל מְגִלְתָא עַל נוּרָא דְעַל גוּמְרֵי:
וְלָא תְוָהוּ וְלָא בְזָעוּ יַת לְבוּשֵׁיהוֹן מַלְכָּא וְכָל עַבְדוֹהִי דְשָׁמְעִין יַת כָּל פִּתְגָמַיָא הָאִלֵין:
וְאַף אֶלְנָתָן וּדְלָיָה וּגְמַרְיָה בְעוֹ מִן מַלְכָּא בְּדִיל דְלָא לְאוֹקָדָא יַת מְגִלְתָא וְלָא קַבֵּיל מִנְהוֹן:
וּפַקֵיד מַלְכָּא יַת יְרַחְמְאֵל בַּר מַלְכָּא וְיַת שְׂרָיָה בַר עַזְרִיאֵל וְיַת שְׁלֶמְיָה בַּר עַבְדִיאֵל לְמִדְבַּר יַת בָּרוּךְ סַפְרָא וְיַת יִרְמְיָה נְבִיָא וְאַגִין עֲלֵיהוֹן יְיָ:
וַהֲוָה פִתְגַם נְבוּאָה מִן קֳדָם יְיָ עִם יִרְמְיָה בָּתַר דְאוֹקֵיד מַלְכָּא יַת מְגִלָתָא וְיַת פִּתְגָמַיָא דִכְתַב בָּרוּךְ מִפּוּמָא דְיִרְמְיָה לְמֵימָר:
תּוּב סַב לָךְ מְגִילָתָא אוֹחֲרֵי וּכְתוֹב עֲלָהּ יַת כָּל פִּתְגְמַיָא קֳדָמָאֵי דַהֲווֹ עַל מְגִלָתָא קַדְמַיְתָא דְאוֹקֵיד יְהוֹיָקִים מְלַךְ שִׁבְטָא דְבֵית יְהוּדָה:
וְעַל יְהוֹיָקִים מְלַךְ שִׁבְטָא דְבֵית יְהוּדָה תֵּימַר כִּדְנַן אֲמַר יְיָ אַתְּ אוֹקֵידְתָּא יַת מְגִלָתָא הָדָא לְמֵימָר מָא דַין כְּתַבְתָּא עֲלָהּ לְמֵימָר מֵיתֵי יֵיתֵי מַלְכָּא דְבָבֶל וִיחַבֵּל יַת אַרְעָא הָדָא וִיבַטֵל מִנָהּ אֱנָשָׁא וּבְעִירָא:
בְּכֵן כִּדְנַן אֲמַר יְיָ עַל יְהוֹיָקִים מְלַךְ שִׁבְטָא דְבֵית יְהוּדָה לָא יְהֵי לֵיהּ יָתֵיב עַל כּוּרְסֵי דָוִד וּנְבֶלְתֵּיהּ תְּהֵא מְשַׁגְרָא לִשְׁרָבָא בִימָמָא וְלִגְלִידָא בְלֵילְיָא:
וְאַסְעַר עֲלוֹהִי וְעַל בְּנוֹהִי וְעַל עַבְדוֹהִי יַת חוֹבֵיהוֹן וְאַיְתִי עֲלֵיהוֹן וְעַל יַתְבֵי יְרוּשְׁלֶם וְעַל אֱנַשׁ יְהוּדָה יַת כָּל בִּישָׁתָא דִגְזָרֵית עֲלֵיהוֹן וְלָא קַבִּילוּ:
וְיִרְמְיָה נְסִיב מְגִלָה אוֹחֲרֵי וְיָהֲבָהּ לְבָרוּךְ בַּר נֵרִיָה סַפְרָא דְיִרְמְיָה יַת כָּל פִּתְגָמֵי סִפְרָא דְאוֹקֵיד יְהוֹיָקִים מְלַךְ שִׁבְטָא דְבֵית יְהוּדָה בְּנוּרָא וְעוֹד אִתּוֹסָפוּ עֲלֵיהוֹן פִּתְגָמִין סַגִיאִין כְּוַתְהוֹן:
