וִיתִיבוּ תְּלַת שְׁנִין לֵית קְרָבָא בֵּין אֲרָם וּבֵין יִשְׂרָאֵל:
וַהֲוָה בְּשַׁתָּא תְּלִיתָאָה וּנְחַת יְהוֹשָׁפָט מֶלֶךְ שִׁבְטָא דְבֵית יְהוּדָה לְוַת מַלְכָּא דְיִשְׂרָאֵל:
וַאֲמַר מַלְכָּא דְיִשְׂרָאֵל לְעַבְדוֹהִי הַיְדַעְתּוּן אֲרֵי דִילָנָא רָמוֹת גִלְעָד וַאֲנַחְנָא שְׁתִיקִין מִלְמֵיסַב יָתָהּ מִיַד מַלְכָּא דַאֲרָם:
וַאֲמַר לִיהוֹשָׁפָט הֲתֵיזִל עִמִי לְאַגָחָא קְרָבָא לְרָמוֹת גִלְעָד וַאֲמַר יְהוֹשָׁפָט לְמַלְכָּא דְיִשְׂרָאֵל אֲנָא כְּוָתָךְ עַמִי כְעַמָךְ וְסוּסְוָתַי כְּסוּסְוָתָךְ:
וַאֲמַר יְהוֹשָׁפָט לְמַלְכָּא דְיִשְׂרָאֵל תְּבַע כְּעַן כְּיוֹמָא דֵין יַת פִּתְגָמָא דַיָי:
וּכְנַשׁ מַלְכָּא דְיִשְׂרָאֵל יַת נְבִיֵי שִׁקְרָא כְּאַרְבַּע מֵאָה גַבְרָא וַאֲמַר לְהוֹן הַאֵיזִיל בְּרָמוֹת גִלְעָד לְאַגָחָא קְרָבָא אִם אִתְמְנַע וַאֲמַרוּ סַק וְיִמְסַר יְיָ בְּיַד מַלְכָּא:
וַאֲמַר יְהוֹשָׁפָט הֲלֵית כָּא נְבִיָא קֳדָם יְיָ עוֹד וְנִתְבַּע מִנֵיהּ:
וַאֲמַר מַלְכָּא דְיִשְׂרָאֵל לִיהוֹשָׁפָט עוֹד גַבְרָא חַד לְמִתְבַּע יַת פִּתְגָמָא דַייָ מִינֵיהּ וַאֲנָא סְנֵיתֵיהּ אֲרֵי לָא יִתְנַבֵּי עֲלַי טַב אֱלָהֵין בִּישׁ מִיכָה בַּר יִמְלָה וַאֲמַר יְהוֹשָׁפָט לָא יֵימַר מַלְכָּא כֵּן:
וּקְרָא מַלְכָּא דְיִשְׂרָאֵל לְגַוְזָאָה חַד וַאֲמַר אוֹחִי תְּבַע מִיכָיְהוּ בַּר יִמְלָה:
וּמַלְכָּא דְיִשְׂרָאֵל וִיהוֹשָׁפָט מַלְכָּא דִיהוּדָה יַתְבִין גְבַר עַל כּוּרְסֵיהּ לְבוּשִׁין לְבִישִׁין וְיַתְבִין בְּאִידְרָא בְּמַעֲלָנָא דִתְרַע שֹׁמְרוֹן וְכָל נְבִיֵי שִׁקְרָא מִתְנַבְּאִין קֳדָמֵיהוֹן:
וַעֲבַד לֵיהּ צִדְקִיָה בַר כְּנַעֲנָה קַרְנִין דְבַרְזֶל וַאֲמַר כִּדְנַן אֲמַר יְיָ בְּאִילֵין תְּקַטֵל יַת אֱנַשׁ אֲרָם עַד דִי תְשֵׁיצִינוּן:
וְכָל נְבִיֵי שִׁקְרָא מִתְנַבַּן כֵּן לְמֵימַר סַק לְרָמוֹת גִלְעָד וְאַצְלַח וְיִמְסַר יְיָ בִּידָא דְמַלְכָּא:
וְאִזְגַדָא דַאֲזַל לְמִקְרֵי מִיכָיְהוּ מַלֵיל עִמֵיהּ לְמֵימַר הָא כְעַן פִתְגָמֵי נְבִיֵי שִׁקְרָא מַמְלַל חַד שַׁוִי קֳדָם מַלְכָּא יְהֵי כְעַן פִּתְגָמָךְ כְּפִתְגָמָא דְחַד מִנְהוֹן וּתְמַלֵיל פִּתְגָמִין תַּקְנִין:
וַאֲמַר מִיכָיְהוּ קַיָם הוּא יְיָ אֲרֵי יַת דְיֵימַר יְיָ לִי יָתֵיהּ אֲמַלֵל:
וַאֲתָא לְוַת מַלְכָּא וַאֲמַר מַלְכָּא לֵיהּ מִיכָיְהוּ הֲנֵיזֵיל רָמוֹת גִלְעָד לַאֲגָחָא קְרָבָא אִם נִתְמְנַע וַאֲמַר לֵיהּ סַק וְאַצְלַח וְיִמְסַר יְיָ בִּידָא דְמַלְכָּא:
וַאֲמַר לֵיהּ מַלְכָּא עַד כַמָא זִמְנִין אֲנָא מְקַיֵם עֲלָךְ דְלָא תְמַלֵיל לִי לְחוֹד קְשׁוֹט בִּשְׁמָא דַיָי:
וַאֲמַר חֲזֵיתִי יַת כָּל יִשְׂרָאֵל מְבַדְרִין עַל טוּרַיָא כַּעֲנָא דְלֵית לְהוֹן רָעִי וַאֲמַר יְיָ לֵית רִבּוֹנִין לְאִלֵין יְתוּבוּן גְבַר לְבֵיתֵיהּ בִּשְׁלָם:
וַאֲמַר מַלְכָּא דְיִשְׂרָאֵל לִיהוֹשָׁפָט הֲלָא אֲמַרִית לָךְ דְלָא יִתְנַבֵּי עֲלֵי טַב אֱלָהִין בִּישׁ:
וַאֲמַר בְּכֵן קַבֵּיל פִּתְגָמָא דַייָ חֲזֵיתִי יַת יְקָרָא דַייָ שְׁרֵי עַל כּוּרְסֵיהּ וְכָל חֵילֵי שְׁמַיָא קַיְמִין קֳדָמוֹהִי מִימִינֵיהּ וּמִסְמָלֵיהּ:
וַאֲמַר יְיָ מַן יַטְעֵי יַת אַחְאָב וְיִסַק וְיִפּוֹק בְּרָמוֹת גִלְעָד וַאֲמַר דֵין בִּכְדֵין וְדֵין אֲמַר בִּכְדֵין:
וּנְפַק רוּחָא וְקָם קֳדָם יְיָ וַאֲמַר אֲנָא אַטְעִינֵיהּ וַאֲמַר יְיָ לֵיהּ בְּמָא:
וַאֲמַר אֶפּוֹק וְאֶהֱוֵי לְרוּחַ שְׁקַר בְּפוּם כָּל נְבִיוֹהִי וַאֲמַר תַּטְעֵי וְאַף תִּכּוּל פּוּק וַעֲבֵיד כֵּן:
וּכְעַן הָא יְהַב יְיָ רוּחַ שְׁקַר בְּפוּם כָּל נְבִיָךְ אִלֵין וַיָי גְזַר לְאַיְתָאָה עֲלָךְ בִּישָׁא:
וּקְרֵיב צִדְקִיָהוּ בַּר כְּנַעֲנָה וּמְחָא יָת מִיכָיְהוּ עַל לְסָתֵיהּ וַאֲמַר אֵי דָא שָׁעָה אִסְתַּלֵק רוּחַ נְבוּאָה מִן קֳדָם יְיָ מִינִי לְמַלְלָא עִמָךְ:
וַאֲמַר מִיכָיְהוּ הָא אַתְּ חָזֵי בְּיוֹמָא הַהוּא דְאַתְּ עָלֵיל תָּוָן בְּגוֹ תָּוָן לְאִטַמָרָא:
וַאֲמַר מַלְכָּא דְיִשְׂרָאֵל דְבַר יַת מִיכָיְהוּ וַאֲתֵיבְהִי לְוַת אָמוֹן רַב קַרְתָּא וּלְוַת יוֹאָשׁ בַּר מַלְכָּא:
וְתֵימַר כִּדְנַן אֲמַר מַלְכָּא שַׁווּ יַת דֵין בְּבֵית אֲסִירַי וַאֲכִילוּהִי לַחְמָא בִּדְחוֹק וּמַיָא בִּדְחוֹק עַד מֵיתָאִי בִּשְׁלָם:
וַאֲמַר מִיכָיְהוּ אִם מֵיתַב תְּתוּב בִּשְׁלָם לָא רַעֲוָא קֳדָם יְיָ בִּי וַאֲמַר שְׁמָעוּ עַמְמַיָא כּוּלְהוֹן:
וּסְלֵיק מַלְכָּא דְיִשְׂרָאֵל וִיהוֹשָׁפָט מֶלֶךְ שִׁבְטָא דְבֵית יְהוּדָה רָמוֹת גִלְעָד:
וַאֲמַר מַלְכָּא דְיִשְׂרָאֵל לִיהוֹשָׁפָט אֲנָא אֶשְׁתַּנֵי וְאֵיעוֹל בִּקְרָבָא וְאַתְּ לְבַשׁ לְבוּשָׁךְ וְאִשְׁתַּנֵי מַלְכָּא דְיִשְׂרָאֵל וְעַל בִּקְרָבָא:
וּמַלְכָּא דַאֲרָם פַּקֵיד יַת רַבָּנֵי רְתִיכָא דִילֵיהּ תְּלָתִין וּתְרֵין מַלְכְּוָן לְמֵימַר לָא תַגִיחוּן קְרָבָא עִם זְעֵירָא וְעִם רַבָּא אֱלָהֵין עִם מַלְכָּא דְיִשְׂרָאֵל בִּלְחוֹדוֹהִי:
וַהֲוָה כַּד חֲזוֹ רַבָּנֵי רְתִיכַיָא יַת יְהוֹשָׁפָט וְאִינוּן אֲמַרוּ בְּרַם מַלְכָּא דְיִשְׂרָאֵל הוּא וְזָרוּ עֲלוֹהִי לְאַגָחָא קְרָבָא בֵיהּ וּצְוַח יְהוֹשָׁפָט:
וַהֲוָה כַּד חֲזּוֹ רַבָּנֵי רְתִיכַיָא אֲרֵי לָא מַלְכָּא דְיִשְׂרָאֵל הוּא וְתָבוּ מֵאֲחוֹרוֹהִי:
וְגַבְרָא נְגַד בְּקַשְׁתָּא לְקִבְלֵיהּ וּמְחָא יַת מַלְכָּא דְיִשְׂרָאֵל בֵּין דִבְקֵי וּבֵין שִׁרְיָנָא וַאֲמַר לְמַרְכַּבְנֵיהּ אִתְחֲזַר לַאֲחוֹרָךְ וְאַפֵּיקְנִי מִן מַשְׁרִיתָא אֲרֵי מְרַע אֲנָא:
וּסְלִיקוּ עָבְדֵי קְרָבָא בְּיוֹמָא הַהוּא וּמַלְכָּא הֲוָה מַתְקֵיף וְקָאִים בִּרְתִיכָא לִקֳבֵיל אֱנַשׁ אֲרָם וּמִית בְּרַמְשָׁא וְאִתְרִיק דְמָא דִמְחָתָא לְגוֹ רְתִיכָא:
וְאַעְבַרוּ כְּרוֹז בְּמַשְׁרִיתָא כְּמֵיעַל שִׁמְשָׁא לְמֵימַר גְבַר לְקַרְתֵּיהּ וּגְבַר לְאַרְעֵיהּ:
וּמִית מַלְכָּא וַאֲתָא לְשֹׁמְרוֹן וּקְבַרוּ יַת מַלְכָּא בְּשֹׁמְרוֹן:
וּשְׁטַף יַת רְתִיכַיָא עַל בְּרֵיכָתָא דְשֹׁמְרוֹן וּלְחָכוּ כַלְבַּיָא יַת דְמֵיהּ וּמָנֵי זְיָנָא שְׁטָפוּ כְּפִתְגָמָא דֵיָי דְמַלֵיל:
וּשְׁאָר פִּתְגָמֵי אַחְאָב וְכָל דַעֲבַד וּבֵית שִׁנָא דְפִיל דִי בְנָא וְכָל קִרְוַיָא דִבְנָא הֲלָא אִינוּן כְּתִיבִין עַל סְפַר פִּתְגָמֵי יוֹמַיָא לְמַלְכֵי יִשְׂרָאֵל:
וּשְׁכִיב אַחְאָב עִם אֲבָהָתוֹהִי וּמְלַךְ אֲחַזְיָהוּ בְרֵיהּ תְּחוֹתוֹהִי:
וִיהוֹשָׁפָט בַּר אָסָא מְלַךְ עַל דְבֵית יְהוּדָה בִּשְׁנַת אַרְבַּע לְאַחְאָב מַלְכָּא דְיִשְׂרָאֵל:
יְהוֹשָׁפָט בַּר תְּלָתִין וַחֲמֵשׁ שְׁנִין כַּד מְלַךְ וְעַשְׂרִין וַחֲמֵשׁ שְׁנִין מְלַךְ בִּירוּשְׁלֵם וְשׁוּם אִמֵיהּ עֲזוּבָה בַּת שִׁלְחִי:
וַאֲזַל בְּכָל אוֹרְחָא אָסָא אֲבוֹהִי לָא סְטָא מִנֵיהּ לְמֶעְבַּד דְכָשַׁר קֳדָם יְיָ:
בְּרַם בְּמָתָא לָא עֲטָרָא עַד כְּעַן עַמָא מְדַבְּחִין וּמַסְקִין בּוּסְמִין עַל בָּמָתָא:
וְאַשְׁלֵים יְהוֹשָׁפָט עִם מַלְכָּא דְיִשְׂרָאֵל:
וּשְׁאַר פִּתְגָמֵי יְהוֹשָׁפָט וּגְבוּרְתֵּיהּ דַעֲבַד וּדְאַגִיחַ קְרָבָא הֲלָא אִינוּן כְּתִיבִין עַל סְפַר פִּתְגָמֵי יוֹמִין לְמַלְכֵי דְבֵית יְהוּדָה:
וּשְׁאַר נַפְקַת בְּרָא דְאִשְׁתָּאַר בְּיוֹמֵי אָסָא אֲבוּהִי פַלֵי מִן אַרְעָא:
וּמַלְכָּא לֵית בֶּאֱדוֹם מְמַנָא אֱלָהֵין אִסְטַרְטִיגֵי מַלְכָּא:
יְהוֹשָׁפָט עֲבַד סְפִינַת אַפְרִיקָא לְמֵיזַל לְאוֹפִיר לְאַיְתָאָה דַהֲבָא וְלָא אֲזַל אַרֵי אִתְבָרוּ סְפִינְתָא בְעֶצְיוֹן גָבֶר:
בְּכֵן אֲמַר אֲחַזְיָהוּ בַּר אַחְאָב לִיהוֹשָׁפָט יֵיזְלוּן עַבְדַי עִם עַבְדָךְ בִּסְפִינָתָא וְלָא אַבָה יְהוֹשָׁפָט:
וּשְׁכִיב יְהוֹשָׁפָט עִם אֲבָהָתוֹהִי וְאִתְקְבַר עִם אֲבָהָתוֹהִי בְּקַרְתָּא דְדָוִד אֲבוּהִי וּמְלַךְ יְהוֹרָם בְּרֵיהּ תְּחוֹתוֹהִי:
אֲחַזְיָהוּ בַּר אַחְאָב מְלַךְ עַל יִשְׂרָאֵל בְּשֹׁמְרוֹן בִּשְׁנַת שְׁבַע עַסְרֵי לִיהוֹשָׁפָט מַלְכָּא שִׁבְטָא דְבֵית יְהוּדָה וּמְלַךְ עַל יִשְׂרָאֵל תַּרְתֵּין שְׁנִין:
וַעֲבַד דְבִישׁ קֳדָם יְיָ וַאֲזַל בְּאוֹרַח אֲבוּהִי וּבְאוֹרַח אִמֵיהּ וּבְאוֹרַח יָרָבְעָם בַּר נְבָט דְחַיֵב יַת יִשְׂרָאֵל:
וּפְלַח יַת בַּעֲלָא וּסְגִיד לֵיהּ וְאַרְגֵיז קֳדָם יְיָ אֱלָהָא דְיִשְׂרָאֵל כְּכָל דַעֲבַד אֲבוּהִי:
