שָׁרַת עֲלַי רוּחַ נְבוּאָה מִן קֳדָם יְיָ וְאַפְקַנִי בְּרוּחַ נְבוּאָה דִי שְׁרַת עֲלַי מִן קֳדָם יְיָ וְאַשְׁרַנִי בְּגוֹ בִקְעֲתָא וְהִיא מַלְיָא גַרְמֵי אֱנָשָׁא:
וְאַעְבְּרַנִי עֲלֵיהוֹן סְחוֹר סְחוֹר וְהָא סַגִיאָן לַחֲדָא עַל אַפֵּי בִקְעֲתָא וְהָא יְבֵישִׁין לַחֲדָא:
וַאֲמַר לִי בַּר אָדָם הֲיֵחוּן גַרְמַיָא הָאִלֵין וַאֲמָרֵית יְיָ אֱלֹהִים אַתְּ גְלֵי קֳדָמָךְ:
וַאֲמַר לִי אִתְנַבֵּי עַל גַרְמַיָא הָאִלֵין וְתֵימַר לְהוֹן גַרְמַיָא יְבֵישַׁיָא קַבִּילוּ פִּתְגָמָא דַיָי:
כִּדְנַן אֲמַר יְיָ אֱלֹהִים לְגַרְמַיָא הָאִלֵין הָא אֲנָא מָעֵיל בְּכוֹן רוּחָא וְתֵיחוּן:
וְאֶתֵּן עֲלֵיכוֹן גִידִין וְאַסֵיק עֲלֵיכוֹן בִּסְרָא וְאֶקְרוֹם עֲלֵיכוֹן מַשְׁכָא וְאֶתֵּן בְּכוֹן רוּחָא וְתֵיחוּן וְתֵידְעוּן אֲרֵי אֲנָא יְיָ:
וְאִתְנַבֵּיתִי כְּמָא דְאִתְפַּקְדֵית וַהֲוָה קַל בְּאִתְנַבָּיוּתִי וְהָא זִיעוּ וּקְרִיבוּ גַרְמַיָא גַרְמָא לְוַת חַבְרֵיהּ:
וַחֲזֵית וְהָא עֲלֵיהוֹן גִידִין וּבִסְרָא סְלֵיק וּקְרֵם עֲלֵיהוֹן מַשְׁכָא מִלְעֵילָא וְרוּחָא לֵית בְּהוֹן:
וַאֲמַר לִי אִתְנַבֵּי עַל רוּחָא אִתְנַבֵּי בַר אָדָם וְתֵימַר לְרוּחָא כִּדְנַן אֲמַר יְיָ אֱלֹהִים מֵאַרְבַּע רוּחַיָא אַתְיָא רוּחָא וְעוּלִי בִקְטִילַיָא הָאִלֵין וְיַחוּן:
וְאִתְנַבֵּיתִי כְּמָא דִי פַּקְדַנִי וְעָלַת בְּהוֹן רוּחָא וְיִחְיוּ וְקָמוּ עַל רִגְלֵיהוֹן מַשִׁרְיָן סַגִיאָן לַחֲדָא לַחֲדָא:
וַאֲמַר לִי בַּר אָדָם גַרְמַיָא הָאִלֵין כָּל בֵּית יִשְׂרָאֵל אִינוּן הָא אִינוּן אָמְרִין יָבִישׁוּ גַרְמָנָא פְּסַק סַבְרָנָא אֲבַדְנָא לָנָא:
בְּכֵן אִתְנַבֵּי וְתֵימַר לְהוֹן כִּדְנַן אֲמַר יְיָ אֱלֹהִים הָא אֲנָא פְּתָחֵית יַת קִבְרֵיכוֹן וְאֶסַק יַתְכוֹן מִקִבְרֵיכוֹן עַמִי וְאָעֵיל יַתְכוֹן לְאַרְעָא דְיִשְׂרָאֵל:
וְתֵידְעוּן אֲרֵי אֲנָא יְיָ כַּד אֶפְתַּח יַת קִבְרֵיכוֹן וְכַד אַסֵיק יַתְכוֹן מִגוֹ קִבְרֵיכוֹן עַמִי:
וְאֶתֵּן רוּחִי בְכוֹן וְתֵיחוּן וְאַשְׁרֵי יַתְכוֹן עַל אַרְעֲכוֹן וְתֵידְעוּן אֲרֵי אֲנָא יְיָ גְזָרִית בְּמֵימְרִי וַאֲקַיֵם אֲמַר יְיָ:
וַהֲוָה פִתְגָם נְבוּאָה מִן קֳדָם יְיָ עִמִי לְמֵימָר:
וְאַתְּ בַּר אָדָם סַב לָךְ לוּחָא חֲדָא וּכְתוֹב עֲלוֹהִי לְשִׁבְטָא דִיהוּדָה וְלִבְנֵי יִשְׂרָאֵל אַחֵיהוֹן וְסַב לוּחָא חֲדָא וּכְתוֹב עֲלוֹהִי לְשִׁבְטָא דְיוֹסֵף דִי הוּא שִׁבְטָא דְאֶפְרַיִם וְכָל בֵּית יִשְׂרָאֵל אֲחֵיהוֹן:
וְתַקְרֵיב יַתְהוֹן חַד לְוַת חַד לָךְ לְלוּחָא חַד וִיהוֹן לְחוֹד בִּידָךְ:
וְכַד יֵימְרוּן לָךְ בְּנֵי עַמָךְ לְמֵימָר הֲלָא תְחַוֵי לָנָא מָה אִלֵין לָךְ:
אִתְנַבֵּי לְהוֹן כִּדְנַן אֲמַר יְיָ אֱלֹהִים הָא אֲנָא מְקָרִיב יַת שִׁבְטָא דְיוֹסֵף דִי הוּא שִׁבְטָא דְאֶפְרַיִם וְשִׁבְטֵי יִשְׂרָאֵל אֲחֵיהוֹן וַאֲחַבֵּיר יַתְהוֹן עֲלוֹהִי יַת שִׁבְטָא דִיהוּדָה וְאַעְבְּדִינוּן לְעַמָא חַד וִיהוֹן חַד קֳדָמָי:
וִיהוֹן לוּחַיָא דִי תִכְתּוֹב עֲלֵיהוֹן בִּידָךְ לְעֵינֵיהוֹן:
וְאִתְנַבֵּי לְהוֹן כִּדְנַן אֲמַר יְיָ אֱלֹהִים הָא אֲנָא מְקָרֵיב יַת בְּנֵי יִשְׂרָאֵל מִבֵּינֵי עַמְמַיָא דְאִתְגְלִיאוּ לְתַמָן וְאֶכְנוֹשׁ יַתְהוֹן מִסְחוֹר סְחוֹר וְאָעֵיל יַתְהוֹן לְאַרְעֲהוֹן:
וְאַעְבֵּיד יַתְהוֹן לְעַמָא חָד בְּאַרְעָא בְּטוּר קוּדְשָׁא דְיִשְׂרָאֵל דְיִשְׂרָאֵל וּמַלְכָּא חַד יְהֵי לְכוּלְהוֹן לְמֶהֱוֵי מַלְכָּא וְלָא יְהוֹן עוֹד לִתְרֵין עַמְמִין וְלָא יִתְפַּלְגוּן עוֹד לְתַרְתֵּין מַלְכְּוָן עוֹד:
וְלָא יִסְתָּאֲבוּן עוֹד בְּמַעֲוָתְהוֹן וּבְשִׁקוּצֵיהוֹן וּבְכָל מֵירְדֵיהוֹן וְאֶפְרוֹק יַתְהוֹן מִכָּל מוֹתְבָנֵיהוֹן דְחָבוּ קֳדָמַי בְּהוֹן וְאַדְכֵּי יַתְהוֹן וִיהוֹן קֳדָמַי לְעַם וַאֲנָא אֶהֱוֵי לְהוֹן לֵאלָהָא:
וְעַבְדִי דָוִד מַלְכָּא עֲלֵיהוֹן וּפַרְנָס חַד יְהֵי לְכוּלְהוֹן וּבְדִינַי יְהָכוּן וּקְיָמַי יִטְרוּן וְיַעְבְּדוּן יַתְהוֹן:
וִיתִיבוּן עַל אַרְעָא דִיהָבִית לְעַבְדִי לְיַעֲקֹב דִיתִיבוּ בָהּ אֲבָהַתְכוֹן וְיַתְבוּן עֲלָהּ אִינוּן וּבְנֵיהוֹן וּבְנֵי בְנֵיהוֹן עַד עַלְמָא וְדָוִד עַבְדִי מַלְכָּא לְהוֹן לַעֲלָם:
וְאֶגְזַר לְהוֹן קְיַם שְׁלָם קְיַם עֲלַם יְהֵי עִמְהוֹן וֶאֱבָרְכִינוּן וְאַסְגֵי יַתְהוֹן וְאֶתֵּן יַת מַקְדְשִׁי בֵּינֵיהוֹן לַעֲלָם:
וְאַשְׁרֵי שְׁכִינְתִּי בֵּינֵיהוֹן וְאֶהֱוֵי לְהוֹן לֵאלָהָא וְאִינוּן יְהוֹן קֳדָמַי לְעָם:
וְיֵדְעוּן עַמְמַיָא אֲרֵי אֲנָא יְיָ מְקַדֵשׁ יַת יִשְׂרָאֵל בְּמֶהֱוֵי מַקְדְשִׁי בֵּינֵיהוֹן לְעַלָם:
