בִּשְׁנַת תַּרְתֵּי עַסְרֵי לְאָחָז מֶלֶךְ שִׁבְטָא דְבֵית יְהוּדָה מְלַךְ הוֹשֵׁעַ בַּר אֵלָה בְּשׁוֹמְרוֹן עַל יִשְׂרָאֵל תְּשַׁע שְׁנִין:
וַעֲבַד דְבִישׁ קֳדָם יְיָ לְחוֹד לָא כְמַלְכֵי יִשְׂרָאֵל דַהֲווֹ קֳדָמוֹהִי:
עֲלוֹהִי סְלִיק שַׁלְמַנְאֶסֶר מַלְכָּא דְאַתּוּר וַהֲוָה לֵיהּ הוֹשֵׁעַ עַבְדָא וַאֲתֵיב לֵיהּ תִּקְרוּבְתָּא:
וְאַשְׁכַּח מַלְכָּא דְאַתּוּר בְּהוֹשֵׁעַ מֵירְדָא דִי שְׁלַח אִזְגַדִין לְוַת סוֹא מַלְכָּא דְמִצְרַיִם וְלָא אַסֵיק תִּקְרוּבְתָּא לְמַלְכָּא דְאַתּוּר כִּשְׁנָא בִשְׁנָא וְאַחְדֵיהּ מַלְכָּא דְאַתּוּר וְאַסְרֵיהּ בְּבֵית אַסִירֵי:
וּסְלִיק מַלְכָּא דְאַתּוּר בְּכָל אַרְעָא וּסְלִיק לְשֹׁמְרוֹן וְצַר עֲלָהּ תְּלַת שְׁנִין:
בְּשַׁתָּא תְשִׁיעֲתָא לְהוֹשֵׁעַ כְּבַשׁ מַלְכָּא דְאַתּוּר יַת שֹׁמְרוֹן וְאַגְלֵי יַת יִשְׂרָאֵל לְאַתּוּר וְאוֹתִיב יַתְהוֹן בַּחְלַח וּבְחָבוֹר נְהַר גוֹזָן וְקִירְוֵי מָדָי:
וַהֲוָה כַד חָבוּ בְנֵי יִשְׂרָאֵל קֳדָם יְיָ אֱלָהָהוֹן דְאַסֵיק יַתְהוֹן מֵאַרְעָא דְמִצְרַיִם מִתְּחוֹת יַד פַּרְעֹה מַלְכָּא דְמִצְרַיִם וּדְחִילוּ טַעֲוַת עַמְמַיָא:
וַאֲזַלוּ בִגְזֵירַת עַמְמַיָא דְתָרִיךְ יְיָ מִן קֳדָם בְּנֵי יִשְׂרָאֵל וּמַלְכֵי יִשְׂרָאֵל דַעֲבָדוּ:
וַאֲמָרוּ בְנֵי יִשְׂרָאֵל מִלִין דְלָא כַשְׁרִין קֳדָם יְיָ אֱלָהָהוֹן וּבְנוֹ לְהוֹן בָּמָן בְכָל קִרְוֵיהוֹן מִמִגְדַל תַּקִיף עַד קִירְוִין כְּרִיכָן:
וַאֲקִימוּ לְהוֹן קָמָן וַאֲשֵׁירִין עַל כָּל רָמָא מְנַטְלָא וּתְחוֹת כָּל אִילָן עַבּוּף:
וְאַסִיקוּ בוּסְמִין תַּמָן עַל כָּל בָּמָתָא בְּעַמְמַיָא דְאַגְלֵי יְיָ מִן קֳדָמֵיהוֹן וַעֲבַדוּ קֳדָמוֹהִי פִּתְגָמִין בִּישִׁין לְאַרְגָזָא קֳדָם יְיָ:
וּפְלָחוּ יַת בְּעָלַיָא דִי אֲמַר יְיָ לְהוֹן לָא תַעַבְדוּן יַת פִּתְגָמָא הָדֵין:
וְאַסְהֵיד יְיָ בְּיִשְׂרָאֵל וּבִיהוּדָה בְיַד כָּל סָפַר כָּל מֵילַף לְמֵימָר תּוּבוּ מֵאוֹרְחַתְכוֹן בִּישָׁתָא וְטַרוּ פִּקוּדֵי וּקְיָמַי כְּכָל אוֹרַיְתָא דְפַקְדִית יַת אֲבָהַתְכוֹן וְדִשְׁלָחִית לְוַתְכוֹן בְּיַד עַבְדֵי נְבִיַיָא:
וְלָא קַבִּילוּ וְאַקְשִׁיאוּ יַת קְדַלְהוֹן כִּקְדָלָא דַאֲבָהַתְהוֹן דְלָא הֵימְנוּ בְּמֵימְרָא דַייָ אֱלָהָהוֹן:
וְקָצוּ בִקְיָמוֹהִי וּבִגְזֵירָתֵיהּ דִי גְזַר עִם אֲבָהַתְהוֹן וּבְסָהָדְוָתֵיהּ דְאַסְהִיד בְּהוֹן וּטְעוֹ בָּתַר טַעֲוָתָא וַהֲווֹ לִלְמָא וּבָתַר עַמְמַיָא דִי בְסַחְרָנֵיהוֹן דִי פַקֵיד יְיָ יַתְהוֹן בְּדִיל דְלָא לְמֶעְבַּד כְּוַתְהוֹן:
וּשְׁבַקוּ יַת כָּל פִקוּדַיָא דַייָ אֱלָהָהוֹן וַעֲבַדוּ לְהוֹן מַתְּכָא תְּרֵין עֶגְלִין וַעֲבָדוּ אֲשֵׁירָתָא וּסְגִידוּ לְכָל חֵילֵי שְׁמַיָא וּפְלַחוּ יַת בְּעָלָא:
וְאַעֲבָרוּ יַת בְּנֵיהוֹן וְיַת בְּנָתֵיהוֹן בְּנוּרָא וּקְסָמוּ קִסְמִין וּנְחִישׁוּ וַחֲשִׁיבוּ לְמֶעְבַּד דְבִישׁ קֳדָם יְיָ לְאַרְגָזָא קֳדָמוֹהִי:
וַהֲוָה רְגַז מִן קֳדָם יְיָ לַחֲדָא בְיִשְׂרָאֵל וְאַגְלִינוּן מֵאַרְעָא בֵית שְׁכִינְתֵּיהּ לָא אִשְׁתָּאַר לְחוֹד שִׁבְטָא דְבֵית יְהוּדָה בִּלְחוֹדוֹהִי:
אַף דְבֵית יְהוּדָה לָא נְטָרוּ יַת פִּקוּדַיָא דַייָ אֱלָהָהוֹן וַאֲזַלוּ בִּגְזֵירַת יִשְׂרָאֵל דַעֲבָדוּ:
וְקָץ יְיָ בְּכָל זַרְעָא דְיִשְׂרָאֵל וְעַנִינוּן וּמַסְרִינוּן בְּיַד בְּזוֹזִין עַד דְאַגְלִינוּן מֵאֲתַר בֵּית שְׁכִנְתֵּיהּ:
אֲרֵי אִתְפְּלִיגוּ בֵית יִשְׂרָאֵל עַל דְבֵית יְהוּדָה וְאַמְלִיכוּ יַת יָרָבְעָם בַּר נְבָט וְאַטְעֵי יָרָבְעָם יַת יִשְׂרָאֵל מִבָּתַר פּוּלְחָנָא דַייָ וְחַיְבִינוּן חוֹבָא רַבָּא:
וַאֲזַלוּ בְּנֵי יִשְׂרָאֵל בְּכָל חוֹבֵי יָרָבְעָם דַעֲבֵד לָא סְטוֹ מִנָהּ:
עַד דְאַגְלֵי יְיָ יַת יִשְׂרָאֵל מֵאַרְעָא בֵּית שְׁכִנְתֵּיהּ כְּמָא דְמַלֵל בְּיַד כָּל עַבְדוֹהִי נְבִיַיָא וּגְלוֹ יִשְׂרָאֵל מֵעַל אַרְעֲהוֹן לְאַתּוּר עַד יוֹמָא הָדֵין:
וְאַיְיתֵי מַלְכָּא דְאַתּוּר מִבָּבֶל וּמִכּוּתָה וּמֵעַוָה וּמֵחֲמָת וּמִסְפַרְוַיִם וְאוֹתִיב בְּקִירְוֵי שֹׁמְרוֹן חֲלַף בְּנֵי יִשְׂרָאֵל וִירִיתוּ יַת שֹׁמְרוֹן וִיתִיבוּ בְּקִירְוָהָא:
וַהֲוָה בִּשְׁרָיוּת לְמֵיתַבְהוֹן תַּמָן לָא הֲווֹ דַחֲלִין מִן קֳדָם יְיָ וְגָרֵי יְיָ בְּהוֹן יַת אַרְיְוָתָא וַהֲווֹ קַטְלִין בְּהוֹן:
וַאֲמַרוּ לְמַלְכָּא דְאַתּוּר לְמֵימָר עַמְמַיָא דְאַגְלֵיתָא וְאוֹתַבְתָּא בְּקִירְוֵי שֹׁמְרוֹן לָא הֲווֹ יָדְעִין יַת דִין אֱלָהָא דְאַרְעָא וְגָרֵי בְהוֹן יַת אַרְיְוָתָא וְהָא אִינוּן מְקַטְלִין יַתְהוֹן בִּדְלֵית אִינוּן יָדְעִין יַת דִין אֱלָהָא דְאַרְעָא:
וּפַקֵיד מַלְכָּא דְאַתּוּר לְמֵימָר אוֹבִילוּ לְתַמָן חַד מִכַּהֲנַיָא דְאַגְלְתִינוּן מִתַּמָן וְיֵזְלוּן וְיַתְבוּן תַּמָן וְיַלְפוּן יַת דִי אֱלָהָא דְאַרְעָא:
וַאֲתָא חַד מִכַּהֲנַיָא דְאִתְגְלִיאוּ מִשֹׁמְרוֹן וִיתֵיב בְּבֵית אֵל וַהֲוָה מַלֵיף יַתְהוֹן אֵכְדֵין יִדְחֲלוּן מִן קֳדָם יְיָ:
וַהֲווֹ עָבְדִין עַמָא עַמָא טַעֲוָתֵיהוֹן וַאֲחִיתוּ בְּבֵית בְּמָתָא דַעֲבָדוּ שֹׁמְרוֹנָאֵי עַמָא עַמָא בְּקִירְוֵיהוֹן דְאִינוּן יַתְבִין תַּמָן:
וֶאֱנָשֵׁי בָבֶל עֲבָדוּ יַת סֻכּוֹת בְּנוֹת וֶאֶנָשֵׁי כוּת עֲבַדוּ יַת נִרְגַל וֶאֱנָשֵׁי חֲמַת עֲבַדוּ יַת אֲשִׁימָא:
וַעֲוָאֵי עֲבַדוּ נִבְחַז וְיַת תַּרְתָּק וֶאֱנַשׁ סְפַרְוָאֵי מוֹקְדִין יַת בְּנֵיהוֹן בְּנוּרָא לְאַדְרַמֶלֶךְ וַעֲנַמֶלֶךְ טַעֲוַת סְפַרְוָיִם:
וַהֲווֹ דַחֲלִין מִן קֳדָם יְיָ וַעֲבַדוּ לְהוֹן מִקְצַתְהוֹן כּוּמָרֵי בָמָתָא וַהֲווֹ עָבְדִין לְהוֹן בְּבֵית בָּמָתָא:
מִן קֳדָם יְיָ הֲווֹ דַחֲלִין וְיַת טַעֲוַתְהוֹן הֲווֹ פַלְחִין כְּנִימוּסֵי עַמְמַיָא דְאַגְלִיאוּ יַתְהוֹן מִתַּמָן:
עַד יוֹמָא הָדֵין אִינוּן עָבְדִין כְּנִימוּסֵי קַדְמָאֵי לֵיתִינוּן דַחֲלִין מִן קֳדָם יְיָ וְלֵיתִינוּן עָבְדִין כִּגְזֵירַתְהוֹן וְכִדְחָזֵי לְהוֹן וּכְאוֹרַיְתָא וּכְתַפְקֵידְתָּא דְפַקֵיד יְיָ יַת בְּנֵי יַעֲקֹב דְשַׁוִי שְׁמֵיהּ יִשְׂרָאֵל:
וּגְזַר יְיָ עִמְהוֹן קְיָם וּפַקְדִינוּן לְמֵימָר לָא תִדְחֲלוּן לְטַעֲוַת עַמְמַיָא וְלָא תִסְגְדוּן לְהוֹן וְלָא תִפְלְחִנוּן וְלָא תְדַבְּחוּן לְהוֹן:
אֱלָהֵן קֳדָם יְיָ דְאַסֵיק יַתְכוֹן מֵאַרְעָא דְמִצְרַיִם בְּחֵיל רַב וּבִדְרַע מְרָמַם יָתֵיהּ תִדְחֲלוּן וּקְדָמוֹהִי תִסְגְדוּן וְעַל מַדְבְּחֵיהּ תְּדַבְחוּן:
וְיַת קְיָמַיָא וְיַת דִינַיָא וְאוֹרַיְתָא וְתַפְקֵידְתָּא דִכְתַב לְכוֹן תִּטְרוֹן לְמֶעְבַּד כָּל יוֹמַיָא וְלָא תִדְחֲלוּן לְטַעֲוַת עַמְמַיָא:
וּקְיָמָא דִגְזָרִית עִמְכוֹן לָא תִתְנְשׁוּן וְלָא תִדְחֲלוּן לְטַעֲוַת עַמְמַיָא:
אֱלָהֵין מִן קֳדָם יְיָ אֱלָהָכוֹן תִּדַחֲלוּן וְהוּא יְשֵׁיזֵיב יַתְכוֹן מִיַד כָּל בַּעֲלֵי דְבָבֵיכוֹן:
וְלָא קַבִּילוּ אֱלָהֵין כְּנִימוּסֵיהוֹן קַדְמָאֵי אִינוּן עָבְדִין:
וַהֲווֹ עַמְמַיָא הָאִילֵין דַחֲלִין מִן קֳדָם יְיָ וְיַת צַלְמָנֵיהוֹן הֲווֹ פַלְחִין אַף בְּנֵיהוֹן וּבְנֵי בְנֵיהוֹן כְּמָא דַעֲבָדוּ אֲבָהַתְהוֹן אִינוּן עָבְדִין עַד יוֹמָא הָדֵין:
