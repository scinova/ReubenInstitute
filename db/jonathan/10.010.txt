וּמַלְכַּת שְׁבָא שְׁמַעַת יַת שְׁמַע שְׁלֹמֹה לִשְׁמָא דַייָ וַאֲתַת לְנַסָיוּתֵיהּ בְּחוֹדָאִין:
וַאֲתַת לִירוּשְׁלֵם בְּמַשִׁרְיַן סַגִיאִין לַחֲדָא גַמְלִין טְעוּנִין בּוּסְמָנִין וְדַהֲבָא סַגִי לַחֲדָא וְאַבְנִין טָבָן וַאֲתַת לְוַת שְׁלֹמֹה וּמַלִילַת עִמֵיהּ יַת כָּל דִי הֲוָה עִם לִבָּהּ:
וְחַוִי לָהּ שְׁלֹמֹה יַת כָּל פִּתְגָמָהָא לָא הֲוָה מִדָעַם דְאִתְכַּסָא מִן מַלְכָּא דְלָא חַוִי לָהּ:
וַחֲזַת מַלְכַּת שְׁבָא יַת כָּל חוּכְמַת שְׁלֹמֹה וּבֵיתָא דִי בְנָא:
וּמֵיכַל פָּתוֹרֵיהּ וְאַסְחָרוּת עַבְדוֹהִי וּמְקָם מְשׁוּמְשָׁנוֹהִי וּלְבוּשֵׁיהוֹן וְשַׁקָיוּתֵיהּ וַעֲלָתֵיהּ דְמַסִיק בֵּית מַקְדְשָׁא דַייָ וְלָא אִשְׁתָּאֵר בָּהּ עוֹד רוּחַ:
וַאֲמַרַת לְמַלְכָּא קוּשְׁטָא הֲוָה פִּתְגָמָא דִי שְׁמָעִית בְּאַרְעִי עַל פִּתְגָמָךְ וְעַל חוּכְמָתָךְ:
וְלָא הֵימָנִית לְפִתְגָמַיָא עַד דַאֲתֵיתִי וַחֲזָאָה עֵינַי וְהָא לָא אִתְחַוָא לִי פַּלְגוּ מַה דַחֲזֵיתִי אוֹסַפְתָּא חוּכְמְתָא וְטַבְתָא עַל שְׁמוּעֲתָא דִי שְׁמָעִית:
טוּבֵי גַבְרָךְ טוּבֵי עַבְדָךְ אִלֵין דְקַיְמִין קֳדָמָךְ תְּדִירָא דְשָׁמְעִין יַת חוּכְמָתָךְ:
יְהִי יְיָ אֱלָהָךְ בְּרִיךְ דְאִתְרְעֵי בָךְ לְמִתְּנָךְ עַל כּוּרְסֵי מַלְכוּתָא דְיִשְׂרָאֵל בִּרְחִימַת יְיָ יַת יִשְׂרָאֵל לְעָלָם וּמַנְיָךְ לְמֶהֱוֵי מַלְכָּא לְמֶעְבַּד דִין דִקְשׁוֹט וּזְכוּ:
וִיהָבַת לְמַלְכָּא מְאָה וְעַסְרִין כִּכְּרִין דַהֲבָא וּבוּסְמָנִין סַגִי לַחֲדָא וְאַבְנִין טָבָן לָא אֲתָא כְּבוּסְמָנַיָא הָאִינוּן עוֹד לִסְגֵי דִי יַהֲבַת מַלְכַּת שְׁבָא לְמַלְכָּא שְׁלֹמֹה:
וְאַף סְפִינָתָא דְחִירָם דְמַיְתָא דַהֲבָא מֵאוֹפִיר אַיְתִיאַת מֵאוֹפִיר אֲעֵי אַלְמוֹגִין סַגִי לַחֲדָא וְאַבְנִין טָבָן:
וַעֲבַד מַלְכָּא יַת אֲעֵי אַלְמוֹגַיָא סָעִיד לְבֵית מַקְדְשָׁא דַייָ וּלְבֵית מַלְכָּא וְכִנוֹרַיָא וְנִבְלַיָא לִבְנֵי לֵוִי לָא אֲתוֹ כֵן אֲעֵי אַלְמוֹגִין וְלָא אִתְחֲזִיאוּ עַד יוֹמָא הָדֵין:
וּמַלְכָּא שְׁלֹמֹה יְהַב לְמַלְכַּת שְׁבָא יַת כָּל רְעוּתָהּ דִשְׁאֵלִית בַּר מִדְמַנֵי לָהּ כְּעוּתַר יְדָא דְמַלְכָּא שְׁלֹמֹה וְאִתְפְּנִיאַת וַאֲזַלַת לְאַרְעָהּ הִיא וְעַבְדָהָא:
וַהֲוָה מַתְקַל דַהֲבָא דִי הֲוָה אָתֵי לִשְׁלֹמֹה בְשַׁתָּא חֲדָא שִׁיתּ מְאָה שִׁתִּין וְשִׁיתּ כִּכְּרֵי דַהֲבָא:
בַּר מַאֲגַר אוּמָנַיָא וּסְחוֹרַת תַּגָרַיָא וְכָל מַלְכֵי סוּמְכְוָתָא וְשִׁלְטוֹנֵי אַרְעָא:
וַעֲבַד מַלְכָּא שְׁלֹמֹה מָאתָן תְּרֵיסִין דַהֲבָא טָבָא שִׁית מְאָה דַהֲבָא מַסִיק עַל תְּרֵיסָא חֲדָא:
וּתְלַת מְאָה מָגִנִין דַהֲבָא טָבָא תְּלָתָא מָנָן דַהֲבָא מַסֵיק עַל מָגִינָא חֲדָא וּמַנִינוּן מַלְכָּא בֵּית מְקִירַת מַלְכַיָא:
וַעֲבַד מַלְכָּא כּוּרְסַיָא דְשִׁינָא רַבָּא וַחֲפָהִי דַהֲבָא טָבָא:
שִׁיתָּא דַרְגִין לְכוּרְסַיָא וְרֵישׁ סְגַלְגַל לְכוּרְסַיָא מֵאַחֲרוֹהִי וְאַשְׁדָתָא מִכָּא וּמִכָּא עַל אֲתַר מוֹתָבָא וּתְרֵין אַרְיְוָן קַיְמִין בִּסְטַר אַשְׁדָתָא:
וּתְרֵי עֲשַׂר אַרְיְוָן קַיְמִין תַּמָן עַל שִׁתָּא דַרְגִין מִכָּא וּמִכָּא לָא אִתְעֲבַד כֵּן לְכָל מַלְכְּוָתָא:
וְכָל מָנֵי מִשְׁתַּיָא דְמַלְכָּא שְׁלֹמֹה דַהֲבָא וְכָל מָנֵי בֵּית מְקֵרַת מַלְכָּא דַהֲבָא טָבָא לֵית כַּסְפָּא לָא אִתְחֲשִׁיב בְּיוֹמֵי שְׁלֹמֹה לְמִדָעַם:
אֲרֵי סְפִינָתָא אַפְרִיקָא לְמַלְכָּא בְּיַמָא עִם סְפִינָתָא דְחִירָם חֲדָא לִתְלַת שְׁנִין אַתְיָא סְפִינַת אַפְרִיקָא טְעִינָא דְהַב וּכְסַף שֵׁן דְפִיל וְקוֹפִין וְטַוָסִין:
וּרְבָא מַלְכָּא שְׁלֹמֹה מִכָּל מַלְכֵי אַרְעָא לְעוֹתֵר וּלְחוּכְמָא:
וְכָל דַיְיָרֵי אַרְעָא בָּעָן לְמֶחֱוֵי יַת סְבַר אַפֵּי שְׁלֹמֹה לְמִשְׁמַע יַת חוּכְמְתֵיהּ דִיהַב יְיָ בְּלִבֵּיהּ:
וְאִינוּן מַיְתַן גְבַר קוּרְבָּנֵיהּ מָנֵי כַסְפָא וּמָנֵי דַהֲבָא וּלְבוּשִׁין וּמָנֵי זְיָנָא וּבוּסְמָנִין סוּסְוָן וְכוּדַנְוָן גְזֵירַת שְׁנָא בִּשְׁנָא:
וּכְנַשׁ שְׁלֹמֹה רְתִיכִין וּפָרָשִׁין וַהֲווֹ לֵיהּ אֶלֶף וְאַרְבַּע מְאָה רְתִיכִין וּתְרֵי עֲסַר אַלְפִין פָּרָשִׁין וְאַשְׁרִינוּן בְּקִירְוֵי רְתִיכַיָא בַּר מִדַהֲווֹ עִם מַלְכָּא בִּירוּשְׁלֵם:
וִיהַב מַלְכָּא יַת כַּסְפָּא בִּירוּשְׁלֵם כְּאַבְנַיָא וְיַת אַרְזַיָא יְהַב כְּשִׁקְמַיָא דְבִשְׁפֶלְתָּא לִסְגֵי:
וּמִתַּפְקִין סוּסְוָתָא דְלִשְׁלֹמֹה מִמִצְרָיִם וּמִקְוֵה תַּגְרֵי מַלְכָּא זַבְּנִין מִקְוֵה בְּדָמִין:
וְסַלְקָא וְנַפְקָא רְתִיכָא מִמִצְרַיִם בְּשִׁית מְאָה כַסְפָּא וְסוּסְוָתָא בְּמֵאָה וְחַמְשִׁין וְכֵן לְכָל מַלְכֵי חִתָּאֵי וּמַלְכֵי אֲרַם בִּרְעוּתְהוֹן נַפְקִין:
