וּסְלִיק נְבִיָא בִּשְׁלִיחוּת מִן קֳדָם יְיָ מִן גִלְגְלָא לַבּוֹכִים וַאֲמַר אֲסֵקִית יַתְכוֹן מִמִצְרַיִם וְאַעֲלִית יַתְכוֹן לְאַרְעָא דִי קַיֵמִית לַאֲבָהַתְכוֹן וַאֲמַרִית לָא אַשְׁנֵי קְיָמַי עִמְכוֹן לְעֳלָם:
וְאַתּוּן לָא תִגְזְרוּן קְיָם לְיַתְבֵי אַרְעָא הָדָא אֲגוֹרֵיהוֹן תְּתָרְעוּן וְלָא קַבֵּלְתּוּן לְמֵימְרִי מָא דָא עֲבַדְתּוּן:
וְאַף אַמְרֵית לָא אֲתָרֵיךְ יַתְהוֹן מִן קֳדָמֵיכוֹן וִיהוֹן לְכוֹן לִמְעִיקִין וְטַעֲוַתְהוֹן יְהוֹן לְכוֹן לְתַקָלָא:
וַהֲוָה כַּד מַלֵיל נְבִיָא דַייָ יַת פִּתְגָמַיָא הָאִלֵין לְכָל בְּנֵי יִשְׂרָאֵל וַאֲרִימוּ עַמָא יַת קַלְהוֹן וּבְכוֹ:
וּקְרוֹ שְׁמָא דְאַתְרָא הַהוּא בּוֹכִים וּנְכִיסוּ תַמָן נִכְסַת קוּדְשַׁיָא קֳדָם דַיָי:
וּשְׁלַח יְהוֹשֻׁעַ יַת עַמָא וַאֲזַלוּ בְּנֵי יִשְׂרָאֵל גְבַר לְאַחְסַנְתֵּיה לְמֵירַת יַת אַרְעָא:
וּפְלַח עַמָא קֳדָם יְיָ כָּל יוֹמֵי יְהוֹשֻׁעַ וְכֹל יוֹמֵי סָבַיָא דִי אוֹרִיכוּ יוֹמִין בָּתַר יְהוֹשֻׁעַ דַחֲזוֹ יַת כָּל עוֹבָדָא דַייָ רַבָּא דִי עֲבַד לְיִשְׂרָאֵל:
וּמִית יְהוֹשֻׁעַ בַּר נוּן עַבְדָא דַייָ בַּר מְאָה וַעֲשַׂר שְׁנִין:
וּקְבַרוּ יָתֵיהּ בִּתְחוּם אַחְסַנְתֵּיהּ בְּתִמְנַת חֶרֶס בְּטוּרָא דְבֵית אֶפְרַיִם מִצִיפּוּנָא לְטוּר גָעַשׁ:
וְאַף כָּל דָרָא הַהוּא אִתְכְּנִישׁוּ לְוַת אֲבָהָתוֹהִי וְקָם דָרָא אוֹחֲרָנָא בַּתְרֵיהוֹן דְלָא יַדְעִין לְמִדְחַל מִן קֳדָם יְיָ וְאַף יַת עוֹבָדָא דִי עֲבַד לְיִשְׂרָאֵל:
וַעֲבַדוּ בְּנֵי יִשְׂרָאֵל יַת דְבִישׁ קֳדָם יְיָ וּפְלַחוּ יַת בַּעֲלַיָא:
וּשְׁבַקוּ יַת פּוּלְחָנָא דַייָ אֱלָהָא דַאַבָהַתְהוֹן דְאַפֵּיק יַתְהוֹן מֵאַרְעָא דְמִצְרַיִם וּטְעוֹ בָּתַר טַעֲוַת עַמְמַיָא מִטַעֲוַת עַמְמַיָא דִי בְסַחֲרָנֵיהוֹן וּסְגִידוּ לָהֶן וְאַרְגִיזוּ קֳדָם יְיָ:
וּשְׁבַקוּ יַת פּוּלְחָנָא דַייָ וּפְלַחוּ לְבַעֲלָא וּלְעַשְׁתָּרָתָא:
וּתְקֵף רוּגְזָא דַייָ בְּיִשְׂרָאֵל וּמַסְרִינוּן בְּיַד בוֹזְזֵיהֶן וּבְזוֹ יַתְהוֹן וּמַסְרִינוּן בְּיַד בַּעֲלֵי דְבָבֵיהוֹן מִסְחוֹר סְחוֹר וְלָא יָכִילוּ עוֹד לְמֵיקָם קֳדָם בַּעֲלֵי דְבָבֵיהוֹן:
בְּכָל אֲתַר דִי נְפַקוּ מָחָא מִן קֳדָם יְיָ הֲוַת בְּהוֹן לְבִישָׁא כְּמָא דְמַלֵיל יְיָ וּכְמָא דְקַיֵם יְיָ לְהוֹן וְעָקַת לְהוֹן לַחֲדָא:
וַאֲקֵים יְיָ נְגִידִין וּפַרְקִינוּן מִיַד בָּזְזֵיהוֹן:
וְאַף מִן נְגִידֵיהוֹן לָא קַבִּילוּ אֲרֵי טָעוּ בָּתַר טַעֲוַת עַמְמַיָא וּסְגִידוּ לְהוֹן סָטוּ בִּפְרִיעַ מִן אוֹרְחָא דַאֲזַלוּ אֲבָהַתְהוֹן לְקַבָּלָא פִּקוּדַיָא דַייָ לָא עֲבַדוּ כֵן:
וַאֲרֵי אָקֵים יְיָ לְהוֹן נְגִידִין וַהֲוָה מֵימְרָא דַייָ בְּסַעֲדֵיהּ דִנְגִידָא וּפָרִיק לְהוֹן מִיַד בַּעֲלֵי דְבָבֵיהוֹן כָּל יוֹמֵי נְגִידָא אֲרֵי מְתִיב מִמֵימְרָא דַאֲמַר יְיָ וּמְקַבֵּל צְלוֹתְהוֹן וּפָרִיק לְהוֹן מִן קֳדָם דַחֲקֵיהוֹן וּמְשַׁעְבְּדֵיהוֹן:
וַהֲוָה כַּד מָאִית נְגִידָא תַּיְבִין וּמְחַבְּלִין מֵאֲבָהַתְהוֹן לִמְהַךְ בָּתַר טַעֲוַת עַמְמַיָא לְמִפְלַחְהֶן וּלְמִסְגַד לָהֶן לָא שַׁבְקִין מֵעוֹבְדֵיהוֹן בִּישַׁיָא וּמֵאוֹרְחַתְהוֹן מְקַלְקְלָתָא:
וּתְקֵיף רוּגְזָא דַייָ בְּיִשְׂרָאֵל וַאֲמַר חֲלַף דִי עֲבָרוּ עַמָא הָדֵין עַל קַיָמִי דִי פַקֵדִית יַת אֲבָהַתְהוֹן וְלָא קַבִּילוּ לְמֵימְרִי:
אַף אֲנָא לָא אוֹסִיף לְתָרָכָא אֱנָשׁ מִן קֳדָמֵיהוֹן מִן עַמְמַיָא דִי שְׁבַק יְהוֹשֻׁעַ וּמִית:
בְּדִיל לְנַסָאָה בְּהוֹן יַת יִשְׂרָאֵל הֲנַטְרִין אִינוּן יַת אוֹרְחָן דְתַקְנַן קֳדָם יְיָ לִמְהַךְ בְּהוֹן כְּמָא דִי נְטַרוּ אֲבָהַתְהוֹן אִם לָא:
וּשְׁבַק יְיָ יַת כָּל עַמְמַיָא הָאִלֵין בְּדִיל דְלָא לְתָרָכוּתְהוֹן בִּפְרִיעַ וְלָא מַסְרִינוּן בִידָא דִיהוֹשֻׁעַ:
