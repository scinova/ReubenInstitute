יֶחְדוּן דְיַתְבִין בְּמַדְבְּרָא בְּאַרְעָא צִיחַיָא וְיִבְעוּן דְשָׁרָן בְּמֵישְׁרָא וְיִזְהֲרוּן כְּשׁוֹשַׁנְיָא:
מֵידָץ יְדוּצוּן וְיֶחְדוּן אַף בִּיעַ וְחֶדְוָא יְקַר לִבְנָן יִתְיְהֵיב לְהוֹן זִיו כַּרְמְלָא וְשָׁרוֹנָא בֵּית יִשְׂרָאֵל דִלְהוֹן אָמְרִין אִלֵין אִינוּן יֶחֱזוּן יְקָרָא דַייָ זִיו תּוּשְׁבַּחְתָּא דֶאֱלָהָנָא:
אֲמַר נְבִיָא תַּקִיפוּ יָדִין דִמְרַשְׁלָן וּרְכוּבִין דִרָעֲלָן חֲסִינוּ:
אֲמַרוּ לְדָמוֹחָן בְּלִבְּהוֹן לְמֶעְבַּד אוֹרַיְתָא תְּקָפוּ וְלָא תִדַחֲלוּן הָא אֶלָהָכוֹן לְמֶעְבַּד פּוּרְעֲנוּת דִין אִתְגְלֵי מָרֵי גְמַלְיָא יְיָ הִיא יִתְגְלֵי וְיִפְרְקִינְכוֹן:
בְּכֵן יִתְפַּתְּחָן עֵינֵי בֵּית יִשְׂרָאֵל דְאִינוּן כְּסָמָן מִן אוֹרַיְתָא וְאוּדְנֵיהוֹן דִכְּחַרְשִׁין לְקַבָּלָא לְמִלֵי נְבִיָא יְצִיתָן:
בְּכֵן בְּמֶחְזְהוֹן גַלְוָתָא דְיִשְׂרָאֵל דְמִתְכַּנְשִׁין וְסַלְקָן לְאַרְעֲהוֹן הָא כְּאַיָלִין קַלִילִין וְלָא לְאִתְעַכָּבָא וִישַׁבַּח לִישָׁנְהוֹן דַהֲוָה בְּלִים אֲרֵי בְּכֵן אִתְבְּזָעוּ בְמַדְבְּרָא מַיָא וְנַחֲלַיָא בְּמֵישְׁרָא:
וִיהֵי שְׁרוֹבָא לַאֲגַמִין דְמַיָן וּבֵית צָחְוָנָא לְמַבּוּעֵי מַיָא אֲתַר דַהֲוָה יְרוֹדִין שָׁרְיָן תַּמָן יִסְקוּן קָנֶה וְגוֹמָא:
וִיהֵי תַמָן כֵּבַשׁ אוֹרַח וְתִקוּן אוֹרְחָא דְקוּדְשָׁא יִתְקְרֵי לָהּ לָא יֶעְדוּן בֵּיהּ מְסָאֲבִין וְלָא יִפְסְקוּן עָדֵי אוֹרְחָא וּדְלָא אֲלִיפוּ לָא יִטְעוּן:
לָא יְהֵי תַמָן מֶלֶךְ מַבְאֵישׁ וְשִׁלְטוֹן מְעִיק לָא יְגוֹזִינֵהּ וְלָא יִשְׁתַּכְּחוּן תַּמָן וִיהָכוּן פְרִיקִין:
וּפְרִיקַיָא דַייָ יְתוּבוּן וְיַתְכַּנְשׁוּן מִבֵּינֵי גַלְוַתְהוֹן וְיָעַלוּן לְצִיוֹן בְּתוּשְׁבַּחְתָּא וְחֶדְוַת עֲלַם תְּהֵי לְהוֹן דִי לָא פַסְקָא וַעֲנַן יְקָר תְּהֵי מַטַל עַל רֵישֵׁיהוֹן בִּיעַ וְחֶדְוָא יִשְׁכְּחוּן וִיסוּף מִנְהוֹן מִבֵּית יִשְׂרָאֵל דְוָנָא וְתֵינָחָתָא:
