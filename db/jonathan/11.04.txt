וְאִתְּתָא חֲדָא מִנְשֵׁי תַלְמִידֵי נְבִיַיָא מְצַוְחָא קֳדָם אֱלִישָׁע לְמֵימַר עַבְדָךְ עוֹבַדְיָה בַעֲלִי מִית וְאַתְּ יְדַעַתְּ אֲרֵי עַבְדָךְ הֲוָה דָחִיל מִן קֳדָם יְיָ כַּד קַטְלַת אִיזֶבֶל יַת נְבִיַיָא דַייָ דְבַר מִנְהוֹן מְאָה גַבְרִין וְאַטְמְרִינוּן חַמְשִׁין חַמְשִׁין גַבְרִין בִּמְעַרְתָּא וַהֲוָה יָזִיף וּמוֹכֵל לְהוֹן בְּדִיל דְלָא לְאוֹכָלָא יַתְהוֹן מִנִכְסוֹהִי דְאַחְאָב מִן קֳדָם דְאִינוּן אוֹנְסָא וּכְעַן רַשְׁיָא אֲתָא לְמִדְבַּר יַת תְּרֵין בְּנֵי לֵיהּ לְעַבְדִין:
וַאֲמַר לָהּ אֱלִישָׁע מָה אַעְבֵּיד לִיךְ חֲוִי לִי מָא אִית לִיךְ בְבֵיתָא וַאֲַמַרַת לֵית לְאַמְתָךְ כָּל מִדַעַם בְּבֵיתָא אֱלָהֵין מָנָא דְמִשְׁחָא:
וַאֲמַר אֱזִילִי שְׁאִילִי לִיךְ מָנִין מִן בָּרָא מִן כָּל שְׁבַבַיְכִי מָנִין רֵיקָנִין לָא תַזְעִירִין:
וְתַהֲכִין וְתַגִיפִין דָשָׁא בְאַפִּיךְ וּבְאַפֵּי בְּרָךְ וּתְרִיקִין עַל כָּל מָנַיָא הָאִלֵין וּמָנָא דְאִתְמְלִי תְּסַלְקִין:
וַאֲזַלַת מִלְוָתֵיהּ וַאֲגָפַת דָשָׁא בְּאַפָּהָא וּבְאַפֵּי בְנָהָא אִינוּן מְקַרְבִין לָהּ מָנַיָא וְהִיא מְרִיקָא מִשְׁחָא:
וַהֲוָה כְּמִשְׁלַם מָנַיָא וַאֲמַרַת לִבְרָהּ קְרִיב לִי עוֹד מָנָא וַאֲמַר לָהּ לֵית עוֹד מָנָא וּפְסַק מִשְׁחָא:
וַאֲתַת וְחַוִיאַת לִנְבִיָא דַייָ וַאֲמַר אֱזִילִי זַבִּינִי יַת מִשְׁחָא וְשַׁלְמִי לְמָרֵי חוֹבְתִיךְ וְאַתְּ וּבְנַיִיךְ תִּתְפַּרְנְסִין מִדְיִשְׁתָּאר:
וַהֲוָה יוֹמָא וַעֲבַר אֱלִישָׁע לְשׁוּנֵם וְתַמָן אִתְּתָא דַחֲלַת חֶטְאִין וְאַתְקִיפַת בֵּיהּ לְמֵיכַל לַחְמָא וַהֲוָה בִּזְמַן מְעִידוֹהִי מִתְפְּנֵי לְתַמָן לְמֵיכַל לַחְמָא:
וַאֲמַרַת לְבַעֲלָהּ הָא כְעַן יְדַעְנָא אֲרֵי נְבִיָא דַייָ קַדִישָׁא הוּא מִתְפְּנֵי לְוָתָנָא תְּדִירָא:
נַעְבֵּיד כְּעַן עֲלִיַת כָּתְלָא זְעֵירָא וּנְתַקֵן לֵיהּ תַּמָן אֲתַר עַרְסָא וּפְתוֹרָא וְכוּרְסַיָא וּמְנַרְתָּא וִיהֵי בְּמֵיתוֹהִי לְוָתָנָא יִתְפְּנֵי לְתַמָן:
וַהֲוָה יוֹמָא וְעַל לְתַמָן וְזָר לְעִילִיתָא וּשְׁכִיב תַּמָן:
וַאֲמַר לְגֵיחֲזֵי עוּלֵימֵיהּ קְרִי לְשׁוּנַמִיתָא הָדָא וּקְרָא לָהּ וְקָמַת קֳדָמוֹהִי:
וַאֲמַר לֵיהּ אֱמַר כְּעַן לָהּ הָא עַסֵקְתָּא לָנָא יַת כָּל עִסְקָא הָדָא מֶה לְמֶעְבַּד לִיךְ הֲפִתְגַם אִית לִיךְ לְמַלָלָא עִם מַלְכָּא אוֹ עִם רַב חֵילָא וַאֲמַרַת בְּגוֹ עִסְקֵי עַמִי אֲנָא מְסוֹבָרָא:
וַאֲמַר וּמָה לְמֶעְבַּד לָהּ וַאֲמַר גֵיחֲזִי בְּקוּשְׁטָא בַּר לֵית לָהּ וּבַעֲלָהּ סִיב:
וַאֲמַר קְרֵי לָהּ וּקְרָא לָהּ וְקָמַת בְּתַרְעָא:
וַאֲמַר לְזִמְנָא הָדֵין בְּעִדַן דְאַתּוּן קַיָמִין תְּהֵא מְחַבְּקָא בְרָא וַאֲמַרַת בְּבָעוּ רִבּוֹנִי נְבִיָא דַייָ לָא יִתְכַּדֵיב פִּתְגָמָךְ בְאַמְתָךְ:
וְעַדִיאַת אִתְּתָא וִילֵידַת בָּר לְזִמְנָא הָדֵין כְּעִדַן דְהִיא קַיָמָא דִי מַלֵיל עִמָהּ אֱלִישָׁע:
וּרְבָא רַבְיָא וַהֲוָה יוֹמָא וּנְפַק לְוַת אֲבוּהִי לְוַת חַצְדַיָא:
וַאֲמַר לַאֲבוּהִי רֵישִׁי רֵישִׁי וַאֲמַר לְעוּלֵימָא סַבְהִי אוֹבִילֵיהּ לְאִמֵיהּ:
וְנַסְבֵיהּ וְאוֹבִילֵיהּ לְאִמֵיהּ וִיתֵיב עַל בִּרְכָּהָא עַד עִדַן טִהֲרָא וּמִית:
וּסְלֵיקַת וְאַשְׁכַּבְתֵּיהּ עַל עַרְסָא דִנְבִיָא דַייָ וְאַגָפַת בְּאַפּוֹהִי וּנְפָקַת:
וּקְרַת לְבַעֲלָהּ וַאֲמַרַת שַׁדַר כְּעַן לִי חַד מִן עוּלֵימַיָא וַחֲדָא מִן אָתְנַיָא וְאִתְמְטֵי לְוַת נְבִיָא דַייָ וְאָתוּב:
וַאֲמַר מָא דֵין אַתְּ אֲזַלְתָּ לְוָתֵיהּ יוֹמָא דֵין לָא יְרַח וְלָא שַׁבְּתָּא וַאֲמַרַת שְׁלָם:
וְזָרֵיזַת אֲתָנָא וַאֲמַרַת לְעוּלֵמָהָא דְבַר וֶאֱזֵיל לָא תִדְחוֹק עֲלַי לְמִרְכַּב אֱלָהֵין כַּד אֵמַר לָךְ:
וַאֲזַלַת וַאֲתַת לְוַת נְבִיָא דַייָ לְטוּר כַּרְמְלָא וַהֲוָה כַּד חֲזָא נְבִיָא דַייָ יָתָהּ מְקֳבֵיל וַאֲמַר לְגֵיחֲזִי עוּלֵימֵיהּ הָא שׁוּנַמִית דֵיכִי:
כְּעַן רְהוּט כְּעַן לָקֳדָמוּתָהּ וַאֲמַר לָהּ הַשְׁלָם לִיךְ הַשְׁלָם לְבַעֲלִיךְ הַשְׁלָם לְרַבְיָא וַאֲמַרַת שְׁלָם:
וַאֲתַת לְוַת נְבִיָא דַייָ לְטוּרָא וְאַתְקִיפַת בְּרִגְלוֹהִי וּקְרֵיב גֵיחֲזִי לְמִידְחָהּ וַאֲמַר נְבִיָא דַייָ שְׁבוֹק מִינָהּ אֲרֵי נַפְשָׁא מְרִירָא לָהּ וּמִן קֳדָם יְיָ אִתְכַּסָא מִנִי וְלָא אִתְחַוָה לִי:
וַאֲמַרַת הַשְׁאֵילִית בְּרָא מִן רִבּוֹנִי הֲלָא אֲמַרִית לָךְ אִם מִתְיָהֵב לִי בַּר קַיָם אִם לָא לָא תְנַסֵיס יָתִי:
וַאֲמַר לְגֵיחֲזִי זְרִיז חַרְצָךְ וְסַב חוּטְרֵי בִידָךְ וְתֵיזִיל אֲרֵי תִשְׁכַּח גְבַר לָא תִשְׁאַל בִּשְׁלָמֵיהּ וַאֲרֵי יִשְׁאַל בִּשְׁלָמָךְ גַבְרָא לָא תְתֵבִינֵיהּ וּתְשַׁוֵי חוּטְרֵי עַל אַפֵּי רַבְיָא:
וַאֲמַרַת אִמֵיהּ דְרַבְיָא קַיָם הוּא יְיָ וְחַי נַפְשָׁךְ אִם אֶשְׁבְּקִינָךְ וְקָם וַאֲזַל בַּתְרָאָה:
וְגֵיחֲזִי עֲבַר קֳדָמֵיהוֹן וְשַׁוֵי יַת חוּטְרָא עַל אַפֵּי רַבְיָא וְלֵית קָל וְלֵית דְמָצִית וְתַב לָקֳדָמוּתֵיהּ וְחַוֵי לֵיהּ לְמֵימַר לָא אִתְעַר רַבְיָא:
וַאֲתָא אֱלִישָׁע לְבֵיתָא וְהָא רַבְיָא מִית רְמֵי עַל עַרְסֵיהּ:
וְעַל וְאַגַף דָשָׁא בְּאַפֵּי תַרְוֵיהוֹן וְצַלִי קֳדָם יְיָ:
וּסְלִיק וּשָׁכִיב עַל רַבְיָא וְשַׁוִי פוּמֵיהּ עַל פוּמֵיהּ וְעֵינוֹהִי עַל עֵינוֹהִי וִידוֹהִי עַל יְדוֹהִי וְאַלְהֵי עֲלוֹהִי וְחַם בִּיסְרֵיהּ דְרַבְיָא:
וְתַב וְהַלֵיךְ בְּבֵיתָא זִמְנָא חֲדָא לְכָא וְזִמְנָא חֲדָא לְכָא וּסְלִיק וְאַלְהֵי עֲלוֹהִי וְאִתְמוֹרַק רַבְיָא עַד שְׁבַע זִמְנִין וּפְקַח רַבְיָא יַת עֵינוֹהִי:
וּקְרָא לְגֵיחֲזִי וַאֲמַר קְרֵי לְשֻׁנַמִיתָא הֲדָא וּקְרָא לָהּ וַאֲתַת לָקֳדָמוֹהִי וַאֲמַר סָבִי בְרִיךְ:
וַאֲתַת וּנְפָלַת קֳדָם רַגְלוֹהִי וּסְגִידַת עַל אַרְעָא וּנְסֵיבַת יַת בְּרָהּ וּנְפָקַת:
וֶאֱלִישָׁע תַּב לְגִלְגָלָא וְכַפְנָא בְּאַרְעָא וְתַלְמִידֵי נְבִיַיָא יַתְבִין קֳדָמוֹהִי וַאֲמַר לְעוּלֵימֵיהּ תְּפֵי דוּדָא רַבָּא וּבַשֵׁיל תַּבְשִׁילָא לְתַלְמִידֵי נְבִיַיָא:
וּנְפַק חָד לְחַקְלָא לְלַקְטָא יֵרוֹקָנִין וְאַשְׁכַּח גוּפְנָא בְּחַקְלָא וְלַקֵיט מִינֵיהּ פִּקְעֵי חַקְלָא מְלֵי לְבוּשֵׁיהּ וַאֲתָא וּפְרַס וּרְמָא לְדוּדָא דְתַבְשִׁלָא אֲרֵי לָא יְדָעוּ:
וַאֲרִיקוּ לְגַבְרַיָא לְמֵיכַל וַהֲוָה כְּמֵיכַלְהוֹן מִתַּבְשִׁילָא וְאִינוּן צְוָחוּ וַאֲמַרוּ מוֹתָא בְדוּדָא נְבִיָא דַייָ וְלָא יְכִילוּ לְמֵיכַל:
וַאֲמַר אַיְתוֹ קִמְחָא וּרְמוֹ לְדוּדָא וַאֲמַר אֲרֵיק לְעַמָא וְיֵיכְלוּן וְלָא הֲוָה מִדַעַם בִּישׁ בְּדוּדָא:
וְגַבְרָא אָתָא מֵאֲרַע דָרוֹמָא וְאַיְתֵי לִנְבִיָא דַייָ לְחֵם בִּכּוּרִין עַסְרִין טוּלְמִין דִלְחֵם סְעוֹרִין וּפֵירוּכִן בִּלְבוּשֵׁיהּ וַאֲמַר הַב לְעַמָא וְיֵיכְלוּן:
וַאֲמַר מְשׁוּמְשָׁנֵיהּ מָא אֶתֵּן דֵין קֳדָם מְאָה גַבְרָא וַאֲמַר הַב לְעַמָא וְיֵיכְלוּן אֲרֵי כִדְנַן אֲמַר יְיָ יֵיכְלוּן וְיוֹתְרוֹן:
וִיהַב קֳדָמֵיהוֹן וַאֲכַלוּ וְאוֹתִירוּ כְּפִתְגָמָא דַיָי:
