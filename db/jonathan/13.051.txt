כִּדְנַן אֲמַר יְיָ הָא אֲנָא מַיְתֵי עַל בָּבֶל וְעַל יַתְבֵי אַרְעָא כַסְדָאֵי עַמְמִין קְטוֹלִין דְרָם לִבְּהוֹן וְשַׁפִּירִין בְּקוּמָא רוּחֲהוֹן מְחַבְּלָא:
וְאֶשְׁלַח עַל בָּבֶל בָּזוֹזִין וְיִבְזוּנָהּ וִירוֹקְנוּן יַת אַרְעָא אֲרֵי הֲווֹ עֲלָהּ מִסְחוֹר סְחוֹר בְּיוֹם בִּישׁ:
לָא יִמְתַּח דִמְתַּח קַשְׁתֵּיהּ וְלָא יִסַק בְּסִרְיָנַיָהּ וְלָא תְחוּסוּן עַל עוּלֵימָהָא גַמִירוּ כָּל נִכְסָהָא:
וְיִתְרְמוֹן קְטוֹלִין בְּאַרְעָא כַּסְדָאֵי וּמַטְעֲנִין בְּשׁוֹקָהָא:
אֲרֵי לָא שְׁבִיקִין בֵּית יִשְׂרָאֵל וִיהוּדָה מֵאֱלָהָהוֹן מַיָי צְבָאוֹת אֲרֵי אַרְעֲהוֹן אִתְמְלִיאַת חֲיָבֵי קְטוֹל מִמֵימַר קַדִישָׁא דְיִשְׂרָאֵל:
עֲרוּקוּ מִגוֹ בָבֶל וְשֵׁיזָבוּ גְבַר נַפְשֵׁיהּ לָא תִלְקוֹן בְּחוֹבָהּ אֲרֵי עִדַן פּוּרְעֲנוּתָא הִיא קֳדָם יְיָ גִמְלָא הוּא מְשַׁלֵם לָהּ:
הָא כְּכָסָא דְדַהֲבָא דְהוּא רְגִיג מִכָּל מָנַיָא כֵּן עִלֵי חוֹבָא דְבָבֶל בְּכֵן פּוּרְעֲנוּת תַּקִיפָא עֲתִידָא דְיֵיתֵי לָהּ מִן קֳדָם יְיָ עֲתִידִין דְיִתְרַווֹן מִנֵהּ רַוָן כָּל מַלְכֵי עַמְמַיָא וּמִכָּס פּוּרְעֲנוּתָא יִשְׁתּוּן עַמְמַיָא עַל כֵּן יִשְׁתַּגְשׁוּן עַמְמַיָא:
בִּתְכֵּיף נָפְלַת בָּבֶל וְאִתְּבָרַת אֱלִילוּ עֲלָהּ סָבוּ שְׁעִיף לִתְבִירָהּ מָאִים תִּתַּסִי:
אֲסֵינָא יַת בָּבֶל וְלָא מִתַּסְיָא שַׁבְקוּהָא וְנֵיזֵיל גְבַר לְאַרְעֵיהּ אֲרֵי קָרִיב לִשְׁמַיָא תְּבִירָהּ וּמְנַטֵל עַד שְׁמֵי שְׁמַיָא:
גְלָא יְיָ יַת זְכוּתָנָא אֲתוֹ וְנִשְׁתָּעֵי בְצִיוֹן יַת עוֹבָדָא דִי עָבֵיד יְיָ אֱלָהָנָא:
שַׁנִינוּ גִרְרַיָא מְלוֹ שִׁלְטַיָא אָעִיר יְיָ יַת רוּחַ מַלְכֵי מָדַי אֲרֵי עַל בָּבֶל רְעוּתֵיהּ לְחַבְּלוּתָהּ אֲרֵי פּוּרְעֲנוּתָא דַייָ הִיא פּוּרְעֲנּוּת הֵיכְלֵיהּ:
עַל שׁוּרֵי בָּבֶל זְקוּפוּ אָתָא אַתְקִיפוּ מְטַרְתָּא אַקִימוּ נָטְרַיָא אַתְקִינוּ חוּצְנָיָא אֲרֵי אַף חָשַׁב יְיָ אַף עָבֵיד יַת דְמַלֵיל עַל יַתְבֵי בָּבֶל:
דְשַׁרְיָא עַל מַיִין סַגִיאִין סַגִיאִין בֵּית נִנָזָהָא מְטָא יוֹם תַּבְרִיךְ עִדַם סְעוּרַת בִּישׁוּתִיךְ:
קַיָם יְיָ צְבָאוֹת בְּמֵימְרֵיהּ אֱלָהֵין אֲמַלִינִיךְ מַשִׁרְיַת עַמְמִין סַגִיאִין כְּזַחֲלָא וִירִימוּן עֲלָךְ קָלְהוֹן:
אֲמַר אִלֵין דַעֲבַד אַרְעָא בְּחֵילֵיהּ שַׁכְלֵיל תֵּבֵל בְּחוּכְמְתֵיהּ וּבְסוּכְלְתָנוּתֵיהּ תְּלָא שְׁמַיָא:
לְקַל מִתְּנֵיהּ רִכְפַת דְמַיִין בִּשְׁמַיָא וּמַסֵיק עֲנָנִין מִסְיָפֵי אַרְעָא בַּרְקִין לִזְמַן מִטְרָא עָבִיד וּמַפִּיק רוּחִין מִבֵּית גִנְזוֹהִי:
אִטַפָּשׁוּ כָּל עַמְמַיָא מִלְמִידַע חוּכְמְתָא בְּהִית כָּל קִינֵי מִלְמֶעְבַּד צְלַם אֲרֵי שִׁקְרָא אַתִּיכוּנוּן וְלָא רוּחָא בְּהוֹן:
לְמָא אִינוּן עוֹבַד תּוּלְעֲבָא בְּעִדַן דְאַסְעַר עֲלֵיהוֹן חוֹבֵיהוֹן יֵיבְדוּן:
לָא כְאִלֵין חוּלְקֵי דְיַעֲקֹב אֲרֵי דִי בְרָא כוֹלָא הוּא וְיִשְׂרָאֵל חוּלַק שִׁבְטָא דְאַחֲסַנְתֵּיהּ יְיָ צְבָאוֹת שְׁמֵיהּ:
מְבַדַר אַתְּ קֳדָמַי כְּרַכָּא דִי בֵּיהּ מָנֵי קְרָבָא וַאֲבַדַר בָּךְ עַמְמִין וַאֲחַבֵּיל בָּךְ מַלְכְּוָן:
וַאֲבַדַד בָּךְ סוּסְוָן וּרְתִיכֵיהוֹן וַאֲבַדַר בָּךְ רְתִיכִין וְרִכְבֵיהוֹן:
וַאֲבַדַר בָּךְ גְבַר וְאִתָּא וַאֲבַדַר בָּךְ סַב וְיָנִיק וַאֲבַדַר בָּךְ עוּלֵימִין וּבְתוּלָן:
וַאֲבַדַר בָּךְ רַעֲיָא וְעֶדְרֵיהּ וַאֲבַדַר בָּךְ אִכַּר וּפַדְנֵיהּ וַאֲבַדַר בָּךְ שִׁלְטוֹנִין וְטוּרְנִין:
וַאֲשַׁלֵם לְבָבֶל וּלְכֹל יַתְבֵי אַרְעָא כַסְדָאֵי יַת כָּל בִּישַׁתְהוֹן דַעֲבָדוּ בְּצִיוֹן לְעֵינֵיכוֹן אֲמַר יְיָ:
הָא אֲנָא שְׁלַח רוּגְזִי עֲלָךְ כְּרַכָּא מְחַבְּלָא אֲמַר יְיָ דִמְחַבֵּיל יַת כָּל אַרְעָא וַאֲרֵים יַת מְחַת גְבוּרְתִּי עֲלָךְ וַאֲמַגְרִינָךְ מִן תּוּקְפָא וְאֶתְּנִינָךְ לִכְרַךְ יְקִידָא:
וְלָא יִסְבוּן מִנָךְ מְלַךְ לְמָלְכוּ וְשִׁלְטוֹן לְשׁוּלְטָנוּ אֲרֵי צָדוּת עֲלַם תְּהֵי אֲמַר יְיָ:
זְקוּפוּ אָתָא בְאַרְעָא תְּקַעוּ שׁוֹפָרָא בְּעַמְמַיָא זַמִינוּ עֲלָהּ עַמְמַיָא בַּסַרוּ עֲלָהּ מַלְכְּוַת אַרְעָא דְקַרְדוּ מַשִׁרְיַת הוּרְמִינֵי וְהַדֵיב זַמִינוּ עֲלָהּ עָבְדֵי קְרַבָא דִיהוֹן סַלְקִין סוּסְוַתְהוֹן מְפַנְכִין הָא כִזַחֲלָא דִמְצַלְהַב:
זַמִינוּ עֲלָהּ עַמְמַיָא יַת מַלְכֵי מָדָי יַת שִׁלְטוֹנָהָא וְיַת כָּל טוּרְנָהָא וְיַת כָּל אֲרַע שׁוּלְטָנֵיהּ:
וּתְזוּעַ אַרְעָא וְתִתְמְרַע אֲרֵי קָמָא עַל בָּבֶל מַחְשְׁבָתָא דַייָ לְשַׁוָאָה יַת אַרְעָא דְבָּבֶל לְצָדוּ מִבְּלִי יָתֵיב:
אִתְמְנִיעוּ גִבָּרֵי בָּבֶל מִלְאֲגָחָא קְרָבָא יְתִיבוּ בְמִצָדַיָא אִתְּבָרַת גִבָּרוּתְהוֹן הֲווֹ חֲלִישִׁין כִּנְשַׁיָא אִתְדְלָקָא בִּרְנְיָתָהָא אִתְבָּרוּ עַבְּרָהָא:
רָהִיט לִקֳדָמוּת רָהִיט יִרְהוֹט וּמְחַוֵי לִקֳדָמוּת מִחַוֵי לְחַוָאָה לְמַלְכָּא דְבָבֶל אֲרֵי אִתְכְּבִישַׁת קַרְתֵּיהּ מִסוֹפֵיהּ:
וּמְגִיזָתָא אִתְאֲחָדָא וְיַת אֲגַמַיָא אוֹקִידוּ בְנוּרָא וְגַבְרֵי עָבְדֵי קְרָבָא אִתְבְּהִילוּ:
אֲרֵי כִּדְנַן אֲמַר יְיָ צְבָאוֹת אֱלָהָא דְיִשְׂרָאֵל כְּנִשְׁתָּא דְבָבֶל דַמְיָא לְאִדְרָא דִמְטָא עִדַן אִדְרִיכוּתֵיהּ עוֹד (צִיבְחַר) כִּזְעֵיר וְיֵיתוּן בָּזוֹזִין עֲלָהּ:
אֲמָרַת יְרוּשְׁלֵם בָּז נִכְסִי וְשֵׁיצַנִי נְבוּכַדְרֶאצַר מַלְכָּא דְבָבֶל נַפְּצַנִי כְּמָן רֵיקָן בַּלְעֵנִי כְּתַנִינָא מִלָא בֵּית גִנְזֵיהּ מִטוּב אַרְעִי טַלְטְלַנִי:
חֲטוּפִי וּבוּזִי עַל בָּבֶל תֵּימַר כְּנִשְׁתָּא דְצִיוֹן וְחוֹבַת דַם זַכַּאי דְאִתְשַׁד בִּי עַל יַתְבֵי אַרְעָא כַסְדָאֵי תֵּימַר יְרוּשְׁלֵם:
בְּכֵן כִדְנַן אֲמַר יְיָ הָא אֲנָא דָאִין יַת דִינִיךְ וּמִתְפְּרַע יַת פּוּרְעֲנוּתִיךְ וְאַחְרִיב יַת יַמָהּ וְאוֹבִישׁ יַת מַבּוּעָהָא:
וּתְהֵי בָּבֶל לְיַגְרִין מְדוֹר יְרוֹדִין צָדוּ וְאִשְׁתְּמָמוּ מִבְּלִי יָתֵיב:
כַּחֲדָא כְּאַרְיְוָן יִנְהֲמוּן כִּבְנֵי אַרְיְוָן יְרִימוּן קָלְהוֹן:
אַיְתָא עֲלֵיהוֹן עָקָא וִיהוֹן דָמָן לְרַוְיָא בְּדִיל דְלָא יְהוֹן תַּקִיפִין וִימוּתוּן מוֹתָא תִנְיָינָא וְלָא יֵחוּן לְעַלְמָא דְאָתֵי אֲמַר יְיָ:
אֶמְסָרִינוּן כְּתוֹרִין לִקְטָלָא כְּדִכְרִין עִם גַדְיָן:
אֵיכְדֵין אִתְכְּבִישַׁת בָּבֶל וְאִתְאֲחַדַת תּוּשְׁבְּחַת כָּל אַרְעָא אֵיכְדֵין הֲוַת לְצָדוּ בָּבֶל בְּעַמְמַיָא:
סְלִיק עַל בָּבֶל מְלַךְ בְּמַשִׁרְיָתֵיהּ דְסַגִי כְּמֵי יַמָא בְּאִתְרְגוּשַׁת גַלוֹהִי חֲפָא:
הֲוָאָה קִרְוָאָה לְצָדוּ אַרְעָא צַדְיָא וְחָרְבָא אַרְעָא לָא יָתֵיב בָּהּ כָּל אֱנַשׁ וְלָא יֶעְבַר בָּהּ בַּר אֱנַשׁ:
וְאַסְעַר עַל דְפָלְחִין לְבַל בְּבָּבֶל וְאַבֵּיז יַת נִכְסוֹהִי מִנֵיהּ וְלָא יִשְׁתַּמְעוּן לֵיהּ עוֹד עַמְמַיָא אַף שׁוּרִי בָּבֶל יִתְרְמוּן:
פּוּקוּ מִגַוָהּ עַמִי וְשֵׁיזִיבוּ גְבַר יַת נַפְשֵׁיהּ מִתְּקוֹף רוּגְזָא דַיָי:
וְדִלְמָא יְזוּעַ לִבְּכוֹן וְתִדַחֲלוּן מִן קֳדָם בְּסוֹרָתָא דְתִשְׁתְּמַע בְּאַרְעָא וְתֵיתֵי בְשַׁתָּא בְסוֹרְתָא וּבַתְרוֹהִי בְּשַׁתָּא בְסוֹרְתָּא וְיִסְגוֹן חֲטוּפִין בְּאַרְעָא וִיקוּם שִׁלְטוֹן עַל שִׁלְטוֹן:
בְּכֵן הָא יוֹמַיָא אָתָן וְאַסְעַר עַל צַלְמֵי בָבֶל וְכָל אַרְעָא תֶחֱרוֹב וְכָל קְטִילָהָא יִתְמְרוּן בְּגַוָהּ:
וִישַׁבְּחוּן עַל בָּבֶל שְׁמַיָא וְאַרְעָא וְכָל דִי בְהוֹן אֲרֵי מִצְפּוּנָא יֵיתוּן עֲלָה בָּזוֹזִין אֲמַר יְיָ:
אַף בְּבָבֶל אִתְרְמִיאוּ קְטִילֵי יִשְׂרָאֵל אַף לְבָבֶל יִתְרְמוּן קְטִילֵי כָל אַרְעָא:
מְשֵׁיזְבַיָא מֵחַרְבָּא גְלוֹ לָא תִתְעַכְּבוּן אִדְכָּרוּ מֵרָחִיק יַת פּוּלְחָנָא דַייָ וִירוּשְׁלֵם תִּסַק עַל לִבְּכוֹן:
בְּהִיתְנָא אֲרֵי שְׁמַעְנָא חִסוּדִין חֲפַת אִתְכְּנָעוּתָא אַפָּנָא אֲרֵי עָלוּ נוּכְרָאִין וּסְאִיבוּ יַת בֵּית מַקְדָשָׁא דַיָי:
בְּכֵן הָא יוֹמַיָא אָתָן אֲמַר יְיָ וְאַסְעַר עַל צַלְמָהָא וּבְכָל אַרְעָא יִתְרְמוּן קְטִילַיָא:
אֲרֵי תִבְנֵי בָבֶל בִּנְיָנִין דְרָמִין עַד צֵית שְׁמַיָא וַאֲרֵי תִכְרוֹךְ כְּרַכִּין תַּקִיפִין עַד רוּמָא מִן קֳדָמֵי יֵיתוּן בָּזוֹזִין עֲלָהּ אֲמַר יְיָ:
קַל צִוְחָתָא מִבָּבֶל וְתַבְרָא רַבָּא מֵאַרְעָא כַסְדָאֵי:
אֲרֵי אַבֵּיז יְיָ יַת בָּבֶל וְאַבֵיד מִנָהּ מַשִׁרְיָן סַגִיאִין וְיִתְכַּנְשׁוּן עֲלֵיהוֹן מַשִׁרְיַת עַמְמִין סַגִיאִין וִירִימוּן בְּאִתְרְגוּשַׁת קָלְהוֹן:
אֲרֵי אֱתוֹ עֲלָהּ עַל בָּבֶל בָּזוֹזִין וְאִתְאֲחִידוּ גִבְּרָהָא אִתְּבְרָא קַשְׁתּוֹתֵיהוֹן אֲרֵי אֱלָהָא מָרֵי גַמְלַיָא יְיָ שַׁלְמָא יְשַׁלֵם לְהוֹן גְמוּלְהוֹן:
וַאֲרַוֵי רַבְרְבָהָא וְחַכִּימָהָא שׁוּלְטָנָהָא וְטוּרְנָהָא וְגִבָּרָהָא וִימוּתוּן מוֹתָא תִנְיָנָא וְלָא יֵיתוּן לְעַלְמָא דְאָתֵי אֲמַר מַלְכָּא יְיָ צְבָאוֹת שְׁמֵיהּ:
כִּדְנַן אֲמַר יְיָ צְבָאוֹת שׁוּרֵי בָּבֶל פָּתַיָא אִתְחַמָרָא יִתְחַמְרוּן וְתַרְעָהָא רָמַיָא בְּנוּרָא יִתּוֹקְדוּן וְיֶהְלוּן עַמְמַיָא דְבָבֶל לְרֵיקָנוּ וּמַלְכְּוָתָא תְסוּף אֶשְׁתָּא וְיִשְׁתַּלְהוּן:
פִּתְגָמָא דִי פַקֵיד יִרְמְיָה נְבִיָא יַת שְׂרָיָה בַר נֵרִיָה בַר מַחְסֵיָה כַּד אָזֵל בִּשְׁלִיחוּת צִדְקִיָה מְלַךְ שִׁבְטָא דְבֵית יְהוּדָה לְבָבֶל בְּשַׁתָּא רְבִיעֲתָא לְמַמְלְכֵיהּ וּשְׂרָיָה רַב תּוּקְרַבְתָּא:
וּכְתַב יִרְמְיָה יַת כָּל בִּישְׁתָא דַעֲתִידָא לְמֵיתֵי עַל בָּבֶל עַל סִפְרָא חַד יַת כָּל פִּתְגָמַיָא הָאִלֵין דִכְתִיבִין עַל בָּבֶל:
וַאֲמַר יִרְמְיָה לִשְׂרָיָה כְּמֵיתָךְ לְבָבֶל וְתֶחֱזֵי וְתִקְרֵי יַת כָּל פִּתְגָמַיָא הָאִלֵין:
וְתֵימַר יְיָ אַתְּ גָזַרְתְּ עַל אַתְרָא הָדֵין לְשֵׁיצָיוּתֵיהּ בְּדִיל דְלָא לְמֶהֱוֵי בֵּיהּ יָתֵיב לְמֵאֱנָשָׁא וְעַד בְּעִירָא אֲרֵי צָדוּת עֲלַם תְּהֵי:
וִיהֵי כְּשֵׁצָיוּתָךְ לְמִקְרֵי יַת סִפְרָא הָדֵין תִּקְטַר בֵּיהּ אַבְנָא וְתִרְמִינֵיהּ לְגוֹ פְּרָת:
וְתֵימַר כְּדֵין תִּשְׁתְּקַע בָּבֶל וְלָא תְקוּם מִן קֳדָם בִּישְׁתָא דִי אֲנָא מַיְתֵי עֲלָהּ וְיִשְׁתַּלְהוֹן עַד הָכָא נְבוּאַת פִּתְגָמֵי יִרְמְיָהוּ:
