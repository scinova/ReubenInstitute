וַי דְמוֹחֲיָא וּמִתְפָּרְקָא קַרְתָּא דִמְסַגְיָא לְאַרְגָזָא:
לָא שְׁמָעַת בְּקַל עַבְדוֹהִי נְבִיַיָא וְלָא קַבִּילַת אוּלְפַן בְּמֵימְרָא דַייָ לָא אִתְרְחִיצַת וּלְפוּלְחַן דֶאֱלָהָא לָא אִתְקְרָבַת:
רַבְרְבָהָא בְּגַוָה כְּאַרְיְוָן נָהֲמִין דַיָנָהָא דֵיבֵי רַמְשָׁא לָא מוֹרִיכִין לְצַפְרָא:
נְבִיֵי שִׁקְרָא דִי בְגַוָהּ בָּאֲשִׁין גַבְרִין שָׁקְרִין כַּהֲנָתָא מַפְסִין קוּדְשָׁא חָטְפִין בְּאוֹרַיְתָא:
יְיָ זַכָּאָה אֲמַר לְאַשְׁרָאָה שְׁכִנְתֵּיהּ בְּגַוָהּ לֵית קֳדָמוֹהִי לְמֶעְבַּד שְׁקַר הָא כִנְהוֹר צַפְרָא דְאָזֵיל וְתַקִין כֵּן דִינֵיהּ נָפֵיק לְאַפְּרַשׁ לָא מִתְעַכֵּב וְלָא יָדְעִין עַיְלַיָא בַהֲתָא:
שֵׁיצֵיתִי עַמְמַיָא אַצְדִיאָה בִּרְנְיָתְהוֹן אַחֲרִיבַית שׁוּקֵיהוֹן מִבְּלִי עָדֵי צְדִיאָה קִרְוֵיהוֹן מִבְּלִי אֱנַשׁ מִבְּלִי יָתֵיב:
אֲמָרֵית בְּרַם תִּדְחֲלוּן מִן קֳדָמַי תְּקַבְּלוּן אוּלְפַן וְלָא יִפְסוֹק מְדוֹרֵיהוֹן מֵאַרְעָא בֵּית שְׁכִנְתִּי כָּל טַבְוָן דַאֲמָרֵית לְהוֹן אַיְתֵי לְהוֹן בְּכֵן אַקְדִימוּ חַבִּילוּ כָּל עוֹבָדֵיהוֹן:
בְּכֵן סַבָּרוּ לְמֵימְרִי אֲמַר יְיָ לְיוֹם אִתְגְלָיוּתִי לְמֵדַן אֲרֵי דִינַי לְכַנָשָׁא עַמְמַיָא לְקָרָבָא מַלְכוּתָא לְמִשְׁפַּךְ עֲלֵיהוֹן חֵימְתִי כָּל תְּקוֹף רוּגְזִי אֲרֵי בְּאֶשַׁת פּוּרְעֲנוּתִי יְסוּפוּן כָּל רִשְׁעֵי אַרְעָא:
אֲרֵי בְּכֵן אֱשַׁנֵי עַל עַמְמַיָא מַמְלַל חָד בְּהִיר לְצַלָאָה כּוּלְהוֹן בִּשְׁמָא דַייָ לְמִפְלַח קֳדָמוֹהִי כְּתַף חָד:
מֵעֵבֶר לְנַהֲרֵי הוֹדוּ בְּרַחֲמִין יְתוּבוּן גַלְוַת עַמִי דְאִתְגְלִיאָה וִיהוֹן מַיְתָן לְכוֹן הָא בְּקוּרְבָּנִין:
בְּעִדָנָא הַהִיא לָא תִבְהֲתוּן מִכָּל עוֹבָדַיִךְ בִּישַׁיָא דִי מָרַדְתָּ בְּמֵימְרִי אֲרֵי בְכֵן אֲגַלֵי מִגַוֵיךְ תַּקִיפֵי תוּשְׁבַּחְתֵּךְ וְלָא תוֹסִיפוּן לְאִתְרַבְרָבָא עוֹד בְּטוּרָא דְקוּדְשִׁי:
וְאַשְׁאַר בְּגַוֵיךְ עַם עִנְוְתָן וּמְקַבֵּל עוּלְבָן וְיִתְרַחֲצוּן בִּשְׁמָא דַיָי:
שְׁאָרָא דְיִשְׂרָאֵל לָא יַעְבְּדוּן שְׁקַר וְלָא יְמַלְלוּן כַּדְבִין וְלָא יִשְׁתְּכַח בְּפוּמֵיהוֹן לִישַׁן דְנִכְלִין אֲרֵי אִינוּן יִתְפַּרְנְסוּן וְיִשְׁרוֹן וְלֵית דְמָנִיד:
שַׁבְּחִי כְּנִשְׁתָּא דְצִיוֹן וּבוּעוּ יִשְׂרָאֵל חֲדוּ וּבוּעוּ בְּכָל לִבָּא כְּנִשְׁתָּא דִירוּשְׁלֵם:
אַגְלֵי יְיָ דַיָנֵי שִׁקְרָא מִגַוִיךְ פַּלֵי בַּעֲלֵי דְבָבֵיךְ מַלְכֵיהּ דְיִשְׂרָאֵל יְיָ אֲמֵר לְאַשְׁרָאָה שְׁכִנְתֵּיהּ בְּגַוֵיךְ לָא תִדְחֲלִין מִן קֳדָם בִּישָׁא עוֹד:
בְּעִדָנָא הַהִיא יִתְאַמַר לִירוּשְׁלֵם לָא תִדְחֲלוּן צִיוֹן לָא יִתְרַשְׁלָן יְדָךְ:
יְיָ אֱלָהָךְ אֲמַר לְאַשְׁרָאָה שְׁכִנְתֵּיהּ בְּגַוֵיךְ גִבַּר פָּרִיק יֶחְדֵי עֲלָךְ בְּחֶדְוָא יִכְבּוֹשׁ עַל חוֹבָךְ בִּרְחִמְתֵּיהּ יְדוּץ עֲלָךְ בְּדִיץ:
דִי הֲווֹ מְעַכְּבִין בִּיךְ זְמַנֵי מוֹעֲדֵיךְ אַרְחִיקֵית מִנֵיךְ וַי עֲלֵיהוֹן עַל דַהֲווֹ נַטְלִין זְיָנֵיהוֹן לְקִבְלֵיךְ וּמְחַסְדִין לִיךְ:
הָא אֲנָא עָבֵיד גְמִירָא עִם כָּל מְשַׁעְבְּדַיִךְ בְּעִדָנָא הַהִיא וְאֶפְרוֹק יַת מְטַלְטְלַיָא וּמְבַדְרַיָא אֲקַרֵיב וַאֲשַׁוִינוּן לְתוּשְׁבְּחָא וּלְשׁוּם בְּכָל אַרְעָא בַּהֲתָתְהוֹן:
בְּעִדָנָא הַהִיא אַכְנֵישׁ יַתְכוֹן וּבְעִדָנָא הַהִיא אֲקָרֵיב גָלוּתְכוֹן אֲרֵי אֶתֵּן יַתְכוֹן לְשׁוּם וּלְתוּשְׁבְּחָא בְּכָל עַמְמֵי אַרְעָא כַּד אָתֵיב יַת גָלוּתְכוֹן לְעֵינֵיכוֹן אֲמַר יְיָ:
