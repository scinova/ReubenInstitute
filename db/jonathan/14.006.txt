וַהֲוָה פִּתְגַם נְבוּאָה מִן קֳדָם יְיָ עִמִי לְמֵימָר:
בַּר אָדָם קַבֵּל נְבוּאָה עַל טוּרֵי יִשְׂרָאֵל וְאִתְנַבֵּי עֲלֵיהוֹן:
וְתֵימַר טוּרֵי יִשְׂרָאֵל קַבִּילוּ פִתְגָמָא דַייָ אֱלֹהִים כִּדְנַן אֲמַר יְיָ אֱלֹהִים לְטוּרַיָא וּלְרָמָתָא וּלְפַצִידַיָא וּלְחֵילַיָא הָא אֲנָא מַיְתֵי עֲלֵיכוֹן דְקַטְלִין בְּחַרְבָּא וְאוֹבִיד בָּמָתֵיכוֹן:
וְיִצְדוּן אֱגוֹרֵיכוֹן וְיִתְבְּרוּן חֲנִיסְנְסֵיכוֹן וְאַרְמִי קְטִילֵיכוֹן קֳדָם פִּגוּר טַעֲוַתְכוֹן:
וְאֶתֵּן יַת פִּגְרֵי בְּנֵי יִשְׂרָאֵל קֳדָם פִּגּוּר טַעֲוַתְהוֹן וַאֲבַדֵר יַת גַרְמֵיכוֹן סִחְרָנוּת אֱגוֹרֵיכוֹן:
בְּכֵל מוֹתְבָנֵיכוֹן קִרְוַיָא יֶחֱרְבָן וּבְמָתָא יִצְדְיָן בְּדִיל דְיֶחֱרְבוּן וְיִצְדוּן אֱגוֹרֵיכוֹן וְיִתְבְּרָן וְיִבְטְלָן טַעֲוַתְכוֹן וְיִתְקַטְפוּן חֲנִסְנְסֵיכוֹן וְיִתְמְחוּן עוֹבָדֵיכוֹן:
וְיִתְרְמוּן קְטִילִין בֵּינֵיכוֹן וְתֵדְעוּן אֲרֵי אֲנָא יְיָ:
וְאַשְׁאַר בְּמֶהֱוֵי לְכוֹן מְשֵׁזְבֵי חַרְבָּא בֵּינֵי עַמְמַיָא בְּאִתְבַּדָרוּתֵיכוֹן בִּמְדִינָתָא:
וְיִדְכְּרוּן מְשֵׁיזְבֵיכוֹן דְחַלְתִּי בֵּינֵי עַמְמַיָא דְאִשְׁתְּבִיאוּ לְתַמָן דִתְבַרֵית יַת לִבְּהוֹן טִפְּשָׁא דִסְטָא מִבָּתַר פּוּלְחָנִי וְיַת חֵיזוּ עֵינֵיהוֹן דִטְעָא בָּתַר טַעֲוַתְהוֹן וְיִדַנְקוּן וִיהוֹן חָזַן עַל בִּישְׁתָּא דַעֲבַדוּ לְכָל תּוֹעֲבַתְהוֹן:
וְיֵדְעוּן אֲרֵי אֲנָא יְיָ לָא לְמַגָן גְזֵרֵית בְּמֵימְרִי לְמֶעְבַּד לְהוֹן בִּישְׁתָּא הָדָא:
כִּדְנַן אֲמַר יְיָ אֱלֹהִים טְפַּח בִּידָךְ וּרְפוֹס בְּרַגְלָךְ וַאֲמַר וַי עַל כָּל תּוֹעֲבָתָא בִּישְׁתָא בֵּית יִשְׂרָאֵל דִי בְּחַרְבָּא בְּכַפְנָא וּבְמוֹתָא יִתְרְמוּן:
דְרָחִיק בְּמוֹתָא יְמוּת וּדְקָרִיב בְּחַרְבָּא יִתְקְטֵיל וּדְיִשְׁתְּאַר וּדְיֵעוֹל לִכְרַכֵּי צַיְרָא בְּכַפְנָא יְמוּת וְיָחוּל רוּגְזִי בְּהוֹן:
וְתִדְעוּן אֲרֵי אֲנָא יְיָ בְּמֶהֱוֵי קְטִילֵיהוֹן בְּגוֹ פִּגוּר טַעֲוַתְהוֹן סַחֲרָנוּת אִגוֹרֵיהוֹן עַל כָּל רָמָא מְנַטְלָא בְּכָל רֵישֵׁי טוּרַיָא תְּחוֹת כָּל אִילָן עַבּוּף וּתְחוֹת כָּל בּוּטְמָא מְעַפְיָא אֲתַר דִיהַבוּ תַמָן קוּרְבָּנִין לְפוּלְחַן לְכָל טַעֲוַתְהוֹן:
וְאָרִים יַת מְחַת גְבוּרְתִּי עֲלֵיהוֹן וְאֶתֵּן יַת אַרְעָא לְצָדוּ וְאִשְׁתְּמָמוּ מִמִדְבַּר דִבְלָת בְּכָל מוֹתְבָנֵיהוֹן וְיֵדְעוּן אֲרֵי אֲנָא יְיָ:
