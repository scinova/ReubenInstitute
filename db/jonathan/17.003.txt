קַבִּילוּ יַת פִּתְגָמָא הָדֵין דִי גְזַר יְיָ עֲלֵיכוֹן בֵּית יִשְׂרָאֵל עַל כָּל זַרְעֲיָתָא דְאַסְקִית מֵאַרְעָא דְמִצְרַיִם לְמֵימָר:
לְחוֹד בְּכוֹן אִתְרְעֵיתִי מִכָּל זַרְעֲיַת אַרְעָא עַל כֵּן אַסְעַר עֲלֵיכוֹן יַת כָּל חוֹבֵיכוֹן:
הֲיַהֲכוּן תְּרֵין כַּחֲדָא אֱלָהֵין אִם אִזְדַמָנוּ:
הֲיִנְהוֹם אַרְיֵה בְּחוּרְשָׁה וְצֵידָה לֵית לֵיהּ הַיְרִים בַּר אַרְיְוָן קָלֵיהּ מִמַרְבּוּעָתֵיהּ אֱלָהֵין אִם אֲחָד:
הֲתִפּוֹל צִפְּרָא בְּקוּלָא עַל אַרְעָא וְצַיִד לֵית לֵיהּ הֲיִסַק קוּלָא מִן אַרְעָא וּמֵחַד לָא אֲחָד:
אִם יִתְקַע שׁוֹפָרָא בְּקַרְתָּא בְּלָא זִמְנֵיהּ וְעַמָא דִי בָהּ לָא יְזוּעוּן אִם תְּהֵי בִישְׁתָא בְּקַרְתָּא וּמִן קֳדָם יְיָ לָא אִתְעַבְּדָת:
אֲרֵי לָא כָשְׁרִין פִּתְגָמִין דְמִתְעַבְּדִין מִן קֳדָם יְיָ אֱלֹהִים אֱלָהִין גְלָא רָזֵיהּ לְעַבְדוֹהִי נְבִיַיָא:
אַרְיֵה נָהֵים מַן לָא יִדְחַל יְיָ אֱלֹהִים מַלֵיל מַן לָא יְקַבֵּיל נְבוּאֲתָא:
בַּסָרוּ עַל בִּרְנְיָת בְּאַשְׁדוֹד וְעַל בִּרְנְיָת בְּאַרְעָא דְמִצְרָיִם וַאֲמָרוּ אִתְכְּנָשׁוּ עַל טוּרֵי שֹׁמְרוֹן וַחֲזוֹ שִׁגוֹשַׁיָא סַגִיאָן בְּגַוָהּ וַאֲנוּסִין בְּגַוָהּ:
וְלָא יָדְעִין לְמֶעְבַּד אוֹרַיְתָא אֲמַר יְיָ דִי מָלָן אוֹצְרֵיהוֹן חֲטוֹף וּבִזָא בְּבִרְנְיָתְהוֹן:
בְּכֵן כִּדְנַן אֲמַר יְיָ אֱלֹהִים עָקָא תַּקִיפָא לְאַרְעָא וְיִבָּטֵיל מִנִיךְ תּוֹקְפֵיךְ וְיִתְבַּזוּן בִּרְנְיָתָיִךְ:
כִּדְנַן אֲמַר יְיָ כְּמָא דִמְשֵׁיזֵיב רַעֲיָא מִפּוּמָא דְאַרְיָא תַּרְתֵּין כַּרְעִין אוֹ חַסְחוֹס דְאוֹדָן כֵּן יִשְׁתֵּיזְבוּן בְּנֵי יִשְׂרָאֵל דְיָתְבִין בְּשֹׁמְרוֹן בִּתְקוֹף שׁוּלְטַן וְעַל דַמֶשֶׂק רְחִיצִין:
קַבִּילוּ וְאַסְהִידוּ בְּבֵית יַעֲקֹב אֲמַר יְיָ אֱלֹהִים אֱלֹהֵי צְבָאוֹת:
אֲרֵי בְּיוֹם אַסְעֲרוּתִי חוֹבֵי יִשְׂרָאֵל עֲלוֹהִי וְאַסְעַר עַל דְפָלְחִין לֶאֱגוֹרַיָא בְּבֵית אֵל וְיִתְקַטְפָן קַרְנַת אֱגוֹרָא וְיִפְּלוּן לְאַרְעָא:
וְאֶמְחֵי בֵּית סִתְוָא עַל בֵּית קַיְטָא וְיֵיבְדוּן בָּתִּין דִמְכַבְּשִׁין בְּשֵׁן דְפִיל וִיסוּפוּן בָּתִּין סַגִיאִין אֲמַר יְיָ:
