לְמוֹאָב כִּדְנַן אֲמַר יְיָ צְבָאוֹת אֱלָהָא דְיִשְׂרָאֵל וַי עַל נְבוֹ אֲרֵי אִתְבְּזִיזַת בְּהִיתַת אִתְכְּבִישַׁת קִרְיָתַיִם בְּהִיתוּ מִבֵּית רוֹחֲצָנְהוֹן וְאִתְּבָרוּ:
לֵית עוֹד תּוּשְׁבְּחָא לְמוֹאָב בְּחֶשְׁבּוֹן חֲשִׁיבוּ עֲלָהּ בִּישָׁא אֲתוֹ וּנְשֵׁיצִינָא מִלְמֶהֱוֵי עַמָא אַף מַדְמַן תִּתַּבָּרִין בַּתְרָךְ יְהָכוּן קְטוֹלֵי חַרְבָּא:
קַל צִוְחָתָא מֵחוֹרוֹנָיִם בִּזָא וּתְבָרָא רַבָּא:
אִתְּבָרַת מַלְכוּת מוֹאָב בַּסָרוּ צִוְחֲתָא שִׁלְטוֹנֵיהוֹן:
אֲרֵי מַסוּקִית לוּחִית כַּד בָּכָן יִסְקוּן בָּכָן אֲרֵי בְּמַחֲתֵית חוֹרוֹנַיִם צִוְחַת תְּבִירֵי קְרָב יְבַסְרוּן:
עֲרָקוּ שֵׁיזִיבוּ נַפְשֵׁיכוֹן וַהֲווֹ כְּמִגְדַל עֲרוֹעֵר דְשָׁרָן בִשְׁכוּנִין בְּמַדְבְּרָא:
אֲרֵי חֲלַף דְאִתְרְחִיצַת בְּאוֹצָרָךְ וּבְבֵית גִנְזָךְ אַף אַתְּ תִּתְאַחֲדִין וְיִפּוֹק כְּמוֹשׁ בְּגָלוּתָא פָּלְחוֹהִי וְרַבְרְבוֹהִי כַּחֲדָא:
וְיֵיתוּן בָּזוֹזִין עַל כָּל קִרְוֵיהוֹן וְקִרְוָא מִנְהוֹן לָא תִשְׁתֵּיזִיב וְיֵיבְדוּן דְיָתְבִין בְּעוּמָקָא וְיִשְׁתֵּיצוּן דְשָׁרָן בְּמֵישְׁרָא דַאֲמַר יְיָ:
אַעְדוֹ בִתְרָא מִן מוֹאָב אֲרֵי מִגְלָא תִגְלֵי וְקַרְוָהָא לְצָדוּ יֶהֱוְיַן מִבְּלִי יָתֵיב בְּהוֹן:
לִיט דִי עָבֵיד עֲלָהּ עוֹבָדָא מִן קֳדָם יְיָ בְּנִכְלָא וְלִיט דְמָנַע חַרְבֵּיהּ מִקְטוֹל:
שָׁלָן מוֹאֲבָאֵי מִזְעוּרֵיהוֹן וְשָׁקְטִין אִינוּן עַל נִכְסֵיהוֹן וְלָא אִתְגְלִיאוּ מֵאֲתַר לְאַתְרָא וּבְגָלוּתָא לָא הֲלִיכוּ עַל כֵּן אוֹרִיכוּ שְׁלֵיוָא נִכְסֵיהוֹן כְּחַמְרָא דִנְטִיר עַל שָׁמְרֵיהּ וְרֵיחֵיהּ לָא פָג:
בְּכֵן הָא יוֹמַיָא אַתָן אֲמַר יְיָ וְאֶשְׁלַח עֲלֵיכוֹן בָּזוֹזִין וְיִבְזוּנוּנוּן וְנִכְסֵיהוֹן יְרוֹקְנוּן וְטוּב אַרְעֲהוֹן יְגַמְרוּן:
וְיִבְהֲתוּן מוֹאֲבָאֵי דְפָלְחִין לִכְמוֹשׁ כְּמָא דִבְהִיתוּ בֵּית יִשְׂרָאֵל מִבֵּית אֵל רוּחֲצָנְהוֹן:
אֵיכְדֵין תֵּמְרוּן גִבָּרִין אֲנַחְנָא וְגַבְרִין דְחֵילָא לְמֶעְבַּד קְרָבָא:
אִתְבְּזִיזוּ מוֹאֲבָאֵי וְקִרְוֵיהוֹן חָרוֹבָא וּשְׁפַר עוֹלֵימֵיהוֹן יִתְמַסְרוּן לִקְטָלָא אֲמַר מַלְכָּא יְיָ צְבָאוֹת שְׁמֵיהּ:
קָרֵיב תְּבַרְהוֹן דְמוֹאֲבָאֵי לְמֵיתֵי וּבִשְׁתְּהוֹן מוֹחַיָא לַחֲדָא:
דְווֹ עֲלוֹהִי כָּל סַחֲרָנוֹהִי וְכָל יָדְעֵי שְׁמֵיהּ אֲמָרוּ אֵיכְדֵין אִתְּבַר מְלַךְ מַבְאִישׁ שִׁלְטוֹן מָעִיק:
חוּתִי מִיְקָרָא וְתוּבִי בְצַחְיָא מַלְכוּת כְּנִשְׁתָּא דְדִיבוֹן אֲרֵי בָזוֹזִין אֱתוֹ עַל מוֹאָב סְלִיקוּ עֲלָךְ חַבִּילוּ כְּרַכַּיִךְ:
עַל אוֹרְחָא קוּמוּ וּסְכוּ דְיַתְבִין בַּעֲרוֹעֵר שְׁאִילוּ שְׁאִילוּ לְמֶעְרְקַיָא וּלְמִשֵׁיזָבַיָא אֲמָרוּ מָה הֲוָה עַל תְּבָרָא דֵיכִי:
בְּהִיתוּ מוֹאֲבָאֵי אֲרֵי אִתְּבָרוּ אַיְלִילוּ וּצְוָחוּ חֲווּ בְאַרְנוֹן אֲרֵי אִתְבְּזִיזוּ מוֹאֲבָאֵי:
וְעָבְדֵי פּוּרְעֲנוּתָא מָטוּ לַאֲרַע מֵישִׁרָא לְחוֹלוֹן וּלְיַהֵץ וְעַל מֵיפָעַת:
וְעַל דִיבוֹן וְעַל נְבוֹ וְעַל בֵּית דִבְלָתָיִם:
וְעַל קִרְיָתָיִם וְעַל בֵּית גְמוּל וְעַל בֵּית מְעוֹן:
וְעַל קִרְיוֹת וְעַל בָּצְרָה וְעַל כָּל קִרְוֵי אֲרַע מוֹאָב דִרְחִיקִין וּדְקָרִיבִין:
אִתְקַטְפַת מַלְכוּת מוֹאָב וְשִׁלְטָנֵיהוֹן אִתְּבָרוּ אֲמַר יְיָ:
אַיְתָא עֲלֵיהוֹן עָקָא וִיהוֹן דָמָן לְרַוְיָא אֲרֵי עַל עַמָא דַייָ אִתְרַבְרָבוּ וְיִדְרְשׁוּן מוֹאָבָאֵי בְּאַתְרֵיהוֹן לְלָעֵיב אַף אִינוּן:
וְעַל דַחֲדִיתוּן עַל תְּבִירֵיהּ דְיִשְׂרָאֵל בִּדְלָא חֲטוֹ לְכוֹן כְּאִלוּ בְּגַנָבִין אִשְׁתְּכָחוּ וְעַל דְאַסְגֵיתוּן עֲלֵיהוֹן פִּתְגָמִין בְּכֵן תְּטַלְטְלוּן:
שְׁבָקוּ קִרְוַיָא וּשְׁרוֹ בְכֵיפָא יַתְבֵי מוֹאָב וַהֲווֹ כְיוֹנָה דְשַׁבְקָא פוּם שׁוֹבְכָא וְנָחֲתָא וְשַׁרְיָא בְּאַרְעִית גוּבָא:
שְׁמַעְנָא רַבְרְבֵי מוֹאָב דְמִתְגַאָן לַחֲדָא רַבְרְבֵיהוֹן וּמְפַנְקֵיהוֹן וְגֵיוָתָנוּתְהוֹן וְרָמוּת לִבְּהוֹן:
קָדָמַי גְלַן אֲמַר יְיָ עוֹבְדֵי רַבְרְבָנֵיהוֹן וְלָא תַקְנִין יַקִירֵיהוֹן לָא כִדְכָשַׁר עֲבָדוּ:
עַל כֵּן עַל מוֹאָבָאֵי מְיַלְלִין וְעַל מוֹאָבָאֵי כּוּלְהוֹן מְצַוְחִין עַל אֱנַשׁ כְּרַךְ תּוּקְפֵהוֹן דָוָן:
בְּכֵן כְּמָא דְאַיְתֵיתִי מַשִׁרְיָן עַל יַעְזֵר כֵּן אַיְתֵי קְטוֹלִין עַל שִׂבְמָא מְטַלְטְלָךְ גָזוּ עֲבָרוּ יַמָא עַד יַם יַעְזֵר מָטוּ עַל חִצְרִיךְ וְעַל קַטְפִיךְ בָּזוּזִין נָפְלוּ:
וְתִתְכְּנֵישׁ בִּיעַ וְחֶדְוָא מִכַּרְמְלָא וּמֵאֲרַע מוֹאָב וַחֲמַר מִמַעַצְרַיָא אֲבַטְלֵית לָא יַעְצְרוּן עִיצוּרִין וְלָא יְרִימוּן בְּקָלְהוֹן:
מִצִוְחַת חֶשְׁבּוֹן עַד אֶלְעָלֵה עַד יַהַץ אִשְׁתְּמַע קָלְהוֹן מִצוֹעַר עַד חוֹרוֹנַיִם עֶגְלַת תַּלְתִּית רַבְּתָא אֲרֵי אַף מֵי נְמָרִים לְצָדוּ יְהוֹן:
וַאֲבַטֵיל מִן מוֹאָב אֲמַר יְיָ מַקְרִיב קוּרְבַּן עַל בָּמָתָא וּמַסִיק בּוּסְמִין לְטַעֲוָתֵיהּ:
עַל כֵּן בְּלִבְּהוֹן מוֹאָבָאֵי כְּכִינוֹרָא הָמָן וְלִבְּהוֹן עַל אֱנַשׁ כְּרַךְ תּוּקְפֵיהוֹן כִּזְמַר אֲבוּבִין דָוֵי עַל כֵּן שְׁאָר נִכְסֵיהוֹן דִקְנוֹ יִתְבַּזְזוּן:
אֲרֵי כָּל סְעַר רֵישׁ נְתַר וְכָל דְקַן גִלוּחַ עַל כָּל סִיעַן מִתְהַמְמִין וְעַל חַרְצִין סָק:
עַל כָּל אֱגָרֵיהוֹן דְמוֹאֲבָאֵי וּבִרְחוֹבֵיהוֹן כּוּלְהוֹן עָבְדִין מִסְפֵּדָא אֲרֵי תְבָרֵית יַת מַלְכוּת מוֹאָב כְּמָנָא דְלֵית צְרוֹךְ בֵּיהּ אֲמַר יְיָ:
אֵיכְדֵין אִתְּבָרוּ אַיְלִילוּ אֵיכְדֵין אַחְזָרוּ קְדִלְהוֹן מוֹאֲבָאֵי בְּהִיתוּ וִיהוֹן מוֹאֲבָאֵי לְחוֹךְ וְלִתְבִיר לְכָל סָחֳרָנֵיהוֹן:
אֲרֵי כִדְנַן אֲמַר יְיָ הָא כְנִשְׁרָא דְמִשְׁתְּדֵי כֵּן יִסַק מֶלֶךְ בְּמַשִׁרְיָתֵיהּ וְיִשְׁרֵי עַל מוֹאָב:
וְתִתְכְּבֵישׁ כְּרַכָּא וּמְצָדָתָא יִתְאַחְדָן וִיהֵי לֵב גִבָּרֵיהוֹן דְמוֹאֲבָאֵי בְּיוֹמָא הַהוּא כְּלֵב אִתְּתָא דְעָקָא לָהּ בְּחֶבְלָהָא:
וְיִשְׁתֵּצוּן מוֹאֲבָאֵי מִלְמֶהֱוֵי עַם אֲרֵי עַל עַמֵיהּ דַייָ אִתְרָבְרָב:
דַחֲלָא וְכוּמְצָא וּמְצָדָא עֲלָךְ יָתֵיב מוֹאָב אֲמַר יְיָ:
דְיֶעְרוֹק מִן קֳדָם דַחֲלָא יִפּוֹל לְגוֹ כוּמְצָא וּדְיִסַק מִגוֹ כוּמְצָא אִתְאַחַד בִּסְרִגְתָּא אֲרֵי אַיְתֵי עֲלֵיהוֹן עַל מוֹאֲבָאֵי שְׁנַת סְעוּרַן חוֹבֵיהוֹן אָמַר יְיָ:
עַל דִאֶשְׁתְּמָעוּ לְיַתְבֵי חֶשְׁבּוֹן גְלוֹ אִתְקְצַר חֵילֵיהוֹן עֲרָקוּ אֲרֵי קִידוּם תַּקִיף כְּאֶשְׁתָּא נְפַק מֵחֶשְׁבוֹן וְעָבְדֵי קְרָבָא כְּשַׁלְהוֹבִיתָא מִבֵּית סִיחוֹן וְקַטִילוּ רַבְרְבֵי מוֹאָב וְיַקִירֵי בְּנֵי אִתְרְגוּשָׁא:
וַי לְכוֹן מוֹאֲבָאֵי אֲבַדְתּוּן עַמָא דְפַלְחִין לִכְמוֹשׁ אֲרֵי אִדַבְּרוּ בְנָךְ בְּשִׁבְיָא וּבְנָתָךְ בְּגָלוּתָא:
וַאֲתֵיב גָלוּת מוֹאָב בְּסוֹף יוֹמַיָא אֲמַר יְיָ עַד הָכָא לְמֶעְבַּד פּוּרְעֲנוּת דִין מִן מוֹאָב:
