וַהֲוָה בְּיוֹמַיָא הָאִינוּן וּכְנָשׁוּ פְּלִשְׁתָּאֵי יַת מַשִׁירְיַתְהוֹן לְחֵילָא לְאַגָחָא קְרָבָא בְיִשְׂרָאֵל וַאֲמַר אָכִישׁ לְדָוִד מִדַע תֵּדַע אֲרֵי עִמִי תִּפּוֹק בְּמַשְׁרִיָתָא אַתְּ וְגַבְרָךְ:
וַאֲמַר דָוִד לְאָכִישׁ בְּכֵן אַתְּ תְּדַע יַת דִי יַעְבַּד עַבְדָךְ וַאֲמַר אָכִישׁ לְדָוִד בְּכֵן נַטְרָא לְרֵישֵׁי אֲשַׁוִינָךְ כָּל יוֹמַיָא:
וּשְׁמוּאֵל מִית וּסְפָדוּ עֲלוֹהִי כָּל יִשְׂרָאֵל וּקְבָרוֹהִי בְרָמָתָא וּסְפָדוּ עֲלוֹהִי אֱנָשׁ בְּקַרְתֵּיהּ וְשָׁאוּל פַּלִי יַת בִּידִין וּזְכוּרוּ מִן אַרְעָא:
וְאִתְכְּנִישׁוּ פְּלִשְׁתָּאֵי וַאֲתוֹ וּשְׁרוֹ בְשׁוּנֵם וּכְנַשׁ שָׁאוּל יַת כָּל יִשְׂרָאֵל וּשְׁרוֹ בַּגִלְבּוֹעַ:
וַחֲזָא שָׁאוּל יַת מַשִׁרְיַת פְּלִשְׁתָּאֵי וּדְחִיל וְזָע לִבֵּיהּ לַחֲדָא:
וּשְׁאִיל שָׁאוּל בְּמֵימְרָא דַייָ וְלָא קַבֵּל צְלוֹתֵיהּ יְיָ אַף בְּחֶלְמַיָא אַף בְּאוּרַיָא אַף בְּסַפְרַיָא:
וַאֲמַר שָׁאוּל לְעַבְדוֹהִי בְּעוֹ לִי אִתְּתָא דְיַדְעָא לְאַסָקָא בִּידִין וֶאֱהַךְ לְוָתָהּ וְאֶשְׁאַל בָּהּ וַאֲמַרוּ עַבְדוֹהִי לֵיהּ הָא אִתְּתָא דְיַדְעָא לְאַסָקָא בִידִין בְּעֵין דוֹר:
וְאִשְׁתַּנֵי שָׁאוּל וּלְבַשׁ לְבוּשִׁין אוֹחֲרָנִין וַאֲזַל הוּא וּתְרֵין גַבְרָא עִמֵיהּ וַאֲתוֹ לְוַת אִתְּתָא בְּלֵילְיָא וַאֲמַר קָסְמִי כְעַן לִי בְּבִידִין וְאַסְקֵי לִי יַת דְאֵימַר לָךְ:
וַאֲמַרַת אִתְּתָא לֵיהּ הָא אַתְּ יָדַעְתָּא יַת דַעֲבַד שָׁאוּל דְשֵׁיצֵי יַת בִּידִין וְיַת זְכוּרוּ מִן אַרְעָא וּלְמָא אַתְּ מִתְגָרֵי בְּנַפְשִׁי לְמִקְטְלִי:
וְקַיֵם לָהּ שָׁאוּל בְּמֵימְרָא דַייָ לְמֵימַר קַיָם הוּא יְיָ אִם יְעַרְעִינָךְ חוֹבָא בְּפִתְגָמָא הָדֵין:
וַאֲמַרַת אִתְּתָא יַת מַן אַסִיק לָךְ וַאֲמַר יַת שְׁמוּאֵל אַסִיקִי לִי:
וַחֲזַת אִתְּתָא יַת שְׁמוּאֵל וּצְוָחַת בְּקַל רַב וַאֲמָרַת אִתְּתָא לְשָׁאוּל לְמֵימָר לְמָה שְׁקַרְתְּ בִּי וְאַתְּ שָׁאוּל:
וַאֲמַר לָהּ מַלְכָּא לָא תִדְחֲלִין אֲרֵי מֶה חָזִית וַאֲמַרַת אִתְּתָא לְשָׁאוּל מַלְאָכָא דַייָ חֲזֵיתִי דְסָלִיק מִן אַרְעָא:
וַאֲמַר לָהּ מֶה חֶזְוֵיהּ וַאֲמַרַת גַבְרָא סָבָא סְלִיק וְהוּא עַטִיף מְעִילָא וִידַע שָׁאוּל אֲרֵי שְׁמוּאֵל הוּא וּכְרַע עַל אַפּוֹהִי עַל אַרְעָא וּסְגִיד:
וַאֲמַר שְׁמוּאֵל לְשָׁאוּל לְמָה אַזְעַתָּנִי לְאַסָקָא יָתִי וַאֲמַר שָׁאוּל עָקַת לִי לַחֲדָא וּפְלִשְׁתָּאֵי מְגִיחִין קְרָבָא בִּי וּמֵימְרָא דַייָ רָחִיק יָתִי וְלָא קַבֵּיל צְלוֹתִי עוֹד אַף בְּיַד סַפְרַיָא אַף בְּחֶלְמַיָא וּקְרֵית לָךְ לְהוֹדָעוּתַנִי מֶה אַעְבֵּיד:
וַאֲמַר שְׁמוּאֵל וּלְמָא אַתְּ שָׁאֵיל יָתִי וּמֵימְרָא דַייָ רָחִיק יָתָךְ וַהֲוָה בְּסַעֲדֵיהּ דְגַבְרָא דִי אַתְּ בְּעֵיל דְבָבֵיהּ:
וַעֲבַד יְיָ לֵיהּ כְּמָא דְמַלֵיל בִּידִי וְאַעֲדֵי יְיָ יַת מַלְכוּתָא מִנָךְ וְיַהֲבָהּ לְחַבְרָךְ לְדָוִד:
כְּמָא דְלָא קַבֵּילְתָּא בְּמֵימְרָא דַייָ וְלָא עֲבַדְתָּא תְקֵיף רוּגְזֵיהּ בַּעֲמָלֵק עַל כֵּן פִּתְגָמָא הָדֵין עֲבַד לָךְ יְיָ יוֹמָא הָדֵין:
וְיִמְסַר יְיָ אַף יַת יִשְׂרָאֵל עִמָךְ בִּידָא דִפְלִשְׁתָּאֵי וּמְחַר אַתְּ וּבְנָךְ עִמִי אַף יַת מַשְׁרִיתָא דְיִשְׂרָאֵל יִמְסַר יְיָ בִּידָא דִפְלִשְׁתָּאֵי:
וְאוֹחֵי שָׁאוּל וּנְפַל מְלֵי קוֹמָתֵיהּ לְאַרְעָא וּדְחִיל לַחֲדָא מִפִּתְגָמֵי שְׁמוּאֵל אַף חֵילָא לָא הֲוָה בֵיהּ אֲרֵי לָא אֲכָל לַחְמָא כָּל יוֹמָא וְכָל לֵילְיָא:
וְאָתַת אִתְּתָא לְוַת שָׁאוּל וַחֲזַת אֲרֵי אִתְבָהֵיל לַחֲדָא וַאֲמַרַת לֵיהּ הָא קַבִּילַת אַמְתָךְ מִנָךְ וּמְסָרִית נַפְשִׁי כְּעַל גַב לְאִתְקְטָלָא וּשְׁמָעִית יַת פִּתְגָמָךְ דְמַלֵילְתָּא עִמִי:
וּכְעַן קַבֵיל כְּעַן אַף אַתְּ לְמֵימָר אַמְתָךְ וַאֲשַׁוֵי קֳדָמָךְ פִּתָּא דְלַחְמָא וֶאֱכוֹל וִיהֵי בָךְ חֵילָא אֲרֵי תֵיזֵיל בְּאוֹרְחָא:
וְסָרִיב וַאֲמַר לָא אֵכוּל וְאַתְקִיפוּ בֵיהּ עַבְדוֹהִי וְאַף אִתְּתָא וְקַבֵּיל לְמֵימְרֵיהוֹן וְקָם מֵאַרְעָא וְיָתֵיב עַל עַרְסָא:
וּלְאִתְּתָא עֵגֶל פַּטִים בְּבֵיתָא וְאוֹחִיאַת וּנְכֵסְתֵּיהּ וּנְסִיבַת קִמְחָא וְלָשַׁת וַאֲפִיתֵיהּ פַּטִיר:
וְקָרֵיבַת קֳדָם שָׁאוּל וְקָדָם עַבְדוֹהִי וַאֲכַלוּ וְקָמוּ וַאֲזַלוּ בְּלֵילְיָא הַהוּא:
