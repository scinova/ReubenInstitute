מַטַל מַשִׁרְיָן דְאָתְיָן מִמַדְבְּרָא כְּמֵי יַמָא דְנָגְדִין כְּעַלְעוֹלִין דְאָתָן בְּאוֹרַח דָרוֹמָא מַדְבְּרָא וְאַתְיָן מִמַדְבְּרָא מֵאַרְעָא דְאִתְעֲבִידָא בָּהּ חֲסִינִין:
אֲמַר נְבִיָא נְבוּאָה קָשְׁיָא אִתְחַוָאָה לִי אֲנִיסַיָא מִתְאַנְסִין וּבָזְזַיָא מִתְבַּזְזִין סַקוּ עֵילָמָאֵי תְקִיפוּ מְדָאֵי כָּל דַהֲווֹ מִתְאַנְחִין מִן קֳדָם מַלְכָּא דְבָבֶל אַנְחֵית לְהוֹן:
עַל כֵּן אִתְמְלִיאוּ חַרְצֵיהוֹן זִיעַיָא דַחֲלָא אֲחַדְתִּינוּן הָא כְּחֶבְלִין לְיָלֵידְתָּא אִטַפָּשׁוּ מִלְמִשְׁמַע טָעוּ מִלְמֶחֱזֵי:
טְעָה לִבְּהוֹן עָקָא וּבִיעוּתִין אַחֲדוּנוּן אֲתַר בֵּית רוֹחְצָנֵיהוֹן הֲוָה לְהוֹן לִתְבַר:
סַדַרוּ פָּתוֹרִין אֲקִימוּ סַכְוָאִין אֲכוֹלוּ שְׁתוּ קוּמוּ רַבְרְבַיָא מְרִיקוּ וְצַחְצַחוּ זְיָנָא:
אֲרֵי כִדְנַן אֲמַר לִי יְיָ אֱזֵיל אֲקֵים סַכְוָאָה דְיֶחֱזֵי חַוֵי:
וְיֶחֱזֵי רְתַךְ אֱנַשׁ וְעִמֵהּ זוֹג פָּרָשִׁים רָכֵב עַל חֲמַר רָכִב עַל גַמְלָא אֲמַר נְבִיָא אֲצֵיצִית אֲצָתָא וַחֲזֵית וְהָא מַשִׁרְיָן לַחֲדָא:
אֲמַר נְבִיָא קַל מַשִׁרְיָן דְאַתְיָן בְּסִיטוּנֵיהוֹן כְּאַרְיָא וְעַל סָכוּתָא קֳדָם יְיָ אֲנָא קָאִים תְּדִירָא בִּימָמָא וְעַל מַטַרְתִּי אֲנָא מְעַתַּד כָּל לֵילְיָא:
וְהָא דֵין אָתָא רְתַךְ אֱנַשׁ וְעִמֵיהּ זוֹג פָּרָשִׁין וְאָתִיב וַאֲמַר נְפַלַת וְאַף עֲתִידָא לְמִפַּל בָּבֶל וְכָל צַלְמֵי טַעֲוָתָהָא יְדַקְדְקוּן לְאַרְעָא:
מַלְכִין דְאוּמָנִין לְאַגָחָא קְרָבָא יֵיתוּן עֲלָהּ לְמִבְזָהָא כְּאִכָּרָא דְאוּמָן לְמֵידַשׁ יַת אִדְרָא אֲמַר נְבִיָא דִשְׁמַעֵית יַת קַל מֵימְרָא דַייָ צְבָאוֹת אֱלָהָא דְיִשְׂרָאֵל חַוֵיתִי לְכוֹן:
מַטַל כָּס דִלְוָט לְאַשְׁקָאָה יַת דוּמָה עֲלֵי אַכְלֵי מִן שְׁמַיָא נְבִיָא פָּרֵישׁ לְהוֹן יַת נְבוּאֲתָא נְבִיָא פָּרֵישׁ לְהוֹן מָה דַעֲתִיד לְמֵיתֵי:
אֲמַר נְבִיָא אִית אֲגַר לְצַדִיקַיָא וְאִית פּוּרְעֲנוּת לְרַשִׁיעַיָא אִם תַּיְבִין אַתּוּן תּוּבוּ עַד דְאַתּוּן יָכְלִין לְמֵגָתָב:
מַטַל כָּס דִלְוָט לְאַשְׁקָאָה יַת עֲרָבָאֵי בְּחוּרְשָׁא בְרַמְשָׁא יְבִיתִין שְׁיָרַת בְּנֵי דְדָן:
לְאַפֵּי צַחְיָא פּוּקוּ נַגְדִין דְמַיָא דְיָתְבֵי בְּאַרְעָא דְרוֹמָא לַחְמָא דְאַתּוּן אָכְלִין זַמִינוּ לִמְעַרְקַיָא:
אֲרֵי מִן קֳדָם קְטוֹל עֲרָקוּ מִן קֳדָם חַרְבָּא שְׁלִיפָא וּמִן קֳדָם קֶשֶׁת מְתִיחָא וּמִן קֳדָם תְּקוֹף קְרָבָא:
אֲרֵי כִדְנַן אֲמַר יְיָ לִי בְּסוֹף שְׁנַיָא כִשְׁנֵי אֲגִירָא וְיָסוּף כָּל יְקָרְהוֹן דַעֲרָבָאֵי:
וּתְקוֹף עָבְדֵי קְרָב גִבָּרֵי בְּנֵי עֲרָבָאֵי יִזְעֲרוּן אֲרֵי בְּמֵימְרָא דַייָ אֱלָהָא דְיִשְׂרָאֵל גְזִיר כֵּן:
