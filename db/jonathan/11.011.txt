וַעֲתַלְיָה אִמֵיהּ דַאֲחַזְיָה חֲזַת אֲרֵי מִית בְּרָהּ וְקָמַת וְאַבְדַת יַת כָּל זַרְעָא דְמַלְכוּתָא:
וּנְסִיבַת יְהוֹשֶׁבַע בַּת מַלְכָּא יוֹרָם אֲחָתֵיהּ דַאֲחַזְיָה יַת יוֹאָשׁ בַּר אֲחַזְיָה וּגְנָבַת יָתֵיהּ מִגוֹ בְּנֵי מַלְכָּא דְמִתְקַטְלִין וְאַטְמָרַת יָתֵיהּ וְיַת מֵנִקְתֵּיהּ בְּאִידְרוֹן בֵּית עַרְסָתָא וְאַטְמָרוּ יָתֵיהּ מִן קֳדָם עֲתַלְיָה וְלָא אִתְקְטֵיל:
וַהֲוָה עִמָה בְּבֵית מַקְדְשָׁא דַייָ מִטַמַר שִׁית שְׁנִין וַעֲתַלְיָה מַלְכַת עַל אַרְעָא:
וּבְשַׁתָּא שְׁבִיעֲתָא שְׁלַח יְהוֹיָדָע וּדְבַר יַת רַבָּנֵי מַאֲוָתָא לְגִבָּרַיָא וּלְרָהֲטַיָא וְאָעֵיל יַתְהוֹן לְוָתֵיהּ לְבֵית מַקְדְשָׁא דַייָ וּגְזַר לְהוֹן קְיָם וְאוֹמֵי יַתְהוֹן בְּבֵית מַקְדְשָׁא דַייָ וְאַחְזֵי יַתְהוֹן יַת בַּר מַלְכָּא:
וּפַקְדִינוּן לְמֵימַר דֵין פִּתְגָמָא דְתַעְבְּדוּן תִּלְתָּא מִנְכוֹן עָלֵי שַׁבָּא וְנַטְרֵי מַטְרַת בֵּית מַלְכָּא:
וְתִלְתָּא בִּתְרַע גִבָּרַיָא וְתִלְתָּא בְּתַרְעָא דְבַתְרוֹהִי תְּרַע רָהֲטַיָא וְתִטְרוּן יַת מַטְרַת בֵּיתָא מִדְאִשְׁתְּלֵי:
וּתְרֵין חוּלָקִין בְּכוֹן כָּל נַפְקֵי שַׁבָּא וְתִטְרוּן יַת מַטְרַת בֵּית מַקְדְשָׁא דַייָ עַל מַלְכָּא:
וְתַקְפוּן עַל מַלְכָּא סְחוֹר סְחוֹר גְבַּר וּזְיָנֵיהּ בִּידֵיהּ וּדְיֵעוּל לְגוֹ מִן סִדְרָא יִתְקְטֵיל וַהֲווֹ עִם מַלְכָּא בְּמִפְקֵיהּ וּבְמֵיעֲלֵיהּ:
וַעֲבַדוּ רַבָּנֵי מַאֲוָתָא כְּכֹל דְפַקֵיד יְהוֹיָדָע כַּהֲנָא וּדְבָרוּ גְבַר יַת גַבְרוֹהִי עָלֵי שַׁבְּתָא עִם נַפְקֵי שַׁבְּתָא וַאֲתוֹ לְוַת יְהוֹיָדָע כַּהֲנָא:
וִיהַב יְהוֹיָדָע כַּהֲנָא לְרַבָּנֵי מַאֲוָתָא יַת מוּרְנִיתָא וְיַת שְׁלָטַיָא דִי לְמַלְכָּא דָוִד דִי בְּבֵית מַקְדְשָׁא דַיָי:
וְקָמוּ רָהֲטַיָא גְבַר וּזְיָנֵיהּ בִּידֵיהּ מֵעִיבַר בֵּיתָא מִימִינָא עַד עִיבַר בֵּיתָא מִסְמָלָא לְמַדְבְּחָא וּלְבֵית כַּפּוֹרֵי עַל מַלְכָּא סְחוֹר סְחוֹר:
וְאַפֵּיק יַת בַּר מַלְכָּא וִיהַב עֲלוֹהִי יַת כְּלִילָא וְיַת סַהֲדוּתָא וְאַמְלִיכוּ יָתֵיהּ וּמְשָׁחוּהִי וּטְפָחוּ יַד וַאֲמָרוּ יַצְלַח מַלְכָּא:
וּשְׁמַעַת עֲתַלְיָה יַת קַל דְרַקְדָנִין עַמָא וַאֲתָת לְוַת עַמָא לְבֵית מַקְדְשָׁא דַיָי:
וַחֲזַת וְהָא מַלְכָּא קָאִים עַל אִסְטְוָנָא כְּהִלְכָתָא וְרַבְרְבַיָא עִמֵיהּ וַחֲצוֹצְרָתָא קָדָם מַלְכָּא וְכָל עַמָא דְאַרְעָא חָדָן וְתָקְעִין בַּחְצוֹצְרָתָא וּבְזָעַת עֲתַלְיָה יַת לְבוּשָׁהָא וּקְרַת מֵירְדָא מֵירְדָא:
וּפַקֵיד יְהוֹיָדָע כַּהֲנָא יַת רַבָּנֵי מַאֲוָתָא דִמְמַנָן עַל חֵילָא וַאֲמַר לְהוֹן אֲפִּיקוּ יָתָהּ לְגָיו מִן סִדְרָא וּדְיֵיעוּל בַּתְרָהָא יִתְקְטֵיל בְּחַרְבָּא אֲרֵי אֲמַר כַּהֲנָא לָא תִתְקְטֵיל בְּבֵית מַקְדְשָׁא דַיְיָ:
וְשַׁוִיאוּ לָהּ אַתְרָא וְעָלַת בְּאוֹרַח מַעֲלָנָא דְסוּסַיָא לְבֵית מַלְכָּא וְאִתְקְטִלַת תַּמָן:
וּגְזַר יְהוֹיָדָע יַת קְיָמָא בֵּין מֵימְרָא דַייָ וּבֵין מַלְכָּא וּבֵין עַמָא לְמֶהֱוֵי לְעָם מְשַׁמֵשׁ קֳדָם יְיָ בֵּין מַלְכָּא וּבֵין עַמָא:
וְעָלוּ כָל עַמָא דְאַרְעָא לְבֵית בְּעָלָא וְסַתְּרוּהִי וְיַת אֱגוֹרוֹהִי וְיַת צַלְמוֹהִי תְּבִירוּ יָאוּת וְיַת מַתַּן כּוּמְרַיָא דִבְעָלָא קְטַלוּ קֳדָם אֱגוֹרָא וּמַנֵי כַּהֲנָא מַטְרָן עַל בֵּית מַקְדְשָׁא דַיְיָ:
וּדְבַר יַת רַבָּנֵי מַאֲוָתָא וְיַת גִבָּרַיָא וְיַת רָהֲטַיָא וְיַת כָּל עַמָא דְאַרְעָא וַאֲחִיתוּ יַת מַלְכָּא מִבֵּית מַקְדְשָׁא דַייָ וְעָלוּ בְּאוֹרַח תְּרַע רָהֲטַיָא דְבֵית מַלְכָּא וִיתֵיב עַל כּוּרְסֵי מַלְכַיָא:
וַחֲדִי כָל עַמָא דְאַרְעָא וְקַרְתָּא שְׁדוֹכַת וְיַת עֲתַלְיָה קְטַלוּ בְּחַרְבָּא בֵּית מַלְכָּא:
