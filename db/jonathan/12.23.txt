מַטָל כָּס דִלְוָט לְאַשְׁקָאָה יַת צוֹר אֵילִילוּ נָחֲתֵי סְפִינֵי יַמָא אֲרֵי אִתְבְּזִיזוּ מְחוֹזֵיהוֹן מִלְמֵיתֵי מֵאַרְעָא כִסְאֵי אָתֵי עֲלֵיהוֹן:
אִתְּבַּרוּ יָתְבֵי נִיסָא תַּגְרֵי צִידוֹן דַהֲווֹ עָדָן בְּיַמָא מַלוּךְ:
דַהֲוַת מְסַפְּקָא סְחוֹרָא לְעַמְמִין סַגִיאִין דְבַחֲצַד כְּנִישַׁת נַהֲרָא עֲלַלְתָּהּ נַהֲרָא סְחוֹרָא לְעַמְמַיָא:
בְּהִיתוּ צִידוֹנָאֵי אֲרֵי אֲמַר מְעַרְבָאָה דִי יָתֵיב בִּתְקוֹף יַמָא לְמֵימָר לְוַי לָא מַרְעֵית לָא אַעְדֵיתִי וְלָא רַבֵּיתִי עוּלֵמִין פַּנְקֵית בְּתוּלָן:
כְּמָא דִי שְׁמָעוּ לְמַחֲתָא דִלְקוֹ מִצְרָאֵי זָעוּ כַּד שְׁמַעוּ צוֹרָאֵי:
גְלוֹ לִמְדִינַת יַמָא אֵילִילוּ יָתְבֵי גִיסָא:
הֲהָדָא לְכוֹן תַּקִיפְתָּא צוֹר מִיוֹמֵי קֶדֶם מִלְקָדְמִין הֲווֹ מַיְתָן לָהּ תַּקְרוּבְתָּא מֵאֲרַע רְחִיקָא כְדוֹ לְאִתּוֹתָבָא הָא גַלְיָא:
מַן מְלִיךְ דָא עַל צוֹר דַהֲיַת מְמַלְכָּא מַלְכִין דִי תַגָרָהָא רַבְרְבִין שִׁלְטוֹנָאָה יְקִירֵי אַרְעָא:
יְיָ צְבָאוֹת מִלְכָהּ לְאַפָסָא יְקָר כָּל חֶדְוָתָא לְאַקָלָא כָּל יַקִירֵי אַרְעָא:
גְלָא מֵאַרְעִיךְ כְּמֵי נַהֲרָא עֲרוֹקִי לִמְדִינַת יַמָא לֵית תְּקוֹף עוֹד:
גְבוּרְתֵּהּ מְרַמְמָא עַל יַמָא לַאֲזָעָא מַלְכְּוָתָא יְיָ פַּקֵיד עַל כְּנַעֲנָאָה לְשֵׁיצָאָה תּוּקְפָאָה:
וַאֲמַר לָא תוֹסְפוּן עוֹד לְמִתְקַף דַהֲוֵיתוּן אָנְסִין לְעַמָא דִי בְצִידוֹן לְכִתִּים קוּמוּ גְלוֹ אַף לְתַמָן לָא יְהֵי לְכוֹן נְיָח:
הָא אַרְעָא כַּסְדָאֵי דֵין עַמָא כַּד לָא הֲוָה אַתּוּרָאָה שַׁכְלְלָהּ לְנַגְוַן אֲקִימוּ חֶזְוָתָהָא פַּגְרוּ בִרְנְיָתָהָא שַׁוְיוּהָא לִדְמְחַמְרָא:
אֵילִילוּ נָחֲתֵי סְפִינֵי יַמָא אֲרֵי אִתְבְּזִיזוּ מְחוֹזֵי תוּקְפֵכוֹן:
וִיהֵי בְּעִדָנָא הַהִיא וְתִתְרְחֵיק צוֹר שַׁבְעִין שְׁנִין כְּיוֹמֵי מַלְכָּא חַד מִסוֹף שַׁבְעִין שְׁנִין יְהֵי לְצוֹר כְּתוּשְׁבְּחַת נַפְקַת בָּרָא:
אִתְחַלֵיף יְקָרִיךְ אִטַלְטֵל לִמְדִינָא קַרְתָּא דִי הֲוֵת כְנַפְקַת בְּרָא אִתְרַחֲקַת הֲפוּכִי כִנָרַיִךְ לְאֶבְלָא וּזְמָרַיִךְ לְאִלְיָא מָאִים יֵעוּל דָכְרָנֵיךְ:
וִיהֵי מִסוֹף שַׁבְעִין שְׁנִין יֵעוּל דָכְרָנָהּ דְצוֹר קֳדָם יְיָ וּתְתוּב לְאַתְרָהּ וּתְהֵי מְסַפְּקָא סְחוֹרָא לְכָל מַלְכְּוַת עַמְמַיָא דִי עַל אַפֵּי אַרְעָא:
וִיהֵי סְחוֹרְתָּהּ וְאַגְרָהּ קוּדְשָׁא קֳדָם יְיָ לָא יִתְאַצֵר וְלָא יִתְגְנַז אֲרֵי לְדִמְשַׁמְשִׁין קֳדָם יְיָ יְהֵי אַגְרָה לְמֵיכַל לְמִסְבַּע וְלִכְסוּ דִיקָר:
