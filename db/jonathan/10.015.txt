וּבִשְׁנַת תַּמְנֵי עַסְרֵי לְמֶלֶךְ יְרָבְעָם בַּר נְבָט מְלַךְ אֲבִיָם עַל דְּבֵית יְהוּדָה:
תְּלַת שְׁנִין מְלַךְ בִּירוּשְׁלֵם וְשׁוּם אִמֵיהּ מַעֲכָה בַּת אֲבִישָׁלוֹם:
וַאֲזֵל בְּכָל חוֹבֵי אֲבוּהִי דַעֲבַד קָדָמוֹהִי וְלָא הֲוָה לִבֵיהּ שְׁלִים בִּדְחַלְתֵּיהּ דַייָ אֱלָהֵיהּ כְּלִבֵּיהּ דְדָוִד אֲבוּהִי:
אֲרֵי בְּדִיל דָוִד יְהַב יְיָ אֱלָהֵיהּ לֵיהּ מַלְכוּ בִּירוּשְׁלֵם לַאֲקָמָא יַת בְּרֵיהּ תְּחוֹתוֹהִי וּלְקַיָמָא יַת יְרוּשְׁלֵם:
דַעֲבַד דָוִד יַת דְכָשַׁר קֳדָם יְיָ וְלָא סְטָא מִכָּל דְפַקְדֵיהּ כָּל יוֹמֵי חַיוֹהִי לְחוֹד בְּפִתְגָמָא דְאוּרִיָה חִתָּאָה:
וּקְרָבָא הֲוַת בֵּין רְחַבְעָם וּבֵין יְרָבְעָם כָּל יוֹמֵי חַיוֹהִי:
וּשְׁאָר פִּתְגָמֵי אֲבִיָם וְכָל דַעֲבַד הֲלָא אִינוּן כְּתִיבִין עַל סְפַר פִּתְגָמֵי יוֹמַיָא לְמַלְכֵי דְבֵית יְהוּדָה וּקְרָבָא הֲוַת בֵּין אֲבִיָם וּבֵין יָרָבְעָם:
וּשְׁכִיב אֲבִיָם עִם אֲבָהָתוֹהִי וּקְבָרוּ יָתֵיהּ בְּקַרְתָּא דְדָוִד וּמְלַךְ אָסָא בְּרֵיהּ תְּחוֹתוֹהִי:
וּבִשְׁנַת עַסְרִין לְיָרָבְעָם מַלְכָּא דְיִשְׂרָאֵל מְלַךְ אָסָא עַל שִׁבְטָא דְבֵית יְהוּדָה:
וְאַרְבְּעִין וַחֲדָא שְׁנִין מְלַךְ בִּירוּשְׁלֵם וְשׁוּם אִמֵיהּ מַעֲכָה בַּת אֲבִישָׁלוֹם:
וַעֲבַד אָסָא דְכָשַׁר קֳדָם יְיָ כְּדָוִד אֲבוּהִי:
וּפַלֵי נַפְקַת בָּרָא מִן אַרְעָא וְאַעְדֵי יַת כָּל טַעֲוָתָא דַעֲבַדוּ אֲבָהָתוֹהִי:
וְאַף יַת מַעֲכָה אִמֵיהּ אַעְדָהּ מִמַלְכוּתָהּ דַעֲבַדַת טַעֲוָתָא לַאֲשֵׁרָתָא וְקָץ אָסָא יַת טַעֲוָתָא וְאוֹקֵיד בְּנַחְלָא דְקִדְרוֹן:
וּבָמָתָא לָא עַטְרָא לְחוֹד לִבָּא דְאָסָא הֲוָה שְׁלִים בִּדְחַלְתֵּיהּ דַייָ כָּל יוֹמוֹהִי:
וְאַיְתֵי יַת קוּדְשֵׁי אָבוּהִי וְקוּדְשׁוֹהִי לְבֵית מַקְדְשָׁא דַייָ כַּסְפָּא וְדַהֲבָא וּמָנַיָא:
וּקְרָבָא הֲוַת בֵּין אָסָא וּבֵין בַּעְשָׁא מַלְכָּא דְיִשְׂרָאֵל כָּל יוֹמֵיהוֹן:
וּסְלִיק בַּעְשָׁא מַלְכָּא דְיִשְׂרָאֵל עַל דְבֵית יְהוּדָה וּבְנָא יַת רָמָתָא בְּדִיל דְלָא לְמִשְׁבַּק לְמִתַּן נָפִיק וְעָלֵיל לְאָסָא מַלְכָּא דְבֵית יְהוּדָה:
וּנְסֵיב אָסָא יַת כָּל כַּסְפָּא וְדַהֲבָא דְאִשְׁתָּאֲרוּ בְּאוֹצְרוֹת בֵּית מַקְדְשָׁא דַייָ וְיַת אוֹצָרֵי בֵּית מַלְכָּא וּמְנִינוּן בִּידָא דְעַבְדוֹהִי וְשַׁדְרִינוּן מַלְכָּא אָסָא לְוַת בַּר הֲדַד בַּר טַבְרִימוֹן בַּר חֶזְיוֹן מַלְכָּא דַאֲרָם דְיָתֵיב בְּדַמֶשֶׂק לְמֵימָר:
קְיָם בֵּינִי וּבֵינָךְ וּבֵין אַבָּא וּבֵין אֲבוּךְ הָא שַׁדְרֵית לָךְ שׁוֹחָדָא כַּסְפָּא וְדַהֲבָא אֱזִיל אַשְׁנֵי יַת קְיָמָךְ עִם בַּעְשָׁא מַלְכָּא דְיִשְׂרָאֵל וְיִסְתַּלֵק מִנִי:
וְקַבֵּל בַּר הֲדַד מִן מַלְכָּא אָסָא וּשְׁדַר יַת רַבָּנֵי חֵילְוָתָא דִילֵיהּ עַל קִרְוֵי יִשְׂרָאֵל וּמְחָא יַת עִיוֹן וְיַת דָן וְיַת אָבֵל בֵּית מַעֲכָה וְיַת כָּל גִינוֹסַר עַל כָּל אֲרַע שִׁבְטָא דְבֵית נַפְתָּלִי:
וַהֲוָה כַּד שְׁמַע בַּעְשָׁא וְאִתְמְנַע מִלְמִבְנֵי יַת רָמָתָא וִיתֵב בְּתַרְעִיתָא:
וּמַלְכָּא אָסָא כְּנַשׁ יַת כָּל דְבֵית יְהוּדָה לֵית דִנְקֵי וּנְטַל יַת אַבְנֵי רָמָתָא וְיַת אֲעָהָא דִי בְנָא בַּעְשָׁא וּבְנָא בְהוֹן מַלְכָּא אָסָא יַת גִבְעֲתָא דִבְנֵי בִנְיָמִן וְיַת מִצְפַּיָא:
וּשְׁאַר כָּל פִּתְגָמֵי אָסָא וְכָל גְבוּרְתֵּיהּ וְכָל דִי עֲבַד וְקִירְוַיָא דִי בְנָא הֲלָא אִינוּן כְּתִיבִין עַל סְפַר פִּתְגָמֵי יוֹמַיָא לְמַלְכָּא דְבֵית יְהוּדָה לְחוֹד לְעִדַן סִיבְתֵיהּ אִתְמְרַע בְּרִגְלוֹהִי:
וּשְׁכִיב אָסָא עִם אֲבָהָתוֹהִי וְאִתְקְבַר עִם אֲבָהָתוֹהִי בְּקַרְתָּא דְדָוִד אֲבוּהִי וּמְלַךְ יְהוֹשָׁפָט בְּרֵיהּ תְּחוֹתוֹהִי:
וְנָדָב בַּר יָרָבְעָם מְלַךְ עַל יִשְׂרָאֵל בִּשְׁנַת תַּרְתֵּין לְאָסָא מַלְכָּא דְשִׁבְטָא דְבֵית יְהוּדָה וּמְלַךְ עַל יִשְׂרָאֵל תַּרְתֵּין שְׁנִין:
וַעֲבַד דְבִישׁ קֳדָם יְיָ וַאֲזַל בְּאוֹרַח אֲבוּהִי וּבְחוֹבֵיהּ דְחַיֵב יַת יִשְׂרָאֵל:
וּמְרַד עֲלוֹהִי בַּעְשָׁא בַר אֲחִיָה מִשִׁבְטָא דְבֵית יִשָׂשׂכָר וּמְחָהִי בַּעְשָׁא בְּגִבְּתוֹן דִי לִפְלִשְׁתָּאֵי וְנָדָב וְכָל יִשְׂרָאֵל צַיְרִין עַל גִבְּתוֹן:
וְקַטְלֵיהּ בַּעְשָׁא בִּשְׁנַת תְּלַת לְאָסָא מַלְכָּא דְשִׁבְטָא דְבֵית יְהוּדָה וּמְלַךְ תְּחוֹתוֹהִי:
וַהֲוָה כְמִמְלְכֵיהּ קְטַל יַת בֵּית יָרָבְעָם לָא אַשְׁאַר כָּל נִשְׁמָא לִיְרָבְעָם עַד דִי שֵׁיצֵיהּ כְּפִתְגָמָא דַייָ דְמַלֵיל בְּיַד עַבְדֵיהּ אֲחִיָה שִׁילוֹנָאָה:
עַל חוֹבֵי יָרָבְעָם דְחַב וּדְחַיֵב יַת יִשְׂרָאֵל בְּאַרְגְזָיוּתֵיהּ דְאַרְגֵיז קֳדָם יְיָ אֱלָהָא דְיִשְׂרָאֵל:
וּשְׁאָר פִּתְגָמֵי נָדָב וְכָל דִי עֲבַד הֲלָא אִינוּן כְּתִיבִין עַל סְפַר פִּתְגָמֵי יוֹמַיָא לְמַלְכֵי יִשְׂרָאֵל:
וּקְרָבָא הֲוַת בֵּין אָסָא וּבֵין בַּעְשָׁא מַלְכָּא דְיִשְׂרָאֵל כָּל יוֹמֵיהוֹן:
בִּשְׁנַת תְּלַת לְאָסָא מֶלֶךְ שִׁבְטָא דְבֵית יְהוּדָה מְלַךְ בַּעְשָׁא בַר אֲחִיָה עַל כָּל יִשְׂרָאֵל בְּתִרְצָה עַסְרִין וְאַרְבַּע שְׁנִין:
וַעֲבַד דְבִישׁ קֳדָם יְיָ וַאֲזַל בְּאוֹרַח יָרָבְעָם וּבְחוֹבֵיהּ דְחַב דִי חַיַב יַת יִשְׂרָאֵל:
