וַהֲוָה בָּתַר דְמִית יְהוֹשֻׁעַ וּשְׁאִילוּ בְּנֵי יִשְׂרָאֵל בְּמֵימְרָא דַייָ לְמֵימַר מַן יִסַק לָנָא לְוַת כְּנַעֲנָאָה בְּקַדְמֵיתָא לְאַגָחָא קְרָבָא בֵּיהּ:
וַאֲמַר יְיָ יְהוּדָה יִסַק הָא מַסְרִית יַת יַתְבֵי אַרְעָא בִּידֵיהּ:
וַאֲמַר יְהוּדָה לְשִׁמְעוֹן אֲחוּהִי סַק עִמִי בְּעַדְבִי וּנְגִיחַ קְרָבָא עִם כְּנַעֲנָאָה וְאֵהַךְ אַף אֲנָא עִמָךְ בְּעַדְבָךְ וַאֲזַל עִמֵיהּ שִׁמְעוֹן:
וּסְלִיקוּ דְבֵית יְהוּדָה וּמְסַר יְיָ יַת כְּנַעֲנָאָה וּפְרִיזָאָה בִּידֵיהוֹן וּמְחָנוּן בְּבֶזֶק עַסְרָא אַלְפִין גַבְרָא:
וְאַשְׁכָּחוּ יַת אֲדֹנִי בֶזֶק בְּבֶזֶק וְאַגִיחוּ קְרָבָא בֵּיהּ וּמְחוֹ יַת כְּנַעֲנָאֵי וְיַת פְּרִיזָאֵי:
וַעֲרַק אֲדֹנִי בֶזֶק וּרְדָפוּ בַּתְרוֹהִי וַאֲחַדוּ יָתֵיהּ וְקַצִיצוּ יַת קַרְסֻלֵי יְדוֹהִי וְרַגְלוֹהִי:
וַאֲמַר אֲדֹנִי בֶזֶק שַׁבְעִין מַלְכִין קַרְסֻלֵי יְדֵיהוֹן וְרַגְלֵיהוֹן מְקַצְצִין הֲווֹ מְלַקְטִין לַחְמָא תְּחוֹת פָּתוֹרִי כְּמָא דְעַבְדִית כֵּן שַׁלֵים לִי יְיָ וְאַיְתוֹהִי לִירוּשְׁלֵם וּמִית תַּמָן:
וְאַגִיחוּ קְרָבָא בְנֵי יְהוּדָה בִּירוּשְׁלֵם וּכְבַשׁוּ יָתָהּ וּמְחוֹהָא לְפִתְגַם דְחָרֶב וְיַת קַרְתָּא אוֹקִידוּ בְנוּרָא:
וּבָתַר כֵּן נְחָתוּ בְּנֵי יְהוּדָה לְאַגָחָא קְרָבָא בִּכְנַעֲנָאֵי יָתֵיב טוּרָא וְדָרוֹמָא וּשְׁפֶלְתָּא:
וַאֲזַלוּ דְבֵית יְהוּדָה לְוַת כְּנַעֲנָאֵי דְיַתְבִין בְּחֶבְרוֹן וְשׁוּם חֶבְרוֹן מִלְקַדְמִין קִרְיַת אַרְבַּע וּמְחוֹ יַת שֵׁשַׁי וְיַת אֲחִימָן וְיַת תַּלְמָי:
וַאֲזַלוּ מִתַּמָן לְוַת יַתְבֵי דְבִיר וְשׁוּם דְבִיר מִלְקַדְמִין קִרְיַת אַרְכֵי:
וַאֲמַר כָּלֵב דְיִמְחֵי יַת קִרְיַת אַרְכֵי וְיִכְבְּשִׁינָהּ וְאֶתֵּן לֵיהּ יַת עַכְסָה בְּרַתִּי לְאִתּוּ:
וְכַבְּשָׁהּ עָתְנִיאֵל בַּר קְנַז אֲחוֹהִי דְכָלֵב דִזְעֵיר מִנֵיהּ וִיהַב לֵיהּ יַת עַכְסָה בְרַתֵּיהּ לְאִתּוּ:
וַהֲוָה בְּמֵעֲלָהּ וְאַמְלַכְתֵּיהּ לְמִשְׁאַל מִן אֲבוּהָא אַחְסַנְתָּא וְאִתְרְכִינַת מֵעַל חֲמָרָא וַאֲמַר לָהּ כָּלֵב מַה לִיךְ:
וַאֲמָרַת לֵיהּ הַב לִי אַחֲסַנְתָּא אֲרֵי לַאֲרַע דָרוֹמָא יְהַבְתָּנִי וְתִתֶּן לִי אֲתַר בֵּית שַׁקְיָא דְמַיָא וִיהַב לָהּ כָּלֵב יַת בֵּית שַׁקְיָא עִלָאָה וְיַת בֵּית שַׁקְיָא תַּחְתָּאָה:
וּבְנֵי שַׁלְמָאָה חָמוּהִי דְמֹשֶׁה סְלִיקוּ מִן קַרְתָּא יְרִיחוֹ עִם בְּנֵי יְהוּדָה מִדְבַּר יְהוּדָה דִי בְּדָרוֹם עֲרָד וַאֲזַלוּ וִיתַבוּ עִם עַמָא:
וַאֲזַל יְהוּדָה עִם שִׁמְעוֹן אֲחוֹהִי וּמְחוֹ יַת כְּנַעֲנָאָה יָתֵיב צְפַת וְגַמַרוּ יָתָהּ וּקְרָא יַת שְׁמָא דְקַרְתָּא חָרְמָה:
וּכְבַשׁ יְהוּדָה יַת עַזָה וְיַת תְּחוּמָא וְיַת אַשְׁקְלוֹן וְיַת תְּחוּמָא וְיַת עֶקְרוֹן וְיַת תְּחוּמָא:
וַהֲוָה מֵימְרָא דַייָ בְּסַעֲדֵיהּ דְבֵית יְהוּדָה וְתָרִיכוּ יַת יַתְבֵי טוּרָא וּבָּתַר כֵּן דְחָבוּ לָא יְכִילוּ לְתָרָכָא יַת יַתְבֵי מֵישְׁרָא אֲרֵי דְתִיכִין דְבַרְזְלָא לְהוֹן:
וִיהַבוּ לְכָלֵב יַת חֶבְרוֹן כְּמָא דְמַלֵיל מֹשֶׁה וְתָרֵיךְ מִתַּמָן יַת תְּלָתָא בְּנֵי גִבָּרַיָא:
וְיַת יְבוּסָאֵי יַתְבֵי יְרוּשְׁלֵם לָא תָרִיכוּ בְּנֵי בִנְיָמִין וִיתִיבוּ יְבוּסָאֵי עִם בְּנֵי בִנְיָמִין בִּירוּשְׁלֵם עַד יוֹמָא הָדֵין:
וּסְלִיקוּ דְבֵית יוֹסֵף אַף אִנוּן לְבֵית אֵל וּמֵימְרָא דַייָ בְּסַעְדְהוֹן:
וּשְׁלַחוּ מְאַלְלִין דְבֵית יוֹסֵף לְבֵית אֵל וְשׁוּם קַרְתָּא מִלְקַדְמִין לוּז:
וַחֲזוֹ נָטְרַיָא גַבְרָא נָפֵיק מִן קַרְתָּא וַאֲמַרוּ לֵיהּ אַחֲזֵינָא כְעַן יַת מַעֲלָנָא דְקַרְתָּא וְנַעְבֵּד עִמָךְ טִיבוּ:
וְאַחֲזִינוּן יַת מַעֲלָנָא דְקַרְתָּא וּמְחוֹ יַת קַרְתָּא לְפִתְגַם דְחָרֶב וְיַת גַבְרָא וְיַת כָּל זַרְעִיתֵיהּ שֵׁזִיבוּ:
וַאֲזַל גַבְרָא לְאַרְעָא חִתָּאֵי וּבְנָא קַרְתָּא וּקְרָא שְׁמָהּ לוּז הוּא שְׁמָהּ עַד יוֹמָא הָדֵין:
וְלָא תָרִיכוּ דְבֵית מְנַשֶׁה יַת בֵּית שְׁאָן וְיַת כַּפְרְנָהָא וְיַת תַּעֲנָךְ וְיַת כַּפְרְנָהָא וְיַת יַתְבֵי דוֹר וְיַת כַּפְרְנָהָא וְיַת יַתְבֵי יִבְלְעָם וְיַת כַּפְרְנָהָא וְיַת יַתְבֵי מְגִדוֹ וְיַת כַּפְרְנָהָא וּשְׁרֵי כְנַעֲנָאָה לְמִתַּב בְּאַרְעָא הָדָא:
וַהֲוָה כַּד תַּקֵיף יִשְׂרָאֵל וְשַׁוֵי יַת כְּנַעֲנָאָה לְמַסְקֵי מִסִין וְתָרָכָא לָא תָרְכִינוּן:
וּדְבֵית אֶפְרַיִם לָא תָרִיךְ יַת כְּנַעֲנָאֵי דְיַתְבִין בְּגָזֶר וִיתִבוּ כְּנַעֲנָאֵי בֵינֵיהוֹן בְּגָזֶר:
דְּבֵית זְבוּלוּן לָא תָרִיכוּ יַת יַתְבֵי קִטְרוֹן וְיַת יַתְבֵי נַהֲלֹל וִיתִבוּ כְנַעֲנָאֵי בֵּינֵיהוֹן וַהֲווֹ לְמַסְקֵי מִסִין:
דְבֵית אָשֵׁר לָא תָרִיכוּ יַת יַתְבֵי עַכּוֹ וְיַת יַתְבֵי צִידוֹן וְיַת אַחְלָב וְיַת אַכְזִיב וְיַת חֶלְבָּה וְיַת אֲפִיק וְיַת רְחוֹב:
וִיתִיבוּ דְבֵית אָשֵׁר בְּגוֹ כְּנַעֲנָאֵי יַתְבֵי אַרְעָא אֲרֵי לָא תָרְכִינוּן:
דְבֵית נַפְתָּלִי לָא תָרִיכוּ יַת יַתְבֵי בֵית שֶׁמֶשׁ וְיַת יַתְבֵי בֵית עֲנָת וִיתִיבוּ בְּגוֹ כְנַעֲנָאֵי יַתְבֵי אַרְעָא וְיַתְבֵי בֵית שֶׁמֶשׁ וּבֵית עֲנָת הֲווֹ לְהוֹן לְמַסְקֵי מִסִין:
וּדְחַקוּ אֱמוֹרָאֵי יַת בְּנֵי דָן לְטוּרָא אֲרֵי לָא שַׁבְקוּנוּן לְמֵחַת לְמֵישְׁרָא:
וּשְׁרִי אֱמוֹרָאָה לְמִתַּב בְּטוּר חֶרֶס בְּאַיָלוֹן וּבְשַׁעַלְבִים וּתְקֵיפַת יַד בֵּית יוֹסֵף וַהֲווֹ לְמַסְקֵי מִסִין:
וּתְחוּם אֱמוֹרָאָה מִמַסְקְנֵי עַקְרַ בִּים מִן כֵּיפָא וּלְעֵילָא:
