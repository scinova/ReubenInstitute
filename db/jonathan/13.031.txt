בְּעִדָנָא הַהִיא אֲמַר יְיָ אֶהֱוֵי לֵאלָהָא לְכָל זַרְעֲיַת יִשְׂרָאֵל וְאִינוּן יְהוֹן קֳדָמָי לְעָם:
כִּדְנַן אֲמַר יְיָ דִיהַב רַחֲמִין לְעַמָא דִנְפַק מִמִצְרַיִם סִפּוּק צָרְכֵיהוֹן בְּמַדְבְּרָא כַּד הֲווֹ מְעָרְקִין מִן קֳדָם דְקַטְלִין בְּחַרְבָּא דַבֵּר בְּמֵימְרֵיהּ לְאַשְׁרָאָה בֵּית נְיָחָא יִשְׂרָאֵל:
אֲמַרֵית יְרוּשְׁלֵם מִלְקַדְמִין יְיָ אִתְגְלֵי עַל אֲבָהָתָנָא נְבִיָא אֲמַר לְהוֹן הָא רַחֲמַת עַלְמָא רְחֵימַת יַתְכוֹן עַל כֵּן דַבָּרִית יַתְכוֹן בְּטַבְוָן:
עוֹד אֲקֵימִינָךְ וְתִתְקַיְמִין כְּנִשְׁתָּא דְיִשְׂרָאֵל עוֹד תִּתְקְנִין בְּתִקֻנִיךְ וְתִפְקִין בְסִיעַת מְשַׁבְּחִין:
עוֹד תִּצְבִין כַּרְמִין בְּטוּרֵי שֹׁמְרוֹן צוּבוּ שְׁתִילִין וְאֲחִילוּ:
אֲרֵי אִית אֲרִיכוּת יוֹמַיָא וְסַגִי טוּבָא דַעֲתִיד לְמֵיתֵי לְצַדִיקַיָא דִנְטָרוּ אוֹרַיְתִי מִלְקַדְמִין חוּלְקֵיהוֹן בְּאַרְעָא דְיִשְׂרָאֵל דַהֲווֹ מְחַמְדִין לִשְׁנֵי נֶחֶמְתָא דְאַתְיָן דְאָמְרִין אֵימָתַי נָקוּם וְנִסַק לְצִיוֹן וְנִתְחֲזֵי קֳדָם יְיָ אֱלָהָנָא:
אֲרֵי כִדְנַן אֲמַר יְיָ שַׁבָּחוּ דְבֵית יַעֲקֹב בְּחֶדְוָא וְדוּצוּ בְּרֵישׁ גְלֵי לְעֵינֵי כָּל עַמְמַיָא בְּשַׂרוּ שְׁבָחוּ וֶאֱמָרוּ פְּרֵיק יְיָ יַת עַמֵיהּ יַת שְׁאָרָא דְיִשְׂרָאֵל:
הָאֲנָא מַיְתֵי לְהוֹן מֵאַרַע צִפּוּנָא וַאֲכַנְשִׁינוּן מִסְיָפֵי אַרְעָא דַהֲווֹ בְהוֹן עִוְרִין וְחִגְרִין יִסְגְיָן בְּהוֹן מְעִדְיָן וְיָלְדָן כַחֲדָא בְּמַשִׁרְיָן סַגִיאָן יְתוּבוּן הַלְכָא:
בְּמִגְלְהוֹן כַּד הֲווֹ בָכָן גְלוֹ בִּמְתָבְהוֹן מִבֵּינֵי גַלְוַתְהוֹן בְּרַחֲמִין סַגִיאִין אֱקָרְבִינוּן אֱדַבְּרִינוּן עַל פְּצִירֵי מַיָא בְּאוֹרַח תַּקְנָא לָא יִתְקְלוּן בָּהּ אֲרֵי הֲוָה מֵימְרִי לְיִשְׂרָאֵל כְּאָב וְאֶפְרַיִם חָבִיב הוּא קָדְמַי:
קַבִּילוּ פִתְגָמָא דַייָ עַמְמַיָא וַחֲווּ בְגַוְוָן מֵרָחִיק וֶאֱמָרוּ מַן דְבַדְרֵיהּ לְיִשְׂרָאֵל יְכַנְשִׁינֵיהּ וְיִטְרִינֵיהּ כְרָעֲיָא לְעֶדְרֵיהּ:
אֲרֵי פָרֵיק יְיָ יַת דְבֵית יַעֲקֹב וְשֵׁיזְבִינוּן מִיַד דְתַקִיפִין מִנְהוֹן:
וְיֵיתוּן וִישַׁבְּחוּן בְּטוּר בֵּית מַקְדְשָׁא דְאִתְבְּנֵי בְצִיוֹן וְיִתְפַּנְקוּן בְּטוּבָא דִיהַב לְהוֹן יְיָ בְּעִבּוּרָא וּבַחֲמָרָא וּבְמִשְׁחָא וּבְנֵי עָאן וְתוֹרִין וּתְהֵי נַפְשְׁהוֹן מַלְיָא תַפְנוּקִין כְּגִנַת שַׁקְיָא דִמְרַוְיָא וְלָא יוֹסְפוּן לְיַצָפָא עוֹד:
בְּכֵן תֶּחְדִין כְּנִשְׁתָּא דְיִשְׂרָאֵל כַּד בְּחִנְגִין וְעוּלֵימִין וְסָבִין כַּחֲדָא וֶאֶהֱפוֹךְ אֶבְלְהוֹן לְבִיעַ וֶאֱנָחֵמִינוּן וְאַחֲדִינוּן מִדְווֹנְהוֹן:
וְאֶשְׂבַּע נְפַשׁ כַּהֲנַיָא תַּפְנוּקִין וְעַמִי בֵּית יִשְׂרָאֵל יַת טוּבִי דִיהָבִית לְהוֹן יִשְׂבְּעוּן אֲמַר יְיָ:
כִּדְנַן אֲמַר יְיָ קָלָא בְרוּם עַלְמָא אִשְׁתְּמַע בֵּית יִשְׂרָאֵל דְבָכָן וּמִתְאַנְחָן בָּתַר יִרְמְיָה נְבִיָא כַּד שְׁלַח יָתֵיהּ נְבוּזַרְאֲדָן רַב קָטוֹלַיָא מֵרָמָתָא אִלְיָא וּדְבָכָן בִּמְרַר יְרוּשְׁלֵם מְבַכְיָא עַל בְּנָהָא מְסַרְבָא לְאִתְנֶחָמָא עַל בְּנָהָא אֲרֵי גְלוֹ:
כִּדְנַן אֲמַר יְיָ מַנְעִי קָלִיךְ מִלְמִבְכֵּי וְעֵינַיִךְ מִלְשַׁגְרָא דִמְעָן אֲרֵי אִית אֲגַר לְעוֹבָדַי אַבְהָתָךְ צַדִיקַיָא אֲמַר יְיָ וִיתוּבוּן מֵאַרַע בַּעֲלֵי דְבָבֵיהוֹן:
וְאִית סִבוּר לְסוֹפִיךְ אֲמַר יְיָ וִיתוּבוּן בָּנַיָא לִתְחוּמְהוֹן:
שְׁמִיעַ וּגְלֵי קָדָמַי בֵּית יִשְׂרָאֵל דְבָבָן וּמִתְאַנְחִין עַל דְאִטַלְטָלוּ אָמְרִין אַיְתֵיתָא עֲלָן יְסוּרִין וְלָא אֲלֵיפְנָא כְּעֵגַל לָא מַלִיף נִיר כְּדוּ אָתֵיבְנָא לְפוּלְחָנָךְ וּנְתֵיב אֲרֵי אַתְּ יְיָ אֱלָהָנָא:
אֲרֵי כַּד אֲנַחְנָא תָיְבִין לְאוֹרַיְתָא מִתְנַחֵם עֲלָנָא וּכְדוּ דְאִתְגְלֵי לָנָא טִפַּחְנָא עַל יְרַכָנָא בְּהֵיתְנָא וְאַף אִתְכְּנָעֲנָא אֲרֵי קַבֵּילְנָא חִסוּדֵי חוֹבָנָא דְמִלְקָדְמִין:
הַכְבַר חֲבִיב קָדָמַי יִשְׂרָאֵל הֲלָא רָבֵי רָחֵים הוּא אֲרֵי בִזְמַן דִמְשַׁוֵי פִּתְגָמֵי אוֹרַיְתִי עַל לִבֵּיהּ לְמֶעְבַּדְהוֹן מִדְכַר דָכֵירְנָא לְאוֹטָבָא לֵיהּ עוֹד עַל כֵּן אִתְגוֹלָלוּ רַחֲמַי עֲלֵיהוֹן רַחֲמָא אֲרַחֵים עֲלֵיהוֹן אֲמַר יְיָ:
כְּנִשְׁתָּא דְיִשְׂרָאֵל אֶדְכְּרִי לִיךְ עוֹבָדֵי אֲבָהָתִיךְ תַּקְנַיָא שְׁפוּכִי תַחֲנוּנִין בִּמְרַר שַׁוִיאִי לִבִּיךְ אִסְתַּכָּלִי בְּעוֹבָדִין דַעֲבַדְתְּ אִם תַּקְנִין דְכֵן גְלֵית לְאוֹרַח רְחִיקָא כְּדוֹ תּוּבוּ כְּנִשְׁתָּא דְיִשְׂרָאֵל וּתְתוּבוּן לִקְרְוָךְ אִלֵין:
עַד אֵימָתַי אַתְּ מִתְחַסְנָא לְמִתַּב כְּנִשְׁתָּא דְסַגִיאִין תִּיוּבְתָּהָא אֲרֵי הָא יְיָ בָרָא חַדְתָּא בְּאַרְעָא וְעַמָא בֵּית יִשְׂרָאֵל יִתְנְהוֹן לְאוֹרַיְתָא:
כִּדְנַן אֲמַר יְיָ צְבָאוֹת אֱלָהָא דְיִשְׂרָאֵל עוֹד יֵימְרוּן יַת פִתְגָמָא הָדֵין בְּאַרְעָא דְבֵית יְהוּדָה וּבְקִרְווֹהִי כַּד אֲתֵיב יַת גַלְוַתְהוֹן יְבָרְכִינָךְ יְיָ מְדוֹר קוּשְׁטֵיהּ טוּרָא קַדִישָׁא:
וְיֵתְבוּן בָּהּ דְבֵית יְהוּדָה וְכָל קִרְוֵיהוֹן כַּחֲדָא אִיכָּרִין וּדְנַטְלִין בְּסַיְעָן:
אֲרֵי שִׂבָעֵית נְפַשׁ מְשַׁלְהֲיָא וְכָל נְפַשׁ יִצְפָּא מַלֵיתִי כָּל טוּב:
אֲמַר נְבִיָא עַל דָא בְּשׂוֹרְתָא דְיוֹמֵי נֶחָמָתָא דַעֲתִידִין לְמֵיתֵי אִתְעְרֵית וַחֲזֵית תָּבֵית דְמָכֵית וְשִׁנְתִי הֲנָאַתְנִי:
הָא יוֹמַיָא אָתָן אֲמַר יְיָ וַאֲקֵים יַת דְבֵית יְהוּדָה אַסְגִינוּן כֶּאֱנָשָׁא וְאַצְלְחִינוּן כִּבְעִירָא:
וִיהִי כְמָה דְחַשְׁבֵית עֲלֵיהוֹן לְמֶעְקַר וּלְתַרְעָא וּלְפַגְרָא וּלְאַבָּדָא וּלְאַבְאָשָׁא כֵּן יֶחְדֵי מֵימְרֵי עֲלֵיהוֹן לְמֶבְנֵי וּלְקָיָמָא אֲמַר יְיָ:
בְּיוֹמַיָא הָאִינוּן לָא יֵמְרוּן עוֹד אֲבָהָתְנָא חָטַן וּבְנַיָא לָקָן:
אֱלָהֵין אֱנַשׁ בְּחוֹבֵיהּ יְמוּת כָּל אֱנַשׁ דְיֶחְטֵי הוּא יְמוּת:
הָא יוֹמַיָא אָתָן אֲמַר יְיָ וְאֶגְזַר עִם דְבֵית יִשְׂרָאֵל וְעִם דְבֵית יְהוּדָה קַיָמָא חַדְתָּא:
לָא כִקְיָמָא דִגְזָרֵית עִם אֲבָהַתְהוֹן בְּיוֹמָא דְאַתְקְפֵית בִּידֵיהוֹן לְמִפַּקְהוֹן מֵאַרְעָא דְמִצְרַיִם דְאִינוּן אַשְׁנִיאוּ יַת קְיָמִי וַאֲנָא אִתְרְעֵיתִי בְהוֹן אֲמַר יְיָ:
אֲרֵי דֵין קְיָמָא דְאַגְזַר עִם בֵּית יִשְׂרָאֵל בָּתַר יוֹמַיָא הָאִינוּן אֲמַר יְיָ אֶתֵּן יַת אוֹרַיְתִי בִּמְעֵיהוֹן וְעַל לִבְּהוֹן אֶכְתְּבִינָא וֶאֱהֶוֵי לְהוֹן לֶאֱלָהּ וְאִינוּן יְהוֹן קָדָמַי לְעַם:
וְלָא יֵילְפוּן עוֹד גְבַר יַת חַבְרֵיהּ וּגְבַר יַת אֲחוֹהִי לְמֵימַר דְעוּ לְמִדְחַל מִן קֳדָם יְיָ אֲרֵי כוּלְהוֹן יֵילְפוּן לְמִדַע דַחַלְתִּי לְמִזְעֵירְהוֹן וְעַד רַבְּהוֹן אֲמַר יְיָ אֲרֵי אֶשְׁבּוֹק לְחוֹבֵיהוֹן וַחֲטָאֵיהוֹן לָא יִדְכְּרוּן עוֹד:
כִּדְנַן אֲמַר יְיָ דִיהַב שִׁמְשָׁא לְאַנְהָרָא בִּימָמָא גְּזֵירָת סִיהֲרָא וְכוֹכְבַיָּא לְאַנְהָרָא בְלֵילְיָא נְזַף בְּיַמָא וְהָמָן גַלוֹהִי יְיָ צְבָאוֹת שְׁמֵיהּ:
כְּמָה דְלֵית אֶפְשַׁר דִיעִידוּן קְיָמַיָא הָאִלֵין מִן קָדָמַי אֲמַר יְיָ אַף זַרְעָא דְיִשְׂרָאֵל לָא יִבַטְלוּן מִלְמֶהֱוֵי עַם מְשַׁמֵשׁ קָדָמַי כָּל יוֹמַיָא:
כִּדְנַן אֲמַר יְיָ כְּמָה דְלֵית אֶפְשַׁר דִידַע אֱנַשׁ מִשְׁחַת שְׁמַיָא מִלְעֵילָא וְאַף לָא לְמִבְדַק יְסוֹדֵי אַרְעָא מִלְרַע אַף מֵימְרִי לָא יְרַחֵיק בְּכָל זַרְעָא דְיִשְׂרָאֵל עַל כָּל דַעֲבָדוּ קָדָמַי אֲמַר יְיָ:
הָא יוֹמַיָא אָתָן אֲמַר יְיָ וְתִתְבְּנֵי קַרְתָּא דִירוּשְׁלֵם קֳדָם יְיָ מִמִגְדַל פִּקוּס וְעַד תְּרַע זִוְיָתָא:
וִיהֵי נָפֵיק עוֹד חוּט מְשַׁח בִּנְיָנָא לְקִבְלָא עַד מְטֵי לְגִבְעֲתָא דִסְמִיכָא לְגָרֵב וִיהֵי מִסְתְּחַר לִבְרֵיכַת עִיגְלָה:
וְכָל מֵישְׁרָא אֲתַר דִנְפָלוּ תַמָן פִּגְרֵי מַשִׁרְיַת אַתּוּרָאָה וְכָל אֲרִיתָא עַד נַחְלָא דְקִידְרוֹן עַד זָוִית תְּרַע בֵּית רֵיסָא דְמַלְכָּא מַדִינְחָא קוּדְשָׁא קֳדָם יְיָ לָא יִתְעַקֵר וְלָא יִתְפַּגֵר עוֹד לַעֲלָם:
