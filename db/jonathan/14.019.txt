וְאַתְּ טוּל אִלְיָא עַל רַבְרְבֵי יִשְׂרָאֵל:
וְתֵימַר מַה כְּנִשְׁתָּא דְיִשְׂרָאֵל דַהֲוַת דַמְיָא לְלַיְתָא בֵּינֵי מַלְכַיָא רַבָּא בֵּינֵי שִׁלְטוֹנַיָא רַבִּיאַת בְּנָהָא:
וְרַבִּיאַת חַד מִבְּנָהָא מַלְכָּא הֲוָה וַאֲלִיף לְמִקְטַל קְטוֹל אֱנָשָׁא קְטַל:
וּשְׁמָעוּ עֲלוֹהִי עַמְמַיָא בִּסְרִיגַתְהוֹן אִתְאֲחַד וְאַיְתִיאוּהוּ בְּשַׁלְשְׁלָן לְאַרְעָא דְמִצְרָיִם:
וַחֲזַת אֲרֵי פְּסַק אֲבַד סִבּוּרָה וְרַבִּיאַת חַד מִבְּנָהָא מַלְכָּא שַׁוְיָתֵיהּ:
וְהַלֵיךְ בְּגוֹ מַלְכַיָא מַלְכָּא הֲוָה וַאֲלֵיף לְמִקְטַל קְטוֹל אֱנָשָׁא קְטָל:
וְאַצְדֵי בִּרְנְיָתֵיהּ וְקִרִוֵיהוֹן אַחְרִיב וּצְדִיאַת אַרְעָא וּמְלוֹאָהּ מִקָל אִתְרְגֻשְׁתֵּהּ:
וְאִתְכְּנִישׁוּ עֲלוֹהִי עַמְמַיָא סְחוֹר סְחוֹר מִמְדִינָתָא וּפְרָסוּ עֲלוֹהִי מְצָדַתְהוֹן בִּסְרִיגַתְהוֹן אִתְאַחַד:
וְיַהֲבוּהִי בְּקוֹלָרִין בְּשִׁלְשְׁלָן וְאַיְתִיאוּהוּ לְוַת מַלְכָּא דְבָבֶל אַיְתִיאוּהוּ בִּמְצַדְיָא בְּדִיל דְלָא יִשְׁתְּמַע קָלֵיהּ עַד עַל טוּרֵי יִשְׂרָאֵל:
כְּנִשְׁתָּא דְיִשְׂרָאֵל בְּמַעְבְּדָה אוֹרַיְתָא דַמְיָא כְגוּפַן דִשְׁתִילָא עַל מַבּוּעַ דְמַיָן דְנוֹפָהּ שַׁפִּיר וְסוֹכָהּ מַטְלָן הִיא שְׁתִילָה עַל מַיִין סַגִיאִין:
וַהֲווֹ בָּהּ שִׁכלְטוֹנִין תַּקִיפִין וּמַלְכִין דְגִבָּרִין לְשַׁעְבְּדָא מַלְכוּ וְרָאֵים בְּתוּקְפֵיהּ עַל מַלְכוּתֵיהּ וְאִתְנְצַח בְּסוּמְכְוָתֵיהּ בִּסְגֵי גִבָּרוֹהִי:
וְאִתְעַקְרַת בִּתְקוֹף מֵאַרְעָא בֵּית שְׁכִנְתָּא לְאַרְעָא אוֹחֲרֵי אִתְגְלִיאַת וּמְלַךְ דְתַקִיף כְּרוּחַ קִידוּמָא קְטֵיל עַמָהּ אִטַלְטָלוּ וּגְלוֹ שִׁלְטוֹנֵי תוּקְפָא וְעַמְמַיָא דְתַקִיפִין כְּאֶשְׁתָּא שֵׁיצִיאִינוּן:
וּכְעַן דַמְיָא לְגוּפַן דִשְׁתִילָא בִּמַדְבְּרָא בְּאַרְעָא צַדְיָא וַחֲרֵיבָא:
וַאֲתוֹ עַמְמַיָא דְתַקִיפִין כְּאֶשְׁתָּא מֵחוֹבֵי זְדוֹנָהּ עַמָהּ קְטִילוּ וְלָא הֲווֹ בָהּ שִׁלְטוֹנִין תַּקִיפִין מַלְכִין דְגִבָּרִין לְשַׁעְבָּדָא מַלְכוּ אֲמַר נְבִיָא אִלְיָא הִיא נְבוּאֲתָא וּתְהֵי לְאִלְיָא:
