וַהֲוָה פִּתְגַם נְבוּאָה מִן קֳדָם יְיָ עִמִי לְמֵימָר:
בַּר אָדָם אִתְנַבֵּי עַל תַּרְתֵּין מְדִינָן דְאִינוּן כְּתַרְתִּין נְשִׁין דִי בְנַת אִמָא חֲדָא הֲוָאָה:
וְטַעֲאָה בְמִצְרַיִם בָּתַר פּוּלְחַן טַעֲוָתְהוֹן טִעֲאָה תַּמָן פַּלְחָא לְטַעֲוָתָא וְתַמָן קִלְקְלָא עוֹבְדֵיהֶן:
וּשְׁמָהָתְהֶן אָהֳלָה רַבְּתָא וְאָהָלִיבָה אֲחָתָהּ וַהֲוָאָה מְשַׁמְשָׁן קָדָמַי וְאַצְלְחָא בִּבְנִין וּבְנָן וּשְׁמָהָתְהֶן שׁוֹמְרוֹן אָהֳלָה וִירוּשְׁלֵם אָהֳלִיבָה:
וּטְעַת אָהֳלָה מִבָּתַר פוּלְחָנִי וְאִתְרְעִיאַת לְמִטְעֵי בָתַר רְחִימָהָא בָּתַר בְּנֵי אַתּוּר קָרִיבָהָא:
לְבוּשֵׁי גְמִיר שִׁלְטוֹנִין וְטוּרָנִין עוּלֵימֵי חֶמְדָתָא כּוּלְהוֹן פָּרָשִׁין רָכְבֵי סוּסְוָן:
וְאִתְרְעִיאַת לְמִטְעֵי בַתְרֵיהוֹן עוּלֵימֵי בְנֵי אַתּוּר כוּלְהוֹן וּבְכָל דְאִתְרְעִיאַת בְּכָל פּוּלְחַן טַעֲוָתְהוֹן אִסְתַּאֲבָת:
וְיַת פּוּלְחַן טַעֲוָתָא דַהֲוָה עִמָהּ מִמִצְרַיִם לָא שַׁבְקַת אֲרֵי יָתָהּ אַפְלָחוּ לְטַעֲוָתָהּ וְאִינוּן אַלִפוּהָא עוֹבָדִין בִּישִׁין וּשְׁפָכוּ טַעֲוָתְהוֹן עֲלָהּ:
בְּכֵן מְסַרְתִיהָא בְּיַד רְחִימָהָא בְּיַד בְּנֵי אַתּוּוּן דְאִתְרְעִיאַת לְמִטְעֵי בַתְרֵיהוֹן:
אִינוּן גַלִיאוּ קְלָנָהּ בְּנָהָא וּבְנָתָהָא שְׁבוֹ וְעַמָהּ בְּחַרְבָּא קְטִילוּ וַהֲוַת שׁוּם לִמְדִינָתָא וּפוּרְעֲנוּת דִינִין עֲבָדוּ בְּגַוָהּ:
וַחֲזַת אֲחָתָהּ אָהֳלִיבָה וְחַבִּילַת רְחִימְתָּא מִנָהּ וְיַת טַעֲוָתָהָא מִטַעֲוַת אֲחָתָהּ:
בָּתַר בְּנֵי אַתּוּר אִתְרְעִיאַת לְמִטְעֵי שִׁלְטוֹנִין וְטוּרְנִין קָרִיבִין לְבוּשֵׁי גְמַר וּפָרָשִׁין רָכְבֵי סוּסְוָן עוּלֵימֵי חֶמְדָתָא כּוּלְהוֹן:
וּגְלֵי קֳדָמֵי אֲרֵי אִסְתַּאֲבַת אוֹרְחָא חֲדָא שַׁוִי לְתַרְתֵּיהֶן:
וְאוֹסִיפַת עַל טַעֲוָתָהּ וַחֲזַת גַבְרִין מְצַיְרִין עַל כָּתְלָא צַלְמֵי כַסְדָאֵי מְצַיְרִין בְּסַמְמָנַיָא:
אֲסִירִין זַרְזִין בְּחַרְצֵיהוֹן מָחֲתִין קוּלְסִין בְּרֵישֵׁיהוֹן חֵיזוּ גִבָּרִין כּוּלְהוֹן דְמוּת בְּנֵי בָבֶל מְדִינַת כַּסְדָאֵי אֲרַע יַלְדוּתְהוֹן:
וְאִתְרְעִיאַת לְמִיטְעֵי בַתְרֵיהוֹן בָּתַר חֵיזוּ עֵינָהָא וּשְׁלָחַת אִזְגַדִין לְוַתְהוֹן לִמְדִינַת אֲרַע כַּסְדָאֵי:
וַאֲתוֹ לְוָתָהּ בְּנֵי בָבֶל לְמֶעְבַּד רַחֲמוּ וְסָאִיבוּ יָתָהּ בְּטַעֲוָתְהוֹן וְאִסְתַּאֲבַת בְּהוֹן עַד דְקָצַת נַפְשָׁהּ בְּהוֹן:
וְאִתְגְלִיאַת בַּהֲתָתָהּ וְאִתְחַזֵי קְלָנָהּ וְרַחֵיק מֵימְרִי יָתָהּ כְּמָא דְרָחֵיק מֵימְרִי יַת אֲחָתָהּ:
וְאַסְגִיאַת יַת טַעֲוָתָהּ לְאַדְכָּרָא יַת יוֹמִין דְמִלְקָדְמִין דִטְעַת בְּהוֹן כַּד הֲוַת בְּאַרְעָא דְמִצְרַיִם:
וְאִתְרְעִיאַת לְמֶהֱוֵי שְׁמוֹטָא לְהוֹן דִבְסַר חֲמָרִין בִּסְרֵיהוֹן וְצַחֲנַת סוּסְוָן צַחֲנַתְהוֹן:
וְאַדְכַּרְתְּ יַת חוֹבֵי עֲצַת חֶטְאָךְ דְמִלְקַדְמִין בְּמַעְבְּדִיךְ מִמִצְרַיִם רְחִימוּ בְּדִיל חֶטְאֵי יַנְקוּתִיךְ:
בְּכֵן אָהֳלִיבָה כִּדְנַן אֲמַר יְיָ אֱלֹהִים הָא אֲנָא מַיְתֵי בִּגְלֵי יַת רְחִימָךְ עֲלָךְ יַת דְקָצַת נַפְשִׁיךְ בְּהוֹן וְאַיְתִינוּן עֲלָךְ מִסְחוֹר סְחוֹר:
בְּנֵי בָבֶל וְכָל כַּסְדָאֵי פְּקוּדָאֵי וְשׁוֹעָאֵי וְקוֹעָאֵי כָּל בְּנֵי אַתּוּר עִמְהוֹן עוּלֵימֵי חֶמְדָתָא שִׁלְטוֹנִין וְטוּרָנִין כּוּלְהוֹן גִבָּרִין מְזַמְנִין רָכְבֵי סוּסְוָן כּוּלְהוֹן:
וְיֵיתוּן עֲלָךְ בְּמָאנֵי זְיָנָא בִּרְתִיכִין וְגַלְגַלִין וּמַשִׁרְיַת עַמְמִין דִמְזַיְנִין בַּעֲגִילִין וְקוּלְסִין יְמַנוּן עֲלָךְ סְחוֹר סְחוֹר וְאֶתֵּן קֳדָמֵיהוֹן פּוּרְעֲנוּת דִינִין וְיִתְפְּרָעוּן מִנִיךְ בְּנִימוּסֵיהוֹן:
וְאֶתֵּן פּוּרְעֲנוּתִי בִּיךְ וְיִתְפַּרְעוּן מִנִיךְ בִּרְגַז רַבְרְבַיִךְ וְיַקִירַיִךְ יִגְלוֹן וְעַמִיךְ בְּחַרְבָּא יִתְקַטְלוּן אִינוּן בְּנָךְ וּבְנָתָךְ יִשְׁבּוּן וּשְׁפַר אַרְעִיךְ יִתּוֹקַד בְּנוּרָא:
וִישַׁלְחוּנִיךְ לְבוּשֵׁי יְקָרִיךְ וְיִסְבוּן מָאנֵי תוּשְׁבַּחְתִּיךְ:
וַאֲבַטֵל חוֹבֵי עֲצַת חֶטְאָךְ מִנִיךְ וְיַת פּוּלְחַן טַעֲוָתִיךְ דַהֲוַת עִמִיךְ כַּד הֲוֵית בְּאַרְעָא דְמִצְרַיִם וְלָא תִזְקְפִין עַיְנָךְ לְהוֹן וְטַעֲוַת מִצְרָאֵי לָא תִדְכְּרִין עוֹד:
אֲרֵי כִדְנַן אֲמַר יְיָ אֱלֹהִים הָא אֲנָא מָסַר לִיךְ בְּיַד דִסְנֵית בְּיַד דְקָצַת נַפְשִׁיךְ בְּהוֹן:
וְיִתְפָּרְעוּן מִנִיךְ כִּדְבָבוּ וְיִסְבוּן יַת כָּל לֵיאוּתִיךְ וְיִשְׁבְּקוּנִיךְ מְטַלְטְלָא וּשְׁבִיקָא וְתִתְגְלֵי בַהֲתַת טְעוּתִיךְ וְחוֹבֵי עֵיצַת חֶטְאָךְ וּזְדוֹנִיךְ:
חוֹבַיִךְ גְרָמוּ אִלֵין לִיךְ בִדְטָעֵית בָּתַר עַמְמַיָא וְאִסְתְּאַבְתְּ בְּפוּלְחַן טַעֲוָתְהוֹן:
בְּאוֹרַח אֲחָתִיךְ הַלֵיכְתְּ וְאֶתֵּן כָּס פּוּרְעֲנוּתָהּ בִידִיךְ:
כִּדְנַן אֲמַר יְיָ אֱלֹהִים כָּס פּוּרְעֲנוּת אֲחָתֵיךְ תִּשְׁתִּין תַּקִיפָא וַחֲסֵינְתָּא תְּהִין לְחוֹךְ וְלִלְעֵיב מִסְגֵי עָקָא דְתֵיתֵי עֲלָךְ:
הָא כְרַוְיָא וְרַוָן תִּתְמְלִין כָּס צָדוּ וְאִישְׁתֵּמָמוּ כָּס פּוּרְעֲנוּת אֲחָתִיךְ שׁוֹמְרוֹן:
וְתִשְׁתִּין יָתָהּ וְתִגְמְרִין וְיַת פוּרְעֲנוּתָהָא תְקַבְּלִין וּבִסְרִיךְ תִּתְלְשִׁין אֲרֵי אֲנָא גְזָרֵית בְּמֵימְרִי אֲמַר יְיָ אֱלֹהִים:
בְּכֵן כִּדְנַן אֲמַר יְיָ אֱלֹהִים חֲלַף דִשְׁבַקְתָּ פּוּלְחָנִי וְאַרְחֵיקְתְּ דְחַלְתִּי מִקָבֵל עֵינָךְ וְאַף אַתְּ קַבִּילִי חוֹבֵי עֵיצַת חֶטְאִיךְ וְטַעֲוָתַיִךְ:
וַאֲמַר יְיָ לִי בַּר אָדָם הֲתוֹכַח יַת אָהֳלָה וְיַת אָהֳלִיבָה וְחַוֵי לְהֶן יַת תּוֹעֶבָתְהָן:
אֲרֵי זַנָאִין בְּהֶן וְדַם זַכַּאי אֲשָׁדוּ בִידֵיהֶן וּבָתַר פוּלְחַן טַעֲוַתְהֶן טְעוֹ וְאַף יַת בְּנֵיהֶן דַעֲתִידִין לְמִיפַּק מִנְהֶן תּוּלְדַת קוּדְשָׁא קֳדָמַי אַעֲבִירוּ לְהֵן לְפוּלְחַן:
עוֹד אַרְגִיזוּ קֳדָמַי סָאִיבוּ יַת בֵּית מַקְדְשִׁי בְּיוֹמָא הַהוּא וְיַת יוֹמֵי שַׁבַּיָא דִילִי אָחִילוּ:
וּבְמִכְסוּתְהֶן יַת בְּנֵיהוֹן לְפוּלְחַן טַעֲוַתְהוֹן וְעָלוּ לְבֵית מַקְדְשִׁי בְיוֹמָא הַהוּא לְאָחָלוּתֵיהּ וְהָא כְדֵין עֳבָדוּ בְּגוֹ בֵּית מַקְדְשֵׁי:
וְאַף אֲרֵי שְׁלַחְתְּ לְגַבְרַיָא דְאָתָן מֵרָחִיק דְאִזְגַד שְׁלִיחַ לְוַתְהוֹן וְהָא אֲתוֹ לַאֲתַר דְזַמִינְתְּ אַתְקִינְתְּ שׁוּקִין וּמַנִית פַּלְטוֹרִין:
וּרְמֵית דַרְגְשִׁין מַכִּיכִין דִיקַר וּפְתוֹרִין מְלַן כָּל טוּב סִדַרְתְּ קֳדָמֵיהוֹן וּקְטוֹרֶת בּוּסְמִי וּמִשְׁחִי שַׁוֵית עֲלֵיהוֹן:
וְקַל אִתְרְגוּשְׁתְּהוֹן דִי שְׁלַן שְׁלֵיוָא בְהוֹן וְלִתְקוֹף גִבָּרִין מִסְגֵי אֱנָשָׁא אָתָן מִסְחוֹר סְחוֹר מִמַדְבְּרָא וִיהָבוּ שֵׁירִין עַל יְדֵיהוֹן וְכֶתֶר דְתוּשְׁבְּחָתָא עַל רֵישֵׁיהוֹן:
וַאֲמָרֵית לִכְנִשְׁתָּא דְיִשְׂרָאֵל דִבְלִיעַת עַמָה בְּחוֹבִין כְּעַן תִּשְׁבוֹק מִטַעֲוָתָהּ וּתְתוּב לְפוּלְחָנִי וְלָא תָבָת:
וְעָלוּ לְוָתָהּ כְּמָא דְעָלִין לְוַת אִיתְּתָא פּוּנְדְקִיתָא כֵּן עָלוּ לְאָהֳלָה וּלְאָהֳלִיבָה מְזִינְתָא דְעֵיצַתְהוֹן חֶטְאִין:
וְגַבְרִין זַכָּאִין בְּקִבְלֵיהוֹן יִתְפָּרְעוּן מִנְהֶן פּוּרְעֲנוּת נַיָפִין וּפוּרְעֲנוּת אַשְׁדַת דַם אֲרֵי זַנְאִין בְּהֶן וְדַם זַכַּאִי אֲשָׁדוּ בִּידֵיהוֹן:
כִּדְנַן אֲמַר יְיָ אֱלֹהִים אִיתְנַבֵּי דְיִסְקוּן עֲלֵיהוֹן מַשִׁרְיָן וְאֶמְסַר יַתְהוֹן לִתְבִיר וּלְבִזָא:
וְיִרְגְמוּ יַתְהֶן בְּאַבְנֵי קָלַע מַשִׁרְיָן וְיִבְזְעוּן יַתְהֶן בְּסַיְפֵיהוֹן בְּנֵיהוֹן וּבְנָתְהוֹן יִקְטְלוּן וּבָתֵּיהוֹן יוֹקְדוּן בְּנוּרָא:
וַאֲבַטֵיל חוֹבֵי עֵיצַת חֶטְאִין מִן אַרְעָא וְיִתְרַדְיָן כָּל מְדִינְתָא וְלָא יַעְבְּדָן כְּעֵיצַת חֲטָאֵיכוֹן:
וְיִתְיַהֲבוּן חוֹבֵי עֵיצַת חֶטְאֵיכוֹן עֲלֵיכוֹן וְחֶטְאֵי פּוּלְחַן טַעֲוַתְכוֹן תְּקַבְּלָן וְתֵידְעוּן אֲרֵי אֲנָא יְיָ אֱלֹהִים:
