וְאוֹבִילַנִי לְתַרְעָא תַּרְעָא דִפְתִיחַ לְאוֹרַח מְדִינְחָא:
וְהָא יְקָר אֱלָהָא דְיִשְׂרָאֵל אִתְגְלֵי מֵאוֹרַח מַדִינְחָא וְקַל מְבָרְכֵי שְׁמֵהּ כְּקַל מַיִין סַגִיאִין וְאַרְעָא נַהֲרַת מִזִיו יְקָרֵהּ:
וּכְחֵיזוּ חֶזְוָא דַחֲזֵיתִי כְּחֶזְוָא דַחֲזֵיתִי בְּאִתְנַבְּיוּתִי לְחַבָּלָא יַת קַרְתָּא וְחֶזְוָא כְּחֵיזוּ דַחֲזִיתִי עַל נְהַר כְּבַר וְאִשְׁתַּטְחֵית עַל אַפָּי:
וִיקָרָא דַייָ אִתְגְלֵי בְּבֵיתָא בְּאוֹרַח תַּרְעָא דִפְתִיחַ לְאוֹרַח מַדִינְחָא:
וּנְטַלְתַּנִי רוּחָא וְאַעֲלַתְנִי לְדַרְתָּא גַוַיְתָא וְהָא אִתְמְלֵי יְקָרָא דַייָ בֵּיתָא:
וּשְׁמָעֵית דִי מְמַלֵל עִמִי מִלְגָיו וְגַבְרָא הֲוָה קָאִים לְקִבְלִי:
וַאֲמַר לִי בַּר אָדָם דֵין אֲתַר בֵּית כּוּרְסֵי יְקָרִי וְדֵין אֲתַר בֵּית אַשְׁרָיוּת שְׁכִנְתִּי דְאַשְׁרֵי שְׁכִנְתִּי תַמָן בְּגוֹ בְּנֵי יִשְׂרָאֵל לַעֲלָם וְלָא יִסְאֲבוּן עוֹד בֵּית יִשְׂרָאֵל שְׁמָא דְקוּדְשִׁי אִינוּן וּמַלְכֵיהוֹן בְּטַעֲוַתְהוֹן וּבְפִגְרֵי מַלְכֵיהוֹן בָּמוֹתֵיהוֹן:
בְּמִתַּנְהוֹן סִפֵּיהוֹן לָקֳבֵיל סִפֵי בֵית מַקְדְשִׁי וּבִנְיָנֵיהוֹן בִּסְטַר עֲזָרָתִי וְכוֹתֶל בֵּית מַקְדָשִׁי בֵּין מֵימְרִי וּבֵינֵיהוֹן וּסְאִיבוּ יַת שְׁמָא דְקוּדְשִׁי בְּתוֹעֲבַתְהוֹן דַעֲבָדוּ וְשֵׁיצֵתִי יַתְהוֹן בְּרוּגְזִי:
כְּעַן יְרַחֲקוּן יַת טַעֲוָתְהוֹן וּפִגְרֵי מַלְכֵיהוֹן מִלְמֶחֱטֵי קֳדָמַי וְאַשְׁרֵי שְׁכִנְתִּי בֵינֵיהוֹן לַעֲלָם:
אַתְּ בַּר אָדָם חַוִי יַת בֵּית יִשְׂרָאֵל יַת בֵּיתָא וְיִתְכַּנְעוּן מֵחוֹבֵיהוֹן וְיִמְשְׁחוּן יַת טִקוּסֵיהּ:
וְאִם יִתְכַּנָעוּן מִכָּל דַעֲבֵדוּ בְמֶחְזֵיהוֹן צוּרַת בֵּיתָא וְטִקוּסֵיהּ וּמַפְקָנוֹהִי וּמַעֲלָנוֹהִי וְכָל צוּרָתֵהּ וְיַת כָּל גְזֵירָתֵהּ וְכָל צוּרָתֵהּ וְכָל דַחֲזֵי לֵיהּ הוֹדַע יַתְהוֹן וּכְתוֹב לְעֵינֵיהוֹן וְיִטְרוּן יַת כָּל צוּרָתֵהּ וְיַת כָּל גְזֵירָתֵהּ וְיַעְבְּדוּן יַתְהוֹן:
דָא אוֹרַיְתָא דְבֵיתָא עַל רֵישׁ טוּרָא כָּל תְּחוּמֵהּ סְחוֹר סְחוֹר קוֹדֶשׁ קוֹדְשַׁיָא הָא דָא אוֹרַיְתָא דְבֵיתָא:
וְאִלֵין מִשְׁחַת מַדְבְּחָא בְּאַמַיָא בְּאַמְתָא דְהִיא אַמָה וּפְשַׁךְ וְתַשְׁוִיתָא אַמְתָא וְאַמְתָא פוּתְיָא וּגְדַנְפָא מַקַף לָהּ עַל סִפְתֵהּ סְחוֹר סְחוֹר זֵירְתָא חֲדָא וְדֵין טִקוּס מַדְבְּחָא:
וּמִתַּשְׁוֵית אַרְעָא עַד מְסַתִמְתָא אַרְעִיתָא תַּרְתֵּין אַמִין וּפוּתְיָא אַמְתָא חֲדָא וּמִמְסַמְתָּא זְעֶרְתָּא עַד מְסַמְתָא רַבְּתָא אַרְבַּע אַמִין וּפוּתְיָא אַמְתָא:
וּמַדְבְּחָא אַרְבַּע אַמִין וּמִמַדְבְּחָא וּלְעֵלָא קַרְנָתָא אַרְבַּע:
וּמַדְבְּחָא תַּרְתָּא עַסְרֵי אוּרְכָּא בְּתַרְתָּא עַסְרֵי פּוּתְיָא מְרֻבַּע עַל אַרְבְּעָא סִיטְרוֹהִי:
וּמִסַמְתָּא אַרְבַּע אַסְרֵי פוּתְיָא עַל אַרְבַּע סִיטְרָהָא וּגְדַנְפָא מַקַף לָהּ סְחוֹר סְחוֹר פַּלְגוּת אַמְתָא וְתַשְׁוִיתָא לָהּ אַמְתָא סְחוֹר סְחוֹר וְדַרְגָן דְסָלְקִין לָה מִתְפַּנָן לְמַדִינְחָא:
וַאֲמַר לִי בַּר אָדָם כִּדְנַן אֲמַר יְיָ אֱלֹהִים אִלֵין גְזֵירַת מַדְבְּחָא בְּיוֹם אִיתְעַבְּדוּתֵהּ לְאַסָקָא עֲלוֹהִי עֲלָתָא וּלְמִזְרָק עֲלוֹהִי דָם נִכְסַת קוּדְשַׁיָא:
וְתִתֵּן לְכַהֲנַיָא לֵיוָאֵי דִי אִינוּן מִזַרְעָא דְצָדוֹק דְקָרִיבִין לְפוּלְחָנִי אֲמַר יְיָ אֱלֹהִים לְשַׁמָשָׁא קֳדָמַי תּוֹר בַּר תּוֹרֵי לְחַטָאתָא:
וְתִיסַב מִדְמֵיהּ וְתִתֵּן עַל אַרְבַּע קַרְנָתֵיהּ וְעַל אַרְבַּע זִוְיַת מְסַמְתָא וְעַל תְּחוּמָה סְחוֹר סְחוֹר וּתְדַכֵּי יָתֵיהּ וּתְכַפֵּר עֲלוֹהִי:
וְתִסַב יַת תּוֹרָא דְחַטָאתָא וְיוֹקְדִינֵיהּ בַּאֲתַר דַחֲזִי לְבֵיתָא מִבָּרָא לְמַקְדְשָׁא:
וּבְיוֹמָא תִנְיָנָא תַקְרִיב צְפִיר בַּר עִזַי שְׁלִים לְחַטָאתָא וְתִדְכֵּי יַת מַדְבְּחָא כְּמָא דְדַכִּיאוּ בְּתוֹרָא:
בְּשֵׁיצָיוּתָךְ מִלְדַכָּאָה תַּקְרִיב תּוֹר בַּר תּוֹרֵי שְׁלִים וּדְכַר מִן עָנָא שְׁלִים:
וְתִקְרְבִינוּן קֳדָם יְיָ וְיִתְּנוּן כַּהֲנַיָא עֲלֵיהוֹן מִילְחָא וְיִסְקוּן יַתְהוֹן עֲלָתָא קֳדָם יְיָ:
שַׁבְעָא יוֹמִין תַּעֲבֵּיד צְפִירָא דְחַטָאתָא לְיוֹמָא וְתוֹר בַּר תּוֹרֵי וּדִכַר מִן עָנָא שְׁלֵימִין יַעַבְדוּן:
שַׁבְעָה יוֹמִין יְכַפְּרוּן עַל מַדְבְּחָא וְיִדְכּוּן יָתֵיהּ וְיַקְרִיבוּן קוּרְבָּנֵיהּ:
וְיִשְׁלְמוּן יַת יוֹמַיָא וִיהֵי בְּיוֹמָא תְמִינָאָה וּלְהָלָא יַעַבְדוּן כַּהֲנַיָא עַל מַדְבְּחָא יַת עַלְוָתְכוֹן וְיַת נִכְסַת קוּדְשֵׁיכוֹן וְיִתְקַבְּלוּן קוּרְבָּנֵיכוֹן בְּרַעֲוָא אֲמַר יְיָ אֱלֹהִים:
