אֲרֵי יְרַחֵם יְיָ עַל דְבֵית יַעֲקֹב וְיִתְרְעֵי עוֹד בְּיִשְׂרָאֵל וְיִשְׁרִינוּן עַל אַרְעֲהוֹן וְיִתּוֹסְפוּן גִיוֹרִין עֲלֵיהוֹן וְיִסְתַּמְכוּן עַל בֵּית יַעֲקֹב:
וְיַדְבְּרִינוּן עַמְמִין וְיוֹבְלוּנוּן לְאַרְעֲהוֹן וְיַחְסְנוּן בֵּית יִשְׂרָאֵל עַל אֲרַע שְׁכִנְתָּא דַייָ לְעַבְדִין וּלְאַמְהָן וִיהוֹן שָׁבִין לְשׁוֹבֵיהוֹן וְיִפְלְחוּן בִּמְשַׁעְבְּדֵיהוֹן:
וִיהֵי בְּיוֹמָא דִינִיחַ יְיָ לָךְ מִצַעֲרָךְ וּמִמַרְוָתָךְ וּמִן פוּלְחָנָא קַשְׁיָא דְאִתְפְּלַח בָּךְ:
וְתִטוֹל מַתְלָא הָדֵין עַל מַלְכָּא דְבָבֶל וְתֵימַר אֵיכְדֵין בְּטֵילַת מַרְוַת מְשַׁעְבְּדָנָא סָף תְּקוֹף חַיָבָא:
תְּבַר יְיָ תְּקוֹף רַשִׁיעִין שׁוּלְטַן חַיָבִין:
דַהֲוָה מָחֵי עַמְמִין בִּתְקוֹף מְחָא דִי לָא פְּסִיק מַפְלַח בִּתְקוֹף עַמְמִין מַפְלַח וְלָא מְנָע:
נָחַת שְׁדוֹכַת כָּל אַרְעָא יַבְעוּ תּוּשְׁבְּחָא:
אַף שִׁלְטוֹנִין חֲדִיאוּ עֲלָךְ עַתִּירֵי נִכְסַיָא אָמְרִין מֵעִידַן דִשְׁכֵיבְתָּא לָא סְלִיק מְחַבְּלָא עֲלָנָא:
שְׁאוֹל מִלְרַע זָעַת לָךְ לִקֳדָמוּת מֵיתָךְ אֲעִירַת לָךְ גִבָּרִין כָּל עַתִּירֵי נִכְסַיָא אָקִימוּ מִכּוּרְסְוַתְהוֹן כָּל מַלְכֵי עַמְמַיָא:
כּוּלְהוֹן יְתִיבוּן וְיֵימְרוּן לָךְ אַף אַתְּ מְרַעְתָּא כְוָתָנָא לָנָא דְמִיתָא:
אִתְּחַת לִשְׁאוֹל יְקָרָךְ אִתְרְגוּשַׁת תּוּשְׁבְּחַת זְמָרָךְ תְּחוֹתָךְ יְשַׁווּן רִמָה וְעֵיל מִנָךְ רִיחֲשָׁא:
אֵיכְדֵין אִתְרְכִינְתָּא מִן רוּמָא דַהֲוֵיתָא זִיוְתָן בְּגוֹ בְְנֵי אֱנָשָׁא כְּכוֹכָב נוֹגְהָא בֵּין כּוֹכְבַיָא אִתְרַטֵישְׁתָּא לְאַרְעָא דַהֲוֵיתָא קְטוֹל בְּעַמְמַיָא:
וְאַתְּ אֲמַרְתְּ בְּלִבָּךְ לְרוּמָא אֶסַק עֵיל מִן עַמֵיהּ דֵאלָהָא אֲשַׁוֵי כוּרְסֵי מַלְכוּתִי וְאֶתֵּיב בְטוּר זְמַן בִּסְיָפֵי צִפּוּנָא:
אֶסַק עִלַוֵי כָל עַמָא אֶהִי עִילָאִי מִכּוּלְהוֹן:
בְּרַם לִשְׁאוֹל תִּתְחַת לִסְיָפֵי גוֹב בֵּית אַבְדָנָא:
חָזָךְ עֲלָךְ יְדוּקוּן בָּךְ יִסְתַּכְּלוּן וְיֵימְרוּן הָדֵין גַבְרָא אֲזִיעַ אַרְעָא אַצְדֵי מַלְכְּוָתָא:
שַׁוֵי תֵּבֵל כְּמַדְבְּרָא וְקִרְווֹהִי פִּגֵר לַאֲסִירוֹהִי לָא פְּתַח תַּרְעָא:
כָּל מַלְכֵי עַמְמַיָא כּוּלְהוֹן שָׁכְבִין בִּיקָר גְבַר בְּבֵית עַלְמֵיהּ:
וְאַתְּ אִתְרְכִיתְּנָא מִקִבְרָךְ כְּחָט טְמִיר חֲפֵי קְטִילִין מְטַעֲנֵי חֶרֶב נַחֲתִין לְגוֹב אַבְנֵי בֵּית אַבְדָנָא כִּפְגַר מְדַשְׁדָשׁ:
לָא תְהֵי כְּחַד מִנְהוֹן בִּקְבוּרָה אֲרֵי אַרְעָךְ חַבֵּילְתָּא וְעַמָךְ קָטֵילְתָּא לָא יִתְקַיֵים לַעֲלָם זֶרַע מַבְאָשִׁין:
אַתְקִינוּ לִבְנֵיהוֹן קִטוֹל בְּחוֹבַי אֲבָהַתְהוֹן דִלְמָא יְקוּמוּן וְיַחְסְנוּן אַרְעָא וְיִתְמְלוּן אַפֵּי תֵבֵל בְּעֵיל דְבָבוּ:
וְאִתְגְלֵי לְאִתְפְּרָעָא מִנְהוֹן אֲמַר יְיָ צְבָאוֹת וֶאֱשֵׁיצֵי לְבָבֶל שׁוּם וּשְׁאָר בַּר וּבַר בַּר אֲמַר יְיָ:
וַאֲשַׁוִינָהּ לִירוּתַת קוּפְדִין בֵּית צָדוּ בִּיצִין דְמַיָן וֶאֱהַמְמִינָהּ כְּמָא דִמְהַמְמִין בִּמְבִינָא וַאֲתַבְּבִינָהּ לְשֵׁיצָאָה אֲמַר יְיָ צְבָאוֹת:
קַיָם יְיָ צְבָאוֹת לְמֵימָר אִם לָא כְּמָא דַחֲשֵׁיבִית כֵּן תְּהֵא וּכְמָא דְיַעֲצִית הִיא תָקוּם:
לְמִתְבַּר אַתּוּרָאָה בְּאַרְעִי וְעַל טוּרֵי עַמִי אַדוֹשְׁשִׁינֵיהּ וְתֶעְדֵי מִנְהוֹן מַרְוָתֵיהּ וְנִירֵיהּ מֵעַל צַוְרֵיהוֹן יִתְפְּסַק:
דָא מַלְכָּא דִי מְלִיךְ עַל כָּל יַתְבֵי אַרְעָא וְדָא גְבוּרְתָּא דִי מְרַמְמָא דְשַׁלִיט עַל כָּל מַלְכְּוָתָא:
אֲרֵי יְיָ צְבָאוֹת מְלַךְ וּמַן יַעְדֵי וּגְבוּרְתֵּיהּ מְרַמְמָא וְלֵית דִיתִיבִינֵיהּ:
בְּשַׁתָּא דְמִית מַלְכָּא אָחָז הֲוָה מַטַל נְבוּאָתָא הָדָא:
לָא תֵיחְדוּן פְּלִשְׁתָּאֵי כּוּלְכוֹן אֲרֵי אִתְּבַּר שׁוּלְטַן דַהֲוָה מַפְלַח בְּכוֹן אֲרֵי מִבְּנֵי בְנוֹהִי דְיִשָׁי יִפּוֹק מְשִׁיחָא וִיהוֹן עוֹבָדוֹהִי בְּכוֹן כְּחֵיוֵי מַפְרִיחַ:
וְיִתְפַּרְנְסוּן חֲשֵׁיכֵי עַמָא וְעִנְוְתָנַיָא בְּיוֹמוֹהִי לְרוֹחֲצָן יִשְׁרוֹן וְיֵמִית בְּכַפְנָא בְּנֵךְ וּשְׁאַר עַמִיךְ יִקְטוֹל:
אֱלִילִי עַל תַּרְעָךְ צַוְחִי עַל קִרְוָךְ אִתְבַּרְתּוּן פְּלִשְׁתָּאֵי כּוּלְכוֹן אֲרֵי מִצִפּוּנָא פּוּרְעֲנוּתָא אָתֵי וְלֵית דִמְאַחַר בְּזִמְנוֹהִי:
וּמַה יְבַסְרוּן אִזְגַדֵי עַמְמַיָא אֲרֵי יְיָ שַׁכְלֵיל צִיוֹן וּבָהּ יִתְרַחֲצוּן חֲשִׁיכֵי עַמָא:
