וּכְעַן שְׁמַע יַעֲקֹב עַבְדִי וְיִשְׂרָאֵל דְאִתְרְעֵיתִי בֵיהּ:
כִּדְנַן אֲמַר יְיָ דְעַבְדָךְ וּדְאַתְקְנָךְ מִמְעִין יְסַעְדִינָךְ לָא תִדְחַל עַבְדִי יַעֲקֹב וְיִשְׂרָאֵל דְאִתְרְעִיתִי בֵיהּ:
אֲרֵי כְמָא דְמִתְיַהֲבִין מַיָא עַל אֲרַע בֵּית צַחֲוָנָא וּמִתְנַגְדִין עַל יַבֶּשְׁתָּא כֵּן אֶתֵּן רוּחַ קוּדְשִׁי עַל בְּנָךְ וּבִרְכָתִי עַל בְּנֵי בְנָךְ:
וְיִתְרַבּוּן צַדִיקַיָא רַכִּיכִין וּמְפַנְקִין כְּלַבְלַבֵי עֲסַב כְּאִילָן דִמְשַׁלַח שָׁרְשׁוֹהִי עַל נַגְדִין דְמַיִין:
דֵין יֵימַר מִדָחֲלַיָא דַייָ אֲנָא וְדֵין יְצַלֵי בְשׁוּם יַעֲקֹב וְדֵין יְקָרֵיב קוּרְבָּנֵהּ קֳדָם יְיָ וּבִשְׁמָא דְיִשְׂרָאֵל יִתְקְרֵי:
כִּדְנַן אֲמַר יְיָ מַלְכָּא דְיִשְׂרָאֵל וּפַרְקֵהּ יְיָ צְבָאוֹת אֲנָא הוּא דְמִלְקַדְמִין אַף עַלְמֵי עָלְמַיָא דִילִי אִינוּן וּבַר מִנִי לֵית אֱלֶהּ:
וּמַן כְּוָתִי דִין עַרְעִינָהּ וִיחַוִינָהּ וִיסַדְרִינָהּ קֳדָמַי מִשַׁוָיוּתִי עַמָא דְמִן עַלְמָא וּדְאָתְיָן וְדַעֲתִידִין לְמֵיתֵי יְחַווּן לָנָא:
לָא תִדַחֲלוּן וְלָא תִתַּבְּרוּן הֲלָא מִבְּכֵן בְסַרְתָּךְ וְחַוֵיתִי וְאַתּוּן סָהֲדִין קֳדָמַי הֲאִית אֱלָהָא בַּר מִנִי וְלֵית דְתַקִיף אֱלָהֵן דְמִן קֳדָמַי מִתְיְהַב לֵיהּ תְּקוֹף:
עַבְדֵי צַלְמַיָא כּוּלְהוֹן לְמָא וּפָלְחֵיהוֹן דְלָא יַהֲנוּן לְהוֹן וְסָהֲדִין אִינוּן בְּנַפְשָׁתֵיהוֹן דְלָא חָזָן וְלָא יָדְעִין בְּדִין דְיִבַּהֲתוּן:
מַן עֲבַד דַחֲלָא וְצַלְמָא אַתִיךְ בְּדִיל דְלָא לַהֲנָאָה:
הָא כָּל פָּלְחֵיהוֹן יִבָּהֲתוּן וְאוּמָנִין עוּבְדָנִין מִבְּנֵי אֱנָשָׁא יִתְכַּנְסוּן כּוּלְהוֹן יְקוּמוּן יִתַּבְּרוּן יִבַהֲתוּן כַּחֲדָא:
נַפְחָא מִבַּרְזְלָא חֲצִינָא עָבֵיד וּנְפַח נוּר בְּשִׁיחוּרִין וּבְמַקוּבִין מַתְקֵיף לֵיהּ וְעָבִיד לֵיהּ בִּתְקוֹף חֵילֵהּ דְאִם יִכְפַּן עַבְדֵהּ וְלָא יֵיכוּל לָא יְהֵי בֵיהּ חֵיל וְאִם יִצְחֵי וְלָא יִשְׁתֵּי מַיָא יְהֵי מְשַׁלְהֵי:
נַגָר אָעִין נָפֵיץ בְּחוּטָא מְסַר לֵיהּ בְּמַשְׁקוּלְתָּא מְגַזֵי לֵיהּ בְּאִזְמֵילַיָא וּבִנְצוּרִין מְאַחֵיד לֵיהּ וְעָבֵיד לֵיהּ כִּדְמוּת גְבַר כְּתוּשְׁבְּחַת אִתְּתָא לְמִיתַּב בְּבֵיתָא:
לְמִקַץ לֵיהּ אַרְזִין וּנְסִיב תְּרֵז וּבְלוּט וּמַתְקֵיף לֵיהּ בַּאֲעֵי חוּרְשָׁא נְצַב אוּרְנָא וּמִטְרָא יְרַבֵּי:
וֶהֱוֵי לֶאֱנָשָׁא לְאַדְלָקָא וּנְסֵיב מִנְהוֹן וּשְׁחֵין אַף אֲזָא וְאָפָא לְחֵם אַף עָבְדֵהּ דַחֲלָא וּסְגֵיד אִתְּכֵהּ צַלְמָא וּבְעָא מִנֵהּ:
פַּלְגֵהּ אוֹקִיד בְּנוּרָא עַל פַּלְגֵהּ בִּסְרָא אָכִיל טְוָא טָוִי וּסְבַע אַף שְׁחִין וַאֲמַר אָח שַׁחֵינִית חֲזֵיתִי נוּר:
וּשְׁאָרֵהּ לְדַחֲלָא עָבַד לְצַלְמָא סְגִיד לֵיהּ וְאֶשְׁתַּעְבֵּד לֵיהּ וּבְעָא מִנֵהּ וַאֲמַר שֵׁיזְבָנִי אֲרֵי דְחַלְתִּי אָתְּ:
לָא יָדְעוּן וְלָא מִסְתַּכְּלִין אֲרֵי מְטַמְטְמָן מִלְמֶחֱזֵי עֵינֵיהוֹן מִלְאִסְתַּכָּלָא בְּלִבְּהוֹן:
וְלָא מֵתִיב לְלִבֵּהּ וְלָא מַדַע וְלָא סוּכְלְתָנוּ לְמֵימַר פַּלְגֵהּ אוֹקֵידַת בְּנוּרָא וְאַף אֲפִיתִי עַל גוּמְרוֹהִי לְחֵם טְוֵית בִּסְרָא וַאֲכַלֵית וּשְׁאֵרֵהּ לְתוֹעֵבָא אַעְבֵּיד לְיִבְלֵי אָעָא אֶסְגוֹד:
הָא דְחַלְתָּא פַלְגֵהּ קִטְמָא לִבֵּהּ שַׁטְיָא אַטְעֲיֵהּ וְלָא יְשֵׁיזֵיב יַת נַפְשֵׁהּ וְלָא יֵימַר הֲלָא שִׁקְרָא עֲבָדֵית בִּימִינִי:
אִדְכַּר אִלֵין יַעֲקֹב וְיִשְׂרָאֵל אֲרֵי עַבְדִי אַתְּ אַתְקֵינְתָּךְ לְמֶהֱוֵי עֲבֵיד פְּלַח קֳדָמַי אַתְּ יִשְׂרָאֵל לָא תִתְנְשֵׁי דְחַלְתִּי:
מְחֵיתִי כְעֵיבָא חוֹבָךְ וְכַעֲנָן עֲדֵי כָּל חֶטְאָךְ תּוּב לְפוּלְחָנִי אֲרֵי פְרַקְתָּךְ:
שַׁבַּחוּ שְׁמַיָא אֲרֵי עֲבַד יְיָ פוּרְקַן לְעַמֵהּ יַבִּיבוּ יְסוֹדֵי אַרְעָא בּוּעוּ טוּרַיָא תּוּשְׁבַּחְתָּא חוּרְשָׁא וְכָל אִילָנַיָא דְבֵהּ אֲרֵי פְרַק יְיָ יַעֲקֹב וּבְיִשְׂרָאֵל יִשְׁתַּבַּח:
כִּדְנַן אֲמַר יְיָ דְפַרְקָךְ וּדְאַתְקְנָךְ מִמְעַיִין אֲנָא יְיָ עָבֵיד כּוֹלָא תְּלֵית שְׁמַיָא בְּמֵימְרִי שַׁכְלְלֵית אַרְעָא בִּגְבוּרְתִּי:
מְבַטֵל אֶתְוָת בַּדִין וְקִסוּמִין מְשַׁגֵשׁ מַתִיב חַכִּימָא לַאֲחוֹרָא וְדַעְתְּהוֹן מְקַלְקֵל:
מְקַיֵם פִּתְגָמֵי עַבְדוֹהִי צַדִיקַיָא וּמְלַךְ שְׁלוּחוֹהִי מַשְׁלִים דַאֲמַר לִירוּשְׁלָם תִּתּוֹתַב וּלְקִרְוַיָא דְבֵית יְהוּדָה יִתְבַּנְיָן וְחָרְבָתָהָא אֲקוֹמֵם:
דַּאֲמַר עַל בָּבֶל תֶּחְרוֹב וְנַהֲרָתָא אֲיַבֵּשׁ:
דַאֲמַר עַל כּוֹרֶשׁ לְמִתַּן לֵיהּ מַלְכוּ וְכָל רְעוּתִי יְקַיֵם וְלִדְאֲמַר לִירוּשְׁלֶם תִּתְבְּנֵי וְהֵיכְלָא יִשְׁתַּכְלֵל:
