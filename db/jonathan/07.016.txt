וַאֲזַל שִׁמְשׁוֹן לְעַזָתָא וַחֲזָא תַמָן אִתְּתָא פּוּנְדְקִיתָא וְעַל לְוָתָהּ:
לְעַזָתָאֵי אִתְחַוָה לְמֵימָר אֲתָא שִׁמְשׁוֹן הַלְכָא וְאַסְחֲרוּ וּכְמָנוּ לֵיהּ כָּל לֵילְיָא בְּמַעֲלָנָא דִתְרַע קַרְתָּא וּשְׁתִיקוּ כָּל לֵילְיָא לְמֵימָר עַד נֵיהוֹר צַפְרָא וְנִקְטְלִינֵיהּ:
וּשְׁכַב שִׁמְשׁוֹן עַד פַּלְגוּת לֵילְיָא וְקָם בְּפַלְגוּת לֵילְיָא וַאֲחַד בְּדָשֵׁי תְּרַע קַרְתָּא וּבִתְרֵין סִפַּיָא וְנַטְלִינוּן עִם עַבְּרָא וְשַׁוִי עַל כְּתֵפוֹהִי וְאַסְקִינוּן לְרֵישׁ טוּרָא דְעַל אַפֵּי חֶבְרוֹן:
וַהֲוָה בָּתַר כֵּן וּרְחֵים אִתְּתָא בְּנַחֲלָא דְשׂוֹרֵק וּשְׁמָהּ דְלִילָהּ:
וּסְלִיקוּ לְוָתָהּ טוּרְנֵי פְּלִשְׁתָּאֵי וַאֲמַרוּ לָהּ שְׁדִילִי יָתֵיהּ וַחֲזִי בַּמֶה חֵילֵיהּ רַבָּא וּבַמֶה נֵיכוּל לֵיהּ וְנִסְרִינֵיהּ לְעַנָיוּתֵיהּ וַאֲנַחְנָא נִתַּן לָךְ גְבַר אֲלַף וּמְאָה סַלְעִין דִכְסַף:
וַאֲמַרַת דְלִילָה לְשִׁמְשׁוֹן חֲוֵי כְעַן לִי בְּמֶה חֵילָךְ סַגִי וּבְמָה תִתְאַסַר לְעַנְיוּתָךְ:
וַאֲמַר לָהּ שִׁמְשׁוֹן אִם יֵיסְרוּנַנִי בְּשַׁבְעָא יְתָרִין רְטִיבִין דְלָא יַבִּישׁוּ וְאֶחֱלוֹשׁ וְאֶהֱוֵי כְּחַד מִן בְּנֵי אֱנָשָׁא:
וְאַסִיקוּ לָהּ טוּרְנֵי פְּלִשְׁתָּאֵי שַׁבְעָא יְתָרִין רְטִיבִין דְלָא יַבִּישׁוּ וַאֲסַרְתֵּיהּ בְּהוֹן:
וּכְמָנָא יְתַב לָהּ בְּאִדְרוֹן בֵּית מִשְׁכְּבָא וַאֲמַרַת לֵיהּ פְּלִשְׁתָּאֵי אֲתוֹ עֲלָךְ שִׁמְשׁוֹן וּפְסַק יַת יְתָרַיָא כְּמָא דְמִתְפְּסַק חוּטָא דְכִתָּנָא דַאֲרַח בֵיהּ נוּרָא וְלָא אִתְיְדַע חֵילֵיהּ:
וַאֲמַרַת דְלִילָה לְשִׁמְשׁוֹן הָא שְׁקַרְתָּ בִּי וּמַלֵילְתָּא עִמִי כַּדְבִין כְּעַן חַוֵי כְעַן לִי בְּמַה תִּתְאַסָר:
וַאֲמַר לָהּ אִם מֵיסַר יֵיסְרוּנַנִי בִּגְדִילָן חַדְתָּן דְלָא אִתְעֲבִידָא בְהוֹן עֲבִידְתָּא וְאֶחֱלוֹשׁ וְאֶהֱוֵי כְּחַד מִבְּנֵי אֱנָשָׁא:
וּנְסִיבַת דְלִילָה גְדִילָן חַדְתָּן וַאֲסַרְתֵּיהּ בְּהוֹן וַאֲמַרַת לֵיהּ פְּלִשְׁתָּאֵי אֲתוֹ עֲלָךְ שִׁמְשׁוֹן וּכְמָנָא יָתֵיב בְּאִידְרוֹן בֵּית מִשְׁכְּבָא וּפַסְקִינוּן מֵעַל דַרְעוֹהִי כְּחוּטָא:
וַאֲמַרַת דְלִילָה לְשִׁמְשׁוֹן עַד הָכָא שְׁקַרְתָּ בִּי וּמַלֵילְתָּא עִמִי כַּדְבִין חַוִי כְעַן לִי בַּמֶה תִּתְאַסָר וַאֲמַר לָהּ אִם תִּשְׁתִּין יַת שְׁבַע גְדִילַת רֵישִׁי עִם אַכְסְנָא דְמַשְׁתֵּיתָא::
וְנִקְשַׁת בְסִכְתָא וַאֲמַרַת לֵיהּ פְּלִשְׁתָּאֵי אֲתוֹ עֲלָךְ שִׁמְשׁוֹן וְאִתְעַר מִשִׁנְתֵיהּ וּנְטַל יַת אַכְסַן דְגַרְדָאִין וְיַת מַשְׁתֵּיתָא:
וַאֲמַרַת לֵיהּ אֵיכְדֵין תֵּימַר רְחִימְתִּיךְ וְלִבָּךְ לֵית עִמִי דְנַן תְּלַת זִמְנִין שַׁקַרְתָּא בִּי וְלָא חַוֵיתָ לִי בְּמָה חֵילָךְ סַגִי:
וַהֲוָה כַּד אֲעִיקַת לֵיהּ בְּפִתְגָמָהָא כָּל יוֹמַיָא וּדְחַקְתֵּיהּ וְעָקַת נַפְשֵׁיהּ לִמְמָת:
וְחַוֵי לָהּ יַת כָּל לִבֵּיהּ וַאֲמַר לָהּ מִסְפַּר לָא עֲדָא עַל רֵישִׁי אֲרֵי נְזִירָא דַייָ אֲנָא מִבִּטְנָא דְאִמָא אִם אִתְגְלַח וְיַדְעִי מִנִי חֵילִי וְאֶחֱלוֹשׁ וְאֶהֱוֵי כְּחַד מִן בְּנֵי אֱנָשָׁא:
וַחֲזַת דְלִילָה אֲרֵי חַוֵי לָהּ יַת כָּל לִבֵּיהּ וּשְׁלָחַת וּקְרַת לְטוּרְנֵי פְּלִשְׁתָּאֵי לְמֵימָר סַקוּ בְּרַם זִמְנָא הָדָא אֲרֵי חֲוֵי לִי יַת כָּל לִבֵּיהּ וּסְלִיקוּ לְוָתָהּ טוּרְנֵי פְּלִשְׁתָּאֵי וְאַסִיקוּ כַסְפָּא בִּידֵיהוֹן:
וְאַשְׁכַּבְתֵּיהּ עַל בִּרְכָּהָא וּקְרַת לְגַבְרָא וּגְדַעַת יַת שְׁבַע גְדִילַת רֵישֵׁיהּ וּשְׁרִיאַת לְעַנָיוּתֵיהּ וַעֲדָא חֵילֵיהּ מִנֵיהּ:
וַאֲמַרַת פְּלִשְׁתָּאֵי אֲתוֹ עֲלָךְ שִׁמְשׁוֹן וְאִתְעַר מִשִׁנְתֵיהּ וַאֲמַר אֶפּוֹק כִּזְמַן בִּזְמַן וְאִתְגַבַּר וְהוּא לָא יְדַע אֲרֵי גְבוּרְתָּא דַייָ עֲדָת מִנֵיהּ:
וַאֲחָדוּהוּ פְּלִשְׁתָּאֵי וְנַקְרוּ יַת עֵינוֹהִי וְאַחִיתוּ יָתֵיהּ לְעַזָת וַאֲסָרוּהוּ בְּשַׁלְשְׁלָן דִנְחָשׁ וַהֲוָה טָחִין רֵחַיָא בְּבֵית אֲסִירַיָא:
וּשְׁרֵי שְׂעַר רֵישֵׁיהּ לְצַמָחָא בָּתַר דְאִתְגַלָח:
וְטוּרְנֵי פְלִשְׁתָּאֵי אִתְכְּנִישׁוּ לְדַבָּחָא דְבַח רַבָּא לְדָגוֹן טַעֲוַתְהוֹן וּלְחֶדְוָא וַאֲמַרוּ מְסָרַת דְחַלְתָּנָא בִּידָנָא יַת שִׁמְשׁוֹן בְּעֵיל דְבָבָנָא:
וַחֲזוֹ יָתֵיהּ עַמָא וְשַׁבָּחוּ יַת טַעֲוַתְהוֹן אֲרֵי אֲמַרוּ מְסָרַת דְחַלְתָּנָא בִּידָנָא יַת בְּעֵיל דְבָבָנָא וְיַת מַחֲרִיב אַרְעָנָא וְדִי מַסְגֵי יַת קְטִילָנָא:
וַהֲוָה כַּד שְׁפַר לִבְּהוֹן וַאֲמַרוּ קְרוּ לְשִׁמְשׁוֹן וִיחַיִךְ לָנָא וּקְרוֹ לְשִׁמְשׁוֹן מִבֵּית אֲסִירַיָא וְחַיִיךְ קֳדָמֵיהוֹן וַאֲקִימוּ יָתֵיהּ בֵּין עַמוּדַיָא:
וַאֲמַר שִׁמְשׁוֹן לְעוּלֵימָא דְמַתְקֵיף בִּידֵיהּ שְׁבוֹק מִנִי וַאֲמִישֵׁנִי יַת עַמוּדַיָא דְבֵיתָא מְתַקֵן עִלַוֵיהוֹן וְאִסְתְּמִיךְ עִלַוֵיהוֹן:
וּבֵיתָא מָלֵי גַבְרַיָא וּנְשַׁיָא וְתַמָן כָּל טוּרְנֵי פְלִישְׁתָּאֵי וְעַל אִגְרָא כִּתְלָתָא אַלְפִין גַבְרָא וְאִתְּתָא דְחָזָן בְּחוֹכָא דְשִׁמְשׁוֹן:
וְצַלִי שִׁמְשׁוֹן קֳדָם יְיָ וַאֲמַר יְיָ אֱלֹהִים יֵיעוּל דָכְרָנִי קֳדָמָךְ וְתַקֵפְנִי כְעַן בְּרַם זִמְנָא הָדָא יְיָ וְאִתְפְּרַע פּוּרְעָנוּתָא חָדָא מִתַּרְתֵּי עֵינַי דְנַקְרוּ לִי פְּלִשְׁתָּאֵי:
וַאֲחַד שִׁמְשׁוֹן יַת תְּרֵין עַמוּדֵי מְצִיעָאֵי דְבֵיתָא מַתְקַן עִלַוֵיהוֹן וְאִסְתְּמִיךְ עֲלֵיהוֹן חַד בִּימִינֵיהּ וְחַד בִּסְמָלֵיהּ:
וַאֲמַר שִׁמְשׁוֹן תְּמוּת נַפְשִׁי עִם פְּלִשְׁתָּאֵי וּנְגַד בְּחֵילָא וּנְפַל בֵּיתָא עַל טוּרְנַיָא וְעַל כָּל עַמָא דִי בֵיהּ וַהֲווֹ קְטִילַיָא דִי קְטַל בְּמוֹתֵיהּ סַגִיאִין מִדִי קְטַל בְּחַיוֹהִי:
וּנְחָתוּ אֲחוֹהִי וְכָל בֵּית אֲבוּהִי וּנְטָלוּ יָתֵיהּ וּסְלִיקוּ וּקְבָרוּ יָתֵיהּ בֵּין צָרְעָה וּבֵין אֶשְׁתָּאֹל בְּקִבְרָא דְמָנוֹחַ אֲבוּהִי וְהוּא דָן יַת יִשְׂרָאֵל עַסְרִין שְׁנִין:
