וַאֲמַר דָוִד בְּלִבֵּיהּ דִלְמָא אִתְמְסִיר יוֹמָא חָד בִּידָא דְשָׁאוּל לֵית לִי מִדַעַם דְטַב אֲרֵי אִשְׁתֵּיזָבָא אִשְׁתֵּיזֵב לְאַרַע פְלִשְׁתָּאֵי וִינוּחַ מִנִי שָׁאוּל לְמִבְעֵנִי עוֹד בְּכָל תְּחוּם אַרְעָא דְיִשְׂרָאֵל וְאִשְׁתֵּיזֵיב מִן יְדֵיהּ:
וְקָם דָוִד וַעֲבַר הוּא וְשִׁית מְאָה גַבְרָא דְעִמֵיהּ לְוַת אָכִישׁ בַּר מָעוֹךְ מַלְכָּא דְגַת:
וִיתֵיב דָוִד עִם אָכִישׁ בְּגַת הוּא וְגַבְרוֹהִי גְבַר וֶאֱנַשׁ בֵיתֵיה דָוִד וְתַרְתֵּין נְשׁוֹהִי אֲחִנוֹעַם דְמִיזְרְעֶאל וַאֲבִיגַיִל אִתַּת נָבָל דְמִכַּרְמְלָא:
וְאִתְחַוָא לְשָׁאוּל אֲרֵי עֲרַק דָוִד לְגַת וְלָא אוֹסִיף עוֹד לְמִבְעוֹהִי:
וַאֲמַר דָוִד לְאָכִישׁ אִם כְּעַן אַשְׁכָּחֵית רַחֲמִין בְּעֵינָךְ יִתְּנוּן לִי אֲתַר בַּחֲדָא מִן קִרְוֵי חַקְלָא וְאֵתֵיב תַּמָן וּלְמָא יְתֵיב עַבְדָךְ בְּקִירְוַת מַלְכוּתָא עִמָךְ:
וִיהַב לֵיה אָכִישׁ בְּיוֹמָא הַהוּא יַת צִקְלָג בְּכֵן הֲוָת צִקְלָג לְמַלְכַיָא דְבֵית יְהוּדָה עַד יוֹמָא הָדֵין:
וַהֲוָה מִנְיַן יוֹמַיָא דִיתֵיב דָוִד בְּקִרְוֵי פְלִשְׁתָּאֵי יוֹמִין עִדַן בְּעִדַן וְאַרְבְּעָא יַרְחִין:
וּסְלִיק דָוִד וְגַבְרוֹהִי וְאִתְנְגִידוּ עַל גְשׁוּרָאֵי וְעַל גִזְרָאֵי וַעֲמַלְקָאֵי אֲרֵי אִינוּן יַתְבַת אַרְעָא דְמֵעַלְמָא מַעֲלָנָא דְחַגְרָא וְעַד אַרְעָא דְמִצְרַיִם:
וּמְחֵי דָוִד יַת יַתְבֵי אַרְעָא וְלָא מְקַיֵם גְבַר וְאִתָּא וּשְׁבָא עָאן וְתוֹרִין וַחֲמָרִין וְגַמְלִין וּנְסִיב לְבוּשִׁין וְתָאִיב וְאָתֵי לְוַת אָכִישׁ:
וַאֲמַר אָכִישׁ לְאָן אִתְנְגַדְתּוּן יוֹמָא דֵין וַאֲמַר דָוִד עַל דָרוֹם יְהוּדָה וְעַל דָרוֹם יְרַחְמְאֵלָה וּלְדָרוֹם שַׁלְמָאָה:
וּגְבַר וְאִתָּא לָא מְקַיֵם דָוִד לְאַיְתָאָה לְגַת לְמֵימָר דִלְמָא יְחַווּן עֲלָנָא לְמֵימָר כְּדֵין עֲבַד דָוִד וּכְדֵין הֲוַת הִלְכָתֵיהּ כָּל יוֹמִין דַהֲוָה יָתֵיב בְּקִירְוֵי פְלִשְׁתָּאֵי:
וְהֵימִין אָכִישׁ בְּדָוִד לְמֵימָר אִתְגְרָאָה אִתְגָרֵי בְּעַמֵיהּ בְּיִשְׂרָאֵל וִיהֵי לִי עֲבַד פְּלַח לַעֲלָם:
