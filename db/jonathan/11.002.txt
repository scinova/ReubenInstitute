וַהֲוָה בְּאַסָקוּת יְיָ יַת אֵלִיָהוּ בְּעַלְעוֹלָא לְצֵית שְׁמַיָא וַאֲזַל אֵלִיָהוּ וֶאֱלִישָׁע מִן גִלְגָלָא:
וַאֲמַר אֵלִיָהוּ לֶאֱלִישָׁע תּוּב כְּעַן הָכָא אֲרֵי יְיָ שַׁלְחַנִי עַד בֵּית אֵל וַאֲמַר אֱלִישָׁע קַיָם הוּא יְיָ וְחַי נַפְשָׁךְ אִם אֶשְׁבְּקִינָךְ וּנְחָתוּ לְבֵית אֵל:
וּנְפַקוּ תַּלְמִידֵי נְבִיַיָא דִי בְּבֵית אֵל לְוַת אֱלִישָׁע וַאֲמַרוּ לֵיהּ הַיְדַעְתָּא אֲרֵי יוֹמָא דֵין יְיָ דְבַר יַת רִבּוֹנָךְ מֵעַל רֵישָׁךְ וַאֲמַר אַף אֲנָא יָדַעְנָא שְׁתוּקוּ:
וַאֲמַר לֵיהּ אֵלִיָהוּ אֱלִישָׁע תּוּב כְּעַן הָכָא אֲרֵי יְיָ שַׁלְחַנִי לִירִיחוֹ וַאֲמַר קַיָם הוּא יְיָ וְחֵי נַפְשָׁךְ אִם אֶשְׁבְּקִינָךְ וַאֲתוֹ לִירִיחוֹ:
וּקְרִיבוּ תַלְמִידֵי נְבִיַיָא דִי בִירִיחוֹ לְוַת אֱלִישָׁע וַאֲמַרוּ לֵיהּ הַיְדַעְתָּא אֲרֵי יוֹמָא דֵין יְיָ דְבַר יַת רִבּוֹנָךְ מִנָךְ וַאֲמַר אַף אֲנָא יְדַעְנָא שְׁתוּקוּ:
וַאֲמַר לֵיהּ אֵלִיָהוּ תּוּב כְּעַן הָכָא אֲרֵי יְיָ שַׁלְחֵנִי עַד יַרְדְנָא וַאֲמַר קַיָם הוּא יְיָ וְחַי נַפְשָׁךְ אִם אֶשְׁבְּקִינָךְ וַאֲזַלוּ תַּרְוֵיהוֹן:
וְחַמְשִׁין גַבְרָא מִתַּלְמִידֵי נְבִיַיָא אֲזָלוּ וְקָמוּ מִקֳבֵיל מֵרָחִיק וְתַרְוֵיהוֹן קָמוּ עַל יַרְדְנָא:
וּנְסִיב אֵלִיָהוּ יַת שׁוֹשִׁיפֵיהּ וְאַחְזַר וּמְחָא יַת מַיָא וְאִתְפְּלִיגוּ לְכָא וּלְכָא וַעֲבָרוּ תַרְוֵיהוֹן בְּיַבֶּשְׁתָּא:
וַהֲוָה כְמֶעְבַּרְהוֹן וְאֵלִיָהוּ אֲמַר לֶאֱלִישָׁע שְׁאַל מָה אַעְבֵּיד לָךְ עַד דְלָא אִידַבַּר מִלְוָתָךְ וַאֲמַר אֱלִישָׁע וִיהֵי כְעַן עַל חַד תְּרֵין בְּרוּחַ נְבוּאֲתָךְ עִמִי:
וַאֲמַר אַקְשֵׁיתָא לְמִשְׁאַל אִם תֶּחְזֵי יָתִי דַאֲנָא מִידַבֵּר מִלְוָתָךְ יְהֵי לָךְ כֵּן וְאִם לָא לָא יְהֵי:
וַהֲוָה עַד דְאִינוּן אָזְלִין מֵיזַל וּמִמַלְלִין וְהָא רְתִיכִין דְאֶשְׁתָּא וְסוּסְוָן דְאֶשְׁתָּא וְאַפְרִישׁוּ בֵּין תַּרְוֵיהוֹן וּסְלִיק אֵלִיָהוּ בְּעִלְעוֹלָא לְצֵית שְׁמַיָא:
וֶאֱלִישָׁע חָזִי וְהוּא מְצַוַח רַבִּי רַבִּי דְטַב לְהוֹן לְיִשְׂרָאֵל בִּצְלוּתֵיהּ מֵרְתִיכִין וּפָרָשִׁין וְלָא חֲזָהִי עוֹד וְאַתְקֵיף בִּלְבוּשֵׁיהּ וְסַדְקִינוּן לִתְרֵין סְדִיקִין:
וַאֲרֵים יַת שׁוֹשִׁיפָא דְאֵלִיָהוּ דִי נְפָלַת מִנֵיהּ וְתַב וְקָם עַל כֵּיף יַרְדְנָא:
וּנְסִיב יַת שׁוֹשִׁיפָא דְאֵלִיָהוּ דִנְפָלַת מִנֵיהּ וּמְחָא יַת מַיָא וַאֲמַר קַבֵּיל בָּעוּתִי יְיָ אֱלָהֵיהּ דְאֵלִיָהוּ אַף הוּא וּמְחָא יַת מַיָא וְאִתְפְּלִיגוּ לְכָא וּלְכָא וַעֲבַר אֱלִישָׁע:
וַחֲזוֹהִי תַלְמִידֵי נְבִיַיָא דְבִירִיחוֹ מִקָבֵיל וַאֲמַרוּ נְחַת רוּחֵיהּ דְאֵלִיָהוּ עַל אֱלִישָׁע וַאֲתוֹ לִקֳדָמוּתֵיהּ וּסְגִידוּ לֵיהּ עַל אַרְעָא:
וַאֲמַרוּ לֵיהּ הָא כְעַן אִית עִם עַבְדָךְ חַמְשִׁין גַבְרִין בְּנֵי חֵילָא יֵיזְלוּן כְּעַן וְיִבְעוּן יַת רִבּוֹנָךְ דִלְמָא נַטְלֵיהּ רוּחַ מִן קֳדָם יְיָ וּרְמָהִי בְּחַד מִן טוּרַיָא אוֹ בְּחַד מִן חֵילַיָא וַאֲמַר לָא תִשְׁלְחוּן:
וְאַתְקִיפוּ בֵּיהּ עַד סַגְיָא וַאֲמַר שְׁלָחוּ וּשְׁלַחוּ חַמְשִׁין גַבְרָא וּבְעוֹ תְלָתָא יוֹמִין וְלָא אַשְׁכְּחוֹהִי:
וְתָבוּ לְוָתֵיהּ וְהוּא יָתֵיב בִּירִיחוֹ וַאֲמַר לְהוֹן הֲלָא אֲמַרִית לְכוֹן לָא תֵיזְלוּן:
וַאֲמַרוּ אֱנָשֵׁי קַרְתָּא לֶאֱלִישָׁע הָא כְעַן מוֹשַׁב קַרְתָּא שַׁפִּיר כְּמָא דְרִבּוֹנִי חֲזֵי וּמַיָא בִּישִׁין וְאַרְעָא מְתַכְּלָא:
וַאֲמַר אַיְתוֹ לִי צְלוּחִיתָא חַדְתָּא וְשַׁווּ תַמָן מִלְחָא וְאַיְתִיאוּ לְוָתֵיהּ:
וּנְפַק לְמַפְקָנָא דְמַיָא וּרְמָא תַּמָן מִלְחָא וַאֲמַר כִּדְנַן אֲמַר יְיָ אַסֵיתִי לְמַיָא הָאִלֵין וְלָא יְהֵי מִתַּמָן עוֹד מוֹתָא וּמְתַכְּלָא:
וְאִתַּסִיאוּ מַיָא עַד יוֹמָא הָדֵין כְּפִתְגָמָא דֶאֱלִישָׁע דְמַלֵיל:
וּסְלִיק מִתַּמָן לְבֵית אֵל וְהוּא סְלִיק בְּאוֹרְחָא וְיַנְקִין זְעֵירִין נְפַקוּ מִן קַרְתָּא וְאִתְלַעֲבוּ בֵיהּ וַאֲמַרוּ לֵיהּ סַק קַרְחָא סַק קַרְחָא:
וְאִתְפְּנֵי לַאֲחוֹרוֹהִי וַחֲזִינוּן וְלַטִינוּן בִּשְׁמָא דַייָ וּנְפַקוּ תַּרְתֵּין דוּבִין מִן חוּרְשָׁא וּבְזָעָא מִנְהוֹן אַרְבְּעִין וּתְרֵין יַנְקִין:
וַאֲזַל מִתַּמָן לְטוּר כַּרְמְלָא וּמִתַּמָן תַּב לְשֹׁמְרוֹן:
