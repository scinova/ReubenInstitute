וַהֲוָה פִּתְגַם נְבוּאָה מִן קֳדָם יְיָ עִמִי לְמֵימָר:
בַּר אָדָם אִתְנַבֵּי לִבְנֵי עַמָךְ וְתֵימַר לְהוֹן אַרְעָא אֲרֵי אַיְתֵי עֲלָהּ דְקַטְלִין בְּחַרְבָּא וְיִדְבְּרוּן עַמָא דְאַרְעָא גַבְרָא חַד מִבֵּינֵיהוֹן וִימַנוּן יָתֵיהּ לְהוֹן לְאַזְהָרָא:
וְיֶחֱזֵי יַת דְקַטְלִין בִּחַרְבָּא אַתְיָן עַל אַרְעָא וְיִתְקַע בְּשׁוֹפְרָא וְיַזְהַר יַת עַמָא:
וְיִשְׁמַע דִי שְׁמַע יַת קַל שׁוֹפָרָא וְלָא אִזְדְהַר וַאֲתוֹ קָטוֹלֵי חַרְבָּא וְדַבְּרוֹהִי חוֹבַת קְטוֹלֵיהּ בְּרֵישֵׁיהּ יְהֵי:
יַת קַל שׁוֹפָרֵיהּ שְׁמַע וְלָא אִזְדְהַר חוֹבַת קָטוֹלֵי בֵּיהּ תְּהֵי וְהוּא אִזְדְהַר נַפְשֵׁיהּ שֵׁיזֵיב:
וּמַזְהֲרָנָא אֲרֵי יֶחֱזֵי יַת דְקַטְלִין בְּחַרְבָּא אָתָן וְלָא תְקַע בְּשׁוֹפָרָא וְעַמָא לָא אִזְדְהַר וַאֲתוֹ קָטוֹלֵי חַרְבָּא וְדַבָּרוּ מִנְהוֹן נַפְשָׁא הוּא בְּחוֹבֵיהּ אִידְכַּר וּדְמֵהּ מִיַד אַזְהֲרָנָא אֶתְבַּע:
וְאַתְּ בַּר אָדָם מַלֵיף מַנִיתָךְ לְבֵית יִשְׂרָאֵל וּתְקַבֵּיל מִמֵימְרִי פִתְגַם וְתַזְהֵר יַתְהוֹן מִלְמֶחֱטֵי קֳדָמַי:
בִּדְאֲמָרֵית לְרַשִׁיעָא רַשִׁיעָא מְמָת תְּמוּת וְלָא מַלֵילְתָּא לְאַזְהָרָה רַשִׁיעָא מֵאוֹרְחֵיהּ הוּא רַשִׁיעָא בְּחוֹבֵהּ יְמוּת וּדְמֵיהּ מִן יְדָךְ אֶתְבַּע:
וְאַתְּ אֲרֵי אַזְהַרְתָּ רַשִׁיעָא מֵאוֹרְחֵהּ לְמֵיתַב מִנָהּ וְלָא תַב מֵאוֹרְחֵיהּ הוּא בְּחוֹבֵיהּ יְמוּת וְאַתְּ נַפְשָׁךְ שֵׁיזֵיבְתָּא:
וְאַתְּ בַּר אָדָם אֱמַר לְבֵית יִשְׂרָאֵל כֵּן אֲמַרְתּוּן לְמֵימָר אֲרֵי חוֹבָנָא וְחַטָאנָא עֲלָנָא מִתְיַהֲבִין וּבְהוֹן אֲנַחְנָא מִתְמְסִין וְאֵיכְדֵין נְחֵי:
אֱמַר לְהוֹן קַיָם אֲנָא אֲמַר יְיָ אֱלֹהִים אִם רָעֵינָא בְּמִית רַשִׁיעָא אֱלָהֵין כַּד יְתוּב רַשִׁיעָא מֵאוֹרְחֵיהּ וְיֵחִי תּוּבוּ מֵאוֹרְחַתְכוֹן בִּישָׁתָא וְלָמָה תְמוּתוּן בֵּית יִשְׂרָאֵל:
וְאַתְּ בַּר אָדָם אֱמַר לִבְנֵי עַמָךְ זְכוּת זַכָּאָה לָא תְשֵׁיזְבִינֵיהּ בְּיוֹמָא דְחוֹבְתֵּהּ וְחוֹבַת חַיָבָא לָא יִתְקַל בָּהּ בְּיוֹמָא דִיתוּב מֵחוֹבֵיהּ וְצַדִיקָא לָא יֵכוּל לְאִתְקַיְמָא בֵיהּ בְּיוֹמָא דְחוֹבְתֵּיהּ:
בִּדְאֲמָרֵית לְזַכָּאָה מֵיחָא יְחֵי וְהוּא יִתְרְחִיץ עַל זְכוּתֵיהּ וַעֲבַד שְׁקַר כָּל זָכוּתֵיהּ לָא יִדְכְּרַן וּבְשִׁקְרֵיהּ דַעֲבֵד בֵּיהּ יְמוּת:
וּבִדְאֲמָרֵית לְרַשִׁיעָא מְמַת תְּמוּת וִיתוּב מֵחוֹבוֹהִי וְיַעְבֵיד דִין דִקְשׁוֹט וּזְכוּ:
מַשְׁכּוֹנָא יְתִיב רַשִׁיעָא גְזֵילָא שַׁלֵים בִּגְזֵירַת חַיַיָא הֲלֵיךְ בְּדִיל דְלָא לְמֶעְבַּד שְׁקַר מֵיחָא יְחֵי וְלָא יְמוּת:
כָּל חוֹבוֹהִי דְחַב לָא יִדְכְרַן לֵיהּ דִין דִקְשׁוֹט וּזְכוּ עֲבַד מֵיחָא יְחֵי:
וְיֵימְרוּן בְּנֵי עַמָךְ לָא מְפָרְשָׁן לָנָא אוֹרְחַת טוּבָא דַייָ וְאִינוּן אוֹרְחַתְהוֹן דִילְהוֹן לָא תִקְנָן:
כַּד יְתוּב זַכָּאָה מִזַכְוָתֵהּ וְיַעְבֵּיד שְׁקַר וִימוּת בְּהוֹן:
וְכַד יְתוּב רַשִׁיעָא מֵחוֹבוֹהִי וְיַעְבֵּיד דִין דִקְשׁוֹט וּזְכוּ עֲלֵיהוֹן הוּא יִתְקַיָם:
וַאֲמַרְתּוּן לָא מְפָרְשָׁן לָנָא אוֹרְחַת טוּבָא דַייָ אֱנַשׁ כְּאוֹרְחֵהּ אֶתְפְּרַע מִנְכוֹן בֵּית יִשְׂרָאֵל:
וַהֲוָה בְּתַרְתֵּי עַסְרֵי שְׁנִין בַּעֲשִׂירָאָה בְּחַמְשָׁא לְיַרְחָא לְגָלוּתָנָא אֲתָא לְוָתִי מְשֵׁיזְבָא מִירוּשְׁלֵם לְמֵימָר אִתְּרַעַת קַרְתָּא:
וּנְבוּאָה מִן קֳדָם יְיָ הֲוַת עִמִי בְּרַמְשָׁא קֳדָם מֵיתֵי מְשֵׁיזְבָא וּפְתַח יַת פּוּמִי עַד מֵיתֵי לְוָתִי בְצַפְרָא וְאֶפְתַּח פּוּמִי וְלָא שַׁתְקֵית עוֹד:
וַהֲוָה פִּתְגָם נְבוּאָה מִן קֳדָם יְיָ עִמִי לְמֵימָר:
בַּר אָדָם יָתְבֵי חוּרְבָתָא הָאִילֵן עַל אַרְעָא דְיִשְׂרָאֵל אָמְרִין לְמֵימָר חַד הֲוָה אַבְרָהָם בְּעַלְמָא וִירֵית יַת אַרְעָא וַאֲנַחְנָא סַגִיאִין לָנָא יְהִיבָא אַרְעָא לְמֵירְתָהּ:
בְּכֵן אֲמַר לְהוֹן כִּדְנַן אֲמַר יְיָ אֱלֹהִים עַל דַם זַכַּאי אַתּוּן אָכְלִין וְעֵינֵיכוֹן אַתּוּן זָקְפִין לְפוּלְחַן טַעֲוַתְכוֹן וְדַם זַכַּאי אַתּוּן אֶשְׁדִין וְאַרְעָא אַתּוּן מְדַמָן לְמֵירְתָהּ:
וְאִתְעַתַּדְתּוּן עַל תּוּקְפֵיכוֹן עֲבַדְתּוּן תּוֹעֵיבָא וּגְבַר יַת אִתַּת חַבְרֵיהּ סְאֵיבְתּוּן וְאַרְעָא אַתּוּן מְדַמָן לְמֵירְתָהּ:
כִּדְנַן תֵּימַר לְהוֹן כִּדְנַן אֲמַר יְיָ אֱלֹהִים קַיָם אֲנָא אִם לָא דִי בְּחָרְבָא בְּחַרְבָּא יִתְקַטְלוּן וְדִי עַל אַפֵּי בָּרָא לְחַיְתָא מְסַרְתֵּיהּ לְשֵׁיצָיוּתְהוֹן וְדִי בִּמְצָדְתָא וּבִמְעֲרָתָא בָּמוֹתָא יְמוּתוּן:
וְאֶתֵּן יַת אַרְעָא לְצָדוּ וְאִשְׁתְּמָמוּ וְיִבְטַל יְקַר תּוּקְפָהּ וְיִצְדוּן טוּרֵי יִשְׂרָאֵל מִבְּלֵי עָדֵי:
וְיֵדְעוּן אֲרֵי אֲנָא יְיָ בִּדְאֶתֵּן יַת אַרְעָא לְצָדוּ וְאִשְׁתְּמָמוּ עַל כָּל תּוֹעֲבָתְהוֹן דַעֲבָדוּ:
וְאַתְּ בַּר אָדָם בְּנֵי עַמָךְ דְדָנָן בָּךְ בִּסְטַר כָּתְלַיָא וּבְתַרְעֵי בָתַּיָא וִימַלֵיל חַד עִם חַד גְבַר יַת אֲחוּהִי לְמֵימָר אֱתוּ דִנְפַק מִן קֳדָם יְיָ:
וְיֵיתוּן עֲלָךְ כְּמֵיתֵי גַבְרִין תַּלְמִידִין וִיתִיבוּן קֳדָמָךְ עַמִי וְשָׁמְעִין יַת פִּתְגָמָךְ וְיַתְהוֹן לָא יַעְבְּדוּן אֲרֵי תּוּלְעָבָא בְּפוּמְהוֹן אִינוּן עָבְדִין בָּתַר מָמוֹן אוּנְסֵיהוֹן לִבְּהוֹן טָעֵי:
וְהָא אַתְּ לְהוֹן כִּזְמַר אֲבּוּבִין דִבְּסִים קָלֵיהּ וּמוֹטַב לְנַגָנָא וְשָׁמְעִין יַת פִּתְגָמָךְ וּלְמֶעְבַּדְהוֹן לֵית אִינוּן צָבָן:
וּבְמֵיתֵיךְ הָא אַתְיָא וְיֵדְעוּן אֲרֵי נְבִיָא הֲוָה בֵינֵיהוֹן:
