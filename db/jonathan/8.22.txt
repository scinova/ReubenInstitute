וַאֲזַל דָוִד מִתַּמָן וְאִשְׁתֵּיזֵיב לִמְעָרַת עֲדֻלָם וּשְׁמָעוּ אַחוֹהִי וּבֵית אֲבוּהִי וּנְחָתוּ לְוָתֵיהּ לְתַמָן:
וְאִתְכְּנִישׁוּ לְוָתֵיהּ כָּל גְבַר מָעִיק רוּחַ וְכָל גְבַר דִי לֵיהּ מָרֵי רְשׁוּ וְכָל גְבַר מָרִיר נְפַשׁ וַהֲוָה עֲלֵיהוֹן לְרַבָּא וַהֲווֹ עִמֵיהּ כְּאַרְבַּע מְאָה גַבְרָא:
וַאֲזַל דָוִד מִתַּמָן לְמִצְפַּיָא דְמוֹאָב וַאֲמַר לְמַלְכָּא דְמוֹאָב יִפּוֹק כְּעַן אַבָּא וְאִמָא עִמְכוֹן עַד דְאִידַע מַה יַעְבֵּיד לִי יְיָ:
וְאַשְׁרִינוּן קֳדָם מַלְכָּא דְמוֹאָב וִיתִיבוּ עִמֵיהּ כָּל יוֹמִין דַהֲוָה דָוִד מִטְמַר בַמְצוּדַיָא:
וַאֲמַר גָד נְבִיָא לְדָוִד לָא תִתֵּב בַּמְצוּדַיָא אִיתָא וֶאֱזֵיל לָךְ לַאֲרַע יְהוּדָה וַאֲזַל דָוִד וַאֲתָא לְחוּרְשָׁא דְחָרֶת:
וּשְׁמַע שָׁאוּל אֲרֵי אִתְיְדַע דָוִד וְגַבְרַיָא דִי עִמֵיהּ וְשָׁאוּל יָתֵיב בְּגִבְעָתָא תְּחוֹת אַשְׁלָא בְרָמָתָא וּמוּרְנִיתֵיהּ בִּידֵיהּ וְכָל עַבְדוֹהִי קַיְמִין עֲלוֹהִי:
וַאֲמַר שָׁאוּל לְעַבְדוֹהִי דְקַיְמִין עֲלוֹהִי שְׁמָעוּ כְעַן בְּנֵי שִׁבְטָא דְבֵית בִּנְיָמִין אַף לְכֻלְכוֹן יִתֵּן בַּר יִשַׁי חַקְלִין וְכַרְמִין לְכֻלְכוֹן יְמַנֵי רַבָּנֵי אַלְפִין וְרַבָּנֵי מַאֲוָתָא:
אֲרֵי מְרַדְתּוּן כֻּלְכוֹן עֲלֵי וְלֵית דְחַוֵי לִי כַּד גְזַר קְיַם בְּרִי עִם בַר יִשַׁי וְלֵית דְכָאִיב מִנְכוֹן עֲלַי וּמְחַוֵי לִי אֲרֵי אָקִים בְּרִי יַת עַבְדִי עֲלֵי לִכְמָנָא כְּיוֹמָא הָדֵין:
וַאֲתֵיב דוֹאֵג אֱדוֹמָאָה וְהוּא מְמַנָא עַל עַבְדֵי שָׁאוּל וַאֲמַר חַזֵיתִי יַת בַּר יִשַׁי אֲתָא לְנוֹב לְוַת אֲחִימֶלֶךְ בַּר אֲחִיטוּב:
וְשָׁאִיל לֵיהּ בְּמֵימְרָא דַייָ וּזְוָדִין יְהַב לֵיהּ וְיַת חַרְבָּא דְגָלְיַת פְּלִשְׁתָּאָה יְהַב לֵיהּ:
וּשְׁלַח מַלְכָּא לְמִקְרֵי יַת אֲחִימֶלֶךְ בַּר אֲחִיטוּב כַּהֲנָא וְיַת כָּל בֵּית אֲבוּהִי כַּהֲנַיָא דִי בְנוֹב וַאֲתוֹ כֻּלְהוֹן לְוַת מַלְכָּא:
וַאֲמַר שָׁאוּל שְׁמַע כְּעַן בַּר אֲחִיטוּב וַאֲמַר הָא אֲנָא רִבּוֹנִי:
וַאֲמַר לֵיהּ שָׁאוּל לְמָה מְרַדְתּוּן עֲלַי אַתְּ וּבַר יִשַׁי בְמַתְּנָךְ לֵיהּ מָזוֹן וְחַרְבָּא וּשְׁאָלִית לֵיהּ בְּמֵימְרָא דַייָ לְמֵיקַם עֲלַי לִכְמָנָא כְּיוֹמָא הָדֵין:
וַאֲתֵיב אֲחִימֶלֶךְ יַת מַלְכָּא וַאֲמַר וּמָן בְּכָל עַבְדָךְ כְּדָוִד מְהֵימָן וְחַתְנָא דְמַלְכָּא וְרַב עַל מִשְׁמַעְתָּךְ וְיַקִיר בְּבֵיתָךְ:
יוֹמָא דֵין שְׁרִיתִי לְמִשְׁאַל לֵיהּ בְּמֵימְרָא דַייָ חַס לִי לָא יְשַׁוֵי מַלְכָּא בְּעַבְדֵיהּ פִּתְגָמָא בְּכָל בֵּית אַבָּא אֲרֵי לָא יְדַע עַבְדָךְ בְּכָל דָא פִּתְגַם זְעֵיר אוֹ רַב:
וַאֲמַר מַלְכָּא מֵמַת תְּמוּת אֲחִימֶלֶךְ אַתְּ וְכָל בֵּית אָבוּךְ:
וַאֲמַר מַלְכָּא לִרְהָטַיָא דְקַיְמִין עֲלוֹהִי אִסְתְּחָרוּ וּקְטֻלוּ כַּהֲנַיָא דַייָ אֲרֵי אַף יְדֵיהוֹן עִם דָוִד וַאֲרֵי יְדָעוּ אֲרֵי עֲרִיק הוּא וְלָא חַווּ לִי וְלָא אֲבוֹ עַבְדֵי מַלְכָּא לְאוֹשָׁטָא יַת יְדֵיהוֹן לְמִשְׁלַט בְּכַהֲנַיָא דַיָי:
וַאֲמַר מַלְכָּא לְדוֹאֵג אִסְתְּחַר אַתְּ וּשְׁלוֹט בְּכַהֲנַיָא וְאִסְתְּחַר דוֹאֵג אֱדוֹמָאָה וּשְׁלַט הוּא בְּכַהֲנַיָא וּקְטַל בְּיוֹמָא הַהוּא תַּמְנָן וְחַמְשָׁא גַבְרִין דְכַשְׁרִין לְמִלְבַּשׁ אֵפוֹד דְבוּץ:
וְיַת נוֹב קִרְיַת כַּהֲנַיָא קְטַל לְפִתְגַם דְחָרֶב מִגְבַר וְעַד אִתְּתָא מֵעוּלֵים וְעַד יַנִיק וְתוֹר וַחֲמָר וְאִימַר לְפִתְגָם דְחָרֶב:
וְאִשְׁתֵּיזֵיב בְּרָא חֲדָא לַאֲחִימֶלֶךְ בַּר אֲחִטוּב וּשְׁמֵיהּ אֶבְיָתָר וַעֲרַק בָּתַר דָוִד:
וְחַוֵי אֶבְיָתָר לְדָוִד אֲרֵי קְטַל שָׁאוּל יַת כַּהֲנַיָא דַיָי:
וַאֲמָר דָוִד לְאֶבְיָתָר יָדַעְנָא בְּיוֹמָא הַהוּא אֲרֵי תַמָן דוֹאֵג אֱדוֹמָאָה אֲרֵי חַוָאָה יְחַוֵי לְשָׁאוּל אֲנָא גְרָמִית לְאִסְתַּקָפָא בְּכָל נְפַשׁ בֵּית אֲבוּךְ:
תִּיב עִמִי לָא תִדְחַל אֲרֵי דְבָעֵי לְמִקְטְלִי בָּעֵי לְמִקְטְלָךְ אֲרֵי נָטִיר אַתְּ עִמִי:
