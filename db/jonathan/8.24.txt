וּסְלֵיק דָוִד מִתַּמָן וִיתֵב בִּמְצַדְתָּא עֵין גֶדִי:
וַהֲוָה כַּד תַּב שָׁאוּל מִבָתַר פְּלִשְׁתָּאֵי וְחַוִיאוּ לֵיה לְמֵימַר הָא דָוִד בְּמַדְבְּרָא דְעֵין גֶדִי:
וּדְבַר שָׁאוּל תְּלָתָא אַלְפִין גְבַר עוּלֵים מִכָּל יִשְׂרָאֵל וַאֲזַל לְמִבְעֵי יַת דָוִד וְגַבְרוֹהִי עַל אַפֵּי שְׁקִיפֵי כֵיפַיָא:
וַאֲתָא לְחַטְרֵי עָאנָא דְעַל אוֹרְחָא וְתַמָן מְעַרְתָּא וְעַל שָׁאוּל לְמֶעְבַּד צוֹרְכֵיהּ וְדָוִד וְגַבְרוֹהִי בִסְיָפֵי מְעַרְתָּא יָתְבִין:
וַאֲמַרוּ גַבְרֵי דָוִד לֵיהּ הָא יוֹמָא דַאֲמַר יְיָ לָךְ הָא אֲנָא מְסַר יַת בְּעֵיל דְבָבָךְ בִּידָךְ וְתַעֲבֵּיד לֵיהּ כְּמָא דְתַקֵן בְּעֵינָךְ וְקָם דָוִד וּפְסַק יַת כְּנַף מְעִילָא דִלְשָׁאוּל בְּרָז:
וַהֲוָה בָתַר כֵּן וְחָשׁ דָוִד בְּלִבֵּיהּ עַל דִי פְסַק יַת כַּנְפָא דִלְשָׁאוּל:
וַאֲמַר לְגַבְרוֹהִי חָס לִי מִן קֳדָם יְיָ אִם אַעְבֵּיד יַת פִּתְגָמָא הָדֵין לְרִבּוֹנִי לִמְשִׁיחָא דַייָ לְאוֹשָׁטָא יְדִי בֵּיהּ אֲרֵי מְשִׁיחָא דַייָ הוּא:
וּפַיֵס דָוִד יַת גַבְרוֹהִי בְּפִתְגָמַיָא וְלָא שַׁבְקִינוּן לְמֵיקַם עַל שָׁאוּל וְשָׁאוּל קָם מִמְעַרְתָּא וַאֲזַל בְּאוֹרְחָא:
וְקָם דָוִד בָּתַר כֵּן וּנְפַק מִן מְעַרְתָּא וּקְרָא בָּתַר שָׁאוּל לְמֵימָר רִבּוֹנִי מַלְכָּא וְאִסְתְּכֵי שָׁאוּל לַאֲחוֹרוֹהִי וּכְרַע דָוִד עַל אַפּוֹהִי עַל אַרְעָא וּסְגִיד:
וַאֲמַר דָוִד לְשָׁאוּל לְמָא אַתְּ שָׁמַע יַת פִּתְגָמֵי בְנֵי אֱנָשָׁא דְאָמְרִין הָא דָוִד כָּמֵין בִּישְׁתָּךְ:
הָא יוֹמָא דֵין חֲזָאָה עֵינָךְ יַת דְמַסְרָךְ יְיָ יוֹמָא דֵין בִּידִי בִּמְעַרְתָּא וַאֲמַרוּ אוֹחֲרָנִין לְמִקְטְלָךְ וְחָסִית עֲלָךְ וַאֲמַרִית לָא אוֹשִׁיט יְדִי בְּרִבּוֹנִי אֲרֵי מְשִׁיחָא דַייָ הוּא:
וְרִבּוֹנִי אִסְתַּכַּל אַף חֲזֵי יַת כְּנַף מְעִילָךְ בִּידִי אֲרֵי בְּמִפְסְקִי יַת כְּנַף מְעִילָךְ וְלָא קְטַלְתִּיךְ דַע וַחֲזוֹ אֲרֵי לֵית בִּידִי חוֹב וְסוּרְחַן וְלָא חָטִית לָךְ וְאַתְּ כָּמִין עַל נַפְשִׁי לְמִקְטְלָהּ:
יְדִין יְיָ בֵּינָא וּבֵינָךְ וְיִתְפְּרַע לִי יְיָ מִנָךְ וִידִי לָא תְהִי בָךְ:
כְּמָא דְאָמְרִין מַתְלָא דְמִלְקָדְמִין מֵרַשִׁיעִין תִּפּוֹק חוֹבַתְהוֹן וִידִי לָא תְהֵי בָךְ:
בָּתַר מַן נְפַק מַלְכָּא דְיִשְׂרָאֵל בָּתַר מַן אַתְּ רָדִיף בָּתַר חֲלַשׁ חָד בָּתַר הֶדְיוֹט חָד:
וִיהֵי יְיָ לְדַיָין וְיַעְבֵּד דִין בֵּינָא וּבֵינָךְ וּגְלֵי קֳדָמוֹהִי וִידוּן דִינִי וְיִתְפְּרַע עוּלְבָּנִי מִמָךְ:
וַהֲוָה כַּד שֵׁיצֵי דָוִד לְמַלָלָא יַת פִּתְגָמַיָא הָאִלֵין עִם שָׁאוּל וַאֲמַר שָׁאוּל הֲקָלָךְ דֵין בְּרִי דָוִד וַאֲרֵים שָׁאוּל קָלֵיהּ וּבְכָא:
וַאֲמַר לְדָוִד זַכַּאי אַתְּ מִנִי אֲרֵי אַתְּ גְמַלְתַּנִי טַבְתָּא וַאֲנָא גְמַלְתָּךְ בִּישְׁתָא:
וְאַתְּ חֲוֵיתָא יוֹמָא דֵין יַת דַעֲבַדְתָּ עִמִי טִיבוּ וְיַת דְמַסְרַנִי יְיָ בִּידָךְ וְלָא קְטַלְתָּנִי:
וַאֲרֵי יַשְׁכַּח גְבַר יַת בְּעֵיל דְבָבֵיהּ וְיִשְׁלְחִנֵיהּ בְּאוֹרַח תַּקָנָא וַיָי יְשַׁלְמִנָךְ טַבְתָא חֲלַף יוֹמָא דֵין דַעֲבַדְתָּא לִי:
וּכְעַן הָא יָדַעְנָא אֲרֵי מִמְלַךְ תִּמְלַךְ וְתִתְקַיֵים בִּידָךְ מַלְכוּתָא דְיִשְׂרָאֵל:
וּכְעַן קַיֵים לִי בְּמֵימְרָא דַייָ דְלָא תְשֵׁיצֵי יַת בְּנַי בַּתְרָי וְלָא תִפְסוֹק יַת שְׁמִי מִבֵּית אַבָּא:
וְקַיֵם דָוִד לְשָׁאוּל וַאֲזַל שָׁאוּל לְבֵיתֵיהּ וְדָוִד וְגַבְרוֹהִי סְלִיקוּ לִמְצַדְתָּא:
