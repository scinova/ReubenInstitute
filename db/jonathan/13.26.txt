בְּרֵישׁ מַלְכוּת יְהוֹיָקִים בַּר יֹאשִׁיָה מְלַךְ שִׁבְטָא דְבֵית יְהוּדָה הֲוָה פִּתְגָמָא הָדֵין מִן קֳדָם יְיָ לְמֵימָר:
כִּדְנַן אֲמַר יְיָ קוּם בְּדָרַת בֵּית מִקְדָשָׁא דַייָ וְתִתְנַבֵּי עַל כָּל קִרְוַיָא דְבֵית יְהוּדָה דְעָלִין לְמִסְגַד בְּבֵית מַקְדְשָׁא דַייָ יַת כָּל פִּתְגָמַיָא דְפַקֵידְתָּךְ לְאִתְנַבָּאָה לְהוֹן לָא תִמְנַע פִּתְגָם:
מָאִים יִשְׁמְעוּן וִיתוּבוּן גְבַר מֵאוֹרְחֵיהּ בִּישָׁא וַאֲתֵיב מִן בִּישְׁתָּא דַאֲנָא חָשֵׁיב לְמֶעְבַּד לְהוֹן מִקֳדָם בִּישׁוּת עוֹבָדֵיהוֹן:
וְתֵימַר לְהוֹן כִּדְנַן אֲמַר יְיָ אִם לָא תְקַבְּלוּן לְמֵימְרִי לִמְהַךְ בְּאוֹרַיְתִי דִיהָבֵית קֳדָמֵיכוֹן:
לְקַבְּלָא לְפִּתְגָמֵי עַבְדֵי נְבִיַיָא דַאֲנָא שְׁלַח לְוַתְכוֹן מַקְדֵם וּמַשְׁלַח וְלָא קַבֵּילְתּוּן:
וְאֶתֵּן יַת בֵּיתָא הָדֵין כְּשִׁלוֹ וְיַת קַרְתָּא הָדָא אֶתֵּן לִלְוָטָא לְכָל עַמְמֵי אַרְעָא:
וּשְׁמָעוּ כַּהֲנַיָא וְסַפְרַיָא וְכָל עַמָא דְיִרְמְיָה מִתְנַבֵּי יַת פִּתְגָמַיָא הָאִלֵין בְּבֵית מַקְדְשָׁא דַיָי:
וַהֲוָה כַּד שֵׁיצֵי יִרְמְיָה לְאִתְנַבָּאָה יַת כָּל דְפַקֵיד יְיָ לְמַלָלָא עִם כָּל עַמָא וַאֲחָדוּ יָתֵיהּ כַּהֲנַיָא וְסַפְרַיָא וְכָל עַמָא לְמֵימַר מֵימַת תְּמוּת:
מַדֵין אִתְנַבֵּיתָא בִּשְׁמָא דַייָ לְמֵימָר כְּשִׁלוֹ יְהֵי בֵיתָא הָדֵין וְקַרְתָּא הָדָא תֶּחֱרוֹב מִבְּלִי יָתֵיב וְאִתְכְּנֵישׁ כָּל עַמָא עַל יִרְמְיָה בְּבֵית מַקְדְשָׁא דַיָי:
וּשְׁמָעוּ רַבְרְבֵי יְהוּדָה יַת פִּתְגָמַיָא הָאִלֵין וּסְלִיקוּ מִבֵּית מַלְכָּא לְבֵית מַקְדְשָׁא דַייָ וְתִיבוּ בְּמַעֲלָנָא דִתְרַע בֵּית מַקְדְשָׁא דַייָ מַדִינְחָאָה:
וַאֲמָרוּ כַּהֲנַיָא וְסַפְרַיָא לְרַבְרְבַיָא וּלְכָל עַמָא לְמֵימַר חוֹבַת דִין דִקְטוֹל לְגַבְרָא הָדֵין אֲרֵי אִתְנַבֵּי עַל קַרְתָּא הָדָא כְּמָה דִשְׁמַעְתּוּן בְּאוּדְנֵיכוֹן:
וַאֲמַר יִרְמְיָה לְכָל רַבְרְבַיָא וּלְכָל עַמָא לְמֵימַר יְיָ שַׁלְחַנִי לְאִתְנַבָּאָה עַל בֵּיתָא הָדֵין וְעַל קַרְתָּא הָדָא יַת כָּל פִּתְגָמַיָא דִשְׁמַעֲתּוּן:
וּכְעַן אַתְקִינוּ אוֹרְחַתְכוֹן וְעוֹבָדֵיכוֹן וְקַבִּילוּ לְמֵימְרָא דַייָ אֱלָהָכוֹן וִיתוּב יְיָ מִן בִּישְׁתָּא דְמַלֵיל עֲלֵיכוֹן:
וַאֲנָא הָאֲנָא מְסִיר בְּיֶדְכוֹן עֲבִידוּ לִי כְּמָה דְתַקִין וּכְמָה דְכָשַׁר בְּעֵינֵיכוֹן:
בְּרַם מֵידַע תֵּידְעוּן אֲרֵי אִם קַטְלִין אַתּוּן יָתִי אֲרֵי דַם זַכַּאי אַתּוּן יָהֲבִין עֲלֵיכוֹן וְעַל קַרְתָּא הָדָא וְעַל יַתְבָהָא אֲרֵי בִקְשׁוֹט שַׁלְחַנִי יְיָ עֲלֵיכוֹן לְאִתְנַבָּאָה קֳדָמֵיכוֹן יַת כָּל פִּתְגָמַיָא הָאִלֵין:
וַאֲמָרוּ רַבְרְבַיָא וְכָל עַמָא לְכַהֲנַיָא וּלְסַפְרַיָא לֵית לְגַבְרָא הָדֵין חוֹבַת דִין דִקְטוֹל אֲרֵי בִשְׁמָא דַייָ אֱלָהָנָא מַלֵיל עִמָנָא:
וְקָמוּ גַבְרַיָא מִסָבֵי אַרְעָא וַאֲמָרוּ לְכָל קְהָלָא דְעַמָא לְמֵימָר:
מִיכָה דְמִמָרֵישָׁא הֲוָה מִתְנַבֵּי בְּיוֹמֵי חִזְקִיָה מְלַךְ שִׁבְטָא דְבֵית יְהוּדָה וַאֲמַר לְכָל עַמָא דְבֵית יְהוּדָה לְמֵימָר כִּדְנַן אֲמַר יְיָ צְבָאוֹת צִיוֹן חֲקַל תִּתְרְדֵי וִירוּשְׁלֵם לְיַגְרִין תְּהֵי וְטוּר בֵּית מַקְדְשָׁא דַייָ לְחִישַׁת חוּרְשָׁא:
הֲמִקְטַל קַטְלוֹהִי חִזְקִיָה מְלַךְ שִׁבְטָא דְבֵית יְהוּדָה הֲלֹא דָחִיל מִן קֳדָם יְיָ וְצַלֵי קֳדָם יְיָ וְתַב יְיָ מִן בִּישְׁתָא דְמַלֵיל עֲלֵיהוֹן וַאֲנַחְנָא גַרְמִין לְאַיְתָאָה בִישְׁתָא רַבְּתָא עַל נַפְשָׁתָא:
וְאַף גַבְרָא הֲוָה מִתְנַבֵּי בִּשְׁמָא דַייָ אוּרִיָה בַּר שְׁמַעְיָה מִקִרְיַת יְעָרִים וְאִתְנַבֵּי עַל קַרְתָּא הָדָא וְעַל אַרְעָא הָדָא כְּכֹל פִּתְגָמֵי יִרְמְיָה:
וּשְׁמַע מַלְכָּא יְהוֹיָקִים וְכָל גִבָּרוֹהִי וְכָל רַבְרְבֵיָא יַת פִּתְגָמוֹהִי וּבְעָא מַלְכָּא לְמִקְטְלֵיהּ וּשְׁמַע אוּרִיָה וּדְחִיל וַעֲרַק וַאֲתָא לְמִצְרָיִם:
וּשְׁלַח מַלְכָּא יְהוֹיָקִים גַבְרִין לְמִצְרָיִם יַת אֶלְנָתָן בַּר עַכְבּוֹר וְגַבְרִין עִמֵיהּ לְמִצְרָיִם:
וְאַפִיקוּ יַת אוּרִיָה מִמִצְרַיִם וְאַיְתִיוּהִי לְוַת מַלְכָּא יְהוֹיָקִים וְקַטְלֵיהּ בְּחַרְבָּא וּרְמָא יַת נְבֵלְתֵּיהּ לְקִבְרֵי גָלַיָא:
בְּרַם יַד אֲחִיקָם בַּר שָׁפָן הֲוָת עִם יִרְמְיָה בְּדִיל דְלָא לְמִמְסַר יָתֵיהּ בִּידָא דְעַמָא לְמִקְטְלֵיהּ:
