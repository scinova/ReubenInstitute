וַהֲוָה פִּתְגַם נְבוּאָה מִן קֳדָם יְיָ עִמִי לְמֵימָר:
בַּר אָדָם אֱמַר לְמַלְכָּא דְצוֹר כִּדְנַן אֲמַר לִיבָּךְ וַאֲמַרְתְּ דַחֲלָא אֲנָא יְיָ אֱלֹהִים חֲלַף דְרָאֵים מוֹתַב תַּקִיפִין יְתִיבֵית בְּגוֹ יַמַיָא וְאַתְּ אֱנָשָׁא וְלֵית בָּךְ צְרוֹךְ וַחֲשַׁבְתָּא בְלִיבָּךְ דְאַתְּ דַחֲלָא:
הָא חַכִּים אַתְּ מִדָנִיאֵל וְכָל רָז לָא יִתְכַּסָא מִנָךְ:
בְּחוּכְמְתָךְ וּבְסִכְלְתָנוּתָךְ קְנִיתָא לָךְ נִכְסִין וּכְנִשְׁתָּא דַהֲבָא וְכַסְפָּא בְּאוֹצְרָךְ:
בִּסְגִי חוֹכְמְתָךְ וּבִסְחוֹרָתָךְ סַגִיאָה מַשִׁרְיָתָךְ וְרָאֵים לִבָּךְ בְמַשִׁרְיָתָךְ:
בְּכֵן כִּדְנַן אֲמַר יְיָ אֱלֹהִים חֲלַף דַחֲשַׁבְתָּא בְּלִבָּךְ דְאַתְּ דַחֲלָא:
בְּכֵן הָא אֲנָא מַיְתֵי עֲלָךְ נוּכְרָאִין תַּקִיפֵי עַמְמַיָא וְיִשְׁלְפוּן סַיְפֵיהוֹן עַל תִּקוּן חוּכְמְתָךְ וִיחַבְּלוּן זִיו אֵימָתָךְ:
לְחַבָּלָא יַחְתוּנָךְ וּתְמוּת מְמַת מְטַעֲנֵי חַרְבָּא בְּגוֹ יַמַיָא:
הֲמֵימַר תֵּימַר דַחֲלָא אֲנָא קֳדָם דְקַטְלָךְ וְאַתְּ אֱנָשָׁא וְלֵית בָּךְ צְרוֹךְ בְּיַד מְחַבְּלָךְ:
מְמַת רַשִׁיעִין תְּמוּת בְּיַד נוּכְרָאִין אֲרֵי אֲנָא גְזַרֵית בְּמֵימְרִי אֲמַר יְיָ אֱלֹהִים:
וַהֲוָה פִּתְגַם נְבוּאָה מִן קֳדָם יְיָ עִמִי לְמֵימָר:
בַּר אָדָם טוֹל אִילְיָא עַל מַלְכָּא דְצוֹר וְתֵימַר לֵיהּ כִּדְנַן אֲמַר יְיָ אֱלֹהִים אַתְּ דָמֵי לְמָנָא דְצוּרְתָא דִי מְתַקֵן בְּחוּכְמְתָא וּמְשֻׁכְלַל בְּשׁוּפְרֵיהּ:
בִּסְגִי טָבְוָן וְתַפְנוּקִין אַתְּ מִתְפְּנַק כְּאִילוּ בְּעֵדֶן גִנְתָא דַייָ אַתְּ שָׁרַי כָּל עֻתְרָא גֵיוְתָא וִיקָרָא מִתְיַהַב לָךְ כָּל אַבְנִין טָבִין מְסַרְדָן עַל לְבוּשָׁךְ סַמְקָן יַרְקָן וְסִבְהֲלוֹם כְּרוּם יַמָא וּבוּרְלָא וּפַנְתֵּירִין שִׁבְזֵיז אִזְמַררְגְדִין וּבַרְקָן מְשַׁקְעִן בִּדְהַב כָּל אִלֵין עֲבִידַת תִּקוּנְךְ בְּכֵן רָאֵים לִבָּךְ בְּרַם לָא אִסְתַּכַּלְתְּ בְּפִגְרָךְ דְאִתְעֲבִיד חֲלָלִין וּנְקָבִין דְאִינוּן צוֹרְכָן דְלָא אֶפְשַׁר לָךְ דְתִתְקַיֵים בְּלָא בְּהוֹן מִיוֹמָא דְאִתְבְּרֵיתָא עִמָךְ מְתַקְנִין:
אַתְּ מְלַךְ מְרַבָּא לְמָלְכוּ וִיהֵבִית לָךְ רְבוּתָא וְאַתְּ עַל טוּר קוּדְשָׁא דַייָ בְּסַרְתָּא וּבְעַמָא קַדִישָׁא סְבַרְתָּא לְמִשְׁלַט:
שְׁלִים אַתְּ בְּאוֹרְחָךְ מִיוֹמָא דְאִתְבְּרֵיתָא עַד דְאִשְׁתַּכַּח שְׁקַר בִּיךְ:
בִּסְגִיאוּת סְחוֹרְתָּךְ אִתְמְלִיאוּ אוֹצְרָךְ חֲטוֹף וְחַבְתָּא וַאֲפִסִינָךְ עַל דִבְסַרְתָּ עַל טוּר קוּדְשָׁא דַייָ וְאַבְּדִינָךְ מַלְכָּא דְרַבְּתָנָא עַל דִי סְבַרְתָּא לְמִשְׁלַט בְּעַמָא קַדִישָׁא:
רָאֵים לִבָּךְ בְּתוּקְפָךְ אִתְחַבַּלְתָּ בְּחוּכְמְתָךְ עַל זִיו אֵימְתָךְ עַל אַרְעָא רְטַשְׁתָּךְ אַזְהָרָא לְמַלְכַיָא יְהַבְתִּיךְ לְאִסְתַכְּלָא בָךְ:
מִסְגֵי חוֹבָךְ בִּשְׁקַר סְחוֹרְתָּךְ אָפֵיסְתָּא מַקְדְשָׁךְ וְאַיְתֵי עַמְמַיָא דְתַקִיפִין כְּאֶשְׁתָּא מֵחוֹבֵי זְדוֹנָךְ אִינוּן יְשֵׁיצוּנָךְ וְאֶתְּנִינָךְ כִּקְטַם עַל אַרְעָא לְעֵינֵי כָּל חָזָךְ:
כָּל דְיָדְעָךְ בְּעַמְמַיָא צְדִיאוּ עֲלָךְ כִּדְלָא הֲוֵית אֶתְנִינָךְ וְכֵן תְּהֵי עַד עָלְמָא:
וַהֲוָה פִתְגָם נְבוּאָה מִן קֳדָם יְיָ עִמִי לְמֵימָר:
בַּר אָדָם קַבֵּיל נְבוּאָה עַל צִידוֹן וְאִתְנַבֵּי עֲלָה:
וְתֵימַר כִּדְנָן אֲמַר יְיָ אֱלֹהִים הָא אֲנָא שְׁלַח רוּגְזִי עֲלָךְ צִידוֹן וְאִתְיַקַר בְּנַיִיךְ וְיֵדְעוּן אֲרֵי אֲנָא יְיָ כַּד אַעְבֵּיד בִּיךְ פּוּרְעֲנוּת דִינִין וְאֶתְקַדֵשׁ בָּהּ:
וַאֲשַׁלַח בָּהּ מוֹתָא וּקְטָלָא בְשׁוּקָהָא כַּד יִתְרְמוּן קְטִילִין בְּגַוָהּ בְּאַיְתָיוּתִי עֲלָהּ דְקָטְלִין בְּחַרְבָּא מִסְחוֹר סְחוֹר וְיֵדְעוּן אֲרֵי אֲנָא יְיָ:
וְלָא יְהֵי עוֹד לְבֵית יִשְׂרָאֵל מְלַךְ מַבְאִישׁ וְשִׁלְטוֹן מָעֵיק מִכָּל סָחֳרָנֵיהוֹן דְבָזְזִין יַתְהוֹן וְיֵדְעוּן אֲרֵי אֲנָא יְיָ אֱלֹהִים:
כִּדְנַן אֲמַר יְיָ אֱלֹהִים כַּד אַכְנַשׁ יַת בֵּית יִשְׂרָאֵל מִבֵּינֵי עַמְמַיָא דְאִתְבַּדַרוּ לְתַמָן וְאִתְקַדֵשׁ בְּהוֹן לְעֵינֵי עַמְמַיָא וִיתוּבוּן עַל אַרְעֲהוֹן דִיהֲבִית לְעַבְדִי לְיַעֲקֹב:
וִיתוּבוּן עֲלָהּ לְרוֹחֲצָן וְיִבְנוּן בָּתִּין וְיִצְבוּן כַּרְמִין וְיֵתְבוּן לְרוֹחֲצָן כַּד אַעְבֵיד פּוּרְעֲנוּת דִינִין מִכֹּל דְבָּזְזִין יַתְהוֹן מִסָחְרָנֵיהוֹן וְיֵדְעוּן אֲרֵי אֲנָא יְיָ אֱלָהָהוֹן:
