וַי דְנַחֲתִין לְמִצְרַיִם לְסָעִיד וְעַל סוּסְוָתָא מִסְתַּמְכִין וּמִתְרַחֲצִין עַל רְתִיכַיָא אֲרֵי סַגִיאִין וְעַל פָּרָשַׁיָא אֲרֵי תַקִיפִין לַחֲדָא וְלָא אִסְתְּמִיכוּ עַל מֵימַר קַדִישָׁא דְיִשְׂרָאֵל וְאוּלְפַן מִן קֳדָם יְיָ לָא תְבָעוּ:
וְאַף הוּא בְּחוּכְמְתֵיהּ אַיְתֵי בִישְׁתָּא עַל דְעַבְדֵהּ וְיַת פִּתְגָמוֹהִי לָא בַטֵיל וַאֲקֵים עַל בֵית מַבְאֲשִׁין וְעַל דְסָעֲדִין לֵיאוּת שְׁקָר:
וּמִצְרַיִם אֱנַשׁ וְלָא רָב וְסוּסְוָתְהוֹן בְּסַר וְלָא רוּחַ וַיָי יָרִים מְחַת גְבוּרְתֵּהּ וְיִתְקַל סָעִיד וְיִפּוֹל סָעִיד וְכַחֲדָא כּוּלְהוֹן יִשְׁתֵּיצוּן:
אֲרֵי כִדְנַן אֲמַר יְיָ לִי כְּמָא דְמַכְלֵי אַרְיָא בַּר אַרְיָוָן עַל צֵידֵיהּ דְמִזְדַמְנִין עֲלוֹהִי דִיר רַעֲיָן מִקָלְהוֹן לָא אִתְבַּר וּמֵאִתְרְגוּשְׁתְּהוֹן לָא מִזְדַעְזַע כֵּן תִּתְגְלֵי מַלְכוּתָא דַייָ צְבָאוֹת לְמִשְׁרֵי עַל טוּרָא דְצִיוֹן וְעַל רָמָתָהּ:
כְּעוֹפָא דְטָאִים כֵּן תִּתְגְלֵי גְבוּרְתֵּהּ דַייָ צְבָאוֹת עַל יְרוּשְׁלֵם יָגֵין וִישֵׁיזֵיב יַצִיל וְיַעְדֵי:
תּוּבוּ לְאוֹרַיְתָא אֲרֵי אַסְנֵיתוּן לְמֶחְטֵי בְּנֵי יִשְׂרָאֵל:
אֲרֵי בְעִדָנָא הַהִיא יְרַחֲקוּן גְבַר טַעֲוַת כַּסְפְּהוֹן וְטַעֲוַת דְהַבְהוֹן דַעֲבָדוּ לְכוֹן יְדֵיכוֹן חוֹבָא:
וְיִפּוֹל אַתּוּרָאָה בְּחֶרֶב לָא אֱנַשׁ וְחֶרֶב לָא אֱנשׁ תְּשֵׁיצִינֵהּ וְיֶעְרוֹק לֵיהּ כְּמִן קֳדָם דְקַטְלִין בְּחַרְבָּא וְגִבְרוֹהִי לִתְּבַר יְהוֹן:
וְשִׁלְטוֹנוֹהִי מִן קֳדָם דַחֲלָא יֶעְדוּן וִיתַּבְרוּן מִן קֳדָם נִיסָא רַבְרְבוֹהִי אֲמַר יְיָ דְזֵיהוֹר לֵיהּ בְּצִיוֹן לְעָבְדֵי אוֹרַיְתָא וְתַנוּר בָּעֵיר לֵיהּ דְאֶשָׁא בִירוּשְׁלֵם לִדְעֲבָרוּ עַל מֵימְרֵהּ:
