וְתֵימַר בְּעִדָנָא הַהִיא אוֹדִי קֳדָמָךְ יְיָ אֲרֵי עַל דְחָבֵית קֳדָמָךְ הֲוָה רוּגְזָךְ עֲלָי וּכְדוּ דְתָבֵית לְאוֹרַיְתָא יְתוּב רוּגְזָךְ מִנִי וּתְרַחֵם עֲלָי:
הָא עַל מֵימָר אֱלָהָא פּוּרְקָנִי אֲנָא רָחִיץ וְלָא אִזְדַעְזָא אֲרֵי תוּקְפִי וְתוּשְׁבַּחְתִּי דְחִילָא יְיָ אֲמַר בְּמֵימְרֵיהּ וְהַוָה לִי לְפָרִיק:
וּתְקַבְּלוּן אוּלְפַן חֲדַת בְּחֶדְוָא מִבְּחִירֵי צַדִיקַיָא:
וְתֵימְרוּן בְּעִדָנָא הַהִיא אוֹדוּ קֳדָם יְיָ צַלוּ בִּשְׁמֵיהּ חֲווּ בְּעַמְמַיָא עוֹבָדוֹהִי אִדְכַּרוּ אֲרֵי תַקִיף שְׁמֵיהּ:
שַׁבְּחוּ קֳדָם יְיָ אֲרֵי גְבוּרָן עֲבַד גַלְיָא דָא בְּכָל אַרְעָא:
בּוּעִי וְשַׁבְּחִי כְּנִשְׁתָּא דְצִיוֹן אֲרֵי רַבָּא אֲמַר לְאַשְׁרָאָה שְׁכִינְתֵּיהּ בְּגַוָךְ קַדִישָׁא דְיִשְׂרָאֵל:
