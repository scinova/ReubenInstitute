וַאֲמַר אֱלִישָׁע קַבִּילוּ פִתְגָמָא דַייָ כִּדְנַן אֲמַר יְיָ כְּעִדְנָא הָדֵין מְחַר סְאָה סוּלְתָּא בְסִלְעָא וְסָאתָן סְעוֹרִין בְּסִלְעָא בִּתְרַע דְשֹׁמְרוֹן:
וַאֲתֵיב גִבָּרָא דְמַלְכָּא דְסָמִיךְ עַל יְדוֹהִי יַת נְבִיָא דַייָ וַאֲמַר אִילוּ יְיָ פְּתַח כַּוִין וּמָחִית טוּבָא מִן שְׁמַיָא הַיְהֵי פִּתְגָמָא הָדֵין וַאֲמַר הָא אַתְּ חָזֵי בְּעֵינָךְ וּמִתַּמָן לָא תֵיכוֹל:
וְאַרְבָּעָה גַבְרִין הֲווֹ סְגִירִין יַתְבִין בְּמַעֲלָנָא דְתַרְעָא וַאֲמַרוּ גְבַר לְחַבְרֵיהּ מָא אֲנַחְנָא יַתְבִין כָּא עַד דִנְמוּת:
אִם נֵימַר נֵיעוֹל לְקַרְתָּא וְכַפְנָא בְקַרְתָּא וּנְמוּת תַּמָן וְאִם נִתֵיב הָכָא וּנְמוּת וּכְעַן אֱתוֹ וְנִשְׁתְּמַע לְמַשִׁרְיַת אֲרָם אִם יְחַיוּנָנָא וְנֵחִי אִם יִקְטְלוּנָנָא וְנִתְקַטֵיל:
וְקָמוּ בְקִבְלָא לְמֵיעַל לְמַשִׁרְיַת אֲרַם וַאֲתוֹ עַד סְיָפֵי מַשִׁרְיַת אֲרָם וְהָא לֵית תַּמָן אֱנָשׁ:
וּמִן קֳדָם יְיָ אִשְׁתְּמַע לְמַשִׁרְיַת אֲרָם קַל רְתִיכִין קַל סוּסְוָן קַל מַשִׁרְיַן סַגִיאִן וַאֲמַרוּ גְבַר לְחַבְרֵיהּ הָא אֲגַר עֲלָנָא מַלְכָּא דְיִשְׂרָאֵל יַת מַלְכֵי חִתָּאֵי וְיַת מַלְכֵי מִצְרָאֵי לְמֵיתֵי עֲלָנָא:
וְקָמוּ וַעֲרָקוּ בְקִבְלָא וּשְׁבָקוּ יַת מַשְׁכְּנֵיהוֹן וְיַת סוּסְוָתְהוֹן וְיַת חֲמָרֵיהוֹן בְּמַשְׁרִיתָא כְּמָא דְהִיא וַעֲרָקוּ לְשֵׁיזָבָא נַפְשֵׁיהוֹן:
וַאֲתוֹ סְגִירַיָא הָאִלֵין עַד סְיָפֵי מַשְׁרִיתָא וַעֲלוּ לְמַשְׁכְּנָא חַד וַאֲכַלוּ וּשְׁתִיאוּ וּנְסִיבוּ מִתַּמָן כַּסְפָּא וְדַהֲבָא וּלְבוּשִׁין וַאֲזָלוּ וְאַטְמָרוּ וְתָבוּ וַעֲלוּ לְמַשְׁכְּנָא אוֹחֲרָנָא וּנְסִיבוּ מִתַּמָן וַאֲזַלוּ וְאַטְמָרוּ:
וַאֲמַרוּ גְבַר לְחַבְרֵיהּ לָא כָשַׁר מַה דַאֲנַחְנָא עַבְדִין יוֹמָא הָדֵין יוֹם בְּסוֹרָא הוּא וְאִלוּ אֲנַחְנָא שַׁתְקִין וּמוֹרְבִין עַד נֵיהוֹר צַפְרָא וִיעַרְעִינָנָא חוֹבָא וּכְעַן אִיתוּ וְנִתְמְטֵי וּנְחַוֵי בֵּית מַלְכָּא:
וַאֲתוֹ וּקְרוֹ לְנַטְרֵי תְרַע קַרְתָּא וְחַוִיאוּ לְהוֹן לְמֵימַר אָתִינָא לְמַשִׁרְיַת אֲרָם וְהָא לֵית תַּמָן גְבַר וְקַל אֱנַשׁ אֱלָהֵין סוּסְוָן אֲסִירִין וַחֲמָרִין אֲסִירִין וּמַשְׁכְּנִין כְּמָא דְאִינוּן:
וּקְרָא נַטְרֵי תַרְעָא וְחַוִיאוּ בֵּית מַלְכָּא לְגָיו:
וְקָם מַלְכָּא בְּלֵילְיָא וַאֲמַר לְעַבְדוֹהִי אֲחַוֵי כְעַן לְכוֹן יַת דַעֲבָדוּ לָנָא אֱנַשׁ אֲרָם יָדְעִין אֲרֵי כְפָנִין אֲנַחְנָא וּנְפַקוּ מִן מַשְׁרִיתָא לְאִטַמָרָא בְחַקְלָא לְמֵימַר אִם יַפְקוּן מִן קַרְתָּא וְנִיחוּדוּנוּן כַּד חַיִין וּלְקַרְתָּא נֵיעוֹל:
וַאֲתֵיב חַד מֵעַבְדוֹהִי וַאֲמַר וְיִדְבְּרוּן כְּעַן חַמְשָׁא מִן סוּסְוָתָא מִן שַׁאֲרָא דְאִשְׁתַּאֲרוּ בָהּ הָא אִינוּן כְּכָל הֲמוֹנָא דְיִשְׂרָאֵל דְאִשְׁתַּאֲרוּ בָהּ אִם יֵבְדוּן הָא אִינוּן כְּכָל הֲמוֹנָא דְיִשְׂרָאֵל דְסָפוּ וְנִשְׁלַח וְנֶחְזֵי:
וְדַבָּרוּ תְרֵין רִכְבֵי סוּסִין וּשְׁלַח מַלְכָּא בָּתַר מַשִׁרְיַת אֲרָם לְמֵימַר אֱזִילוּ וַחֲזוֹ:
וַאֲזַלוּ בַתְרֵיהוֹן עַד יַרְדְנָא וְהָא כָל אוֹרְחָא מַלְיָא לְבוּשִׁין וּמָנִין דִרְמוֹ אֱנַשׁ אֲרָם בְּאִתְבָּעוּתְהוֹן לְמֶעְרַק וְתָבוּ אִזְגַדַיָא וְחַוִיאוּ לְמַלְכָּא:
וּנְפַק עַמָא וּבְזוֹ יַת מַשִׁרְיַת אֲרָם וַהֲוָה סְאָה סוּלְתָּא בְּסִלְעָא וְסָאתָן סְעוֹרִין בְּסִלְעָא כְּפִתְגָמָא דַיָי:
וּמַלְכָּא מַנֵי יַת גִבָּרָא דִסְמִיךְ עַל יְדוֹהִי עַל תַּרְעָא וְדָשׁוּהִי עַמָא בְּתַרְעָא וּמִית כְּמָא דְמַלֵיל נְבִיָא דַייָ דְמַלֵיל כַּד נְחַת מַלְכָּא לְוָתֵיהּ:
וַהֲוָה כַּד מַלֵיל נְבִיָא דַייָ עִם מַלְכָּא לְמֵימָר סָאתָן סְעוֹרִין בְּסִלְעָא וּסְאָה סוּלְתָּא בְּסִלְעָא יִהֵי כְּעִדָנָא הָדֵין מְחַר בְתַרְעָא דְשֹׁמְרוֹן:
וַאֲתֵיב גִבָּרָא יַת נְבִיָא דַייָ וַאֲמַר וְאִלוּ יְיָ פָּתַח כַּוִין וּמָחִית טוּבָא מִן שְׁמַיָא הַיְהֵי כְּפִתְגָמָא הָדֵין וַאֲמַר הָא אַתְּ חָזֵי בְּעֵינָךְ וּמִתַּמָן לָא תֵיכוּל:
וַהֲוָה לֵיהּ כֵּן וְדָשׁוּ יָתֵיהּ עַמָא בְּתַרְעָא וּמִית:
