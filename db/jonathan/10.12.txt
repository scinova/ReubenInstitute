וַאֲזַל רְחַבְעָם לִשְׁכֶם אֲרֵי לִשְׁכֶם אֲתוֹ כָל יִשְׂרָאֵל לְאַמְלָכָא יָתֵיהּ:
וַהֲוָה כַּד שְׁמַע יָרָבְעָם בַּר נְבָט וְהוּא עַד כְּעַן בְּמִצְרַיִם דַעֲרַק מִן קֳדָם מַלְכָּא שְׁלֹמֹה וִיתֵב יָרָבְעָם בְּמִצְרָיִם:
וּשְׁלָחוּ וּקְרוֹ לֵיהּ וַאֲתָא יָרָבְעָם וְכָל קְהַל יִשְׂרָאֵל וּמַלִילוּ עִם רְחַבְעָם לְמֵימָר:
אֲבוּךְ אַקְשֵׁי יַת נִירָנָא וְאַתְּ כְּעַן אֲקֵיל יַת פּוּלְחָנָא דַאֲבוּךְ קַשְׁיָא וּמִנִירֵיהּ תַּקִיפָא דִיהַב עֲלָנָא וְנַפְלְחִינָךְ:
וַאֲמַר לְהוֹן אֱזִילוּ עוֹד תְּלָתָא יוֹמִין וְתוּבוּ לְוָתִי וַאֲזַלוּ עַמָא:
וְאִתְמְלִיךְ מַלְכָּא רְחַבְעָם בְּסָבַיָא דַהֲווֹ מְשַׁמְשִׁין קֳדָם שְׁלֹמֹה אֲבוּהִי כַּד הֲוָה קַיָם לְמֵימָר אֵיכְדֵין אַתּוּן מִתְמַלְכִין לַאֲתָבָא יַת עַמָא הָדֵין פִּתְגָמָא:
וּמַלִילוּ עִמֵיהּ לְמֵימַר אִם יוֹמָא דֵין תִּתְעַלֵיב לְעַמָא הָדֵין וּתִפְלְחִינוּן וְתַעֲנִינוּן מִלִין טָבָן וּתְמַלֵיל עִמְהוֹן פִּתְגָמִין תַּקְנִין וִיהוֹן לָךְ עַבְדִין כָּל יוֹמַיָא:
וּשְׁבַק יַת מַלְכָּא דְסָבַיָא דִי מַלְכוּהִי וְאִתְמְלִיךְ בְּיַנְקַיָא דִרְבִיאוּ עִמֵיהּ דִמְשַׁמְשִׁין קֳדָמוֹהִי:
וַאֲמַר לְהוֹן מָה אַתּוּן מִתְמַלְכִין וּנְתֵיב פִּתְגָמָא יַת עַמָא הָדֵין דְמַלִילוּ עִמִי לְמֵימָר אֲקֵיל מִן נִירָא דִיהַב אֲבוּךְ עֲלָנָא:
וּמַלִילוּ עִמֵיהּ יַנְקַיָא דִרְבִיאוּ עִמֵיהּ לְמֵימָר כְּדֵין תֵּימַר לְעַמָא הָדֵין דְמַלִילוּ עִמָךְ לְמֵימַר אֲבוּךְ אַתְקֵיף יַת נִירָנָא וְאַתְּ אֲקֵיל מִנָנָא כְּדֵין תֵּימַר לְהוֹן חֲלָשׁוּתִי תַקִיפָא מֵעוּלֵימוּתֵיהּ דְאַבָּא:
וּכְעַן אַבָּא רְמָא עֲלֵיכוֹן נִיר תַּקִיף וַאֲנָא אוֹסִיף עַל נִירֵכוֹן אַבָּא רְדָא יַתְכוֹן בְּשַׁרְבִיטִין וַאֲנָא אֶרְדֵי יַתְכוֹן בְּמַרְגְנִין:
וַאֲתָא יְרָבְעָם וְכָל עַמָא לְוַת רְחַבְעָם בְּיוֹמָא תְלִיתָאָה כְּמָא דְמַלֵיל מַלְכָּא לְמֵימַר תּוּבוּ לְוָתִי בְּיוֹמָא תְלִיתָאָה:
וְאָתִיב מַלְכָּא יַת עַמָא פִּתְגַם קָשֵׁי וּשְׁבַק יַת מִלְכָּא דְסָבַיָא דִי מְלַכוּהִי:
וּמַלֵיל עִמְהוֹן כְּמִלְכָּא דְיַנְקַיָא לְמֵימָר אַבָּא אַתְקֵיף יַת נִירְכוֹן וַאֲנָא אוֹסִיף עַל נִירְכוֹן אַבָּא רְדָא יַתְכוֹן בְּשַׁרְבִיטִין וַאֲנָא אֶרְדֵי יַתְכוֹן בְּמַרְגְנִין:
וְלָא קַבֵּיל מַלְכָּא מִן עַמָא אֲרֵי הֲוַת פְּלוּגְתָּא מִן קֳדָם יְיָ בְּדִיל לַאֲקָמָא יַת פִּתְגָמָא דְמַלֵיל יְיָ בְּיַד אֲחִיָה שִׁילוֹנָאָה עַל יָרָבְעָם בַּר נְבָט:
וַחֲזוֹ כָל יִשְׂרָאֵל אֲרֵי לָא קַבֵּיל מַלְכָּא מִנְהוֹן וַאֲתִיבוּ עַמָא יַת מַלְכָּא פִּתְגָמָא לְמֵימַר לֵית לָנָא חוּלַק בְּדָוִד וְלָא אַחֲסָנָא בְּבַר יִשַׁי גְבַר לְקִרְוָךְ יִשְׂרָאֵל כְּעַן מְלוֹךְ עַל אֱנַשׁ בֵּיתָךְ דָוִד וַאֲזַל יִשְׂרָאֵל לְקִרְוֹהִי:
וּבְנֵי יִשְׂרָאֵל דְיַתְבִין בְּקִרְוַיָא דְבֵית יְהוּדָה וּמְלַךְ עֲלֵיהוֹן רְחַבְעָם:
וּשְׁלַח מַלְכָּא רְחַבְעָם יַת אֲדוֹרָם דִמְמַנָא עַל מַסְקֵי מִסִין וּרְגָמוּ כָל יִשְׂרָאֵל יָתֵיהּ בְּאַבָנָא וּמִית וּמַלְכָּא רְחַבְעָם אִתְקַף לְמִסַק בִּרְתִיכָא לְמִפַּךְ יְרוּשְׁלָם:
וּמְרַדוּ בֵּית יִשְׂרָאֵל בְּבֵית דָוִד עַד יוֹמָא הָדֵין:
וַהֲוָה כַּד שְׁמָעוּ כָל יִשְׂרָאֵל אֲרֵי תַב יְרָבְעָם וּשְׁלַחוּ וּקְרוֹ יָתֵיהּ לְוַת כְּנִשְׁתָּא וְאַמְלִיכוּ יָתֵיהּ עַל כָּל יִשְׂרָאֵל לָא הֲוָה בָּתַר בֵּית דָוִד אֱלָהֵין שִׁבְטָא דִיהוּדָה בִּלְחוֹדוֹהִי:
וַאֲתָא רְחַבְעָם לִירוּשְׁלֵם וּכְנַשׁ יַת כָּל בֵּית יְהוּדָה וְיַת שִׁבְטָא דְבֵית בִּנְיָמִין מְאָה וְתַמְנָן אַלְפִין גְבַר גִבַּר עָבְדֵי קְרָבָא לְאַגָחָא קְרָבָא עִם בֵּית יִשְׂרָאֵל לַאֲתָבָא יַת מַלְכוּתָא לִרְחַבְעָם בַּר שְׁלֹמֹה:
וַהֲוָה פִתְגַם נְבוּאָה מִן קֳדָם יְיָ עִם שְׁמַעְיָה נְבִיָא דַייָ לְמֵימָר:
אֲמַר לִרְחַבְעָם בַּר שְׁלֹמֹה מַלְכָּא דְבֵית יְהוּדָה וּלְכָל שִׁבְטָא דְּבֵית יְהוּדָה וְשִׁבְטָא בִּנְיָמִן וּשְׁאָר עַמָא לְמֵימָר:
כִּדְנַן אֲמַר יְיָ לָא תִסְקוּן וְלָא תְגִיחוּן קְרָבָא עִם אֲחֵיכוֹן בְּנֵי יִשְׂרָאֵל תּוּבוּ גְבַר לְבֵיתֵיהּ אֲרֵי מִן קֳדָם מֵימְרִי הֲוָה פִתְגָמָא הָדֵין וְקַבִּילוּ לְפִתְגָמָא דַייָ וְתָבוּ לְמֵיזַל כְּפִתְגָמָא דַיְיָ:
וּבְנָא יְרָבְעָם יַת שְׁכֶם בְּטוּרָא דְבֵית אֶפְרַיִם וִיתֵב בָּהּ וּנְפַק מִתַּמָן וּבְנָא יַת פְּנוּאֵל:
וַאֲמַר יָרָבְעָם בְּלִבֵּיהּ כְּעַן דִלְמָא תְּתוּב מַלְכוּתָא לְבֵית דָוִד:
אִם יִסַק עַמָא הָדֵין לְמֶעְבַּד נִכְסַת קוּדְשִׁין בְבֵית מַקְדְשָׁא דַייָ בִּירוּשְׁלֵם וְיִתְפְּנֵי לִבָּא דְעַמָא הָדֵין בָּתַר רִבּוֹנְהוֹן בָּתַר רְחַבְעָם מֶלֶךְ שִׁבְטָא דְבֵית יְהוּדָא וְיִקְטְלֻנַנִי וִיתוּבוּן לְוַת רְחַבְעָם מֶלֶךְ שִׁבְטָא דְבֵית יְהוּדָה:
וְאִתְמְלִיךְ מַלְכָּא וַעֲבַד תְּרֵי עֶגְלִין דְדַהֲבָא וַאֲמַר לְהוֹן סַגִיאָה לְכוֹן אָרְחָא מִלְמֵסַק לִירוּשְׁלֵם הָא אִינוּן דַחֲלָתָךְ יִשְׂרָאֵל דְאַסְקוּךְ מֵאַרְעָא דְמִצְרָיִם:
וְשַׁוִי יַת חַד בְּבֵית אֵל וְיַת חַד יְהַב בְּדָן:
וַהֲוָה פִתְגָמָא הָדֵין לְחוֹבָא וַאֲזַלוּ עַמָא קֳדָם חַד עַד דָן:
וַעֲבַד יַת בֵּית בָּמָתָא וַעֲבַד כּוּמָרַיָא קְצַת מִן עַמָא דְלָא הֲווֹ מִבְּנֵי לֵוִי:
וַעֲבַד יָרָבְעָם חַגָא בְּיַרְחָא תְמִינָאָה בְּחַמְשָׁא עַסְרָא יוֹמָא לְיַרְחָא כְּחַגָא דְשִׁבְטָא דִיהוּדָה וְאַסֵיק עַל אֱגוֹרָא כֵּן עֲבַד בְּבֵית אֵל לְדַבְּחָא לְעֶגְלַיָא דִי עֲבַד וַאֲקֵים בְּבֵית אֵל יַת כּוּמָרֵי בְמָתָא דִי עֲבַד:
וְאַסֵיק עַל אֱגוֹרָא דַעֲבַד בְּבֵית אֵל בְּחַמְשָׁא עַסְרָא יוֹמָא לְיַרְחָא תְמִינָאָה בְּיַרְחָא דַעֲדָא מֵרְעוּתֵיהּ וַעֲבַד חַגָא לִבְנֵי יִשְׂרָאֵל וּסְלִיק עַל אֱגוֹרָא לְאַקְטָרָא:
