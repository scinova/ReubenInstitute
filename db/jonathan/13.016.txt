וַהֲוָה פִתְגָם נְבוּאָה מִן קֳדָם יְיָ עִמִי לְמֵימָר:
לָא תִסַב לָךְ אִתְּתָא וְלָא יְהוֹן לָךְ בְּנִין וּבְנָן בְּאַתְרָא הָדֵין:
אֲרֵי כִדְנַן אָמַר יְיָ עַל בְּנִין וְעַל בְּנָן דְמִתְיַלְדִין בְּאַתְרָא הָדֵין וְעַל אִמָתְהוֹן דְיָלְדִין יַתְהוֹן וְעַל אֲבָהַתְהוֹן דְמוֹלִידִין יַתְהוֹן בְּאַרְעָא הָדָא:
מְמָת מְמַרְעֵי כְפַן יְמוּתוּן לָא יִסְתַּפֵּדוּן וְלָא יִתְקַבְּרוּן לִזְבַל מְבַדֵר עַל אַפֵּי אַרְעָא יְהוֹן וּבְחַרְבָּא וּבְכַּפְנָא יְסוּפוּן וּתְהֵי נְבַלְתְּהוֹן מְשַׁגְרָא לְמֵיכַל לְעוֹפָא דִשְׁמַיָא וְלִבְעִירָא דְאַרְעָא:
אֲרֵי כִדְנַן אָמַר יְיָ לָא תֵעוֹל לְבֵית מַרְזְחָא וְלָא תְהָךְ לְמִסְפַּד וְלָא תִדְנֵי לְהוֹן אֲרֵי כְנֵשֵׁית יַת שְׁלָמִי מִן עַמָא הָדֵין אֲמַר יְיָ יַת חִסְדָא וְיַת רַחֲמֵי:
וִימוּתוּן רַבְרְבִין וְדַעְדְקִין בְּאַרְעָא הָדָא לָא יִתְקַבְּרוּן וְלָא יִסְפְּדוּן לְהוֹן וְלָא יִתְחַמְמוּן וְלָא יְמִרְטוּן לְהוֹן:
וְלָא יִפְרְסוּן לְהוֹן עַל אֶבְלָא לְנֶחָמוּתְהוֹן עַל מִיתָא וְלָא יַשְׁקוּן יַתְהוֹן כָּס תַּנְחוּמִין גְבַר עַל אֲבוּהִי וְעַל אִמֵיהּ:
וּבֵית אַסְחָרוּתָא לָא תֵעוֹל לְאַסְחָרָא עִמְהוֹן לְמִיכַל וּלְמִשְׁתֵּי:
אֲרֵי כִּדְנַן אֲמַר יְיָ צְבָאוֹת אֱלָהָא דְיִשְׂרָאֵל הָא אֲנָא מְבַטֵל מִן אַתְרָא הָדֵין לְעֵינֵיכוֹן וּבְיוֹמֵיכוֹן קָל בְיָע וְקָל חֶדְוָא קָל חַתְנִין וְקָל כַּלָן:
וִיהֵי אֲרֵי תְחַוֵי לְעַמָא הָדֵין יַת כָּל פִּתְגָמַיָא הָאִלֵין וְיֵמְרוּן לָךְ עַל מַה גְזַר יְיָ עֲלָנָא יַת קַל בִּישְׁתָּא רַבְּתָא הָדָא וּמַה חוֹבָנָא וּמַה סוּרְחָנָנָא דְחַבְנָא קֳדָם יְיָ אֱלָהָנָא:
וְתֵימַר לְהוֹן עַל דִשְׁבַקוּ אֲבָהָתְכוֹן פּוּלְחָנִי אָמַר יְיָ וּטְעוֹ בָתַר טַעֲוָת עַמְמַיָא וּפְלַחוּנוּן וּסְגִידוּ לְהוֹן וּפוּלְחָנִי שְׁבַקוּ וְיַת אוֹרַיְתִי לָא נְטַרוּ:
וְאַתּוּן אַבְאַשְׁתּוּן לְמֶעְבַּד מֵאֲבָהַתְכוֹן וְהָא אַתּוּן אָזְלִין גְבַר בָּתַר הִרְהוּר לִבֵּיהּ בִּישָׁא בְּדִיל דְלָא לְקַבָּלָא לְמֵימְרִי:
וַאֲטַלְטֵל יַתְכוֹן מֵעַל אַרְעָא הָדָא לְאַרְעָא דְלָא יְדַעְתּוּן אַתּוּן וַאֲבָהָתְכוֹן וְתִפְלְחוּן תַּמָן לְעַמְמַיָא פָּלְחֵי טַעֲוָתָא יְמָם וְלֵילֵי דְלָא יְהֵי לְכוֹן רַחֲמִין:
בְּכֵן הָא יוֹמַיָא אָתָאן אָמַר יְיָ וְלָא יְהוֹן מִשְׁתְּעָן עוֹד בִּגְבוּרְתָּא דַייָ דְאַסֵיק יַת בְּנֵי יִשְׂרָאֵל מֵאַרְעָא דְמִצְרָיִם:
אֱלָהֵן יְהוֹן מִשְׁתְּעָן בִּגְבוּרְתָּא דַייָ דְאַסִיק יַת בְּנֵי יִשְׂרָאֵל מֵאַרְעָא צִפּוּנָא וּמִכָּל מְדִינָתָא דְאַגְלֵינוּן לְתַמָן וַאֲתֵיבִינוּן לְאַרְעֲהוֹן דִיהָבֵית לַאֲבָהַתְהוֹן:
הָא אֲנָא שְׁלַח לְעַמְמִין קָטוֹלִין אֲמַר יְיָ וִיקַטְלוּנוּן וּבָתַר כֵּן אֶשְׁלַח לְסַגִיאִין צְיָדִין וִיצָדוּנוּן מֵעַל כָּל כָּל טוּר וּמֵעַל כָּל רָמָתָא וּמַשְׁקִיפֵי כֵּיפַּיָא:
אֲרֵי קֳדָמַי גְלַן כָּל אוֹרְחַתְהוֹן לָא מְסַתְּרָן מִן קֳדָמַי וְלָא מְטַמְרָן חוֹבֵיהוֹן מִלְקָבֵיל מֵימְרִי:
וַאֲשַׁלֵם לְתִנְיָנִין כְּקַדְמְאִין עַל חַד תְּרֵין חוֹבֵיהוֹן וְחַטָאֵיהוֹן עַל דְאַחִילוּ יַת אֲרַע בֵּית שְׁכִנְתִּי בְּנִבְלַת שִׁקוּצֵיהוֹן וְתוֹעֲבַתְהוֹן מְלוֹ יַת אַחְסַנְתִּי:
יְיָ תּוּקְפִי וְרוּחְצָנִי וּמְשֵׁיזְבִי בְּעִדַן עָקָא לְשֵׁמַע גְבוּרְתָּךְ עַמְמִין יֵיתוּן מִסְיָפֵי אַרְעָא וְיֵמְרוּן בְּרַם לְשִׁקְרָא פְּלָחוּ אֲבָהָתָנָא לִלְמָא וְלֵית בְּהוֹן הֲנָאָה:
הַאֶפְשַׁר דְיַעְבֵּד לֵיהּ אֱנַשׁ דַחֲלַן וְאִינוּן טַעֲוָן דְלֵית בְּהוֹן צְרוֹךְ:
בְּכֵן הָא אֲנָא מְחַוֵי לְהוֹן בְּזִמְנָא הָדָא אֲחַוִינוּן יַת פּוּרְעֲנוּתִי וְיַת מְחַת גְבוּרְתִּי וְיֵדְעוּן אֲרֵי שְׁמִי יְיָ:
