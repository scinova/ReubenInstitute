וַהֲוָה פִּתְגַם נְבוּאָה מִן קֳדָם יְיָ עִמִי לְמֵימַר:
בַּר אָדָם קַבֵּיל נְבוּאָה עַל גוֹג אַרְעָא מָגוֹג רַב רֵישׁ מֶשֶׁךְ וְתֻבָל וְאִתְנַבֵּי עֲלוֹהִי:
וְתֵימַר כִּדְנַן אֲמַר יְיָ אֱלֹהִים הָא אֲנָא שְׁלַח רוּגְזִי עֲלָךְ גוֹג רַב רֵישׁ מֶשֶׁךְ וְתֻבָל:
וַאֲשַׁדְלִינָךְ וְאֶתֵּן שֵׁירִין עַל לִיסָתָךְ וְאַפֵּיק יָתָךְ וְיַת כָּל מַשְׁרִיָתָךְ סוּסְוָן וּפָרָשִׁין לְבוּשֵׁי גְמִיר כּוּלְהוֹן מַשִׁרְיָן סַגִיאָן דִי מְזַיְנִין בַּעֲגִילִין וּתְרֵיסִין אָחֳדֵי סַיְפִין כּוּלְהוֹן:
פַּרְסָאֵי כּוּשְׁאֵי וּפוּטָאֵי עִמְהוֹן כּוּלְהוֹן מְזַיְנִין בַּעֲגִילִין וְקוּלְסִין:
גוֹמֶר וְכָל מַשִׁרְיָתָהָא מְדִינַת גַרְמַמְיָא סְיָפֵי צִפּוּנָא וְיַת כָּל מַשִׁרְיָתְהוֹן עַמְמִין סַגִיאִין דִי עִמָךְ:
אִזְדַרַז וְאַתְקֵין לָךְ אַתְּ וְכָל מַשִׁרְיָתָךְ דְאִתְכְּנִישׁוּ עֲלָךְ וּתְהֵי לְהוֹן לְמַטָרָא:
מִיוֹמִין סַגִיאִין תְּמַנִי מַשִׁרְיָתָךְ בְּסוֹף שְׁנַיָא תְהַךְ עַל אַרְעָא דְתָבוּ עֲלָה מְקַטְלֵי חַרְבָּא דְאִתְכְּנָשׁוּ מֵעַמְמִין סַגִיאִין עַל טוּרֵי אַרְעָא דְיִשְׂרָאֵל דַהֲווֹ לְחָרְבָא תְּדִירָא וְאִינוּן מִבֵּינֵי עַמְמַיָא אִתְגְלִיאוּ וִיתִיבוּ לְרוֹחֲצָן כּוּלְהוֹן:
וְתִסַק בְּאִתְרְגוּשְׁתָּא תֵּיתֵי כַּעֲנָנָא דְסָלֵיק וְחָפֵי יַת אַרְעָא תְּהֵי אַתְּ וְכָל מַשִׁרְיָתָךְ וְעַמְמִין סַגִיאִין דִי עִמָךְ:
כִּדְנַן אֲמַר יְיָ אֱלֹהִים וִיהֵי בְּעִדָנָא הַהִיא יִסְקוּן הִרְהוּרִין עַל לִבָּךְ וּתְחַשֵׁב מַחְשְׁבָן בִּישָׁן:
וְתֵימַר אֶסַק עַל אַרְעָא דְיָתְבָא שְׁלֵיוָא בְּקִרְוֵי פִצְחַיָא אֱהַךְ עַל עַמָא דְשָׁקְטִין וְשָׁרָן לְרוֹחֲצָן כּוּלְהוֹן יָתְבִין בְּלָא שׁוּרָא וְעָבְרִין וְדָשִׁין לֵית לְהוֹן:
לְמִבַּז בִּזָא וּלְמֶעְדֵי עֲדָאָה לְכַנָשָׁא מַשִׁרְיָתָךְ עַל חָרְבָתָא דְאִתְיַתְבָא וְעַל עַמָא דְאִתְכַּנְשׁוּ מִבֵּינֵי עַמְמַיָא דְאִצְלָחוּ בְּנִכְסִין וְקִנְיָנִין יָתְבִין עַל תּוּקְפָא דְאַרְעָא:
שְׁבָא וּדְדָן וְתַגְרֵי יַמָא וְכָל מַלְכָּהָא יֵימְרוּן לָךְ הַלְמִבַּז בִּזָא אַתְּ אָתֵי הַלְמֶעֱדֵי עָדָאָה כְּנִישַׁת מַשִׁרְיָתָךְ לְמִיסַב כְּסַף וּדְהַב לְמַשְׁבֵי נִכְסִין וְקִנְיָנִין לְמֶעְדֵי עֲדִי סַגִי:
בְּכֵן אִתְנַבֵי בַּר אָדָם וְתֵימַר לְגוֹג כִּדְנַן אֲמַר יְיָ אֱלֹהִים הֲלָא בְּעִדָנָא הַהִיא כַּד יַתְבוּן עַמִי יִשְׂרָאֵל לְרוֹחֲצָן תֵּידַע פוּרְעֲנוּת גְבוּרְתִּי:
וְתֵיתֵי מֵאַתְרָךְ מִסְיָפֵי צִפּוּנָא אַתְּ וְעַמְמִין סַגִיאִין דִי עִמָךְ רוֹכְבֵי סוּסְוָן כּוּלְהוֹן מַשִׁרְיָן סַגִיאִין וְעַם סַגִי:
וְתִיסַק עַל עַמִי יִשְׂרָאֵל כַּעֲנָנָא דְסָלֵיק וְחָפֵי יַת אַרְעָא בְּסוֹף יוֹמַיָא תְּהֵי וְאַיְתִינָךְ עַל אַרְעָא בֵּית שְׁכִנְתִּי בְּדִיל דְיִדְעוּן עַמְמַיָא פּוּרְעֲנוּת גְבוּרְתִּי כַּד אִתְקַדֵשׁ בָּךְ דִיהוֹן חָזָן בְּפוּרְעֲנוּתָךְ גוֹג:
כִּדְנַן אֲמַר יְיָ אֱלֹהִים הַאַתְּ הוּא דְמַלֵלִית בְּיוֹמֵי קֶדֶם בְּיַד עַבְדֵי נְבִיֵי יִשְׂרָאֵל דְאִתְנַבִיאוּ בְּיוֹמַיָא הָאִינוּן מִלְקַדְמַת דְנָא שָׁנִין סַגִיאִין לְאַיְתָאָה יָתָךְ עֲלֵיהוֹן:
וִיהֵי בְּעִדָנָא הַהִיא בְּיוֹם מֵיתֵי גוֹג עַל אַרְעָא דְיִשְׂרָאֵל אֲמַר יְיָ אֱלֹהִים יַדְלִיק רוּגְזִי וְחֵימְתִי:
וּבְאִתְגְלָאָה פּוּרְעֲנוּתִי בְמִדְלַק אֶשַׁת רוּגְזִי גְזָרֵית בְמֵימְרִי אִם לָא בְּיוֹמָא הַהוּא יְהֵי זִיעַ סַגִי עַל אַרְעָא דְיִשְׂרָאֵל:
וִיזוּעוּן מִן קֳדָמַי נוּנֵי יַמָא וְעוֹפָא דִשְׁמַיָא וְחַיַת בָּרָא וְכָל רִחֲשָׁא דְרָחֵישׁ עַל אַרְעָא וְכָל אֱנָשָׁא דִי עַל אַפֵּי אַרְעָא וְיִתְפַּגְרוּן טוּרַיָא וְיִתְתַּלְשׁוּן מִגְדָלַיָא וְכָל שׁוּר רָם לְאַרְעָא יִתְחַמָר:
וַאֲזַמְנִינֵיהּ לְמִפַּל בְּחַרְבָּא עַל טוּרֵי יִשְׂרָאֵל עַמִי אֲמַר יְיָ אֱלֹהִים חֶרֶב גְבַר בַּאֲחוּהִי תְהֵי:
וְאֶתְפְּרַע מִנֵיהּ בְּמוֹתָא וּבִקְטָלָא וּמְטַר מִגְבַּר וְאַבְנֵי אֶלְגָבִישׁ אֶשְׁתָּא וְגוּפְרִיתָא אֶמְטַר עֲלוֹהִי וְעַל מַשִׁרְיָתֵיהּ וְעַל עַמְמִין סַגִיאִין דִי עִמֵיהּ:
וְאִתְרַבֵּי וְאִתְקַדֵשׁ וַאֲגַלֵי גְבוּרְתִּי לְעֵינֵי עַמְמִין סַגִיאִין וְיֵדְעוּן אֲרֵי אֲנָא יְיָ:
