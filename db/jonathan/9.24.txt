וְאוֹסִיף רוּגְזָא דַייָ לְמִתְקַף בְּיִשְׂרָאֵל וְגָרֵי יַת דָוִד בְּהוֹן לְמֵימַר אֱזִיל מְנֵי יַת יִשְׂרָאֵל וְיַת דְבֵית יְהוּדָה:
וַאֲמַר מַלְכָּא לְיוֹאָב רַב חֵילָא דְעִמֵיהּ הֲלֵיךְ כְּעַן בְּכָל שִׁבְטָא דְיִשְׂרָאֵל מִדָן וְעַד בְּאֵר שֶׁבַע וּמְנֵי יַת עַמָא וְאִידַע יַת חֻשְׁבַּן עַמָא:
וַאֲמַר יוֹאָב לְמַלְכָּא וְיוֹסֵף יְיָ אֱלָהָךְ עַל עַמָא כְּאִלֵין וּכְאִלֵין מְאָה זִמְנִין וְעֵינֵי רִבּוֹנִי מַלְכָּא חָזָן וְרִבּוֹנִי מַלְכָּא לְמָא אִתְרְעֵי בְּפִתְגָמָא הָדֵין:
וּתְקֵיף פִּתְגָמָא דְמַלְכָּא עַל יוֹאָב וְעַל רַבָּנֵי חֵילָא וּנְפַק יוֹאָב וְרַבָּנֵי חֵילָא קֳדָם מַלְכָּא לְמִמְנֵי יַת עַמָא יַת יִשְׂרָאֵל:
וַעֲבַרוּ יַת יַרְדְנָא וּשְׁרוֹ בַעֲרוֹעֵר דָרוֹם קַרְתָּא דִבְגוֹ נַחְלָא דְשִׁבְטָא גָד וְאֶל יַעְזֵר:
וַאֲתוֹ לְגִלְעָד וּלְאַרְעָא דָרוֹמָא לְחָדְשִׁי וַאֲתוֹ לְדָן יַעַן וּמִתַּמָן אִסְתְּחַר לְצִידוֹן:
וַאֲתוֹ לְקִירְוִין כְּרִיכָן וְכָל קִרְוֵי חִוָאֵי וּכְנַעֲנָאֵי וּנְפַקוּ לְדָרוֹם יְהוּדָה לִבְאֵר שָׁבַע:
וְהֲלִיכוּ בְּכָל אַרְעָא וַאֲתוֹ מִסְיָפֵי תִשְׁעַת יַרְחִין וְעַסְרִין יוֹמִין לִירוּשְׁלֵם:
וִיהַב יוֹאָב יַת חֻשְׁבַּן מִנְיַן עַמָא לְמַלְכָּא וַהֲווֹ יִשְׂרָאֵל תַּמְנֵי מְאָה אַלְפִין גְבַר גִבַּר שַׁלְפֵי סַיָפָא וֶאֱנַשׁ יְהוּדָה חֲמֵשׁ מְאָה אַלְפִין גַבְרָא:
וְחַשׁ דָוִד בְּלִבֵּיהּ בָּתַר כֵּן דִמְנָא יַת עַמָא וַאֲמַר דָוִד קֳדָם יְיָ חָבִית לַחֲדָא דַעֲבָדִית הָדָא וּכְעַן יְיָ אַעֲבַּר כְּעַן יַת חוֹבָא דְעַבְדָךְ אֲרֵי אִתְתַּכָּלִית לַחֲדָא:
וְקָם דָוִד בְּצַפְרָא וּפִתְגַם נְבוּאָה מִן קֳדָם יְיָ הֲוָה עִם גָד נְבִיָא חֶזְוַיָא דְדָוִד לְמֵימָר:
אֱזִיל וּתְמַלֵיל עִם דָוִד כִּדְנַן אֲמַר יְיָ חֲדָא מִתְּלַת אֲנָא רָמֵי עֲלָךְ בְּחַר לָךְ חַד מִנְהוֹן וְאַעְבֵּד לָךְ:
וַאֲתָא גָד לְוַת דָוִד וְחַוֵי לֵיהּ וַאֲמַר לֵיהּ הֲתָעוֹל לָךְ שְׁבַע שְׁנִין כַּפְנָא בְּאַרְעָךְ אִם תְּלָתָא יַרְחִין תְּהֵי עָרִיק קֳדָם סַנְאָךְ וְהוּא רָדִיף לָךְ וְאִם מֶהֱוֵה תְּלָתָא יוֹמִין מוֹתָא בְּאַרְעָךְ כְּעַן דַע וַחֲזֵי מָה אֲתִיב שֹׁלְחִי פִּתְגָמָא:
וַאֲמַר דָוִד לְגָד עָקַת לִי לַחֲדָא נִתְמְסַר כְּעַן בְּיַד מֵימְרָא דַייָ אֲרֵי סַגִיאִין רַחֲמוֹהִי וּבִידָא דֶאֱנָשָׁא לָא אִתְמְסָר:
וִיהַב יְיָ מוֹתָא בְּיִשְׂרָאֵל מֵעִידַן דְמִתְנְכִיס תְּמִידָא וְעַד דְמִתַּסֵק וּמִית מִן עַמָא מִדָן וְעַד בְּאֵר שֶׁבַע שַׁבְעִין אַלְפִין גַבְרָא:
וְאוֹשִׁיט יְדֵיהּ מַלְאָכָא לִירוּשְׁלֵם לְחַבָּלוּתָהּ וְתַב יְיָ מִן בִּישְׁתָא וַאֲמַר לְמַלְאָכָא דִמְחַבֵּל בְּעַמָא מִסַת כְּעַן אַגַח יְדָךְ וּמַלְאָכָא דַייָ הֲוָה שָׁרֵי בְּבֵית אִידְרֵי דְאַרְנוֹן יְבוּסָאָה:
וַאֲמַר דָוִד קֳדָם יְיָ כַּד חֲזָא יַת מַלְאָכָא דְמִקַטֵל בְּעַמָא וַאֲמַר הָא אֲנָא חָבִית וַאֲנָא סְרָחִית וְאִלֵין עַמָא דְאִינוּן כְּעָנָא בְּיַד רַעְיָא מָה עֲבָדוּ תְּהֵי כְּעַן מָחֲתָךְ בִּי וּבְבֵית אַבָּא:
וַאֲתָא גָד לְוַת דָוִד בְּיוֹמָא הַהוּא וַאֲמַר לֵיהּ סַק אֲקֵים קֳדָם יְיָ מַדְבְּחָא בְּבֵית אִידְרֵי דַאֲרַוָן יְבֻסָאָה:
וּסְלִיק דָוִד כְּפִתְגָמָא דְגָד כְּמָא דְפַקֵיד יְיָ:
וְאִסְתְּכִי אֲרַוָן וַחֲזָא יַת מַלְכָּא וְיַת עַבְדוֹהִי עָבְרִין עֲלוֹהִי וּנְפַק אֲרְוָן וּסְגִיד לְמַלְכָּא אַפּוֹהִי עַל אַרְעָא:
וַאֲמַר אֲרַוָן מָא דֵין אֲתָא רִבּוֹנִי מַלְכָּא לְוַת עַבְדֵיהּ וַאֲמַר דָוִד לְמִזְבַן מִנָךְ יַת בֵּית אִידְרֵי לְמִבְנָא מַדְבְּחָא קֲדָם יְיָ וְיִתְכְּלִי מוֹתָנָא מֵעַל עַמָא:
וַאֲמַר אֲרַוָן לְדָוִד יִסַב וְיִסַק רִבּוֹנִי מַלְכָּא דְתַקִין בְּעֵינוֹהִי חֲזֵי תּוֹרַיָא דִכְשֵׁרִין לַעֲלָתָא וּמוֹרִיגַיָא וּמָנֵי תוֹרַיָא לְאָעַיָא:
כֹּלָא יְהַב אֲרַוָן מַלְכָּא לְמַלְכָּא דִי בְעָא מִנֵיהּ מַלְכָּא וַאֲמַר אֲרַוָן לְמַלְכָּא יְיָ אֱלָהָךְ יְקַבֵּיל קוּרְבָּנָךְ בְּרַעֲוָא:
וַאֲמַר מַלְכָּא לַאֲרַוָן לָא אֲרֵי מִזְבַּן אֶזְבּוּן מִנָךְ בְּדָמִין וְלָא אַסִיק קֳדָם יְיָ אֱלָהָי עֲלַוָן מַגָן וּזְבַן דָוִד יַת בֵּית אִידְרֵי וְיַת תּוֹרַיָא בִּכְסַף סַלְעִין חַמְשִׁין:
וּבְנָא תַמָן דָוִד מַדְבְּחָא קֳדָם יְיָ וְאַסֵיק עֲלַוָן וְנִכְסַת קוּדְשִׁין וְקַבֵּיל יְיָ צְלוֹת דַיָרֵי אַרְעָא וְאִתְכְּלִי מוֹתָנָא מֵעַל יִשְׂרָאֵל:
