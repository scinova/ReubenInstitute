וּבְפַלְגוּתְהוֹן יַת אַרְעָא בְאַחֲסָנָא תִּפְרְשׁוּן אַפְרָשׁוּתָא קֳדָם יְיָ קוּדְשָׁא מִן אַרְעָא אוּרְכָּא עַסְרִין וְחַמְשָׁא אַלְפִין אוּרְכָּא וּפוּתְיָא עַסְרָא אַלְפִין קוּדְשָׁא הוּא בְּכָל תְּחוּמָה סְחוֹר סְחוֹר:
יְהֵי מִדֵין לְקוּדְשָׁא חֲמֵשׁ מְאָה בַּחֲמֵשׁ מְאָה מְרֻבַּע סְחוֹר סְחוֹר וְחַמְשִׁין אַמִין רוּחָא לֵיהּ סְחוֹר סְחוֹר:
וּמִן מִשְׁחָתָא הָדָא תִּמְשַׁח אוּרְכָּא עַסְרִין וְחַמְשָׁא אַלְפִין וּפוּתְיָא עַסְרָא אַלְפִין וּבֵיהּ יְהֵי מִקְדְשָׁא קוֹדֶשׁ קוּדְשִׁין:
קוּדְשָׁא מִן אַרְעָא הוּא לְכַהֲנַיָא דִי מְשַׁמְשִׁין בְּבֵית מַקְדְשָׁא דַייָ יְהֵי דְקָרִיבִין לְשַׁמָשָׁא קֳדָם יְיָ וִיהֵי לְהוֹן אֲתַר שְׁבִיק לְבֵיתָא וְכִבְשָׁא לְמַקְדְשָׁא:
וְעַסְרִין וְחַמְשָׁא אַלְפִין אוּרְכָּא וְעַסְרָא אַלְפִין פּוּתְיָא וִיהֵי לְלֵיוָאֵי מְשַׁמְשֵׁי בֵיתָא לְהוֹן לְאַחֲסָנָא עַסְרִין לִשְׁכָּן:
וְאַחֲסָנַת קַרְתָּא תִּתְּנוּן חַמְשָׁא אַלְפִין פּוּתְיָא וְאוּרְכָּא עַסְרִין וְחַמְשָׁא אַלְפִין לָקֳבֵיל אַפְרָשׁוּת קוּדְשָׁא לְכָל בֵּית יִשְׂרָאֵל יְהֵי:
וּלְרַבָּא מִכָּא וּמִכָּא לְאַפְרָשׁוּת קוּדְשָׁא וּלְאַחְסָנַת קַרְתָּא לָקֳבֵיל אַפְרָשׁוּת קוּדְשָׁא וְלִקֳבֵיל אַחְסָנַת קַרְתָּא מֵרוּחַ מַעַרְבָא מַעַרְבָא וּמֵרוּחַ מַדִינְחָא מַדִינְחָא וְאוּרְכָּא לָקֳבֵיל חַד מִן חוּלְקַיָא מִתְּחוּם מֵעַרְבָא עַד תְּחוּם מַדִינְחָא:
דָא אַרְעָא תְּהֵי לְרַבָּא לְאַחְסַנְתָּא בְּיִשְׂרָאֵל וְלָא יֵנִיסוּן עוֹד רַבְרְבֵי יַת עַמִי וְאַרְעָא יִתְּנוּן לְבֵית יִשְׂרָאֵל לְשִׁבְטֵיהוֹן:
כִּדְנַן אֲמַר יְיָ אֱלֹהִים סַגִי לְכוֹן רַבְרְבֵי יִשְׂרָאֵל דַטְפָא וּבִזָא אַעְדוּ וְדִין דִקְשׁוֹט וּזְכוּ עֲבִידוּ סְלִיקוּ תִקְלַתְכוֹן מִגוֹ עַמִי אֲמַר יְיָ אֱלֹהִים:
מוֹזְנִוָן דִקְשׁוֹט וּמְכִילַן דִקְשׁוֹט וּבָתִּין דִקְשׁוֹט יְהֵי לְכוֹן:
מְכִלְתָּא וּבֵיתָא טִקוּס חַד יְהֵי לְכוֹן סְכוֹם תְּלַת סָאִין לְמִסַב מַעֲשַׂר כּוֹרָא בִּמְכִלְתָּא רַטִיבָא בֵּיתָא וַעֲשַׂר כּוֹרָא יַבִּישָׁא מְכִלְתָּא מִן כּוֹרָא יְהֵי טְקוּסֵיהּ:
וְסַלְעָא עַסְרִין מָעִין תִּלְתּוּת מָנַיָא עַסְרִין סַלְעִין מָנֵי כַסְפָּא עַסְרִין וַחֲמֵשׁ סַלְעִין רִבְעוּת מָנַיָא חֲמֵשׁ עַסְרֵי סַלְעִין כּוּלְהוֹן שִׁתִּין מָנִין וּמָנֵי רַבָּא קוּדְשָׁא יְהֵי לְכוֹן:
דָא אַפְרָשׁוּתָא דְתִפְרְשׁוּן חַד מָן שִׁתָּא בִמְכִלְתָּא מִכּוֹר חִטִין וְחַד מִן שִׁתָּא בִמְכִלְתָּא מִכּוֹר סַעֲרִין:
וְדַחֲזֵי לְמִסַב מִן מִשְׁחָא בִּמְכִלְתָּא רַטִיבָא מֵעִישׂוּרָא בֵיתָא מִן כּוֹרָא חַד מִן עַסְרָא הַוְיָא בֵיתָא בְּכּוֹרָא אֲרֵי עֲסַר בַּתִּין כּוֹרָא:
וְאִימַר חַד מִן עֲנָא מִן מָאתָן דַחֲזֵי לְמִסַת מִפַּטְמָא דְיִשְׂרָאֵל לְמִנְחֲתָא וְלַעֲלָתָא וּלְנִכְסַת קוּדְשַׁיָא לְכַפָּרָא עֲלֵיהוֹן אֲמַר יְיָ אֱלֹהִים:
כָּל עַמָא דְאַרְעָא יְהוֹן עָבְדִין יַת אַפְרָשׁוּתָא הָדָא לְרַבָּא בְּיִשְׂרָאֵל:
וְעַל רַבָּא יְהֵי עֲלָתָא וּמִנְחֲתָא וְנִסְכַיָא בְּחַגַיָא וּבְיַרְחַיָא וּבְשַׁבַּיָא וּבְכָל מוֹעֲדֵי בֵּית יִשְׂרָאֵל הוּא יַעֲבֵיד יַת חַטָאתָא וְיַת מִנְחָתָא וְיַת עֲלָתָא וְיַת נִכְסַת קוּדְשַׁיָא לְכַפָּרָא עַל בֵּית יִשְׂרָאֵל:
כִּדְנַן אֲמַר יְיָ אֱלֹהִים בְּקַדְמָאָה בְּחַד לְיַרְחָא תִּסַב תּוֹר בַּר תּוֹרֵי שְׁלִים וְתִדְכֵּי יַת מַקְדְשָׁא:
וְיִסַב כַּהֲנָא מִדְמָא דְחַטָאתָא וְיִתֵּן עַל סְקוֹפַת בֵּיתָא וְעַל אַרְבַּע זִוְיַת מְסַמָתָא לְמַדְבְּחָא וְעַל סְקוֹפַת תַּרְעָא דְדַרְתָּא גַוַיְתָא:
וְכֵן תַּעְבֵיד בְּשַׁבְעָא בְּיַרְחָא מִגְבַר דְמִשְׁתְּלֵי וּמִטְפַשׁ וּתְכַפְּרוּן עַל בֵּיתָא:
בְּנִיסָן בְּאַרְבְּעָא עַסְרָא יוֹמָא לְיַרְחָא יְהֵי לְכוֹן פִּסְחָא חַגָא שַׁבְעָא יוֹמִין פַּטִירַיָא יִתְאֲכֵיל:
וְיַעְבֵּיד רַבָּא בְּיוֹמָא הַהוּא חִלוֹפוֹהִי וַחֲלַף כָּל עַמָא דְאַרְעָא תּוֹרָא דְחַטָאתָא:
וְשַׁבְעָא יוֹמֵי דְחַגָא יַעְבֵּיד עֲלָתָא קֳדָם יְיָ שַׁבְעָא תוֹרִין וְשַׁבְעָא דִכְרִין שָׁלְמִין לְיוֹמָא שַׁבְעַת יוֹמִין וְחַטָאתָא צְפִיר בַּר עִזֵי לְיוֹמָא:
וּמִנְחָתָא מְכִלְתָּא לְתוֹרָא וּמְכִלְתָּא לְדִכְרָא יַעְבֵּיד וּמִשְׁחָא מְלֵי הִינָא לִמְכִלְתָּא:
בִּשְׁבִיעָאָה בְּחַמְשַׁת עַסְרָא יוֹמָא לְיַרְחָא בְּחַגָא יַעְבֵּיד כְּאִלֵין שִׁבְעָא יוֹמִין כְּאִלֵין שִׁבְעָא יוֹמִין כְּחַטָאתָא וְכַעֲלָתָא וּכְמִנְחָתָא וּכְמִשְׁחָא:
