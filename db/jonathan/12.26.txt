בְּעִדָנָא הַהִיא יְשַׁבְּחוּן תוּשְׁבַּחְתָּא חַדְתָּא בְּאַרְעָא דְבֵית יְהוּדָה קִרְיַת תּוּקְפָא לָנָא פוּרְקָן יִתְשָׂם עַל שׁוּרָהָא וְרַחֲמִין:
פְּתַחוּ תַּרְעַיָא וְיֵעוּל עַמָא זַכָּאָה דִי נְטָרוּ אוֹרַיְתָא בְּלֵבָב שָׁלֵם:
בְּלֵבָב שָׁלֵם נְטָרוּ שְׁלָמָא שְׁלָמָא יִתְעֲבֵיד לְהוֹן אֲרֵי בְמֵימְרָךְ אִתְרְחִיצוּ:
אִתְרְחִיצוּ עַל מֵימְרָא דַייָ לְעָלַם וּלְעָלְמֵי עָלְמַיָא בְּכֵן תִּתְפַרְקוּן בְּמֵימַר דְחִילָא דַייָ תַּקִיף עָלְמַיָא:
אֲרֵי אַמְאִיךְ יַתְבֵי רוּמָא קִרְיָא תַקִיפָא יַמְאֲכִינָא וְיַרְמִינָהּ עַד אַרְעָא יִמְטִינָהּ עַד עַפְרָא:
יְדוּשׁוּנָה רִגְלִין רִגְלֵי צַדִיקַיָא פַּרְסַת עִנְוְתָנַיָא חֲשִׁיכֵי עַמָא:
אוֹרְחַת צַדִיקַיָא כַּוָנָן אַף עֲבַדוּ אוֹרְחַת צַדִיקַיָא תְּתַקֵן:
אַף לְאוֹרַח דִינָךְ יְיָ סְבַרְנָא לִשְׁמָךְ וּלְדָכְרָנָךְ חֲמֵדַת נַפְשָׁנָא:
נַפְשִׁי מַחְמְדָא לְצַלָאָה קֳדָמָךְ בְּלֵילְיָא אַף רוּחִי בִמְעַי מְבָרְכָא לָךְ אֲרֵי כְּמָא דְדִינָךְ תַּקִין לְאַרְעָא קוּשְׁטָא יִתְאַלְפוּן לְמֶעְבַד דְדָיְרִין בְּתֵבֵל:
יְהַבְתָּא לְהוֹן אַרְכָּא לְרַשִׁיעַיָא אִם יְתוּבוּן לְאוֹרַיְתָךְ וְלָא תָבוּ כָּל יוֹמִין דַהֲווֹ קַיָמִין יַעְבְּדוּן קְשׁוֹט עַל אַרְעָא חֲווֹ שַׁקָרִין אַף אִינוּן לָא יֶחֱזוּן בְּתוּשְׁבְּחַת יְקָרָךְ יְיָ:
יְיָ כַּד תִּתְגְלֵי בִּגְבוּרְתָּךְ לְאוֹטָבָא לִדְחַלְתָּךְ לָא יִנְהַר לְסָנְאֵי עַמְךְ יֶחֱזוּן רַשִׁיעַיָא וְיִבְהֲתוּן פּוּרְעֲנוּת עַמָא תַחְפִּינוּן אַף אֶשְׁתָּא לְבַעֲלֵי דְבָבָךְ תְּשֵׁיצֵי:
יְיָ תְּתַקֵין שְׁלָמָא לָנָא אֲרֵי בְּכָל עִדָן חֲסֵיכְנָא מֵחוֹבָנָא אַתְּ עֲבֵיד עִמָנָא:
יְיָ אֱלָהָנָא שַׁלִיטוּ בָנָא עַמְמַיָא לְמֶהֱוֵי עֲלָנָא רִבּוֹנִין בַּר מִנָךְ בְּרַם בְּמֵימְרָךְ אֲנַחְנָא רְחִיצִין בִּשְׁמָךְ אֲנַחְנָא מוֹדָן:
אִינוּן פַּלְחִין לְמֵתִין דְלָא יֵחוּן לִגְבָּרֵהוֹן דְלָא יְקוּמוּן בְּכֵן כַּד תַּסְעַר עֲלֵיהוֹן חוֹבֵיהוֹן וּתְשֵׁיצִינוּן וְתוֹבִיד כָּל דָכְרָנְהוֹן:
אִתְגְלֵיתָא לְכַנָשָׁא מְבַדְרֵי עַמָךְ יְיָ אַף עָתִיד אַתְּ לְקָרָבָא גָלוּתְהוֹן לְאִתְגְלָאָה בִגְבוּרְתֵּךְ לְרַחָקָא כָּל רַשִׁיעַיָא לִגְהִנָם:
יְיָ בְּעָקָא הֲווֹ דְכִירִין לִדְחַלְתָּךְ בְּעָקַתְהוֹן הֲווֹ מַלְפִין אוּלְפַן אוֹרַיְתָךְ בַּחֲשָׁי:
כִּמְעַדְיָא דִמְטָא דִבְעִדָן מֵילְדָה זָיְעָא וּמְצַוְחָא בְּחֶבְלָהָא כֵּן הֲוֵינָא מִן קֳדָם דְחַבְנָא קֳדָמָךְ יְיָ:
אֲחַדְתָּנָא עָקָא כִּמְעַדְיָא דִמְטָת עִדָן מֵילְדָהּ קַלִיל כְּרוּחָא פּוּרְקַן לָא אַיְתִיאוּ אַרְעָא לָא עֲבַדוּ אַף פְּרִישָׁן לָא יָכְלִין לְמֶעְבַּד דְדָיְרִין בְּתֵבֵל:
אַתְּ הוּא מְחַיֵי מֵיתִין גַרְמֵי נְבֶלְתְּהוֹן אַתְּ מֵקִים יֵחוּן וִישַׁבְּחוּן קֳדָמָךְ כָּל דַהֲווֹ רְמָן בְּעַפְרָא אֲרֵי טַל נְהוֹר טַלָךְ לְעָבְדֵי אוֹרַיְתָךְ וְרַשִׁיעַיָא דִי יְהַבְתָּא לְהוֹן גְבוּרָא וְאִינוּן עֲבַרוּ עַל מֵימְרָךְ לִגְהִנָם תִּמְסַר:
אֱזֵיל עַמִי עֲבֵיד לָךְ עוֹבָדִין טָבִין דְיִגְנוּן עֲלָךְ בְּעִדַן עָקָא אִטַמַר כִּזְעֵיר זְמַן עַד דְיֶעְדֵי לְוָט:
אֲרֵי הָא יְיָ מִתְגְלֵי מֵאֲתַר שְׁכִנְתֵּהּ לְאַסְעָרָא חוֹב יָתֵיב אַרְעָא עֲלוֹהִי וּתְגַלֵי אַרְעָא עַל דַם זַכַּאי דְאִתְשֵׁיד בָּהּ וְלָא תְכַסֵי עוֹד עַל קְטִילָהָא:
