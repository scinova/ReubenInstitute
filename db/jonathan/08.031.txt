וּפְלִשְׁתָּאֵי מַגִיחִין קְרָבָא בְּיִשְׂרָאֵל וְאַפִּיכוּ אֱנָשֵׁי יִשְׂרָאֵל מִן קֳדָם פְלִשְׁתָּאֵי וּנְפַלוּ קְטִילִין בְּטוּרָא דְגִלְבּוֹעַ:
וְאַדְבִּיקוּ פְלִשְׁתָּאֵי יַת שָׁאוּל וְיַת בְּנוֹהִי וּקְטָלוּ פְלִשְׁתָּאֵי יַת יְהוֹנָתָן וְיַת אֲבִינָדָב וְיַת מַלְכִּי שׁוּע בְּנֵי שָׁאוּל:
וְתַקִיפוּ עָבְדֵי קְרָבָא עַל שָׁאוּל וְאַשְׁכְּחוֹהִי קַשְׁתַּיָא גַבְרִין דְאוּמָנִין לְמֵיגַד בְּקַשְׁתָּא וּדְחִיל לַחֲדָא מִקַשְׁתַּיָא:
וַאֲמַר שָׁאוּל לְנָטֵל זְיָנֵיהּ שְׁלוֹף חַרְבָּךְ וְקַטְלַנִי בָהּ דִלְמָא יֵיתוּן עֲרֵלַיָא הָאִלֵין וְיִקְטְלוּנַנִי וְיִתְלַעֲבוּן בִּי וְלָא אָבָא נָטֵל זְיָנֵיהּ אֲרֵי דָחִיל לַחֲדָא וּנְסִיב שָׁאוּל יַת חַרְבָּא וּנְפַל עֲלָהּ:
וַחֲזָא נָטֵל זְיָנֵיהּ אֲרֵי מִית שָׁאוּל וּנְפַל אַף הוּא עַל חַרְבֵּיהּ וּמִית עִמֵיהּ:
וּמִית שָׁאוּל וּתְלָתָא בְּנוֹהִי וְנָטֵל זְיָנֵיהּ אַף כָּל גַבְרוֹהִי בְּיוֹמָא הַהוּא כַּחֲדָא:
וַחֲזוֹ אֱנָשֵׁי יִשְׂרָאֵל דִבְּעִבְרָא דְמֵישְׁרָא וְדִבְּעִבְרָא דְיַרְדְנָא אֲרֵי אַפִיכוּ אֱנָשֵׁי יִשְׂרָאֵל וַאֲרֵי מִיתוּ שָׁאוּל וּבְנוֹהִי וּשְׁבָקוּ יַת קִרְוַיָא וְאַפִּיכוּ וַאֲתוֹ פְלִשְׁתָּאֵי וִיתִיבוּ בְהוֹן:
וַהֲוָה בְּיוֹמָא דְבַתְרוֹהִי וַאֲתוֹ פְלִשְׁתָּאֵי לְחַלָצָא קְטִילַיָא וְאַשְׁכָּחוּ יַת שָׁאוּל וְיַת תְּלָתָא בְּנוֹהִי קַטִילִין רְמָן בְּטוּרָא דְגִלְבּוֹעַ:
וּפְסַקוּ יַת רֵישֵׁיהּ וְחַלִיצוּ זְיָנֵיהּ וּשְׁלָחוּ בְּאַרְעָא דִפְלִשְׁתָּאֵי סְחוֹר סְחוֹר לְבַסָרָא בֵּית טַעֲוַתְהוֹן וְיַת עַמָא:
וְשַׁוִיאוּ יַת זְיָנֵיהּ בֵּית עַשְׁתְּרָתָא וְיַת גוּפֵיהּ צְלָבוּ בְּשׁוּרָא דְבֵית שָׁן:
וּשְׁמָעוּ עֲלוֹהִי יַתְבֵי יָבֵשׁ גִלְעָד יַת דַעֲבַדוּ פְּלִשְׁתָּאֵי לְשָׁאוּל:
וְקָמוּ כָּל גְבַר גִבַּר וַאֲזַלוּ כָּל לֵילְיָא וּנְסִיבוּ יַת גוּפָא דְשָׁאוּל וְיַת גוּפֵי בְּנוֹהִי מִשׁוּרָא דְבֵית שָׁן וַאֲתוֹ לְיָבֵשׁ וּקְלוֹ עֲלֵיהוֹן כְּמָא דְקָלַן עַל מַלְכַיָא תַּמָן:
וּנְסִיבוּ יַת גַרְמֵיהוֹן וּקְבָרוּ תְּחוֹת אֲשֵׁלָא בְּיָבֵשׁ וְצָמוּ שַׁבְעָא יוֹמִין:
