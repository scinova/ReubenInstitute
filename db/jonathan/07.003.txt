וְאִלֵין עַמְמַיָא דִי שְׁבַק יְיָ לְנַסָאָה בְהוֹן יַת יִשְׂרָאֵל יַת כָּל דְלָא הֲווֹ יָדְעִין יַת כָּל קְרָבֵי כְנָעַן:
לְחוֹד בְּדִיל דְיֵדְעוּן דָרַיָא דִבְנֵי יִשְׂרָאֵל לַאֲלָפוּתְהוֹן קְרָבָא לְחוֹד דְמִלְקַדְמִין לָא הֲווֹ יָדְעִין:
חַמְשָׁא טוּרְנֵי פְּלִשְׁתָּאֵי וְכָל כְּנַעֲנָאֵי וְצִידוֹנָאֵי וְחִוָאֵי יַתְבֵי טוּרָא דִלְבָנוֹן מִטוּר מֵישַׁר חֶרְמוֹן עַד מַעֲלָנָא דַחֲמָת:
וַהֲווֹ לְנַסָאָה בְּהוֹן יַת יִשְׂרָאֵל לְמִידַע הַיְקַבְּלוּן יַת פִּקוּדַיָא דַייָ דְפַקֵד יַת אֲבָהַתְהוֹן בִּידָא דְמֹשֶׁה:
וּבְנֵי יִשְׂרָאֵל יְתִיבוּ בְּגוֹ כְנַעֲנָאֵי חִתָּאֵי וֶאֱמוֹרָאֵי וּפְרִיזָאֵי וְחִוָאֵי וִיבוּסָאֵי:
וּנְסִיבוּ יַת בְּנָתֵיהוֹן לְהוֹן לִנְשִׁין וְיַת בְּנַתְהוֹן יְהִיבוּ לִבְנֵיהוֹן וּפְלַחוּ יַת טַעֲוַתְהוֹן:
וַעֲבַדוּ בְנֵי יִשְׂרָאֵל יַת דְבִישׁ קֳדָם יְיָ וְאִתְנְשִׁיאוּ יַת פּוּלְחָנָא דַייָ אֱלָהָהוֹן וּפְלַחוּ יַת בְּעָלַיָא וְלַאֲשָׁרָתָא:
וּתְקֵיף רוּגְזָא דַייָ בְּיִשְׂרָאֵל וּמַסְרִינוּן בְּיַד כּוּשַׁן חַיָבָא מַלְכָּא דַאֲרָם דִי עַל פְּרָת וּפְלַחוּ בְּנֵי יִשְׂרָאֵל יַת כּוּשַׁן חַיָבָא תְּמָנֵי שְׁנִין:
וּזְעִיקוּ בְּנֵי יִשְׂרָאֵל קֳדָם יְיָ וְאֲקֵים יְיָ פָרִיק לִבְנֵי יִשְׂרָאֵל וּפַרְקִינוּן יַת עָתְנִיאֵל בַּר קְנַז אֲחוּהִי דְכָלֵב דִזְעֵיר מִנֵיהּ:
וּשְׁרַת עֲלוֹהִי רוּחַ נְבוּאָה מִן קֳדָם יְיָ וְדָן יַת יִשְׂרָאֵל וּנְפַק לְאַגָחָא קְרָבָא וּמְסַר יְיָ בִּידוֹהִי יַת כּוּשַׁן חַיָבָא מַלְכָּא דַאֲרָם דִי עַל פְרָת וּתְקֵפַת יְדֵיהּ עַל כּוּשַׁן חַיָבָא:
וּשְׁדוֹכַת אַרְעָא דְיִשְׂרָאֵל אַרְבְּעִין שְׁנִין וּמִית עָתְנִיאֵל בַּר קְנַז:
וְאוֹסִיפוּ בְּנֵי יִשְׂרָאֵל לְמֶעְבַּד דְבִישׁ קֳדָם יְיָ וּתְקֵיף יְיָ יַת עֶגְלוֹן מַלְכָּא דְמוֹאָב עַל יִשְׂרָאֵל עַל דַעֲבָדוּ יַת דְבִישׁ קֳדָם יְיָ:
וּכְנַשׁ לְוָתֵיה יַת בְּנֵי עַמוֹן וַעֲמָלֵק וַאֲזַל וּמְחָא יַת יִשְׂרָאֵל וִירִיתוּ יַת קַרְתָּא יְרִיחוֹ:
וּפְלַחוּ בְּנֵי יִשְׂרָאֵל יַת עֶגְלוֹן מַלְכָּא דְמוֹאָב תַּמְנֵי עַסְרֵי שְׁנִין:
וּזְעִיקוּ בְנֵי יִשְׂרָאֵל קֳדָם יְיָ וַאֲקֵים יְיָ לְהוֹן פָּרִיק יַת אֵהוּד בַּר גֵרָא בַּר שִׁבְטָא דְבִנְיָמִין גַבְרָא גְמִיד בִּידֵיהּ דְיַמִינָא וְשַׁדָרוּ בְנֵי יִשְׂרָאֵל בִּידֵיהּ תִּקְרוּבְתָּא לְעֶגְלוֹן מַלְכָּא דְמוֹאָב:
וַעֲבַד לֵיהּ אֵהוּד חַרְבָּא וְלָהּ תְּרֵין פּוּמִין גַרְמִידָא אָרְכָּא וְזָרֵיז יָתָהּ מִלְרַע לִלְבוּשׁוֹהִי עַל יַרְכֵיהּ דְיַמִינֵיהּ:
וְקָרֵיב יַת תִּקְרוּבְתָּא לְעֶגְלוֹן מַלְכָּא דְמוֹאָב וְעֶגְלוֹן גְבַר פַטִים לַחֲדָא:
וַהֲוָה כַּד שֵׁיצֵי לְקָרָבָא יַת תִּקְרוּבְתָּא וּשְׁלַח יַת עַמָא נַטְלֵי תִּקְרוּבְתָּא:
וְהוּא תַב מִן מַחֲצָבַיָא דְעִם גִלְגָלָא וַאֲמַר פִּתְגָמָא דְסִתְרָא אִית לִי לְמַלָלָא עִמָךְ מַלְכָּא וַאֲמַר סְלִיק וּנְפַקוּ מֵעִלַווֹהִי כָּל דְקַיְמִין עֲלוֹהִי:
וְאֵהוּד עָאל לְוָתֵיהּ וְהוּא יָתֵיב בַּעֲלִיַת בֵּית קַיְטָא דִילֵיהּ בִּלְחוֹדוֹהִי וַאֲמַר אֵהוּד פִּתְגָמָא דַייָ לִי לְמַלָלָא עִמָךְ וְקָם מֵעַל כּוּרְסֵיהּ:
וְאוֹשִׁיט אֵהוּד יַת יַד שְׂמָאלֵיהּ וּנְסִיב יַת חַרְבָּא מֵעַל יַרְכָא דְיַמִינֵיהּ וּקְבָעָה בִּמְעוֹהִי:
וְעַל אַף דִסְתְּקָא בָּתַר שְׁנָנָא וְטִמְטֵם שׁוּמְנָא בְּאַפֵּי שְׁנָנָא אֲרֵי לָא שְׁלַף חַרְבָּא מִמְעוֹהִי וּנְפַק אוּכְלֵיהּ שְׁפִיךְ:
וּנְפַק אֵהוּד לְאַכְסַדְרָא וַאֲחַר דָשֵׁי עִילִיתָא בְאַפּוֹהִי וְאַגָף:
וְהוּא נְפַק וְעַבְדוֹהִי עֲלוֹ וַחֲזוֹ וְהָא דָשֵׁי עִילִיתָא אֲחִידָן וַאֲמַרוּ בְּרַם עָבִיד הוּא יַת צוֹרְכֵיהּ בְּאִדְרוֹן בֵּית קַיְטָא:
וְאוֹרִיכוּ עַד סַגִי וְהָא לֵיתוֹהִי פָּתַח דָשֵׁי עִלִיתָא וּנְסִיבוּ יַת מַפְתְּחָא וּפְתַחוּ וְהָא רִבּוֹנְהוֹן טָרִיף רְמֵי עַל אַרְעָא מִית:
וְאֵהוּד אִשְׁתְּזִיב עַד אִתְעַכְּבֵיהוֹן וְהוּא עֲבַר יַת מַחְצְבַיָא וְאִשְׁתְּזֵיב לִשְׂעִירָתָא:
וַהֲוָה בְּמֵיתוֹהִי וּתְקַע בְּשׁוֹפָרָא בְּטוּרָא דְבֵית אֶפְרָיִם וּנְחָתוּ עִמֵיהּ בְּנֵי יִשְׂרָאֵל מִן טוּרָא וְהוּא קֳדָמֵיהוֹן:
וַאֲמַר לְהוֹן רְדִיפוּ בַתְרַי אֲרֵי מְסַר יְיָ יַת בַּעֲלֵי דְבָבֵיכוֹן יַת מוֹאֲבָאֵי בִּידֵיכוֹן וּנְחָתוּ בַתְרוֹהִי וַאֲחַדוּ יַת מְגִיזַת עָבְרָאֵי יַרְדְנָא עַל מוֹאָב וְלָא שְׁבַקוּ אֱנַשׁ לְמֶעְבַר:
וּקְטַלוּ יַת מוֹאָב בְּעִדָנָא הַהִיא כְּעַסְרָא אַלְפִין גַבְרָא כָּל אֵימְתָן וְכָל גְבַר גִבַּר חֵילָא וְלָא אִישְׁתְּזֵיב אֱנָשׁ:
וְאִתַּבָּרוּ מוֹאֲבָאֵי בְּיוֹמָא הַהוּא תְּחוֹת יַד אֱנַשׁ יִשְׂרָאֵל וּשְׁדוֹכַת אַרְעָא דְיִשְׂרָאֵל תְּמָנָן שְׁנִין:
וּבַתְרוֹהִי הֲוָה שַׁמְגַר בַּר עֲנָת וּקְטַל יַת פְּלִשְׁתָּאֵי שִׁית מְאָה גַבְרִין בִּפְרַשׁ תּוֹרַיָא וּפְרַק אַף הוּא יַת יִשְׂרָאֵל:
