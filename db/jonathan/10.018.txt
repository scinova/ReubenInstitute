וַהֲוָה מִזְמַן יוֹמִין סַגִיאִין וּפִתְגַם נְבוּאָה מִן קֳדָם יְיָ הֲוָה עִם אֵלִיָהוּ בְּשַׁתָּא תְלִיתָאָה לְמֵימַר אֱזֵיל אִתְחַזֵי לְאַחְאָב וְאֶתֵּן מִטְרָא עַל אַפֵּי אַרְעָא:
וַאֲזַל אֵלִיָהוּ לְאִתְחַזָאָה לְאַחְאָב וְכַפְנָא תַקִיף בְּשׁוֹמְרוֹן:
וּקְרָא אַחְאָב לְעוֹבַדְיָהוּ דִי מְמַנָא עַל בֵּיתָא וְעוֹבַדְיָהוּ הֲוָה דָחִיל מִן קֳדָם יְיָ לַחֲדָא:
וַהֲוָה כַּד קְטַלַת אִיזֶבֶל יַת נְבִיאַיָא דַייָ וּדְבַר עוֹבַדְיָהוּ מְאָה גַבְרִין נְבִיאִין וְאַטְמְרִינוּן חַמְשִׁין גַבְרָא בִּמְעַרְתָּא וְסוֹבְרִינוּן בְּלַחְמָא וּבְמַיָא:
וַאֲמַר אַחְאָב לְעוֹבַדְיָהוּ הֲלִיךְ בְּאַרְעָא עַל כָּל מַבּוּעֵי מַיָא וְעַל כָּל נַחְלַיָא מָאִים נִשְׁכַּח עִסְבָּא וּנְקַיֵם סוּסְוָּן וְכוּדַנְוָן וְלָא נִפְסוֹק מִבְּעִירָא:
וּפְּלִיגוּ לְהוֹן יַת אַרְעָא לְמֶעְבַּר בָּהּ אַחְאָב אֲזַל בְּאוֹרְחָא חֲדָא בִּלְחוֹדוֹהִי וְעוֹבַדְיָהוּ אֲזַל בְּאוֹרְחָא חֲדָא בִּלְחוֹדוֹהִי:
וַהֲוָה עוֹבַדְיָהוּ אָזֵיל בְּאוֹרְחָא וְהָא אֵלִיָהוּ לָקֳדָמוּתֵיהּ וְאִשְׁתְּמוֹדָעֵיהּ וּנְפַל עַל אַפּוֹהִי וַאֲמַר הַאַתְּ דֵין רִבּוֹנִי אֵלִיָהוּ:
וַאֲמַר לֵיהּ אֲנָא אֱזֵיל אֲמַר לְרִבּוֹנָךְ הָא אֵלִיָהוּ:
וַאֲמַר מַה חָבִית אֲרֵי אַתְּ מְסַר יַת עַבְדָךְ בִּידָא דְאַחְאָב לְמִקְטְלִי:
קַיָם הוּא יְיָ אֱלָהָךְ אִם אִית עַם וּמַלְכוּ דְלָא שְׁלַח רִבּוֹנִי לְתַמָן לְמִבְעָךְ וַאֲמַרוּ לָא הֲוָה כָּא וּמוֹמֵי יַת מַלְכוּתָא וְיַת עַמָא אֲרֵי לָא מַשְׁכַּח לָךְ:
וּכְעַן אַתְּ אֲמַר אֱזֵיל אֲמַר לְרִבּוֹנָךְ הָא אֵלִיָהוּ:
וִיהֵי עוֹד דַאֲנָא אָזֵיל מִלְוָתָךְ וְרוּחָא מִן קֳדָם יְיָ יִטְלִינָךְ לַאֲתַר דְלָא אִידַע וֶאֱהַךְ לְחַוָאָה לְאַחְאָב וְלָא יִשְׁכְּחִינָךְ וְיִקְטְלִינַנִי וְעַבְדָךְ דָחֵיל מִן קֳדָם יְיָ מִזְעוּרִי:
הֲלָא אִתְחַוָא לְרִבּוֹנִי יַת דַעֲבָדִית כַּד קְטָלַת אִיזֶבֶל יַת נְבִיאַיָא דַייָ וְאַטְמְרֵית מִנְבִיאַיָא דַייָ מְאָה גַבְרִין חַמְשִׁין חַמְשִׁין גְבַר בִּמְעַרְתָּא וְסוֹבַרְתִּינוּן בְּלַחְמָא וּבְמַיָא:
וּכְעַן אַתְּ אָמַר אֱזֵיל אֱמַר לְרִבּוֹנָךְ הָא אֵלִיָהוּ וְיִקְטְלִינַנִי:
וַאֲמַר אֵלִיָהוּ קַיָם הוּא יְיָ צְבָאוֹת דִי מְשַׁמֵשְׁנָא קֳדָמוֹהִי אֲרֵי יוֹמָא דֵין אִתְחַזֵי לֵיהּ:
וַאֲזַל עוֹבַדְיָה לָקָדָמוּת אַחְאָב וְחַוֵי לֵיהּ וַאֲזַל אַחְאָב לָקָדָמוּת אֵלִיָהוּ:
וַהֲוָה כַּד חֲזָא אַחְאָב יַת אֵלִיָהוּ וַאֲמַר אַחְאָב לֵיהּ הַאַתְּ דֵין עָכְרֵיהּ דְיִשְׂרָאֵל:
וַאֲמַר לָא עֲכַרִית יַת יִשְׂרָאֵל אֱלָהִין אַתְּ וּבֵית אֲבוּךְ בִּדְשַׁבַקְתּוּן יַת פִּקוּדַיָא דַייָ וַאֲזַלְתּוּן בָּתַר בְּעָלַיָא:
וּכְעַן שְׁלַח כְּנוֹשׁ לְוָתִי יַת כָּל יִשְׂרָאֵל לְטוּר כַּרְמְלָא וְיַת נְבִיֵי בַּעְלָא אַרְבַּע מְאָה וְחַמְשִׁין וּנְבִיֵי אֲשֵׁרָתָא אַרְבַּע מְאָה אָכְלֵי פְּתוֹרָא דְאִיזֶבֶל:
וּשְׁלַח אַחְאָב בְּכָל בְּנֵי יִשְׂרָאֵל וּכְנַשׁ יַת נְבִיֵי שִׁקְרָא לְטוּר כַּרְמֶלָא:
וּקְרִיב אֵלִיָהוּ לְוַת כָּל עַמָא וַאֲמַר עַד אֵימָתַי אַתּוּן פְּלִיגִין לִתְרֵין פַּלְגְוָן הֲלָא יְיָ אֱלֹהִים פְלָחוּ קֳדָמוֹהִי בִּלְחוֹדוֹהִי וּלְמָא אַתּוּן טָעָן בָּתַר בַּעֲלָא דְלֵית בֵּיהּ צְרוֹךְ וְלָא אֲתִיבוּ עַמָא יָתֵיהּ פִּתְגָם:
וַאֲמַר אֵלִיָהוּ לְעַמָא אֲנָא אִשְׁתָּאֲרִית נְבִיָא קֳדָם יְיָ בִּלְחוֹדִי וּנְבִיֵי בַעֲלָא אַרְבַּע מְאָה וְחַמְשִׁין גַבְרָא:
וְיִתְיַהֲבוּן לָנָא תְּרֵין תּוֹרִין וְיִבְחֲרוּן לְהוֹן תּוֹרָא חַד וִיהַדְמוּנֵיהּ וִישַׁווּן עַל אָעַיָא וְאֶשְׁתָּא לָא יְשַׁווּן וַאֲנָא אַעְבֵּיר יַת תּוֹרָא חַד וְאֶתֵּן עַל אָעַיָא וְאֶשְׁתָּא לָא אֲשַׁוֵי:
וְתִקְרוּן בְּשׁוּם טַעֲוַתְכוֹן וְלָא תִתְעַנוּן מִן קֳדָם דְלֵית בֵּיהּ צְרוֹךְ בְּרַם אֲנָא אֲצַלֵי בִשְׁמָא דַייָ וִישַׁלַח מֵימְרֵיהּ וִיחוֹת אֶשְׁתָּא אֲרֵי יְיָ הוּא אֱלֹהִים וַאֲתֵיב כָּל עַמָא וַאֲמַרוּ תַּקִין פִּתְגָמָא:
וַאֲמַר אֵלִיָהוּ לִנְבִיֵי בַעֲלָא בְחָרוּ לְכוֹן תּוֹרָא חָד וַעֲבִידוּ קֳדָמֵיתָא אֲרֵי אַתּוּן סַגִיאִין וּקְרוֹ בְּשׁוּם טַעֲוַתְכוֹן וְאֶשְׁתָּא לָא תְשַׁווּן:
וּנְסִיבוּ יַת תּוֹרָא דִיהַב לְהוֹן וַעֲבַדוּ וּקְרוֹ בְּשׁוּם בַּעֲלָא מִצַפְרָא וְעַד עִדָן טִיהֲרָא לְמֵימַר בַּעֲלָא עֲנֵינָא וְלֵית קַל וְלֵית דַעֲנֵי וְאִשְׁתַּטוּ עַל אֱגוֹרָא דַעֲבָד:
וַהֲוָה בְּעִדָן טִיהֲרָא וְחַיָךְ בְּהוֹן אֵלִיָהוּ וַאֲמַר קְרוֹ בְקַל רַב אֲרֵי אַתּוּן אָמְרִין דַחֲלָא הוּא דִלְמָא שׁוּעִי לֵיהּ אוֹ אִשְׁתְּדָפָא אִשְׁתְּדִיף אוֹ אוֹרַח הֲוַת לֵיהּ אוֹ דִלְמָא דְמוּךְ הוּא וְיִתְעַר:
וּקְרוֹ בְּקַל רַב וְאִתְחַמָמוּ כְּנִימוּסֵיהוֹן בְּסַיְפִין וּבְרוֹמְחִין עַד דְאִשְׁתַּפִיךְ דְמָא עֲלֵיהוֹן:
וַהֲוָה כְּמֶעְבַּר טִיהֲרָא וְאִשְׁתַּטִיאוּ עַד לְמִיסַק מִנְחָתָא וְלֵית קַל וְלֵית דְעָנֵי וְלֵית דְמַצִית:
וַאֲמַר אֵלִיָהוּ לְכָל עַמָא אִתְקְרִיבוּ לְוָתִי וּקְרִיבוּ כָל עַמָא לְוָתֵיהּ וּבְנָא יַת מַדְבְּחָא דַייָ דַהֲוָה מְפַגָר:
וּנְסִיב אֵלִיָהוּ תַּרְתָּא עַסְרֵי אַבְנַיָא לְמִנְיַן שִׁבְטֵי בְנֵי יַעֲקֹב דִי הֲוָה פִתְגָמָא דַייָ עִמֵיהּ לְמֵימַר יִשְׂרָאֵל יְהֵי שְׁמָךְ:
וּבְנָא יַת אַבְנַיָא מַדְבְּחָא בִּשְׁמָא דַייָ וַעֲבַד מְזִיקָתָא כְּבֵית סָאתָן זַרְעָא סְחוֹר סְחוֹר לְמַדְבְּחָא:
וְסַדֵר יַת אָעַיָא וַהֲדֵים יַת תּוֹרָא וְשַׁוֵי עַל אָעַיָא:
וַאֲמַר מְלוּ אַרְבַּע כַּדִין מַיָא וַאֲרִיקוּ עַל עֲלָתָא וְעַל אָעַיָא וַאֲמַר תְּנוֹ וּתְנוֹ וַאֲמַר תַּלִיתוּ וּתְלִיתוּ:
וַאֲזַלוּ מַיָא סְחוֹר סְחוֹר לְמַדְבְּחָא וְאַף יַת מְזֵיקָתָא מְלֵי מַיָא:
וַהֲוָה בְּמִיסַק מִנְחָתָא וּקְרִיב אֵלִיָהוּ נְבִיָא וַאֲמַר יְיָ אֱלָהֵיהּ דְאַבְרָהָם יִצְחָק וְיִשְׂרָאֵל יוֹמָא דֵין יִתְיְדַע אֲרֵי אַתְּ יְיָ דִשְׁכִנְתָּךְ שַׁרְיָא בְיִשְׂרָאֵל וַאֲַנָא עַבְדָךְ וּבְפִתְגָמָךְ עֲבָדִית יַת כָּל פִּתְגָמַיָא הָאִילֵין:
קַבֵּל צְלוֹתִי יְיָ בְּאֶשְׁתָּא קַבֵּל צְלוֹתִי יְיָ בְּמִטְרָא וְיֵדְעוּן עַמָא הָדֵין בְּמֶעְבְּדָךְ לְהוֹן נִסָא אֲרֵי אַתְּ יְיָ אֱלֹהִים וְאַתְּ בִּרְחֵימְתָךְ יַתְהוֹן מִשְׁתְּאֵיל לְהוֹן בְּמֵימְרָךְ לַאֲתָבוּתְהוֹן לִדְחַלְתָּךְ וְאִינוּן יְהָבוּ יַת לִבְּהוֹן פְּלִיג:
וּנְפָלַת אֶשְׁתָּא מִן קֳדָם יְיָ וַאֲכָלַת יַת עֲלָתָא וְיַת אָעַיָא וְיַת אַבְנַיָא וְיַת עַפְרָא וְיַת מַיָא דְמִבְזִיקָתָא לְחֵיכַת:
וַחֲזוֹ כָל עַמָא וּנְפַלוּ עַל אֲפֵּיהוֹן וַאֲמַרוּ יְיָ הוּא אֱלֹהִים יְיָ הוּא אֱלֹהִים:
וַאֲמַר אֵלִיָהוּ לְהוֹן אֲחוּדוּ יַת נְבִיֵי בַעֲלָא גְבַר אַל יִשְׁתֵּיזֵיב מִנְהוֹן וְאַחְדוּנוּן וַאֲחָתִינוּן אֵלִיָהוּ לְנַחְלָא דְקִישׁוֹן וְנַכְסִינוּן תַּמָן:
וַאֲמַר אֵלִיָהוּ לְאַחְאָב סַק אֱכוֹל וּשְׁתֵי אֲרֵי קַל אִתְרְגוּשַׁת מִטְרָא:
וּסְלֵיק אַחְאָב לְמֵיכַל וּלְמִשְׁתֵּי וְאֵלִיָהוּ סְלִיק לְרֵישׁ טוּרָא דְכַרְמֶלָא וּנְחִין לְאַרְעָא וְשַׁוֵי אַפּוֹהִי בֵּין בִּרְכוֹהִי:
וַאֲמַר לְעוּלֵימֵיהּ סַק כְּעַן סְכִי לְאוֹרַח מַעֲרָבָא וּסְלִיק וּסְכָא וַאֲמַר לֵית מִדַעַם וַאֲמַר תּוּב שְׁבַע זִמְנִין:
וַהֲוָה בְּזִמְנָא שְׁבִיעֵיתָא וַאֲמַר הָא עֲנָנָא סְלִיק זְעֵירָא כְּפִיסַת יַד גְבַר סָלֵיק מִמְעַרְבָא וַאֲמַר סַק אֱמַר לְאַחְאָב אִזְדָרַז וְחוּת וְלָא יְעַכְּבִינָךְ מַטְרָא:
וַהֲוָה עַד דְמִזְדָרַז וְנָחִית וּשְׁמַיָא אִתְחַפִּיאוּ עַנָנָא וְרוּחָא וַהֲוָה מְטַר סַגִי וּרְכִיב אַחְאָב וַאֲזַל לְיִזְרְעֶאל:
וְרוּחַ גְבוּרָא מִן קֳדָם יְיָ הֲוַת עִם אֵלִיָהוּ וְזָרֵיז חַרְצֵיהּ וּרְהַט קֳדָם אַחְאָב עַד דַאֲתָא לְיִזְרְעֶאל:
