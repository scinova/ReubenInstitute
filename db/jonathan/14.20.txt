וַהֲוָה בְשַׁתָּא שְׁבִיעֲתָא בְּחַמְשָׁא בְּעַסְרָא לְיַרְחָא אֲמַר נְבִיָא אֲתוֹ גַבְרִין מִסָבֵי יִשְׂרָאֵל לְמִתְבַּע אוּלְפַן מִן קֳדָם יְיָ וִיתִיבוּ קֳדָמַי:
וַהֲוָה פִתְגַם נְבוּאָה מִן קֳדָם יְיָ עִמִי לְמֵימָר:
בַּר אָדָם אִתְנַבֵּי לְסָבֵי יִשְׂרָאֵל וְתֵימַר לְהוֹן כִּדְנַן אֲמַר יְיָ אֱלֹהִים הַלְמִתְבַּע אוּלְפַן מִן קֳדָמַי אַתּוּן אָתָן קַיָם אֲנָא אִם אֶשְׁתְּאֵיל לְכוֹן בְּמֵימְרִי אֲמַר יְיָ אֱלֹהִים:
הֲתוֹכַח יַתְהוֹן הֲתוֹכַח בַּר אָדָם יַת תּוֹעֲבַת אֲבָהַתְהוֹן הוֹדָעִינוּן:
וְתֵימַר לְהוֹן כִּדְנַן אֲמַר יְיָ אֱלֹהִים בְּיוֹמָא דְאִתְרְעֵיתִי בְיִשְׂרָאֵל וְקַיְמֵית לְהוֹן בְּמֵימְרִי לְזַרְעָא דְבֵית יַעֲקֹב וְאִתְגְלֵיתִי לְמִפְרַקְהוֹן בְּאַרְעָא דְמִצְרַיִם וְקַיְמֵית בְּמֵימְרִי לְהוֹן לְמֵימָר אֲנָא יְיָ אֱלָֹהָכוֹן:
בְּיוֹמָא הַהוּא קַיְמִית לְהוֹן בְּמֵימְרִי לְאַפָּקוּתְהוֹן מֵאַרְעָא דְמִצְרַיִם לְאַרְעָא דִיהָבֵית לְהוֹן עַבְדָא חֲלַב וּדְבָשׁ תּוּשְׁבְּחָא הִיא לְכָל מְדִינָתָא:
וַאֲמָרֵית לְהוֹן גְבַר יַת שִׁקוּצוֹהִי אַרְחִיקוּ מִקֳבֵיל עֵינוֹהִי וּבְפוּלְחַן טַעֲוַת מִצְרָאֵי לָא תִּסְתַּאֲבוּן אֲנָא יְיָ אֱלָהָכוֹן:
וְסָרִיבוּ עַל מֵימְרִי וְלָא אָבוּ לְקַבָּלָא לִנְבִיֵי גְבַר יַת שִׁקוּצֵי עֵינֵיהוֹן לָא אַרְחִיקוּ וְיַת פּוּלְחַן טַעֲוַת מִצְרָאֵי לָא שְׁבָקוּ וַאֲמָרֵית לְמִשְׁפַּךְ חֵימְתִי עֲלֵיהוֹן דְיָחוּל רוּגְזִי בְּהוֹן בְּגוֹ אַרְעָא דְמִצְרָיִם:
וַעֲבָדֵית בְּדִיל שְׁמִי בְּדִיל דְלָא יִתְחַל לְעֵינֵי עַמְמַיָא דִי אִינוּן בֵּינֵיהוֹן דְאִתְגְלֵיתִי לְמִפְרַקְהוֹן לְעֵינֵיהוֹן לְאַפָּקוּתְהוֹן מֵאַרְעָא דְמִצְרָיִם:
וְאַפֵּיקְתִּינוּן מֵאַרְעָא דְמִצְרָיִם וְאָעֵלְתִּינוּן לְמַדְבְּרָא:
וִיהָבֵית לְהוֹן יַת קְיָמַי וְיַת דִינַי הוֹדְעֵית יַתְהוֹן דְאִם יַעְבֵּיד יַתְהוֹן אֱנָשָׁא וִיחֵי בְהוֹן בְּחַיֵי עַלְמָא:
וְאַף יַת יוֹמֵי שַׁבַּיָא דִילִי יְהָבֵית לְהוֹן לְמֶהֱוֵי לְאָת בֵּין מֵימְרִי וּבֵינֵיהוֹן לְמֵדַע אֲרֵי אֲנָא יְיָ מְקַדְשֵׁיהוֹן:
וְסָרִיבוּ עַל מֵימְרִי בֵּית יִשְׂרָאֵל בְּמַדְבְּרָא בִּקְיָמַי לָא הֲלִיכוּ וְיַת דִינַי קָצוּ דְאִם יַעְבֵּיד יַתְהוֹן אֱנָשָׁא וְיֵיחֵי בְהוֹן חַיֵי עַלְמָא וְיַת יוֹמֵי שַׁבַּיָא דִילִי אֲחִילוּ לַחֲדָא וַאֲמָרֵית לְמִשְׁפַּךְ חֲמָתִי עֲלֵיהוֹן בְּמַדְבְּרָא לְשֵׁיצָיוּתְהוֹן:
וַעֲבָדֵית בְּדִיל שְׁמִי בְּדִיל דְלָא יִתְחַל לְעֵינֵי עַמְמַיָא דְאַפֵיקְתִּינוּן לְעֵינֵיהוֹן:
וְאַף אֲנָא קַיְמֵת בְּמֵימְרִי לְהוֹן בְּמַדְבְּרָא בְּדִיל דְלָא לְאָעֲלָא יַתְהוֹן לְאַרְעָא דִי יְהָבֵית עָבְדָא חֲלַב וּדְבַשׁ תּוּשְׁבְּחָא הִיא לְכָל מְדִינָתָא:
חֲלַף דִי בְדִינַי קָצוּ וְיַת קְיָמַי לָא הַלִיכוּ בְּהוֹן וְיַת יוֹמֵי שַׁבַּיָא דִילִי אֲחִילוּ אֲרֵי בָתַר פוּלְחַן טַעֲוָתְהוֹן לִבְּהוֹן טָעֵי:
וְחָס מֵימְרִי עֲלֵיהוֹן מִלְחַבְּלוּתְהוֹן וְלָא עֲבָדִית עִמְהוֹן גְמִירָא בְּמַדְבְּרָא:
וַאֲמָרֵית לִבְנֵיהוֹן בְּמַדְבְּרָא בִּגְזֵירַת אֲבָהַתְכוֹן לָא תְהָכוּן וְיַת נִימוּסֵיהוֹן לָא תִטְרוּן וּבְפוּלְחַן טַעֲוָתְהוֹן לָא תִסְתַּאֲבוּן:
אֲנָא יְיָ אֱלָהָכוֹן בִּקְיָמֵי הֲלִיכוּ וְיַת דִינֵי טָרוּ וַעֲבִידוּ יַתְהוֹן:
וְיַת יוֹמֵי שַׁבַּיָא דִילִי קַדִישׁוּ וִיהוֹן לְאָת בֵּין מֵימְרִי וּבֵינֵיכוֹן לְמֵידַע אֲרֵי אֲנָא יְיָ אֱלָהָכוֹן:
וְסָרִיבוּ עַל מֵימְרִי בְּנַיָא בִּקְיָמַי לָא הֲלִיכוּ וְיַת דִינַי לָא נְטָרוּ לְמֶעְבַּד יַתְהוֹן דְאִם יַעְבֵּיד יַתְהוֹן אֱנָשָׁא וִיחֵי בְהוֹן בְחַיֵי עַלְמָא יַת יוֹמֵי שַׁבַּיָא דִילִי אֲחִילוּ וַאֲמָרֵית לְמִשְׁפַּךְ חֵימְתִי עֲלֵיהוֹן דְיָחוּל רוּגְזִי בְהוֹן בְּמַדְבְּרָא:
וַאֲתֵיב יַת מְחַת גְבוּרְתִּי וַעֲבָדֵית בְּדִיל שְׁמִי בְּדִיל דְלָא יִתְחַל לְעֵינֵי עַמְמַיָא דְאַפְקֵית יַתְהוֹן לְעֵינֵיהוֹן:
אַף אֲנָא קַיְמֵית בְּמֵימְרִי לְהוֹן בְּמַדְבְּרָא לְאַגְלָאָה יַתְהוֹן לְבֵינֵי עַמְמַיָא וּלְבַדָרָא יַתְהוֹן בִּמְדִינָתָא:
חֲלַף דְדִינֵי לָא עֲבָדוּ וּבִקְיָמַי קָצוּ וְיַת יוֹמֵי שַׁבַּיָא דִילִי אֲחִילוּ וּבָתַר טַעֲוַת אֲבָהַתְהוֹן הֲוָאָה עֵינֵיהוֹן:
וְאַף אֲנָא מִדְמָרְדוּ בְמֵימְרֵי וְלָא אָבוּ לְקַבָּלָא לִנְבִיאַי אַרְחֵיקְתִּינוּן וּמְסַרְתִּינוּן בְּיַד שַׂנְאֵיהוֹן וּבָתַר יִצְרֵיהוֹן טִפְּשָׁא אֲזָלוּ וַעֲבָדוּ גְזֵירָן דְלָא תַקְנָן וְנִימוּסִין דְלָא אִתְקַיֵמְתּוּן בְּהוֹן:
וְסָאִיבֵית יַתְהוֹן בְּמַתְּנָתְהוֹן בְּאַעֲבְּרָא כָּל פְּתַח וַלְדָא בְדִיל דְאַצְדִינוּן בְּדִיל דְיֵדְעוּן דִי אֲנָא יְיָ:
בְּכֵן אִתְנַבֵּי לְבֵית יִשְׂרָאֵל בַּר אָדָם וְתֵימַר לְהוֹן כִּדְנַן אֲמַר יְיָ אֱלֹהִים עוֹד דָא אַרְגִיזוּ קָדְמַי אֲבָהָתְכוֹן בִּדְשַׁקָרוּ בְּמֵימְרִי שְׁקַר:
וְאָעֵילְתִּינוּן לְאַרְעָא דְקַיֵמִית בְּמֵימְרִי לְמִתַּן יָתָהּ לְהוֹן וַחֲזוֹ כָּל רָמָא מְנַטְלָא וְכָל אִילַן עַבּוּף וּנְכִיסוּ תַמָן אַרְגָזָא קוּרְבָּנֵיהוֹן וְשַׁוִיאוּ תַמָן פּוּלְחַן דִבְחֵיהוֹן וּנְסִיכוּ תַמָן יַת נִכְסֵיהוֹן:
וַאֲמָרֵית לְהוֹן מָא בָמָתָא דְאַתּוּן אָתָן לְאִשְׁתְּטָאָה תַמָן וְאִתְקְרִי שְׁמָהּ בָּמָתָא עַד יוֹמָא הָדֵין:
בְּכֵן אֲמַר לְבֵית יִשְׂרָאֵל כִּדְנַן אֲמַר יְיָ אֱלֹהִים הַבְּאוֹרַח אֲבָהַתְכוֹן אַתּוּן מִסְתַּאֲבִין וּבָתַר שִׁקוּצֵיהוֹן אַתּוּן טָעָן:
וּבְמַטַל מַתְּנָתֵיכוֹן בְּאַעֲבָרָא בְּנֵיכוֹן בְּנוּרָא אַתּוּן מִסְתַּאֲבִין לְכָל פּוּלְחַן טַעֲוָתְכוֹן עַד יוֹמָא דֵין וַאֲנָא אֶשְׁתָּאֵיל לְכוֹן בְּמֵימְרִי בֵּית יִשְׂרָאֵל קַיָם אֲנָא אֲמַר יְיָ אֱלֹהִים אִם אֶשְׁתְּאֵין לְכוֹן בְּמֵימְרִי:
וּדְאַתּוּן מְהַרְהֲרִין וְדִסְלִיקַת עַל לִבְּכוֹן קֳדָמַי גַלְיָן דְאַתּוּן אָמְרִין נְהֵי כְעַמְמַיָא כְּזַרְעֲיַת מְדִינָתָא לְשַׁמְשָׁא אָעָא וְאַבְנָא:
קַיָם אֲנָא אֲמַר יְיָ אֱלֹהִים אִם לָא בְּיַד תַּקִיפָא וּבְדַרְעָא מְרוֹמָם וּבִרְגַז שְׁפִיךְ אֶמְלוֹךְ עֲלֵיכוֹן:
וְאַפֵּיק יַתְכוֹן מִבֵּינֵי עַמְמַיָא וְאַכְנֵישׁ יַתְכוֹן מִן מְדִינָתָא דְאִתְבַּדַרְתּוּן בְּהוֹן בְּיַד תַּקִיפָא וּבְדַרְעָא מְרוֹמָם וּבִרְגַז שְׁפִיךְ:
וְאָעֵיל יַתְכוֹן לְמִדְבַּר עַמְמַיָא וְאִתְפְּרַע מִנְכוֹן תַּמָן אַפִּין בְּאַפִּין:
כְּמָא דְאִתְפְּרָעֵית מִן אֲבָהַתְכוֹן בְּמַדְבְּרָא אַרְעָא דְמִצְרָיִם כֵּן אִתְפְּרַע מִנְכוֹן אֲמַר יְיָ אֱלֹהִים:
וַאֲנִיחַ עֲלֵיכוֹן גְזֵירַת דִינִי וְאָעֵיל יַתְכוֹן בְּמַסוֹרֶת קַיָמָא:
וְאַפְרִישׁ מִנְכוֹן דִי מְרָדוּ וְדִי שַׁקָרוּ בְמֵימְרִי מֵאַרְעָא תוֹתְבוּתְהוֹן אַפֵיק יַתְהוֹן וּלְאַרְעָא דְיִשְׂרָאֵל לָא יַעֲלוּן וְתֵידְעוּן אֲרֵי אֲנָא יְיָ:
וְאַתּוּן בֵּית יִשְׂרָאֵל כִּדְנַן אֲמַר יְיָ אֱלֹהִים גְבַר פּוּלְחַן טַעֲוָתֵהּ אֱזִילוּ פְלָחוּ וּבָתַר דְתַבְתּוּן לֵיתֵיכוֹן צָבָן לְקַבָּלָא לְמֵימְרִי וְיַת שְׁמָא דְקוּדְשִׁי לָא תַחֲלוּן עוֹד בְּמַתְּנָתֵכוֹן וּבְטַעֲוָתְכוֹן:
אֲרֵי בְטוּר קוּדְשִׁי בְּטוּר קוּדְשָׁא דְיִשְׂרָאֵל אֲמַר יְיָ אֱלֹהִים תַּמָן יִפְלְחוּן קֳדָמַי כָּל בֵּית יִשְׂרָאֵל כּוּלְהוֹן בְּאַרְעָא חַיַיָא תַּמָן יִתְקַבֵּל קוּרְבָּנֵיהוֹן בִּפְרִיעַ וְתַמָן אֶתְבַּע יַת אַפְרָשׁוּתְכוֹן וְיַת רֵישׁ אַצְוָתְכוֹן בְּכָל קוּדְשֵׁיכוֹן:
קוּרְבַּן לְרַעֲוָא יִתְקַבֵּל קוּרְבַּנְכוֹן בִּפְרִיעַ כַּד אַפֵּיק יַתְכוֹן מִבֵּינֵי עַמְמַיָא וְאַכְנֵישׁ יַתְכוֹן מִן מְדִינָתָא דְאִתְבַּדַרְתּוּן בְּהָן וְאֶתְקַדֵשׁ בְּכוֹן לְעֵינֵי עַמְמַיָא:
וְתֵידְעוּן אֲרֵי אֲנָא יְיָ כַּד אָעֵיל יַתְכוֹן לְאַרְעָא דְיִשְׂרָאֵל דִי קַיֵמִית בְּמֵימְרִי לְמִתַּן יָתָהּ לַאֲבָהָתְכוֹן:
וְתִדְכְּרוּן תַּמָן יַת אוֹרְחָתְכוֹן וְיַת כָּל עוֹבָדֵיכוֹן דְאִסְתַּאַבְתּוּן בְּהוֹן וְתִידַנְקוּן וּתְהוֹן חָזָן בְּכָל בִּישַׁתְכוֹן דַעֲבַדְתּוּן:
וְתֵידְעוּן אֲרֵי אֲנָא יְיָ כַּד אַעְבֵּיד עִמְכוֹן בְּדִיל שְׁמִי לָא כְאוֹרְחָתְכוֹן בִּישְׁתָא וּכְעוֹבָדֵיכוֹן מְחַבְּלַיָא בֵּית יִשְׂרָאֵל אֲמַר יְיָ אֱלֹהִים:
