וּשְׁלֹמֹה הֲוָה שַׁלִיט בְּכָל מַלְכוּתָא מִן נְהַר פְּרָת עַד אֲרַע פְלִשְׁתָּאֵי וְעַד תְּחוּם מִצְרָיִם מְקַרְבִין קֻרְבָּנָא וּפַלְחִין יַת שְׁלֹמֹה כָּל יוֹמֵי חַיוֹהִי:
וַהֲוָה לַחֲמָא דִשְׁלֹמֹה לְיוֹמָא חָד תְּלָתִין כֹּרִין סֻלְתָּא וְשִׁתִּין כֹּרִין קִמְחָא:
עַסְרָא תוֹרִין דְפִטְמָא וְעַסְרִין תּוֹרִין דְרַעֲיָא וּמְאָה עָנָא בַּר מֵאַיָלִין וְטַבְיָן וְיַחְמוֹרִין וְעוֹפַיָא פַּטִימַיָא:
אֲרֵי הוּא מַפְלַח בְּכָל עִבַּר פְּרָת מִתִּפְסַח וְעַד עַזָה בְּכָל מַלְכֵי עִבַּר פְּרָת וּשְׁלָמָא הֲוָה לֵיהּ מִכָּל עַבְרוֹהִי מִסְחוֹר סְחוֹר:
וִיתֵיב יְהוּדָה וְיִשְׂרָאֵל לְרוֹחֲצָן גְבַר תְּחוֹת פֵּרֵי גוּפְנוֹהִי וּתְחוֹת פֵּרֵי תְאֵנוֹהִי מִדָן וְעַד בְּאֵר שָׁבַע כֹּל יוֹמֵי שְׁלֹמֹה:
וַהֲווֹ לִשְׁלֹמֹה אַרְבְּעִין אַלְפִין אֻרְוַת סוּסַוָן לִרְתִיכוֹהִי וּתְרֵי עֲסַר אַלְפִין פָּרָשִׁין:
וּמְסוֹבְרִין אִסְטַרְטִיגַיָא הָאִלֵין יַת מַלְכָּא שְׁלֹמֹה וְיַת כָּל דְקָרִיב לְפָתוֹרָא דְמַלְכָּא שְׁלֹמֹה גְבַר יַרְחֵיהּ לָא מְעַכְּבִין מִדָעַם:
וּסְעוֹרַיָא וְתִבְנָא לְסוּסְוָתָא וּלְרַכְשָׁא מַיְתָן לְאַתְרָא דִיהֵי תַמָן גְבַר כְּהִלְכְתֵיהּ:
וִיהַב יְיָ חוּכְמְתָא לִשְׁלֹמֹה וְסוּכְלְתָנוּתָא סַגִי לַחֲדָא וּפַתְיוּת לִבָּא כְּחָלָא דְעַל כַּף יַמָא:
וּסְגִיאַת חוּכְמַת שְׁלֹמֹה מֵחוּכְמַת כָּל בְּנֵי מַדִינְחָא וּמִכֹּל חוּכְמַת מִצְרָיִם:
וַחֲכָם מִכָּל אֱנָשָׁא מֵאֵיתָן בַּר זָרַח וְהֵימָן וְכַלְכֹּל וְדַרְדַע בְּנֵי מָחוֹל וַהֲוָה שְׁמֵיהּ סַגִי בְכָל עַמְמַיָא סְחוֹר סְחוֹר:
וּמַלֵל תְּלָתָא אַלְפִין מַתְלִין וַהֲוָאָה תוּשְׁבְּחָתֵיהּ אֲלַף וַחֲמֵשׁ:
וְאִתְנַבֵּי עַל מַלְכֵי בֵית דָוִד דַעֲתִידִין לְמִשְׁלַט בְּעַלְמָא הָדֵין וּבְעַלְמָא דְאָתֵי דִמְשִׁיחָא וְאִתְנַבֵּי עַל בְּעִירָא וְעַל עוֹפָא וְעַל רִיחֲשָׁא וְעַל נוּנַיָא:
וַאֲתוֹ מִכָּל עַמְמַיָא לְמִשְׁמַע יַת חוּכְמַת שְׁלֹמֹה מִכָּל מַלְכֵי אַרְעָא דְשָׁמְעִין יַת חוּכְמְתֵיהּ:
וּשְׁלַח חִירָם מַלְכָּא דְצוֹר יַת עַבְדוֹהִי לְוַת שְׁלֹמֹה אֲרֵי שְׁמַע אֲרֵי יָתֵיהּ מְשַׁחוּ לְמֶהֱוֵי מַלְכָּא תְּחוֹת אֲבוּהִי אֲרֵי רְחִימָא הֲוָה חִירָם לְדָוִד כָּל יוֹמַיָא:
וּשְׁלַח שְׁלֹמֹה לְוַת חִירָם לְמֵימָר:
אַתְּ יְדַעְתָּ יַת דָוִד אַבָּא אֲרֵי לָא יָכִיל לְמִבְנֵי בֵיתָא לִשְׁמָא דַייָ אֱלָהֵיהּ מִן קֳדָם עַבְדֵי קְרָבָא דְאַקְפוֹהִי עַד דִמְסַר יְיָ יַתְהוֹן תְּחוֹת פַּרְסַת רִגְלוֹהִי:
וּכְעַן אַנִיחַ יְיָ אֱלָהִי לִי מִסְחוֹר סְחוֹר לֵית שָׂטָן וְלֵית פְּגַע בִּישׁ:
וְהָא אֲנָא אֲמַר לְמִבְנֵי בֵיתָא לִשְׁמָא דַייָ אֱלָהִי כְּמָא דְמַלֵיל יְיָ עִם דָוִד אַבָּא לְמֵימַר בְּרִיךְ דְאֶתֵּן תְּחוֹתָךְ עַל כּוּרְסֵי מַלְכוּתָךְ הוּא יִבְנֵי בֵיתָא לִשְׁמִי:
וּכְעַן פַּקֵיד וִיקַצְצוּן לִי אַרְזִין מִן לִבְנָן וְעַבְדַי יְהוֹן עִם עַבְדָיִךְ וַאֲגַר עַבְדָיִךְ אֶתֵּן לָךְ כְּכֹל דְתֵימַר אֲרֵי אַתְּ יְדַעְתָּ אֲרֵי לֵית בָּנָא גַבְרָא דִידַע לְמִקַץ אָעִין כְּצִידוֹנָאֵי:
וַהֲוָה כַּד שְׁמַע חִירָם יַת פִּתְגָמֵי דִשְׁלֹמֹה וַחֲדִי לַחֲדָא וַאֲמַר בְּרִיךְ יְיָ יוֹמָא דֵין דִיהַב לְדָוִד בַּר חַכִּים עַל עַמָא סַגִיאָה הָדֵין:
וּשְׁלַח חִירָם לְוַת שְׁלֹמֹה לְמֵימַר שְׁמָעִית יַת דִשְׁלַחְתָּא לְוָתִי אֲנָא אֲסַפֵּיק יַת כָּל צוֹרְכָךְ בַּאֲעֵי אַרְזִין וּבַאֲעֵי בִירְוָן:
עַבְדַי יֵיחֲתוּן מִן לִבְנָן לְיַמָא וַאֲנָא אֲשַׁוִינוּן תּוּרְנָסִין בְּיַמָא עַד אַתְרָא דְתִשְׁלַח לִי וְאֶרְמִינוּן תַּמָן וְאַתְּ תִּטוֹל וְאַתְּ תְּסַפֵּיק צוֹרְכָי לְמִתַּן מָזוֹן לֶאֱנַשׁ בֵּיתִי:
וַהֲוָה חִירָם יָהֵיב לִשְׁלֹמֹה אֲעֵי אַרְזִין וַאֲעֵי בִּירְוָן כָּל צוֹרְכֵיהּ:
וּשְׁלֹמֹה יָהֵיב לְחִירָם עַסְרִין אַלְפִין כּוֹרִין חִיטִין פַּרְנוּס לֶאֱנַשׁ בֵּיתֵיהּ וְעַסְרִין כּוֹרִין מִשְׁחָא כְּתִישָׁא כְּדֵין יְהֵיב שְׁלֹמֹה לְחִירָם שְׁנָא בִּשְׁנָא:
וַיָי יְהֵב חוּכְמְתָא לִשְׁלֹמֹה כְּמָא דְמַלֵיל לֵיהּ וַהֲוָה שְׁלָמָא בֵּין חִירָם וּבֵין שְׁלֹמֹה וּגְזָרוּ קְיַם תַּרְוֵיהוֹן:
וּמַנֵי מַלְכָּא שְׁלֹמֹה מַסְקֵי מִסִין מִכָּל יִשְׂרָאֵל וַהֲוָה מַסְקֵי מִסִין תְּלָתִין אַלְפִין גַבְרָא:
וְשַׁלְחִינוּן לְלִבְנָן עַסְרָא אַלְפִין יַרְחָא מִשְׁתַּנַן יַרְחָא הֲווֹ בְּלִבְנָן תְּרֵין יַרְחִין גְבַר בְּבֵיתֵיהּ וַאֲדוֹנִירָם מְמַנָא עַל מַסְקֵי מִסִין:
וַהֲוָה לִשְׁלֹמֹה שַׁבְעִין אַלְפִין דְנַטְלִין בְּכַתְפָא וְתַמְנָן אַלְפִין דְפַסְלִין בְּטוּרָא:
בַּר מֵרַבָּנֵי אִיסְטַרְטִיגַיָא דִמְמַנַן עַל עֱבִידְתָּא דִשְׁלֹמֹה תְּלָתָא אַלְפִין וּתְלַת מְאָה דִמְפַלְחִין בְּעַמָא דְעָבְדִין בַּעֲבִידְתָּא:
וּפַקֵיד מַלְכָּא וַעֲקָרוּ אַבְנִין רַבְרְבָן אַבְנִין יְקָרָן לְשַׁכְלָלָא בֵיתָא אַבְנִין פְּסִילָן:
וּפְסָלוּ אַרְדִיכְלֵי שְׁלֹמֹה וְאַרְדִיכְלֵי חִירוֹם וְאַרְגוּבְלַיָא וְאַתְקִינוּ אֲעַיָא וְאַבְנַיָא לְמִבְנֵי בֵיתָא:
