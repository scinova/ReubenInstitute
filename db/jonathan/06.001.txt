וַהֲוָה בָּתַר דְמִית מֹשֶׁה עַבְדָא דַייָ וַאֲמַר יְיָ לִיהוֹשֻׁעַ בַּר נוּן מְשַׁמְשׁוּנֵיהּ דְמֹשֶׁה לְמֵימָר:
מֹשֶׁה עַבְדִי מִית וּכְעַן קוּם עִיבַּר יַת יַרְדְנָא הָדֵין אַתְּ וְכָל עַמָא הָדֵין לְאַרְעָא דַאֲנָא יְהִיב לְהוֹן לִבְנֵי יִשְׂרָאֵל:
כָּל אַתְרָא דְתִדְרַךְ פַּרְסַת רַגְלְכוֹן בֵּיהּ לְכוֹן יְהַבִתָהּ כְּמָא דְמַלֵילִית עִם מֹשֶׁה:
מִמַדְבְּרָא וּלְבָנוֹנָא הָדֵין וְעַד נַהֲרָא רַבָּא נַהֲרָא דִפְרָת כֹּל אַרַע חִתָּאֵי וְעַד יַמָא רַבָּא מַעֲלָנֵי שִׁמְשָׁא יְהֵא תְחוּמְכוֹן:
לָא יִתְעַתַּד אֱנָשׁ קֳדָמָךְ כָּל יוֹמֵי חַיָיךְ כְּמָא דַהֲוָא מֵימְרִי בְּסַעֲדֵיהּ דְמֹשֶׁה כֵּן אֶהֱוֵי עִמָךְ לָא אַשְׁבְּקִינָךְ וְלָא אַרְחְקִינָךְ:
תְּקֵיף וַעְלַם אֲרֵי אַתְּ תַּחְסֵין יַת עַמָא הָדֵין יַת אַרְעָא דְקַיֵימִית לַאֲבָהַתְהוֹן לְמִתַּן לְהוֹן:
לְחוֹד תְּקֵיף וֶעְלֵם לַחֲדָא לְמִטַר לְמֶעְבַּד כְּכָל אוֹרַיְתָא דְפַקְדָךְ מֹשֶׁה עַבְדִי לָא תִסְטֵי מִינֵיהּ לְיַמִינָא וְלִשְׂמָאלָא בְּדִיל דְתִצְלַח בְּכָל אֲתַר דִתְהָךְ:
לָא יַעְדֵי סִיפְרָא דְאוֹרַיְתָא הָדֵין מִפּוּמָךְ וּתְהֵי הֲגֵי בֵּיהּ יֵימָם וְלֵילְיָא בְּדִיל דְתִטַר לְמֶעְבַּד כְּכָל דִכְתִיב בֵּיהּ אֲרֵי בְּכֵן תַּצְלַח יַת דַרְכָּךְ וּבְכֵן תַּכְשָׁט:
הֲלָא פַקֵידְתָּךְ תְּקֵף וֶעְלֵם לָא תִדְחַל וְלָא תִתְּבַר אֲרֵי בְּסַעֲדָךְ מֵימְרָא דַייָ אֱלָהָךְ בְּכָל אֲתַר דִתְהָךְ:
וּפַקֵיד יְהוֹשֻׁעַ יַת סַרְכֵי עַמָא לְמֵימָר:
עִבָרוּ בְּגוֹ מַשְׁרִיתָא וּפַקִידוּ יַת עַמָא לְמֵימָר אַתְקִינוּ לְכוֹן זְוָדִין אֲרֵי בְּסוֹף תְּלָתָא יוֹמִין אַתּוּן עָבְרִין יַת יַרְדְנָא הָדֵין לְמֵיעַל לְמֵירַת יַת אַרְעָא דַייָ אֱלָהָכוֹן יָהֵיב לְכוֹן לְמִירְתָהּ:
וּלְשֵׁבֶט רְאוּבֵן וּלְשֵׁבֶט גָד וּלְפַלְגוּת שִׁבְטָא דִמְנַשֶׁה אֲמַר יְהוֹשֻׁעַ לְמֵימָר:
הֲווֹ דְכִירִין יַת פִּתְגָמָא דִפַקֵיד יַתְכוֹן מֹשֶׁה עַבְדָא דַייָ לְמֵימָר יְיָ אֱלָהָכוֹן מַנִיחַ לְכוֹן וִיהַב לְכוֹן יַת אַרְעָא הָדָא:
נְשֵׁיכוֹן טַפְלֵיכוֹן וּבְעִירְכוֹן יֵתְבוּן בְּאַרְעָא דִיהַב לְכוֹן מֹשֶׁה בְּעִבְרָא דְיַרְדְנָא וְאַתּוּן תְעָבְרוּן מְזָרְזִין קֳדָם אֲחוּכוֹן כֹּל גִבָּרֵי חֵילָא וּתְסַעֲדוּן יַתְהוֹן:
עַד דִינִיחַ יְיָ לַאֲחוּכוֹן כְּוַתְכוֹן וְיֵירְתוּן אַף אִינוּן יַת אַרְעָא דַייָ אֱלָהָכוֹן יָהֵיב לְכוֹן וּתְתוּבוּן לְאַרַע יְרוּתַתְכוֹן וְתֵירְתוּן יָתָהּ דִיהַב לְכוֹן מֹשֶׁה עַבְדָא דַייָ בְּעִבְרָא דְיַרְדְנָא מַדְנַח שִׁמְשָׁא:
וַאֲתִיבוּ יַת יְהוֹשֻׁעַ לְמֵימָר כֹּל דְפַקֵדְתָּנָא נַעֲבַד וּלְכָל אֲתַר דְתִשְׁלַחְנָנָא נְהַךְ:
כְּכֹל דְקַבֵּלְנָא מִן מֹשֶׁה כֵּן נְקַבֵּל מִנָךְ לְחוֹד יְהֵי מִימְרָא דַייָ אֱלָהָךְ בְּסַעֲדָךְ כְּמָא דַהֲוָה בְּסַעֲדֵיהּ דְמֹשֶׁה:
כָּל גְבַר דִיסַרֵב עַל מֵימְרָךְ וְלָא יְקַבֵל יַת פִּתְגָמָךְ לְכֹל דְתַפְקְדִנֵיהּ יִתְקְטֵיל לְחוֹד תְּקֵף וֶעְלָם:
