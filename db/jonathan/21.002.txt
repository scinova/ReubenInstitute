הָא עַל טוּרֵי אַרְעָא דְיִשְׂרָאֵל רַגְלֵי מְבַשֵׂר מַשְׁמַע שְׁלַם חוֹגִי יְהוּדָה חַגֵךְ שַׁלִימִי נִדְרֵךְ אֲרֵי לָא יוֹסְפוּן עוֹד דְיֶעְדוּן בִּיךְ רַשִׁיעִין כּוּלְהוֹן אֶשְׁתַּצִיאוּ:
דַהֲווֹ סַלְקִין וּמִתְבַּדְרִין עַל אַרְעִיךְ צָיְרִין עֲלָךְ בִּצְיָר מְקַיְמִין סִכְוָאִין עַל אוֹרְחָתָךְ תַּקִיפֵי קְדַל חֲסִינֵי חֵיל לַחֲדָא:
אֲרֵי אָתִיב יְיָ תּוּקְפֵיהּ לְיַעֲקֹב רְבוּתֵיהּ לְיִשְׂרָאֵל אֲרֵי בְזוּנוּן בָּזוֹזִין וְקִרְוֵי תוּשְׁבַּחְתְּהוֹן חַבִּילוּ:
תְּרִיסֵי גִבָּרֵיהוֹן מְסַמְקִין גַבְרֵי עָבְדֵי קְרָבָא לְבִישִׁין דְזָהֲרִיתָא בְּאֶשְׁתָּא פִּילֵי רְתִיכֵיהוֹן זַמִינִין לְיוֹם קִשׁוּטֵיהוֹן וְרַבָּנֵי מַשִׁרְיָתְהוֹן מְסַרְבְּלִין בְּצִבְעוֹנִין:
בְּשׁוּקַיָא מִשְׁתַּגְשְׁגִין רְתִיכַיָא קַל נִקוּשׁ זְיָנֵיהוֹן מִשְׁתְּמַע בִּרְחוֹבֵי קִרְוַיָא חֶזְוֵיהוֹן כִּבְעוּרִין כִּבְרָקִין טְרִידִין:
יְמַנוּן רַבָּנֵי מַשִׁרְיָתְהוֹן יִתְקְלוּן בְּמַהֲכֵיהוֹן יוֹחוּן יִכְבְּשׁוּן שׁוּרָא יִבְנוּן מִגְדָלַיָא:
גִשְׁרֵי נַהֲרְוָתָא אִתְפְּסַקָא וּמַלְכָּא בְּהֵיכְלֵיהּ זָע:
וּמַלְכְּתָא יָתְבַת צִיבָא בְגָלַיָא נַפְקַת וְאַמְהָתָהָא מַדְבְּרָן אָזְלָן בַּתְרָאָה מְנַהֲמָן כְּקַל יוֹנִים מְטַרְפָן עַל לִבֵּיהוֹן:
וְנִינְוֵה כְּבֵית כְּנִישַׁת מַיָא מִיוֹמֵי קֳדָם הִיא וְאִינוּן עֲרָקוּ קוּמוּ קוּמוּ וְלֵית דְמַפְנֵי לְמֵיקָם:
בּוֹזוּ כַסְפָּא בּוֹזוּ דַהֲבָא וְלֵית סוֹף לְאוֹצְרַיָא סָפוּ מִכָּל מָנֵי חֶמְדָתָא:
בְּזִיזָא וּמִתְבַּזְזָא וּפְתִיחַ תַּרְעָא לְסָנְאָה וְלֵב מְסֵי וְנִקוּשׁ רְכוּבִין וְזִיוּעַ בְּכָל חַרְצִין וְאַפֵּי כּוּלְהוֹן אִתְחַפִּיאוּ אִכְּרוּם אוֹכָמִין כְּקִידְרָא:
אָן מְדוֹרֵי מַלְכַיָא וּבֵית מֵישְׁרֵי שִׁלְטוֹנַיָא אֲתַר דַהֲווֹ אָזְלִין מַלְכַיָא תַמָן בְּנֵיהוֹן שָׁבְקִין הָא כְאַרְיָה דְשָׁרֵי עַל צֵידֵיהּ לְרוֹחֲצָן וְלֵית דְמָנִיד:
מַלְכַיָא מַיְתָן חֲטוֹף לִנְשֵׁיהוֹן וּבִזַיָא לִבְנֵיהוֹן וּמְמַלָן אוֹצְרֵיהוֹן חֲטוֹף וּבִרְנְיָתֵיהוֹן בִּזָא:
הָא אֲנָא שְׁלַח רוּגְזִי עֲלָךְ אֲמַר יְיָ צְבָאוֹת וְאַדְלֵיק בְאֶשְׁתָּא רְתִיכָךְ וְכַפְרָנָךְ תִּקְטוֹל חַרְבָּא וֶאֱשֵׁיצֵי מֵאַרְעָא סְחוֹרְתִּיךְ וְלָא יִשְׁתְּמַע עוֹד קַל אִזְגַדָךְ:
