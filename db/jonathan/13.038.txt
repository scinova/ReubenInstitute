וּשְׁמַע שְׁפַטְיָה בַר מַתָּן וּגְדַלְיָהוּ בַר פַשְׁחוּר וְיוּכַל בַּר שֶׁלֶמְיָהוּ וּפַשְׁחוּר בַּר מַלְכִּיָה יַת פִּתְגָמַיָה דְיִרְמְיָה מְמַלֵיל עִם כָּל עַמָא לְמֵימָר:
כִּדְנַן אֲמַר יְיָ דְיָתֵיב בְּקַרְתָּא הָדָא יִתְקְטֵיל בְּחַרְבָּא בְּכַפְנָא וּבְמוֹתָא וּדְיִפּוֹק וְיִשְׁתְּמַע לְכַסְדָאֵי וִיחֵי וּתְהֵי לֵיהּ נַפְשֵׁיהּ לְשֵׁיזָבָא וְיִתְקַיֵם:
כִּדְנַן אֲמַר יְיָ אִתְמְסָרָא תִתְמְסַר קַרְתָּא הָדָא בְּיַד מַשִׁרְיַת מַלְכָּא דְבָבֶל וְיִכְבְּשִׁינָהּ:
וַאֲמַרוּ רַבְרְבַיָא לְמַלְכָּא יִתְקְטֵיל כְּעַן גַבְרָא הָדֵין אֲרֵי עַל כֵּן הוּא מְרַשֵׁל יַת יְדֵי גַבְרֵי עָבְדֵי קְרָבָא דְאִשְׁתָּאֲרוּ בְּקַרְתָּא הָדָא וְיַת יְדֵי כָל עַמָא לְמֵלָלָא עִמְהוֹן כְּפִתְגָמַיָא הָאִלֵין אֲרֵי גַבְרָא הָדֵין לֵיתוֹהִי תָּבַע לִשְׁלָמָא לְעַמָא הָדֵין אֱלָהֵן לְבִישׁוּ:
וַאֲמַר מַלְכָּא צִדְקִיָה הָא הוּא מְסִיר בְּיֶדְכוֹן אֲרֵי לֵית מַלְכָּא יָכִיל לַאֲתָבָא יַתְכוֹן פִּתְגָם:
וְדַבָּרוּ יַת יִרְמְיָה וּרְמוֹ יָתֵיהּ לְגוּבָא דְמַלְכִּיָה בַר מַלְכָּא דִבְדָרַת בֵּית אֲסִירַיָא וְשַׁרְבִּיבוּ יַת יִרְמְיָה בְּאַטוּנַיָא וּבְגוּבָא לֵית מַיָא אֱלָהֵן טִינָא וּטְבַע יִרְמְיָה בְּטִינָא:
וּשְׁמַע עַבְדָא דְמַלְכָּא צִדְקִיָה גְבַר רַב וְהוּא בְּבֵית מַלְכָּא אֲרֵי יְהָבוּ יַת יִרְמְיָה לְגוּבָא וּמַלְכָּא יָתֵיב בִּתְרַע שִׁבְטָא בִנְיָמִין:
וּנְפֵק עַבְדָא דְמַלְכָּא מִבֵּית מַלְכָּא וּמַלֵיל עִם מַלְכָּא לְמֵימָר:
רִבּוֹנִי מַלְכָּא אַבְאִישׁוּ גַבְרַיָא הָאִלֵין יַת כָּל דַעֲבַדוּ לְיִרְמְיָה נְבִיָא יַת דִרְמוֹהִי לְגוּבָא כְּבַר מִית בְּאַתְרֵיהּ מִן קֳדָם כַּפְנָא אֲרֵי לֵית לַחְמָא עוֹד בְּקַרְתָּא:
וּפַקֵיד מַלְכָּא יַת עַבְדָא דְמַלְכָּא צִדְקִיָה לְמֵימָר דְבַר עִמָךְ מִכָּא תְּלָתִין גַבְרִין וְתַסֵיק יַת יִרְמְיָה נְבִיָא מִן גוּבָא עַד לָא יְמוּת:
וּדְבַר עַבְדָא דְמַלְכָּא יַת גַבְרַיָא עִמֵיהּ וְעָל לְבֵית מַלְכָּא לִתְחוֹת אוֹצְרָא וּנְסֵיב מִתַּמָן בְּלָאֵי טַפְסָן וּבְלָאֵי שַׁחְקָן וְשַׁרְבְבִינוּן לְוַת יִרְמְיָה לְגוּבָא בְּאַטוּנַיָא:
וַאֲמַר עַבְדָא דְמַלְכָּא צִדְקִיָה לְיִרְמְיָה שַׁוֵי כְעַן בְּלָאֵי טַפְסָן וְשַׁחְקָן תְּחוֹת (מַדְבְּקֵי) יְדָךְ מִלְרַע לַאֲטוּנַיָא וַעֲבַד יִרְמְיָה כֵּן:
וּנְגָדוּ יַת יִרְמְיָה בְּאַטוּנָא וְאַסִיקוּ יָתֵיהּ מִן גוּבָא וִיתֵיב יִרְמְיָה בְּדָרַת בֵּית אֲסִירַיָא:
וּשְׁלַח מַלְכָּא צִדְקִיָה וּדְבַר יַת יִרְמְיָה נְבִיָא לְוָתֵיהּ לְמַעֲלָנָא תְּלִיתְאָה דִי בְּבֵית מַקְדְשָׁא דַייָ וַאֲמַר מַלְכָּא לְיִרְמְיָה שְׁאַל אַנָא יָתָךְ פִּתְגָמָא לָא תְכַסֵי מִנִי פִּתְגָמָא:
וַאֲמַר יִרְמְיָה לְצִדְקִיָה אֲרֵי אֱחַוֵי לָךְ הֲלָא מִקְטַל תִּקְטְלִינַנִי וַאֲרֵי אַמְלְכִינָךְ לָא תְקַבֵּיל מִנִי:
וְקַיָם מַלְכָּא צִדְקִיָה לְיִרְמְיָה בְּסִתְרָא לְמֵימָר קַיָם הוּא יְיָ יַת דְקַיֵם לָנָא יַת נַפְשָׁתָא הָאִלֵין אִם אֶקְטְלִינָךְ וְאִם אֶמְסְרִינָךְ בְּיַד גַבְרַיָא הָאִלֵין דְבָעָן לְמִקְטְלָךְ:
וַאֲמַר יִרְמְיָה לְצִדְקִיָה כִּדְנַן אֲמַר יְיָ צְבָאוֹת אֱלָהָא דְיִשְׂרָאֵל אִם מִפַּק תִּפּוֹק לְוַת רַבְרְבֵי מַלְכָּא דְבָבֶל וְתִתְקַיֵם נַפְשָׁךְ וְקַרְתָּא הָדָא לָא תִתּוֹקַד בְּנוּרָא וּתְחֵי אַתְּ וֶאֱנַשׁ בֵּיתָךְ:
וְאִם לָא תִפּוֹק לְוַת רַבְרְבֵי מַלְכָּא דְבָבֶל וְתִתְמְסַר קַרְתָּא הָדָא בִּידָא דְכַסְדָאֵי וְיוֹקְדִינָהּ בְּנוּרָא וְאַתְּ לָא תִשְׁתֵּיזֵיב מִידֵיהוֹן:
וַאֲמַר מַלְכָּא צִדְקִיָה לְיִרְמְיָה אֲנָא יָצֵיף מִן יְהוּדָאֵי דְאִשְׁתְּמָעוּ לְכַסְדָאֵי דִלְמָא יִמְסְרוּן יָתִי בִּידֵיהוֹן וְיִתְלַעֲבוּן בִּי:
וַאֲמַר יִרְמְיָה לָא יִמְסְרוּן קַבֵּיל כְּעַן לְמֵימְרָא דַייָ לְדַאֲנָא מְמַלֵיל עִמָךְ וְיֵיטַב לָךְ וְתִתְקַיֵם נַפְשָׁךְ:
וְאִם לֵית אַתְּ צָבֵי לְמִפַּק דֵין פִּתְגָמָא דְאַחְזְיַנִי יְיָ:
הָא כָּל נְשַׁיָא דְאִשְׁתְּאָרָא בְּבֵית מְלַךְ שִׁבְטָא דְבֵית יְהוּדָה מִתַּפִקָן לְוַת רַבְרְבֵי מַלְכָּא דְבָבֶל וְאִינוּן אָמְרִין אַטְעֲיוּךְ וִיכִילוּ לָךְ אֱנַשׁ שְׁלָמָךְ אִשְׁתְּקָעָא בִּבְהָתָא רִגְלָךְ אַסְתַּחָרָא לַאֲחָרָא:
וְיַת כָּל נְשָׁךְ וְיַת בְּנָךְ מִתַּפְקִין לְוַת כַּסְדָאֵי וְאַתְּ לָא תִשְׁתֵּיזֵיב מִיָדְהוֹן אֲרֵי בְיַד מַלְכָּא דְבָבֶל תִּתְמַסֵר וְיַת קַרְתָּא הָדָא תִּתּוֹקָד בְּנוּרָא:
וַאֲמַר צִדְקִיָה לְיִרְמְיָה אֱנַשׁ לָא יְדַע בְּפִתְגָמַיָא הָאִלֵין וְלָא תְמוּת:
וַאֲרֵי יִשְׁמְעוּן רַבְרְבַיָא אֲרֵי מַלְלִית עִמָךְ וְיֵתוּן לְוָתָךְ וְיֵימְרוּן לָךְ חַוֵי כְעַן לָנָא מַה מַלְלִית עִם מַלְכָּא לָא תְכַסֵי מִנָנָא וְלָא נִקְטְלִינָךְ וּמַה מַלֵיל עִמָךְ מַלְכָּא:
וְתֵימַר לְהוֹן בְּעוּ אֲנָא בְּעֵי קֳדָם מַלְכָּא בְּדִיל דְלָא לְאַתָבוּתִי בֵּית יְהוֹנָתָן לְמֵימָת תַּמָן:
וַאֲתוּ כָּל רַבְרְבַיָא לְוַת יִרְמְיָה וּשְׁאִילוּ יָתֵיהּ וְחַוֵי לְהוֹן כְּכָל פִתְגָמַיָא הָאִלֵין דְפַקֵיד מַלְכָּא וּשְׁתִיקוּ מִנֵיהּ אֲרֵי לָא יִשְׁתְּמַע פִּתְגָמָא:
וִיתֵיב יִרְמְיָה בְּדָרַת בֵּית אֲסִירַיָא עַד יוֹמָא דְאִתְכְּבִישַׁת יְרוּשְׁלֵם וַהֲוָה כַּד אִתְכְּבִישַׁת יְרוּשְׁלֵם:
