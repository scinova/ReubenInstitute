נְבִיַיָא אִתְנַבִּיאוּ תַנְחוּמִין עַל עַמִי אֲמַר אֱלָהָכוֹן:
מַלִילוּ עַל לִבָּא דִירוּשְׁלֵם וְאִתְנַבּוּ עֲלָהּ אֲרֵי עֲתִידָא דְתִתְמְלֵי מֵעַם גַלְוָתָהָא אֲרֵי אִישְׁתְּבִיקוּ לָהּ חוֹבָהָא אֲרֵי קַבִּילַת כָּס תַּנְחוּמִין מִן קֳדָם יְיָ כְּאִלוּ לָקַת עַל חַד תְּרֵין בְּכָל חַטָאתָהָא:
קַל דְמַכְלֵי בְּמַדְבְּרָא פַּנוּ אוֹרַח מִן קֳדָם עַמָא דַייָ כְּבִישָׁא בְּמֵישְׁרָא כִּבְשִׁין קֳדָם כְּנִשְׁתָּא דֶאֱלָהָנָא:
כָּל חֵילַיָא יְרָאמוּן וְכָל טוּר וְרָמָא יִמְאֲכוּן וִיהֵי כִפְלָא לְמֵישְׁרָא וּבֵית גִדוּדִין לְבִקְעָה:
וְיִתְגְלֵי יְקָרָא דַייָ וְיֶחֱזוּן כָּל בְּנֵי בִשְׂרָא כַּחֲדָא אֲרֵי בְמֵימְרָא דַייָ גְזַר כֵּן:
קַל דַאֲמַר אִתְנַבֵּי וּמְתֵיב וַאֲמַר מָה אִתְנַבֵּי כָּל רַשִׁיעַיָא כְּעִשְׂבָּא וְכָל תּוּקְפְהוֹן כְּמוֹצָא דְחַקְלָא:
יָבֵשׁ עִשְׂבָּא נְתַר נִצֵיהּ אֲרֵי רוּחָא מִן קֳדָם יְיָ נְשָׁבַת בֵּיהּ בְּכֵן כְּעִשְׂבָּא חֲשִׁיבִין רַשִׁיעַיָא בְּעַמָא:
מִית רַשִׁיעַיָא אֲבָדוּ עֶשְׁתּוֹנוֹהִי וּפִּתְגָמָא דְאֶלָהָנָא קַיָם לְעָלְמִין:
עַל טוּר רָם סַקוּ לְכוֹן נְבִיַיָא דִי מְבַשְׂרִין לְצִיוֹן אֲרִימוּ בְחֵילָא קָלְכוֹן דִי מְבַשְׂרִין לִירוּשְׁלֵם אֲרִימוּ לָא תִדְחֲלוּן אֲמָרוּ לְקִרְוַיָא דְבֵית יְהוּדָה אִתְגְלִיאַת מַלְכוּתָא דֶאֱלָהָכוֹן:
הָא יְיָ אֱלֹהִים בִּתְקוֹף מִתְגְלֵי וּתְקוֹף דְרַע גְבוּרְתֵּהּ שַׁלְטָא קֳדָמוֹהִי הָא אֲגַר עָבְדֵי מֵימְרֵהּ עִמֵהּ דְכָל עוֹבָדֵיהוֹן גְלַן קֳדָמוֹהִי:
כְּרָעַיָא דְעֶדְרֵהּ רָעֵי בְּדַרְעֵהּ מַכְנַשׁ אִמְרִין וּבְחֵינֵהּ מְסוֹבַר רַכִּיכִין מֵינִיקָתָא בְּנִיחַ מְדַבָּר:
מַן אֲמַר אִלֵין קַיָם אֲמַר וַעֲבֵד דְכָל מֵי עַלְמָא חֲשִׁיבִין קֳדָמוֹהִי כְּטִפָּא בִשְׁעוֹלָא וּמִשְׁחַת שְׁמַיָא כְּאִלוּ בְּזֵירְתָא מְתַקְנִין וְעַפְרָא דְאַרְעָא כְּאִלוּ בִּמְכִילָא אִתְכַּל וְטוּרַיָא כְּאִלוּ מַתְקַל תַּקִילִין וְרָמָתָא כַּד בְּמוֹזְנַיָא:
מַן תַּקִין יַת רוּחַ קוּדְשָׁא בְּפוּם כָּל נְבִיַיָא הֲלָא יְיָ וְצַדִיקַיָא עָבְדֵי מֵימְרֵהּ פִּתְגָמֵי רְעוּתֵהּ הוֹדְעִינוּן:
לְהוֹן דִבְעוֹן מִן קֳדָמוֹהִי אֲסַבַּר חוּכְמָא וְאַלְפִינוּן אוֹרַח דְדִין וִיהַב לִבְנֵיהוֹן אוֹרַיְתָא וְאוֹרַח דְסוּכְלְתָנוּ לִבְנֵי בְנֵיהוֹן הוֹדְעִינוּן:
הָא עַמְמַיָא כְּטִפָּא מִדוּל וּכְעִיוּל מוֹזְנַיָא חֲשִׁיבִין הָא נַגְוָתָא כְּדוֹקָא דְפָרַח:
וְאִילָנֵי לִבְנָן לָא יְסַפְקוּן לְאַדְלָקָא וְחֵיוָתָא דְבָהּ לֵית הִיא כְּמִיסַת לַעֲלָתָא:
כָּל עַמְמַיָא כִּלְמָא עוֹבָדֵיהוֹן גְמִירָא וְשֵׁיצָאָה אִינוּן חֲשִׁיבִין קֳדָמוֹהִי:
וּלְמַן אַתּוּן מְדַמִין קֳדָם אֵל וּמַה דְמוּת אַתּוּן מְסַדְרִין קֳדָמוֹהִי:
הָא צִלְמָא נַגְרָא עֲבֵיד וְקֵינָאָה בִּדְהַב מְחַפֵּי לֵיהּ וְשַׁלְשְׁלַן דִכְסַף קֵינָאָה מְאָחֵיד לֵיהּ:
אוֹרָן בָּרֵי אָעָא דְלָא אָחִיד בֵּיהּ רִקְבָא בָּחַר נַגַר אוּמַן בְּעֵי לֵיהּ לְאַתְקָנָא צְלַם דְלָא יִצְטְלֵי:
הֲלָא יְדַעְתּוּן הֲלָא שְׁמַעְתּוּן הֲלָא אִיתְחַוָאָה מִן אַוְלָא עוֹבַד סִדְרֵי בְרֵאשִׁית לְכוֹן הֲלָא תִּסְתַּכְּלוּן לְמִדְחַל קֳדָם מִן דִבְרָא יְסוֹדֵי אַרְעָא:
דְאַשְׁרֵי בִּתְקוֹף רוּמָא שְׁכִינַת יְקָרֵהּ וְכָל יַתְבֵי אַרְעָא חֲשִׁיבִין קֳדָמוֹהִי כְּקַמְצִין דְמָתַח כִּזְעֵיר שְׁמַיָא וּפַרְסִינוּן כְּמַשְׁכַּן יְקָרָא לְבֵית שְׁכִינְתָּא:
דִמְסַר שִׁלְטוֹנִין לְחוּלְשָׁא דַיָנֵי אַרְעָא כִּלְמָא עֲבֵיד:
אֲפִילוּ יִפְשׁוּן אֲפִילוּ יִסְגוּן אֲפִילוּ יִתְרַבּוּן בְּאַרְעָא בְנֵיהוֹן וְאַף רוּגְזֵהּ יְשַׁלַח בְּהוֹן וְיִבְהֲתוּן וּמֵימְרֵהּ כְּעַלְעֵילָא לְקַשָׁא יְדַבֵּר יַתְהוֹן:
וּלְמַן תְּדַמוּן קֳדָמַי וּתְשַׁווֹן יֵימַר קַדִישָׁא:
זְקוּפוּ לְרוּמָא עֵינֵיכוֹן וַחֲזוּ לְמִדְחַל קֳדָם מַן דִבְרָא אִלֵין דְאַפֵּיק בְּמִנְיָן חֵילֵי שְׁמַיָא לְכוּלְהוֹן בְּשִׁמְהָן קָרֵי מַסְגֵי גְבוּרָן וּתְקוֹף חֵיל חַד מִסִדְרֵהּ לָא מִתְעַכָּב:
לְמָה תֵימַר יַעֲקֹב וּתְמַלֵיל יִשְׂרָאֵל מְטַמְרָא אוֹרְחִי מִן קֳדָם יְיָ וּמִן קֳדָם אֱלָהִי דִינִי יְעָדֵי:
הֲלָא יְדַעְתָּא אִם לָא שְׁמַעְתָּא אֱלָהָא עַלְמָא יְיָ דִי בְרָא יְסוֹדֵי אַרְעָא לָא בַעֲמַל וְלָא בְלֵיאוּ לֵית סוֹף לְסוּכְלְתָנוּתֵיהּ:
דִיהַב לְצַדִיקַיָא דִמְשַׁלְהַן לְפִתְגָמֵי אוֹרַיְתָא חוּכְמָא וּדְלֵית לְהוֹן תְּקוֹף נִכְסִין מַסְגֵי:
וְיִשְׁתַּלְהוּן עוּלֵימִין חַיָבִין וְיִלְאוּן וּרְוָקֵי רַשִׁיעַיָא אִיתְּקָלָא יִתְקְלוּן:
וּדְסַבָּרוּ לְפוּרְקָנָא דַייָ יִתְכַּנְשׁוּן מִבֵּינֵי גָלוּתְהוֹן וְיוֹסְפוּן חֵיל וְיִתְחַדְתּוּן לְעוּלֵימַתְהוֹן כְּצִימוּחַ דְסָלֵיק עַל גַדְפֵי נִשְׁרִין יְרַהֲטוּן וְלָא יִלְאוּן יַהֲכוּן וְלָא וְלָא יִשְׁתַּלְהוּן:
