וְאוֹסִיפוּ בְּנֵי יִשְׂרָאֵל לְמֶעְבַּד דְבִישׁ קֳדָם יְיָ וְאֵהוּד מִית:
וּמְסַרְנוּן יְיָ בְּיַד יָבִין מַלְכָּא דִכְנַעַן דִי מְלַךְ בְּחָצוֹר וְרַב חֵילֵיה סִיסְרָא וְהוּא יָתֵיב בִּתְקוֹף כְּרַכֵּי עַמְמַיָא:
וּצְעָקוּ בְנֵי יִשְׂרָאֵל קֳדָם יְיָ אֲרֵי תְשַׁע מֵאָה רְתִיכִין דְבַרְזְלָא לֵיהּ וְהוּא דְחַק יַת בְּנֵי יִשְׂרָאֵל בְּתוּקְפָא עַסְרִין שְׁנִין:
וּדְבוֹרָה אִתְּתָא נְבִיאֲתָא אִיתַּת לַפִּידוֹת הִיא דַיְנָא יַת יִשְׂרָאֵל בְּעִדָנָא הַהִיא:
וְהִיא יַתְבָה בְּקַרְתָּא בַּעֲטָרוֹת דְבוֹרָה מִתְפַּרְנְסָא מִן דִילָהּ וְלָה דִקְלִין בִּירִיחוֹ פַּרְדְסִין בְּרָמָתָא זֵיתִין עָבְדִין מְשַׁח בְּבִקְעֲתָא בֵּית שַׁקְיָא בְּבֵית אֵל עֲפַר חִוַר בְּטוּר מַלְכָּא וְסַלְקִין לְוָתָהּ בְנֵי יִשְׂרָאֵל לְדִינָא:
וּשְׁלָחַת וּקְרַת לְבָרָק בַּר אֲבִינוֹעַם מִקֶדֶשׁ דְבֵית נַפְתָּלִי וַאֲמַרַת לֵיהּ הֲלָא פָּקַד יְיָ אֱלָהָא דְיִשְׂרָאֵל אֱזֵיל וְתִתְנְגִיד בְּטוּרָא דְתָבוֹר וְתִדְבַּר עִמָךְ עַסְרָא אַלְפִין גַבְרָא מִבְּנֵי נַפְתָּלִי וּמִבְּנֵי זְבוּלֻן:
וֶאֱתַתֵּיב עַל יְדָךְ לְנַחְלָא דְקִישׁוֹן יַת סִיסְרָא רַב חֵילָא דְיָבִין וְיַת רְתִיכוֹהִי וְיַת מַשִׁרְיָתֵיהּ וְאֶמְסְרִנֵהּ בִידָךְ:
וַאֲמַר לָהּ בָּרָק אִם תֵּיזְלִין עִמִי וְאֵיזִיל וְאִם לָא תֵּיזְלִין עִמִי לָא אֵיזִיל:
וַאֲמַרַת מֵיזַל אֵיזִיל עִמָךְ לְחוֹד אֲרֵי לָא תְהֵי תוּשְׁבַּחְתָּךְ עַל אוֹרְחָא דְאַתְּ אָזֵיל אֲרֵי בְּיַד אִתְּתָא יִמְסוֹר יְיָ יַת סִיסְרָא וְקָמַת דְבוֹרָה וַאֲזַלַת עִם בָּרָק לְקֶדֶשׁ:
וּכְנַשׁ בָּרָק יַת שִׁבְטָא דִזְבוּלֻן וְיַת שִׁבְטָא דְנַפְתָּלִי לְקֶדֶשׁ וּסְלִיקוּ עִמֵיהּ עַסְרָא אַלְפִין גַבְרָא וּסְלִיקַת עִמֵיהּ דְבוֹרָה:
וְחֶבֶר שַׁלְמָאָה אִתְפָּרַשׁ מִשַׁלְמָאֵי מִבְּנֵי חוֹבָב חֲמוּהִי דְמֹשֶׁה וּפַרְסֵיהּ לְמַשְׁכְּנֵיהּ עַד מֵישַׁר אַגְנַיָא דְעִם קֶדֶשׁ:
וְחַוִיאוּ לְסִיסְרָא אֲרֵי סְלִיק בָּרָק בַּר אֲבִינוֹעַם לְטוּר תָּבוֹר:
וּכְנַשׁ סִיסְרָא יַת כָּל רְתִיכֵיהּ תְּשַׁע מְאָה רְתִיכִין דְבַרְזְלָא וְיַת כָּל עַמָא דְעִמֵיהּ מִתְקוֹף כְּרַכֵּי עַמְמַיָא לְנַחֲלָא דְקִישׁוֹן:
וַאֲמַרַת דְבוֹרָה לְבָרָק קוּם אֲרֵי דֵין יוֹמָא דִמְסַר יְיָ יַת סִיסְרָא בִּידָךְ הֲלָא מַלְאָכָא דַייָ נְפַק לְאַצְלָחָא קֳדָמָךְ וּנְחַת בָּרָק מִטוּרָא דְתָבוֹר וְעַסְרָא אַלְפִין גַבְרָא בַּתְרוֹהִי:
וְתַבֵּר יְיָ יַת סִיסְרָא וְיַת כָּל רְתִיכַיָא וְיַת כָּל מַשְׁרִיתָא לְפִתְגַם דְחֶרֶב קֳדָם בָּרָק וּנְחַת סִיסְרָא מֵעַל רְתִיכָא וְאַפַּךְ בְּרַגְלוֹהִי:
וּבָרָק רְדַף בָּתַר רְתִיכָא וּבָתַר מַשְׁרִיתָא עַד תְּקוֹף כְּרַכֵּי עַמְמַיָא וְאִתְקְטָלָא כָּל מַשִׁרְיַת סִיסְרָא לְפִתְגַם דְחָרֶב לָא אִשְׁתָּאַר עַד חָד:
וְסִיסְרָא אַפֵּךְ בְּרַגְלוֹהִי לְמַשְׁכְּנָא דְיָעֵל אִתַּת חֶבֶר שַׁלְמָאָה אֲרֵי שְׁלָמָא בֵּין יָבִין מַלְכָּא דְחָצוֹר וּבֵין בֵּית חֶבֶר שַׁלְמָאָה:
וְנַפְקַת יָעֵל לִקְדָמוּת סִיסְרָא וַאֲמַרַת לֵיהּ זוּר רִבּוֹנִי זוּר לְוָתִי לָא תִדְחַל וְזַר לְוָתָהּ לְמַשְׁכְּנָא וְכַסִיתֵיהּ בְּגוּנְכָא:
וַאֲמַר לָהּ אַשְׁקִינִי כְעַן זְעֵיר מַיָא אֲרֵי צָחֵי אֲנָא לְמִשְׁתֵּי וּפְתַחַת יַת נוֹדָא דְחֶלְבָּא וְאַשְׁקִיתֵיהּ וכַסִיתֵיהּ:
וַאֲמַר לָה קוּמִי בִּתְרַע מַשְׁכְּנָא וִיהֵי אִם אֱנַשׁ יֵיתֵי וְיִשְׁאֲלִנָיךְ וְיֵמַר הֲאִית כָּא גַבְרָא וְתֵימְרִין לָא:
וּנְסִיבַת יָעֵל אִתַּת חֶבֶר יַת סִכַּת מַשְׁכְּנָא וְשַׁוִיאַת יַת אַרְזַפְתָּא בִּידָהּ וַעֲלַת לְוָתֵיהּ בְּרָז וּנְקִישַׁת יַת סִכְּתָא בְּצִדְעֵיהּ וּנְעָצַת בְּעַרְעָא וְהוּא דְמוּך וְאִשְׁתַּלְהִי וּמִית:
וְהָא בָרָק רְדַף יַת סִיסְרָא וְנַפְקַת יָעֵל לִקְדָמוּתֵיהּ וַאֲמַרַת לֵיהּ אִיתָא וְאַחֲזִינָךְ יַת גַבְרָא דְאַתְּ בָּעֵי וְעַל לְוָתָהּ וְהָא סִיסְרָא טָרִיף וּמִית וְסִכְּתָא נְעִיצָא בְּצִדְעֵיהּ:
וּתְבַר יְיָ בְּיוֹמָא הַהוּא יַת יָבִין מַלְכָּא דִכְנָעַן קֳדָם בְּנֵי יִשְׂרָאֵל:
וְאַזְלַת יַד בְּנֵי יִשְׂרָאֵל אָזְלָא וְתָקְפָא עַל יָבִין מַלְכָּא דִכְנָעַן עַד דִי שֵׁצִיאוּ יַת יָבִין מַלְכָּא דִכְנָעַן:
