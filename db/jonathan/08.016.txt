וַאֲמַר יְיָ לִשְׁמוּאֵל עַד אֵימָתַי אַתְּ מִתְאַבֵּל אֶל שָׁאוּל וַאֲנָא רְחַקְתֵּיהּ מִלְמֶהֱוֵי מַלְכָּא עַל יִשְׂרָאֵל מְלֵי קַרְנָךְ מִשְׁחָא וְאִיתָא אֶשְׁלְחִינָךְ לְוַת יִשַׁי דְמִבֵּית לֶחֶם אֲרֵי גְלֵי קֳדָמַי בִּבְנוֹהִי כָּשַׁר קֳדָמַי לְמֶהֱוֵי מַלְכָּא:
וַאֲמַר שְׁמוּאֵל אֵיכְדֵין אֵיזֵל וְאִם יִשְׁמַע שָׁאוּל וְיִקְטְלִינַנִי וַאֲמַר יְיָ עֶגְלַת תּוֹרִין תִּסַב בִּידָךְ וְתֵימַר לְדַבָּחָא קֳדָם יְיָ אָתֵיתִי:
וְתַזְמִין לְיִשַׁי בְּשֵׁירוּתָא וַאֲנָא אַחֲזִינָךְ יַת דְתַעְבֵּד וְתִמְשַׁח קֳדָמַי יַת דְאֵימַר לָךְ:
וַעֲבַד שְׁמוּאֵל יַת דְמַלֵיל יְיָ וַאֲתָא לְבֵית לֶחֶם וְאִתְכַּנָשׁוּ סָבֵי קַרְתָּא לִקְדָמוּתֵיהּ וַאֲמַרוּ שְׁלָם מֵיתָךְ:
וַאֲמַר שְׁלָם לְדַבָּחָא קֳדָם יְיָ אָתֵיתִי אִזְדַמָנוּ וְתֵיעֲלוּן עִמִי לְשֵׁירוּתָא וְזַמֵין יַת יִשַׁי וְיַת בְּנוֹהִי וּקְרָא לְהוֹן לְנִכְסַת קוּדְשָׁא:
וַהֲוָה בְּמֵיעַלְהוֹן וַחֲזָא יַת אֱלִיאָב וַאֲמַר בְּרַם תַּקִין קֳדָם יְיָ מְשִׁיחֵיהּ:
וַאֲמַר יְיָ לִשְׁמוּאֵל לָא תִסְתַּכֵּל בְּחֶזְוֵיהּ וּבְרוּם קוֹמָתֵיהּ אֲרֵי רְחַקְתֵּיה אֲרֵי לָא כְּמוֹ דְחָזָן בְּנֵי אֱנָשָׁא אֲרֵי בְּנֵי אֱנָשָׁא חָזָן בְּעֵינַיְהוֹן וְקֳדָם יְיָ גַלְיָן מַחְשְׁבַת לִבָּא:
וּקְרָא יִשַׁי לַאֲבִינָדָב וְאַעֲבִירֵיהּ קֳדָם שְׁמוּאֵל וַאֲמַר אַף בְּדֵין לָא רַעֲוָא קֳדָם יְיָ:
וְאַעֲבַר יִשַׁי שַׁמָה וַאֲמַר אַף בְּדֵין לָא רַעֲוָא קֳדָם יְיָ:
וְאַעֲבַר יִשַׁי שַׁבְעָא בְנוֹהִי קֳדָם שְׁמוּאֵל וַאֲמַר שְׁמוּאֵל לְיִשַׁי לָא רַעֲוָא קֳדָם יְיָ בְּאִלֵין:
וַאֲמַר שְׁמוּאֵל לְיִשַׁי הַשְׁלִימוּ עוּלֵימַיָא וַאֲמַר עוֹד אִשְׁתָּאַר זְעֵירָא וְהָא רָעֵי בְעָנָא וַאֲמַר שְׁמוּאֵל לְיִשַׁי שְׁלַח וְאַיְתִינֵיהּ אֲרֵי לָא נִסְתְּחַר עַד מֵיתוֹהִי הָכָא:
וּשְׁלַח וְאַיְתֵיהּ וְהוּא סַמוֹק עֵינוֹהִי יָאִין וְשַׁפִּיר בְּרֵיוֵיהּ וַאֲמַר יְיָ קוּם מַשְׁחֵיהּ אֲרֵי דֵין הוּא:
וּנְסִיב שְׁמוּאֵל יַת קַרְנָא דְמִשְׁחָא וּמְשַׁח יַת דָוִד בְּגוֹ אֲחוֹהִי וּשְׁרַת רוּחַ נְבוּאָה מִן קֳדָם יְיָ עַל דָוִד מִיוֹמָא הַהוּא וּלְעֵלָא וְקָם שְׁמוּאֵל וַאֲזַל לְרָמָתָא:
וְרוּחַ גְבוּרָה מִן קֳדָם יְיָ דַהֲוַת עִם שָׁאוּל עֲדָת מִנֵיהּ וּמְבַעֲתָא לֵיהּ רוּחַ בִּישָׁא מִן קֳדָם יְיָ:
וַאֲמַרוּ עַבְדֵי שָׁאוּל לֵיהּ הָא כְעַן רוּחַ בִּישָׁא מִן קֳדָם יְיָ מְבַעֲתָא לָךְ:
יֵימַר כְּעַן רִבּוֹנָנָא עַבְדָךְ קֳדָמָךְ יִבְעוּן גַבְרָא דְיָדַע לְנַגְנָא בְּכִנוֹרָא וִיְהֵי כַּד תִשְׁרֵי עֲלָךְ רוּחַ בִּישָׁא מִן קֳדָם יְיָ וִינַגֵן בִּידֵיהּ וְיֵיטַב לָךְ:
וַאֲמַר שָׁאוּל לְעַבְדוֹהִי חֲזוֹ כְעַן לִי גַבְרָא דְמוֹטִיב לְנַגָנָא וְתַיְתוּנֵיהּ לְוָתִי:
וְאָתֵיב חַד מֵעוּלֵימַיָא וַאֲמַר הָא חֲזֵתִי בַּר לְיִשַׁי דְמִבֵּית לֶחֶם יָדַע לְנַגָנָא וְגִבַּר חֵילָא וְגִבַּר עָבִיד קְרָבִין וְסוּכְלָתָן בְּעֵצָה וְגַבְרָא שַׁפִּיר בְּרֵיוָא וּמֵימְרָא דַייָ בְּסַעֲדֵיהּ:
וּשְׁלַח שָׁאוּל אִזְגַדִין לְוַת יִשָׁי וַאֲמַר שַׁדַר לִי יַת דָוִד בְּרָךְ דִי בְּעָנָא:
וּנְסִיב יִשַׁי טְעוּן חַמְרָא טְעוּן דְלַחְמָא וּגְרַב דַחֲמָר וְגַדְיָא בַר עִזֵי חַד וְשַׁדַר בְּיַד דָוִד בְּרֵיהּ לְוַת שָׁאוּל:
וַאֲתָא דָוִד לְוַת שָׁאוּל וְשַׁמֵשׁ קֳדָמוֹהִי וְרִחֲמֵיהּ לַחֲדָא וַהֲוָה לֵיהּ נָטַל זְיָנִין:
וּשְׁלַח שָׁאוּל לְוַת יִשַׁי לְמֵימָר יְשַׁמֵשׁ כְּעַן דָוִד קֳדָמַי אֲרֵי אַשְׁכַּח רַחֲמִין קֳדָמַי:
וַהֲוָה כַּד שַׁרְיָא רוּחַ בִּישָׁא מִן קֳדָם יְיָ לְוַת שָׁאוּל וּנְסִיב דָוִד יַת כִּנוֹרָא וּמְנַגֵן בִּידֵיהּ וּמִתְרְוַח לְשָׁאוּל וְטָב לֵיהּ וּמִסְתַּלְקָא מִנֵיהּ רוּחַ בִּישָׁא:
