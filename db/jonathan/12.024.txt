הָא יְיָ מַבֵּיז אַרְעָא וּמְסַר לָהּ לְסָנְאָה וְתַחְפֵּי בַהֲתָא אַפֵּי רַבְרְבָהָא עַל דַעֲבַרוּ עַל אוֹרַיְתָא וִיבַדַר יַת דְיָתְבִין בָּהּ:
וִיהֵי חִילוֹנָאָה כְּכַהֲנָא עַבְדָא כְּרִיבּוֹנֵהּ אַמְתָא כְּמֶרָתָהּ זָבְנָא כִּמְזַבְּנָא יָזְפָא כְּמוֹזְפָא רָשָׁא כְּמַן דְרָשֵׁי בֵיהּ:
אִתְבְּזָזָא תִּתְבְּזֵז אַרְעָא וְאִתְדָשָׁא תִּתְדָשׁ אֲרֵי יְיָ מַלֵל יַת פִּתְגָמָא הָדֵין:
אִתְאַבְּדַת חֲרוֹבַת אַרְעָא צְדִיאַת חֲרוֹבַת תֵּבֵל סְפוּ תְקוֹף עַמָא דְאַרְעָא:
וְאַרְעָא חָבַת תְּחוֹת יַתְבָהָא אֲרֵי עֲבַרוּ עַל אוֹרַיְתָא אַבְטִילוּ מוֹעֲדַיָא אַשְׁנִיאוּ קְיָמָא דִי מִן עַלְמָא:
עַל כֵּן מִן קֳדָם מוֹמֵי דִשְׁקַר חֲרוֹבַת אַרְעָא וּצְדִיאוּ דְיָתְבִין בָּהּ עַל כֵּן סָפוּ יָתְבֵי אַרְעָא וְאִשְׁתְּאַר אֱנָשָׁא כִּזְעֵיר:
אִתְאַבַּלוּ כָל שָׁתֵי חֲמַר אֲרֵי אִתְפְּרִיכוּ גוּפְנַיָא אִתְאֲנַחוּ כָּל חָדֵי לִבָּא:
בְּטֵלַת חֶדְוַת תּוּפִּין אִתְמְנַעַת אִתְרְגוּשַׁת תַּקִיפִין פִּסְקַת חֶדְוַת כִּנָרָא:
עַל יְדֵי זְמָר לָא יִשְׁתּוּן חֲמַר יֵימַר עֲתִּיקָא לְשָׁתוֹהִי:
אִתְּבַרַת קַרְתְּהוֹן צְדִיאַת אֲחִידוּ כָל בָּתַּיָא מִלְמֵעַל:
מְצַוְחִין עַל חַמְרָא בְּשׁוּקַיָא שְׁלֵימַת כָּל חֶדְוָתָא גְלָא בִּיעַ מִן אַרְעָא:
אִשְׁתְּאַר בְּקַרְתָּא צְדוֹ וְאִתְרְגוּשְׁתָּא בְפִגוּר תַּרְעִין:
אֲרֵי כְדֵין יִשְׁתְּאֲרוּן יְחִידָאִין צַדִיקַיָא בְגוֹ אַרְעָא בֵּינֵי מַלְכְּוָתָא כְּבִיעוּר זֵיתָא כְּעוֹלְלָן בָּתַר קָטַף:
אִנוּן יִרְמוֹן קָלֵיהוֹן יְשַׁבְּחוּן בְּמֵימְרָא דַייָ יִבְעוֹן כְּמָא דִבְעוֹ עַל גְבוּרָן דְאִתְעֲבִידָא לְהוֹן עַל יַמָא:
עַל כֵּן בְּמֵיתֵי נְהוֹרָא לְצַדִיקַיָא יִקְרוֹן קֳדָם יְיָ בְּנִיסֵי יַמָא יְהוֹן מוֹדָן וּמְבָרְכִין שְׁמָא דַייָ אֱלָהָא דְיִשְׂרָאֵל:
מִבֵּית מַקְדְשָׁא דְמִתַּמָן עֲתִיד לְמִפַּק חֶדְוָא לְכָל יַתְבֵי אַרְעָא תּוּשְׁבְּחָא שְׁמַעְנָא לְצַדִיקַיָא אָמַר נְבִיָא רָב אֲגַר לְצַדִיקַיָא לְרַשִׁיעַיָא אִתְגְלֵי לִי וַי לַאֲנוּסַיָא דְמִתְאַנְסִין וּלְבִזוּז בָּזְזִין דְהָא מִתְבָּזִין:
דַחֲלָא וְכוּמְצָא וּמְצָדָא עֲלָךְ יָתֵיב אַרְעָא:
וִיהֵי דְיֶעֱרוֹק מִן קֳדָם דַחֲלָא יִפּוֹל לְגוֹ כּוּמְצָא וְדִיַסֵק מִגוֹ כּוּמְצָא יִתְאַחַד בִּסְרִיגְתָּא אֲרֵי גְבוּרָן בִּשְׁמַיָא אִתְעֲבִידָא וְזָעוּ יְסוֹדֵי אַרְעָא:
מַזָע תְּזוּעַ אַרְעָא מֵנָד תָּנוּד אַרְעָא פְּרוֹק תִּתְפְּרַק אַרְעָא:
אִשְׁתְּדָאָה תִּשְׁתְּדֵי אַרְעָא כְּרַוְיָא וּתְהֵי אַזְלָא וְאַתְיָא כְּעַרְסְלָא וְיִתְקְפוּן עֲלָהּ חוֹבָהָא וְתִפּוֹל וְלָא תוֹסִיף לְמֵקָם:
וִיהֵי בְּעִדָנָא הַהִיא יַסְעַר יְיָ עַל חַלְוָתָא תוּקְפָא דְיָתְבִין בְּתוּקְפָא וְעַל מַלְכֵי בְּנֵי אֱנָשָׁא דְדָיְרִין עַל אַרְעָא:
וְיִכְנְשׁוּנוּן מִכְנַשׁ לְבֵית אַסִירֵי וִיעַנְגוּנוּן לְבֵית עַגְנָה וּמַסְגֵי יוֹמִין יֵעוּל דָכְרָנֵיהוֹן:
וְיִבְהֲתוּן דְפָלְחִין לְסִיהֲרָא וְיִתְכַּנְעוּן דְסָגְדִין לְשִׁמְשָׁא אֲרֵי תִתְגְלֵי גְבוּרְתָּא דַייָ צְבָאוֹת בְּטוּרָא דְצִיוֹן וּבִירוּשְׁלֶם וּקְדָם סָבֵי עַמֵהּ בִּיקָר:
