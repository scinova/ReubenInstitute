וַהֲוָה פִּתְגָם נְבוּאָה מִן קֳדָם יְיָ עִמִי לְמֵימָר:
בַּר אָדָם קַבֵּיל נְבוּאָה לְאוֹרַח דָרוֹמָא וְאַלֵיף עַל דָרוֹמָא וְאִתְנְבֵּי עַל חוּרְשָׁא דַחֲקַל דָרוֹמָא:
וְתֵימַר לְחוּרְשָׁא דְדָרוֹמָא קַבֵּיל פִּתְגָמָא דַייָ כִּדְנַן אֲמַר יְיָ אֱלֹהִים הָא אֲנָא מַדְלִיק בָּךְ אֶשְׁתָּא וְתִקְטֵיל בָּךְ כָּל אִילַן יַבִּישׁ לָא תִטְפֵי לְשַׁלְהוֹבִיתָא דִמְשֵׁיצְיָא וְיִתְרַשְׁמוּן בָּהּ כָּל אַפַּיָא מִדְרוֹם לְצִפּוּנָא:
וְיֶחֱזוּן כָּל בְּנֵי בִסְרָא אֲרֵי אֲנָא יְיָ אַדְלִיקְתָּהּ לָא תִטְפֵי:
וַאֲמָרֵית קַבֵּיל בָּעוּתִי יְיָ אֱלֹהִים הָא אִינוּן אָמְרִין לִי הֲלָא מַמְתֵיל מַתְלִין הוּא:
וַהֲוָה פִּתְגַם נְבוּאָה מִן קֳדָם יְיָ עִמִי לְמֵימָר:
בַּר אָדָם קַבֵיל נְבוּאָה עַל יְרוּשְׁלֵם וְאַלֵיף עַל מַקְדְשִׁין וְאִתְנַבֵּי עַל אַרְעָא דְיִשְׂרָאֵל:
וְתֵימַר לְאַרְעָא דְיִשְׂרָאֵל כִּדְנַן אֲמַר יְיָ הָא אֲנָא שְׁלַח רוּגְזִי עֲלָךְ וְאַפֵּיק חַרְבִּי מִלִדְנָהּ וְאַגְלֵי מִנִיךְ זַכָּאָךְ בְּדִיל לְשֵׁיצָאָה חַיָבָךְ:
חֲלַף דְאַגְלֵי מִנִיךְ זַכָּאָךְ בְּדִיל לְשֵׁיצָאָה חַיָבָךְ בְּכֵן תִּפּוֹק חַרְבִּי מִלִדְנָהּ עַל כָּל בִּסְרָא מִדָרוֹם לְצִפּוּנָא:
וְיֵדְעוּן כָּל בְּנֵי בִּסְרָא אֲרֵי אֲנָא יְיָ אַפְקֵית חַרְבִּי מִלִדְנָהּ לָא תְתוֹב עוֹד:
וְאַתְּ בַּר אָדָם אִתְאַנַח בְּזִיעַ חַרְצִין וּבְמֶרֶר תִּתְאַנַח לְעֵינֵיהוֹן:
וִיהֵי אֲרֵי יֵמְרוּן לָךְ עַל מָה אַתְּ מִתְאַנַח וְתֵימַר עַל בְּסוֹרְתָא אֲרֵי אַתְיָא וְיִתְמְסֵי כָּל לֵב וְיִתְרַשְׁלָן כָּל יְדֵין וְתֶעֱמֵי כָל רוּחַ וְכָל רְכוּבַיָא יִתְאַשְׁדָן מַיָא הָא אַתְיָא וְתִתְקַיֵם אֲמַר יְיָ אֱלֹהִים:
וַהֲוָה פִּתְגַם נְבוּאָה מִן קֳדָם יְיָ עִמִי לְמֵימָר:
בַּר אָדָם אִתְנַבֵּי וְתֵימַר כִּדְנַן אֲמַר יְיָ אֱלֹהִים אֲמַר חַרְבָּא דְמַלְכָּא דְבָבֶל תִּקְטוֹל וְחַרְבָּא דִבְנֵי עַמוֹן תִּתְּנֵי וּתְשֵׁיצֵי:
בְּדִיל לְמִקְטַל קְטוֹל אִשְׁתְּלִיפַת בְּדִיל לְמֶעְבַּד פּוּרְעֲנוּת אִשְׁתַּנִינַת עַל דְחָדִיאוּ שִׁבְטָא דְבֵית יְהוּדָה וּבִנְיָמִין לְשִׁבְטֵי יִשְׂרָאֵל כַּד גְלוֹ עַל דִפְלָחוּ לְטַעֲוָתָא וְתָבוּ אִינוּן לְמִטְעֵי בָּתַר צַלְמֵי אָעַיָא:
וִיהַב פּוּרְעֲנוּתְהוֹן לְאִתְמַסְרָא בְּיַד מַלְכָּא דְבָבֶל הִיא מְחַת חַרְבָּא תְּהֵי פּוּרְעֲנוּתְהוֹן לְמִמְסַר יַתְהוֹן בְּיַד קָטוֹלִין:
צְוַח וְאַיְלֵיל בַּר אָדָם אֲרֵי הִיא הֲוַת בְּעַמִי הִיא וְכָל רַבְרְבֵי יִשְׂרָאֵל קְטוֹלֵי חַרְבָּא אִתְכַּנְשׁוּ וְאָתוּ עַל עַמִי מִסְחוֹר סְחוֹר בְּכֵן טְפַח עַל יַרְכָא:
אֲרֵי אִתְנַבִּיאוּ עֲלֵיהוֹן נְבִיָא וְלָא תָבוּ וּמָה יְהֵי בְסוֹפֵיהוֹן אֱמַר וְאַף שִׁבְטָא דְבֵית יְהוּדָה וּבִנְיָמִין מִגְלָא יַגְלוּן וּבְעוֹבְדֵיהוֹן בִּישַׁיָא לָא יִתְקַיְמוּן אֲמַר יְיָ אֱלֹהִים:
וְאַתְּ בַּר אָדָם אִתְנַבֵּי וּטְפַח יַד עַל יַד וְתִתְכְּפֵיל חַרְבָא תְלִיתָאָה חֲרַב קְטוֹלִין הִיא חֲרָב פּוּרְעֲנוּת קְטוֹלָא רַבָּא דִמְזִיעָא לְהוֹן:
בְּדִיל לְמִתְבַּר לִבְּהוֹן וּלְאַסְגָאָה מַתְקַלְהוֹן עַל כָּל קִרְוֵיהוֹן מַנִיתִי קָטְלֵי חַרְבָּא וֵי חַרְבָּא אִשְׁתְּלִיפַת לְמִקְטֵל אִשְׁתְּנִינִת לְשֵׁיצָאָה:
אִשְׁתְּלִיפִי וּקְטִילִי מִדָרוֹמָא וְשֵׁיצֵי מִן צִפוּנָא לְאָן אַפָּךְ מְזַמְנִין:
וְאַף אֲנָא אַיְתֵי פּוּרְעֲנוּת עַל פּוּרְעֲנוּת וַאֲנִיחַ חֵימְתִי אֲנָא יְיָ גְזַרֵית בְּמֵימְרִי:
וַהֲוָה פִּתְגָם נְבוּאָה מִן קֳדָם יְיָ עִמִי לְמֵימָר:
וְאַתְּ בַּר אָדָם אַתְקִין לָךְ תַּרְתֵּין אוֹרְחָן דְיֵיתוּן בְּהוֹן קְטוֹלֵי חַרְבָּא דְמַלְכָּא דְבָבֶל מִמְדִינְתָא הֲדָא יִפְקוּן תַּרְוֵיהוֹן וְאַתְרָא אַתְקִין בְּרֵישׁ פְּרָשׁוּת אוֹרַח קַרְתָּא אַתְרָא אַתְקֵין:
אוֹרְחָא אַתְקֵין דְיֵיתוּן בָּהּ קְטוֹלֵי חַרְבָּא מֵרַבַּת בְּנֵי עַמוֹן וְעַל דְבֵית יְהוּדָה דִי נְפַקוּ מִירוּשְׁלֵם לְמֵיתַב בְּקִרְוִין כְּרִיכָן:
אֲרֵי קָם מַלְכָּא דְבָבֶל עַל אֵם פְּרָשׁוּת אוֹרַח בְּרֵישׁ תַּרְתֵּין אוֹרַח לְמִקְסָם קָסֶם קְשַׁת בְּגִרְרַיָא שָׁאַל בְּצַלְמְנַיָא חֲזָא בְּכַבְדָא:
בִימִינֵיהּ הֲוָה קִסְמָא נָפִיל לִמְהַךְ לִירוּשְׁלֵם לְמַנְאָה אָפְרוּרְוָן לְפַתָּרָא תַרְעִין דְיַעֲלוּן בְּהוֹן קָטוֹלִין לַאֲרָמָא קַל בְּאַצְוָחַת קַרְנָא לְמַנְאָה אָפְרוּרְוָן עַל תַּרְעֵן לְמִצְבַּר מִלַיְתָא וּלְמִבְנֵי כַרְקוֹם:
וַהֲוָה לְהוֹן קִסוּמֵיהוֹן כַּדִיבִין בְּעֵינֵיהוֹן נַחֲשִׁין דַהֲוָה מְנַחַשׁ לְהוֹן וְאִינוּן לָא יָדְעִין דְאַרְבְּעִין וּתְשַׁע זַמְנִין הֲוָה כְמַן וּמִתְּתֵיב עַל פִּתְגָמָא עַד דְתֵיתֵי לֵיהּ שַׁעְתָּא דְיִתְמַסְרוּן בִּידֵיהּ:
בְּכֵן כִּדְנַן אֲמַר יְיָ אֱלֹהִים חֲלַף דְאַדְכַּרְתּוּן חוֹבֵיכוֹן בְּאִתְגְלָאָה מְרָדֵיכוֹן לְאִתְחֲזָאָה חֶטְאֵיכוֹן בְּכֹל עוֹבָדֵיכוֹן חֲלַף דְאַדְכַּרְתּוּן בְּיַד מַלְכָּא דְבָבֶל תִּתְמַסְרוּן:
וְאַתְּ חַיָב קְטָלָא רַשִׁיעָא רַבָּא דְיִשְׂרָאֵל דִמְטָא יוֹם תַּבְרֵיהּ עִדַן תּוּשְׁלְמַת חוֹבוֹהִי:
כִּדְנַן אֲמַר יְיָ אֱלֹהִים אַעְדֵי מִצְנָפְתָּא מִשְׁרָיָה כַּהֲנָא רַבָּא וַאֲבַטֵיל כִּתְרָא מִצִדְקִיָה מַלְכָּא אֲמַר דֵין וְדֵין בְּאַתְרֵיהּ לָא יִתְקַיֵם מִגְלָא יִגְלוֹן גְדַלְיָה בַּר אֲחִיקָם דְלָא הֲוָה דִילֵיהּ יִסְבִינָהּ וְצִדְקִיָה דְדִילֵיהּ הִיא תַּעְדֵי מִנֵיהּ:
חוֹבֵיהוֹן דְחָבוּ כְּחוֹבֵיהוֹן אֶתְפְּרַע מִנְהוֹן אַף לְדֵין לָא יִתְקַיֵם לֵיהּ עַד דְאַיְתֵי עֲלוֹהִי פּוֹרְעֲנוּת יִשְׁמָעֵאל בַּר נְתַנְיָה וְאֶמְסְרִינֵיהּ בִּידֵיהּ:
וְאַתְּ בַּר אָדָם אִתְנַבֵּי וְתֵימַר כִּדְנַן אֲמַר יְיָ אֱלֹהִים עַל בְּנֵי עַמוֹן וְעַל חִסוּדֵיהוֹן וְתֵימַר חַרְבָּא חַרְבָּא אִשְׁתַּלְפַת לְמִקְטַל אִשְׁתַּנִינַת לְשֵׁיצָאָה בְּדִיל לְמֶהֱוֵי לָהּ צַחְצוּחִין:
בְּאִתְנַבָּאָה לָךְ שְׁקַר בְּאַלְפָא לָךְ כִּדְבָן לְמִמְסַר יַת צַוְרָךְ כְּצַוְרֵי קְטִילֵי רַשִׁיעַיָא דִמְטָא יוֹם תְּבִירְהוֹן עִדַן סְעוּרַן בִּישַׁתְהוֹן:
אֲתִיב חַרְבָּא לִנְדָנָהּ בְּאַתְרָא דְאִתְבְּרֵיתוּן בַּאֲרַע תּוֹתָבוּתְכוֹן אֶתְפְּרַע מִנְכוֹן:
וְאֶשְׁפּוֹךְ עֲלָךְ חֵימָתִי בְּאֵשַׁת פּוּרְעֲנוּתִי אַתְקֵיף עֲלָךְ וְאֶמְסְרִינָךְ בְּיַד גִבָּרִין רַשִׁיעִין דְאוֹמָנִין לְחַבָּלָא:
לְעַמָמַיָא דְתַקִיפִין כְּאֶשְׁתָּא תִּתְמָסְרוּן לְאִשְׁתֵּיצָאָה חוֹבַת קְטוֹלָךְ תְּהֵי בְגוֹ אַרְעָא לָא יֵעוּל דָכְרָנִיךְ אֲרֵי אֲנָא יְיָ גְזֵרִית בְּמֵימְרִי:
