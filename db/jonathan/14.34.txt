וַהֲוָה פִּתְגַם נְבוּאָה מִן קֳדָם יְיָ עִמִי לְמֵימָר:
בַּר אָדָם אִתְנַבֵּי עַל פַּרְנָסֵי יִשְׂרָאֵל אִתְנַבֵּי וְתֵימַר לְהוֹן לְפַרְנְסַיָא כִּדְנַן אֲמַר יְיָ אֱלֹהִים וַי עַל פַּרְנְסַיָא דְיִשְׂרָאֵל דַהֲווֹ מְפַרְנְסִין נַפְשֵׁיהוֹן הֲלָא לְפַרְנָסָא עַמָא אִתְמְנִיאוּ פַּרְנָסַיָא:
יַת טוּבָא אַתִּין אָכְלִין וַעֲמַר נָקִי אַתּוּן לָבְשִׁין פַטִימַיָא אַתּוּן נָכְסִין לְעַמָא לֵיתֵיכוֹן מְפַרְנְסִין:
יַת כְּאֵיבַיָא לָא תַקֵיפְתּוּן וְלִמְרִיעַיָא לָא אֲסֵיתוּן וְלִתְבִירַיָא לָא צְמַדְתּוּן וְיַת דְאִיטַלְטַלוּ לָא אֲתֵיבְתּוּן וְיַת דַאֲבַדוּ לָא בְעִיתוּן וּבְתוּקְפָא פְלַחְתּוּן בְּהוֹן וּבְקַשְׁיוּ:
וְאִתְבַּדָרוּ מִבְּלִי פַרְנֵס וְאִתְמְסָרוּ לְאִשְׁתֵּיצָאָה לְכָל מַלְכְּוַת עַמְמַיָא וְאִיטַלְטָלוּ:
טְעוֹ עַמִי בְּכָל טוּרַיָא וְעַל כָּל רָמָא מְנַטְלָא וְעַל כָּל אַפֵּי אַרְעָא אִתְבְּדָרוּ עַמִי וְלֵית דְתָבַע וְלֵית דְבָעֵי:
בְּכֵן פַּרְנָסִין קַבִּילוּ יַת פִּתְגָמָא דַיָי:
קַיָם אֲנָא אֲמַר יְיָ אֱלֹהִים אִם לָא חֲלַף דַהֲווֹ עַמִי לְבִזָא וְאִתְמְסַרוּ עַמִי לְאִשְׁתֵּיצָאָה לְכָל מַלְכְּוַת עַמְמַיָא מִבְּלִי פַּרְנָס וְלָא תְבָעוּ פַּרְנָסַיָא יַת עַמִי וּפַרְנִיסוּ פַּרְנָסַיָא יַת נַפְשֵׁיהוֹן וְיַת עַמִי לָא פַרְנִיסוּ:
בְּכֵן פַּרְנָסַיָא רַשִׁיעַיָא תּוּבוּ לְאוֹרַיְתָא וַאֲנָא עֲתִיד לְרַחָמָא עֲלֵיכוֹן אֲצִיתוּ לְאוּלְפַן אוֹרַיְתָא וְקַבִּילוּ פִּתְגָמָא דַיָי:
כִּדְנַן אֲמַר יְיָ אֱלֹהִים הָא אֲנָא שְׁלַח רוּגְזִי עַל פַּרְנָסַיָא וְאֶתְבַּע יַת עַמִי מִידֵיהוֹן וַאֲבַטְלִינוּן מִלְפַרְנָסָא עַמָא וְלָא יְפַרְנְסוּן עוֹד פַרְנָסַיָא נַפְּשֵׁיהוֹן וֶאֱשֵׁיזֵיב עַמִי מִידֵיהוֹן וְלָא יִתְמַסְרוּן בִּידֵיהוֹן לְאִשְׁתֵּיצָאָה:
אֲרֵי כִּדְנַן אֲמַר יְיָ אֱלֹהִים הָא אֲנָא מִתְגְלִי וְאֶתְבַּע יַת עַמִי וַאֲבַקְרִינוּן:
כְּמָא דִמְבַקֵר רָעֲיָא עֶדְרֵיהּ בְּיוֹמָא דַהֲוָה בְּגוֹ עָנֵיהּ וּמַפְרֵשׁ לְהוֹן כֵּן אֲבַקֵר יַת עַמִי וֶאֱשֵׁיזֵיב יַתְהוֹן מִכָּל אַתְרַיָא דְאִתְבַּדָרוּ לְתַמָן בְּיוֹם עֲנָנָא וַאֲמִטְתָא:
וְאַפִיקִינוּן מִבֵּינֵי עַמְמַיָא וְאַכְנְשִׁינוּן מִן מְדִינָתָא וְאָעִילִינוּן לְאַרְעֲהוֹן וַאֲפַּרְנְסִינוּן עַל טוּרֵי יִשְׂרָאֵל בְּפַצִידַיָא וּבְכָל מוֹתְבֵי אַרְעָא:
בְּפַרְנוּס טַב אֲפַרְנֵיס יַתְהוֹן וּבְטוּר קוּדְשָׁא דְיִשְׂרָאֵל יְהֵיבֵית מֵישְׁרֵיהוֹן תַּמָן יִשְׁרוּן בַּאֲתַר דְיָאֵי וּפַרְנוּס טַב יִתְפַּרְנְסוּן עַל טוּרֵי יִשְׂרָאֵל:
אֲנָא אֲפַרְנֵס יַת עַמִי וַאֲנָא אַשְׁרִינוּן לְרוֹחֲצָן אֲמַר יְיָ אֱלֹהִים:
יַת דַאֲבָּדוּ אֶתְבַּע וְיַת דְאִיטַלְטָלוּ אֲתִיב וְלִתְבִירַיָא אֶצְמוֹד וְלִמְרִיעַיָא אֱסוֹבַר וְיַת חֲטָאַיָא וְיַת חַיָבַיָא אֶשֵׁיצֵי וַאֲפַרְנֵיס עַמִי בְדִינָא:
וְאַתּוּן עַמִי כִּדְנַן אֲמַר יְיָ אֱלֹהִים הָא אֲנָא דָאֵין בֵּין גְבַר לִגְבַר לַחֲטָאַיָא וּלְחַיָבַיָא:
הַזְעֵיר לְכוֹן דְפַרְנוּס טַב אַתּוּן מִתְפַּרְנְסִין וּמוֹתַר פַּרְנוּסֵכוֹן דָיְשִׁין בְּרַגְלֵיכוֹן וּמִשְׁתֵּי מְתַקֵן אַתּוּן שָׁתָן וְיַת מוֹתְרָא בְּרַגְלֵיכוֹן אַתּוּן דָיְשִׁין:
וְעַמִי מוֹתַר מֵיכַל שַׁמְשֵׁיכוֹן אָכְלִין וּמוֹתַר מַשְׁתֵּי שַׁמְשֵׁיכוֹן שָׁתָן:
בְּכֵן כִּדְנַן אֲמַר יְיָ אֱלֹהִים לְהוֹן הָא אֲנָא מִתְגְלֵי וְאֵידוּן בֵּין גְבַר עַתִּיר וּבֵין גְבַר מִסְכֵּן:
חֲלַף דְבִרְשַׁע וּבִדְרַע דְחֵיקְתּוּן וּבְתוּקְפָא תְבַרְתּוּן כָּל כְּאַיבַיָא עַד דְבַדַרְתּוּן יַתְהוֹן לְבֵינֵי מְדִינָתָא:
וְאֶפְרוֹק לְעַמִי וְלָא יִתְמַסְרוּן עוֹד לְבִזָא וְאֵידוּן בֵּין גְבַר לִגְבַר:
וְאָקֵים עֲלֵיהוֹן פַּרְנֵס חַד וִיפַרְנֵיס יַתְהוֹן וְהוּא יְהֵי לְהוֹן לְפַרְנֵס:
וַאֲנָא יְיָ אֶהֱוֵי לְהוֹן לֵאלָהָא וְעַבְדִי דָוִד מַלְכָּא בֵינֵיהוֹן אֲנָא יְיָ גְזָרֵית בְּמֵימְרִי:
וְאֶגְזַר לְהוֹן קְיָם שְׁלָם וַאֲבַטֵל חַיְתָא בִּישְׁתָא מִן אַרְעָא וִיתִיבוּן בְּמַדְבְּרָא לְרוֹחֲצָן וְיִתְעַתְּקוּן בִּרְחוּשָׁא:
וְאַשְׁרֵי יַתְהוֹן סְחוֹר סְחוֹר לְבֵית מַקְדְשִׁי וִיהוֹן מְבוֹרְכָן וְאָחֵית לְהוֹן מְטַר בַּכִּיר בְּעִדָנֵיהּ מַטְרִין דְבָרְכָן יְהוֹן:
וְיִתֵּן אִילַן חַקְלָא יַת אִבֵּיהּ וְאַרְעָא תִּתֵּן עֲלַלְתָּהּ וִיהוֹן עַל אַרְעֲהוֹן לְרוֹחֲצָן וְיֵדְעוּן אֲרֵי אֲנָא יְיָ כַּד אֶתְבַּר יַת נִיר תּוּקְפֵיהוֹן וְאֵשֵׁיזְבִינוּן מִיַד דִמְפַלְחִין בְּהוֹן:
וְלָא יְהוֹן עוֹד בִּזָא לְעַמְמַיָא וּמַלְכְּוַת אַרְעָא לָא יְשֵׁיצִינוּן וְיַתְבוּן לְרוֹחֲצָן וְלֵית דְמָנִיד:
וְאָקִים לְהוֹן נִצְבָא לְקַיָמָא וְלָא יְהוֹן עוֹד מְטַלְטְלֵי כַּפְנָא בְּאַרְעָא וְלָא יְקַבְּלוּן עוֹד אִתְכְּנָעוּת עַמְמַיָא:
וְיֵדְעוּן אֲרֵי אֲנָא יְיָ אֱלָהָהוֹן מֵימְרִי בְּסַעֲדְהוֹן וְאִינוּן עַמִי בֵּית יִשְׂרָאֵל אֲמַר יְיָ אֱלֹהִים:
וְאַתּוּן עַמִי עַם דְאִתְקְרֵי שְׁמִי עֲלֵיהוֹן בֵּית יִשְׂרָאֵל אַתּוּן אֲנָא אֱלָהָכוֹן אֲמַר יְיָ אֱלֹהִים:
