וַהֲוָה כַּד שֵׁיצֵי שְׁלֹמֹה לְמִבְנֵי יַת בֵּית מַקְדְשָׁא דַייָ וְיַת בֵּית מַלְכָּא וְיַת כָּל רְעוּת שְׁלֹמֹה דְאִתְרְעֵי לְמֶעְבַּד:
וְאִתְגְלֵי יְיָ לִשְׁלֹמֹה תִּנְיָנוּת כְּמָא דְאִתְגְלֵי לֵיהּ בְּגִבְעוֹן:
וַאֲמַר לֵיהּ שְׁמִיעַ קֳדָמַי צְלוֹתָךְ וּבָעוּתָךְ דִבְעֵיתָא מִן קֳדָמַי אַקְדֵישִׁית יַת בֵּיתָא הָדֵין דִי בְנֵיתָא לְאַשְׁרָאָה שְׁכִנְתִּי תַמָן עַד עַלְמָא וּתְהֵי שְׁכִנְתִּי שַׁרְיָא בֵיהּ אִם רְעוּתִי מִתְעַבְּדָא תַמָן כָּל יוֹמַיָא:
וְאַתְּ אִם תְהַךְ קֳדָמַי כְּמָא דְהַלִיךְ דָוִד אֲבוּךְ בְּקַשִׁיטוּת לִבָּא וּבִתְרִיצוּתָא לְמֶעְבַּד כְּכֹל דְפַקֵידְתָּךְ קְיָמַי וְדִינַי תִּטַר:
וַאֲקֵים יַת כּוּרְסֵי מַלְכוּתָךְ עַל יִשְׂרָאֵל לְעַלָם כְּמָא דְמַלֵלִית עִם דָוִד אֲבוּךְ לְמֵימַר לָא יִפְסוֹק לָךְ גְבַר מֵעַל כּוּרְסֵי מַלְכוּתָא דְיִשְׂרָאֵל:
אִם מֵיתַב תְּתוּבוּן אַתּוּן וּבְנֵיכוֹן מִבָּתַר פּוּלְחָנִי וְלָא תִטְרוּן פִּקוּדַי קְיָמַי דִי יְהָבִית קֳדָמֵיכוֹן וְתַהֲכוּן וְתִפְלְחוּן לְטַעֲוַת עַמְמַיָא וְתִסְגְדוּן לְהוֹן:
וַאֲשֵׁיצֵי יַת יִשְׂרָאֵל מֵעַל אַפֵּי אַרְעָא דִי יְהָבִית לְהוֹן וְיַת בֵּיתָא דְאַקְדֵישִׁית לִשְׁמִי אַרְחִיק מִקֳבֵיל מֵימְרִי וִיהֵי יִשְׂרָאֵל לִמְתַל וּלְשׁוֹעֵי בְּכָל עַמְמַיָא:
וּבֵיתָא הָדֵין דַהֲוָה עִילָאֵי יְהֵי חֲרִיב כָּל דְיֶעֶבַּר עֲלוֹהִי יַכְלֵי וְיָנוּד וְיֵימְרוּן עַל מָה עֲבַד יְיָ כְּדֵין לְאַרְעָא הָדָא וּלְבֵיתָא הָדֵין:
וְיֵמְרוּן עַל דִי שְׁבַקוּ יַת פּוּלְחָנָא דַייָ אֱלָהָהוֹן דְאַפֵיק יַת אֲבָהַתְהוֹן מֵאַרְעָא דִמִצְרַיִם וְאַתְקִיפוּ בְּטַעֲוַת עַמְמַיָא וּסְגִידוּ לְהוֹן וּפְלָחוּנוּן עַל כֵּן אַיְתֵי יְיָ עֲלֵיהוֹן יַת כָּל בִּישְׁתָּא הָדָא:
וַהֲוָה מִסוֹף עַשְׂרִין שְׁנִין דִי בְנָא שְׁלֹמֹה יַת תְּרֵין בָּתַּיָא יַת בֵּית מַקְדְשָׁא דַייָ וְיַת בֵּית מַלְכָּא:
חִירָם מַלְכָּא דְצוֹר סוֹבַר יַת שְׁלֹמֹה בַּאֲעֵי אַרְזִין וּבַאֲעֵי בִירְוָן וּבְדַהֲבָא לְכָל צוֹרְכֵיהּ בְּכֵן יָהִיב מַלְכָּא שְׁלֹמֹה לְחִירָם עַסְרִין קִרְוִין בְּאַרְעָא גְלִילָא:
וּנְפַק חִירָם מִצוֹר לְמִסְעַר יַת קִירְוַיָא דִיהַב לֵיהּ שְׁלֹמֹה וְלָא כַשְׁרִין בְּעֵינוֹהִי:
וַאֲמַר מַה קִרְוַיָא הָאִלֵין דִי יְהַבְתְּ לִי אֲחִי וּקְרָא לְהוֹן אַרְעָא כָּבוּל עַד יוֹמָא הָדֵין:
וּשְׁדַר חִירָם לְמַלְכָּא מְאָה וְעַשְׂרִין כִּכָּרֵי דַהֲבָא:
וְדֵין פִּתְגָם מַסְגֵי מִסִין דְמַנֵי מַלְכָּא שְׁלֹמֹה לְמִבְנֵי יַת בֵּית מַקְדְשָׁא דַייָ וְיַת בֵּיתֵיהּ וְיַת מִלֵיתָא וְיַת שׁוּרָא דִירוּשְׁלֵם וְיַת חָצוֹר וְיַת מְגִדוֹ וְיַת גָזֶר:
פַּרְעֹה מַלְכָּא דְמִצְרַיִם סְלִיק וּכְבַשׁ יַת גֶזֶר וְאוֹקְדָהּ בְּנוּרָא וְיַת כְּנַעֲנָאָה דְיָתֵיב בְּקַרְתָּא קְטַל וִיהֲבָהּ מַתְנָן לִבְרַתֵּיהּ אִתַּת שְׁלֹמֹה:
וּבְנָא שְׁלֹמֹה יַת גָזֶר וְיַת בֵּית חוֹרוֹן אַרְעָאָהּ:
וְיַת בַּעֲלַת וְיַת תַּדְמוּר בְּמַדְבְּרָא בְּאַרְעָא:
וְיַת כָל קִירְוֵי אוֹצָרַיָא דִי הֲוָאָה לִשְׁלֹמֹה וְיַת קִירְוֵי רְתִיכַיָא וְיַת קִירְוֵי פָּרָשַׁיָא וְיַת כָּל רְעוּת שְׁלֹמֹה דְאִתְרְעֵי לְמִבְנֵי בִירוּשְׁלֵם וּבְלִיבְנָן וּבְכָל אַרַע שׁוּלְטָנֵיהּ:
כָּל עַמָא דְאִשְׁתָּאַר מִן אֱמוֹרָאָה חִתָּאֵי פְרִיזָאֵי חִוָאֵי וִיבוּסָכאֵי דְלָא מִבְּנֵי יִשְׂרָאֵל אִינוּן:
בְּנֵיהוֹן דְאִשְׁתָּאֲרוּ בַתְרֵיהוֹן בְּאַרְעָא דְלָא יְכִילוּ בְּנֵי יִשְׂרָאֵל לְגַמָרוּתְהוֹן וּמַנִינוּן שְׁלֹמֹה לְמַסְקֵי מִסִין עַד יוֹמָא הָדֵין:
וּמִבְּנֵי יִשְׂרָאֵל לָא יְהַב שְׁלֹמֹה לְמֶהֱוֵי פַלָחִין אֲרֵי אִינוּן גְבַר עָבְדֵי קְרָבָא וְעַבְדוֹהִי וְרַבְרְבָנוֹהִי וְגִבָּרוֹהִי וְרַבָּנֵי רְתִיכוֹהִי וּפָרָשׁוֹהִי:
אִילֵין רַבָּנֵי אִסְטַרְטִיגַיָא דִי מְמַנָן עַל עֲבִדְתָּא דִשְׁלֹמֹה חֲמֵשׁ מְאָה וְחַמְשִׁין דִי מְפַלְחִין בְּעַמָא דְעָבְדִין בְּעִבִדְתָּא:
בְּרַם בַת פַּרְעֹה סְלֵיקַת מִקַרְתָּא דְדָוִד לְבֵיתָא דִי בְנָא לָהּ בְּכֵן בְּנָא יַת מַלֵיתָא:
וְאַסֵיק שְׁלֹמֹה תְּלַת זִמְנִין בְּשַׁתָּא עֲלַוָן וְנִכְסַת קוּדְשִׁין עַל מַדְבְּחָא דִבְנָא קֳדָם יְיָ וּמַקְטַר עֲלוֹהִי קְטוֹרֶת בּוֹסְמַיָא דִי קֳדָם יְיָ וְשַׁלִים יַת בֵּיתָא:
וּסְפִינָתָא עֲבַד שְׁלֹמֹה מַלְכָּא בְּעֶצְיוֹן גֶּבֶר דְעִם אֵילוֹת עַל כֵּיף יַמָא דְסוּף בְּאַרְעָא דֶאֱדוֹם:
וּשְׁדַר חִירָם בִּסְפִינָתָא יַת עַבְדוֹהִי גַבְרִין סַפָנִין דְאוּמָנִין לְדַבָּרָא בְּיַמָא עִם עַבְדֵי שְׁלֹמֹה:
וַאֲתוֹ לְאוֹפִיר וּנְסִיבוּ מִתַּמָן דַהֲבָא אַרְבַּע מְאָה וְעַשְׂרִין כִּכָּרִין וְאַיְתִיאוּ לְוַת מַלְכָּא שְׁלֹמֹה:
