וַהֲוָה פִּתְגַם נְבוּאָה מִן קֳדָם יְיָ עִמִי לְמֵימָר:
בַּר אָדָם אִתְנַבֵּי וְתֵימַר כִּדְנַן אֲמַר יְיָ אֱלֹהִים אַיְלִילוּ וַאֲמָרוּ וַי מִן קֳדָם יוֹמָא:
אֲרֵי קָרִיב יוֹמָא וְקָרִיב יוֹמָא דַעֲתִיד לְמֵיתֵי מִן קֳדָם יְיָ יוֹם עַנַן זִיעַ תְּבַר עַמְמַיָא יְהֵי:
וְתָחוֹל חַרְבָּא בְּמִצְרַיִם וִיהֵי זִיעַ בְּכוּשׁ כַּד יִתְרְמוּן קְטִילִין בְּמִצְרַיִם וְיִשְׁבּוּן אִתְרְגוּשְׁתָּהּ וְיִתְפַּגְרוּן אוּשְׁוָתָהָא:
כּוּשָׁאֵי וּפוּטָאֵי וְלוּדָאֵי וְכָל סוּמְכָוָתָא וְכוּב וּבְנֵי אַרְעָא קְיָמָא עִמְהוֹן בְּחַרְבָּא יִתְקַטְלוּן:
כִּדְנַן אֲמַר יְיָ וְיִתְקַטְלוּן סַעֲדֵי מִצְרַיִם וְיִבַטֵל יְקַר תּוּקְפָהּ מִמִגְדוֹל סְוָנֵה בְּחַרְבָּא יִתְקַטְלוּן בָּהּ אֲמַר יְיָ אֱלֹהִים:
וְיִצְדוּן בְּגוֹ מְדִינָן דְצַדְיָן וְקִרְווֹהִי בְּגוֹ קִרְוֵין דַהֲוָאָה חָרְבָן יֶהֱוְיָן:
וְיֵדְעוּן אֲרֵי אֲנָא יְיָ בִּדְאֶתֵּן עַמְמַיָא דְתַקִיפִין כְּאֶשְׁתָּא עַל מִצְרַיִם וְיִתְּבְרוּן כָּל סְעִידָהָא:
בְּעִדָנָא הַהִיא יִפְקוּן אִזְגַדִין מִן קֳדָמַי בְּלִגְיוֹנִין לַאֲזָעָא כּוּשׁ דְשַׁרְיָא לְרוֹחֲצָן וִיהֵי זִיעָא בְּהוֹן כְּיוֹם מִצְרַיִם אֲרֵי הָא אַתְיָא:
כִּדְנַן אֲמַר יְיָ אֱלֹהִים וַאֲבַטֵיל יַת אִתְרְגוּשַׁת מִצְרַיִם בְּיַד נְבוּכַדְרֶאצַר מַלְכָּא דְבָבֶל:
הוּא וְעַמֵיהּ עִמֵיהּ תְּקִיפֵי עַמְמַיָא אֶתֵּן לְחַבָּלָא אַרְעָא וְיִשְׁלְפוּן סַיְפֵיהוֹן עַל מִצְרַיִם וְיִמְלוֹן יַת אַרְעָא קְטִילִין:
וְאֶתֵּן נַהֲרֵיהוֹן יַבֶּשְׁתָּא וְאֶמְסַר יַת אַרְעָא בְּיַד מַבְאֲשִׁין וְאַצְדֵי אַרְעָא וּמְלָהָא בְּיַד רְשִׁיעִין אֲנָא יְיָ גְזָרִית בְּמֵימְרִי:
כִּדְנַן אֲמַר יְיָ אֱלֹהִים וְאַבֵיד דְפָלְחִין לְטַעֲוָתָא וֶאֱשֵׁיצֵי דְפָלְחִין לְצַלְמָנַיָא מִמָפֵס וּמַלְכָּא מֵאַרְעָא דְמִצְרַיִם דַחֲלָא בְּמִצְרַיִם:
וְאַצְדֵי יַת פַּתְרוֹס וְאַיְתֵי עַמָא דְתַקִיפִין כְּאֶשְׁתָּא עַל טָאנֵיס וְאַעְבֵּיד פּוּרְעֲנוּת דִינִין בְּאַלַכְסַנְדַרְיָא:
וְאֶשְׁפּוֹךְ חֵימְתִי עַל סִין תְּקוֹף מִצְרַיִם וֶאֱשֵׁיצֵי יַת אִתְרְגוּשַׁת אֲלַכְסַנְדַרְיָא:
וְאַיְתֵי עַמְמַיָא דְתַקִיפִין כְּאֶשְׁתָּא בְּמִצְרַיִם מִזַע תְּזוּעַ סִין וַאֲלַכְסַנְדַרְיָא תְּהֵי מְתַרְעָא מִשׁוּר וּמָפֵיס בַּעֲלֵי דְבָבוּ יִקְפוּנָהּ יוֹם יוֹם:
עוּלֵימֵי אָוֶן וּפִיבֶסֶת בְּחַרְבָּא יִתְקַטְלוּן וּפָלְחֵיהוֹן בְּשִׁבְיָא יָהֲכוּן:
וְעַל תַּחְפַּנְחֵס אַיְתֵי חֲשׁוֹךְ בִּימָם כַּד אֶתְבַּר תַּמָן יַת תְּקוֹף מִצְרַיִם וִיבַטֵל בָּהּ יְקַר תּוּקְפָהּ מְלַךְ בְּמַשִׁרְיָתֵיהּ יַחְפִּינָהּ כַּעֲנָנָא דְסָלֵיק וְחָפֵי יַת אַרְעָא וְיַתְבֵי כַפְרָנָהָא בְּשִׁבְיָא יָהֲכוּן:
וְאַעְבֵּד פּוּרְעֲנוּת דִינִין בְּמִצְרַיִם וְיֵדְעוּן אֲרֵי אֲנָא יְיָ:
וַהֲוָה בַּחֲדָא עַשְׂרֵי שְׁנִין בְּקַדְמָאָה בְּשַׁבְעָא לְיַרְחָא הֲוָה פִתְגַם נְבוּאָה מִן קֳדָם יְיָ עִמִי לְמֵימָר:
בַּר אָדָם יַת תְּקִיף פַּרְעֹה מַלְכָּא דְמִצְרַיִם תַּבְרֵית וְהָא לָא מִתַּקְפָא לְאִתַּסָאָה לְמִתַּן עֵיצָה לְמֶחְכַּם לְאִתַּקָפָא לְמֵיחַד מַלְכּוּ:
בְּכֵן כִּדְנַן אֲמַר יְיָ אֱלֹהִים הָא אֲנָא שְׁלַח רוּגְזִי עַל פַּרְעֹה מַלְכָּא דְמִצְרַיִם וְאֶתְּבַר יַת סַעֲדוֹהִי וְיַת שִׁלְטוֹנוֹהִי תַקִיפַיָא דִבְהוֹן הוּא תְבַר יַת מַלְכְּוָתָא וְאֶרְמֵי יַת תּוּקְפָא מִנֵיהּ:
וְאַגְלֵי יַת מִצְרָאֵי בֵּינֵי עַמְמַיָא וַאֲבַדְרִינוּן בִּמְדִינָתָא:
וְאַתְקֵיף יַת מַלְכוּת מַלְכָּא דְבָבֶל וְאֶתֵּן יַת תּוּקְפִי בִּידֵיהּ וְאֶתְּבַר יַת תְּקוֹף פַּרְעֹה וְיִנְהֵם נְהוֹם מְטַעֲנֵי חֶרֶב קֳדָמוֹהִי:
וְאַתְקֵיף יַת מַלְכוּת מַלְכָּא דְבָבֶל וּמַלְכוּת פַּרְעֹה תִּבְטֵל וְיֵדְעוּן אֲרֵי אֲנָא יְיָ כַּד אֶתֵּן תּוּקְפִי בְּיַד מַלְכָּא דְבָבֶל וְיִתְפְּרַע פּוּרְעֲנוּת מִן אַרְעָא דְמִצְרָיִם:
וְאַגְלֵי יַת מִצְרָאֵי לְבֵינֵי עַמְמַיָא וֶאֱבַדַר יַתְהוֹן בִּמְדִינָתָא וְיֵדְעוּן אֲרֵי אֲנָא יְיָ:
