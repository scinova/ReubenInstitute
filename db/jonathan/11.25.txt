וַהֲוָה בְשַׁתָּא תְשִׁיעֵיתָא לְמִמְלְכֵיהּ בְּיַרְחָא עֲשִׂירָאָה בְּעַשְׂרָא לְיַרְחָא אֲתָא נְבוּכַדְנֶאצַר מַלְכָּא דְבָבֶל הוּא וְכָל מַשְׁרִיתֵיהּ עַל יְרוּשְׁלֵם וּשְׁרָא עֲלָה וּבְנוֹ עֲלָה בִּרְקוּם סְחוֹר סְחוֹר:
וַעֲלַת קַרְתָּא בִּצְיָרָא עַד חֲדָא עַשְׂרֵי שְׁנִין לְמַלְכָּא צִדְקִיָהוּ:
בְּתִשְׁעָא לְיַרְחָא וּתְקֵיף כַּפְנָא בְּקַרְתָּא וְלָא הֲוָה מֵיכַל לְעַמָא דְאַרְעָא:
וְאִתְרָעַת קַרְתָּא וְכָל גַבְרֵי עַבְדֵי קְרָבָא בְּלֵילְיָא בְּאוֹרַח תַּרְעָא בֵּין שׁוּרַיָא דִי עַל גִינְתָא דְמַלְכָּא וְכַסְדָאֵי שָׁרָן עַל קַרְתָּא סְחוֹר סְחוֹר וַאֲזַל בְּאוֹרַח מֵישְׁרָא:
וּרְדָפוּ מַשִׁרְיַת כַּסְדָאֵי בָּתַר מַלְכָּא וְאַדְבִּיקוּ יָתֵיהּ בְּמֵישְׁרֵי יְרֵיחוֹ וְכָל מַשִׁרְיָתֵיהּ אִתְבַּדָרַת מִנֵיהּ:
וַאֲחָדוּ יַת מַלְכָּא וְאַסִיקוּ יָתֵיהּ לְוַת מַלְכָּא דְבָבֶל לְרִבְלָת וּמַלִילוּ עִמֵיהּ פִּתְגָמֵי דִינִין:
וְיַת בְּנוֹהִי דְצִדְקִיָהוּ נְכִיסוּ לְעֵינוֹהִי וְיַת עֵינֵי צִדְקִיָהוּ עַוֵר וְאַסְרֵיהּ בְשִׁלְשְׁלָן דִנְחַשׁ וְאוֹבִילֵיהּ לְבָבֶל:
וּבְיַרְחָא חֲמִישָׁאָה בְּשַׁבְעָא לְיַרְחָא הִיא שְׁנַת תְּשַׁע עַסְרֵי שְׁנִין לְמַלְכָּא נְבוּכַדְנֶאצַר מַלְכָּא דְבָבֶל אֲתָא נְבוּזַרְאֲדָן רַב קָטוֹלַיָא עַבְדָא דְמַלְכָּא דְבָבֶל לִירוּשְׁלֵם:
וְאוֹקִיד יַת מַקְדְשָׁא דַייָ וְיַת בֵּית מַלְכָּא וְיַת כָּל בָּתֵּי יְרוּשְׁלֵם וְיַת כָּל בָּתֵּי רַבְרְבַיָא אוֹקִיר בְּנוּרָא:
וְיַת שׁוּרֵי יְרוּשְׁלֵם סְחוֹר סְחוֹר תְּרָעוּ כָּל מַשִׁרְיַת כַּסְדָאֵי דְעִם רַב קָטוֹלַיָא:
וְיַת שְׁאָר עַמָא דְאִשְׁתַּאֲרוּ בְקַרְתָּא וְיַת שְׁמִיעַיָא דְאִשְׁתַּמָעוּ עַל מַלְכָּא דְבָבֶל וְיַת שְׁאָר הֲמוֹנָא אַגְלֵי נְבוּזַרְאֲדָן רַב קָטוֹלַיָא:
וּמַחֲשִׁיכֵי עַמָא דְאַרְעָא אַשְׁאַר רַב קָטוֹלַיָא לְמֶהֱוֵי מַפְלְחִין בְּחַקְלִין וּבְכַרְמִין:
וְיַת עֲמוּדֵי נְחָשָׁא דִבְבֵית מַקְדְשָׁא דַייָ וְיַת בְּסִיסַיָא וְיַת יַמָא דִנְחָשָׁא דִי בְּבֵית מַקְדְשָׁא דַייָ תְּבַרוּ כַסְדָאֵי וְאוֹבִילוּ יַת נְחָשֵׁיהוֹן לְבָבֶל:
וְיַת דוּדַיָא וְיַת מַגְרוֹפְיָתָא וְיַת מְזַמְרַיָא וְיַת בָּזִיכַיָא וְיַת כָּל מָנֵי נְחָשָׁא דַהֲווֹ מְשַׁמְשִׁין בְּהוֹן נְסִיבוּ:
וְיַת מַחְתְּיָתָא וְיַת מִזְרְקַיָא דִי דַהֲבָא דַהֲבָא וְדִי כַסְפָּא כַסְפָּא נְסִיב רַב קָטוֹלַיָא:
עֲמוּדַיָא תְרֵין וְיַמָא חַד וּבְסִיסַיָא דַעֲבַד שְׁלֹמֹה לְבֵית מַקְדְשָׁא דַייָ לָא הֲוָה מַתְקַל לִנְחֻשְׁתְּהוֹן דְכָל מְנַיָא הָאִלֵין:
תַּמְנֵי עַסְרֵי אַמִין רוּמֵיהּ דְעַמוּדָא חָד וּקְרוּנְתְּהוֹן עֲלוֹהִי דִנְחָשָׁא וְרוּמָא דִקְרוּנְתְּהוֹן תְּלַת אַמִין וּסְרִגְתָּא וְרִמוֹנַיָא עַל קְרוּנְתְּהוֹן סְחוֹר סְחוֹר כּוֹלָא נְחָשָׁא וּכְאִלֵין לְעַמוּדָא תִנְיָנָא עַל סְרִגְתָּא:
וּדְבַר רַב קָטוֹלַיָא יַת שְׂרָיָה כַּהֲנָא רַבָּא וְיַת צְפַנְיָה סְגַן כַּהֲנַיָא וְיַת תְּלָתָא אֲמַרְכְּלַיָא:
וּמִן קַרְתָּא דְבַר גַוָזָא חַד דְהוּא מְמַנָא עַל גַבְרֵי עָבְדֵי קְרָבָא וְחַמְשָׁא גַבְרִין דְחָזָן אַפֵּי מַלְכָּא דְאִשְׁתְּכָחוּ בְקַרְתָּא וְיַת סַפְרָא רַב חֵילָא דִכְנַשׁ יַת עַמָא דְאַרְעָא וְשִׁתִּין גַבְרָא מֵעַמָא דְאַרְעָא דְאִשְׁתְּכָחוּ בְקַרְתָּא:
וּדְבַר יַתְהוֹן נְבוּזַרְאֲדָן רַב קָטוֹלַיָא וְאוֹבִיל יַתְהוֹן לְוַת מַלְכָּא דְבָבֶל לְרִבְלָת:
וּמְחָא יַתְהוֹן מַלְכָּא דְבָבֶל וְקַטְלִינוּן בְּרִבְלָא בְּאַרְעָא דַחֲמָת וּגְלוֹ דְבֵית יְהוּדָה מֵעַל אַרְעֲהוֹן:
וְעַמָא דְאִשְׁתְּאַר בְּאַרְעָא דְבֵית יְהוּדָה דְאַשְׁאַר נְבוּכַדְנֶאצַר מַלְכָּא דְבָבֶל וּמַנֵי עֲלֵיהוֹן יַת גְדַלְיָה בַר אֲחִיקָם בַּר שָׁפָן:
וּשְׁמָעוּ כָל רַבָּנֵי חֵילְוָתָא אִינוּן וְגַבְרֵיהוֹן אֲרֵי מַנֵי עֲלֵיהוֹן מַלְכָּא דְבָבֶל יַת גְדַלְיָה וַאֲתוֹ לְוַת גְדַלְיָה לְמִצְפְּיָא וְיִשְׁמָעֵאל בַּר נְתַנְיָה וְיוֹחָנָן בַּר קָרֵחַ וּשְׂרָיָה בַר תַּנְחֻמֶת דְמִנְטוֹפָה וְיַאֲזַנְיָה בַר מַעֲכָת אִינוּן וְגַבְרֵיהוֹן:
וְקַיֵם לְהוֹן גְדַלְיָה וּלְגַבְרֵיהוֹן וַאֲמַר לְהוֹן לָא תִדְחֲלוּן מֵעַבְדֵי כַסְדָאֵי תְּבוּ בְאַרְעָא וּפְלָחוּ יַת מַלְכָּא דְבָבֶל וְיִטַב לְכוֹן:
וַהֲוָה בְּיַרְחָא שְׁבִיעָאָה אֲתָא יִשְׁמָעֵאל בַּר נְתַנְיָה בַּר אֱלִישָׁמָע מִזַרְעָא דְמַלְכוּתָא וְעַסְרָא גַבְרִין עִמֵיהּ וּמְחוֹ יַת גְדַלְיָה וּמִית וְיַת יְהוּדָאֵי וְיַת כַּסְדָאֵי דַהֲווֹ עִמֵיהּ בְּמִצְפְּיָא:
וְקָמוּ כָל עַמָא מִזְעֵירָא וְעַד רַבָּא וְרַבָּנֵי חֵילְוָתָא וַאֲתוֹ לְמִצְרָיִם אֲרֵי דְחִילוּ מִן קֳדָם כַסְדָאֵי:
וַהֲוָה בִּתְלָתִין וּשְׁבַע שְׁנִין לְגָלוּת יְהוֹיָכִין מֶלֶךְ שִׁבְטָא דְבֵית יְהוּדָה בִּתְרֵי עֲשַׂר יַרְחִין בְּעַשְׂרִין וְשַׁבְעָא לְיַרְחָא רַבֵּי אֱוִיל מְרוֹדָךְ מַלְכָּא דְבָּבֶל בִּשְׁנַת מַמְלְכֵיהּ יַת רֵישׁ יְהוֹיָכִין מֶלֶךְ שִׁבְטָא דְבֵית יְהוּדָה מִבֵּית אֲסִירֵי:
וּמַלֵיל עִמֵיהּ פִּתְגָמִין תַּקְנִין וִיהַב יַת כּוּרְסֵיהּ עֵיל מִן כּוּרְסֵי מַלְכַיָא דִי עִמֵיהּ בְּבָבֶל:
וְשַׁנִי יַת לְבוּשֵׁי אִסוּרֵיהּ וְאָכִיל לַחֲמָא תְדִירָא קֳדָמוֹהִי כָּל יוֹמֵי חַיוֹהִי:
וְשֵׁירוּתֵיהּ שֵׁירוּת תְּדִירָא מִתְיַהֲבָא לֵיהּ מִן קֳדָם מַלְכָּא פִּתְגַם יוֹם בְּיוֹמֵיהּ כָּל יוֹמֵי חַיוֹהִי:
