בְּעִדָנָא הַהִיא יְהֵי אוּלְפַן אוֹרַיְתָא גְלֵי כְּמַבּוּעַ דְמַיִין לְבֵית דָוִד וּלְיָתְבֵי יְרוּשְׁלֵם וְאֶשְׁבּוֹק לְחוֹבֵיהוֹן כְּמָא דִמְדַכָּן בְּמֵי אַדְיוּתָא וּבִקְטָם תּוֹרָתָא דְחַטָאתָא:
וִיהֵי בְעִדָנָא הַהִיא אֲמַר יְיָ צְבָאוֹת אֱשֵׁיצֵי יַת שׁוּם טַעֲוַת עַמְמַיָא מִן אַרְעָא וְלָא יְהֵי דָכְרָן מִשׁוּמְהוֹן עוֹד וְאַף יַת נְבִיֵי שִׁקְרָא וְיַת רוּחַ מְסַאֲבָא אֲבַטֵיל מִן אַרְעָא:
וִיהֵי אֲרֵי אִתְנַבֵּי גְבַר עוֹד וְיֵימְרוּן לֵיהּ אֲבוּהִי וְאִמֵיהּ יַלְדוֹהִי לָא תִתְקַיֵם אֲרֵי שִׁקְרָא אִתְנַבֵּיתָא בִּשְׁמָא דַייָ וְיֶחְדוּן בֵּיהּ אֲבוּהִי וְאִמֵיהּ יַלְדוֹהִי בְּאִתְנַבָּיוּתֵיהּ:
וִיהֵי בְּעִדָנָא הַהִיא יִבְהֲתוּן נְבִיֵי שִׁקְרָא גְבַר מִנְבוּאֲתֵהּ בְּאִתְנַבָּיוּתֵיהּ וְלָא יִתְנַבּוּן עוֹד שְׁקַר בְּדִיל לְאַלְפָא כַּדִיבִין:
וְיֵימַר לָא נְבִיָא אֲנָא גְבַר פָּלַח בְּאַרְעָא אֲנָא אֲרֵי אֱנָשָׁא אַקְנִיַנִי מִזְעוּרִי:
וְיֵימַר לֵיהּ מָה מְחָתָא הָאִלֵין דַאֲתָאָה עֲלָנָא הֲלָא בְּעוֹבָדֵי יְדָךְ וְיֵימַר יָאוּת לָקֵינָא בְּחוֹבִין דְרַחֲמָנָא:
חַרְבָּא אִתְגְלָא עַל מַלְכָּא דְבָבֶל וְעַל שִׁלְטוֹנָא חַבְרֵיהּ דִכְוָתֵיהּ דְדָמֵיהּ לֵיהּ אֲמַר יְיָ צְבָאוֹת קְטוֹל יַת מַלְכָּא דְבָבֶל וְיִתְבַּדְרוּן שִׁלְטוֹנַיָא וַאֲתִיב יַת מְחַת גְבוּרְתִּי עַל תִּנְיָנַיָא:
וִיהֵי בְּכָל אַרְעָא אֲמַר יְיָ תְּרֵין חוּלְקִין דְבָהּ יִשְׁתֵּיצוּן יְסוּפוּן (וּתְלִיתָאָה) יִשְׁתָּאֵר בָּהּ:
וְאָעֵיל יַת תִּלְתָּא בְּעָקָא בְּכוּר דְאָשָׁא וְאַצְרוֹפִינוּן כְּמָא דְצָרְפִין יַת כַּסְפָּא וְאֶבְחֲרִינוּן כְּמָא דְבָחֲרִין יַת דַהֲבָה הוּא יְצַלֵי בִשְׁמִי וַאֲנָא אֲקַבֵּיל צְלוֹתֵיהּ אֲמָרֵית עַמִי הוּא וְהוּא יֵימַר יְיָ אֱלֹהָי:
