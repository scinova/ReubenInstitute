וְאַתְּ בֵּית לֶחֶם אֶפְרָתָה כִּזְעֵיר הֲוֵיתָא לְאִתְמַנָאָה בְּאַלְפַיָא דְבֵית יְהוּדָה מִנָךְ קֳדָמַי יִפּוֹק מְשִׁיחָא לְמֶהֱוֵי עֲבֵיד שׁוּלְטַן עַל יִשְׂרָאֵל דִי שְׁמֵיהּ אָמִיר מִלְקָדְמִין מִיוֹמֵי עָלְמָא:
בְּכֵן יִתְמַסְרוּן כְּעִדַן יָלֵידְתָּא לְמֵילַד וּשְׁאַר אֲחֵיהוֹן יִסְתַּמְכוּן עֲלֵיהוֹן בְּנֵי יִשְׂרָאֵל:
וִיקוּם וְיִשְׁלוֹט בִּתְקוֹף מִן קֳדָם יְיָ בִּרְבוֹת שְׁמָא דַייָ אֱלָהֵיהּ וִיתוּבוּן מִבֵּינֵי גָלוּתְהוֹן אֲרֵי כְעַן יַסְגֵי שְׁמֵיהּ עַד סְיָפֵי אַרְעָא:
וִיהֵי מִבְּכֵן שְׁלָמָא לָנָא אֲתוּרָאָה אֲרֵי יֵעוּל בְּאַרְעָנָא וַאֲרֵי יִדְרוֹךְ בְּבִרְנְיָתָנָא וּנְמַנֵי עֲלָנָא שַׁבְעָא מַלְכִין וּתְמַנְיָא רַבְרְבֵי אֱנָשָׁא:
וְיִרְעוּן יַת אַרְעָא דְאַתּוּר בְּחַרְבָּא וְיַת אַרְעָא נִמְרוֹד בִּתְקוֹף כְּרַכָּהָא וִישֵׁיזְבִינָנָא מֵאַתּוּר דְלָא יֵעוֹל בְּאַרְעָנָא וְלָא יִדְרוֹךְ בִּתְחוּמָנָא:
וִיהֵי שְׁאָרָא דְבֵית יַעֲקֹב בְּגוֹ עַמְמִין סַגִיאִין כְּטַלָא קֳדָם יְיָ וְכִרְסִיסֵי מַלְקוֹשָׁא דִי עַל עִסְבָּא דִי לָא מוֹרִיךְ לֶאֱנָשָׁא וְלָא מַכְתַּר לְבַר אֱנָשָׁא:
וִיהֵי שְׁאָרָא דְיַעֲקֹב בְּעַמְמַיָא בְּגוֹ עַמְמִין סַגִיאִין כְּאַרְיָא בִּבְעִירָא בְחוּרְשָׁא וּכְלַיְתָא בְּעֶדְרֵי עָנָא דְאִם יַעֲדֵי וְיִרְמוֹס וְיִקְטוֹל וְלֵית דִי מְשֵׁיזִיב:
סַגִי מִבְּכֵן תִּתְקֵיף יְדָךְ יִשְׂרָאֵל עַל סַנְאָךְ וְכָל בַּעֲלֵי דְבָבָךְ יִשְׁתֵּיצוּן:
וִיהֵי בְעִדָנָא הַהִיא אֲמַר יְיָ וֶאֱשֵׁיצֵי סוּסְוַת עַמְמַיָא מִבֵּינָךְ וְאוֹבֵיד רְתִיכֵיהוֹן:
וֶאֱשֵׁיצֵי קִרְוֵי עַמְמַיָא מֵאַרְעָךְ וַאֲפַגֵיר כָּל כְּרַכֵּיהוֹן תַּקִיפַיָא:
וֶאֱשֵׁיצֵי חֲרָשִׁין מִבֵּינָךְ וּמְעַנְנִין לָא יְהוֹן לָךְ:
וֶאֱשֵׁיצֵי צַלְמֵי עַמְמַיָא וְקָמָתְהוֹן מִבֵּינָךְ וְלָא תִשְׁתַּעְבֵּיד עוֹד לְעוֹבָדֵי יְדָךְ:
וְאַעֲקַר שְׁתִילֵי עַמְמַיָא מִבֵּינָךְ וֶאֱשֵׁיצֵי בַּעֲלֵי דְבָבָךְ:
וְאַעְבֵּיד בִּרְגַז וּבְחֵימָא יַת פּוּרְעֲנוּת דִין מִן עַמְמַיָא דְלָא קַבִּילוּ אוּלְפַן אוֹרַיְתָא:
