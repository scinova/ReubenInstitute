וַהֲוָה לִזְמַן יוֹמִין בְּיוֹמֵי חֲצַד חִטִין וְאַדְכַּר שִׁמְשׁוֹן יַת אִתְּתֵיהּ בְּגַדְיָא בַר עִזֵי וַאֲמַר אֵיעוּל לְוַת אִתְּתִי לְאִדְרוֹן בֵּית מִשְׁכְּבָא וְלָא שַׁבְקֵיהּ אֲבוּהָא לְמֵיעַל:
וַאֲמַר אֲבוּהָא מֵימַר אֲמַרִית אֲרֵי מִסְנָא סְנֵיתָהּ וִיהָבִיתָהּ לְחַבְרָךְ הֲלָא אֲחָתָהּ זְעֶרְתָּא שַׁפִירָא מִנָהּ תְּהֵי כְעַן לָךְ חִלוּפָא:
וַאֲמַר לְהוֹן שִׁמְשׁוֹן זַכַּאי אֲנָא זִמְנָא הָדָא מִפְּלִשְׁתָּאֵי אֲרֵי עָבִיד אֲנָא עִמְהוֹן בִּישָׁא:
וַאֲזַל שִׁמְשׁוֹן וַאֲחַד תְּלַת מֵאָה תַעֲלִין וּנְסִיב בְּעוּרַיָא וּקְטַר דַנְבָא בְּרֵישׁ דַנְבָא וְשַׁוִי בְעוּרָא חָד בֵּין תְּרֵין דִנְבַיָא בִּמְצִיעָן:
וְאַדְלֵיק אֶשְׁתָּא בִּבְעוּרַיָא וּשְׁלַח בְּקָמַת פְּלִשְׁתָּאֵי וְאַדְלֵיק מִגָדִישַׁיָא וְעַד כַּרְמַיָא וְעַד זֵיתָא:
וַאֲמַרוּ פְלִשְׁתָּאֵי מַן עֲבַד דָא וַאֲמַרוּ שִׁמְשׁוֹן חֲתָנֵיהּ דְתִמְנִי אֲרֵי נְסִיב יַת אִתְּתֵיהּ וְיַהֲבָהּ לְחַבְרֵיהּ וּסְלִיקוּ פְלִשְׁתָּאֵי וְאוֹקִידוּ יָתָהּ וְיַת אֲבוּהָא בְּנוּרָא:
וַאֲמַר לְהוֹן שִׁמְשׁוֹן אִם תַּעְבְדוּן כַּהֲדָא אֱלָהֵן כַּד אֶתְפְּרַע מִנְכוֹן וּבָתַר כֵּן אֶתְמְנָע:
וּמְחָא יָתְהוֹן פָּרָשִׁין עִם רִגְלָאִין מְחָא סַגִיאָה וּנְחַת וִיתֵיב בִּשְׁקֵיף כֵּיף עֵיטָם:
וּסְלִיקוּ פְלִשְׁתָּאֵי וּשְׁרוֹ בְּאַרְעָא דְבֵית יְהוּדָה וְאִתְרְטִישׁוּ עַל לֶחִי:
וַאֲמַרוּ אֱנָשֵׁי יְהוּדָה לְמָא סְלֵקְתּוּן עֲלָנָא וַאֲמַרוּ לְמֵיסַר יַת שִׁמְשׁוֹן סְלִיקְנָא לְמֶעְבַּד לֵיהּ כְּמָא דַעֲבַד לָנָא:
וּנְחָתוּ תְּלָתָא אַלְפִין גַבְרָא מִדְבֵית יְהוּדָה לִשְׁקֵיף כֵּיף עֵיטָם וַאֲמַרוּ לְשִׁמְשׁוֹן הֲלָא יְדַעְתָּא אֲרֵי שַׁלְטִין בָּנָא פְּלִשְׁתָּאֵי וּמַה דָא עֲבַדְתְּ לָנָא וַאֲמַר לְהוֹן כְּמָא דַעֲבַדוּ לִי כֵּן עַבְדֵית לְהוֹן:
וַאֲמַרוּ לֵיהּ לְמֵיסְרָךְ נְחַתְנָא לְמִמְסְרָךְ בִּידָא דִפְלִשְׁתָּאֵי וַאֲמַר לְהוֹן שִׁמְשׁוֹן קַיִימוּ לִי דִלְמָא תִשְׁלְטוּן בִּי אַתּוּן:
וַאֲמַרוּ לֵיהּ לְמֵימָר לָא אֲרֵי מֵיסַר נִיסְרִינָךְ וְנִמְסְרִינָךְ בִּידֵיהוֹן וּמִקְטַל לָא נִקְטְלִנָךְ וְאַסְרוּהִי בְּתַרְתֵּין גְדִילַן חַדְתִּין וְאַסְקוּהִי מִן כֵּיפָא:
הוּא אֲתָא עַד לֶחִי וּפְלִשְׁתָּאֵי יַבִּיבוּ לָקֳדָמוּתֵיהּ וּשְׁרַת עֲלוֹהִי רוּחַ גְבוּרָא מִן קֳדָם יְיָ וַהֲוָאָה גְדִילָתָא דְעַל דַרְעוֹהִי כְּכִתָּנָא דְאָרַח בֵּיהּ נוּרָא וְאִתְמְסִיוּ אֱסָרוֹהִי מֵעַל יְדוֹהִי:
וְאַשְׁכַּח לוֹעָא דַחֲמָרָא בְּטִינָתָא וְאוֹשִׁיט יְדֵיהּ וְנַסְבָהּ וּקְטַל בָּהּ אֲלַף גַבְרָא:
וַאֲמַר שִׁמְשׁוֹן בְּלוֹעָא דַחֲמָרָא רְמִיתִנוּן דְגוֹרִין בְּלוֹעָא דַחֲמָרָא קְטָלִית אֲלַף גַבְרָא:
וַהֲוָה כְּשֵׁיצָיוּתֵיהּ לְמַלָלָא וּרְמָא לוֹעָא מִן יְדֵיהּ וּקְרָא לְאַתְרָא הַהוּא רָמַת לְחִי:
וּצְחִי לַחֲדָא וְצַלִי קֳדָם יְיָ וַאֲמַר אַתְּ עֲבַדְתָּ עַל יַד עַבְדָךְ יַת פּוּרְקָנָא רַבָּא הָדָא וּכְעַן אֵימוּת בְּצָחוּתָא וְאִתְמְסַר בִּידָא דַעֲרֵלַיָא:
וּבְזַע יְיָ יַת כֵּיפָא דִי בְּלוֹעָא וּנְפַקוּ מִנֵיהּ מַיָא וּשְׁתִי וְתָבַת רוּחֵיהּ וַחֲיָא עַל כֵּן אִתְקְרֵי שְׁמָהּ עֵינָא דְאִתְיְהִיבַת בִּצְלוֹתָא דְשִׁמְשׁוֹן הִיא קַיְמָא בַּלֶחִי עַד יוֹמָא הָדֵין:
וְדָן יַת יִשְׂרָאֵל בְּיוֹמֵי פְּלִשְׁתָּאֵי עַסְרִין שְׁנִין:
