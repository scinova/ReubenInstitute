וַהֲוָה בְאַרְבַּע מְאָה וְתַמְנָן שְׁנִין לְמִפַּק בְּנֵי יִשְׂרָאֵל מֵאַרְעָא דְמִצְרָיִם בְּשַׁתָּא רְבִיעֲתָא בְּיֶרַח זִיו נִיצָנַיָא הוּא יַרְחָא תִנְיָנָא לְמִמְלַךְ שְׁלֹמֹה עַל יִשְׂרָאֵל וּבְנָא בֵיתָא קֳדָם יְיָ:
וּבֵיתָא דִי בְנָא מַלְכָּא שְׁלֹמֹה קֳדָם יְיָ שִׁתִּין אַמִין אוּרְכֵּיהּ וְעַסְרִין פּוּתְיֵהּ וּתְלָתִין אַמִין רוּמֵיהּ:
וְאוּלַמָא עַל אַפֵּי הֵיכְלָא דְבֵיתָא עַסְרִין אַמִין אוּרְכֵּיהּ עַל אַפֵּי פוּתְיָא דְבֵיתָא עֲסַר אַמִין פּוּתְיָא עַל אַפֵּי בֵיתָא:
וַעֲבַד לְבֵיתָא כַּוִין פְּתִיחָן מִלְגָיו וּסְתִימָן מִלְבָרָא וְחָפָן כְּלִינְסָא בֵּית אַנְחַת רֵישֵׁי שִׁרְיַת אַרְזַיָא:
וּבְנָא עַל כָּתְלֵי בֵיתָא זִיוָא סְחוֹר סְחוֹר לְהֵיכָלָא וּלְבֵית כַּפּוֹרֵי וַעֲבַד מְחִיצַיָא סְחוֹר סְחוֹר:
מְחִיצָתָא אַרְעִיתָא חֲמֵשׁ אַמִין פוּתְיָהּ וּמְצִיעֲתָא שִׁית אַמִין פוּתְיָהּ וּתְלִיתֵיתָא שְׁבַע אַמִין פוּתְיָהּ אֲרֵי נְפַקְתָּא עֲבַד לְבֵיתָא סְחוֹר סְחוֹר לְבָרָא מִן מְחִיצַיָא לְמֶהֱוֵי רֵישֵׁי שִׁרְיָתָא נְיָחִין עַל זִיזְיָא וְלָא יֶהֶוְיַן שִׁרְיָתָא מְעַבְּרַן בְּכָתְלֵי בֵיתָא:
וּבֵיתָא בְּאִתְבְּנָיוּתֵיהּ אַבְנִין שְׁלֵמִין מַטְקַס אִתְבָּנֵי וְאַרְזוֹבִין וְחוֹלִילִין כָּל מָאן דְּפַרְזֶל לָא אִשְׁתְּמַע בְּבֵיתָא בְּאִתְבְּנָיוּתָא:
תַּרְעָא דִמְחִיצָתָא אַרְעִיתָא פְּתִיחַ לְעִבַּר בֵּיתָא מִיָמִינָא וּבִמְסִבָּתָא סַלְקִין מִן אַרְעִיתָא לְעִילִיתָא בְּאוֹרַח מְצִיעֵיתָא:
וּבְנָא יַת בֵּיתָא וְשַׁכְלְלֵיהּ וְטַלֵיל יַת בֵּיתָא בְּהַנְתּוּכִין וְעֵילָא מִנְהוֹן סִדְרָא דְרִכְפַּת רֵישֵׁיהּ שִׁרְיָתָא אַרְזַיָא:
וּבְנָא יַת תֵּיקָא עַל כָּל בֵּיתָא חֲמֵשׁ אַמִין רוּמֵיהּ וְטַלֵיל יַת בֵּיתָא בַּעֲאֵי אַרְזַיָא:
וַהֲוָה פִּתְגָם נְבוּאָה מִן קֳדָם יְיָ עִם שְׁלֹמֹה לְמֵימָר:
בֵּיתָא הָדֵין דִי אַתְּ בָּנֵי אִם תְּהַךְ בִּקְיָמַי וְיַת דִינַי תַּעְבֵּיד וְתִטַר יַת כָּל פִּקוּדַי לִמְהַךְ בְּהוֹן וְאָקִים יַת פִּתְגָמַי עִמָךְ דִי מַלֵילִית עִם דָוִד אֲבוּךְ:
וְאַשְׁרֵי שְׁכִנְתִּי בְּגוֹ בְּנֵי יִשְׂרָאֵל וְלָא אַרְחֵיק יַת עַמִי יִשְׂרָאֵל:
וּבְנָא שְׁלֹמֹה יַת בֵּיתָא וְשַׁכְלְלֵיהּ:
וּבְנָא יַת כּוֹתְלֵי בֵיתָא מִלְגָאו בִּמְחִיצַת אַרְזַיָא מֵאַרְעִית בֵּיתָא עַד כּוֹתְלַיָא עַד מָטֵי שְׁמֵי שַׁרְיָתָא חָפֵי נִסְרִין דְאַרְזָא מִלְגָיו וַחֲפָא יַת אוּשֵׁי בֵיתָא בִּמְחִיצַת בִּירְוָן:
וּבְנָא יַת עַסְרִין אַמִין מִסְיָפֵי בֵיתָא בֵמְחִיצַת אַרְזַיָא מִן אוּשַׁיָא עַד כָּתְלַיָא וּבְנָא לֵיהּ מִלְגָיו לְבֵית כֵּפּוֹרֵי לְקוֹדֶשׁ קוּדְשַׁיָא:
וְאַרְבְּעִין אַמִין הֱוֵי בֵיתָא הוּא הֵיכָלָא לְקִבְלֵיהּ:
וְנִסְרִין דְאַרְזָא חֲפֵי בֵיתָא מִלְגָיו וּגְלַף חֵיזוּ בֵיעִין וְאַטוּנִין שׁוֹשַׁנִין כָּל בֵּיתָא חֲפֵי נִסְרִין דְאַרְזָא לֵית אַבְנָא מִתְחַזְיָא:
וּבֵית כַּפּוֹרֵי מִגוֹ בֵיתָא מִלְגָיו אַתְקֵין לְמִתַּן תַּמָן יַת אֲרוֹן קְיָמָא דַיָי:
וְקָדָם בֵּית כַּפּוֹרֵי עַסְרִין אַמִין אוֹרְכָּא וְעַסְרִין אַמִין פּוּתְיָא וְעַסְרִין אַמִין רוּמֵיהּ וַחֲפָהִי דַהֲבָא טָבָא וַחֲפָא מַדְבְּחָא נִסְרִין דְאַרְזָא:
וַחֲפָא שְׁלֹמֹה יַת בֵּיתָא מִלְגָיו דַהֲבָא טָבָא וְאַעְבֵּר בְּשִׁלְשְׁלָן דִדְהַב קֳדָם בֵּית כַּפּוֹרֵי וַחֲפָהִי דַהֲבָא:
וְיַת כָּל בֵּיתָא חֲפָּא דַהֲבָא עַד דִשְׁלִים כָּל בֵּיתָא וְכָל מַדְבְּחָא לְבֵית כַּפּוֹרֵי חֲפָא דַהֲבָא:
וַעֲבַד בְּבֵית כַּפּוֹרֵי תְּרֵין כְּרוּבִין דַאֲעֵי זֵיתָא עֲסַר אַמִין רוּמֵיהּ:
וַחֲמֵשׁ אַמִין כַּנְפָא דִכְרוּבָא חָד וַחֲמֵשׁ אַמִין כַּנְפָא דִכְרוּבָא תִּנְיָנָא עֲסַר אַמִין מִקְצַת כַּנְפוֹהִי וְעַד קְצַת כַּנְפוֹהִי:
וַעֲסַר אַמִין כְּרוּבָא תִנְיָנָא מִשְׁחָא חֲדָא וְעֶבְרָא חָד לִתְרֵין כְּרוּבַיָא:
רוּמֵיהּ דִכְרוּבָא חָד עֲסַר אַמִין וְכֵן כְּרוּבָא תִנְיָנָא:
וִיהַב יַת כְּרוּבַיָא בְּגוֹ בֵיתָא גַוָאָה וּפְרָסוּ יַת כַּנְפֵי כְרוּבַיָא וּמַטְיָא כַנְפָא דְחָד בְּכוֹתְלָא וְכַנְפָא דִכְרוּבָא תִנְיָנָא מַטְיָא בְּכוֹתְלָא תִנְיָנָא וְכַנְפֵיהוֹן לְגוֹ בֵיתָא מַטְיָא כְּנַף לִכְנָף:
וַחֲפָא יַת כְּרוּבַיָא דַהֲבָא:
וְיַת כָּל כּוֹתְלֵי בֵיתָא אַקִיף צִיוּרִין וּגְלַף קְלִיעַן כְּרוּבִין וְצוּרַת דִקְלִין וְאָטוּנִין שׁוֹשַׁנִין מִלְגָיו וּלְבָרָא:
וְיַת אוּשֵׁי בֵיתָא חֲפָא דַהֲבָא מִלְגָיו וּלְבָרָא:
וְלִתְרַע בֵּית כַּפּוֹרֵי עֲבַד דָשִׁין דַאֲעֵי זֵיתָא אֵלָהִי וְסִפּוֹהִי מְטַקְסִין:
וְתַרְתֵּין דָשִׁין דַאֲעֵי זֵיתָא וּגְלַף עֲלֵיהוֹן קְלִיעַן כְּרוּבִין וְצוּרַת דִקְלִין וְאַטוּנִין שׁוֹשַׁנִין וַחֲפָא דַהֲבָא וְנַסִיךְ עַל כְּרוּבַיָא וְעַל צוּרַת דִקְלִין יַת דַהֲבָא:
וְכֵן עֲבַד לִתְרַע הֵיכָלָא סִפִּין דַאֲעֵי זֵיתָא מִן אַרְבְּעָה סִטְרֵיהוֹן מְרַבְּעִין:
וְתַרְתֵּין דָשִׁין דַאֲעֵי בִרְוָן תְּרֵין צִירִין דְדָשָׁא חֲדָא סְגַלְגְלִין וּתְרֵין צִירִין דְדָשָׁא תִנְיֵתָא סְגַלְגְלִין:
וּגְלַף כְּרוּבַיָא וְצוּרַת דִקְלִין וְאַטוּנִין שׁוֹשַׁנִין וַחֲפָא דַהֲבָא מְכַבֵּשׁ עַל צוּרָתָא:
וּבְנָא יַת דַרְתָּא גַוְיָתָא תְּלָתָא סִדְרִין דְאֶבֶן טָבָא וְסִדְרָא חָד דְמָרֵישׁ אֲעֵי אַרְזַיָא:
וּבְשַׁתָּא רְבִיעִיתָא שָׁרֵי לְאִשְׁתַּכְלָלָא בֵּית מַקְדְשָׁא דַייָ בְּיֶרַח זִיו נִצָנַיָא:
וּבְשַׁתָּא חֲדָא עַסְרֵי בְּיֶרַח מֵיסַף אִבְבַיָא הוּא יַרְחָא תְמִינָאָה אִשְׁתַּכְלַל בֵּיתָא לְכָל גְזֵרָתֵיהּ וּלְכָל דַחֲזֵי לֵיהּ וּבְנָהִי שְׁבַע שְׁנִין:
