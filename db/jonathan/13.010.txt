קַבִּילוּ יַת פִּתְגָמָא דְמַלֵיל יְיָ עֲלֵיכוֹן בֵּית יִשְׂרָאֵל:
כִּדְנַן אֲמַר יְיָ מֵאוֹרְחַת עַמְמַיָא לָא תֵילְפוּן וּמֵאַתְוָן דְמִשְׁתַּנִין בִּשְׁמַיָא לָא תִתַּבְּרוּן אֲרֵי מִתַּבְּרִין עַמְמַיָא מִנְהוֹן:
אֲרֵי גְזֵירַת עַמְמַיָא לְמָא אִינוּן אֲרֵי אָעָא מֵחוּרְשָׁא קָיִץ לֵיהּ עוֹבַד יְדֵי נַגָר בַּחֲצִינָא:
בְּכַסְפָּא וּבְדַהֲבָא חָפֵי לֵיהּ בְּמַסְמְרִין וּבְמַקָבִין מַתְקֵיף לֵיהּ דְלָא יִצְטְלֵי:
עוֹבַד נְגִיד זְקִיפִין אִינוּן וְלָא מְמַלְלִין אִתְנַטָלָא יִתְנַטְלוּן אֲרֵי לֵית בְּהוֹן רוּחָא לַהֲלָכָא לָא תִדַחֲלוּן מִנְהוֹן אֲרֵי לָא מַבְאֲשִׁין וְאַף לְאֵיטָבָא לֵית אִינוּן יָדְעִין:
לֵית בַּר מִנָךְ יְיָ רַב אַתְּ וְרַב שְׁמָךְ בְּגִבָּרוּתָא:
מַן לָא יִדְחַל מִן קֳדָמָךְ מֶלֶךְ עַל כָּל עַמְמַיָא אֲרֵי דִילָךְ הִיא מַלְכוּתָא אֲרֵי בְּכָל חַכִּימֵי עַמְמַיָא וּבְכָל מַלְכְּוָתְהוֹן לֵית בַּר מִנָךְ:
וּבַחֲדָא יְחוֹבוּן עַמְמַיָא פַּלְחֵי כּוֹכְבַיָא עַל דְאִטַפָּשׁוּ עַל דִפְלָחוּ לְטַעֲוָתָא דִלְמָא אִינוּן:
כַּסְפָּא דִמְחַפָּן מֵאַפְרִיקָא מַיְתָן וְדַהֲבָא מֵאוֹפִיר עוֹבַד נַגָר וִידֵי קִינֵי תִּכְלָא וְאַרְגְוָנָא לְבוּשְׁהוֹן עוֹבַד חֲכִּימִין כּוּלְהוֹן:
וַיְיָ אֱלֹהִים קְשׁוֹט הוּא אֱלָהּ קַיָם וּמְלַךְ עַלְמִין מֵרוּגְזֵיהּ תְּזוּעַ אַרְעָא וְלָא יָכְלִין עַמְמַיָא לְסוֹבָרָא רוּגְזֵיהּ:
דְנָא פַתְשֶׁגֶן אִגַרְתָּא דִשְׁלַח יִרְמִיָה נְבִיָא לְוַת שְׁאָר סָבֵי גַלְוָתָא דִי בְּבָבֶל וְאִם יֵמְרוּן לְכוֹן כַּשְׂדָאֵי דְאַתּוּן בֵּינֵיהוֹן פְּלָחוּ לְטַעֲוָתָא בֵּית יִשְׂרָאֵל כְּדֵין תְּתִיבוּ וּכְדֵין תֵּימְרוּן לְהוֹן טַעֲוָן דְּאַתּוּן פְלָחִין לְהוֹן טַעֲוַת דְלֵית בְּהוֹן צְרוֹךְ אִינוּן מִן שְׁמַיָא לָא יָכְלִין לְאַחָתָא מִטְרָא וּמִן אַרְעָא לָא יָכְלִין לְצַמָחָא פֵרִין אִינוּן וּפָלְחֵיהוֹן יֵיבְדוּ מֵאַרְעָא וְיִשְׁתֵּיצוּן מִן תְּחוֹת שְׁמַיָא אִלֵין:
וְכִדְנָא תֵימְרוּן לְהוֹן אֲנַחְנָא פַּלְחִין לִדְעָבֵיד אַרְעָא בְּחֵילֵי שַׁכְלֵיל תֵּבֵל בְּחוּכְמָתֵיהּ בְּסוּכְלָתָנוּתֵיהּ תְּלָא שְׁמַיָא:
לְקַל מִתְּנִיָהּ רִכְפַת דְמַיִין בִּשְׁמַיָא וּמֵסִיק עֲנָנִין מִסְיָפֵי אַרְעָא בַּרְקִין לִזְמַן מִטְרָא עָבַד וּמַפֵּיק רוּחִין מִבֵּית גִנְזוֹהִי:
אִתַּבָּרוּ כָל פַּלְחֵי כּוֹכְבַיָא מִלְמִידַע חָכְמָא בְּהִית כָּל קִינִי מִלְמֶעֱבַּד צְלֵם אֲרֵי שִׁקְרָא אַתְּכוּנוּן וְלָא רוּחָא בְהֵין:
לְמָא אִינוּן עוֹבֵד תּוּלְעָבָא בְּעִדַן דְאַסְעַר עֲלֵיהוֹן חוֹבֵיהוֹן יֵיבְדוּן:
לָא כְאִלֵין חוּלְקֵיהּ דְיַעֲקֹב אֲרֵי דִבְרָא כוֹלָא הוּא וְיִשְׂרָאֵל שִׁבְטָא דְאַחֲסַנְתֵּיהּ יְיָ צְבָאוֹת שְׁמֵיהּ:
כְּנוּשִׁי מֵאַרְעָא סְחוֹרְתִּיךְ דְיַתְבָא בְתוּקְפָא בְּקִרְוִין כְּרִיכָן:
אֲרֵי כִדְנַן אֲמַר יְיָ הָא אֲנָא מַיְתִי עָקָא עַל עַמָא הָדֵין וּכְמָה דְשָׁרָן אַבְנָא בְקִלְעָא כֵּן אֱבַדַר יַת יַתְבֵי אַרְעָא בְּזִמְנָא הָדָא וְאָעֵיק לְהוֹן בְדִיל דִיקַבְּלוּן פּוּרְעָנוּת חוֹבֵיהוֹן:
וַי לִי עַל תַּבְרִי מְמַרְעָא מַחְתִי וַאֲנָא אֲמָרֵית בְּרַם דֵין מַרְעִי וֶאֱסוֹבְרִינֵיהּ:
אַרְעִי חֲרוֹבַת וְכָל קִרְוֵי אִתְבְּזִיזָא עַמִי גָלוֹ וְלֵיתִינוּן לֵית דְמָתַח עוֹד מַשְׁכְּנִי וּמְקַיֵם יְרִיעָתָי:
אֲרֵי אִטַפָּשׁוּ מַלְכַיָא וְאוּלְפַן מַן קֳדָם יְיָ לָא תְבָעוּ עַל כֵּן לָא יִצְלְחוּן וְכָל עַמְהוֹן יִתְבַּדְרוּן:
קַל בְּשׂוֹרְתָא הָא אַתְיָא וּתְבָרָא רַבָּא מֵאֲרַע צִפּוּנָא לְשַׁוְאָה יַת קִרְוַיָא דְבֵית יְהוּדָה לְצָדוּ מְדוֹר יְרוֹדִין:
יְדַעְנָא יְיָ אֲרֵי לָא לֶאֱנָשָׁא אוֹרְחֵיהּ וְלָא לִגְבַר דְאָזִיל וּמַתְקִין יַת פְּסִיעָתֵיהּ:
אַיְתִי עֲלֵיהוֹן יְסוּרִין יְיָ בְּרַם בְּדִין חֲשׂוֹךְ לָא יִתְקַף רוּגְזָךְ בְּהוֹן דִלְמָא יִזְעָרוּן:
שְׁפוֹךְ חֵימָתָךְ עַל עַמְמַיָא דְלָא אֲלִיפוּ לְמֵידַע דְחַלְתָּךְ וְעַל זַרְעִיתָא דְכַשְׂדָאֵי דִי בְפּוּלְחַן שְׁמָךְ לָא תְבָעוּ אֲרֵי שְׁלִיטוּ בִּדְבֵית יַעֲקֹב וּבְזוּ נִכְסֵהוֹן וְשֵׁיצִיאוּנוּן וְיַת מְדוֹרְהוֹן אַצְדִיאוּ:
