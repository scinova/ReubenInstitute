וּנְטָלַת יָתִי רוּחָא וְאָעַלָת יָתִי לִתְרַע בֵּית מַקְדְשָׁא דַייָ מַדִינְחָאָה דִפְתִיחַ לְמַדִינְחָא וְהָא בְּמַעֲלָנָא דִתְרַע עַסְרִין וְחַמְשָׁא גַבְרִין וַחֲזֵית בֵּינֵיהוֹן יַת יַאֲזַנְיָה בַּר עַזֻר וְיַת פְּלַטְיָא בַּר בְּנָיָא רַבְרְבֵי עַמָא:
וַאֲמַר לִי בַּר אָדָם אִלֵין גַבְרַיָא דִמְחַשְׁבִין לְמֵינַס וּדְמָלְכִין מְלַךְ בִּישׁ בְּקַרְתָּא הָדָא:
דְאָמְרִין לָא בְקָרִיב לְמִבְנֵי בָתִּין הִיא חֲשִׁיבָא לָנָא כְדוּדָא וַאֲנַחְנָא חֲשִׁיבִין בְּגַוָהּ כְּבִסְרָא דִבְשִׁיל בְּגוֹ דוּדָא:
בְּכֵן אִתְנַבֵּי עֲלֵיהוֹן אִתְנַבֵּי בַּר אָדָם:
וּשְׁרַת עֲלֵי רוּחַ נְבוּאָה מִן קֳדָם יְיָ וַאֲמַר לִי אֱמַר כִּדְנַן אֲמַר יְיָ כֵּן אֲמַרְתּוּן בֵּית יִשְׂרָאֵל וְדִי אַתּוּן מְהַרְהֲרִין וְדִסְלִיקַת עַל לִבְּכוֹן קֳדָמַי גְלַן:
אַסְגֵיתוּן קְטִילֵכוֹן בְּקַרְתָּא הָדָא וּמְלֵיתוּן שׁוּקָהָא קְטִילִין:
בְּכֵן כִּדְנַן אֲמַר יְיָ אֱלֹהִים קְטִילֵיכוֹן דְשַׁוִיתוּן בְּגַוָהּ אִינוּן כְּבִסְרָא וְהָא כְדוּדָא וְיַתְכוֹן אַגְלֵי מִגַוָהּ:
מִן קֳדָם דְקָטְלִין בְּחַרְבָּא דְחַלְתּוּן וּדְקָטְלִין בְּחַרְבָּא אַיְתִי עֲלֵיכוֹן אֲמַר יְיָ אֱלֹהִים:
וְאַגְלֵי יַתְכוֹן מִגַוָהּ וְאֶמְסַר יַתְכוֹן בְּיַד נוּכְרָאִין וְאַעְבֵּיד בְּכוֹן פּוּרְעֲנוּת דִינִין:
בְּחַרְבָּא תִּתְקַטְלוּן עַל תְּחוּם אַרְעָא דְיִשְׂרָאֵל אִתְפְּרַע מִנְכוֹן וְתֵידְעוּן אֲרֵי אֲנָא יְיָ:
הִיא לָא תְהֵי לְכוֹן כְּדוּדָא בְּרַם אַתּוּן תְּהוֹן בְּגַוָהּ כְּבִסְרָא דִבְשִׁיל בְּגוֹ דוּדָא עַל תְּחוּם אַרְעָא דְיִשְׂרָאֵל אֶתְפְּרַע מִנְכוֹן:
וְתֵידְעוּן אֲרֵי אֲנָא יְיָ דִי בִקְיָמַי לָא הַלֵיכְתּוּן וְדִינַי לָא עֲבַדְתּוּן וּכְנִימוּסֵי עַמְמַיָא דִי בְסָחְרָנֵיכוֹן עֲבַדְתּוּן:
וַהֲוָה כְּאִתְנַבָּיוּתִי וּפְלַטְיָה בַר בְּנָיָה מִית וְאִשְׁתַּטְחֵית עַל אַפֵּי וּצְוָחֵית בְּקַל רַב וַאֲמָרֵית קַבֵּיל בָּעוּתִי יְיָ אֱלֹהִים גְמִירָא אַתְּ עֲבִיד עִם שְׁאָרָא דְיִשְׂרָאֵל:
וַהֲוָה פִּתְגָם נְבוּאָה מִן קֳדָם יְיָ עִמִי לְמֵימָר:
בַּר אָדָם אֲחַךְ אֲחַךְ אֱנַשׁ קָרִיבָךְ וְכָל בֵּית יִשְׂרָאֵל כּוּלְהוֹן דְאָמְרִין לְהוֹן יַתְבֵי יְרוּשְׁלֵם אִתְרְחָקוּ מִדְחַלְתָּא דַייָ דִילָנָא הִיא אִתְיְהִיבַת אַרְעָא לְמֵירְתָּה:
בְּכֵן כִּדְנַן אֲמַר יְיָ אֱלֹהִים אֲרֵי אַרְחֵיקְתִּינוּן לְבֵינֵי עַמְמַיָא וַאֲרֵי בְדַרְתִּינוּן בִמְדִינָתָא וִיהָבֵית לְהוֹן בָתֵּי כְנִשְׁתָּא תִּנְיָן לְבֵית מַקְדְשִׁי וְאִינוּן כִּזְעֵיר בִּמְדִינָתָא דְאִתְגִלִיאוּ לְתַמָן:
בְּכֵן אֲמַר כִּדְנַן אֲמַר יְיָ אֱלֹהִים וַאֲקָרֵיב יַתְכוֹן מִן בֵּינֵי עַמְמַיָא וְאַכְנֵישׁ יַתְכוֹן מִן מְדִינָתָא דְאִתְבַּדַרְתּוּן בְּהֵן וְאֶתֵּן לְכוֹן יַת אַרְעָא דְיִשְׂרָאֵל:
וְיֶעֱלוּן לְתַמָן וְיַעְדוּן יַת כָּל שִׁקוּצָא וְיַת כָּל תּוֹעֲבָתָא מִנָהּ:
וְאֶתֵּן לְהוֹן לֵב דְחִיל וְרוּחַ דְחִילָא אֶתֵּן בְּמֵעֵיהוֹן וְאֶתְּבַר לִבָּא דְרַשִׁיעָא דְהוּא תַּקִיף כְּאַבְנָא מִבִּסְרֵיהוֹן וְאֶתֵּן לְהוֹן לֵב דָחִיל קֳדָמַי לְמֶעְבַד רְעוּתָי:
בְּדִיל דְבִקְיָמַי יְהָכִין וְיַת דִינַי יִטְרוּן וְיַעַבְּדוּן יַתְהוֹן וִיהוֹן קֳדָמַי לְעַם וַאֲנָא אֶהֱוֵי לְהוֹן לֶאֱלָהָא:
בָּתַר פּוּלְחַן שִׁקוּצֵיהוֹן וְתוֹעֲבָתְהוֹן לִבְּהוֹן טָעֵי פּוּרְעֲנוּת אוֹרְחֵיהוֹן בְּרֵישֵׁיהוֹן יְהָבֵית אֲמַר יְיָ אֱלֹהִים:
וּנְטָלוּ כְרוּבַיָא יַת גַפֵּיהוֹן וְגַלְגְלַיָא לְקִבְלֵיהוֹן וִיקָר אֱלָהָא דְיִשְׂרָאֵל עֲלֵיהוֹן מִלְעֵילָא:
וְאִסְתַּלֵק יְקָרָא דַייָ מִן גוֹ קַרְתָּא וּשְׁרָא עַל טוּר זֵיתָא דְמִמִדְנַח לְקַרְתָּא:
וְרוּחַ נְטַלְתַּנִי וְאֹבִילַתְנִי לִמְדִינַת אַרְעָא כַסְדָאֵי לְוַת בְּנֵי גָלוּתָא בְּחֶזְוָא בְּרוּחַ נְבוּאָה דִי שְׁרָת עֲלֵי מִן קֳדָם יְיָ וְאִסְתַּלֵק מִנִי חֲזְוָא דַחֲזֵיתִי:
וְאִתְנַבֵּיתִי לִבְנֵי גָלוּתָא יַת כָּל פִּתְגָמַיָא דַייָ דְאַחְזִינִי:
