וּשְׁלַח מַלְכָּא וּכְנָשׁוּ לְוָתֵיהּ כָּל סָבֵי יְהוּדָה וִירוּשְׁלֵם:
וּסְלֵיק מַלְכָּא לְבֵית מַקְדְשָׁא דַייָ וְכָל אֱנַשׁ יְהוּדָה וְיַתְבֵי יְרוּשְׁלֵם עִמֵיהּ וְכַהֲנַיָא וְסַפְרַיָא וְכָל עַמָא לְמִזְעֵירָא וְעַד רַבָּא וּקְרָא קֳדָמֵיהוֹן יַת כָּל פִּתְגָמָא סַפְרָא דִקְיָמָא דְאִשְׁתְּכַח בְּבֵית מַקְדְשָׁא דַיָי:
וְקָם מַלְכָּא עַל אִסְטְוָנָא וּגְזַר יַת קְיָמָא קָדָם יְיָ לִמְהַךְ בָּתַר פּוּלְחָנָא דַייָ וּלְמִטַר פִּקוּדוֹהִי וְיַת סְהִידְוָתוֹהִי וְיַת קְיָמוֹהִי בְּכָל לִבָּא וּבְכָל נְפַשׁ לְקַיָמָא יַת פִּתְגָמֵי קְיָמָא הָדֵין דִכְתִיבִין עַל סִפְרָא הָדֵין וְקַבִּילוּ כָל עַמָא עֲלֵיהוֹן קְיָמָא:
וּפַקֵיד מַלְכָא יַת חִלְקִיָהוּ כַהֲנָא רַבָּא וְיַת סְגַן כַּהֲנַיָא וְיַת אֲמַרְכְּלַיָא לְאַפָּקָא מֵהֵיכָלָא דַייָ יַת כָּל מָנַיָא דַעֲבִידִין לִבְעָלָא וְלַאֲשֵׁירְתָא וּלכָל חֵילֵי שְׁמַיָא וְאוֹקְדִינוּן מִבָּרָא לִירוּשְׁלֵם בְמֵישַׁר קִדְרוֹן וְאוֹבִיל יַת עֲפַרְהוֹן לְבֵית אֵל:
וּבַטֵיל יַת כּוּמְרַיָא דִי יְהָבוּ מַלְכַיָא דְבֵית יְהוּדָה וְאַסִיקוּ בוּסְמִין עַל בָּמָתָא בְּקִרְוַיָא דְבֵית יְהוּדָה וּבְסַחֲרָנֵי יְרוּשְׁלֵם וְיַת דַאֲסִיקוּ בוּסְמִין לִבְעָלָא וּלְשִׁמְשָׁא וּלְסִיהֲרָא וּלְמַזְלָתָא וּלְכָל חֵילֵי שְׁמַיָא:
וְאַפֵיק יַת אֲשֵׁירָתָא מִבֵּית מַקְדְשָׁא דַייָ מִבָּרָא לִירוּשְׁלֵם לְמֵישַׁר קִדְרוֹן וְאוֹקִיד יָתָהּ בְּנַחְלָא דְקִדְרוֹן וְאַדְקָהּ לְעַפְרָא וּרְמָא יַת עַפְרָהּ לְקִבְרֵי בְּנֵי גָלַיָא:
וְתָרַע יַת בָּתֵּי הֶקְדֵשׁ טַעֲוָתָא דִי בְבֵית מַקְדְשָׁא דַייָ דִי נְשַׁיָא מָחֲיָן תַּמָן מְכוֹלִין לַאֲשֵׁירָתָא:
וְאַיְתֵי יַת כָּל כּוּמְרַיָא מִקִירְוַיָא דְבֵית יְהוּדָה וְסָאֵיב יַת בָּמָתָא דְאַסִיקוּ בוּסְמִין תַּמָן כּוּמְרַיָא מִגֶבַע עַד בְּאֵר שָׁבַע וְתָרַע יַת בָּמַת תַּרְעַיָא דִי קֳדָם תְּרַע יְהוֹשֻׁעַ רַב קַרְתָּא דִי עַל שְׂמָאלָא גַבְרָא בְּמֵיעֲלֵיהּ תְּרַע קַרְתָּא:
בְּרַם לָא סַלְקִין כַּהֲנַיָא כּוּמָרֵי בָמָתָא לְדַבָחָא עַל מַדְבְּחָא דַייָ בִּירוּשְׁלֵם אֱלָהֵין אֲכָלוּ פַטִיר בְּגוֹ אֲחֵיהוֹן:
וְסָאֵיב יַת תּוֹפֶת דִי בְחֵילַת בַּר הִנוֹם בְּדִיל דְלָא לְאַעֲבָרָא גְבַר יַת בְּרֵיהּ וְיַת בְּרַתֵּיהּ בְּנוּרָא לַמוֹלֶךְ:
וּבַטֵל יַת סוּסְוָתָא דִיהָבוּ מַלְכַיָא דְבֵית יְהוּדָה לְשִׁמְשָׁא מִמַעֲלָנָא דְבֵית מַקְדְשָׁא דַייָ לְלִשְׁכַּת נְתַן מֶלֶךְ גַוְזָאָה דִבְפַרְוָרַיָא וְיַת רְתִיכֵי שִׁמְשָׁא אוֹקִיד בְּנוּרָא:
וְיַת אֱגוֹרַיָא דִי עַל אִגַר עִלִית אָחָז דַעֲבָדוּ מַלְכַיָא דְבֵית יְהוּדָה וְיַת אֱגוֹרַיָא דַעֲבַד מְנַשֶׁה בְּתַרְתֵּין דָרַת בֵּית מַקְדְשָׁא דַייָ תְּרַע מַלְכָּא וְאַרְחֵיק מִתַּמָן וּרְמָא יַת עֲפַרְהוֹן לְנַחֲלָא דְקִדְרוֹן:
וְיַת בָּמָתָא דִי עַל אַפֵּי יְרוּשְׁלֵם דְמִדָרוֹם לְטוּר זֵיתַיָא דִי בְנָא שְׁלֹמֹה מַלְכָּא דְיִשְׂרָאֵל לְעַשְׁתּוֹרֶת שִׁקֻץ צִידוֹנָאֵי וְלִכְמוֹשׁ שִׁקֻץ מוֹאֲבָאֵי וּלְמִלְכּוֹם רִיחוּק בְּנֵי עַמוֹן סָאֵיב מַלְכָּא:
וְתַבַּר יַת קָמָתָא וְקָץ יַת אֲשֵׁירָתָא וּמַלֵא יַת אַתְרֵהוֹן גַרְמֵי אֱנָשָׁא:
וְאַף יַת אֱגוֹרָא דִי בְבֵית אֵל בָּמָתָא דַעֲבַד יָרָבְעָם בַּר נְבָט דְחַיֵב יַת יִשְׂרָאֵל אַף יַת אֱגוֹרָא הַהִיא וְיָת בָּמָתָא תָּרַע וְאוֹקִיד יַת בָּמָתָא אַדְקָה לְעַפְרָא וְאוֹקִיד אֲשֵׁירָתָא:
וְאִתְפְּנֵי יֹאשִׁיָהוּ וַחֲזָא יַת קִבְרַיָא דְתַמָן בְּטוּרָא וּשְׁלַח וּנְסִיב יַת גַרְמַיָא מִן קִבְרַיָא וְאוֹקִיד עַל אֱגוֹרַיָא וְסַאֲבֵיהּ כְּפִתְגָמָא דַייָ דְמַלֵיל נְבִיָא דַייָ דְאִתְנַבֵּי יַת פִּתְגָמַיָא הָאִלֵין:
וַאֲמַר מָה צִיוֹנָא הָדֵין דִי אֲנָא חֲזֵי וַאֲמַרוּ לֵיהּ אֱנָשֵׁי קַרְתָּא קִבְרָא דִנְבִיָא דַייָ דַאֲתָא מִשִׁבְטָא דִיהוּדָה וְאִתְנַבֵּי יַת פִּתְגָמַיָא הָאִלֵין דַעֲבַדְתָּא עַל אֱגוֹרַיָא בְּבֵית אֵל:
וַאֲמַר שְׁבוּקוּ מִנֵיהּ אֱנַשׁ לָא יָנִיד גַרְמוֹהִי וְשַׁזֵיבוּ גַרְמוֹהִי (עִם) יַת גַרְמֵי נְבִיָא שִׁקְרָא דַאֲתָא מִשֹׁמְרוֹן:
וְאַף יַת כָּל בָּתֵּי בָּמָתָא דִי בְּקִירְוֵי שֹׁמְרוֹן דִי עֲבָדוּ מַלְכֵי יִשְׂרָאֵל לְאַרְגָזָא פַּלֵי יֹאשִׁיָהוּ וַעֲבַד לְהוֹן כְּכָל עֲבִידָא דַעֲבַד בְבֵית אֵל:
וּנְכִיס יַת כָּל כּוּמְרֵי בָמָתָא דִי תַמָן עַל אֱגוֹרַיָא וְאוֹקִיד יַת גַרְמֵי אֱנָשָׁא עֲלֵיהוֹן וְתַב לִירוּשְׁלֵם:
וּפַקֵיד מַלְכָּא יַת כָּל עַמָא לְמֵימַר עֱבִידוּ פִסְחָא קֳדָם יְיָ אֱלָהָכוֹן כְּמָא דִכְתִיב עַל סִפְרָא דִקְיָמָא הָדֵין:
אֲרֵי לָא אִתְעֲבִיד כְּפִסְחָא הָדֵין מִיוֹמֵי נְגִידַיָא דְדָנוּ יַת יִשְׂרָאֵל וְכָל יוֹמֵי מַלְכֵי יִשְׂרָאֵל וּמַלְכַיָא דְבֵית יְהוּדָה:
אֱלָהִין בְּתַמְנֵי עַסְרֵי שְׁנִין לְמַלְכָּא יֹאשִׁיָהוּ אִתְעַבַד פִּסְחָא הָדֵין קֳדָם יְיָ בִּירוּשְׁלֵם:
וְאַף יַת בִּידִין וְיַת זְכוּרוּ וְיַת צַלְמָנַיָא וְיַת טַעֲוָתָא וְיַת כָּל שִׁקוּצַיָא דְאִתְחַזִיאוּ בְאַרְעָא דְבֵית יְהוּדָה וּבִירוּשְׁלֵם פַּלֵי יֹאשִׁיָהוּ בְּדִיל לְקַיָמָא יַת פִּתְגָמֵי אוֹרַיְתָא דִכְתִיבִין עַל סִפְרָא דְאַשְׁכַּח חִלְקִיָה כַהֲנָא בְּבֵית מַקְדְשָׁא דַיָי:
וּכְוָתֵיהּ לָא הֲוָה קֳדָמוֹהִי מַלְכָּא דְתַב לְפוּלְחָנָא דַייָ בְּכָל לִבֵּיהּ וּבְכָל נַפְשֵׁיהּ וּבְכָל נִכְסוֹהִי כְּכֹל אוֹרַיְתָא דְמֹשֶׁה וּבַתְרוֹהִי לָא קָם כְּוָתֵיהּ:
בְּרַם לָא תַב יְיָ מִתְּקוֹף רוּגְזֵיהּ רַבָּא דִתְקֵיף רוּגְזֵיהּ בִּדְבֵית יְהוּדָה עַל כָּל אַרְגְזוּתָא דִי אַרְגֵיז קֳדָמוֹהִי מְנַשֶׁה:
וַאֲמַר יְיָ אַף יַת דְבֵית יְהוּדָה אַגְלֵי מֵאַרְעָא בֵּית שְׁכִנְתִּי כְּמָא דְאַגְלֵיתִי יַת יִשְׂרָאֵל וְאַרְחִיק יַת קַרְתָּא הָדָא דְאִתְרְעֵיתִי בִּירוּשְׁלֵם וּבְבֵיתָא דַאֲמָרֵית יְהִי שְׁמִי תַּמָן:
וּשְׁאָר פִּתְגָמֵי יֹאשִׁיָהוּ וְכָל דִי עֲבַד הֲלָא אִינוּן כְּתִיבִין עַל סְפַר פִּתְגָמֵי יוֹמִין לְמַלְכַיָא דְבֵית יְהוּדָה:
בְּיוֹמוֹהִי סְלִיק פַּרְעֹה חֲגִירָא מַלְכָּא דְמִצְרַיִם עַל מַלְכָּא דְאַתּוּר עַל נְהַר פְּרָת וַאֲזַל מַלְכָּא יֹאשִׁיָהוּ לָקֳדָמוּתֵיהּ וְקַטְלֵיהּ בִּמְגִדוֹ כַּד חֲזָא יָתֵיהּ:
וְאַחְתוּהִי עַבְדוֹהִי כַּד מִית מִמְגִדוֹ וִאַיְתִיאוּהִי לִירוּשְׁלֵם וְקַבְרוּהִי בִּקְבוּרְתֵּיהּ וּדְבַר עַמָא דְאַרְעָא יַת יְהוֹאָחָז בַּר יֹאשִׁיָהוּ וּמְשָׁחוּ יָתֵיהּ וְאַמְלִיכוּ יָתֵיהּ תְּחוֹת אֲבוּהִי:
בַּר עַסְרִין וּתְלַת שְׁנִין יְהוֹאָחָז כַּד מְלַךְ וּתְלָתָא יַרְחִין מְלַךְ בִּירוּשְׁלֵם וְשׁוּם אִמֵיהּ חֲמוּטַל בַּת יִרְמִיָה מִן לִבְנָה:
וַעֲבַד דְבִישׁ קֳדָם יְיָ כְּכָל דַעֲבָדוּ אֲבָהָתוֹהִי:
וְאַסְרֵיהּ פַרְעֹה חֲגִירָא בְרִבְלָה בְּאַרְעָא דַחֲמָת מִלְמִמְלַךְ בִּירוּשְׁלֵם וּרְמָא מַגְבֵּיתָא עַל עַמָא דְאַרְעָא מֵאָה כִּכְּרֵי כַסְפָא וְכִכְּרָא דְדַהֲבָא:
וְאַמְלִיךְ פַרְעֹה חֲגִירָא יַת אֶלְיָקִים בַּר יֹאשִׁיָהוּ תְּחוֹת יֹאשִׁיָהוּ אֲבוּהִי וְשַׁוִי יַת שְׁמֵיהּ יְהוֹיָקִים וְיַת יְהוֹאָחָז דְבַר וַאֲתָא לְמִצְרַיִם וּמִית תַּמָן:
וְכַסְפָא וְדַהֲבָא יְהַב יְהוֹיָקִים לְפַרְעֹה בְּרַם רְמָא מַגְבִּיתָא עַל עַמָא דְאַרְעָא לְמִתַּן יַת כַּסְפָּא עַל מֵימְרָא דְפַרְעֹה גְבַר כַּד חֲזֵי טַקִיס יַת כַּסְפָּא וְיַת דַהֲבָא מִן עַמָא דְאַרְעָא לְמִתַּן לְפַרְעֹה חֲגִירָא:
בַּר עַסְרִין וְחָמֵשׁ שְׁנִין יְהוֹיָקִים כַּד מְלַךְ וַחֲדָא עַסְרֵי שְׁנִין מְלַךְ בִּירוּשְׁלֵם וְשׁוּם אִמֵיהּ זְבוּדָה בַּת פְּדָיָה מִן רוּמָה:
וַעֲבַד דְבִישׁ קֳדָם יְיָ כְּכֹל דַעֲבָדוּ אֲבָהָתוֹהִי:
