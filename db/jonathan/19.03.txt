וַהֲוָה פִּתְגַם נְבוּאָה מִן קֳדָם יְיָ עִם יוֹנָה תִּנְיָנוּת לְמֵימָר:
קוּם אֱזֵיל לְנִינְוֵה קַרְתָּא רַבְּתָא וְאִתְנַבֵּי עֲלָהּ יַת נְבוּאֲתָא דִי אֲנָא מְמַלֵיל עִמָךְ:
וְקָם יוֹנָה וַאֲזַל לְנִינְוֵה כְּפִתְגָמָא דַייָ וְנִינְוֵה הֲוַת קַרְתָּא רַבְּתָא קֳדָם יְיָ מַהֲלַךְ תְּלָתָא יוֹמִין:
וּשְׁרֵי יוֹנָה לְמֵיעַל בְּקַרְתָּא מְהַלֵיךְ יוֹמָא חָד וְאַכְרֵיז וַאֲמַר בְּסוֹף אַרְבְּעִין יוֹמִין וְנִינְוֵה מִתְהַפְּכָא:
וְהֵימִינוּ אַנְשֵׁי נִינְוֵה בְּמֵימְרָא דַייָ וּגְזָרוּ צוֹמָא וַאֲסָרוּ שַׂקִין מֵרַבְּהוֹן וְעַד זְעָרֵיהוֹן:
וּמְטָא פִּתְגָמָא לְוַת מַלְכָּא דְנִינְוֵה וְקָם מִכּוּרְסֵי מַלְכוּתֵיהּ וְאַעְדִי לְבוּשֵׁי יְקָרֵיהּ מִנֵיהּ וְאִתְכַּסֵי שַׂקָא וִיתִיב עַל קִיטְמָא:
וְאַכְרֵיז וַאֲמַר בְּנִינְוֵה מִגְזֵירַת מַלְכָּא וְרַבְרְבָנוֹהִי לְמֵימָר אֱנָשָׁא וּבְעִירָא תּוֹרָא וַעֲנָא לָא יִטְעֲמוּן מִדַעַם וְלָא יִרְעוּן וּמַיָא לָא יִשְׁתּוּן:
וְיִתְכַּסוּן סַקִין אֱנָשָׁא וּבְעִירָא וִיצַלוּן קֳדָם יְיָ בִּתְקוֹף וִיתוּבוּן גְבַר מֵאוֹרְחֵיהּ בִּישְׁתָּא וּמִן חֲטוֹפָא דְבִידֵיהוֹן:
מַן יְדַע דְאִית בִּידֵיהּ חוֹבִין יְתוּב מִנְהוֹן וְיִתְרַחֵים עֲלוֹהִי מִן קֳדָם יְיָ וִיתוּב מִתְּקוֹף רוּגְזֵיהּ וְלָא נוֹבֵד:
וּגְלַן קֳדָם יְיָ עוֹבָדֵיהוֹן אֲרֵי תָבוּ מֵאוֹרְחַתְהוֹן בִּישְׁתָא וְתָב יְיָ מִן בִּישְׁתָא דִי מַלֵיל לְמֶעְבַּד לְהוֹן וְלָא עֲבָד:
