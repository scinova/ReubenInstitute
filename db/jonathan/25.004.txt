וְתַב מַלְאֲכָא דִמְמַלֵיל עִמִי וַאֲעִירַנִי כְּגַבְרָא דְמִתְעַר מִשְׁנָתֵיהּ:
וַאֲמַר לִי מַה אַתְּ חָזֵי וַאֲמָרֵית חֲזֵיתִי וְהָא מְנַרְתָּא דְדַהֲבָא כֻלָהּ וְגוּלְתָהּ עַל רֵישָׁא וְשַׁבְעָא בוֹצִינָהָא דַעֲלָהּ שַׁבְעָא וְשַׁבְעָא דִמְרִיקִין מִנְהוֹן מִשְׁחָא לְבוֹצִינַיָא דְעַל רֵישָׁהּ:
וְתַרְתֵּין זֵיתִין דַעֲלָהּ חַד מִימִין גּוּלְתָא וְחַד עַל סְמָלְתָה:
וַאֲתִיבֵית וַאֲמָרֵית לְמַלְאֲכָא דְמַלֵיל עִמִי לְמֵימַר מַה אִלֵין רִבּוֹנִי:
וַאֲתִיב מַלְאֲכָא דְמַלֵיל עִמִי וַאֲמַר לִי הֲלָא יְדַעְתָּא מַן אִינוּן אִלֵין וַאֲמָרֵית לִי רִבּוֹנִי:
וַאֲתִיב וַאֲמַר לִי לְמֵימָר דֵין פִּתְגָמָא דַייָ עִם זְרֻבָּבֶל לְמֵימָר לָא בְחֵילָא וְלָא בִּגְבוּרָא אֱלָהֵין בְּמֵימְרִי אֲמַר יְיָ צְבָאוֹת:
מָה אַתְּ חָשִׁיבָא מַלְכוּתָא טִפְּשָׁתָא קֳדָם זְרֻבָּבֶל הֲלָא כְמֵשְׁרָא וְיִגְלֵי יַת מְשִׁיחֵיהּ דַאֲמִיר שְׁמֵיהּ מִלְקָדְמִין וְיִשְׁלוֹט בְּכָל מַלְכְּוָתָא:
וַהֲוָה פִּתְגַם נְבוּאָה מִן קֳדָם יְיָ עִמִי לְמֵימָר:
יְדֵי זְרֻבָּבֶל שְׁרִיאָה לְמִבְנֵי בֵיתָא הָדֵין וִידוֹהִי יְשַׁכְלְלוּנֵיהּ וְתֵידְעוּן אֲרֵי יְיָ צְבָאוֹת שַׁלְחַנִי לְאִתְנַבָּאָה לְכוֹן:
אֲרֵי מַן הוּא דֵכִי דְבַסַר יוֹמָא דֵין עַל בִּנְיָנָא דְהוּא זְעֵיר הֲלָא יְתוּב וְיֶחְדֵי כַּד יֶחֱזֵי יַת אֶבֶן מַשְׁקוֹלְתָּא בִידָא דִזְרֻבָּבֶל שַׁבְעָא נִדְבְּכִין כְּאִלֵין קֳדָם יְיָ גְלַן עוֹבָדֵי בְּנֵי אֱנָשָׁא בְּכָל אַרְעָא:
וַאֲתִיבֵית וַאֲמָרֵית לֵיהּ מָא תְרֵין זֵיתַיָא הָאִלֵין עַל יְמִין מְנַרְתָּא וְעַל סְמָלָא:
וַאֲתִיבֵית תִּנְיָנוּת וַאֲמָרֵית לֵיהּ מָא תְּרֵין שׁוּבְלֵי זֵיתַיָא דִבְיַד תְּרֵין אִסְקְרִיטְוָן דִדְהַב דִמְרִיקִין מִנְהוֹן מִשְׁחָא לְבוֹצִינֵי דַהֲבָא:
וַאֲמַר לִי לְמֵימַר הֲלָא יְדַעַתְּ מַן אִלֵין וַאֲמָרֵית לָא רִבּוֹנִי:
וַאֲמַר אִלֵין תְּרֵין בְנֵי רַבְרְבַיָא דְקַיְמִין קֳדָם רִבּוֹן כָּל אַרְעָא:
