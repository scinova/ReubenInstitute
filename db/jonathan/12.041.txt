אֲצִיתוּ לְמֵימְרִי נַגְוָן וּמַלְכְּוָן יוֹסְפוּן חֵיל יִתְקָרְבוּן בְּכֵן יְמַלְלוּן כַּחֲדָא לְדִינָא נִתְקְרָב:
מַן אַיְתֵי בִגְלַאי מִמַדִינְחָא אַבְרָהָם בְּחִיר צַדִיקַיָא בִּקְשׁוֹט קַרְבֵהּ לְאַתְרֵהּ מְסַר קֳדָמוֹהִי עַמְמִין וּמַלְכִין תַּקִיפִין תְּבַר רְמָא כְּעַפְרָא קָטוֹלִין קֳדָם חַרְבֵּהּ כְּקַשָׁא רְדוֹפִין קֳדָם קַשְׁתֵּהּ:
רְדָפִינוּן עֲדָא שְׁלָם חַיָלַת אוֹרַח בְּרִגְלוֹהִי לָא עֲלָת:
מַן אֲמַר אִלֵין קַיָם אֲמַר וַעֲבִיד וְסִדַר דָרַיָא מִלְקָדְמִין אֲנָא יְיָ בָּרֵית עַלְמָא מִבְּרֵאשִׁית אַף עַלְמֵי עָלְמַיָא דִילִי אִינוּן בַּר מִנִי לֵית אֱלָהָא:
חֲזוֹ נַגְוָתָא וִידַחֲלוּן דְבִסְיָפֵי אַרְעָא יְזוּעוּן יִתְקַרְבוּן וְיֵיתוּן:
גְבַר יַת חַבְרֵהּ יִסְעֲדוּן וְלַאֲחוֹהִי יֵימַר תַּקִיף:
הֲלָא יְבַהֲתוּן בְּעוֹבָדֵיהוֹן דְמַתְקֵיף נַגָרָא עִם קֵינָאָה מָחֵי בְּפַטִישָׁא עִם דִמְטַפַּח בְקוּרְנְסָא זִמְנָא אֲמַר עַל דִבְקָא תַּקִין הוּא וּמַתְקֵיף לֵיהּ בְמַסְמְרִין דְלָא יִצְטְלֵי:
וְאַתְּ יִשְׂרָאֵל עַבְדִי יַעֲקֹב דְאִתְרְעֵיתִי בָּךְ זַרְעֵהּ דְאַבְרָהָם רְחִימִי:
דְקָרֵיבְתָּךְ מִזַרְעִית אַרְעָא וּמִמַלְכְּוָתָא בְּחַרְתָּךְ וַאֲמָרֵית לָךְ עַבְדִי אַתְּ אִתְרְעֵיתִי בָּךְ וְלָא אַרְחֵקִינָךְ:
לָא תִדְחַל אֲרֵי בְסַעֲדָךְ מֵימְרִי לָא תִתְּבַר אֲרֵי אֲנָא אֱלָהָךְ אִתְקְפִינָךְ אַף אֶסְעַדִינָךְ אַף אַחֲדִינָךְ בִּימִין קוּשְׁטִי:
הָא יִבָּהֲתוּן וְיִתְכַּנְעוּן כָּל עַמְמַיָא דַהֲווֹ מִתְגָרִין בָּךְ יְהוֹן כִּלְמָא וְיֵבְדוּן אֱנָשֵׁי דִינָךְ:
תַּבְעִינוּן וְלָא תַשְׁכְּחִינוּן לְאַנְשֵׁי מַצוּתָךְ יְהוֹן כִּלְמָא וּכְלָא מִדָעַם גַבְרִין דַהֲווֹ מִתְגָרִין לְמֶעְבַּד עִמָךְ קְרָב:
אֲרֵי אֲנָא יְיָ אֱלָהָךְ מַתְקֵיף יְמִינָךְ דַאֲמָרִית לָךְ לָא תִדְחַל מֵימְרִי בְסַעֲדָךְ:
לָא תִדַחְלוּן שִׁבְטַיָא דְבֵית יַעֲקֹב זַרְעִיתָא דְיִשְׂרָאֵל מֵימְרִי בְסַעֲדְכוֹן אֲמַר יְיָ וּפָרִיקְכוֹן קַדִישָׁא דְיִשְׂרָאֵל:
הָא שַׁוִיתֵיכוֹן לְמוֹרַג תַּקִיף חֲדַת מָלֵי סִמְפּוֹרִין תְּקָטֵיל פַּלְחֵי כּוֹכְבַיָא וּתְשֵׁיצֵי וּמַלְכוּתָא כְּמוֹצָא תְשַׁוֵי:
תְּדִרִינוּן וְרוּחָא תִטְלוּנוּן וּמֵימְרֵהּ כְּעַלְעוֹלָא לְקַשָׁא יְבַדֵר יַתְהוֹן וְאַתְּ תְּבוּעַ בְּמֵימְרָא דַייָ בְּקַדִישָׁא דְיִשְׂרָאֵל תִּשְׁתַּבַּח:
עִנְוְתָנַיָא וַחֲשִׁיכַיָא דִמְחַמְדִין לְאוּלְפָנָא הָא כְּצַחְיָא לְמַיָא וְלָא מְשַׁכְּחִין רוּחֵיהוֹן בְּסִגוּפָא מְשַׁלְהַיָא אֲנָא יְיָ אֲקַבֵּיל צְלוֹתְהוֹן אֱלָהָא דְיִשְׂרָאֵל לָא אֲרַקְחִינוּן:
אֲקָרֵב גָלוּתְהוֹן מִבֵּינֵי עַמְמַיָא וְאֶדְבְּרִינוּן בְּאוֹרַח תַקָנָא וְאֶפְתַּח לְהוֹן עַל נַגְדִין נַהֲרִין וּבְגוֹ מַשִׁרְיָן מַבּוּעִין אֲשַׁוֵי מַדְבְּרָא לַאֲגַמִין דְמַיָן וַאֲרַע בֵּית צַחְוָנָא לְמַבּוּעֵי מַיָא:
אֶתֵּן בְּמַדְבְּרָא אַרְזִין שִׁיטִין וַהֲדַסִין וְאָעִין דִמְשַׁח אֱרַבֵּי בְמֵישְׁרָא בּוּרְוָן מוּרְנְיָן וְאֶשְׁכְּרוֹעִין כַּחֲדָא:
בְּדִיל דְיֶחֱזוּן וְיֵדְעוּן וִישַׁווּן דְחַלְתִּי עַל לִבְּהוֹן וְיִסְתַּכְּלוּן כַּחֲדָא אֲרֵי גְבוּרְתָא דַייָ עֲבָדַת דָא וְקַדִישָׁא דְיִשְׂרָאֵל בְּרָא יָתָהּ:
קְרִיבוּ דִינְכוֹן אֲמַר יְיָ אָעִילוּ חֶזְוַתְכוֹן אֲמַר מַלְכֵּיהּ דְיַעֲקֹב:
יִתְקָרְבוּן וִיחַווּן לָנָא יַת דְעַרְעִינָנָא קַדְמֵיתָא מַה אִינוּן הֲווֹ וּנְשַׁוֵי לִבְּנָא וְנֵדָע סוֹפְהוֹן אוֹ דַעֲתִידִין לְמֵיתֵי בַּסְרוּנָא:
חֲווֹ דְאַתְיָן לְסוֹפָא וְנִדַע אִם לְטַעֲוָן דְאִית בְּהוֹן צְרוֹךְ אַתּוּן פָּלְחִין אִם יָכְלִין אִינוּן לְאֵיטָבָא וּלְאַבְאָשָׁא וְנִסְתַּכֵּיל וְנֵידוֹן כַּחֲדָא:
הָא אַתּוּן לְמָא וְעוֹבָדֵיכוֹן לָא מִדָעַם תּוֹעֵיבָא דְאַתְרְעִיתוּן בֵּיהּ לְכוֹן:
אַיְתֵיתֵי בִּגְלַאִי מְלַךְ דְתַקִיף מֵרוּחַ צִפּוּנָא וְיֵיתֵי כְּמִפַּק שִׁמְשָׁא בִּגְבוּרְתֵּהּ מִמַדִנְחָא אַגְבְּרִינֵהּ בִּשְׁמִי וְיֵיתִי וְיֵדוּשׁ שִׁלְטוֹנֵי עַמְמַיָא כְּמָא דְדָיְשִׁין יַת עַפְרָא וּכְפַחְרָא דְעָרִיךְ יַת טִינָא:
מַן חַוֵי מִן אַוְלָא וְנֵדַע וּמִלְקָדְמִין וְנֵימַר קְשׁוֹט אַף לֵית דִי מְחַוֵי אַף לֵית דִמְבַסֵר אַף לֵית דְשָׁמַע לְמִילֵיכוֹן:
פִּתְגָמֵי נֶחְמָתָא דְאִתְנַבִּיאוּ נְבִיָיא מִלְקָדְמִין עַל צִיוֹן הָא אֲתוֹ וְלִירוּשְׁלֵם מְבַסֵר אֶתֵּן:
וּגְלֵי קֳדְמַי וְלֵית גְבַר דְלֵיהּ עוֹבָדִין טָבִין וּמֵאִלֵין לֵית דְמָלִיךְ מֵילַךְ וּשְׁאִילְתִּינוּן אִם יְתוּבוּן פִּתְגָם:
הָא כוּלְהוֹן לְמָא וְלָא מִדָעַם עוֹבָדֵיהוֹן בִּזָא וּתְבִרָא עֶשְׁתּוֹנֵיהוֹן:
