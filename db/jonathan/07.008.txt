וַאֲמַרוּ לֵיהּ אֱנָשׁ אֶפְרַיִם מָא פִּתְגָמָא הָדֵין עֲבַדְתָּא לָנָא בְּדִיל מַה לָא קְרֵיתָא לָנָא כִּדְאַזַלְתָּא לְאַגָחָא קְרָבָא בְּמִדְיָן וְדַיְנִין עִמֵיהּ בִּתְקוֹף:
וַאֲמַר לְהוֹן מֶה עַבְדֵית כְּעַן כְּוַתְכוֹן הֲלָא טָבִין חַלָשַׁיָא דְבֵית אֶפְרַיִם מִתַּקְפַיָא דְבֵית אֲבִיעֶזֶר:
בְּיֶדְכוֹן מְסַר יְיָ יַת רַבְרְבֵי מִדְיָן יַת עוֹרֵב וְיַת זְאֵב וּמַה יְכֵלִית לְמֶעְבַּד כְּוַתְכוֹן בְּכֵן נָחִת רוּחֵיהוֹן מִנֵיהּ בְּמַלָלוּתֵיה פִּתְגָמָא הָדֵין:
וַאֲתָא גִדְעוֹן לְיַרְדְנָא עֲבַר הוּא וּתְלַת מְאָה גַבְרַיָא דְעִמֵיהּ כַּד מְשַׁלְהֵן וְרָדְפִין:
וַאֲמַר לֶאֱנָשֵׁי סֻכּוֹת הָבוּ כְעַן פִּתָּא דְלַחְמָא לְעַמָא דִי עִמִי אֲרֵי מְשַׁלְהָן אִנוּן וַאֲנָא רָדִיף בָּתַר זֶבַח וְצַלְמֻנָע מַלְכֵי מִדְיָן:
וַאֲמַר רַבְרְבֵי סֻכּוֹת הָא כְּאִלּוּ זֶבַח וְצַלְמֻנָע כְעַן מְסִירִין בִּידָךְ אֲרֵי נִתַּן לְחֵילָךְ לַחֲמָא:
וַאֲמַר גִדְעוֹן בְּכֵן כַּד יִמְסַר יְיָ יַת זֶבַח וְיַת צַלְמֻנָע בִּידִי וְאָדוּשׁ יַת בִּסְרֵיכוֹן עַל כּוּבֵי מַדְבְּרָא וְעַל בַּרְקָנַיָא:
וּסְלִיק מִתַּמָן לִפְנוּאֵל וּמַלֵל לְוַתְהוֹן כְּדָא וַאֲתִיבוּ יָתֵיה אֱנָשֵׁי פְּנוּאֵל כְּמָא דַאֲתִיבוּ אֱנָשֵׁי סֻכּוֹת:
וַאֲמַר אַף לֶאֱנָשֵׁי פְּנוּאֵל לְמֵימַר כַּד אֵיתוּב בִּשְׁלָם אֶתְרַע יַת מַגְדְלָא הָדֵין:
וְזֶבַח וְצַלְמֻנָע בַּקַרְקֹר וּמַשִׁרְיָתֵיהוֹן עִמְהוֹן כַּחֲמֵשֶׁת עֲסַר אַלְפִין כֹּל דְאִשְׁתָּאֲרוּ מִכֹּל מַשִׁרְיַת בְּנֵי מַדִנְחָא וּדְאִתְקַטָלוּ מְאָה וְעַשְׂרִין אַלְפִין גַבְרָא שְׁלִיפֵי חַרְבָּא:
וּסְלֵיק גִדְעוֹן בְּאוֹרַח מַשִׁרְיַת עֲרָבָאֵי דְשָׁרָן בִּשְׁכוּנִין בְּמַדְבְּרָא מִמַדִינְחָא לְנוֹבַח וְרָמָתָא וּמְחָא יַת מַשְׁרִיתָא וּמַשְׁרִיתָא שַׁרְיָא לְרוֹחְצָן:
וְאַפָכוּ זֶבַח וְצַלְמֻנָע וּרְדַף בַּתְרֵיהוֹן וַאֲחַד יַת תְּרֵין מַלְכֵי מִדְיָן יַת זֶבַח וְיַת צַלְמֻנָע וְכָל מַשְׁרִיתָא אָזִיעַ:
וְתַב גִדְעוֹן בַּר יוֹאָשׁ מִן קְרָבָא עַד לָא מֵעַל שִׁמְשָׁא:
וַאֲחַד עוּלֵימָא מֵאֱנָשֵׁי סֻכּוֹת וְשַׁאֲלֵיהּ וּכְתַב לֵיהּ יַת רַבְרְבֵי סֻכּוֹת וְיַת סָבָהָא שַׁבְעִין וְשִׁבְעָה גוּבְרִין:
וַאֲתָא לְוַת אֱנָשֵׁי סֻכּוֹת וַאֲמַר הָא זֶבַח וְצַלְמֻנָע דְחַסִדְתּוּן יָתִי לְמֵימָר הָא כְאִלוּ זֶבַח וְצַלְמֻנָע כְּעַן מְסִירִין בִּידָךְ אֲרֵי נִתַּן לְגַבְרָךְ דִמְשַׁלְהֵן לַחְמָא:
וּדְבַר יַת סָבֵי קַרְתָּא וְיַת כּוּבֵי מַדְבְּרָא וְיַת בַּרְקָנַיָא וְתַבַּר עֲלֵיהוֹן יַת אֱנָשֵׁי סֻכּוֹת:
וְיַת מִגְדַל פְּנוּאֵל תְּרַע וּקְטַל יַת אֱנָשֵׁי קַרְתָּא:
וַאֲמַר לְזֶבַח וּלְצַלְמֻנָע הֵיכָא גַבְרַיָא דִקְטַלְתּוּן בְּתָבוֹר וַאֲמַרוּ כְּוָתָךְ כֵּן אַף אִינוּן חַד מִנְהוֹן כְּחֶזְוָא דִבְנֵי מַלְכָּא:
וַאֲמַר אַחַי בְּנֵי אִמָא אִנוּן קַיָם הוּא יְיָ אֵלוּ קַיֵמְתּוּן יַתְהוֹן לָא קַטְלֵית יַתְכוֹן:
וַאֲמַר לְיֶתֶר בּוּכְרֵיהּ קוּם קְטוֹל יַתְהוֹן וְלָא שְׁלַף עוּלֵימָא חַרְבֵּיהּ אֲרֵי דָחִיל אֲרֵי עַד כְּעַן יָנִיק:
וַאֲמַר זֶבַח וְצַלְמֻנָע קוּם אַתְּ וּשְׁלַט בָּנָא אֲרֵי כְגַבְרָא כֵּן גְבוּרְתֵּיהּ וְקָם גִדְעוֹן וּקְטַל יַת זֶבַח וְיַת צַלְמֻנָע וּנְסִיב יַת עִנְקַיָא דִי בְצַוְארֵי גְמַלֵיהוֹן:
וַאֲמַרוּ אֱנָשׁ יִשְׂרָאֵל לְגִדְעוֹן מְלוֹךְ עֲלָנָא אַף אַתְּ אַף בְּרָךְ אַף בַּר בְּרָךְ אֲרֵי פְרַקְתָּנָא מִן יַד מִדְיָנָאֵי:
וַאֲמַר לְהוֹן גִדְעוֹן לָא אֶמְלוֹךְ אֲנָא עֲלֵיכוֹן וְלָא יִמְלוֹךְ בְּרִי עֲלֵיכוֹן יְיָ יִמְלוֹךְ עֲלֵיכוֹן:
וַאֲמַר לְהוֹן גִדְעוֹן שָׁאֵיל אֲנָא מִנְכוֹן שְׁאֵילָה וְהָבוּ לִי גְבַר קָדָשֵׁיהּ מִן בִּזְתָא אֲרֵי קָדָשֵׁי דְדַהֲבָא לְהוֹן אֲרֵי עֲרָבָאֵי אִינוּן:
וַאֲמַרוּ מִתַּן נִתָּן וּפָרְשׁוּ יַת כְּסוּתָא וּרְמוּ לְתַמָן גְבַר קָדְשֵׁיהּ מִן בִּזְתָא:
וַהֲוָה מַתְקַל קָדָשֵׁי דְדַהֲבָא דִשְׁאַל אֶלֶף וּשְׁבַע מְאָה דַהֲבָא בַּר מִן סַהֲרָנַיָא וּכְלִילָא וּלְבוּשֵׁי אַרְגְוָנָא דִי עַל מַלְכֵי מִדְיָן וּבַר מִן עֲנָקַיָא דִי בְּצַוְרֵי גְמַלֵיהוֹן:
וַעֲבַד יָתֵיה גִדְעוֹן לְאֵיפוֹדָא וַאֲקֵים יָתֵיהּ בְּקַרְתִּיהּ בְּעָפְרָה וּטְעוֹ כָל יִשְׂרָאֵל בַּתְרוֹהִי לְתַמָן וַהֲוָה לְגִדְעוֹן וּלְבֵיתֵיהּ לְתַקָלָא:
וְאִתְּבַרוּ מִדְיָנָאֵי קֳדָם בְּנֵי יִשְׂרָאֵל וְלָא אוֹסִיפוּ לַאֲרָמָא רֵישֵׁיהוֹן וּשְׁדוֹכַת אַרְעָא דְיִשְׂרָאֵל אַרְבְּעִין שְׁנִין בְּיוֹמֵי גִדְעוֹן:
וַאֲזַל יְרֻבַּעַל בַּר יוֹאָשׁ וִיתֵיב בְּבֵיתֵיהּ:
וּלְגִדְעוֹן הֲווֹ שַׁבְעִין בְּנִין נַפְקֵי יַרְכֵיהּ אֲרֵי נְשִׁין סַגִיאִין הֲוָאָה לֵיהּ:
וּלְחֵנָתֵיהּ דִי בִשְׁכֶם יְלִידַת לֵיהּ אַף הִיא בַּר וְשַׁוִי יַת שְׁמֵיהּ אֲבִימֶלֶךְ:
וּמִית גִדְעוֹן בַּר יוֹאָשׁ בְּסֵיבוּ טָבָא וְאִתְקְבַר בְּקִבְרָא דְיוֹאָשׁ אֲבוּהִי בְּעָפְרָה דַאֲבוּהִי דְעָזֶר:
וַהֲוָה כַּד מִית גִדְעוֹן וְתָבוּ בְּנֵי יִשְׂרָאֵל וְטָעוּ בָּתַר בַּעֲלַיָא וְשַׁוִיאוּ לְהוֹן בָּעֵיל קְיָם לְטַעֲוָתָא:
וְלָא אִדְכָּרוּ בְּנֵי יִשְׂרָאֵל יַת פּוּלְחָנָא דַייָ אֱלָהָהוֹן דִי שֵׁיזִיב יַתְהוֹן מִיַד כָּל בַּעֲלֵי דְבָבֵיהוֹן מִסְחוֹר סְחוֹר:
וְלָא עֲבָדוּ טֵיבוּ עִם בֵּית יְרֻבַּעַל גִדְעוֹן כְּכָל טַבְתָא דִי עֲבַד עִם יִשְׂרָאֵל:
