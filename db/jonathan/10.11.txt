וּמַלְכָּא שְׁלֹמֹה רְחֵים נְשִׁין בְּנַת עַמְמִין סַגִיאִין וְיַת בַת פַּרְעֹה מוֹאֲבָאָן עֲמוֹנָאָן אֲדוֹמָאָן וְצִידוֹנָאָן חִתָּאָן:
מִן עַמְמַיָא דִי אֲמַר יְיָ לִבְנֵי יִשְׂרָאֵל לָא תִתְעָרְבוּן בְּהוֹן וְאִינוּן לָא יִתְעָרְבוּן בְּכוֹן דִלְמָא יַטְעוּן יַת לִבְּכוֹן בָּתַר טַעֲוַתְהוֹן בְּהוֹן אִתְרְעֵי שְׁלֹמֹה לְמִרְחַם:
וַהֲוָאָה לֵיהּ נְשִׁין מַלְכָּן שְׁבַע מְאָה וּלְחֵינָן תְּלַת מְאָה וְאַטְעִיאָה נְשׁוֹהִי יַת לִבֵּיהּ:
וַהֲוָה לְעִדָן דְסִיב שְׁלֹמֹה נְשׁוֹהִי אַטְעִיאָה יַת לִבֵּיהּ בָּתַר טַעֲוַת עַמְמַיָא וְלָא הֲוָה לִבֵּיהּ שְׁלִים בִּדְחַלְתָּא דַייָ אֱלָהֵיהּ כְּלִבָּא דְדָוִד אֲבוּהִי:
וַאֲזַל שְׁלֹמֹה בָּתַר עַשְׁתּוֹרֶת טַעֲוַת צִידוֹנָאֵי וּבָתַר מִלְכּוֹם שִׁקוּץ עַמוֹנָאֵי:
וַעֲבַד שְׁלֹמֹה דְבִישׁ קֳדָם יְיָ וְלָא אַשְׁלִים בָּתַר דַחַלְתָּא דַייָ כְּדָוִד אֲבוּהִי:
בְּכֵן בְּנָא שְׁלֹמֹה בָּמָתָא לִכְמוֹשׁ שִׁקוּץ מוֹאָב בְּטוּרָא דִי עַל אַפֵּי יְרוּשְׁלֵם וּלְמוֹלֶךְ שִׁקוּץ בְּנֵי עַמוֹן:
וְכֵן עֲבַד לְכָל נְשׁוֹהִי בְּנַת עַמְמַיָא מַסְקִין בּוּסְמִין וּמְדַבְּחַן לְטַעֲוַתְהוֹן:
וּתְקֵיף רוּגְזָא דַייָ בִּשְׁלֹמֹה אֲרֵי אִתְפְּנֵי לִבֵּיהּ מִדְחַלְתָּא דַייָ אֱלָהָא דְיִשְׂרָאֵל דְאִתְגְלֵי לֵיהּ תַּרְתֵּין זִמְנִין:
וּפַקֵיד עֲלוֹהִי עַל פִּתְגָמָא הָדֵין בְּדִיל דְלָא לִמְהַךְ בָּתַר טַעֲוַת עַמְמַיָא וְלָא נְטַר יַת דְפַקֵיד יְיָ:
וַאֲמַר יְיָ לִשְׁלֹמֹה חֲלַף דַהֲוַת דָא עִמָךְ וְלָא נְטַרְתָּ פִּקוּדֵי וּקְיָמַי דִי פַקְדֵית עֲלָךְ אַעְדָאָה אֶעְדֵי יַת מַלְכוּתָא מִנָךְ וְאֶתְּנֶנָה לַעַבְדָךְ:
בְּרַם בְּיוֹמָךְ לָא אֶעְבְדִנָה בְּדִיל דָוִד אֲבוּךְ מִיַד בְּנָךְ אֶעְדִינָה:
לְחוֹד יַת כָּל מַלְכוּתָא לָא אֶעְדֵי שִׁבְטָא חַד אֶתֶּן לִבְרָךְ בְּדִיל דָוִד עַבְדִי וּבְדִיל יְרוּשְׁלֵם קַרְתָּא דְאִתְרְעֵיתִי:
וַאֲקֵים יְיָ סָטָן לִשְׁלֹמֹה יַת הֲדַד אֱדוֹמָאָה מִזַרְעָא דְמַלְכָּא הוּא בֶּאֱדוֹם:
וַהֲוָה כַּד הֲוָה דָוִד בֶּאֱדוֹם כַּד סְלִיק יוֹאָב רַב חֵילָא לְחַלָצָא קְטִילַיָא וּקְטַל כָּל דְכוּרָא בֶּאֱדוֹם:
אֲרֵי שִׁתָּא יַרְחִין יְתִיב תַּמָן יוֹאָב וְכָל יִשְׂרָאֵל עַד דִי שֵׁיצֵי כָּל דְכוּרָא בֶּאֱדוֹם:
וַעֲרַק אֲדַד הוּא וְגַבְרִין אֱדוֹמָאִין מֵעַבְדֵי אֲבוּהִי עִמֵיהּ לְמֵיעַל לְמִצְרָיִם וַהֲדַד יָנִיק זְעֵיר:
וְקָמוּ מִמִדְיָן וַאֲתוֹ לְפָארָן וּדְבַרוּ גַבְרִין עִמְהוֹן מִפָּארָן וַאֲתוֹ לְמִצְרַיִם לְוַת פַּרְעֹה מַלְכָּא דְמִצְרַיִם וִיהַב לֵיהּ בֵּיתָא וּמָזוֹן אֲמַר לְסַפְּקָא לֵיהּ וְאַרְעָא יְהַב לֵיהּ:
וְאַשְׁכַּח הֲדַד רַחֲמִין בְּעֵינֵי פַרְעֹה לַחֲדָא וִיהַב לֵיהּ אִתְּתָא יַת אֲחֵת אִתְּתֵיהּ אֲחָתֵיהּ דְתַחְפְּנֵיס מַלְכְּתָא:
וִילֵידַת לֵיהּ אֲחָתָא דְתַחְפְּנֵיס יַת גְנוּבַת בְּרֵיהּ וַחֲסַלְתֵּיהּ תַּחְפְּנֵס בְּגוֹ בֵּית פַּרְעֹה וַהֲוָה גְנוּבַת בְּבֵּית פַּרְעֹה בְּגוֹ בְּנֵי פַרְעֹה:
וַהֲדַד שְׁמַע בְּמִצְרַיִם אֲרֵי שְׁכִיב דָוִד עִם אֲבָהָתוֹהִי וַאֲרֵי מִית יוֹאָב רַב חֵילָא וַאֲמַר הֲדַד לְפַרְעֹה שַׁלְחֵנִי וֶאֱהַךְ לְאַרְעִי:
וַאֲמַר לֵיהּ פַּרְעֹה אֲרֵי מָא אַתְּ חָסֵר עִמִי דְאַתְּ בָּעֵי לְמֵיזַל לְאַרְעָךְ וַאֲמַר לָא אֲרֵי שַׁלָחָא תְּשַׁלְחִינַנִי:
וַאֲקֵם יְיָ לֵיהּ סָטָן יַת רְזוֹן בַּר אֶלְיָדָע דַעֲרַק מִן קֳדָם הֲדַדְעֶזֶר מַלְכָּא דְצוֹבָה רִבּוֹנֵיהּ:
וּכְנַשׁ עֲלוֹהִי גַבְרִין וַהֲוָה רַב מַשְׁרִיתָא כַּד קְטִיל דָוִד יַתְהוֹן וַאֲזַלוּ לְדַמֶשֶׂק וִיתִיבוּ בָהּ וּמְלַכוּ בְּדַמֶשֶׂק:
וַהֲוָה סָטָן לְיִשְׂרָאֵל כָּל יוֹמֵי שְׁלֹמֹה וְיַת בִּישְׁתָא דַעֲבַד הֲדָד וּמְרִיד בְּיִשְׂרָאֵל וּמְלַךְ עַל אֲרָם:
וְיָרָבְעָם בַּר נְבָט אֶפְרָתִי מִן צְרֵדָה וְשׁוּם אִמֵיהּ צְרוּעָה אִתְּתָא אַרְמַלְתָּא עַבְדָא לִשְׁלֹמֹה וַאֲרֵים יְדָא בְּמַלְכָּא:
וְדֵין פִּתְגָמָא דַאֲרֵים יְדָא בְּמַלְכָּא שְׁלֹמֹה בְּנָא יַת מִלֵיתָא אֲחַד יַת תַּרְעֲתָא קַרְתָּא דְדָוִד אֲבוּהִי:
וְגַבְרָא יָרָבְעָם גִבַּר חֵילָא וַחֲזָא שְׁלֹמֹה יַת עוּלֵימָא אֲרֵי עֲבִיד עַבִידְתָּא הוּא וּמַנֵי יָתֵיהּ לְכָל מַסְקֵי מִסִין דְבֵית יוֹסֵף:
וַהֲוָה בְּעִידָנָא הַהִיא וִיָרָבְעָם נְפַק מִירוּשְׁלֵם וְאַשְׁכַּח יָתֵיהּ אֲחִיָה שִׁלוֹנָאָה נְבִיאָה בְּאוֹרְחָא וְהוּא מִתְכַּסֵי בִּכְסוּתָא חַדְתָּא וְתַרְוֵיהוֹן אִינוּן בִּלְחוֹדֵיהוֹן בְּחַקְלָא:
וַאֲחַד אֲחִיָה בִּכְסוּתָא חַדְתָּא דִי עֲלוֹהִי וּסְדִיקָהּ לִתְרֵין עֲסַר סִדְקִין:
וַאֲמַר לְיָרָבְעָם סַב לָךְ עַסְרָא סִדְקִין אֲרֵי כִדְנַן אֲמַר יְיָ אֱלָהָא דְיִשְׂרָאֵל הָא אֲנָא מַעֲדֵי יַת מַלְכוּתָא מִידָא דִשְׁלֹמֹה וְאֶתֵּן לָךְ יַת עַסְרֵי שִׁבְטִין:
וְשִׁבְטָא חָד יְהֵי לֵיהּ בְּדִיל דָוִד עַבְדִי וּבְדִיל יְרוּשְׁלֵם קַרְתָּא דְאִתְרְעֵיתִי בָהּ מִכֹּל שִׁבְטַיָא דְיִשְׂרָאֵל:
חֲלַף דִשְׁבַקוּ פוּלְחָנִי וּסְגִידוּ לְעַשְׁתּוֹרֶת טַעֲוַת צִדוֹנָאֵי וְלִכְמוֹשׁ טַעֲוַת מוֹאֲבָאֵי וּלְמִלְכּוֹם טַעֲוַת בְנֵי עַמוֹן וְלָא הֲלִיכוּ בְּאוֹרְחָן דְתַקְנָן קֳדָמַי לְמֶעְבַּד דְכָשַׁר קֳדָמַי וּקְיָמַי וְדִינַי כְּדָוִד אֲבוּהִי:
וְלָא אֶסַב יַת כָּל מַלְכוּתָא מִינֵיהּ אֲרֵי מַלְכָּא אֲשַׁוִינֵיהּ כָּל יוֹמֵי חַיוֹהִי בְּדִיל דָוִד עַבְדִי דְאִתְרְעֵיתִי בֵיהּ דִנְטַר פִּקוּדֵי וְאוֹרַיְתִי וּקְיָמַי:
וְאֶסַב מַלְכוּתָא מִיַד בְּרֵיהּ וְאֶתְּנָהּ לָךְ יַת עַסְרָא שִׁבְטִין:
וְלִבְרֵיהּ אֶתֵּן שִׁבְטָא חַד בְּדִיל לְקַיָמָא מַלְכוּ לְדָוִד עַבְדִי כָּל יוֹמַיָא קֳדָמַי בִּירוּשְׁלֵם קַרְתָּא דְאִתְרְעֵיתִי לְאַשְׁרָאָה שְׁכִנְתִּי תַּמָן:
וְיָתָךְ אֱקָרֵיב וְתִמְלוֹךְ בְּכָל דְרַעֲוָא נַפְשָׁךְ וּתְהֵי מַלְכָּא עַל יִשְׂרָאֵל:
וִיהֵי אִם תְקַבֵּיל לְמֵימְרִי יַת כָּל דַאֲפַקְדִינָךְ וּתְהַךְ בְּכָל אוֹרְחַן דְתַקְנָן קֳדָמַי וְתַעֲבֵּיד דְכָשַׁר בְּעֵינַי לְמִטַר קְיָמַי וּפִקוּדַי כְּמָא דַעֲבַד דָוִד עַבְדִי וִיהֵי מֵימְרִי בְּסַעֲדָךְ וַאֲקַיֵם לָךְ מַלְכוּ קַיָמָא כְּמָא דְקַיֵמִית לְדָוִד וְאֶתֵּן לָךְ יַת יִשְׂרָאֵל:
וְאַעֲנֵי יַת זַרְעָא דְדָוִד בְּדִיל דָא בְּרַם לָא כָל יוֹמַיָא:
וּבְעָא שְׁלֹמֹה לְמִקְטַל יַת יָרָבְעָם וְקָם יָרָבְעָם וַעֲרַק לְמִצְרַיִם לְוַת שִׁישַׁק מַלְכָּא דְמִצְרַיִם וַהֲוָה בְּמִצְרָיִם עַד דְמִית שְׁלֹמֹה:
וּשְׁאָר פִּתְגָמֵי שְׁלֹמֹה וְכָל דִי עֲבַד וְחוּכְמָתֵיהּ הֲלָא אִינוּן כְּתִיבִין עַל סִפְרָא פִּתְגָמֵי שְׁלֹמֹה:
וְיוֹמַיָא דִמְלַךְ שְׁלֹמֹה בִּירוּשְׁלֵם עַל כָּל יִשְׂרָאֵל אַרְבְּעִין שְׁנִין:
וּשְׁכִיב שְׁלֹמֹה עִם אֲבָהָתוֹהִי וְאִתְקְבַר בְּקַרְתָּא דְדָוִד אֲבוּהִי וּמְלַךְ רְחַבְעָם בְּרֵיהּ תְּחוֹתוֹהִי:
