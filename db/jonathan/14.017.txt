וַהֲוָה פִּתְגַם נְבוּאָה מִן קֳדָם יְיָ עִמִי לְמֵימַר:
בַּר אָדָם חוּד חִידְתָּא וּמְתַל מַתְלָא עַל בֵּית יִשְׂרָאֵל:
וְתֵימַר כִּדְנַן אֲמַר יְיָ אֱלֹהִים נִשְׁרָא רַבָּא רַב גַפִּין אֲרִיךְ אֵיבְרָן חָפֵי כַנְפִין דִילֵהּ צִיוּרִין אֲתָא לְלִבְנָן וּנְסֵיב יַת עוֹבַרְתָּא דְאַרְזָא רָמָא:
יַת רֵישׁ שַׁבִּשְׁתֵּהּ נְסִיב וְאוֹבְלֵיהּ לְאַרְעָא דְבָטְלָא מִפּוּלְחָנָא בְאַרְעָא דִכְנַעַן וְעַד לָא יַעֲלוּן לָהּ בֵּית יִשְׂרָאֵל בְּקִרְיַת תַּגָרִין שַׁוְיֵיהּ:
וּנְסִיב מִזַרְעָא דְאַרְעָא וְיַהֲבֵיהּ בַּחֲקַל טַב לְמִזְרַע נִצְבַהּ עַל מַיִין סַגִיאִין גְפַּן נִצְבָא שַׁוְיֵיהּ:
וּצְמַח וַהֲוָה לְגֶפֶן תַּקִיף מַכִּיךְ בְּקוֹמָא לְכַפָפָא עוֹבַרְתֵּהּ לֵיהּ וְשָׁרָשׁוֹהִי תְּחוֹתוֹהִי יְהוֹן וַהֲוָה לְגֶפֶן וַעֲבַד שַׁבְקִין וּשְׁלַח עוֹבָרָן:
וַהֲוָה נִשְׁרָא חַד רַבָּא רַב גַפִּין וַחֲפֵי כַנְפִין וְהָא גוּפְנָא הָדָא כַּפַּת שָׁרְשׁוֹהִי עֲלוֹהִי וְעוֹבַרְתֵּהּ שְׁלָחַת לֵיהּ לְאַשְׁקָאָה יָתָהּ מֵעוּגְיַת נִצְבָתָהּ:
בַּחֲקַל טַב עֲלֵי מַיִין סַגִיאִין הִיא שְׁתִילָה לְמֶעְבַר עֲנַף וּלְמִטְעַן פֵּרִין לְמֶהֱוֵי לְגֶפֶן תַקִיף:
אֲמַר כִּדְנַן יְיָ אֱלֹהִים תִּצְלַח הֲלָא יַת שָׁרְשׁוֹהִי יַפְסֵיק וְיַת אִבָּהּ יַקְטֵיף וְיִבְשׁוּן כָּל טַרְפֵי צִמְחָהּ יִתְּרוֹן וְלָא בְמַשִׁרְיָן סַגִיאִין וּבְעַם סַגִי יֵיתוּן לְמֶעְקַר יָתָהּ מִשׁוֹרְשָׁהָא:
וְהָא שְׁתִילָא הֲתִצְלַח הֲלָא כְּמִיקְרַב בָּה רוּחַ קִידוּמָא תִיבַשׁ מֵיבַשׁ עַל עוּגְיַת נִצְבְתָהּ תִּיבָשׁ:
וַהֲוָה פִּתְגָם נְבוּאָה מִן קֳדָם יְיָ עִמִי לְמֵימָר:
אֱמַר כְּעַן לְעַמָא סַרְבָנָא הֲלָא יְדַעְתּוּן מָה אִלֵין אֱמַר הָא אָתָא מַלְכָּא דְבָבֶל לִירוּשְׁלֵם וּדְבַר יַת מַלְכָּהּ וְיַת רַבְרְבָהָא וְאוֹבִיל יַתְהוֹן לְוָתֵה לְבָבֶל:
וּדְבַר מִזַרְעָא דְמַלְכוּתָא וּגְזַר עִמֵהּ קְיַם וְאָעֵיל יָתֵהּ בְּמוֹמֵי וְיַת רַבְרְבֵי אַרְעָא דְבַר:
לְמֶהֱוֵי מַלְכוּ חַלָשָׁא בְּדִיל דְלָא לְאִתְרַבְרְבָא לְמִטַר יַת קְיָמַהּ לְמִפְלְחֵהּ:
וּמְרַד בֵּיהּ לְשַׁלָחָא אִזְגַדוֹהִי לְמִצְרַיִם לְמִתַּן לֵיהּ סוּסְוָן וְעַם סַגִי הֲיִצְלַח הֲיִשְׁתֵּיזִיב דְעָבֵיד אִילֵין וּמְשַׁנֵי קַיֵם הֲיִשְׁתֵּיזֵיב:
קַיֵם אֲנָא אֲמַר יְיָ אֱלֹהִים אִם לָא בְּאַתְרָא דְמַלְכָּא דְאַמְלִיךְ יָתֵיהּ דִבְסַר עַל מוֹמָתֵהּ וּדְאַשְׁנֵי יַת קְיָמֵיהּ עִמֵיהּ בְּגוֹ בָבֶל יְמוּת:
וְלָא בְּמַשִׁרְיָן סַגִיאִין וּבְעַם סַגִי יַעְבֵד עִמֵהּ פַרְעֹה קְרָבָא בְּמִצְבַּר מְלֵיתָא וּבְמִבְנֵי כַרְקוֹם לְשֵׁיצָאָה מְדִינַן סַגִיאִין:
וּבְסַר עַל מוֹמֵי לְאַשְׁנָאָה קְיַם וְהָא אַמְטֵי יְדֵיהּ לְפַרְעֹה וְכָל אִלֵין עֲבַד לָא יִשְׁתֵּיזֵיב:
בְּכֵן כִּדְנַן אֲמַר יְיָ אֱלֹהִים קַיָם אֲנָא אִם לָא מוֹמָתִי דִבְסַר וּקְיָמִי דְשַׁנִי וְאֶתֵּן פּוּרְעֲנוּת אוֹרְחֵהּ בְּרֵישֵׁהּ:
וְאֶפְרוֹס עֲלוֹהִי מְצֻדָתִי וְיִתְאַחַד בִּסְרִגְתִּי וְאַגְלִינֵהּ לְבָבֶל וְאִתְפְּרַע מִנֵיהּ תַּמָן שִׁקְרַהּ דִשְׁקַר קֳדָמַי:
וְיַת כָּל גַבְּרוֹהִי בְּכָל מַשִׁרְיָתֵיהּ בְּחַרְבָּא יִתְקַטְלוּן וּדְיִשְׁתָּאֲרוּן לְכָל רוּחַ יִתְבַּדְרוּן וְתֵידְעוּן אֲרֵי אֲנָא יְיָ גְזָרֵית בְּמֵימְרִי:
כִּדְנַן אֲמַר יְיָ אֱלֹהִים וַאֲקָרֵיב אֲנָא מִמַלְכוּתָא דְבֵית דָוִד דִמְתִיל כְּאַרְזָא רָמָה וַאֲקַיְמִינֵהּ מִבְּנֵי בְנוֹהִי יָנִיק וַאֲרַבֵּי וַאֲקַיְמִינֵיהּ בְּמֵימְרִי בְּטוּר רָם וּמְנַטֵל:
בְּטוּר קוּדְשָׁא דְיִשְׂרָאֵל אֲקַיְמִינֵיהּ וְיַכְנֵישׁ מַשִׁרְיָן וְיַעְבֵיד סוּמְכְוָן וִיהֵי לְמֶלֶךְ תַּקִיף וְיִסְתַּמְכוּן עֲלוֹהִי כָּל צַדִיקַיָא וְכָל עִנְוְתָנַיָא בִּטְלַל מַלְכוּתֵיהּ יִשְׁרוּן:
וְיֵדְעוּן כָּל מַלְכֵי עַמְמַיָא אֲרֵי אֲנָא יְיָ אַמְאִיכֵית מַלְכוּת דַהֲוַת תַּקִיפָא וְתַקִיפַת מַלְכוּת דַהֲוַת חַלָשָׁא אַמְאִיכֵית מַלְכוּת עַמְמַיָא דַהֲוַת תַּקִיפָא כְּאָעָא רָטִיב וּתְקִיפִית מַלְכוּת בֵּית יִשְׂרָאֵל דַהֲוַת חַלָשָׁא כְּאָעָא יַבִּישׁ אֲנָא יְיָ גְזָרִית בְּמֵימְרִי וַאֲקַיֵם:
