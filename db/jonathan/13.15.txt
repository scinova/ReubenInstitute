וַאֲמַר יְיָ לִי אִם יְקוּם מֹשֶׁה וּשְׁמוּאֵל לְמִבְעֵי מִן קָדָמַי לָא רַעֲוָא קָדָמַי בְּעַמָא הָדֵין שְׁלַח מֵאֲרַע בֵּית שְׁכִנְתִּי וְיִגְלוֹן:
וִיהֵי אֲרֵי יֵימְרוּן לָךְ לְאָן נִגְלֵי וְתֵימַר לְהוֹן כִּדְנַן אֲמַר יְיָ לִדְמוֹתָא לְמוֹתָא וְדִלְחַרְבָּא לְחַרְבָּא וְדִלְכַפְנָא לְכַפְנָא וְדִי לִשִׁבְיָא לְשִׁבְיָא:
וְאַסְעַר עֲלֵיהוֹן אַרְבַּע פוּרְעֲנוּן בִּישַׁן אֲמַר יְיָ יַת דְקַטְלִין בְּחַרְבָּא לְמִקְטַל וְיַת כַּלְבַיָא לְגָרְרָא וְיַת עוֹפָא דִשְׁמַיָא וְיַת בְּעִירָא דְאַרְעָא לְמֵיכַל וּלְחַבָּלָא:
וְאֶתְּנִינוּן לִזְיַע לְכָל מַלְכְּוַת אַרְעָא עַל דְלָא תָבוּ כִּמְנַשֶׁה בַר יְחִזְקִיָה מְלַךְ שִׁבְטָא דְבֵית יְהוּדָה עַל דְאַסְגֵי לְמֶעְבַּד בִּירוּשְׁלֵם:
אֲרֵי מַן יֵחוּס עֲלָךְ יְרוּשְׁלֵם וּמַן יִדְוֵי עֲלָךְ וּמַן יְזוּר לְמִשְׁאַל לִשְׁלָמָא לָךְ:
אַתְּ שְׁבַקְתְּ פּוּלְחָנִי אֲמַר יְיָ לַאֲחוֹרָא אִסְתַּחַרְתְּ וְאָרֵים יַת מְחַת גְבוּרְתִּי עֲלָךְ וֶאֱחַבְּלִינָךְ עַל דַהֲוֵית יָכְלָא לְמֵיתַב וְלָא תָבְתְּ:
וּכְמָא דְדָרַן בִּמְדִירַיָא כֵּן אֲבַדֵר יַת עַמִי בְּקִרְוֵי אַרְעָא תַּכְלִית אַבְּדֵית יַת עַמִי מֵאוֹרְחַתְהוֹן לָא תָבוּ:
סַגִיאַת קָדָמַי אַרְמְלָתְהוֹן מֵחָל יַמְמִין אַיְתֵיתִי עֲלֵיהוֹן עַל סִיעַת עוּלֵימֵיהוֹן בְּזוֹזִין כַּד בְּטִיהֲרָא אַיְתֵיתִי עֲלֵיהוֹן מַשִׁרְיָן בִּתְכֵּיף וְחַבִּילָא קִרְוֵיהוֹן:
צְדִיאַת אַרְעֲהוֹן מִבְּלִי עַמָא הֲווֹ לְמַפְּחַת נְפַשׁ גְלָא יְקַרְהוֹן בְּחַיֵיהוֹן וְעַד דְאִינוּן קַיְמִין בְּהִיתוּ וְאִתְכְּנָעוּ וּשְׁאַרְהוֹן לְחַרְבָּא אֶמְסַר קֳדָם בַּעֲלֵי דְבָבֵיהוֹן אֲמַר יְיָ:
וַי לִי אִמִי אֲרֵי יְלִדְתִּינִי לְמֶהֱוֵי גְבַר דִין וֶאֱנַשׁ מָצוּ לְכָל יַתְבֵי אַרְעָא לָא רָשָׁן בִּי וּמָא דַאֲנָא דַשֵׁינָא לָא תָבַע מִנְהוֹן כּוּלְהוֹן מְלַטְטִין לִי:
אֲמַר יְיָ אִם לָא סוֹפֵךְ יְהֵא לְטַב אִם לָא יֵיתוּן וְיִבְעוֹן מִנָךְ בְּעִדַן דְאַיְתִי עֲלֵיהוֹן סַנְאָה וְיָעִיק לְהוֹן בָּעֵיל דְבָבָא:
מְלַךְ דְתַקִיף כְּבַרְזְלָא יִסַק לְסָעֵיר מְלַךְ דְתַקִיף כְּבַרְזְלָא וְכִנְחָשָׁא יֵתֵי מִצִפּוֹנָא עֲלָהּ לְמִתְבָּרָא:
נִכְסֵיכוֹן וּבֵית גִנְזֵיכוֹן לְבִזָא אֶמְסַר לָא בְּדָמִין בְּחוֹבִין דַהֲוֵיתוּן פָּלְחִין לְטַעֲוָתָא וּבְכָל תְּחוּמֵיכוֹן:
וְתִשְׁתַּעְבְּדוּן לְבַעֲלֵי דְבָבֵיכוֹן בְּאַרְעָא דְלָא יְדַעְתּוּן אֲרֵי קִידוּם תַּקִיף כְּאֶשְׁתָּא נְפַק מִן קָדָמַי בִּרְגַז עֲלֵיכוֹן יֵחוֹל:
אִתְגְלֵי קָדָמָךְ יְיָ יֵעוֹל דוּכְרָנִי וּפַקְדֵנִי לְאוֹטָבָא לִי וְאַסְעַר עַל בַּעֲלֵי דְבָבֵי אִתְפָּרַע לִי מֵרָדְפַי לָא תִתֵּן אַרְכָּא לְעוּלְבָּנִי גְלֵי קָדָמָךְ דְקַבְּלִית חִסוּדִין עַל מֵימְרָךְ:
קַבְּלֵית לְפִתְגָמָךְ וְקָיֵמְתִּינוּן וַהֲוָה מֵימְרָךְ לִי לְבִיעַ וּלְחֶדְוַת לִבִּי אֲרֵי אִתְקְרֵי שְׁמָךְ עֲלַי יְיָ אֱלֹהִים צְבָאוֹת:
לָא יְתֵיבֵית בְּסִיעַת מְשַׁבְּחִין וְלָא חֲדֵיתִי מִן קֳדָם מֵימְרָךְ פָּרֵישׁ יְתֵיבִית בִית אֲרֵי נְבוּאָה דַלְוָט מַלֵיתָנִי:
לְמָה הֲוָה כֵיבִי לְאַפְרַשׁ וּמְחָתִי תַקִיפָא לִמְסַרְבָא לְאִתַּסָאָה לָא יְהֵי מֵימְרָךְ לִי כַּדְבִין כְּמַבַּע דְמַיִין דְפַסְקִין מוֹהִי:
בְּכֵן כִּדְנַן אֲמַר יְיָ אִם תְּתוּב וַאֲתֵיבִינָךְ קָדָמֵי תְּשַׁמֵשׁ וְאִם תְּתֵיב רַשִׁיעַיָא לְמֶהֱוֵי צַדִיקַיָא רַעֲוַת מֵימְרִי תְקַיֵם יְתוּבוּן אִינוּן לְפִתְגָמָךְ וְאַתְּ לָא תְתוּב לְמִטְעֵי בַּתְרֵיהוֹן:
וְאֶתְּנִינָךְ לְעַמָא הָדֵין כְּשׁוּר דִנְחַשׁ כְּרִיךְ וִיגִיחוּן לְקִבְּלָךְ וְלָא יִכְלוּן לָךְ אֲרֵי בְסַעֲדָךְ מֵימְרִי לְמַפְרָקָךְ וּלְשֵׁיזָבוּתָךְ אָמַר יְיָ:
וֶאֱשֵׁזְבִינָךְ מִיַד מַבְאֲשִׁין וְאֶפְרִיקָנָךְ מִיַד תִּקִיפִין:
