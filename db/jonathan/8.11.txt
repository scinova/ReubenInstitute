וּסְלֵיק נָחָשׁ מַלְכָּא דִבְנֵי עַמוֹן וּשְׁרָא עַל יָבֵשׁ גִלְעָד וַאֲמַרוּ כָּל אֱנָשֵׁי יָבֵשׁ לְנָחָשׁ גְזַר לָנָא קְיָם וְנִפְלְחִינָךְ:
וַאֲמַר לְהוֹן נָחָשׁ מַלְכָּא דִבְנֵי עַמוֹן בְּהָדָא אֶגְזַר לְכוֹן בְּמֵיקַר לְכוֹן כָּל עֵינָא דְיַמִינָא וַאֲשַׁוִינָה חִיסוּדָא עַל כָּל יִשְׂרָאֵל:
וַאֲמַרוּ לֵיהּ סָבֵי יָבֵשׁ אוֹרִיךְ לָנָא שַׁבְעָא יוֹמִין וְנִשְׁלַח אִזְגַדִין בְּכָל תְּחוּם יִשְׂרָאֵל וְאִם לֵית דְפָרִיק יָתָנָא וְנִפּוֹק לְוָתָךְ:
וַאֲתוֹ אִזְגַדַיָא לְגִבְעֲתָא דְשָׁאוּל וּמַלִילוּ פִּתְגָמַיָא קֳדָם עַמָא וַאֲרִימוּ כָל עַמָא יַת קָלֵיהוֹן וּבְכוֹ:
וְהָא שָׁאוּל אָתָא בָּתַר תּוֹרַיָא מִן חַקְלָא וַאֲמַר שְׁאוּל מַה לְעַמָא אֲרֵי בָכָן וְאִשְׁתַּעִיאוּ לֵיהּ יַת פִּתְגָמֵי אֱנָשֵׁי יָבֵשׁ:
וּשְׁרַת רוּחַ גְבוּרָא מִן קֳדָם יְיָ עַל שָׁאוּל כַּד שְׁמַע יַת פִּתְגָמַיָא הָאִלֵין וּתְקֵיף רוּגְזֵיהּ לַחֲדָא:
וּנְסִיב פַּדַן תּוֹרַיָא וְהַדְמֵיהּ וּשְׁלַח בְּכָל תְּחוּם יִשְׂרָאֵל בְּיַד אִזְגַדַיָא לְמֵימַר דְלֵיתוֹהִי נָפֵיק בָּתַר שָׁאוּל וּבָתַר שְׁמוּאֵל כְּדֵין יִתְעֲבֵיד לְתוֹרֵיהּ וּנְפַל זִיעָא מִן קֳדָם יְיָ עַל עַמָא וּנְפָקוּ כְּגַבְרָא חָד:
וּמְנִינוּן בְבָזָק וַהֲווֹ בְנֵי יִשְׂרָאֵל תְּלַת מְאָה אַלְפִין וֶאֱנַשׁ יְהוּדָה תְּלָתִין אַלְפִין:
וַאֲמַרוּ לְאִזְגַדַיָא דַאֲתוֹ כְּדֵין תֵּימְרוּן לֶאֱנָשׁ יָבֵשׁ גִלְעָד מְחַר תְּהִי לְכוֹן פּוּרְקָנָא בְּמֵיחַם שִׁמְשָׁא וַאֲתוֹ אִזְגַדַיָא וְחַוִיאוּ לֶאֱנָשֵׁי יָבֵשׁ וַחֲדִיאוּ:
וַאֲמַרוּ אֱנָשֵׁי יָבֵשׁ מְחַר נִפּוֹק לְוַתְכוּן וְתַעְבְּדוּן לָנָא כְּכוֹל דְתַקִין בְּעֵינֵיכוֹן:
וַהֲוָה בְּיוֹמָא דְבַתְרוֹהִי וּמַנִי שָׁאוּל יַת עַמָא תְּלָתָא מַשִׁירְיָן וַעֲלוֹ בְּגוֹ מַשְׁרִיתָא בְּמַטְרַת צַפְרָא וּמְחוֹ יַת בְּנֵי עַמוֹן עַד מֵיחַם יוֹמָא וַהֲווֹ דְאִשְׁתָּאֲרוֹ וְאִתְבַּדָרוּ וְלָא אִשְׁתָּאֲרוּ בְּהוֹן תְּרֵין דְעַרְקִין כַּחֲדָא:
וַאֲמַר עַמָא לִשְׁמוּאֵל מָא דֵין דִי מְבַסֵר לְמֵימַר לָא כָשַׁר שָׁאוּל לְמִמְלַךְ עֲלָנָא אַיְתוֹ גַבְרַיָא וְנַקְטְלִינוּן:
וַאֲמַר שָׁאוּל לָא יִתְקְטֵיל גַבְרָא בְּיוֹמָא הָדֵין אֲרֵי יוֹמָא דֵין עֲבַד יְיָ פּוּרְקָנָא בְּיִשְׂרָאֵל:
וַאֲמַר שְׁמוּאֵל לְעַמָא אִיתוּ וְנֵזִיל לְגִלְגָלָא וּנְחַדַת תַּמָן מַלְכוּתָא:
וַאֲזַלוּ כָּל עַמָא לְגִלְגָלָא וְאַמְלִיכוּ תַמָן יַת שָׁאוּל קֳדָם יְיָ בְּגִלְגָלָא וּנְכִיסוּ תַמָן נִכְסַת קוּדְשִׁין קֳדָם יְיָ וַחֲדֵי תַמָן שָׁאוּל וְכָל אֱנָשֵׁי יִשְׂרָאֵל עַד לַחֲדָא:
