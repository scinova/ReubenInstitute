וַהֲוָה פִּתְגַם נְבוּאָה מִן קֳדָם יְיָ עִם יוֹנָה בַר אֲמִתַּי לְמֵימָר:
קוּם אֱזֵיל לְנִינְוֵה קַרְתָּא רַבְּתָא וְאִתְנַבֵּי עַלָהּ אֲרֵי סְלִיקַת בִּישַׁתְהוֹן קֳדָמָי:
וְקָם יוֹנָה לְמֶעְרַק לְיַמָא מִן קֳדָם דְאִתְנַבֵּי בִּשְׁמָא דַייָ וּנְחַת לְיָפוֹ וְאַשְׁכַּח אִילְפָא דַאֲזָלָא לְיַמָא וִיהַב אַגְרָא וּנְחַת בָּהּ לְמֵיזַל עִמְהוֹן בְּיַמָא מִן קֳדָם דְאִתְנַבֵּי בִּשְׁמָא דַיָי:
וַיָי אֲרִים רוּחַ רַב עַל יַמָא וַהֲוָה נַחְשׁוֹל רַב בָּיַמָא וְאִילְפָא בָּעֲיָא לְאִתַּבְּרָא:
וּדְחִילוּ סַפְנַיָא וּבְעוֹ גְבַר מִן דְחַלְתֵּיהּ וַחֲזוֹ אֲרֵי לֵית בְּהוֹן צְרוֹךְ וּרְמוֹ יַת מָנַיָא דִבְאִילְפָא לְיַמָא לְאַקָלָא מִנְהוֹן וְיוֹנָה נְחֵית לְאַרְעִית שִׁדָא דְאִילְפָא וּשְׁכִיב וּדְמוּךְ:
וּקְרַב לְוָתֵיהּ רַב סְפַנְיָא וַאֲמַר לֵיהּ מָא אַתְּ דְמוּךְ קוּם צַלֵי קֳדָם אֱלָהָךְ מָאִים יִתְרַחֵים מִן קֳדָם יְיָ עֲלָנָא וְלָא נוֹבֵיד:
וַאֲמָרוּ גְבַר לְחַבְרֵיהּ אֱתוֹ וְנִרְמֵי עַדְבִין וְנִידַע בְּדִיל מָה בִישְׁתָא הָדָא לָנָא וּרְמוֹ עַדְבִין וּנְפַל עַדְבָא עַל יוֹנָה:
וַאֲמַרוּ לֵיהּ חַוֵי כְעַן לָנָא בְּדִיל מָה בִישְׁתָּא הָדָא לָנָא מָה עֲבִידְתָּךְ וּמְנָן אַתְּ אָתֵי אֵי דָא הִיא מְדִינְתָךְ וְאֵי מִדֵין עַמָא אַתְּ:
וַאֲמַר לְהוֹן יְהוּדָאָה אֲנָא וּמִן קֳדָם יְיָ אֱלָהָא דִשְׁמַיָא אֲנָא דָחֵל דַעֲבַד יַת יַמָא וְיַת יַבֶּשְׁתָּא:
וּדְחִילוּ גַבְרַיָא דְחִילָא רַבָּא וַאֲמָרוּ לֵיהּ מַה דָא עַבַדְתָּא אֲרֵי יְדָעוּ גוּבְרַיָא אֲרֵי מִן קֳדָם יְיָ הוּא עָרֵיק אֲרֵי חַוֵי לְהוֹן:
וַאֲמָרוּ לֵיהּ מַה נַעְבֵּיד לָךְ וִינוּחַ יַמָא מִנָנָא אֲרֵי יַמָא אָזֵיל וְנַחְשׁוֹלֵיהּ תַּקִיף:
וַאֲמַר לְהוֹן טְלוּנִי וּרְמוּנִי לְיַמָא וְיָנִיחַ יַמָא מַעֲלֵיכוֹן אֲרֵי יָדַע אֲנָא אֲרֵי בְדִילִי נַחְשׁוּלָא רַבָּא הָדֵיל עֲלֵיכוֹן:
וְשָׁיְטִין גַבְרַיָא לַאֲתָבָא לְיַבֶּשְׁתָּא וְלָא יְכִילוּ אֲרֵי יַמָא אָזֵיל וְנַחְשׁוֹלֵיהּ תַּקִיף עֲלֵיהוֹן:
וְצַלִיאוּ קֳדָם יְיָ וַאֲמָרוּ קַבֵּיל בְּעוּתָנָא יְיָ לָא כְעַן נֵיבַד בְּחוֹבַת נְפַשׁ דְגַבְרָא הָדֵין וְלָא תִתֵּן עֲלָנָא חוֹבַת דַם זַכַּאי אֲרֵי אַתְּ יְיָ כְּמָא דְרַעֲוָא קֳדָמָךְ עֲבַדְתָּא:
וּנְסִיבוּ יַת יוֹנָה וּרְמוֹהִי לְיַמָא וְנַח יַמָא מִנַחְשׁוֹלֵיהּ:
וּדְחִילוּ גַבְרַיָא דְחִילָא רַבָּא מִן קֳדָם יְיָ וַאֲמַרוּ לְדַבָּחָא דְבַח קֳדָם יְיָ וּנְדָרוּ נִדְרִין:
