וְקָם בָּתַר אֲבִימֶלֶךְ לְמִפְרַק יַת יִשְׂרָאֵל תּוֹלָע בַּר פּוּאָה בַּר דּוֹדוֹ גַבְרָא מִשִׁבְטָא דְבֵית יִשָׂשכָר וְהוּא יָתֵיב בְּשָׁמִיר בְּטוּרָא דְבֵית אֶפְרָיִם:
וְדָן יַת יִשְׂרָאֵל עַסְרִין וּתְלַת שְׁנִין וּמִית וְאִתְקְבַר בְּשָׁמִיר:
וְקָם בַּתְרוֹהִי יָאִיר גִלְעָדָאָה וְדָן יַת יִשְׂרָאֵל עַסְרִין וְתַרְתֵּין שְׁנִין:
וַהֲווֹ לֵיהּ תְּלָתִין בְּנִין רְכִיבִין עַל תְּלָתִין עוּלִין וּתְלָתִין קִרְוִין לְהוֹן לְהוֹן קְרַן כַּפְּרָנֵי יָאִיר עַד יוֹמָא הָדֵין דִי בְּאַרְעָא דְגִלְעָד:
וּמִית יָאִיר וְאִתְקְבַר בְּקָמוֹן:
וְאוֹסִיפוּ בְּנֵי יִשְׂרָאֵל לְמֶעְבַּד דְבִישׁ קֳדָם יְיָ וּפְלָחוּ יַת בַּעֲלַיָא וְיַת עַשְׁתְּרָתָא וְיַת טַעֲוַת אֲרָם וְיַת טַעֲוַת צִידוֹן וְיַת טַעֲוַת מוֹאָב וְיַת טַעֲוַת בְּנֵי עַמוֹן וְיַת טַעֲוַת פְּלִשְׁתָּאֵי וּשְׁבַקוּ יַת פּוּלְחָנָא דַייָ וְלָא פְּלָחוּ קֳדָמוֹהִי:
וּתְקֵיף רוּגְזָא דַייָ בְּיִשְׂרָאֵל וּמַסְרִינוּן בְּיַד פְּלִשְׁתָּאֵי וּבְיַד בְּנֵי עַמוֹן:
וּדְחַקוּ וְשַׁעֲבִּידוּ יַת בְּנֵי יִשְׂרָאֵל בְּשַׁתָּא הַהִיא תְּמַנְיָא עַסְרֵי שְׁנִין יַת כָּל בְּנֵי יִשְׂרָאֵל דִבְעִבְרָא דְיַרְדְנָא בְּאַרְעָא אֱמוֹרָאָה דִי בְגִלְעָד:
וַעֲבָרוּ בְנֵי עַמוֹן יַת יַרְדְנָא לְאַגָחָא קְרָבָא אַף בִּיהוּדָה וּבִדְבֵית בִנְיָמִין וּבִדְבֵית אֶפְרַיִם וְעָקַת לְיִשְׂרָאֵל לַחֲדָא:
וּזְעִיקוּ בְּנֵי יִשְׂרָאֵל קֳדָם יְיָ לְמֵימָר חַבְנָא קֳדָמָךְ וַאֲרֵי שְׁבַקְנָא יַת פּוּלְחָנָא דַייָ אֱלָהָנָא וּפְלַחְנָא יַת בַּעֲלַיָא:
וַאֲמַר יְיָ לִבְנֵי יִשְׂרָאֵל הֲלָא מִמִצְרַיִם וּמִן אֱמוֹרָאָה וּמִן בְּנֵי עַמוֹן וּמִפְּלִשְׁתָּאֵי:
וְצִידוֹנָאֵי וַעֲמַלְקָאֵי וֶאֱנָשֵׁי מָעוֹן דָחֲקוּ יַתְכוֹן וּבָעִיתוּן קֳדָמַי וּפַרְקֵית יַתְכוֹן מִידֵיהוֹן:
וְאַתּוּן שְׁבַקְתּוּן פּוּלְחָנִי וּפְלַחְתּוּן לְטַעֲוַת עַמְמַיָא בְּכֵן לָא אוֹסִיף לְמִפְרַק יַתְכוֹן:
אֱזִילוּ וּבְעוֹ מִן טַעֲוָתָא דְאִיתְרְעִיתוּן בְּהוֹן הַאִם יָכְלִין לְמִפְרַקְכוֹן בְּעִידָן עֲקַתְכוֹן:
וַאֲמַרוּ בְנֵי יִשְׂרָאֵל קֳדָם יְיָ חַבְנָא עֲבֵיד אַתְּ לָנָא כְּכָל דְתַקִין קֳדָמָךְ בְּרַם שֵׁיזַבְנָא כְעַן יוֹמָא דֵין:
וְאַעֲדִיאוּ יַת טַעֲוַת עַמְמַיָא מִבֵּינֵיהוֹן וּפְלָחוּ קֳדָם יְיָ וְעָקַת נַפְשֵׁיהּ בַּעֲמַל יִשְׂרָאֵל:
וְאִתְכְּנִישׁוּ בְּנֵי עַמוֹן וּשְׁרוֹ בַּגִלְעָד וְאִתְכְּנִישׁוּ בְּנֵי יִשְׂרָאֵל וּשְׁרוֹ בְּמִצְפָּה:
וַאֲמַרוּ עַמָא רַבְרְבֵי גִלְעָד גְבַר לְחַבְרֵיהּ מַן גַבְרָא דְיִשְׁרֵי לְאַגָחָא קְרָבָא בִּבְנֵי עַמוֹן יְהֵי לְרֵישׁ לְכֹל יַתְבֵי גִלְעָד:
