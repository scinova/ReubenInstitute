וְאַעֲלַנִי לְהֵיכָלָא וּמְשַׁח יַת אֵילֵיָא שִׁית אַמִין פּוּתְיָא מִכָּא וְשִׁית אַמִין פּוּתְיָא מִכָּא פּוּתְיָא דְמַשְׁכְּנָא:
וּפוּתְיָא דְתַרְעָא עֲסַר אַמִין וְכִתְפֵי תַרְעָא חֲמֵשׁ אַמִין מִכָּא וַחֲמֵשׁ אַמִין מִכָּא וּמְשַׁח אוּרְכֵּיהּ אַרְבְּעִין אַמִין וּפוּתְיָא עַסְרִין אַמִין:
וְעַל לְגָיו וּמְשַׁח אֵילָא דְתַרְעָא תַּרְתֵּין אַמִין וְתַרְעָא שִׁית אַמִין וּפוּתְיָא דְתַרְעָא שְׁבַע אַמִין:
וּמְשַׁח יַת אוּרְכֵּיהּ עַסְרִין אַמִין וּפוּתְיָא עַסְרִין אַמִין דִי קֳדָם הֵיכָלָא וַאֲמַר לִי דֵין בֵּית קוֹדֶשׁ קוּדְשַׁיָא:
וּמְשַׁח כּוֹתְלָא דְבֵיתָא שִׁית אַמִין וּפוּתְיָא דִמְחִיצְתָּא אַרְבַּע אַמִין סְחוֹר סְחוֹר מַקַף לְבֵיתָא סְחוֹר סְחוֹר:
וּמְחִיצַיָא מְחִיצָא עַל מְחִיצָא תְּלָתִין וּתְלַת חֲדָא עַסְרֵי בְּסִידְרָא קְבִיעִין בְּכִתְלָא דְמִגָיו לִמְחִיצָא סְחוֹר סְחוֹר לְמֶהֱוֵי רֵישֵׁי שַׁרְיָתָא נָיְחִין עַל זִיזָא וְלָא יֶהֶוְיָן שַׁרְיָתָא מְעַבְּרָן בְּכָתְלֵי בֵּיתָא:
וּפּוּתְיָא מְסִבָּתָא וְסַלְקָא לְעֵלָא לְעֵלָא מִן מְחִיצַיָא אֲרֵי מַקְפָא בֵיתָא לְסַלָקָא לְעֵלָא לְעֵלָא סְחוֹר סְחוֹר לְגָיו עַל כֵּן פָּתֵי בֵיתָא מִלְעֲלָא וְכֵן בִּמְסִבָּתָא סַלְקִין מִן אַרְעִיתָא לְעִלֵיתָא בְּאוֹרַח מְצִיעֲתָא:
וַחֲזֵית לְבֵיתָא רוּמָא סְחוֹר סְחוֹר מְשַׁכְלְלָן מְחִיצַיָא מִשְׁחַת קָנַיָא שִׁית אַמִין רֶוַח:
פּוּתְיָא דְכִתְלָא דִמְחִיצְתָּא דִלְבָרָא חֲמֵשׁ אַמִין וַאֲתַר שְׁבִיק בֵּית מְחִיצַיָא דִלְגָיו:
וּבֵין לִשְׁכָּתָא פּוּתְיָא עַסְרִין אַמִין סְחוֹר סְחוֹר מַקִיף לְבֵיתָא סְחוֹר סְחוֹר:
וְתַרְעָא דִמְחִיצְתָּא לַאֲתַר שְׁבִיק תַּרְעָא חַד פְּתִיחַ לְאוֹרַח צִפּוּנָא וְתַרְעָא חַד פְּתִיחַ לְדָרוֹמָא וּפּוּתְיָא דַאֲתַר שְׁבִיק חֲמֵשׁ אַמִין סְחוֹר סְחוֹר:
וּבִנְיָנָא דִקֳדָם בְּצוּרְתָּא רוּחַ אוֹרַח מַעַרְבָא פּוּתְיָא שַׁבְעִין אַמִין וְכָתְלָא דְבִנְיָנָא חֲמֵשׁ אַמִין פּוּתְיָא סְחוֹר סְחוֹר וְאוּרְכֵּהּ תִּשְׁעִין אַמִין:
וּמְשַׁח יַת בֵּיתָא אוּרְכָּא מְאָה אַמִין וּבְצוּרְתָא וּבִנְיָנָא וְכָתְלַיָא אוּרְכָּא מְאָה אַמִין:
וּפוּתְיָא דִקֳדָם בֵּיתָא וּבְצוּרְתָא לְמַדִינְחָא מְאָה אַמִין:
וּמְשַׁח אוּרְכָּא דְבִנְיָנָא דִקֳדָם בְּצוּרְתָא דְלַאֲחוֹרָאָה וְזִיוְיָתָהָא מִכָּא וּמִכָּא מְאָה אַמִין וְהֵיכְלָא גַוָאָה וְאוּלַמֵי דַרְתָּא:
סִפַּיָא וְכַוִין סְתִּימָן וְעַתִּיקַיָא סְחוֹר סְחוֹר לִתְלָתֵיהוֹן לָקֳבֵיל סִפָּא דְחָפֵי נִיסְרִין דְאַרְזָא סְחוֹר סְחוֹר וְאַרְעָא עַד כַּוַיָא וְכַוִין סְתִימָן:
עַד לְעֵלָא מִן תַּרְעָא וּמִן בֵּיתָא גַוְאָה וּלְבָרָא וְעַל כָּל כּוֹתְלָא סְחוֹר סְחוֹר מִלְגָיו וּמִלְבָרָא מִשְׁחָן:
וּגְלִיף כְּרוּבִין וְכוֹתְרָן וְכוֹתַרְתָּא בֵּין כְּרוּבָא לִכְרוּבָא וּתְרֵין אַפִּין לִכְּרוּבָא:
וְאַפֵּי אֱנָשָׁא עֲבִיד לְכוֹתַרְתָּא מִכָּא וְאַפֵּי אַרְיָא עֲבִיד לְכוֹתַרְתָּא מִכָּא גְלִיף עַל כָּל בֵּיתָא סְחוֹר סְחוֹר:
מֵאַרְעָא עַד לְעֵלָא מִתַּרְעָא כְּרוּבִין וְכוּתְרָן גְלִיפָן וְכֵן לְכָתְלָא דְהֵיכָלָא:
הֵיכָלָא דְסִפּוֹהִי מְרֻבָּעִין וְאַפֵי בֵית כִּפּוּרֵי חֶזְוֵיהּ כְּחֵיזוּ יְקָרֵהּ:
לָקֳבֵל מַדְבְּחָא פְּתוֹרָא דְאָעָא תְּלַת אַמִין רוּמָה וְאוּרְכֵּהּ תַּרְתִּין אַמִין וְזִיוְיָן לֵיהּ וְאוּרְכֵּיהּ וְכָתְלוֹהִי דְאָעָא וּמַלֵיל עִמִי דֵין פָּתוֹרָא דִקֳדָם יְיָ:
וְתַרְתֵּין דָשִׁין לְהֵיכָלָא וּלְבֵית כִּפּוֹרֵי:
וְתַרְתֵּין דָשִׁין אִימְהָן לְדָשַׁיָא תַּרְתֵּין מַקְפָן דָשִׁין תַּרְתֵּין לְדָשָׁא חַד וְתַרְתֵּין דָשִׁין לְאוֹחֲרַנְתָּא:
וּגְלִיף עֲלֵיהוֹן עַל דָשֵׁי הֵיכָלָא כְּרוּבִין וְכוֹתְרָן כְּמָא דִגְלִיף עַל כּוֹתְלַיָא וּסְקִיפָתָא דְאָעָא עַל אַפֵּי אוּלַמָא מִלְבָרָא:
וְכַוִין סְתִימָן וְכוֹתַרְתָּא מִכָּא וּמִכָּא עַל כִּתְפֵי אוּלַמָא וּמְחִיצַת בֵּיתָא וּסְקוּפְתָא:
