וּנְחַת שִׁמְשׁוֹן לְתִמְנָת וַחֲזָא אִתְּתָא בְּתִמְנָת מִבְנַת פְּלִשְׁתָּאֵי:
וּסְלִיק וְחַוִי לַאֲבוּהִי וּלְאִמֵיהּ וַאֲמַר אִתְּתָא חֲזֵיתִי בְּתִמְנָת מִבְּנַת פְּלִשְׁתָּאֵי וּכְעַן סָבוּ יָתָהּ לִי לְאִתּוּ:
וַאֲמַרוּ לֵיהּ אֲבוּהִי וְאִמֵיהּ הָא לֵית בִּבְנַת אַחָךְ וּבְכָל עַמִי אִתְּתָא דְאַתְּ אָזֵיל לְמִיסַב אִתְּתָא מִפְּלִשְׁתָּאֵי עֲרֵלַיָא וַאֲמַר שִׁמְשׁוֹן לַאֲבוּהִי יָתָהּ סַב לִי אֲרֵי הִיא כַּשְׁרַת בְּעֵינָי:
וַאֲבוּהִי וְאִמֵיהּ לָא יָדְעִין אֲרֵי מִן קֳדָם יְיָ הוּא אֲרֵי תוּסְקָפָא הוּא בָעֵי לְאִתְגְרָאָה בִּפְלִשְׁתָּאֵי וּבְעִידָנָא הַהִיא פְּלִשְׁתָּאֵי שַׁלְטִין בְּיִשְׂרָאֵל:
וּנְחַת שִׁמְשׁוֹן וַאֲבוּהִי וְאִמֵיהּ לְתִמְנַת וַאֲתוֹ עַד כַּרְמֵי תִמְנָת וְהָא אַרְיָא בַּר אַרְיְוָן נָהִים לָקֳדָמוּתֵיהּ:
וּשְׁרַת עֲלוֹהִי רוּחַ גְבוּרָא מִן קֳדָם יְיָ וְהַדְמֵיהּ כְּמָא דִמְהַדְמִין גַדְיָא וּמִדַעַם לֵית בִּידֵיהּ וְלָא חַוִי לַאֲבוּהִי וּלְאִמֵיהּ יַת דַעֲבָד:
וּנְחַת וּשְׁאִיל בְּאִתְּתָא וְכַשְׁרַת בְּעֵינֵי שִׁמְשׁוֹן:
וְתַב לִזְמַן יוֹמִין לְמִסְבָּהּ וְסָר לְמֶחֱזֵי יַת פִּגְרָא דְאַרְיָא וְהָא קִינָא דְדַבַּרְיָתָא בְּפִגְרָא דְאַרְיָא וּבֵיהּ דוּבְשָׁא:
וְנַסְחֵיהּ בִּידֵיהּ וַאֲזַל מֵיזַל וְאָכִיל וַאֲזַל לְוָת אָבוּהִי וּלְוַת אִמֵיהּ וִיהַב לְהוֹן וַאֲכַלוּ וְלָא חֲוִי לְהוֹן אֲרֵי מִפִּגְרָא דְאַרְיָא נְסִיחַ דוּבְשָׁא:
וּנְחַת אֲבוּהִי עַל עֵיסַק אִתְּתָא וַעֲבַד תַּמָן שִׁמְשׁוֹן מִשְׁתַּיָא אֲרֵי כֵּן עָבְדִין עוּלֵמַיָא:
וַהֲוָה כַּד חֲזוֹ יָתֵיהּ וְדַבָּרוּ תְּלָתִין חֲבֵרִין וַהֲווֹ עִמֵיהּ:
וַאֲמַר לְהוֹן שִׁמְשׁוֹן אֵיחוּד כְּעַן לְכוֹן חוּדִיתָא אִם חַוָאָה תְחַווּן יָתָהּ לִי שַׁבְעַת יוֹמֵי מִשְׁתַּיָא וְתַשְׁכְּחוּן וְאֶתֵּן לְכוֹן תְּלָתִין פִּלְדָסִין וּתְלָתִין אִצְטָלַוָן דִלְבוּשָׁא:
וְאִם לָא תוּכְלוּן לְחַוָאָה לִי וְתִתְּנוּן אַתּוּן לִי תְּלָתִין פִּלְדָסִין וּתְלָתִין אִצְטָלַוָן דִלְבוּשָׁא וַאֲמַרוּ לֵיהּ חוּד חִידָתָךְ וְנַשְׁמְעִינָהּ:
וַאֲמַר לְהוֹן מֵאָכְלָא נְפַק מֵיכְלָא וּמִתַּקִיפָא נְפַק חוּלְיָא וְלָא יְכִילוּ לְחַוָאָה חוּדִיתָא תְּלָתָא יוֹמִין:
וַהֲוָה בְּיוֹמָא שְׁבִיעָאָה וַאֲמַרוּ לְאִתַּת שִׁמְשׁוֹן שַׁדִילִי יַת בַּעֲלִיךְ וִיחַוֵי לָנָא יַת חוּדִיתָא דִלְמָא נוֹקִיד יָתִיךְ וְיַת בֵּית אָבוּךְ בְּנוּרָא הַלְמַסְכְּנוּתָנָא קְרֵיתוּן יָתָנָא הַלְכָא:
וּבְכַת אִתַּת שִׁמְשׁוֹן עֲלוֹהִי וַאֲמַרַת לְחוֹד שְׂנֵאתַנִי וְלָא רְחַמְתָּנִי חוּדְתָּא חַדְתָּא לִשבְנֵי עַמִי וְלִי לָא חֲוֵיתָא וַאֲמַר לָהּ הָא לְאַבָּא וּלְאִמָא לָא חֲוֵיתִי וְלִיךְ אֵכְדֵין אֲחַוֵי:
וּבְכַת עֲלוֹהִי שַׁבְעָא יוֹמַיָא דַהֲוָה לְהוֹן מִשְׁתַּיָא וַהֲוָה בְּיוֹמָא שְׁבִיעָאָה וְחַוִי לָהּ אֲרֵי דְחַקְתֵּיהּ וְחַוִיאַת חוּדִיתָא לִבְנֵי עַמָהּ:
וַאֲמַרוּ לֵיהּ אֱנָשֵׁי קַרְתָּא בְּיוֹמָא שְׁבִיעָאָה עַד לָא מֵעַל שִׁמְשָׁא מַה חַלֵי מִדוּבְשָׁא וּמַה תַּקִיף מֵאַרְיָא וַאֲמַר לְהוֹן אִילוּלָפוֹן בְּדַקְתּוּן בְּאִתְּתִי לָא אַשְׁכַּחְתּוּן חוּדָתִי:
וּשְׁרַת עֲלוֹהִי רוּחַ גְבוּרָא מִן קֳדָם יְיָ וּנְחַת לְאַשְׁקְלוֹן וּקְטַל מִנְהוֹן תְּלָתִין גַבְרָא וּנְסִיב יַת זְרָזֵהוֹן וִיהַב אִצְטְלָוַיָא לִדְחַוִיאוּ חוּדִיתָא וּתְקֵיף רוּגְזֵיהּ וּסְלִיק לְבֵית אָבוּהִי:
וַהֲוַת אִתַּת שִׁמְשׁוֹן לִדְחַבְרֵיהּ דַהֲוָה שׁוֹשְׁבִינֵיהּ:
