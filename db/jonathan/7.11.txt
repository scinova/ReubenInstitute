וְיִפְתָּח גִלְעָדָא הֲוָה גִבַּר חֵילָא וְהוּא בַּר אִתְּתָא פּוּנְדְקִיתָא וְאוֹלִיד גִלְעָד יַת יִפְתָּח:
וִילֵידַת אִתַּת גִלְעָד לֵיהּ בְּנִין וּדְבִיאוּ בְנֵי אִתְּתָא וְתָרִיכוּ יַת יִפְתָּח וַאֲמַרוּ לֵיהּ לָא תַחְסֵין בְּבֵית אֲבוּנָא אֲרֵי בַּר אִתְּתָא אוֹחֲרִי אַתְּ:
וַעֲרַק יִפְתַּח מִן קֳדָם אֲחוֹהִי וִיתֵיב בְּאַרְעָא טָבָא וְאִתְכְּנִישׁוּ לְוַת יִפְתַּח גַבְרִין סְרִיקִין וּנְפָקוּ עִמֵיהּ:
וַהֲוָה מִזְמַן יוֹמִין וְאַגִיחוּ קְרָבָא בְּנֵי עַמוֹן עִם יִשְׂרָאֵל:
וַהֲוָה כַּד אַגִיחוּ קְרָבָא בְּנֵי עַמוֹן עִם יִשְׂרָאֵל וַאֲזַלוּ סָבֵי גִלְעָד לְמִדְבַּר יַת יִפְתַּח מֵאֲרַע טָב:
וַאֲמַרוּ לְיִפְתַּח אִיתָא וּתְהֵי לָנָא לְרַבָּא וּנְגִיחַ קְרָבָא בִּבְנֵי עַמוֹן:
וַאֲמַר יִפְתָּח לְסָבֵי גִלְעָד הֲלָא אַתּוּן סְנֵיתוּן יָתִי וְתָרִיכְתּוּנִי מִבֵּית אַבָּא וּמָא דֵין אֲתֵיתוּן לְוָתִי כְּעַן כַּד עָקַת לְכוֹן:
וַאֲמַרוּ סָבֵי גִלְעָד לְיִפְתָּח בְּכֵן כְּעַן תַּבְנָא לְוָתָךְ וּתְהַךְ עִמָנָא וְתַגִיחַ קְרָבָא בִּבְנֵי עַמוֹן וּתְהֵי לָנָא לְרֵישׁ לְכֹל יַתְבֵי גִלְעָד:
וַאֲמַר יִפְתָּח לְסָבֵי גִלְעָד אִם מְתִיבִין אַתּוּן יָתִי לְאַגָחָא קְרָבָא בִּבְנֵי עַמוֹן וְיִמְסַר יְיָ יַתְהוֹן קֳדָמַי אֲנָא אֱהֶוֵי לְכוֹן לְרֵישׁ:
וַאֲמַרוּ סָבֵי גִלְעָד לְיִפְתָּח מֵימְרָא דַייָ יְהֵי סָהִיד בֵּינָנָא אִם לָא כְּפִתְגָמָךְ כֵּן נַעְבֵּד:
וַאֲזַל יִפְתָּח עִם סָבֵי גִלְעָד וּמַנִיאוּ עַמָא יָתֵיהּ עֲלֵיהוֹן לְרֵישׁ וּלְרַבָּא וּסְדַר יִפְתָּח יַת כָּל פִּתְגָמוֹהִי קֳדָם יְיָ בְּמִצְפַּיָא:
וּשְׁלַח יִפְתָּח אִזְגַדִין לְוַת מַלְכָּא דִבְנֵי עַמוֹן לְמֵימָר מַה לִי וְלָךְ אֲרֵי אֲתֵיתָא לְוָתִי לְאַגָחָא קְרָבָא בְּאַרְעִי:
וַאֲמַר מַלְכָּא דִבְנֵי עַמוֹן לְאִזְגַדֵי יִפְתָּח אֲרֵי נְסִיב יִשְׂרָאֵל יַת אַרְעִי בְּמִסְקֵיהּ מִמִצְרַיִם מֵאַרְנוֹנָא וְעַד יוּבְקָא וְעַד יַרְדְנָא וּכְעַן אֲתֵיב יַתְהֶן בִּשְׁלָם:
וְאוֹסִיף עוֹד יִפְתָּח וּשְׁלַח אִזְגַדִין לְוַת מַלְכָא דִבְנֵי עַמוֹן:
וַאֲמַר לֵיהּ כִּדְנַן אֲמַר יִפְתָּח לָא נִסִיב יִשְׂרָאֵל יַת אַרְעָא דְמוֹאָב וְיַת אַרְעָא בְּנֵי עַמוֹן:
אֲרֵי בְּמִסַקְהוֹן מִמִצְרָיִם וַאֲזַל יִשְׂרָאֵל בְּמַדְבְּרָא עַד יַמָא דְסוּף וַאֲתָא לִרְקָם:
וּשְׁלַח יִשְׂרָאֵל אִזְגַדִין לְוַת מַלְכָּא דֶאֱדוֹם לְמֵימָר אֶעְבָּר כְּעַן בְּאַרְעָךְ וְלָא שְׁמַע מַלְכָּא דְאֱדוֹם וְאַף לְוַת מַלְכָּא דְמוֹאָב שְׁלַח וְלָא קַבֵּל וִיתֵיב יִשְׂרָאֵל בִּרְקָם:
וַאֲזַל בְּמַדְבְּרָא וְאַקֵיף יַת אֲרַע אֱדוֹם וְיַת אֲרַע מוֹאָב וַאֲתוֹ מִמַדְנַח שִׁמְשָׁא לְאַרְעָא דְמוֹאָב וּשְׁרוֹ בְּעִיבְרָא דְאַרְנוֹן וְלָא עָלוּ בִּתְחוּם מוֹאָב אֲרֵי אַרְנוֹן תְּחוּם מוֹאָב:
וּשְׁלַח יִשְׂרָאֵל אִזְגַדִין לְוַת סִיחוֹן מַלְכָּא דֶאֱמוֹרָאָה מַלְכָּא דְחֶשְׁבּוֹן וַאֲמַר לֵיהּ יִשְׂרָאֵל נֶעְבַּר כְּעַן בְּאַרְעָךְ עַד אַתְרִי:
וְלָא הֵימִין סִיחוֹן יַת יִשְׂרָאֵל לְמֶעְבַּר בִּתְחוּמֵיהּ וּכְנַשׁ סִיחוֹן יַת כָּל עַמֵיהּ וּשְׁרוֹ בְּיָהְצָה וְאַגִיחַ קְרָבָא עִם יִשְׂרָאֵל:
וּמְסַר יְיָ אֱלָהָא דְיִשְׂרָאֵל יַת סִיחוֹן וְיַת כָּל עַמֵיהּ בִּידָא דְיִשְׂרָאֵל וּמְחוֹנוּן וִירֵית יִשְׂרָאֵל יַת כָּל אֲרַע אֱמוֹרָאָה יָתֵיב אַרְעָא הַהִיא:
וִירִיתוּ יַת כָּל תְּחוּם אֱמוֹרָאָה מֵאַרְנוֹנָא וְעַד יוּבְקָא וּמִן מַדְבְּרָא וְעַד יַרְדְנָא:
וּכְעַן יְיָ אֱלָהָא דְיִשְׂרָאֵל תָּרִיךְ יַת אֱמוֹרָאָה מִן קֳדָם עַמֵיהּ יִשְׂרָאֵל וְאַתְּ מְדַמֵי לְמֵירְתֵיהּ:
הֲלָא יַת דְיַחְסְנִנָךְ כְּמוֹשׁ טַעֲוָתָךְ יָתֵיהּ תַּחְסִין וְיַת כָּל דְתָרֵיךְ יְיָ אֱלָהָנָא מִן קֳדָמָנָא יָתֵיהּ נִירָת:
וּכְעַן הֲטַב טַב אַתְּ מִבָּלָק בַּר צִפּוֹר מַלְכָּא דְמוֹאָב הֲמֵידַן דַן עִם יִשְׂרָאֵל אִם קְרָבָא אַגִיחַ בְּהוֹן:
כַּד יָתֵיב יִשְׂרָאֵל בְּחֶשְׁבּוֹן וּבְכַפְרָנָהָא וּבְעַרְעוֹר וּבְכַפְרָנָהָא וּבְכָל קִרְוַיָא דְעַל תְּחוּמֵי אַרְנוֹן תְּלַת מְאָה שְׁנִין וּמָא דֵין לָא שְׁזַבְתּוּן בְּעִדָנָא הַהִיא:
וַאֲנָא לָא חָטֵית לָךְ וְאַתְּ עָבֵיד עִמִי בִּישָׁא לְאַגָחָא קְרָבָא בִי יְדִין יְיָ דַעֲבַד דִינָא יוֹמָא דֵין בֵּין בְּנֵי יִשְׂרָאֵל וּבֵין בְּנֵי עַמוֹן:
וְלָא קַבֵּיל מַלְכָּא דִבְנֵי עַמוֹן לְפִתְגָמֵי יִפְתָּח דִשְׁלַח לְוָתֵיהּ:
וּשְׁרַת עַל יִפְתָּח רוּחַ גְבוּרָא מִן קֳדָם יְיָ וַעֲבַר יַת גִלְעָד וְיַת מְנַשֶׁה וַעֲבַר יַת מִצְפַּיָא דְגִלְעָד וּמִמִצְפַּיָא דְגִלְעָד עֲבַר בְּנֵי עַמוֹן:
וְקַיֵם יִפְתָּח קְיָם קֳדָם יְיָ וַאֲמַר אִם מִמְסַר תִּמְסַר יַת בְּנֵי עַמוֹן בִּידִי:
וִיהֵי דְיִפּוֹק מֵפַק בַּר מִדָשֵׁי בֵיתִי לָקֳדָמוּתִי כַּד אֵתוּב בִּשְׁלָם מִבְּנֵי עַמוֹן וִיהֵי קֳדָם יְיָ וְאַסְקִינֵיהּ עֲלָתָא:
וַעֲבַר יִפְתָּח לְוַת בְּנֵי עַמוֹן לְאַגָחָא קְרָבָא בְהוֹן וּמְסָרִינוּן יְיָ בִּידֵיהּ:
וּמְחָנוּן מֵעֲרוֹעֵר עַד מַעֲלָנָא דְמִנִית עַשְׂרִין קִרְוִין וְעַד מֵישַׁר כְּרָמַיָא מַחֲתָא רַבְּתָא לַחֲדָא וְאִתְבָּרוּ בְּנֵי עַמוֹן מִן קֳדָם בְּנֵי יִשְׂרָאֵל:
וַאֲתָא יִפְתָּח לְמִצְפַּיָא לְבֵיתֵיהּ וְהָא בְרַתֵּיהּ נַפְקַת לָקֳדָמוּתֵיהּ בְּתֻפִין וּבְחִנְגִין וּלְחוֹד הִיא יְחִידָה לֵית לֵיהּ מִנֵיהּ בַּר אוֹ בְרַת:
וַהֲוָה כַּד חֲזֵי יָתָהּ וּבְזַע יַת לְבוּשׁוֹהִי וַאֲמַר וַוי בְּרַתִּי אַכְרָעָא אַכְרַעְתִּינִי וְאַתְּ הֲוֵית בְּעָכְרִי וַאֲנָא פַּתְחֵית פּוּמִי בִּנְדַר קֳדָם יְיָ וְלָא אֵיכוּל לְמֵיתָב:
וַאֲמָרַת לֵיהּ אַבָּא פְּתַחְתָּא יַת פּוּמָךְ בִּנְדַר קֳדָם יְיָ ָעֲבֵיד לִי כְמָא דִי נְפַק מִפּוּמָךְ בָּתַר דַעֲבַד לָךְ יְיָ פּוּרְעָנוּתָא מִבַּעֲלֵי דְבָבָךְ מִבְּנֵי עַמוֹן:
וַאֲמַרַת לַאֲבוּהָא יִתְעֲבֵיד לִי פִּתְגָמָא הָדֵין שְׁבוֹק מִנִי תְּרֵין יַרְחִין וְאֵהַךְ וְאִתְנְגִיד עַל טוּרַיָא וְאֶבְכֶּה עַל בְּתוּלַי אֲנָא וְחַבְרָתִי:
וַאֲמַר אֱזִילִי וּשְׁלַח יָתָהּ תְּרֵין יַרְחִין וַאֲזַלַת הִיא וְחַבְרָתָהָא וּבְכַת עַל בְּתוּלָהָא עַל טוּרַיָא:
וַהֲוָה מִסּוֹף תְּרֵין יַרְחִין וְתָבַת לְוַת אֲבוּהָא וַעֲבַד לָהּ יַת נִדְרֵיהּ דִי נְדַר וְהִיא לָא יָדְעַת גְבַר וַהֲוַת לִגְזֵירָא בְּיִשְׂרָאֵל: [בְּדִיל דְלָא לְאַסָקָא גְבַר יַת בְּרֵיהּ וְיַת בְּרַתֵּיהּ לַעֲלָתָא כְּמָא דַעֲבַד יִפְתַּח גִלְעָדָאָה וְלָא שָׁאֵיל לְפִינְחָס כַּהֲנָא וְאִילוּ שָׁאֵיל לְפִינְחָס כַּהֲנָא הֲוָה פָרִיק יָתָהּ בְּדָמִין:]
מִזְמַן לִזְמַן אַזְלַן בְּנַת יִשְׂרָאֵל לַאֲלָאָה לְבַת יִפְתָּח גִלְעָדָאָה אַרְבְּעָא יוֹמִין בְּשַׁתָּא:
