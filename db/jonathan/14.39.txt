וְאַתְּ בַר אָדָם אִתְנַבֵּי עַל גוֹג וְתֵימַר כִּדְנַן אֲמַר יְיָ אֱלֹהִים הָא אֲנָא שְׁלַח רוּגְזִי עֲלָךְ גוֹג רַב רֵישׁ מֶשֶׁךְ וְתֻבָל:
וַאֲשַׁדְלִינָךְ וְאַטְעִינָךְ וְאַסְקִינָךְ מִסְיָפֵי צִפּוּנָא וְאַיְתִינָךְ עַל טוּרֵי יִשְׂרָאֵל:
וְאֶרְמֵי קַשְׁתָּךְ מִיַד סְמָאלָךְ וְגִרְרָךְ מִיַד יְמִינָה אֲמַגֵיר:
עַל טוּרֵי אַרְעָא דְיִשְׂרָאֵל תִּתְרְמֵי נְבֶלְתַּךְ אַתְּ וְכָל מַשִׁרְיָתָךְ וְעַמְמִין דִי עִמָךְ לְעוֹפָא צִפָּר כָּל דִי פָרַח וּלְחַיַת בָּרָא מְסַרְתָּךְ לְאִשְׁתֵּיצָאָה:
עַל אַפֵּי בְּרָא תִּתְרְמֵי נְבֵילְתָּךְ אֲרֵי אֲנָא גְזָרֵית בְּמֵימְרִי אֲמַר יְיָ אֱלֹהִים:
וְאַדְלֵיק אֶשְׁתָּא בְמָגוֹג וּבְיָתְבֵי נַגְוָתָא דְשָׁרָן לְרוֹחֲצָן וְיֵדְעוּן אֲרֵי אֲנָא יְיָ:
וְיַת שְׁמָא דְקוּדְשִׁי אֲגַלֵי בְּגוֹ עַמִי יִשְׂרָאֵל וְלָא אָחֵיל יַת שְׁמָא דְקוּדְשִׁי עוֹד וְיֵדְעוּן עַמְמַיָא אֲרֵי אֲנָא יְיָ קַדִישָׁא אַשְׁרֵיתִי שְׁכִנְתִּי בְּיִשְׂרָאֵל:
הָא אַתְיָא וְתִתְקַיֵם אֲמַר יְיָ אֱלֹהִים הוּא יוֹמָא דִי גְזָרֵית בְּמֵימְרִי:
וְיִפְקוּן יַתְבֵי קִרְוֵי אַרְעָא דְיִשְׂרָאֵל וִיהוֹן שַׁגְרִין וּמִסְתּוֹבָרִין בְּמָנֵי זְיָנֵי בַעֲגָלִין וּתְרֵיסִין בְּקַשְׁתָּן וּבְגִירְרִין וּבְעָאֵי מוּרְנִין וְרוֹמְחִין וְיַדְלְקוּן בְּהוֹן נוּרָא שְׁבַע שְׁנִין:
וְלָא יִסְבוּן אָעִין מִן חַקְלָא וְלָא יְקַצוּן מִן חוּרְשָׁא אֲרֵי בְמִנֵי זְיָנָא יִדְלְקוּן נוּרָא וְיִבְזוּן יַת בְּזָזֵיהוֹן וְיַעְדוּ יַת עֲדֵיהוֹן אֲמַר יְיָ אֱלֹהִים:
וִיהֵי בְּעִדָנָא הַהוּא אֶתֵּן לְגוֹג אֲתַר כָּשַׁר לְבֵית קְבוּרָא בְּיִשְׂרָאֵל בְּחֵילְתָּא מְגִיזָתָא מַדְנַח יַם גִינוֹסַר וְסָמְכָא הִיא לִתְּרֵין טוּרַיָא וְיִקְבְּרוּן תַּמָן יַת גוֹג וְיַת כָּל אִתְרְגוּשְׁתֵּיהּ וְיִקְרוֹן חֵילַת אִתְרְגוּשַׁת גוֹג:
וְיִקְבְּרוּנוּן בֵּית יִשְׂרָאֵל בְּדִיל לְדַכָּאָה יַת אַרְעָא שִׁבְעָה יַרְחִין:
וְיִקְבְּרוּן כָּל עַמָא דְאַרְעָא וִיהֵי לְהוֹן לְשׁוּם יוֹם אִתְיַקְרוּתִי אֲמַר יְיָ אֱלֹהִים:
וְגַבְרִין יְמַנוּן תְּדִירָא לְמֶהֱוֵי עָדָן בְּאַרְעָא מְקַבְּרִין עַם דְעָדָן יַת דִי אִשְׁתְּאָרוּ עַל אַפֵּי אַרְעָא לְדַכָּיוּתָהּ מִסוֹף שַׁבְעָא יַרְחִין יִשְׁרוּן לְמִבְדָק:
וְעָדֵי אוֹרְחָא דְעָדָן בְּאַרְעָא דִי אִם יֶחֱזֵי גַרְמָא דֶאֱנָשָׁא וְיִבְנֵי לְוָתֵיהּ צִיוּנָא עַד דְיִקְבְּרוּן יָתֵיהּ דִי מְקַבְּרִין בְּחֵילַת אִתְרְגוּשַׁת גוֹג:
וְאַף לְתַמָן יִתְרְמוּן קְטִילֵי רוֹמָא קַרְתָּא דְסַגִיאִין אִתְרְגוּשְׁתָּהָא גוֹג וִידַכּוּן אַרְעָא:
וְאַתְּ בַּר אָדָם כִּדְנַן אֲמַר יְיָ אֱלֹהִים אֱמַר לְעוֹפָא צִפַּר כָּל דִי פְרַח וּלְכָל חַיַת בָּרָא אִתְכְּנִישׁוּ וֶאֱתוֹ אִתְקְרָבוּ מִסְחוֹר סְחוֹר עַל דִי קְטִילֵי דִי אֲנָא קָטֵיל לְכוֹן קְטוֹל סַגִי עַל טוּרֵי יִשְׂרָאֵל וְתֵיכְלוּן בִּסְרָא וְתִשְׁתּוּן דְמָא:
בְּסַר גִבָּרִין תֵּיכְלוּן וְדַם רַבְרְבֵי אַרְעָא תִּשְׁתּוּן מַלְכִין שִׁלְטוֹנִין וְטוּרְנִין גִבָּרִין עַתִּירֵי נִכְסַיָא כּוּלְהוֹן:
וְתֵיכְלוּן תַּרְבָּא לְמִסְבַּע וְתִשְׁתּוּן דְמָא מִבְּסַר קְטִילִין דִי אֲנָא קָטֵיל לְכוֹן:
וְתִסְבְּעוּן עַל טוּרֵי יִשְׂרָאֵל סוּסְוָן וּרְתִיכִין גִבָּרִין וְכָל גַבְרֵי עָבְדֵי קְרָבָא אֲמַר יְיָ אֱלֹהִים:
וְאֶתֵּן יַת יְקָרִי בְּעַמְמַיָא וְיֶחֱזוּן כָּל עַמְמַיָא יַת פּוּרְעֲנוּתִי דַעֲבָדַת וְיַת גְבוּרְתִּי דְשַׁוֵיתִי בְהוֹן:
וְיֵדְעוּן בֵּית יִשְׂרָאֵל אֲרֵי אֲנָא יְיָ אֱלָהָהוֹן מִן יוֹמָא הַהוּא וּלְהָלָא:
וְיֵדְעוּן עַמְמַיָא אֲרֵי בְּחוֹבֵיהוֹן גְלוֹ בֵית יִשְׂרָאֵל עַל דְשַׁקָרוּ בְמֵימְרִי וְסַלִיקֵית שְׁכִנְתִּי מִנְהוֹן וּמְסַרְתִּינוּן בְּיַד סָנְאֵיהוֹן וְאִתְקַטָלוּ בְחַרְבָּא כּוּלְהוֹן:
כְּסוֹאֲבָתְהוֹן וּכְמֵירְדֵיהוֹן אִתְפְּרָעֵית מִנְהוֹן וְסַלְקֵית שְׁכִנְתִּי מִנְהוֹן:
בְּכֵן כִּדְנַן אֲמַר יְיָ אֱלֹהִים כְּעַן אֲתֵיב יַת גָלוּת יַעֲקֹב וְאֶרְחַם כָּל בֵּית יִשְׂרָאֵל וְאֶתְפְּרַע פּוּרְעֲנוּת גְבוּרְתִּי מִן עַמְמַיָא עַל דְאָחִילוּ יַת שְׁמָא דְקוּדְשִׁי:
וִיקַבְּלוּן יַת אִתְכְּנָעוּתְהוֹן וְיַת כָּל שִׁקְרֵיהוֹן דְשַׁקָרוּ בְּמֵימְרִי בְּמֵיתַבְהוֹן עַל אַרְעֲהוֹן לְרוֹחֲצָן וְלֵית דְמָנִיד:
כַּד אֲתִיב יַתְהוֹן מִבֵּינֵי עַמְמַיָא וְאַכְנֵישׁ יַתְהוֹן מִן מְדִינְתָא בַּעֲלֵי דְבָבֵיכוֹן וְאִתְקַדַשׁ בְּהוֹן לְעֵינֵי עַמְמִין סַגִיאִין:
וְיֵדְעוּן אֲרֵי אֲנָא יְיָ אֱלָהָהוֹן וְעַל דְחָבוּ קֳדָמַי אַגְלֵיתִי יַתְהוֹן לְבֵינֵי עַמְמַיָא וּכְדוֹ דְתָבוּ כְּנִשְׁתִּינוּן לְאַרְעֲהוֹן וְלָא אַשְׁאַר עוֹד מִנְהוֹן תַּמָן:
וְלָא אֲסַלֵק עוֹד שְׁכִנְתִּי מִנְהוֹן דִשְׁפִיכֵית יַת רוּחַ קוּדְשִׁי עַל בֵּית יִשְׂרָאֵל אֲמַר יְיָ אֱלֹהִים:
