כִּדִנַן אֲמַר יְיָ שְׁמַיָא כוּרְסֵי יְקָרֵי וְאַרְעָא כְּבַשׁ קֳדָמַי אֵידֵין בֵּיתָא דְתִבְנוֹן קֳדָמַי וְאֵידֵין אֲתַר בֵּית אַשְׁרָיוּת שְׁכִנְתִּי:
וְיַת כָּל אִלֵין גְבוּרְתִי עֲבָדַת וַהֲלָא הֲוָאָה כָל אִלֵין אֲמַר יְיָ וּבְדֵין רַעֲוָא קֳדָמַי לְאִסְתַּכָּלָא בֵּיהּ בִּדְעַנְוְתָן וּמַכִּיךְ רוּחַ וּמִשְׁתְּוֵי לִקֳבֵל פִתְגָמִי:
נָכֵיס תּוֹרָא כְּקָטִיל גְבַר דָבַח אִימַר כְּנָקֵיף כְּלֵב מַסְקֵי קוּרְבַּן דַם חֲזִירָא קוּרְבַּן מַתְּנַתְהוֹן מַתְּנַת אוֹנָס אַף אִינוּן אִתְרְעִיאוּ בְּאוֹרְחַתְהוֹן וּבְשִׁיקוּצֵיהוֹן נַפְשֵׁהוֹן אִתְרְעִיאַת:
אַף אֲנָא אֶצְבֵּי בְתַבְרֵהוֹן וּמִמַה דְיָצְפָן לָא יִשְׁתֵּיזְבוּן חֲלַף דִשְׁלָחֵית נְבִיֵי וְלָא תָבוּ אִתְנַבִיאוּ וְלָא קַבִּילוּ וַעֲבָדוּ דְבִישׁ קֳדָמַי וּבִדְלָא צְבִינָא אִתְרְעִיאוּ:
קַבִּילוּ פִּתְגָמָא דַייָ צַדִיקַיָא דְמִשְׁתְּוָן לִקֳבֵיל פִּתְגָמֵי רְעוּתֵיהּ אָמְרִין אֲחֵיכוֹן סַנְאֵיכוֹן מֵרַחֲקֵיכוֹן בְּדִיל שְׁמִי יִסְגֵי יְקָרָא דַייָ וְנֶחֱזֵי בְחֶדְוַתְכוֹן וְאִינוּן יִבְהֲתוּן:
קַל אִתְרְגוֹשָׁא מֵקַרְתָּא דִירוּשְׁלֵם קָלָא מַהֵיכְלָא קַל מֵימְרָא דַייָ דִמְשַׁלֵים גַמְלָא לְבַעֲלֵי דְבָבוֹהִי:
עַד לָא מֵיתֵי עָקָא לָהּ תִּתְפְּרִיק עַד לָא יֵיתֵי לָהּ זִיעַ בְּחֶבְלִין עַל יְלִידָה יִתְגְלֵי מַלְכָּהּ:
מַן שְׁמַע כַּחֲדָא מַן חֲזָא כְּאִלֵין הָאֶפְשַׁר דְתִתְעֲבֵיד אַרְעָא בְּיוֹמָא חָד אִם יִתְבְּרֵי עַמָא זְמַן חֲדָא אֲרֵי עֲתִידָא דְתִתְנְחֵים צִיוֹן וְתִתְמְלֵי מֵעַם שְׁבִי גַלְוָתָהָא:
אֲנָא אֱלָהָא בְּרִית עַלְמָא מִבְּרֵאשִׁית אֲמַר יְיָ אֲנָא בְרֵית כָל אֱנָשָׁא אֲנָא בַדְרֵית יַתְהוֹן לְבֵינֵי עַמְמַיָא אַף אֲנָא עֲתִיד לְכַנָשָׁא גַלְוָתִיךְ אֲמַר אֱלָהָיִךְ:
חֲדוֹ בִירוּשְׁלֵם וּבוּעוּ בָהּ כָּל רָחֲמָהָא דוּצוּ עִמָהּ דַיִץ כָּל דַהֲווֹ מִתְאַבְּלִין עֲלָה:
בְּדִיל דְתִתְפַּנְקוּן וְתִשְׂבְּעוּן מִבִּזַת תַּנְחוּמָתָהָא בְּדִיל דְתִשְׁתּוּן וְתִתְרַווֹן מַחֲמַד יְקָרָהּ:
אֲרֵי כִדְנַן אֲמַר יְיָ הָא אֲנָא מַיְתִי לָהּ כִּשְׁפַע נְהַר פְּרָת שְׁלַם וְכִנְחַל מַגְבַּר יְקַר עַמְמַיָא וְתִתְפַּנְקוּן עַל גִסְסִין תִּתְנַטְלוּן וְעַל רְכוּבִין תִּתְרַבּוּן:
כִּגְבַר דְאִמֵיהּ מְנַחֲמָא לֵיהּ כֵּן מֵימְרִי יְנַחֵים יַתְכוֹן וּבִירוּשְׁלֵם תִּתְנַחֲמוּן:
וְתֶחֱזוּן וְיֶחְדֵי לִבְּכוֹן וּגְוִיַתְכוֹן כְּדִתְאִין יִזְהֲרוּן וְתִתְגְלֵי גְבוּרְתָּא דַייָ לְאוֹטָבָא לְעַבְדוֹהִי צַדִיקַיָא וְיֵתֵי לְוָט לְבַעֲלֵי דְבָבוֹהִי:
אֲרֵי הָא יְיָ בְאֶשְׁתָּא מִתְגְלֵי וּכְעַלְעוּלִין רְתִיכוֹהִי לַאֲתָבָא בִּתְקוֹף רוּגְזֵיהּ וּמְזוֹפִיתֵיהּ בְּשַׁלְהוֹבִית אֶשְׁתָּא:
אֲרֵי בְּאֶשְׁתָּא עֲתִיד יְיָ לְמִידַן וּבְחַרְבֵּיהּ יַת כָּל בִּשְׂרָא וְסַגִיאִין קְטִילַיָא קֳדָם יְיָ:
דְמִזְדַמְנִין וְדִמְדַכָּן לְגִנָן דְטַעֲוָתָא סִיעָא בָּתַר סִיעָא אָכְלֵי בְּשַׂר חֲזִירָא וְשִׁקְצָא וְעַכְבְּרָא כַּחֲדָא יְסוּפוּן אֲמַר יְיָ:
וּקֳדָמַי גְלַן עוֹבָדֵיהוֹן וְעֶשְׁתּוֹנֵיהוֹן עֲתִידָנָא לְכַנָשָׁא יַת כָּל עַמְמַיָא אוּמַיָא וְלִישָׁנַיָא וְיֵתוּן וְיֶחְזוּן יַת יְקָרִי:
וַאֲשַׁוֵי בְהוֹן אָתָא וַאֲשַׁלַח מִנְהוֹן מְשֵׁיזְבִין לְבֵינֵי עַמְמַיָא לִמְדִינַת יַמָא פּוּלָאֵי וְלוּדָאֵי דְנַגְדִין וּמָחָן בְּקַשְׁתָּא לִמְדִינַת תּוּבַל וְיָוָן נַגְוָתָא רְחִיקָן דְלָא שְׁמָעוּ יַת שְׁמַע גְבוּרְתִי וְלָא חֲזוֹ יַת יְקָרִי וְיַחְווּן יַת יְקָרִי בְּעַמְמֵיָא:
וְיֵיתוּן יַת כָּל אֲחֵיכוֹן מִכָּל עַמְמַיָא קוּרְבָּנָא קֳדָם יְיָ בְסוּסְוָן וּבִרְתִיכִין וּבִרְחֵילִין וּבְכִדְנְוָן וּבְכִרְכְּרָן וּבְתוּשְׁבְּחָן עַל טוּרָא דְקוּדְשָׁא אֲמַר יְיָ כְּמָא דְיֵיתוּן בְּנֵי יִשְׂרָאֵל יַת קוּרְבָּנָא בְּמָנָא דְכֵי לְבֵית מַקְדְשָׁא דַיָי:
וְאַף מִנְהוֹן אֲקָרֵב לְמֶהֱוֵי כַהֲנַיָא וְלֵוָאֵי אֲמַר יְיָ:
אֲרֵי כְּמָא דִשְׁמַיָא חַדְתִּין וְאַרְעָא חַדְתָּא דַאֲנָא עָבִיד קַיְמִין קֳדָמַי אֲמַר יְיָ כֵּן יִתְקַיֵם זַרְעֲכוֹן וְשׁוּמְכוֹן:
וִיהֵי בִּזְמַן רֵישׁ יְרַח בִּירַח וּבִזְמַן שַׁבָּא בְּשַׁבָּא יֵיתוּן כָּל בִּשְׂרָא לְמִסְגַד קֳדָמַי אֲמַר יְיָ:
וְיִפְקוּן וְיֶחְזוּן בְּפִגְרֵי גַבְרַיָא חַיָבַיָא דִמְרָדוּ בְמֵימְרִי אֲרֵי נִשְׁמַתְהוֹן לָא יְמוּתוּן וְאֶשְׁתְּהוֹן לָא תִטְפֵי וִיהוֹן מִתְדָנִין רַשִׁיעַיָא בְּגֵיהִנָם עַד דְיֵימְרוּן עֲלֵיהוֹן צַדִיקַיָא מִיסַת חֲזֵינָא:
