וַאֲמַר יְיָ לִי סַב לָךְ לוּחַ רַב וּכְתוֹב עֲלוֹהִי כְּתַב מְפָרַשׁ מוֹחִי לְמִבַּז בִּזָא וּלְמֶעְדֵי עֲדָאָה:
וְאַסְהֵיד קֳדָמַי סְהִידִין מְהֵמְנָן יַת לְוָטַיָא דַאֲמָרִית לְאַיְתָאָה בִּנְבוּאַת אוּרִיָה כַהֲנָא וְהָא אֲתוֹ אַף כֵּן כָּל נְחָמָתָא דַאֲמָרֵית לְאַיְתָאָה בִּנְבוּאַת זְכַרְיָה בַּר יְבָרֶכְיָה אֲנָא עֲתִיד לְאַיְתָאָה:
וַעֲלֵית לְוַת נְבִיאֲתָא וְעַדִיאַת וִילֵדַת בַּר וַאֲמַר יְיָ לִי קְרִי שְׁמֵיהּ מוֹחִי לְמִבַּז בִּזָא וּלְמֶעְדֵי עֲדָאָה:
אֲרֵי עַד לָא יְדַע עוּלֵימָא מִקְרֵי אַבָּא וְאִמָא יִשְׁבֵי יַת נִכְסֵי דַמֶשֶׂק וְיַת עֲדָאֵי שֹׁמְרוֹן קֳדָם מַלְכָּא דְאַתּוּר:
וְאוֹסִיף מֵימְרָא דַייָ לְמַלָלָא עִמִי עוֹד לְמֵימָר:
חֲלַף דְקָץ עַמָא הָדֵין בְּמַלְכוּתָא דְבֵית דָוִד דִמְדַבֵּר לְהוֹן בְּנִיחַ כְּמֵי שִׁלוּחָא דְנַגְדִין בְּנִיחַ וְאִתְרְעִיאוּ בִרְצִין וּבַר רְמַלְיָה:
בְּכֵן הָא יְיָ מַיְתֵי וּמַסִיק עֲלֵיהוֹן יַת מַשִׁרְיַת עַמְמַיָא דְסַגִיאִין כְּמֵי נַהֲרָא תַּקִיפַיָא וַחֲסִינַיָא יַת מַלְכָּא דְאַתּוּר וְיַת כָּל מַשִׁרְיָתֵיהּ וְיִסַק עַל כָּל פַּצִידוֹהִי וִיהַךְ עַל כָּל כֵּיפוֹהִי:
וְיַעְדֵי בְאַרְעָא דְבֵית יְהוּדָה כְּנַחַל מְגַבֵּר עַד יְרוּשְׁלֵם יִמְטֵי וִיהֵי עַם מַשִׁרְיָתֵיהּ מָלֵי פְתָאֵי אַרְעָךְ עִמָנוּ אֵל:
אִתְחַבָּרוּ עַמְמַיָא וְאִתְבָּרוּ וְאָצִיתוּ כֹּל דִי בִסְיָפֵי אַרְעָא אַתְקִיפוּ וְאִתְבָּרוּ אַתְקִיפוּ וְאִתְבָּרוּ:
אִתְמְלִיכוּ מְלַךְ וְיַעֲדֵי מַלִלוּ פִּתְגָמָא וְלָא יִתְקַיֵם אֲרֵי בְסַעֲדָנָא אֱלָהָא:
אֲרֵי כִדְנַן אֲמַר יְיָ לִי בְּמִתְקַף נְבוּאָתָא וְאֶלְפֵנִי מִלְמֵהַךְ בְּאוֹרַח עַמָא הָדֵין לְמֵימָר:
לָא תֵימְרוּן מְרוֹד לְכָל דְיֵמַר עַמָא הָדֵין מְרוֹד וְלִדְחַלְתֵּיהּ לָא תִדְחֲלוּן וְעַל תּוּקְפֵיהּ לָא תֵמְרוּן תְּקוֹף:
יַת יְיָ צְבָאוֹת יָתֵיהּ תֵּימְרוּן קַדִישׁ וְהוּא דְחַלְתְּכוֹן וְהוּא תוּקְפֵכוֹן:
וְאִם לָא תְקַבְּלוּן וִיהֵי מֵימְרֵיהּ בְכוֹן לְפוּרְעָן וּלְאֶבֶן מָחֵי וּלְכֵיף מְתַקֵל לִתְרֵין בָּתֵּי רַבְרְבֵי יִשְׂרָאֵל לִתְבַר וּלְתַקָלָא עַל דְאִתְפְּלִיגוּ עַל דְבֵית יְהוּדָה דְיָתְבִין בִּירוּשְׁלֵם:
וְיִתְקְלוּן בְּהוֹן סַגִיאִין וְיִפְּלוּן וְיִתַּבְּרוּן וְיִתַּצְדוּן וְיִתְאַחֲדוּן:
נְבִיָא טַר סַהֲדוּתָא לָא תַסְהִיד בְּהוֹן אֲרֵי לָא מְקַבְּלִין חֲתוֹם וְטַמַר אוֹרַיְתָא לָא צָבָן דְיֵלְפוּן בָהּ:
אֲמַר נְבִיָא עַל דָא צַלֵיתִי קֳדָם יְיָ דַאֲמַר לְסַלְקָא שְׁכִנְתֵּיהּ מִדְבֵית יַעֲקֹב וּבְעֵיתִי מִן קֳדָמוֹהִי:
הָא עַד דַאֲנָא קַיֵם וְעוּלֵימַיָא דִיהַב לִי יְיָ יִתְקַיְמוּן בָּנָא אָתִין וּמוֹפְתִין דַאֲמִירָן לְמֵיתֵי עַל יִשְׂרָאֵל דְאִם יֶחֱזוּן וִיתוּבוּן תִּבָטֵיל גְזֵירְתָּא דִי גְזָרָא עֲלֵיהוֹן דְיִגְלוּן מִלְאִתְחֲזָאָה קֳדָם יְיָ צְבָאוֹת דִי שְׁכִנְתֵּיהּ בְּטוּרָא דְצִיוֹן:
וַאֲרֵי יֵימְרוּן לְכוֹן עַמְמַיָא דְאַתּוּן בֵּינֵיהוֹן תְּבָעוּ מִן בִּדִין וּמִן זְכוּרוּ דִמְנַצְפִין וְדִמְנַהֲמִין הֲלָא כֵן אוֹרַח עַמְמַיָא פַלְחֵי טַעֲוָתָא עַמָא מִן טַעֲוָתֵיהּ תַּבְעִין חַיָא מִן מֵיתַיָא:
כְּדֵין תֵּימְרוּן לְהוֹן לְאוֹרַיְתָא דְאִתְיְהֵיבַת לָנָא וּלְסַהֲדוּ אֲנַחְנָא שָׁמְעִין אֶלָא תִגְלוּן לְבֵינֵי עַמְמַיָא וְיֵימְרוּן לְכוֹן כְּפִתְגָמָא הָדֵין מִכְּעַן לֵית לֵיהּ מַן דִשְׁחַר וְיַבְעִינֵיהּ:
וְיַעֲדֵי תַקְלָא בְּאַרְעָא וִיהֵי עָקָא וּכְפַן וִיהֵי אֲרֵי יֶחֱזֵי כְפַן וְסִיגוּף וְיִלוּט וְיִבְזֵי שׁוּם פְּתַכְרֵיהּ וְטַעֲוָתֵיהּ וְיִתְפְּנֵי לְעֵילָא לְמִיבְעֵי פּוּרְקַן בָּתַר דְתִתְחַתֵּם גְזֵירָתָא וְלָא יִכּוֹל:
וְסָעִיד מִיַתְבֵי אַרְעָא יִבְעֵי אֲרֵי יַיְתֵי עֲלוֹהִי עָקָא כְּפָן וְשַׁלְהֲבֵי עָקָא קְבֵל וּבַדוּר:
אֲרֵי לָא יִשְׁתַּלְהֵי כָּל דְיֵיתֵי לַאֲעָקָא לְהוֹן כְּעִידַן קֳדָמַי גְלוֹ עַם אֲרַע זְבוּלֻן וְעַם אֲרַע נַפְתָּלִי וּשְׁאַרְהוֹן מְלַךְ תַּקִיף יַגְלֵי עַל דְלָא אַדְכַּרוּ גְבוּרַת יַמָא וְנִסֵי יַרְדְנָא קְרַב כְּרַכֵּי עַמְמַיָא:
