וְשַׁקָרוּ בְּנֵי יִשְׂרָאֵל שְׁקַר בְּחֶרְמָא וּנְסִיב עָכָן בַּר כַּרְמִי בַר זַבְדִי בַר זֶרַח לְשִׁבְטָא דִיהוּדָה מִן חֶרְמָא וּתְקֵיף רוּגְזָא דַייָ בִּבְנֵי יִשְׂרָאֵל:
וּשְׁלַח יְהוֹשֻׁעַ גֻבְרִין מִירִיחוֹ לְעַי דְעִם בֵּית אָוֶן מִמַדְנַח לְבֵית אֵל וַאֲמַר לְהוֹן לְמֵימַר סַקוּ וְאַלִילוּ יַת אַרְעָא וּסְלִיקוּ גַבְרַיָא וְאַלִילוּ יַת עָי:
וְתָבוּ לְוַת יְהוֹשֻׁעַ וַאֲמַרוּ לֵיהּ לָא יִסַק כָּל עַמָא כִּתְרֵין אַלְפִין גַבְרָא אוֹ כִּתְלָתָא אַלְפִין גַבְרָא יִסְקוּן וְיִמְחוּן יַת עָי לָא תַרְגִישׁ לְתַמָן יַת כָּל עַמָא אֲרֵי זְעִירִין אִינוּן:
וּסְלִיקוּ מִן עַמָא לְתַמָן כִּתְלָתָא אַלְפִין גַבְרָא וְאַפִּיכוּ קֳדָם אֱנָשֵׁי עָי:
וּקְטַלוּ מִנְהוֹן אֱנָשֵׁי עַי כִּתְלָתִין וְשִׁתָּא גַבְרָא וּרְדָפוּנוּן קֳדָם תַּרְעָא עַד דְתַבְּרוּנוּן וּמְחוּנוּן בְּמַחֲתָנָא וְאִתְמְסֵי לִבָּא דְעַמָא וַהֲוָה לְמַיָא:
וּבְזַע יְהוֹשֻׁעַ לְבוּשׁוֹהִי וּנְפַל עַל אַפּוֹהִי עַל אַרְעָא קֳדָם אֲרוֹנָא דַייָ עַד רַמְשָׁא הוּא וְסָבֵי יִשְׂרָאֵל וְאַסִיקוּ עַפְרָא עַל רֵישֵׁיהוֹן:
וַאֲמַר יְהוֹשֻׁעַ קַבֵּיל בָּעוּתִי יְיָ אֱלֹהִים לְמָא אַעֲבַרְתָּא אַעֲבָרָא יַת עַמָא הָדֵין יַת יַרְדְנָא לְמִמְסַר יָתָנָא בִּידָא דֶאֱמוֹרָאָה לְאוֹבָדוּתָנָא וּלְוַי פוֹן דִשְׁרֵינָא וִיתֵיבְנָא בְּעִבְרָא דְיַרְדֵנָא:
בְּבָעוּ אֲדֹנָי מָה אֵימַר בָּתַר דְאַחֲזָרוּ יִשְׂרָאֵל קְדַלְהוֹן לְמֶהְפַךְ קֳדָם בַּעֲלֵי דְבָבֵיהוֹן:
וְיִשְׁמְעוּן כְּנַעֲנָאֵי וְכָל יַתְבֵי אַרְעָא וְיִסְתַּחֲרוּן עֲלָנָא וִישֵׁצוּן יַת שְׁמָנָא מִן אַרְעָא וּמָה תַּעֲבֵּד בְּדִיל שְׁמָךְ רַבָּא:
וַאֲמַר יְיָ לִיהוֹשֻׁעַ קוּם לָךְ לָמָה דְנָן אַתְּ רָמֵי עַל אַפָּךְ:
חֲטָא יִשְׂרָאֵל וְאַף עֲבָרוּ עַל קְיָמִי דְפַקְדֵית יַתְהוֹן וְאַף נְסִיבוּ מִן חֶרְמָא וְאַף גְנָבוּ וְאַף כַּדִיבוּ וְאַף שַׁוִיאוּ בְּמָנֵיהוֹן:
וְלָא יִכְלוּן בְּנֵי יִשְׂרָאֵל לְמֵקַם קֳדָם בַּעֲלֵי דְבָבֵיהוֹן קְדַלְהוֹן יַחְזְרוּן לְמֵיפַּךְ קֳדָם בַּעֲלֵי דְבָבֵיהוֹן אֲרֵי הֲווֹ לְחֶרְמָא לָא אוֹסִיף לְמֶהֱוַי מֵימְרִי בְּסַעְדְכוֹן אִם לָא תְשֵׁיצוּן חֶרְמָא מִבֵּנֵיכוֹן:
קוּם זַמֵן יַת עַמָא וְתֵימַר אִזְדַמְנוּ לִמְחַר אֲרֵי כִּדְנַן אֲמַר יְיָ אֱלָהָא דְיִשְׂרָאֵל חֶרְמָא בֵינָךְ יִשְׂרָאֵל לָא תִכּוּל לְמֵיקַם קֳדָם בַּעֲלֵי דְבָבָךְ עַד דְתַעֲדוּן חֶרְמָא מִבֵּינֵיכוֹן:
וְתִתְקַרְבוּן בְּצַפְרָא לְשִׁבְטֵיכוֹן וִיהֵי שִׁבְטָא דְיִתְאַחַד מִן קֳדָם יָיָ יִקְרַב לְזַרְעֲיָן וְזַרְעִיתָא דְיִתְאַחַד מִן קֳדָם יְיָ תִּקְרַב לְבָתִּין וּבֵיתָא דְיִתְאַחַד מִן קֳדָם יָיָ יִתְקַרַב לְגַבְרַיָא:
וִיהֵי דְיִתְאַחַד בְּחֶרְמָא יִתּוֹקַד בְּנוּרָא יָתֵיהּ וְכָל דִי לֵיהּ אֲרֵי עֲבַר עַל קְיָמָא דַייָ וַאֲרַי עֲבַד דְלָא כָשַׁר בְּיִשְׂרָאֵל:
וְאַקְדֵם יְהוֹשֻׁעַ בְּצַפְרָא וְקָרֵיב יַת יִשְׂרָאֵל לְשִׁבְטוֹהִי וְאִתְאַחַד שִׁבְטָא דְבֵית יְהוּדָה:
וְקָרֵיב יַת זַרְעֲיַת יְהוּדָה וְאַחַד יַת זַרְעֲיַת זָרַח וְקָרֵיב יַת זַרְעֲיַת זָרַח לְגַבְרַיָא וְאִתְאַחַד זַבְדִי:
וְקָרֵיב יַת בֵּיתֵיהּ לְגַבְרַיָא וְאִתְאַחַד עָכָן בַּר כַּרְמִי בַּר זַבְדִי בַּר זֶרַח לְשִׁבְטָא דִיהוּדָא:
וַאֲמַר יְהוֹשֻׁעַ לְעָכָן בְּרִי שַׁוִי כְעַן יְקָרָא קֳדָם יְיָ אֱלָהָא דְיִשְׂרָאֵל וְהַב קֳדָמוֹהִי הוֹדָאָה וְחַוֵי כְּעַן לִי מַה עֲבַדְתָּא לָא תְכַסֵי מִינִי:
וַאֲתֵיב עָכָן יַת יְהוֹשֻׁעַ וַאֲמַר בְּקוּשְׁטָא אֲנָא חָבִית קֳדָם יְיָ אֱלָהָא דְיִשְׂרָאֵל וּכְדֵין וּכְדֵין עֲבָדִית:
וַחֲזִית בְּבִזְתָא אִצְטְלִי בַּבְלִי חָד שַׁפִּיר וּמָאתָן סַלְעִין דִכְסַף וְלִישְׁנָא דְדַהֲבָא חָד חַמְשִׁין סַלְעִין מַתְקְלֵיהּ וְרִגַגְתִינוּן וּנְסֵבְתִּנוּן וְהָא אִנוּן טְמִירִין בְּאַרְעָא בְּגוֹ מַשְׁכְּנִי וְכַסְפָּא תְּחוֹתָא:
וּשְׁלַח יְהוֹשֻׁעַ אִזְגַדוּן וּרְהַטוּ לְמַשְׁכְּנָא וְהָא טְמִירִין בְּמַשְׁכְּנֵיהּ וְכַסְפָּא תְּחוֹתָא:
וּנְסִיבוּנוּן מִגוֹ מַשְׁכְּנָא וְאַיְתִיאוּנוּן לְוַת יְהוֹשֻׁעַ וּלְוַת כָּל בְּנֵי יִשְׂרָאֵל וְאַתִּיכוּנוּן קֳדָם יְיָ:
וּדְבַר יְהוֹשֻׁעַ יַת עָכָן בַּר זֶרַח וְיַת כַּסְפָּא וְיַת אִצְטְלָא וְיַת לִישָׁנָא דְדַהֲבָא וּדְבַר יַת בְּנוֹהִי וְיַת בְּנָתֵיהּ וְיַת תּוֹרֵיהּ וְיַת חֲמָרֵיהּ וְיַת עָנֵיהּ וְיַת מַשְׁכְּנֵיהּ וְיַת כָּל דִילֵיהּ וְכָל יִשְׂרָאֵל עִמֵיהּ וְאַסִיקוּ יַתְהוֹן לְמֵישַׁר עָכוֹר:
וַאֲמַר יְהוֹשֻׁעַ מָה עַכַרְתָּנָא יַעְכְּרִינָךְ יְיָ בְּיוֹמָא הָדֵין וּרְגָמוּ יָתֵיהּ כָּל יִשְׂרָאֵל בְּאַבְנַיָא וְאוֹקִידוּ יַתְהוֹן בְּנוּרָא בָּתַר דִרְגִימוּ יַתְהוֹן בְּאַבְנַיָא:
וְאַקִימוּ עֲלוֹהִי דְגוֹר אַבְנִין רַב עַד יוֹמָא הָדֵין וְתַב יְיָ מִתְקוֹף רוּגְזֵיהּ עַל כֵּן קְרָא שְׁמָא דְאַתְרָא הַהוּא מֵישַׁר עָכוֹר עַד יוֹמָא הָדֵין:
