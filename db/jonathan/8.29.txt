וּכְנָשׁוּ פְּלִשְׁתָּאֵי יַת כָּל מַשִׁרְיַתְהוֹן לַאֲפֵק וְיִשְׂרָאֵל שָׁרָן בְּעֵינָא דִי בְיִזְרְעֶאל:
וְטוּרְנֵי פְּלִשְׁתָּאֵי עָבְרִין לְמָאֲוָן וּלְאַלְפִין וְדָוִד וְגַבְרוֹהִי עָבְרִין בְּבַתְרַיְתָא עִם אָכִישׁ:
וַאֲמַרוּ רַבְרְבֵי פְּלִשְׁתָּאֵי מָה יְהוּדָאֵי הָאִלֵין וַאֲמַר אָכִישׁ לְרַבְרְבֵי פְלִשְׁתָּאֵי הֲלָא דֵין דָוִד עַבְדָא דְשָׁאוּל מַלְכָּא דְיִשְׂרָאֵל דַהֲוָה עִמִי דְנַן יוֹמִין אוֹ דְנַן שְׁנִין וְלָא אַשְׁכָּחִית בֵּיהּ מִדַעַם בִּישׁ מִיוֹמָא דְאִתְפְרַשׁ לְוָתִי עַד יוֹמָא הָדֵין:
וּרְגִיזוּ עֲלוֹהִי רַבְרְבֵי פְלִשְׁתָּאֵי וַאֲמַרוּ לֵיהּ רַבְרְבֵי פְלִשְׁתָּאֵי אָתֵיב יַת גַבְרָא וִיתוּב לְאַתְרֵיהּ דְמַנִיתֵיהּ תַּמָן וְלָא יֵחוֹת עִמָנָא בִּקְרָבָא וְלָא יְהֵי לְשָׂטָן בִּקְרָבָא וּבַמֶה יִתְרְעֵי דֵין לְוַת רִבּוֹנֵיהּ הֲלֹא בְּרֵישֵׁי גַבְרַיָא הָאִינוּן:
הֲלָא דֵין דָוִד דְמַשְׁמְעִין לֵיהּ בְּחִנְגַיָא לְמֵימָר קְטַל שָׁאוּל בְּאַלְפִין וְדָוִד בְּרִבְוָן:
וּקְרָא אָכִישׁ לְדָוִד וַאֲמַר לֵיהּ קַיֵם הוּא יְיָ אֲרֵי כָשַׁר אַתְּ וְטַב בְּעֵינַי מַפְקָךְ וּמַעֲלָךְ עִמִי בְּמַשְׁרִיתָא אֲרֵי לָא אַשְׁכָּחִית בָּךְ מִדַעַם בִּישׁ מְיוֹמָא דַאֲתֵיתָא לְוָתִי עַד יוֹמָא הָדֵין וּבְעֵינֵי טוּרְנַיָא לָא תַקֵין אָתְּ:
וּכְעַן תּוּב וֶאֱזֵיל בִּשְׁלָם וְלָא תַעְבֵיד בִּישׁ בְעֵינֵי טוּרְנֵי פְלִשְׁתָּאֵי:
וַאֲמַר דָוִד לְאָכִישׁ אֲרֵי מָה עֲבָדִית וּמַה אַשְׁכַּחְתָּא בְעַבְדָךְ מִיוֹמָא דַהֲוֵיתִי מְשַׁמֵשׁ קֳדָמָךְ עַד יוֹמָא הָדֵין אֲרֵי לָא אֱהָךְ וְאַגִיח קְרָבָא בְּשַׂנְאֵי רִבּוֹנִי מַלְכָּא:
וַאֲתֵיב אָכִישׁ וַאֲמַר לְדָוִד יְדַעְנָא אֲרֵי תַקִין אַתְּ בְּעֵינַי כְּמַלְאָכָא דַייָ בְּרַם רַבְרְבֵי פְלִשְׁתָּאֵי אֲמַרוּ לָא יִסַק עִמָנָא בִּקְרָבָא:
וּכְעַן אַקְדֵים בְּצַפְרָא וְעַבְדֵי רִבּוֹנָךְ דַאֲתוֹ עִמָךְ וּתְקַדְמוּן בְּצַפְרָא וְיִנְהַר לְכוֹן וְתֵיזְלוּן:
וְאַקְדֵים דָוִד הוּא וְגַבְרוֹהִי לְמֵיזַל בְּצַפְרָא לְמֵתַב לַאֲרַע פְּלִשְׁתָּאֵי וּפְלִשְׁתָּאֵי סְלִיקוּ לְיִזְרְעֶאל:
