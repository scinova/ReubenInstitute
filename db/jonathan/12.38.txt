בְּיוֹמַיָא הָאִנוּן מָרַע חִזְקִיָה לִמְמָת וַאֲתָא לְוָתֵהּ יְשַׁעְיָהוּ בַר אָמוֹץ נְבִיָא וַאֲמַר לֵיהּ כִדְנַן אֲמַר יְיָ פַּקֵד עַל אֱנַשׁ בֵּיתָךְ אֲרֵי מָאִת אַתְּ וְלָא תֵיהֵי מִמַרְעָךְ:
וְאַסְחַר חִזְקִיָהוּ אַפּוֹהִי לְכוֹתֵל בֵּית מַקְדְשָׁא וְצַלִי קֳדָם יְיָ:
וַאֲמַר קַבֵּל בָּעוּתִי יְיָ אִדְכַּר כְּעַן יַת דִי פְלָחִית קֳדָמָךְ בִּקְשׁוֹט וּבְלֵב שָׁלֵם וּדְתַקִין בְּעֵינָךְ עַבְדִית וּבְכָא חִזְקִיָה בְּכִיא סַגִיאָה:
וַהֲוָה פִתְגָמֵי נְבוּאָה מִן קֳדָם יְיָ עִם יְשַׁעְיָה לְמֵימָר:
אֱזֵיל וְתֵימַר לְחִזְקִיָה כִּדְנַן אֲמַר יְיָ אֱלָהֵיהּ דְדָוִד אָבוּךְ שְׁמִיעַ קֳדָמַי צְלוֹתָךְ גַלְיָא קֳדָמַי דִמְעֲתָךְ הָא אֲנָא מוֹסִיף עַל יוֹמָךְ חֲמֵשׁ עַסְרֵי שְׁנִין:
וּמִיַד מַלְכָּא דְאַתּוּר אֲשֵׁיזְבִינָךְ וְיַת קַרְתָּא הָדָא וְאַגִין עַל קַרְתָּא הָדָא:
וְדֵין לָךְ אָתָא מִן קֳדָם יְיָ דְיַעְבֵּיד יְיָ יַת פִּתְגָמָא הָדֵין דִי מַלֵיל:
הָא אֲנָא מֵתִיב יַת טוּל אֶבֶן שְׁעַיָא דְנָחְתַת בְּמַסְקָנָא דְאָחָז שִׁמְשָׁא לַאֲחוֹרוֹהִי עֲסַר שְׁעָן וְתַב שִׁמְשָׁא עֲסַר שְׁעָן בְּצוּרַת אֶבֶן שְׁעַיָא דְנָחֲתַת:
כְּתַב אוֹדָאָה עַל נִיסָא דְאִתְעַבֵיד לְחִזְקִיָה מְלַךְ שִׁבְטָא דְבֵית יְהוּדָה כַּד מְרַע וְאִתְּסֵי מִמַרְעֵהּ:
אֲנָא אֲמָרֵית בִּדְוָן יוֹמֵי אֱהַךְ בְּתַרְעֵי שְׁאוֹל עַל דָכְרָנִי לְטַב אִתּוֹסַף עַל שְׁנָי:
אֲמָרֵית לָא אִתְחַזֵי עוֹד קֳדָם דְחֵילָא דַייָ בְּאַרְעָא בֵית שְׁכִינְתֵּהּ דִי בָה אֲרִיכוּת חַיָא וְלָא אֶפְלַח קֳדָמוֹהִי עוֹד בְּבֵית מַקְדְשָׁא דְמִתַּמָן עֲתִיד לְמִפַּק חֶדְוָא לְכָל יַתְבֵי אַרְעָא מִלְמֵידַר בִּירוּשְׁלֵם קַרְתָּא דְקוּדְשָׁא מְדוֹרֵי פְּסָק:
מִבְּנֵי דָרִי אִתְנְטִילוּ יוֹמֵי אִתְקַצִיצוּ וּגְלוּ מִנִי אִתְכְּפִילוּ כְּמַשְׁכְּנָא דְרָעֵי אִתְקְפָדוּ כְּנַחַל נַדְדָאִין חַיַי מִיקָר מַלְכוּתִי אֲנָא גָּלֵי יְמָמָא וְלֵילְוָתִי שְׁלִימוּ:
נַהֲמִית עַד צַפְרָא כְּאַרְיָא דְנָהֵים וְתָבַר גַרְמַי חֵיוָתָא כֵּן מִתַּבְּרִין מִן קֳדָם דְוָנָא כָּל גַרְמֵי יְמָמַי וְלֵילְוָתִי שְׁלִימוּ:
כְּסוּסְיָא דַאֲחִיד וּמְנַצֵיף כֵּן נְצֵיפַת וְנַהֲמֵית כְּיוֹנָה זְקָפִית עֵינַי דְיֵיתֵי לִי רְוַח מִן קֳדָם דִשְׁכִינְתֵּהּ בִשְׁמֵי מְרוֹמָא יְיָ קַבֵּיל צְלוֹתִי עֲבֵיד בָּעוּתִי:
מָה אֲמַלֵיל תּוּשְׁבַּחְתָּא וְאֵימַר קֳדָמוֹהִי וְהוּא אַסְגֵי טַבְוָן לְמֶעְבַּד עִמִי מָה אֶפְלַח וַאֲשַׁלֵים קֳדָמוֹהִי כָּל שְׁנַיָא דְאוֹסִיף עַל חַיַי וְשֵׁיזֵיב מִמְרִיר נַפְשִׁי:
יְיָ עַל כָּל מֵיתַיָא אֲמַרְתָּא לְאַחָיָאָה וּקְדָם כּוּלְהוֹן אַחְיֵיתָא רוּחִי וְאַחְיֵיתָנִי וְקַיֵימְתָּנִי:
הָא לְעָבְדֵי אוֹרַיְתָא סַגִי שְׁלָמָא קָדָמָךְ וְאַתְּ מַיְיתֵי מְרִירָא לְרַשִׁיעַיָא בְּכֵן כַּד יְדָעִית יוֹם מוֹתִי שְׁפָכִית דִמְעֲתִי בִּצְלוֹ קֳדָמָךְ מַר לִי סַגִי וְאַתְּ אִתְרְעֵיתָא בְּחַיֵי בְּדִיל דְלָא לְחַבָּלָא נַפְשִׁי אֲרֵי אַרְחֵיקְתָּא מִן קֳדָמָךְ כָּל חֲטָאָי:
אֲרֵי לָא דְבִשְׁאוֹל מוֹדַן קֳדָמָךְ מֵיתַיָא לָא מְשַׁבְּחִין לָךְ לָא מְסַבְּרִין נַחְתֵי גוּב בֵּית אָבְדָנָא לְפוּרְקָנָךְ:
דְחַי חַי יוֹדֵי קֳדָמָךְ כְּוָתִי יוֹמָא הָדֵין אֲבָהָן לִבְנֵיהוֹן יְחַווּן גְבוּרְתֵּיךְ וְיוֹדוּן לְמֵימָר דְכָל אִילֵין קְשׁוֹט:
יְיָ לְמִפְרְקָנָא אֲמַר יְיָ וְנִגוּן תּוּשְׁבַּחְתֵּהּ נְנַגֵן כָּל יְמֵי חַיָנָא עַל בֵּית מַקְדְשָׁא דַיָי:
וַאֲמַר יְשַׁעְיָהוּ יִסְבוּן דְבֵילַת תֵּינִין וִישׁוֹרְטוּן עַל שִׁיחֲנָא וְאִתַּסֵי:
וַאֲמַר חִזְקִיָה מָה אָת אֲרֵי אֶסַק לְבֵית מַקְדְשָׁא דַיָי:
