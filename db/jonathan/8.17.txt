וּכְנָשׁוּ פְלִשְׁתָּאֵי יַת מַשְׁרִיתֵיהוֹן לְאַגָחָא קְרָבָא וְאִתְכַּנָשׁוּ לְשׁוֹכֹה דִי לְשֵׁבֶט יְהוּדָה וּשְׁרוֹ בֵּין שׁוֹכֹה וּבֵין עֲזֵקָה בְּאֶפֶס דַמִים:
וְשָׁאוּל וֶאֱנַשׁ יִשְׂרָאֵל אִתְכְּנִישׁוּ וּשְׁרוֹ בְּמֵישַׁר בּוּטְמָא וְסַדָרוּ קְרָבָא לָקֳדָמוּת פְּלִשְׁתָּאֵי:
וּפְלִשְׁתָּאֵי קַיְמִין עַל טוּרָא מִכָּא וְיִשְׂרָאֵל קַיְמִין עַל טוּרָא מִכָּא וְחֵילְתָא בֵּינֵיהוֹן:
וּנְפַק גַבְרָא מִבֵּינֵיהוֹן מִמַשִׁרְיַת פְלִשְׁתָּאֵי גָלְיָת שְׁמֵיהּ מִגָת רוּמֵיהּ שִׁתָּא אַמִין וְזֵירְתָא:
וְקוּלַס דִנְחַשׁ עַל רֵישֵׁיהּ וְשִׁרְיָן גַלְבִּין הוּא לָבֵישׁ וּמַתְקַל שִׁרְיוֹנָא חַמְשָׁא אַלְפִין תִּקְלֵי נְחָשָׁא:
וּטְרַקְלִילִין דִנְחַשׁ עַל רִגְלוֹהִי וּמְסַחֲפָא דִנְחָשָׁא נָפִיק מִן קוּלְסָא וּמְטַל בֵּין כְּתֵפוֹהִי:
וְאָעָא דְמוּרְנִיתֵיהּ כְּאַכְסָן דְגַרְדָאִין וְשִׁנָנָא דְמוּרְנִיתֵיהּ מַתְקַל שִׁית מְאָה תִקְלֵי פַרְזְלָא וְנָטַל תְּרֵיסָא אָזֵיל קֳדָמוֹהִי:
וְקָם וְאַכְרֵיז עַל סִדְרֵי יִשְׂרָאֵל וַאֲמַר לְהוֹן לְמָה תִפְקוּן לְסַדְרָא קְרָבָא הֲלָא אֲנָא פְלִשְׁתָּאָה וְאַתּוּן עַבְדִין לְשָׁאוּל (אֲנָא הוּא גָלְיָת פְּלִשְׁתְּאָה דְמִן גַת דִקְטָלִית תְּרֵין בְּנֵי עֵלִי כַּהֲנֵי חָפְנִי וּפִינְחָס וּשְׁבֵיתִי יַת אַרוֹן קֳיָמָא דַייָ וְאוֹבֵלִית יָתֵיהּ לְבֵית דָגוֹן טַעֲוָתִי וַהֲוָה תַמָן בְּקִירְוֵי פְלִשְׁתָּאֵי שַׁבְעָא יַרְחִין אַף כָּל קְרָב וּקְרָב דַהֲווֹ לְהוֹן לִפְלִשְׁתָּאֵי אֲנָא נָפֵיק בְּרֵישׁ חֵילָא וּנְצַחְנָא בִקְרָבָא וּרְמִינָא קְטִילִין כְּעַפְרָא דְאַרְעָא וְעַד כְּעַן לָא אַכְשָׁרוּ יָתִי פְּלִשְׁתָּאֵי לְמֶהֱוֵי רַב אַלְפָא עִלַוֵיהוֹן וְאַתּוּן בְּנֵי יִשְׂרָאֵל מַה גְבוּרָה עֲבַד לְכוֹן שָׁאוּל בַּר קִישׁ דְמִן גִבְעֲתָא דְמַנִיתוּן יָתֵיהּ מַלְכָּא עִלַוֵיכוֹן אִם גְבַר גִבָּר הוּא יֵיחוֹת וְיַעְבֵּיד קְרָבָא עִמִי וְאִם גְבַר חַלָשׁ הוּא) בְּחַרוּ לְכוֹן גַבְרָא וְיֵחוֹת לְוָתִי:
אִם יִכּוּל לַאֲגָחָא קְרָבָא עִמִי וְיִקְטְלִינַנִי וּנְהֵי לְכוֹן לְעַבְדִין וְאִם אֲנָא אִכּוּל לֵיהּ וְאַקְטִלִינֵיהּ וּתְהוֹן לָנָא לְעַבְדִין וְתִפְלְחוּן יָתָנָא:
וַאֲמַר פְּלִשְׁתָּאָה אֲנָא חֲסָדִית יַת סִדְרֵי יִשְׂרָאֵל יוֹמָא דֵין לְמֵימָר אַיְתוֹ לִי גַבְרָא וְנַגִיחַ קְרָבָא כַּחֲדָא:
וּשְׁמַע שָׁאוּל וְכָל יִשְׂרָאֵל יַת פִּתְגָמֵי פְּלִשְׁתָּאָה הָאִלֵין וְאִתְבָּרוּ וּדְחִילוּ לַחֲדָא:
וְדָוִד בַּר גְבַר אֶפְרָתִי הָדֵין מִבֵּית לֶחֶם דְבֵית יְהוּדָה וּשְׁמֵיהּ יִשַׁי וְלֵיהּ תְּמַנְיָא בְנִין וְגַבְרָא בְּיוֹמֵי שָׁאוּל סִיב מְנִי בִּבְחִירַיָא:
וַאֲזַלוּ תְּלָתָא בְּנֵי יִשַׁי רַבְרְבַיָא אֲזָלוּ בָּתַר שָׁאוּל לִקְרָבָא וְשׁוּם תְּלָתָא בְנוֹהִי דַאֲזָלוּ לִקְרָבָא אֱלִיאָב בּוּכְרָא וְתִנְיָנֵיהּ אֲבִינָדָב וּתְלִיתָאָה שַׁמָה:
וְדָוִד הוּא זְעֵירָא וּתְלָתָא רַבְרְבַיָא אֲזַלוּ בָּתַר שָׁאוּל:
וְדָוִד אֲזַל וְתַב מִלְוַת שָׁאוּל לְמִרְעֵי יַת עֲנָא דַאֲבוּהִי בֵּית לֶחֶם:
וְקָרֵיב פְּלִשְׁתָּאָה וּמַקְדִים וּמַחֲשִׁיךְ וְאִתְעַתַּד אַרְבְּעִין יוֹמִין:
וַאֲמַר יִשַׁי לְדָוִד בְּרֵיהּ סַב כְּעַן לַאֲחָיִךְ מְכִילְתָא דְקָלְיָא הָדֵין וְעַסְרָא טוּלְמִין דְלַחְמָא הָדֵין וְאוֹבִיל לְמַשְׁרִיתָא לַאֲחָיךְ:
וְיַת עֲסַר גוּבְנִין דַחֲלָבָא הָאִלֵין תּוֹבֵיל לְרַב אַלְפָא וְיַת אַחָיךְ תַּסְעַר לִשְׁלָם וְיַת טִיבֵיהוֹן תַּיְתֵי:
וְשָׁאוּל וְאִינוּן וְכָל אֱנַשׁ יִשְׂרָאֵל בְּמֵישַׁר בּוּטְמָא מְגִיחִין קְרָבָא עִם פְּלִשְׁתָּאֵי:
וְאַקְדֵם דָוִד בְּצַפְרָא וּשְׁבַק יַת עָנָא עַל נָטְרָא וּנְסִיב וַאֲזַל כְּמָה דְפַּקְדֵיהּ יִשַׁי וְאָתָא לְכַרְכּוּמָא וּמַשִׁרְיַת עָבְדֵי קְרָבָא נְפָקוּ לְסִידְרָא וְיַבִּיבוּ בִּקְרָבָא:
וְסַדָרוּ יִשְׂרָאֵל וּפְלִשְׁתָּאֵי סִדְרָא לִקֳדָמוּת סִדְרָא:
וּשְׁבַק דָוִד יַת מָנַיָא דִי עֲלוֹהִי עַל יַד נָטִיר מָנַיָא וּרְהַט לְסִדְרָא וְאָתָא וְשָׁאֵל לַאֲחוֹהִי בִּשְׁלָם:
וְהוּא מְמַלֵיל עִמְהוֹן וְהָא גַבְרָא מִבֵּינֵיהוֹן סְלִיק גָלְיַת פְּלִשְׁתָּאָה שְׁמֵיהּ מִגָת מִסִדְרֵי פְלִשְׁתָּאֵי וּמַלֵיל כְּפִתְגָמַיָא הָאִלֵין וּשְׁמַע דָוִד:
וְכָל אֱנַשׁ יִשְׂרָאֵל בְּמֶחֱזֵיהוֹן יַת גַבְרָא וְאַפִּיכוּ מִן קֳדָמוֹהִי וּדְחִילוּ לַחֲדָא:
וַאֲמַר אֱנַשׁ יִשְׂרָאֵל הַחֲזֵיתוּן גַבְרָא דְסָלִיק הָדֵין אֲרֵי לְחַסָדָא יַת יִשְׂרָאֵל סְלִיק וִיהֵי גַבְרָא דְיִקְטְלִינֵיהּ יַעֲתְּרִינֵיהּ מַלְכָּא עוֹתַר סַגִי וְיַת בְּרַתֵּיהּ יִתֵּן לֵיהּ וְיַת בֵּית אֲבוּהִי יַעֲבֵּיד חוֹרִין רַבְרְבִין בְּיִשְׂרָאֵל:
וַאֲמַר דָוִד לְגַבְרַיָא דְקַיְמִין עִמֵיהּ לְמֵימַר מַה יִתְעֲבֵיד לְגַבְרָא דְיִקְטוֹל יַת פְלִישְׁתָּאָה דֵיכִי וְיַעְדִי חִסוּדָא מֵעַל יִשְׂרָאֵל אֲרֵי מַן פְלִשְׁתָּאָה עֲרֵלָא הָדֵין אֲרֵי חֲסַד סִדְרֵי עַבְדֵי קְרָבָא דְעַמֵיהּ דַייָ קַיָמָא:
וַאֲמַר לֵיהּ עַמָא כְּפִתְגָמָא הָדֵין לְמֵימַר כְּדֵין יִתְעֲבֵיד לְגַבְרָא דְיִקְטְלִינֵיהּ:
וּשְׁמַע אֱלִיאָב אֲחוּהִי רַבָּא בְּמַלָלוּתֵיהּ עִם גַבְרַיָא וּתְקֵיף רוּגְזָא דֶאֱלִיאָב בְּדָוִד וַאֲמַר לְמָה דְנַן נְחַתְתָּא וְעַל מַן רְטַשְׁתָּא זְעֵיר עָנָא הָאִינוּן בְּמַדְבְּרָא אֲנָא יְדַעְנָא יַת בַּקְרְנוּתָךְ וְיַת בִּישׁוּת לִבָּךְ אֲרֵי בְּדִיל לְמֶחְזֵי עָבְדֵי קְרָבָא נְחַתְתָּא:
וַאֲמַר דָוִד מָה עֲבָדִית כְּעַן הֲלָא פִּתְגָם הוּא דַאֲמָרִית:
וְאִסְתְּחַר מִלְוָתֵיהּ לִקֳבֵיל אוֹחֲרָן וַאֲמַר כְּפִתְגָמָא הָדֵין וַאֲתִיבוּהִי עַמָא פִּתְגָמָא כְּפִתְגָמָא קַדְמָאָה:
וְאִשְׁתְּמָעוּ פִתְגָמַיָא דְמַלֵיל דָוִד וְחַוִיאוּ קֳדָם שָׁאוּל וְדַבַּרְהִי:
וַאֲמַר דָוִד לְשָׁאוּל לָא יִתְבַּר לִבָּא דֶאֱנָשָׁא מִנֵיהּ עַבְדָךְ יֵיזִיל וְיַגִיחַ קְרָבָא עִם פִּלִשְׁתָּאָה הָדֵין:
וַאֲמַר שָׁאוּל לְדָוִד לָא תִכּוּל לְמֵיזַל עִם פְּלִשְׁתָּאָה הָדֵין לְאַגָחָא קְרָבָא עִמֵיהּ אֲרֵי יַנִיק אַתְּ וְהוּא גְבַר עָבִיד קְרָבִין מִיַנְקוּתֵיהּ:
וַאֲמַר דָוִד לְשָׁאוּל רַעֲיָא הֲוָה עַבְדָךְ לַאֲבוּהִי בַּעֲנָא וְאָתָא אַרְיָא וְאַף דוּבָּא וּנְסִיב אִמְרָא מֵעִדָרָא:
וּנְפָקִית בַּתְרוֹהִי וּמְחִיתֵיהּ וְשֵׁזִיבְתֵיהּ מִפּוּמֵיהּ וְקָם עָלַי וַאֲחָדִית בְּלוֹעֵיהּ וּמְחִיתֵיהּ וּקְטַלְתֵּיהּ:
אַף יַת אַרְיָא אַף דֻבָּא קְטַל עַבְדָךְ וִיהֵי פְּלִשְׁתָּאָה עֲרֵלָא הָדֵין כְּחַד מִנְהוֹן אֲרֵי חֲסַד סִדְרֵי עַמָא דַייָ קַיָמָא:
וַאֲמַר דָוִד יְיָ דְשֵׁיזְבַנִי מִידָא דְאַרְיָא וּמִידָא דְדוּבָּא הוּא יְשֵׁיזְבִינַנִי מִידָא דִפְלִשְׁתָּאָה הָדֵין וַאֲמַר שָׁאוּל לְדָוִד אֱזֵיל וּמֵימְרָא דַייָ יְהֵי בְסַעֲדָךְ:
וְאַלְבֵּישׁ שָׁאוּל יַת דָוִד לְבוּשׁוֹהִי וִיהַב קוּלַס דִנְחָשׁ עַל רֵישֵׁיהּ וְאַלְבֵּישׁ יָתֵיהּ שִׁרְיָנָא:
וְזָרֵיז דָוִד יַת חַרְבֵּיהּ מֵעַל לִלְבוּשׁוֹהִי וְלָא אָבָא לְמֵיזַל אֲרֵי לָא אַלִיף וַאֲמַר דָוִד לְשָׁאוּל לֵית אֲנָא יָכִיל לְמֵיזַל בְּאִלֵין אֲרֵי לֵית בְּהוֹן נִסָא וְאַעְדִינוּן דָוִד מֵעִלַווֹהִי:
וּנְסִיב חוּטְרֵיהּ בִּידֵיהּ וּבְחַר לֵיהּ חַמְשָׁא שְׁעִיעֵי אַבְנַיָא מִן נַחֲלָא וְשַׁוִי יַתְהוֹן בְּמָאן רַעֲיָא דִי לֵיהּ וּבְתַרְמִילֵיהּ וְקַלְעֵיהּ בִּידֵיהּ וּקְרִיב לְוַת פְלִשְׁתָּאָה:
וַאֲזַל פְּלִשְׁתָּאָה אָזֵיל וְקָרֵיב לְוַת דָוִד וְגַבְרָא נָטֵיל תְּרֵיסָא קֳדָמוֹהִי:
וְאִסְתְּכֵי פְלִשְׁתָּאָה וַחֲזָא יַת דָוִד וְשַׁטְיֵהּ אֲרֵי הֲוָה יַנִיק וְסַמוֹק שַׁפִּיר בְּרֵיוֵיהּ:
וַאֲמַר פְּלִשְׁתָּאָה לְדָוִד הֲכַלְבָּא שָׁטֵי אֲנָא דְאַתְּ אָתֵי עָלַי בְּחוּטְרַיָא וְלַיִט פְּלִשְׁתָּאָה יַת דָוִד בְּטַעֲוָתֵיהּ:
וַאֲמַר פְּלִשְׁתָּאָה לְדָוִד אִיתָא לְוָתִי וְאֶתֵּן יַת בִּסְרָךְ לְעוֹפָא דִשְׁמַיָא וְלִבְעִירָא דְאַרְעָא:
וַאֲמַר דָוִד לִפְלִשְׁתָּאָה אַתְּ אָתֵי עָלַי בְּחַרְבָּא וּבְמוּרְנִיתָא וּבְרוֹמְחָא וַאֲנָא אָתֵי עֲלָךְ בִּשְׁמָא דַייָ צְבָאוֹת אֱלָהָא דְסִדְרֵי יִשְׂרָאֵל דַחֲסַדְתָּא:
יוֹמָא הָדֵין יִמְסְרִינָךְ יְיָ בִּידִי וְאֶקְטְלִינָךְ וְאַעְדִי יַת רֵישָׁךְ מִנָךְ וְאֶתֵּן פִּגוּל מַשִׁרְיַת פְּלִשְׁתָּאֵי יוֹמָא הָדֵין לְעוֹפָא דִשְׁמַיָא וּלְחֵיוַת אַרְעָא וְיֵדְעוּן כָּל דַיָרֵי אַרְעָא דְאִית אֱלָהָא בְּיִשְׂרָאֵל:
וְיֵדְעוּן כָּל קְהָלָא הָדֵין אֲרֵי לָא בְּחַרְבָּא וּבְמוּרְנִיתָא פָּרִיק יְיָ אֲרֵי מִן קֳדָם יְיָ נִצְחָן קְרָבַיָא וְיִמְסַר יַתְכוֹן בִּידָנָא:
וַהֲוָה כַּד קָם פְּלִשְׁתָּאָה וַאֲזַל וּקְרַב לִקֳדָמוּת דָוִד וְאוֹחֵי דָוִד וּרְהַט לְסִדְרָא לָקֳדָמוּת פְּלִשְׁתָּאָה:
וְאוֹשִׁיט דָוִד יַת יְדֵיהּ לְמָנָא וּנְסִיב מִתַּמָן אַבְנָא וְאַקְלַע בִּקִלְעָא וּמְחָא יַת פְּלִשְׁתָּאָה עַל בֵּית עֵינוֹהִי וּטְבָעַת אַבְנָא בְּבֵית עֵינוֹהִי וּנְפַל עַל אַפּוֹהִי עַל אַרְעָא:
וּתְקֵיף דָוִד מִן פְּלִשְׁתָּאָה בְּקִלְעָא וּבְאַבְנָא וּמְחָא יַת פְּלִשְׁתָּאָה וְקַטְלֵיהּ וְחַרְבָּא לֵית בִּידָא דְדָוִד:
וּרְהַט דָוִד וְקָם עַל פְּלִשְׁתָּאָה וּנְסִיב יַת חַרְבֵּיהּ וְשַׁלְפָא מִנְדָנָהּ וְקַטְלֵיהּ וּפְסַק בֵּיהּ יַת רֵישֵׁיהּ וַחֲזוֹ פְּלִשְׁתָּאֵי אֲרֵי מִית גִבַּרְהוֹן וְאַפִּיכוּ:
וְקָמוּ אֱנָשֵׁי יִשְׂרָאֵל וִיהוּדָה וְיַבִּיבוּ וּרְדָפוּ יַת פְּלִשְׁתָּאֵי עַד מַעֲלָנָא דְגַיְא וְעַד תַּרְעֵי עֶקְרוֹן וּנְפַלוּ קָטוֹלֵי פְלִשְׁתָּאֵי בְּאוֹרַח שַׁעֲרַיִם וְעַד גַת וְעַד עֶקְרוֹן:
וְתָבוּ בְּנֵי יִשְׂרָאֵל מִלְמִרְדַף בָּתַר פְלִשְׁתָּאֵי וּבְזוֹ יַת מַשִׁרְיַתְהוֹן:
וּנְסִיב דָוִד יַת רֵישָׁא דִפְלִשְׁתָּאָה וְאַיְתֵיהּ לִירוּשְׁלֵם וְיַת זְיָנֵיהּ שַׁוִי בְמַשְׁכְּנֵיהּ:
וְכַד חֲזָא שָׁאוּל יַת דָוִד נָפִיק לָקֳדָמוּת פְּלִשְׁתָּאֵי אֲמַר לְאַבְנֵר רַב חֵילָא בַּר מַן דֵין עוּלֵימָא אַבְנֵר וַאֲמַר אַבְנֵר חַיֵי נַפְשָׁךְ מַלְכָּא אִם יְדַע אֲנָא:
וַאֲמַר מַלְכָּא שְׁאַל אַתְּ בַּר מַן דֵין עוּלֵימָא:
וְכַד תַּב דָוִד מִלְמִקְטַל יַת פְלִשְׁתָּאָה וּדְבַר יָתֵיהּ אַבְנֵר וְאַיְתֵיהּ קֳדָם שָׁאוּל וְרֵישָׁא דִפְלִשְׁתָּאָה בִּידֵיהּ:
וַאֲמַר לֵיהּ שָׁאוּל בַּר מַן אַתְּ עוּלֵימָא וַאֲמַר דָוִד בַּר עַבְדָךְ יִשַׁי דְמִבֵּית לָחֶם:
