פִּתְגַם נְבוּאָה דַהֲוָה עִם יִרְמְיָה עַל כָּל עַמָא דְבֵית יְהוּדָה בְּשַׁתָּא רְבִיעֵיתָא לִיהוֹיָקִים בַּר יֹאשִׁיָה מְלַךְ שִׁבְטָא דְבֵית יְהוּדָה הִיא שַׁתָּא קַדְמֵיתָא לִנְבוּכַדְרֶצַר מַלְכָּא דְבָבֶל:
דְאִתְנַבֵּי יִרְמְיָה נְבִיָא עַל כָּל עַמָא דְבֵית יְהוּדָה וְעַל כָּל יַתְבֵי יְרוּשְׁלֵם לְמֵימָר:
מִן תְּלַת עַשְׂרֵי שְׁנִין לְיֹאשִׁיָה בַּר אָמוֹן מְלַךְ שִׁבְטָא דְבֵית יְהוּדָה וְעַד יוֹמָא הָדֵין דְנַן עַשְׂרִין וּתְלַת שְׁנִין הֲוָה פִּתְגַם נְבוּאָה מִן קֳדָם יְיָ עִמִי וּמַלְלֵית עִמְכוֹן מַקְדֵם וּמְמַלֵיל וְלָא קַבֵּילְתּוּן:
וּשְׁלַח יְיָ לְוַתְכוֹן יַת כָּל עַבְדוֹהִי נְבִיַיָא מַקְדֵם וּמַשְׁלַח וְלָא קַבֵּילְתּוּן וְלָא אַרְכֵינְתּוּן יַת אוּדְנְכוֹן לְקַבְּלָא:
לְמֵימָר תּוּבוּ כְעַן גְבַר מֵאוֹרְחֵיהּ בִּישְׁתָּא וּמִבִּישׁוּת עוֹבָדֵיכוֹן וְתִיבוּ עַל אַרְעָא דִיהַב יְיָ לְכוֹן וְלַאֲבָהַתְכוֹן לְמִן עַלְמָא וְעַד עַלְמָא:
וְלָא תְהָכוּן בָּתַר טַעֲוַת פַּלְחֵי כּוֹכְבַיָא לְמִפְלַחְהָן וּלְמִסְגַד לְהֵן וְלָא תִרְגְזוּן קָדָמַי בְּעוֹבָדֵי יְדֵיכוֹן וְלָא אַבְאֵישׁ לְכוֹן:
וְלָא קַבֵּילְתּוּן לְמֵימְרִי אֲמַר יְיָ בְּדִיל דְלָא לְאַרְגָזָא קָדָמַי בְּעוֹבָדֵי יְדֵיכוֹן לְאַבְאָשָׁא לְכוֹן:
בְּכֵן כִּדְנַן אֲמַר יְיָ צְבָאוֹת חֲלַף דְלָא קַבֵּילְתּוּן יַת פִּתְגָמָי:
הָא אֲנָא שְׁלַח וַאֲדַבֵּר יַת כָּל מַלְכְּוַת צִפּוּנָא אֲמַר יְיָ וְלִנְבוּכַדְרֶאצַר מַלְכָּא דְבָבֶל עַבְדִי וְאַיְתִינוּן עַל אַרְעָא הָדָא וְעַל יַתְבָהָא וְעַל כָּל עַמְמַיָא הָאִלֵין סְחוֹר סְחוֹר וֶאֱגַמְרִינוּן וַאֲשַׁוִינוּן לְצָדוּ וּלְאִשְׁתְּמָמוּ וּלְחָרְבַת עֲלָם:
וְאוֹבַד מִנְהוֹן קַל בִּיעַ וְקַל חֶדְוָא קַל חַתְנִין וְקַל כַּלָן קַל סִיעַן דִמְשַׁבְּחִין בְּנֵיהוֹר בּוּצִין:
וּתְהֵי כָּל אַרְעָא הָדָא לְחָרְבָּא לְצָדוּ וְיִפְלְחוּן עַמְמַיָא הָאִלֵין יַת מַלְכָּא דְבָבֶל שִׁבְעִין שְׁנִין:
וִיהֵי כְמִשְׁלַם שַׁבְעִין שְׁנִין אַסְעַר עַל מַלְכָּא דְבָבֶל וְעַל עַמָא הַהוּא אֲמַר יְיָ יַת חוֹבֵיהוֹן וְעַל אֲרַע כַּסְדָאֵי וַאֲשַׁוֵי יָתֵיהּ לְצָדוּת עֲלָם:
וְאַיְתֵי עַל אַרְעָא הַהִיא יַת כָּל פִתְגָמַיָא דִגְזָרִית עֲלָהּ יַת כָּל דִכְתִיב בְּסִפְרָא הָדֵין דְאִתְנַבֵּי יִרְמְיָה עַל כָּל עַמְמַיָא:
אֲרֵי פְלָחוּ בְהוֹן אַף אִינוּן עַמְמִין סַגִיאִין וּמַלְכִין רַבְרְבִין וַאֲשַׁלֵים לְהוֹן כֶּאֱגוֹרֵיהוֹן וּכְעוֹבְדֵי יְדֵיהוֹן:
אֲרֵי כִדְנַן אֲמַר יְיָ אֱלָהָא דְיִשְׂרָאֵל לִי קַבֵּיל יַת כָּסָא דַחֲמַר לְוָטָא הָדֵין מִן קֳדָמַי וְתַשְׁרֵי יָתֵיהּ יַת כָּל עַמְמַיָא דַאֲנָא שְׁלַח יָתָךְ עֲלֵיהוֹן:
וְיִשְׁתּוּן וְיִשְׁתַּגְשׁוּן וְיִטְרְפוּן מִן קֳדָם דְקַטְלִין בְּחַרְבָּא דַאֲנָא שְׁלַח בֵּינֵיהוֹן:
וְקַבְּלֵית יַת כַּסָא מִן קֳדָם יְיָ וְאַשְׁקֵיתִי יַת כָּל עַמְמַיָא דְשַׁלְחַנִי יְיָ לְוַתְהוֹן:
יַת יְרוּשְׁלֵם וְיַת קִרְוַיָא דְבֵית יְהוּדָה וְיַת מַלְכָהָא וְיַת רַבְרְבָהָא לְמִתַּן יַתְהוֹן לְחַרְבָּא לְצָדוּ לְאִשְׁתְּמָמוּ וּלְלוּטָא כְּיוֹמָא הָדֵין:
יַת פַּרְעֹה מַלְכָּא מִצְרַיִם וְיַת עַבְדוֹהִי וְיַת רַבְרָבוֹהִי וְיַת כָּל עַמֵיהּ:
וְיַת כָּל מַלְכֵי סוּמְכְוָתָא וְיַת כָּל מַלְכֵי אֲרַע עוּץ וְיַת כָּל מַלְכֵי אֲרַע פְלִשְׁתָּאֵי וְיַת אַשְׁקְלוֹן וְיַת עַזָה וְיַת עֶקְרוֹן וְיַת שְׁאָרָא דְאַשְׁדוֹד:
יַת אֱדוֹם וְיַת מוֹאָב וְיַת בְּנֵי עֲמוֹן:
וְיַת כָּל מַלְכֵי צוֹר וְיַת כָּל מַלְכֵי צִידוֹן וְיַת כָּל מַלְכֵי נִיסָא דִי בְעֶבְרָא דְיַמָא:
יַת דְדָן וְיַת תֵּימָא וְיַת בּוּז כָּל מַקְפֵי פָאתָא:
וְיַת כָּל מַלְכֵי עֲרָב וְיַת כָּל מַלְכֵי עֲרָבָאֵי דְשָׁרָן בִּשְׁכוּנִין בְּמַדְבְּרָא:
וְיַת כָּל מַלְכֵי זִמְרִי וְיַת כָּל מַלְכֵי עֵילָם וְיַת כָּל מַלְכֵי מָדָי:
וְיַת כָּל מַלְכֵי צִפּוּנָא דְקָרִיבִין וּדְרָחִיקִין גְבַר לַאֲחוֹהִי וְיַת כָּל מַלְכְּוַת עַמְמַיָא דְעַל אַפֵּי אַרְעָא וּמַלְכָּא דְבָבֶל יִשְׁתֵּי בַּתְרֵיהוֹן:
וְתֵימַר לְהוֹן כִּדְנַן אֲמַר יְיָ צְבָאוֹת אֱלָהָא דְיִשְׂרָאֵל שְׁתוֹ וּרְווֹ וְאִשְׁתַּגִישׁוּ וּנְפָלוּ וְלָא תְקוּמוּן מִן קֳדָם דְקַטְלִין בְּחַרְבָּא דַאֲנָא שְׁלַח בֵּינֵיכוֹן:
וִיהֵי אֲרֵי יְסָרְבוּן לְקַבָּלָא כַּסָא מִן יְדָךְ לְמִשְׁתֵּי וְתֵימַר לְהוֹן כִּדְנַן אֲמַר יְיָ צְבָאוֹת מִשְׁתָּא תִשְׁתּוּן:
אֲרֵי הָא בְקַרְתָּא דְאִתְקְרֵי שְׁמִי עֲלָהּ אֲנָא מַשְׁרֵי לְאַבְאָשָׁא וְאַתּוּן מְדַמָן דְמִזְכָּא תִזְכּוּן לָא תִזְכּוּן אֲרֵי דְקַטְלִין בְּחַרְבָּא אֲנָא מַכְלֵי עַל כָּל יַתְבֵי אַרְעָא אֲמַר יְיָ צְבָאוֹת:
וְאַתְּ תִּתְנַבֵּי לְהוֹן יַת כָּל פִּתְגָמַיָא הָאִלֵין וְתֵימַר לְהוֹן יְיָ מִמְרוֹמָא יַכְלֵי וּמִמְדוֹר קוּדְשֵׁיהּ יָרִים מֵימְרֵיהּ אַכְלָאָה יַכְלֵי דְיֵיתוּן בָּזוֹזִין עַל אֲרַע בֵּית שְׁכִנְתִּי הָא כְּנָחֲתֵי מַעֲצָרָא דִמְרִימִין קָלְהוֹן כֵּן יֵיתֵי זַעֲוָא לְכָל יַתְבֵי אַרְעָא:
מְטָא אִתְרְגוֹשָׁא עַד סְיָפֵי אַרְעָא אֲרֵי דִינָא קֳדָם יְיָ בְּעַמְמַיָא מִתְפְּרַע הוּא לְכָל בִּשְׂרָא רַשִׁיעַיָא דִבְּהוֹן מַסְרִינוּן לְקַטָלָא אֲמַר יְיָ:
כִּדְנַן אֲמַר יְיָ צְבָאוֹת הָא בִּישְׁתָא נַפְקָא מֵעַם לְעַם וְעַמְמִין רַבְרְבִין יֵיתוּן בִּגְלֵי מִסְיָפֵי אַרְעָא:
וִיהוֹן קְטִילַיָא רְמַן קֳדָם יְיָ בְּיוֹמָא הַהוּא מִסְיָפֵי אַרְעָא וְעַד סְיָפֵי אַרְעָא לָא יִסְתַּפְּדוּן וְלָא יִתְכַּנְשׁוּן וְלָא יִתְקַבְּרוּן לְזֵיבַל מְבַדֵר עַל אַפֵּי אַרְעָא יְהוֹן:
אֱלִילוּ מַלְכַיָא וּצְוָחוּ וְחוּפוּ רֵישֵׁיכוֹן בְּקִטְמָא תַּקִיפֵי עַמָא אֲרֵי שְׁלִימוּ יוֹמֵיכוֹן לְקַטְלָא וְתִתְבַּדְרוּן וְתִפְּלוּן דַהֲוֵיתוּן רְגִיגִין כְּמָנֵי חֶמְדָתָא:
וְיֵבַד בֵּית עֱרוֹקֵי מִן מַלְכַיָא וְשֵׁיזָבָא מִתַּקִיפֵי עַמָא:
קַל צִוְחַת מַלְכַיָא וִילְלַת תַּקִיפֵי עַמָא אֲרֵי אַבֵּז יְיָ יַת עַמְהוֹן:
וְיִצְדוֹן מְדוֹרֵי שְׁלָמְהוֹן מִקֳדָם תְּקוֹף רוּגְזָא דַיָי:
גְלָא מַלְכָּא מִכְּרַכֵּיהּ אֲרֵי הֲוַת אַרְעֲהוֹן לְצָדוּ מִן קֳדָם חֶרֶב סַנְאָה דְהִיא כַּחֲמַר מְרַוְיָא וּמִן קֳדָם תְּקוֹף רוּגְזֵיהּ:
