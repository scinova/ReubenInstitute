מַן אֲמַר אִלֵין עָתִיד לְאַיְתָאָה מָחָא עַל אֱדוֹם פּוּרְעָנָא תַקִיפָא עַל בָּצְרָה לְמֶעְבַּד פּוּרְעֲנוּת דִין עַמֵהּ כְּמָא דְקַיֵים לְהוֹן בְּמֵימְרֵהּ אֲמַר הָא אֲנָא מִתְגְלֵי כְּמָא דְמַלֵלִית בְּזָכוּ סַגְיָא קֳדָמַי חֵיל לְמִפְרַק:
מָא דֵין יִסְמְקוּן טוּרִין מִדַם קְטִילִין וּמֵישְׁרִין יִפְקוּן כַּחְמַר בְּמַעְצְרָא:
הָא כְּבִיעוּט דְמִתְבָּעֵיט בְּמַעְצְרָא כֵּן יִסְגֵי קְטוֹל בְּמַשִׁרְיַת עַמְמַיָא וְלָא יְהֵי לְהוֹן תְּקוֹף קֳדָמַי וְאֶקְטְלִינוּן בְּרוּגְזִי וַאֲדוּשִׁינוּן בְּחַמְתִי וְאֶתְבַּר תַּקִיפֵיהוֹן קֳדָמַי וְכָן חַכִּימֵיהוֹן אֱסַלְעִים:
אֲרֵי יוֹם פּוּרְעֲנוּתָא קֳדָמַי וּשְׁנַת פּוּרְקַן עַמִי מְטָת:
וּגְלֵי קֳדָמַי וְלֵית גְבַר דְלֵיהּ עוֹבָדִין טָבִין וִידִיַע קֳדָמַי וְלֵית אֱנַשׁ דִיקוּם וְיִבְעֵי עֲלֵיהוֹן וּפְרַקְתִּינוּן בִּדְרַע תּוּקְפִי וּבְמֵימַר רְעוּתִי סְעַדְתִּינוּן:
וַאֲקַטֵל עַמְמַיָא בְּרוּגְזִי וַאֲדוֹשְׁשִׁינוֹן בְּחֵימְתִי וְאֶרְמֵי לַאֲרַע אַרְעִיתָא קְטִילֵי גִבָּרֵיהוֹן:
אֲמַר נְבִיָא טִיבוּתָא דַייָ אֲנָא מַדְכַּר תּוּשְׁבַּחְתָּא דַייָ כְּעַל כָּל דְגַמְלָנָא יְיָ וְסַגִי טוּבֵהּ לְבֵית יִשְׂרָאֵל דְגַמְלִינוּן כְּרַחֲמוֹהִי וּכְסַגְיוּת טוּבֵהּ:
וַאֲמַר בְּרַם עַמִי אִינוּן בְּנַיָא דְלָא יְשַׁקְרוּן וַהֲוָה מֵימְרֵהּ לְהוֹן פָּרִיק:
בְּכָל עִדַן דְחָבוּ קֳדָמוֹהִי לְאַיְתָאָה עֲלֵיהוֹן עָקָא לָא אָעִיק לְהוֹן וּמַלְאַךְ שְׁלִיחַ מִן קֳדָמוֹהִי פְרֵיקִינוּן בְּרַחֲמָתֵהּ וּבִמְחָסֵהּ עֲלֵיהוֹן הָא שֵׁיזְבִינוּן וְנַטְלִינוּן וְסוֹבְרִינוּן כָּל יוֹמֵי עַלְמָא:
וְאִינוּן סָרִיבוּ וְאַרְגִיזוּ עַל מֵימַר נְבִיֵי קוּדְשֵׁהּ וְאִתְהַפֵּךְ מֵימְרֵהּ לְהוֹן לִבְעֵיל דְבַב הוּא אַגִיחַ קְרָבָא בְּהוֹן:
וְחָס עַל יְקַר שְׁמֵהּ בְּדִיל דוּכְרָן טַבְוָתֵהּ דְמִן עַלְמָא גְבוּרָן דַעֲבַד עַל יְדֵי מֹשֶׁה לְעַמֵהּ דִלְמָא יֵימַר אָן דְאַסְקִינוּן מִן יַמָא אָן דְדַבְּרִינוּן בְּמַדְבְּרָא כְּרַעֲיָא לְעָנֵהּ אָן דְאַשְׁרֵי בֵינֵיהוֹן יַת מֵימַר נְבִיֵי קוּדְשֵׁהּ:
דַבֵּר לְיַמִינָא דְמֹשֶׁה דְרַע תּוּשְׁבַּחְתֵּהּ בְּזַע מֵי יַמָא דְסוּף מִן קֳדָמֵיהוֹן לְמֶעְבַּד לֵיהּ שׁוּם עֲלָם:
דַבְּרִינוּן בֵּין תְּחוֹמַיָא כְּסוּסְיָא דִי בְמַדְבְּרָא לָא מַתְקַל כֵּן אַף אִינוּן לָא אִתְקִילוּ:
כִּבְעִירָא דִי בְּמֵישְׁרָא מַדְבְּרָא מֵימְרָא דַייָ דַבְּרִינוּן כֵּן דִבַּרְתֵּהּ לְעַמָךְ לְמֶעְבַּד לָךְ שׁוּם דְתוּשְׁבְּחָא:
אִסְתְּכֵי מִן שְׁמַיָא וְאִתְגְלֵי מִמְדוֹר קוּדְשָׁךְ וְתוּשְׁבַּחְתָּךְ אָן פּוּרְעֲנוּתָךְ וְגִבַּרְוָתָךְ הֲמוֹן טַבְוָתָךְ וְסַגְיוּת רַחֲמָךְ עֲלַי יֶחְסְנוּן:
אֲרֵי אַתְּ הוּא דְרַחֲמָךְ עֲלָנָא סַגִיאִין כְּאָב עַל בְּנִין אֲרֵי אַבְרָהָם לָא אַסְקָנָא מִמִצְרַיִם וְיִשְׂרָאֵל לָא עָבַד לָנָא פְּרִישָׁן בְּמַדְבְּרָא אַתְּ הוּא יְיָ וְרַחְמָךְ עֲלָנָא סַגִיאִין כְּאָב עַל בְּנִין פִּרְקְנָא מַעֲלַם שְׁמָךְ:
לְמָה תְרַחֲקִינָנָא יְיָ לְמִטְעֵי מֵאוֹרְחָן דְתַקְנָן קֳדָמָךְ כְּעַמְמַיָא דְלֵית לְהוֹן חוּלַק בְּאוּלְפַן אוֹרַיְתָךְ לָא יִתְפְּנֵי לִבָּנָא מִדְחַלְתָּךְ אֲתֵיב שְׁכִנְתָּךְ לְעַמָךְ בְּדִיל עַבְדָךְ צַדִיקַיָא דְקַיֵמְתָּא לְהוֹן בְּמֵימְרָךְ לְמֶעְבַּד בֵּינֵיהוֹן שִׁבְטֵי אַחְסַנְתָּךְ:
בִּזְעֵיר דִירִיתוּ עַמָא דְקוּדְשָׁךְ בַּעֲלֵי דְבָבָנָא דוֹשִׁישׁוּ מַקְדָשָׁךְ:
הָא אֲנַחְנָא עַמָךְ דְמִן עַלְמָא לָא לְעַמְמַיָא יְהַבְתְּ אוּלְפַן אוֹרַיְתָךְ לָא אִתְקְרֵי שְׁמָךְ עֲלֵיהוֹן לָא לְהוֹן אַרְכֵינְתָּא שְׁמַיָא אִתְגְלֵיתָא מִן קֳדָמָךְ טוּרַיָא זָעוּ:
