אִתְגְלָא אִתְגְלָא לִבְשִׁי תוּקְפֵךְ צִיוֹן לִבְשִׁי לְבוּשִׁי תּוּשְׁבַּחְתִּיךְ יְרוּשָׁלַםִ קַרְתָּא דְקוּדְשָׁא אֲרֵי לָא יוֹסְפוּן וְיֶעְדוּן בִּיךְ עוֹד עָרְלִין וּמְסַאֲבִין:
אִתְנַפַּצִי מֵעָפְרָא קוּמִי תְבִי עַל כּוּרְסֵי יְקָרָא יְרוּשְׁלֵם אִתְפְּסָקוּ חִנְקֵי צַוְרֵיכוֹן שְׁבִיָא כְּנִשְׁתָּא דְצִיוֹן:
אֲרֵי כִדְנָן אֲמַר יְיָ מַגָן אִזְדַבִּנְתּוּן וְלָא בִכְסַף תִּתְפַּרְקוּן:
אֲרֵי כִדְנַן אֲמַר יְיָ אֱלֹהִים לְמִצְרַיִם נְחַת עַמִי בְקַדְמֵיתָא לְאִתּוֹתָבָא תַמָן וְאַתּוּרָאָה בִּלְמָא אַנְסֵיהּ:
וּכְעַן עֲתִידְנָא לְמִפְרַק אֲמַר יְיָ אֲרֵי אִזְדַבַּן עַמִי מַגָן עַמְמַיָא דִשְׁלִיטוּ בְהוֹן מִשְׁתַּבְּחִין אֲמַר יְיָ וּתְדִירָא כָּל יוֹמָא עַל פּוּלְחָן שְׁמִי מַרְגְזִין:
בְּכֵן יִתְרַבָּא בְעַמְמַיָא שְׁמִי בְּכֵן בְּעִידָנָא הַהִיא תֵּידְעוּן אֲרֵי אֲנָא הוּא דְמַלְלֵית וּמֵימְרִי קַיָם:
מַה יָאֲוָן עַל טוּרֵי אַרְעָא דְיִשְׂרָאֵל רִגְלֵי מְבַסֵר מַשְׁמַע שְׁלָם מְבַסֵר טַב מַשְׁמַע פּוּרְקָן אֲמַר לִכְנִשְׁתָּא דְצִיוֹן אִיתְגְלִיאַת מַלְכוּתָא דֶאֱלָהָיִךְ:
קַל פַּרְנָסַיִךְ דִמְרִימִין קָלְהוֹן כַּחֲדָא מְשַׁבְּחִין אֲרֵי בְעֵינֵיהוֹן יֶחֱזוּן גְבוּרָן דִי עֲבַד יְיָ כַּד יְתִיב שְׁכִנְתֵּהּ לְצִיוֹן:
בּוּעוּ וְשַׁבַּחוּ כַּחֲדָא חָרְבַת יְרוּשְׁלֵם אֲרֵי עֲתִיד יְיָ לְנַחְמָא עַמֵהּ פָּרִיק יְרוּשְׁלֶם:
גַלֵא יְיָ יַת דְרַע קוּדְשֵׁהּ לְעֵינֵי כָּל עַמְמַיָא וְיֶחֱזוּן כָּל דְבִסְיָפֵי אַרְעָא יַת פּוּרְקָנָא דֶאֱלָהָנָא:
אִתְפְּרַשׁוּ אִתְפְּרַשׁוּ פּוּקוּ מִתַּמָן בִּמְסָאָב לָא תִקְרְבוּן פּוּקוּ מִגַוָהּ אִתְבְּחָרוּ נַטְלֵי מָנֵי בֵית מַקְדְשָׁא דַיָי:
אֲרֵי לָא בִּבְהִילוּ תִּפְּקוּן מִבֵּינֵי עַמְמַיָא וּבְעֵירוֹקֵי לָא תִתּוֹבְלוּן לְאַרְעֲכוֹן אֲרֵי מְדַבֵּר קֳדָמֵיכוֹן יְיָ וַעֲתִיד לְכַנָשָׁא גַלְוַתְכוֹן אֱלָהָא דְיִשְׂרָאֵל:
הָא יַצְלַח עַבְדִי מְשִׁיחָא יְרוּם וְיִסְגֵי וְיִתְקוֹף לַחֲדָא:
כְּמָא דִסְבַרוּ לֵיהּ בֵּית יִשְׂרָאֵל יוֹמִין סַגִיאִין דַהֲוָה חֲשִׁיךְ בֵּינֵי עַמְמַיָא חֶזְוֵהוֹן וְזִיוְהוֹן מִבְּנֵי אֱנָשָׁא:
כֵּן יְבַדַר עַמְמִין סַגִיאִין עֲלוֹהִי יִשְׁתְּקוּן מַלְכִין יְשַׁווּן יְדֵיהוֹן עַל פּוּמְהוֹן אֲרֵי דְלָא אִשְׁתָּעִיאוּ לְהוֹן חֲזוֹ וּדְלָא שְׁמָעוּ אִסְתַּכֵּלוּ:
