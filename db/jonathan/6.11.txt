וַהֲוָה כַּד שְׁמַע יָבִין מַלְכָּא דְחָצוֹר וּשְׁלַח לְוַת יוֹבָב מַלְכָּא דְמָדוֹן וּלְוַת מַלְכָּא דְשִׁמְרוֹן וּלְוַת מַלְכָּא דְאַכְשָׁף:
וּלְוַת מַלְכַיָא דְמִצְפּוּנָא בְּטוּרָא וּבְמֵישְׁרָא דָרוֹם גִינוֹסָר וּבִשְׁפֶלְתָּא וּבְפִלְכֵי דוֹר מַעֲרָבָא:
כְּנַעֲנָאָה מִמַדִינְחָא וּמִמַעֲרָבָא וֶאֱמוֹרָאָה וְחִתָּאָה וּפְרִיזָאָה וִיבוּסָאָה בְּטוּרָא וְחִוָאָה בְּשִׁפּוּלֵי חֶרְמוֹן בַּאֲרַע מִצְפַּיָא:
וּנְפָקוּ אִינוּן וְכָל מַשִׁרְיַתְהוֹן עִמְהוֹן עַם סַגִי כְּחָלָא דְעַל כֵּיף יַמָא לִסְגֵי וְסוּסְוַן וּרְתִיכוּן סַגִי לַחֲדָא:
וְאִזְדַמְנוּ כֹּל מַלְכַיָא הָאִילֵין וַאֲתוֹ וּשְׁרוֹ כַחְדָא עַל מֵי מֵרוֹם לְאַגָחָא קְרָבָא עִם יִשְׂרָאֵל:
וַאֲמַר יְיָ לִיהוֹשֻׁעַ לָא תִדְחַל מִן קֳדָמֵיהוֹן אֲרֵי מְחַר כְּעִידָנָא הָדֵין אֲנָא מְסַר יַת כּוּלְהוֹן קְטִילִין קֳדָם יִשְׂרָאֵל יַת סוּסְוָתְהוֹן תְּעַקֵר וְיַת רְתִיכֵיהוֹן תּוֹקֵד בְּנוּרָא:
וַאֲתָא יְהוֹשֻׁעַ וְכָל עַמָא עָבְדֵי קְרָבָא עִמֵיהּ עֲלֵיהוֹן וְאִינוּן שָׁרָן עַל מֵי מֵרוֹם בִּתְכֵיף וּנְפַלוּ בְהוֹן:
וּמַסְרִינוּן יְיָ בִּידָא דְיִשְׂרָאֵל וּמְחוּנוּן וּרְדָפוּנוּן עַד צִידוֹן רַבְּתָא וְעַד חַרְצֵי יַמָא וְעַד בִּקְעַת מִצְפַּיָא מַדִינְחָא וּמְחוּנוּן עַד דְלָא אִישְׁתָּאַר לְהוֹן מְשֵׁזִיב:
וַעֲבַד לְהוֹן יְהוֹשֻׁעַ כְּמָא דַאֲמַר לֵיהּ יְיָ יַת סוּסְוָתְהוֹן עִקַר וְיַת רְתִיכֵיהוֹן אוֹקֵד בְּנוּרָא:
וְתַב יְהוֹשֻׁעַ בְּעִדָנָא הַהִיא וּכְבַשׁ יַת חָצוֹר וְיַת מַלְכָּהּ קְטַל בְּחַרְבָּא אֲרֵי חָצוֹר מִלְקַדְמִין הִיא רֵישׁ כָּל מַלְכְּוָתָא הָאִלֵין:
וּמְחוֹ יַת כָּל נַפְשָׁתָא דִי בָהּ לְפִתְגַם דְחָרֶב גְמִירָא לָא אַשְׁאַר כָּל נִשְׁמָא וְיַת חָצוֹר אוֹקֵד בְּנוּרָא:
וְיַת כָּל קִרְוֵי מַלְכַיָא הָאִילֵין וְיַת כָּל מַלְכֵיהוֹן כְּבַשׁ יְהוֹשֻׁעַ וּמְחָנוּן לְפִתְגַם דְחָרֶב גַמַר יַתְהוֹן כְּמָא דְפַקֵד מֹשֶׁה עַבְדָא דַיָי:
לְחוֹד כָּל קִרְוַיָא דְקַיְמִין עַל תּוּקְפֵיהוֹן לָא אוֹקְדִינוּן יִשְׂרָאֵל אֱלָהֵין יַת חָצוֹר בִּלְחוֹדָהּ אוֹקֵיד יְהוֹשֻׁעַ:
וְכָל עֲדֵי קִרְוַיָא הָאִלֵין וּבְעִירָא בְּזוֹ לְהוֹן בְּנֵי יִשְׂרָאֵל לְחוֹד יַת כָּל אֱנָשָׁא מְחוֹ לְפִתְגַם דְחָרֶב עַד דִי שֵׁיצִיאוּ יַתְהוֹן לָא אַשְׁאָרוּ כָּל נִשְׁמָא:
כְּמָא דְפַקֵיד יְיָ יַת מֹשֶׁה עַבְדֵיהּ כֵן פַּקֵד מֹשֶׁה יַת יְהוֹשֻׁעַ וְכֵן עֲבַד יְהוֹשֻׁעַ לָא בַטֵיל פִּתְגָמָא מִכָּל דְפַקֵיד יְיָ יַת מֹשֶׁה:
וְאַחְסִין יְהוֹשֻׁעַ יַת כָּל אַרְעָא הָדָא טוּרָא וְיַת כָּל דָרוֹמָא וְיַת כָּל אַרְעָא דְגוֹשֶׁן וְיַת שְׁפֶלְתָּא וְיַת מֵישְׁרָא וְיַת טוּרָא דְיִשְׂרָאֵל וּשְׁפֵלָתֵיהּ:
מִן טוּרָא פְּלִיגָא דְסָלִיק לְשֵׂעִיר וְעַד מֵישַׁר גָד בְּבִקְעַת לִבְנָן שְׁפּוּלֵי טוּרָא דְחֶרְמוֹן וְיַת כָּל מַלְכֵיהוֹן כְּבַשׁ וּמְחָנוּן וְקַטְלִינוּן:
יוֹמִין סַגִיאִין עֲבַד יְהוֹשֻׁעַ יַת כָּל מַלְכַיָא הָאִלֵין קְרָבָא:
לָא הֲוַת קַרְתָּא דְאַשְׁלֵימַת עִם בְּנֵי יִשְׂרָאֵל אֱלָהֵן חִוָאָה יַתְבֵי גִבְעוֹן יַת כּוֹלָא נְסִיבוּ בִּקְרָבָא:
אֲרֵי מִן קֳדָם יְיָ הֲוַת לְתַקָפָא יַת לִבְּהוֹן לְסַדָרָא קְרָבָא עִם יִשְׂרָאֵל בְּדִיל לְגַמָרוּתְהוֹן בְּדִיל דְלָא לְמֶהֱוֵי לְהוֹן רַחֲמִין אֲרֵי בְּדִיל לְשֵׁצָיוּתְהוֹן כְּמָא דְפַקֵד יְיָ יַת מֹשֶׁה:
וַאֲתָא יְהוֹשֻׁעַ בְּעִדָנָא הַהִיא וְשֵׁיצֵי יַת גִבָּרַיָא מִן טוּרָא מִן חֶבְרוֹן מִן דְבִיר מִן עֲנָב וּמִכֹּל טוּרָא דִיהוּדָה וּמִכֹּל טוּרָא דְיִשְׂרָאֵל עִם קִרְוֵיהוֹן גַמְרוּנוּן יְהוֹשֻׁעַ:
לָא אִשְׁתָּאַר גִבָּרַיָא בַּאֲרַע בְּנֵי יִשְׂרָאֵל לְחוֹד בְּעַזָה בְּגַת וּבְאַשְׁדוֹד אִשְׁתָּאֲרוּ:
וּנְסִיב יְהוֹשֻׁעַ יַת כָּל אַרְעָא כְּכֹל דִי מַלֵיל יְיָ עִם מֹשֶׁה וְיַהֲבָהּ יְהוֹשֻׁעַ לְאַחֲסָנָא לְיִשְׂרָאֵל כְּפַלְגוּתֵיהוֹן לְשִׁבְטֵיהוֹן וְאַרְעָא שָׁדוֹכָת מִלְמֶעְבַּד קְרָבָא:
