וִידַע יוֹאָב בַּר צְרוּיָה אֲרֵי לִבָּא דְמַלְכָּא לְמִפַּק עַל אַבְשָׁלוֹם:
וּשְׁלַח יוֹאָב לִתְקוֹעַ וּנְסֵיב מִתַּמָן אִתְּתָא חֲכִּמְתָּא וַאֲמַר לָהּ אִתְאַבְּלֵי כְעַן וּלְבוּשִׁי כְעַן לְבוּשֵׁי אֶבְלָא וְלָא תְשׁוּפִין מִשְׁחָא וּתְהֵא כְּאִתְּתָא דְנַן יוֹמִין סַגִיאִין מִתְאַבְּלָא עַל מֵיתָא:
וְתַהֲכִין לְוַת מַלְכָּא וּתְמַלְלִין עִמֵיהּ כְּפִתְגָמָא הָדֵין וְשַׁוַי יוֹאָב יַת פִּתְגָמַיָא בְּפוּמָהּ:
וַאֲמָרַת אִתְּתָא דְמִתְּקוֹעַ לְמַלְכָּא וּנְפָלַת עַל אַפָּהּ עַל אַרְעָא וּסְגִידַת וַאֲמָרַת פְּרוֹק מַלְכָּא:
וַאֲמַר לָהּ מַלְכָּא מַה לִיךְ וַאֲמָרַת בְּקוּשְׁטָא אִתְּתָא אַרְמַלְתָּא אֲנָא וּמִית בַּעֲלִי:
וּלְאַמְתָךְ תְּרֵין בְּנִין וְאִתְנְצִיאוּ תַרְוֵיהוֹן בְּחַקְלָא וְלֵית דִמְשֵׁיזֵיב בֵּינֵיהוֹן וּמְחוֹ חַד יַת חַד וּקְטַל יָתֵיהּ:
וְהָא קָמַת כָּל זַרְעִיתָא עַל אַמְתָךְ וַאֲמַרוּ הָבִי וְנִקְטוֹל יַת דִקְטַל אֲחוֹהִי וְנִקְטְלִינֵיהּ בְּחוֹבַת נַפְשָׁא דַאֲחוֹהִי דִקְטַל וּנְשֵׁיצֵי אַף יַת יָרוֹתָא וּבְעוֹ לְמִקְטַל יַת חַד יְחִידִי דְאִשְׁתְּאַר בְּדִיל דְלָא לְשַׁוָאָה לְבַעֲלֵי שׁוּם וּשְׁאָר עַל אַפֵּי אַרְעָא:
וַאֲמַר מַלְכָּא לְאִתְּתָא אֱזִילִי לְבֵיתָךְ וַאֲנָא אֲפַקֵיד עֲלָךְ:
וַאֲמַרַת אִתְּתָא דְמִתְּקוֹעַ לְמַלְכָּא עֲלַי דִי לִי רִבּוֹנִי מַלְכָּא חוֹבָא וְעַל בֵּית אַבָּא וּמַלְכָּא וְכוּרְסֵי מַלְכוּתֵיהּ זַכַּאי:
וַאֲמַר מַלְכָּא מַן דְיֵימַר לִיךְ מִלָא וְתַיְתִינֵיהּ לְוָתִי וְלָא יוֹסִיף עוֹד לְאַנְזָקָה בִיךְ:
וַאֲמַרַת יִדְכַּר כְּעַן מַלְכָּא מַה דִכְתִיב בִּסְפַר אוֹרַיְתָא דַייָ אֱלָהָךְ בְּדִיל לְאַסְגָאָה אוֹרְחָא קֳדָם גָאֵל דְמָא לְחַבָּלָא וְלָא יְשֵׁיצוֹן יַת בְּרִי וַאֲמַר קַיָם הוּא יְיָ אִם יִפּוֹל מִסְעַר רֵישׁ בְּרִיךְ לְאַרְעָא:
וַאֲמַרַת אִתְּתָא תְּמַלֵל כְּעַן אַמְתָךְ קֳדָם רִבּוֹנִי מַלְכָּא פִתְגָמָא וַאֲמַר מַלֵילִי:
וַאֲמַרַת אִתְּתָא וּלְמָה חֲשַׁבְתָּא כַּהֲדָא עַל עַמָא דַייָ וּמְמַלֵיל מַלְכָּא פִתְגָמָא הָדֵין כִּגְבַר חַיָב בְּדִיל דְלָא לְאַתָבָא מַלְכָּא יַת בִּדוּרֵיהּ:
אֲרֵי מִיתָא דְמָאִית וּכְמַיָא דְמִתְאַשְׁדִין לְאַרְעָא דְלָא אֶפְשַׁר לְהוֹן דְיִתּוֹסְפוּן כֵּן לֵית אֶפְשַׁר לְדַיָנָא דְקוּשְׁטָא לְקַבָּלָא מָמוֹן דִשְׁקַר וְדִמְחַשֵׁב מֵחְשְׁבָן בְּדִיל דְלָא לְבָדָרָא מִנֵיהּ בִּדוּרָא:
וּכְעַן דַאֲתֵיתִי לְמַלָלָא קֳדָם מַלְכָּא רִבּוֹנִי יַת פִּתְגָמָא הָדֵין אֲרֵי דַחֲלוּנַנִי עַמָא וַאֲמַרַת אַמְתָךְ אֲמַלֵיל כְּעַן קֳדָם מַלְכָא מְאִים יַעְבֵּד מַלְכָּא יַת פִּתְגָמָא דְאַמְתֵיהּ:
אֲרֵי יִשְׁמַע מַלְכָּא לְשֵׁיזָבָא יַת אַמְתֵיהּ מִיַד גַבְרָא לְשֵׁיצָאָה יָתִי וְיַת בְּרִי כַּחֲדָא מֵאַחֲסָנַת עַמָא דַיָי:
וַאֲמַרַת אַמְתָךְ יְהֵי כְעַן פִּתְגָמָא דְרִבּוֹנִי מַלְכָּא לְנִיחָא אֲרֵי כְּמַלְאָכָא דַייָ כֵּן רִבּוֹנִי מַלְכָּא לְמִשְׁמַע טַב וּבִישׁ וּמֵימְרָא דַייָ אֱלָהָךְ יְהֵי בְסַעֲדָךְ:
וַאֲתֵיב מַלְכָּא וַאֲמַר לְאִתְּתָא לָא כְעַן תְּכַסֵי מִנִי פִתְגָמָא דַאֲנָא שְׁאֵל יָתִיךְ וַאֲמַרַת אִתְּתָא יְמַלֵיל כְּעַן רִבּוֹנִי מַלְכָּא:
וַאֲמַר מַלְכָּא הֲיַד יוֹאָב עִמָךְ בְּכָל דָא וַאֲתִיבַת אִתְּתָא וַאֲמַרַת חַיֵי נַפְשָׁךְ רִבּוֹנִי מַלְכָּא אִם אִית לְמִסְטֵי לְיַמִינָא וְלִסְמָלָא מִכֹּל דִי מַלֵיל רִבּוֹנִי מַלְכָּא אֲרֵי עַבְדָךְ יוֹאָב הוּא פַקְדַנִי וְהוּא שַׁוִי בְּפוּמָא דְאַמְתָךְ יַת כָּל פִּתְגָמַיָא הָאִלֵין:
בְּדִיל לְאַקָפָא יַת אַפֵּי פִתְגָמָא עֲבַד עַבְדָךְ יוֹאָב יַת פִּתְגָמָא הָדֵין וְרִבּוֹנִי חַכִּים כְּחָכְמַת מַלְאָכָא דַייָ לְמִידַע יַת כָּל דִי בְאַרְעָא:
וַאֲמַר מַלְכָּא לְיוֹאָב הָא כְעַן עֲבָדִית יַת פִּתְגָמָא הָדֵין וֶאֱזֵיל אֲתֵיב יַת עוּלֵימָא יַת אַבְשָׁלוֹם:
וּנְפַל יוֹאָב עַל אַפּוֹהִי לְאַרְעָא וּסְגִיד וּבָרִיךְ יַת מַלְכָּא וַאֲמַר יוֹאָב יוֹמָא דֵין יְדַע עַבְדָךְ אֲרֵי אַשְׁכָּחִית רַחֲמִין בְּעֵינָךְ רִבּוֹנִי מַלְכָּא דַעֲבַד מַלְכָּא יַת פִּתְגָמָא דְעַבְדֵיהּ:
וְקָם יוֹאָב וַאֲזַל לִגְשׁוּר וְאַיְתֵי יַת אַבְשָׁלוֹם לִירוּשְׁלֵם:
וַאֲמַר מַלְכָּא יִסְתְּחַר לְבֵיתֵיהּ וְאַפֵּי לָא יֶחְזֵי וְאִסְתְּחַר אַבְשָׁלוֹם לְבֵיתֵיהּ וְאַפֵּי מַלְכָּא לָא חֲזָא:
וּכְאַבְשָׁלוֹם לָא הֲוָה גְבַר שַׁפִּיר בְּכָל יִשְׂרָאֵל לְשָׁבָּחָא לַחֲדָא מִפַּרְסַת רִגְלוֹהִי וְעַד מוֹחֵיהּ לָא הֲוָה בֵיהּ מוּמָא:
וּבְסַפָּרוּתֵיהּ יַת רֵישֵׁיהּ וַהֲוָה מִזְמַן עִדַן לְעִדַן דִמְסַפֵר אֲרֵי יַקִיר עֲלוֹהִי וּמְסַפֵּר לֵיהּ וּתְקַל יַת סְעַר רֵישֵׁיהּ מָאתָן סִלְעִין בְּמַתְקְלָא דְמַלְכָּא:
וְאִתְיְלִידוּ לְאַבְשָׁלוֹם תְּלָתָא בְנִין וּבְרַתָּא חֲדָא וּשְׁמָהּ תָּמָר הִיא הֲוַת אִתְּתָא שַׁפִּירַת חֵיזוּ:
וִיתֵב אַבְשָׁלוֹם בִּירוּשְׁלֵם תַּרְתֵּין שְׁנִין וְאַפֵּי מַלְכָּא לָא חֲזָא:
וּשְׁלַח אַבְשָׁלוֹם לְוַת יוֹאָב לְשַׁלְחָא יָתֵיהּ לְוַת מַלְכָּא וְלָא אֲבָה לְמֵיתֵי לְוָתֵיהּ וּשְׁלַח עוֹד תִּנְיָנוּת וְלָא אֲבָה לְמֵיתֵי:
וַאֲמַר לְעַבְדוֹהִי חֲזוֹ אַחְסַנְתֵּיהּ דְיוֹאָב דִסְמִיכָא לְדִילִי וְלֵיהּ תַּמָן סְעוֹרִין אֱזִילוּ וְאַדְלְקוּהָ בְּנוּרָא וְאַדְלִיקוּ עַבְדֵי אַבְשָׁלוֹם יַת אַחְסַנְתֵּיהּ בְּנוּרָא:
וְקָם יוֹאָב וַאֲתָא לְוַת אַבְשָׁלוֹם לְבֵיתָא וַאֲמַר לֵיהּ לְמָה אַדְלִיקוּ עַבְדָיִךְ יַת אַחְסַנְתָּא דִילִי בְּנוּרָא:
וַאֲמַר אַבְשָׁלוֹם לְיוֹאָב הָא שְׁלָחִית לְוָתָךְ לְמֵימָר אִיתָא הַלְכָא וְאִשְׁלְחִינָךְ לְוַת מַלְכָּא לְמֵימַר לְמָה אָתֵיתִי מִגְשׁוּר טַב לִי עוֹד דַאֲנָא תַמָן וּכְעַן אֶחֱזֵי אַפֵּי מַלְכָּא וְאִם אִית בִּי חוֹבָא וְיִקְטְלִינַנִי:
וַאֲתָא יוֹאָב לְוַת מַלְכָּא וְחַוִי לֵיהּ וּקְרָא לְאַבְשָׁלוֹם וַאֲתָא לְוַת מַלְכָּא וּסְגִיד לֵיהּ עַל אַפּוֹהִי עַל אַרְעָא קֳדָם מַלְכָּא וּנְשַׁק מַלְכָּא לְאַבְשָׁלוֹם:
