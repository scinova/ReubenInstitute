וּשְׁלַח יְיָ יַת נָתָן לְוַת דָוִד וַאֲתָא לְוָתֵיהּ וַאֲמַר לֵיהּ תְּרֵין גַבְרִין הֲווֹ בְקַרְתָּא חָדָא חַד עֲתִּירָא וְחַד מִסְכֵּינָא:
לְעַתִּירָא הֲווֹ עָאן וְתוֹרִין סַגִי לַחֲדָא:
וּלְמִסְכֵּינָא לֵית כָּל מִדָעַם אֱלָהֵין אִמַרְתָּא חֲדָא זְעֵירָא דְזַבְּנָא וְקַיְימָהּ וּרְבַת עִמֵיהּ וְעִם בְּנוֹהִי כַּחֲדָא מִפִּתֵּיהּ אֲכָלַת וּמִכּוֹסֵיהּ שַׁתְיָא וּבְחֵיקֵיהּ שָׁכְבָא וַהֲוַת לֵיהּ כִּבְרַתָּא:
וַאֲתָא אוֹרְחָא לְגַבְרָא עַתִּירָא וְחַס לְמֵיסַב מֵעָנֵיהּ וּמִתּוֹרֵיהּ לְמֶעְבַּד לְאוֹרְחָא דַאֲתָא לֵיהּ וּנְסֵיב יַת אִמַרְתָּא דְגַבְרָא מִסְכֵּינָא וְעַבְדָהּ לְגַבְרָא דַאֲתָא לְוָתֵיהּ:
וּתְקֵיף רוּגְזָא דְדָוִד בְּגַבְרָא לַחֲדָא וַאֲמַר לְנָתָן קַיָים הוּא יְיָ אֲרֵי גְבַר חַיַב קְטוֹל גַבְרָא דַעֲבַד דָא:
וְיַת אִמַרְתָּא יְשַׁלֵם עַל חַד אַרְבְּעָא חֲלַף דַעֲבַד יַת פִּתְגָמָא הָדֵין וְעַל דְלָא חָס:
וַאֲמַר נָתָן לְדָוִד אַתְּ גַבְרָא כִּדְנַן אֲמַר יְיָ אֱלָהָא דְיִשְׂרָאֵל אֲנָא מְשַׁחְתָּךְ לְמֶהֱוֵי מַלְכָּא עַל יִשְׂרָאֵל וַאֲנָא שֵׁיזַבְתָּךְ מִידָא דְשָׁאוּל:
וִיהָבֵית לָךְ יַת בֵּית רִבּוֹנָךְ וְיַת נְשֵׁי רִבּוֹנָךְ בְּחֵיקָךְ וִיהָבֵית לָךְ יַת בֵּית יִשְׂרָאֵל וִיהוּדָה וְאִם זְעֵיר וְאוֹסִיף לָךְ כְּאִלֵין וּכְאִלֵין:
מָא דֵין בָּסַרְתָּא עַל פִּתְגָמָא דַייָ לְמֶעְבַּד דְבִישׁ קֳדָמוֹהִי יַת אוּרִיָה חִתָּאָה קְטַלְתָּא בְּחַרְבָּא וְיַת אִתְּתֵיהּ נְסִיבְתָּא לָךְ לְאִתּוּ וְיָתֵיהּ קְטַלְתָּא בְחַרְבָּא דִבְנֵי עַמוֹן:
וּכְעַן לָא תִפְסוֹק חַרְבָּא מִן אֱנַשׁ בֵּיתָךְ עַד עַלְמָא חֲלָף דַאֲשַׁטְתַּנִי וּנְסֵבְתָּא יַת אִתַּת אוּרִיָה חִתָּאָה לְמֶהֱוֵי לָךְ לְאִתּוּ:
כִּדְנַן אֲמַר יְיָ הָא אֲנָא מְקַיֵים עֲלָךְ בִּישָׁא מִבֵּיתָךְ וְאֶדְבַר יַת נְשָׁךְ לְעֵינָךְ וְאֶתֵּן לְחַבְרָךְ וְיִשְׁכּוֹב עִם נְשָׁךְ לְעֵינֵי שִׁמְשָׁא הָדָא:
אֲרֵי אַתְּ עֲבַדְתָּא בְסִתְרָא וַאֲנָא אַעְבֵּיד יַת פִּתְגָמָא הָדֵין קֳדָם כָּל יִשְׂרָאֵל וְקָבֵיל שִׁמְשָׁא:
וַאֲמַר דָוִד לְנָתָן חָבִית קֳדָם יְיָ וַאֲמַר נָתָן לְדָוִד אַף יְיָ אַעְבֵּר חוֹבָךְ לָא תְמוּת:
בְּרַם אֲרֵי מִפְתַּח פְּתַחְתָּא פּוּמָא דְסַנְאֵי עַמָא דַייָ בְּפִתְגָמָא הָדֵין אַף בְּרָא דְאִתְיְלִיד לָךְ מֵמַת יְמוּת:
וַאֲזַל נָתָן לְבֵיתֵיהּ וּתְבַר יְיָ יַת רַבְיָא דִילִידַת אִתַּת אוּרִיָה לְדָוִד וְאִטְעַן:
וּבְעָא דָוִד רַחֲמִין מִן קֳדָם יְיָ עַל רַבְיָא וְצָם דָוִד צוֹמָא וְעַל וּבָת וּשְׁכִיב עַל אַרְעָא:
וְקָמוּ סָבֵי בֵיתֵיהּ עֲלוֹהִי לַאֲקָמוּתֵיהּ מִן אַרְעָא וְלָא אֲבָא וְלָא אֲכָל עִמְהוֹן לַחְמָא:
וַהֲוָה בְּיוֹמָא שְׁבִיעָאָה וּמִית רַבְיָא וּדְחִילוּ עַבְדֵי דָוִד לְחַוָאָה לֵיהּ אֲרֵי מִית רַבְיָא אֲרֵי אֲמַרוּ כַּד הֲוָה רַבְיָא קַיָם מַלֵילְנָא עִמֵיהּ וְלָא קַבֵּיל מִנָנָא וְאֵיכְדֵין נֵימַר לֵיהּ מִית רַבְיָא וְיַעְבֵּד בִּישָׁא:
וַחֲזָא דָוִד אֲרֵי עַבְדוֹהִי מְלַחֲשִׁין וּסְבַר דָוִד אֲרֵי מִית רַבְיָא וַאֲמַר דָוִד לְעַבְדוֹהִי הֲמִית רַבְיָא וַאֲמַרוּ מִית:
וְקָם דָוִד מִן אַרְעָא וּמְשַׁח וְשַׁף וְשַׁנֵי כְּסוּתֵיהּ וְעַל לְבֵית מַקְדְשָׁא דַייָ וּסְגִיד וַאֲתָא לְבֵיתֵיהּ וּשְׁאֵיל וְשַׁוִיאוּ לֵיהּ לַחְמָא וַאֲכַל:
וַאֲמַרוּ עַבְדוֹהִי לֵיהּ מַה פִתְגָמָא הָדֵין דַעֲבַדְתָּא עַד דְרַבְיָא קַיָם צַמְתָּא וּבְכֵית וְכַד מִית רַבְיָא קַמְתָּא וַאֲכַלְתְּ לַחְמָא:
וַאֲמַר עַד דְרַבְיָא קַיָם צָמֵית וּבְכֵיתִי אֲרֵי אֲמַרִית מַן יְדַע דִלְמָא יִתְרְחֵים עֲלַי מִן קֳדָם יְיָ וְיֵחִי רַבְיָא:
וּכְעַן מִית לְמָא דְנַן אֲנָא צָאֵים הַאֵיכוּל לַאֲתָבוּתֵיהּ עוֹד אֲנָא אֵיזֵיל לְוָתֵיהּ וְהוּא לָא תָאֵב לְוָתִי:
וְנָחֵים דָוִד יַת בַּת שֶׁבַע אִתְּתֵיהּ וְעַל לְוָתָהּ וּשְׁכִיב עִמָהּ וִילֵידַת בַּר וּקְרַת יַת שְׁמֵיהּ שְׁלֹמֹה וַיָי רַחְמֵיהּ:
וּשְׁלַח בְּיַד נָתָן נְבִיָא וּקְרָא יַת שְׁמֵיהּ יְדִידְיָה בְּדִיל יְיָ:
וְאַגִיחַ קְרָבָא יוֹאָב בְּרַבַּת בְּנֵי עַמוֹן וּכְבַשׁ יַת קִרְיַת מַלְכוּתָא:
וּשְׁלַח יוֹאָב אִזְגַדִין לְוַת דָוִד וַאֲמַר אַגָחִית קְרָבָא בְרַבָּה וְאַף כְּבָשִׁית יַת קִרְיַת מַלְכוּתָא:
וּכְעַן כְּנוֹשׁ יַת שְׁאַר עַמָא וּשְׁרֵי עַל קַרְתָּא וְכַבְּשָׁהּ דִלְמָא אֶכְבּוֹשׁ אֲנָא יַת קַרְתָּא וְיִתְקְרִי שְׁמִי עֲלָהּ:
וּכְנַשׁ דָוִד יַת כָּל עַמָא וַאֲזַל לְרַבַּת וְאַגִיחַ קְרָבָא בָהּ וְכַבְּשָׁהָא:
וּנְסִיב יַת כְּלִילָא דְמַלְכְּהוֹן מֵעַל רֵישֵׁיהּ וּמַתְקְלָהּ כִּכְּרָא דְדַהֲבָא וּבָהּ אֲבַן טָבָא וַהֲוָה עַל רֵישָׁא דְדָוִד וַעֲדַת קַרְתָּא אַפֵיק סַגִי לַחֲדָא:
וְיַת עַמָא דְבָהּ אַפֵּיק וּמְסַר יַתְהוֹן בִּמְסָרִין וּבְמוֹרִיגֵי פַרְזְלָא וּבְמִגְזְרֵי פַרְזְלָא וּגְרַר יַתְהוֹן בְּשׁוּקַיָא וְכֵן עֲבַד לְכָל קִירְוֵי בְנֵי עַמוֹן וְתַב דָוִד וְכָל עַמָא לִירוּשְׁלֵם:
