גְלוֹ בְּנֵי בִּנְיָמִן מִגוֹ יְרוּשְׁלֵם אֲרֵי קַל צְוָחַת קַרְנָא אִשְׁתְּמָעַת בְּאַרְעָא וְעַל בֵּית בִּקְעַת כַּרְמַיָא זְקוּפוּ עַמְמַיָא סִמְוָתָא אֲרֵי בִישְׁתָא אִסְתְּכִיאַת מִצִפּוּנָא וּתְבִירָא רַבָּא:
יָאֲתָה וּמְפַנַקְתָּא אֵיכְדֵין קַלְקֵילְתָּא יַת אוֹרְחָתַיִךְ בְּכֵן בְּהִיתַת כְּנִשְׁתָּא דְצִיוֹן:
לָהּ יֵיתוּן מַלְכִין וּמַשִׁרְיַתְהוֹן יִגְדוּן עֲלָהּ מַשְׁכְּנִין סְחוֹר סְחוֹר וְסַיְעִין גְבַר יַת חַבְרֵיהּ:
זַמִינוּ עֲלָהּ עָבְדֵי קְרָבָא קוּמוּ וְנִסַק בְּטִיהֲרָא וַי לָנָא אֲרֵי פְנָא יוֹמָא אֲרֵי נְגָדוּ טוּלֵי רַמְשָׁא:
קוּמוּ וְנִסַק בְּלֵילְיָא וּנְחַבֵּל בִּירְנְיָתָהָא:
אֲרֵי כִדְנַן אֲמַר יְיָ צְבָאוֹת קוּצוּ אָעָא וְצַבָּרוּ עַל יְרוּשְׁלֵם מְלֵיתָא הָא קַרְתָּא דְאִתְפַּקִידוּ לָהּ חוֹבָהָא כּוּלְהוֹן אוֹנְסִין בְּגַוָהּ:
כְּגוֹבָא דִמְקַיֵים מוֹהִי כֵּן אוֹרִיכוּ וְאִתְקַיָמוּ בְּגַוָהּ עָבְדֵי בִישְׁתָא קַל חֲטוֹפִין וּבְזוֹזִין מִשְׁתְּמַע בָהּ קֳדָמַי תְּדִירָא עַל כֵּן אַיְתֵיתִי עֲלָהּ מְרַע וּמָחָא:
אִיתְרְדָא יְרוּשְׁלֵם דִלְמָא יְרָחַק מֵימְרִי יָתָךְ דִלְמָא אֲשַׁוִינָךְ צַדְיָא כַּאֲרַע סְדוֹם דְלָא אִתּוֹתָבַת:
כִּדְנַן אֲמַר יְיָ צְבָאוֹת אִתְבְּזָזָא יִתְבַּזְזוּן כְּמָא דִמְהַלְכִין עַל שׁוּלֵי גֶפֶן לְמִקְטַף שְׁאָרָא דְיִשְׂרָאֵל וּתְתֵיב מַחְתָּךְ עֲלֵיהוֹן כִּדְקָטֵיף עַל סַלְסְלִין:
עַל מַה אֲמַלֵיל וְאַסְהֵיד וִיקַבְּלוּן אוּלְפַן הָא אִטַפִּשַׁת אוּדְנְהוֹן וְלָא יָכְלִין לְמֵיצָת הָא פִּתְגָמָא דַייָ הֲוָה לְהוֹן לְחִיסוּדִין מִדְלָא צָבָן בֵּיהּ:
וְיַת נְבוּאַת דַייָ בִּתְקוֹף מִן קֳדָם יְיָ אִתְמְלֵיתִי לְאֵיתִי לְסוֹבָרָא וְלָא יְכֵילְתִּי לְמִשְׁפַּךְ עַל יַנְקִין בְּשׁוֹקָא וְעַל סִיעַת עוּלֵמִין כַּחֲדָא אֲרֵי אַף גְבַר עִם אִתָּא יִתְאֲחָדוּ סָב עִם דִשְׂבַע יוֹמִין:
וְיִסְתַּחֲרוּן בָּתֵּיהוֹן לְאוֹחֲרָנִין חַקְלָן וּנְשִׁין כַּחֲדָא אֲרֵי אָרֵים יַת מְחַת גְבוּרְתִּי עַל יַתְבֵי אַרְעָא אֲמַר יְיָ:
אֲרֵי מִזְעֵירְהוֹן וְעַד רַבְּהוֹן כּוּלְהוֹן אַנְסֵי מָמוֹן וּמִסָפַר וְעַד כָּהֵין כּוּלְהוֹן עָבְדֵי שְׁקָר:
וְאַסִיאוּ יַת תְּבַר כְּנִשְׁתָּא דְעַמִי בְּמִלֵי שִׁקְרֵיהוֹן לְמֵימַר שְׁלָמָא שְׁלָמָא וְלֵית שְׁלָם:
עֲלֵיהוֹן לְמִבְהַת אֲרֵי תּוֹעֵבְתָּא עֲבָדוּ אַף מִבְהַת לֵית אִינוּן בָהֲתִין אַף אִתְכְּנָעָא לֵית אִינוּן יָדְעִין בְּכֵן יִתְרְמוֹן קָטִילִין בְּעִדַן דְאַסְעַר עֲלֵיהוֹן חוֹבֵיהוֹן יִתְקְלוּן אֲמַר יְיָ:
כִּדְנַן אֲמַר יְיָ קוּמוּ עַל אוֹרְחַן וַחֲזוֹ וּשְׁאִילוּ לִשְׁבִילִין דְמִן עַלְמָא אֵידָא אוֹרְחָא דְתַקְנָא וְהַלִיכוּ בָּהּ וְאַשְׁכָּחוּ נְיַח לְנַפְשְׁכוֹן וַאֲמָרוּ לָא נֵיזֵיל:
וַאֲקֵימִית עֲלֵיכוֹן מַלְפִין קַבִּילוּ לְמִלֵי נְבִיָא וַאֲמָרוּ לָא נְקַבֵּיל:
בְּכֵן שְׁמָעוּ עַמְמַיָא וּדְעִי כְּנִשְׁתָּא דְיִשְׂרָאֵל יַת חוֹבֵיהוֹן:
שְׁמָעִי אַרְעָא דְיִשְׂרָאֵל הָא אֲנָא מַיְתֵי בִישְׁתָּא עַל עַמָא הָדֵין תּוּשְׁלְמַת עוֹבָדֵיהוֹן אֲרֵי לְפִתְגָמֵי עַבְדֵי נְבִיַיָא לָא קַבִּילוּ וְאוֹרַיְתִי וְקָצוּ בָּהּ:
לְמָא דְנַן קֳדָמַי לְבוֹנְתָא דְמִשְׁבָא תֵיתֵי וְקָנֵי טַבְתָא מֵאַרַע רְחִיקָא עֲלָוָתְכוֹן לָא לְרַעֲוָא וְנִכְסַת קוּדְשֵׁיכוֹן לָא יִתְקַבְּלוּן קֳדָמָי:
בְּכֵן כִּדְנַן אֲמַר יְיָ הָא אֲנָא מַיְתִי עַל עַמָא הָדֵין תַּקְלָן וְיִתְקְלוּן בְּהוֹן אֲבָהָן וּבְנִין כַּחֲדָא מָגֵיר וְקָרִיבֵיהּ יֵיבְדוּן:
כִּדְנַן אֲמַר יְיָ הָא עַמָא אָתֵי מִצִפּוּנָא וְעַמְמִין רַבְרְבִין יֵיתוּן בִּגְלֵי מִסְיָפֵי אַרְעָא:
קַשְׁתָּן וּתְרֵיסִין אֲחִידִין אַכְזְרְאִין אִינוּן וְלֵית לְהוֹן רַחֲמִין קָלְהוֹן כְּיַמָא בְּאִתְרְגוֹשֵׁיהּ וְעַל סוּסְוָתָא רְכֵיבִין מְסַדְרִין אִינוּן כִּגְבַר עָבֵיד קְרָבִין עֲלָךְ כְּנִשְׁתָּא דְצִיוֹן:
שְׁמַעְנָא יַת שִׁמְעֲהוֹן אִתְרַשְׁלָא יְדָנָא עָקָא אֲחַדְתָּנָא הָא כְּחֶבְלִין לְיָלֵדְתָּא:
לָא תִפְקוּן לְחַקְלָא וּבְאוֹרְחָא לָא תְהָכוּן אֲרֵי חַרְבָּא דְסָנְאָה מְקַטְלָא לְמִתְכַּנְשִׁין מִסְחוֹר סְחוֹר:
כְּנִשְׁתָּא דְעַמִי אֱסָרוּ סַקִין וְחָפוּ רֵישְׁכוֹן בְּקִטְמָא אֵבֶל כְּיָחִיד עֲבִידוּ לְכוֹן מִסְפֵּיד בִּמְרַר אֲרֵי בִּתְּכֵּיף יֵיתוּן בְּזוֹזִין עֲלָנָא:
בְּחִיר יְהַבְתָּךְ בְּעַמִי כִּכְרַךְ תַּקִיף וּתְהוֹדַע וּתְחַוֵי לְהוֹן יַת אוֹרְחַתְהוֹן:
כָּל רַבְרְבֵיהוֹן מָרְדִין מְהַלְכִין בִּנְכִילוּ כְּמָא דִמְעָרֵב נְחָשׁ עִם פַּרְזְלָא כּוּלְהוֹן מְחַבְּלִין אִינוּן:
הָא כְּמֵשֵׁב נַפְּחִין דַחֲרִיךְ בְּגַוָאָה דְנוּרָא כֵּן חֲשָׁא קָלְהוֹן דִנְבִיַיָא דְמִתְנַבָּן לְהוֹן תּוּבוֹ לְאוֹרַיְתָא וְלָא תָבוּ וְכַאֲבָרָא דְמִתְמְסֵי בְּגוֹ כוּרָא כֵּן הֲוָאָה בְּטִילִין בְּעֵינֵיהוֹן מִלֵי נְבִיַיָא דְמִתְנַבָּן לְהוֹן לָא לְהַנָאָה אֲלִיפוּנוּן מַלְפֵיהוֹן וְעוֹבָדֵיהוֹן בִּישַׁיָא לָא שְׁבָקוּ:
הָא מִכְּסַף פְּסוּלָא מְרַחְקִין לְהוֹן אֲרֵי רָחֵיק יְיָ יַתְהוֹן:
