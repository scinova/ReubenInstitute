כִּדְנַן אֲמַר יְיָ לִי אֱזֵיל וְתִזְבּוּן לָךְ זְרַז דְכִתַּן וּתְשַׁוִינֵיהּ עַל חַרְצָךְ וּבְמַיָא לָא תֶעֲלִינֵיהּ:
וּזְבִינֵיהּ יַת זַרְזָא כְּפִתְגָמָא דַייָ וְשַׁוִיתֵיהּ עַל חַרְצִי:
וַהֲוָה פִּתְגַם נְבוּאָה מִן קֳדָם יְיָ עִמִי תִּנְיָנוּת לְמֵימָר:
סַב יַת זַרְזָא דִי זְבִינְתָּא דִי עַל חֲרָצָךְ וְקוּם אֱזֵיל לִפְרָת וּטְמַרְהִי תַמָן בִּשְׁעִיעוּת כֵּיף גְלֵי:
וַאֲזָלַת וּטְמַרְתֵּיהּ בִּפְרָת כְּמָה דִי פַּקֵיד יְיָ יָתִי:
וַהֲוָה מִסוֹף יוֹמִין סַגִיאִין וַאֲמַר יְיָ לִי קוּם אֱזֵיל לִפְרָת וְסַב מִתַמָן יַת זַרְזָא דִי פַקֵדְתָּךְ לְמִטַמְרֵיהּ תַמָן:
וַאֲזָלֵית לִפְרָת וַחֲפָרֵית וּנְסִיבֵית יַת זַרְזָא מִן אַתְרָא דִטְמַרְתֵּיהּ תַּמָן וְהָא אִתְחַבֵּיל זַרְזָא לֵית בֵּיהּ צְרוֹךְ לְכָל מִדַעַם:
וַהֲוָה פִּתְגַם נְבוּאָה מִן קֳדָם יְיָ עִמִי לְמֵימָר:
כִּדְנַן אֲמַר יְיָ כְּדֵין אֲחַבֵּיל יַת תְּקוֹף אֱנַשׁ יְהוּדָה וְיַת תְּקוֹף יַתְבֵי יְרוּשְׁלֵם דַהֲוָה סַגִי:
עַמָא הָדֵין דְעוֹבָדֵיהוֹן בִּישִׁין דְסָרִיבוּ לְקַבָּלָא יַת פִּתְגָמָא דְאָזְלִין בָּתַר הִרְהוּר לִבְּהוֹן וּטְעוֹ בָּתַר טַעֲוַת פַּלְחֵי כּוֹכְבַיָא לְמִפְלַחְהוֹן וּלְמִסְגֵיד לְהוֹן וִיהוֹן כְּזַרְזָא הָדֵין דְלֵית בְּהוֹן צְרוֹךְ לְכָל מִדָעַם:
אֲרֵי כְּמָא דְמִדְבַּק זַרְזָא בְּחַרְצָא דְגַבְרָא כֵּן קָרְבֵית לְפוּלְחָנִי יַת כָּל בֵּית יִשְׂרָאֵל וְיַת כָּל בֵּית יְהוּדָה אֲמַר יְיָ לְמֶהֱוֵי קֳדָמַי לְעַם וּלְשׁוּם וּלְחוּשְׁבְּחָא וְלִרְבוּ וְלָא קַבִּילוּ:
וְתֵימַר לְהוֹן יַת פִּתְגָמָא הָדֵין כִּדְנַן אֲמַר יְיָ אֱלָהָא דְיִשְׂרָאֵל כָּל גְרַב יִתְמְלֵי חֲמַר וְיֵימְרוּן לָךְ הֲמִידַע לָא הֲוֵינָא יָדְעִין אֲרֵי כָּל גְרַב יִתְמְלֵי חֲמָר:
וְתֵימַר לְהוֹן כִּדְנַן אֲמַר יְיָ הָא אֲנָא מְמַלֵי יַת כָּל יַתְבֵי אַרְעָא הָדָא וְיַת מַלְכָּא דְיַתְבִין לְדָוִד עַל כּוּרְסוֹהִי וְיַת כַּהֲנַיָא וְיַת נְבִיַיָא דְשִׁקְאָא וְיַת כָּל יַתְבֵי יְרוּשְׁלֵם עָקָא וִיהוֹן דָמָן לְרַוְיָא:
וֶאֱגָרִינוּן גְבַר בַּאֲחוּהִי וַאֲבָהָן וּבְנִין כַּחֲדָא אֲמַר יְיָ לָא אֶחְמוֹל וְלָא אֵיחוּס וְלָא אֲרַחֵם מִלְחַבָּלוּתְהוֹן:
קַבִּילוּ וְאַצִיתוּ לָא תִתְרַבְרְבוּן אֲרֵי בְמֵימְרָא דַייָ גְזִיר כֵּן:
הָבוּ קֳדָם יְיָ אֱלָהָכוֹן יְקָרָא עַד לָא תֵיתֵי עֲלֵיכוֹן עָקָא וּתְהוֹן דָמִין לְדִמְהַלְכִין כִּדְבְקִבְלָא וְעַד לָא יִתְנַגְפוּן רַגְלֵיכוֹן עַל טוּרֵי קָבָל וְתִסְבְּרוּן לִנְהוֹר וִיהֵי עֲלֵיכוֹן חֲשׁוֹךְ לְזֵיהוֹר וְהָא כְּדִבְקִבְלָא תְהוֹן מְהַלְכִין אַתּוּן:
וְאִם לָא תְקַבְּלוּן אוּלְפַן כַּד בְּסִתְרָא תִבְכֵּי נַפְשִׁי מִן קֳדָם דְיֶדְעֵי מִנְכוֹן יְקַרְכוֹן וּמִדְמַע תִּדְמַע וּתְשַׁגֵר עֵינֵי דִמְעִין אֲרֵי אִשְׁתְּבֵי עַמָא דַיְיָ:
אֲמַר לְמַלְכָּא וּלְמַלְכְּוָתָא אִמְאִיכוּ תָבוּ אֲרֵי גְלָא יְקַרְכוֹן מִנְכוֹן נְפַל כְּלִיל תּוּשְׁבַּתְּכוֹן:
קִרְוֵי דָרוֹמָא אִתְאַחִידוּ וְלֵית דְפָּתַח גְלוֹ דְבֵית יְהוּדָה כּוּלְהוֹן גָלוּ שְׁלֵימָא אִתְקַבְּלוּ תוּשְׁלִימַת עוֹבָדֵיכוֹן:
זְקִיפוּ עֵינֵיכוֹן וַחֲזוֹ דְאָתָן מִצִפּוּנָא אָן עַמָא דְאִתְמָסַר לִיךְ עַם תּוּשְׁבַּחְתִּיךְ:
מַה תֵּימְרוּן אֲרֵי יִסְעַר עֲלָךְ חוֹבָךְ וְאַתְּ אַלֵיפְתְּ יַתְהוֹן לְמֶעְבַּד בִּישִׁין עֲלָךְ הֲווֹ רַבְרְבִין כִּדְמַן אַוְלָא הֲלָא עָקָא וְחֶבְלִין יִיחְדוּנָךְ הָא כְּחֶבְלִין לְיָלֵידְתָּא:
וַאֲרֵי תֵימְרִין בְּלִבִּיךְ מָא דֵין עַרְעוּנִי כָּל אִלֵין עַל דְסַגִיאוּ חוֹבָךְ אִתְגְלִיאַת בַּהֲתָתִיךְ אִתְחֲזֵי קְלָנִיךְ:
הַאֶפְשַׁר דִישַׁנֵי הִנְדוֹאָה מַשְׁכֵיהּ וְנִמְרָא שִׁלְטֵי רִקְמָתֵיהּ אַף אִתּוּן תִּיכְלוּן לְאוֹטָבָא דַהֲוֵיתוּן מִלְפִין לְמֶעְבַּד בִּישִׁין:
וָאֱבַדְרִינוּן כְּקַשָׁא דְעָדֵי מִן קֳדָם רוּחָא לְמַדְבְּרָא:
דֵין עִדְבִיךְ וְחוּלַק אַחְסַנְתִּיךְ מִן קָדָמַי אֲמַר יְיָ דִשְׁבַקְתְּ פּוּלְחָנִי וְאִתְרַחֵיצְתְּ עַל שִׁקְרָא:
וְאַף אֲנָא אַגְלֵי בַּהֲתַת חוֹבָךְ עַל אַפָּךְ וְיִתְחֲזֵי קְלָנִיךְ:
טַפְּשׁוּתִיךְ וּשְׁטוּתִיךְ וַעֲצַת חֶטְאֵי יַנְקוּתִיךְ עַל רָמָתָא בְּחַקְלָא גְלֵי קָדָמַי שִׁקוּצִיךְ וַי לִיךְ יְרוּשְׁלֵם לָא תִדְכִּין עַד כְּעַן לִיךְ אַרְכָא יוֹמִין סַגִיאִין:
