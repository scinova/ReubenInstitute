וּלְאַחְאָב שַׁבְעִין בְּנִין בְּשֹׁמְרוֹן וּכְתַב יֵהוּא אִגָרָן וְשַׁדַר לְשֹׁמְרוֹן לְרַבְרְבֵי יִזְרְעֶאל סָבַיָא וּלְתוּרְבָּנַיָא דְאַחְאָב לְמֵימַר:
וּכְעַן כְּמֵיתָא אִגַרְתָּא הָדֵין לְוַתְכוֹן וְעִמְכוֹן בְּנֵי רִבּוֹנֵיכוֹן וְעִמְכוֹן רְתִיכִין וְסוּסְוָן וְקִרְוִין כְּרִיכָן וּמָאנֵי זְיָנָא:
וְתֶחֱזוּן דְתַקִין וְדִכְשַׁר מִבְּנֵי רִבּוֹנְכוֹן וּתְשַׁווּן עַל כּוּרְסֵי מַלְכוּתָא דַאֲבוּהִי וְתַגִיחוּן קְרָב עַל בֵּית רִבּוֹנְכוֹן:
וּדְחִילוּ לַחֲדָא לַחֲדָא וַאֲמַרוּ הָא תְּרֵין מַלְכַיָא לָא יְכִילוּ לְמֵיקַם קֳדָמוֹהִי וְאַכְדֵין נְקוּם אֲנַחְנָא:
וּשְׁלַח דִמְמַנָא עַל בֵּיתָא וְדִמְמַנָא עַל קַרְתָּא וְסָבַיָא וְתוּרְבְיָנַיָא לְוַת יֵהוּא לְמֵימַר עַבְדָךְ אֲנַחְנָא וְכָל דְתֵימַר לָנָא נַעְבֵּיד לָא נַמְלִיךְ גְבַר דְתַקִין בְּעֵינָךְ עֱבֵיד:
וּכְתַב לְהוֹן אִגַרְתָּא תִּנְיָנוּת לְמֵימַר אִם דִילִי אַתּוּן וּמִנִי אַתּוּן מְקַבְּלִין סָבוּ יַת רֵישֵׁי גַבְרִי בְּנֵי רִבּוֹנְכוֹן וְתַיְתוֹן לְוָתִי כְּעִדָנָא הָדֵין מְחַר לְיִזְרְעֶאל וּבְנֵי מַלְכָּא שַׁבְעִין גַבְרָא יַת רַבְרְבֵי קַרְתָּא מְרַבָּן יַתְהוֹן:
וַהֲוָא כְּמֵיתֵי אִגַרְתָּא לְוַתְהוֹן וְדַבָּרוּ יַת בְּנֵי מַלְכָּא וּנְכִיסוּ שַׁבְעִין גַבְרָא וְשַׁוִיאוּ יַת רֵישֵׁהוֹ בְּסָלַיָא וְשַׁדְרִינוּן לְוָתֵיהּ לְיִזְרְעֶאל:
וַאֲתָא אִזְגַדָא וְחַוִי לֵיהּ לְמֵימַר אַיְתִיאוּ רֵישֵׁי בְּנֵי מַלְכָּא וַאֲמַר שַׁווּ יַתְהוֹן תְּרֵין דְגוֹרִין בְּמַעֲלָנָא דְתַרְעָא עַד צַפְרָא:
וַהֲוָה בְּצַפְרָא וּנְפַק וְקָם וַאֲמַר לְכָל עַמָא זַכָּאִין אַתּוּן הָא אֲנָא מְרָדִית עַל רִבּוֹנִי וּקְטַלְתֵּיהּ וּמַן קְטַל יַת כָּל אִלֵין:
דְעוּ כְעַן אֲרֵי לָא יִבָּטֵל מִפִּתְגָמָא דַייָ דְמַלֵיל יְיָ עַל בֵּית אַחְאָב וַיָי קַיֵם יַת דְמַלֵיל בְּיַד עַבְדֵיהּ אֵלִיָהוּ:
וּמְחָא יֵהוּא יַת כָּל דְאִשְׁתַּאֲרוּ לְבֵית אַחְאָב בְּיִזְרְעֶאל וְכָל רַבְרְבוֹהִי וְקָרִיבוֹהִי וְחַבְרוֹהִי עַד דְלָא אִשְׁתָּאַר לֵיהּ מְשֵׁיזִיב:
וְקָם וַאֲתָא וַאֲזַל לְשֹׁמְרוֹן הוּא בֵּית כְּנִישַׁת רַעֲיָא בְּאוֹרְחָא:
וְיֵהוּא אַשְׁכַּח יַת אֲמֵי אֲחַזְיָה מֶלֶךְ שִׁבְטָא דְבֵית יְהוּדָה וַאֲמַר מַן אַתּוּן וַאֲמַרוּ אֲחֵי אֲחַזְיָהוּ אֲנַחְנָא וּנְחַתְנָא לְמִסְעַד שְׁלַם בְּנֵי מַלְכָּא וּבְנֵי מַלְכְּתָא:
וַאֲמַר אֲחוּדִינוּן כַּד חַיִין וְאַחְדוּנוּן כַּד חַיִין וּנְכוּסוּנוּן לְגוֹ בֵית כְּנִשְׁתָּא אַרְבְּעִין וּתְרֵין גַבְרָא וְלָא אִשְׁתָּאַר אֱנַשׁ מִנְהוֹן:
וַאֲזַל מִתַּמָן וְאַשְׁכַּח יַת יְהוֹנָדָב בַּר רֵכָב לָקֳדָמוּתֵיהּ וּבָרְכֵיהּ וּשְׁאֵיל בִּשְׁלָמֵיהּ וַאֲמַר לֵיהּ הֲאִית לִבָּךְ כָּשַׁר כְּמָא דְלִבִּי עִם לִבָּךְ וַאֲמַר יְהוֹנָדָב אִית וְאִית הַב יַת יְדָךְ וִיהַב יְדֵיהּ וְאַסְקֵיהּ לְוָתֵיהּ לִרְתִיכָא:
וַאֲמַר אִיתָא עִמִי וַחֲזֵי בְקִנְאָתִי דַאֲנָא מְקַנֵא קֳדָם יְיָ וַאֲחִיתוּ יָתֵיהּ בִּרְתִיכֵיהּ:
וַאֲתָא לְשֹׁמְרוֹן וּמְחָא יַת כָּל דְאִשְׁתָּאֲרוּ לְאַחְאָב בְּשֹׁמְרוֹן עַד דִי שֵׁיצֵיהּ כְּפִתְגָמָא דַייָ דְמַלֵיל עִם אֵלִיָהוּ:
וּכְנַשׁ יֵהוּא יַת כָּל עַמָא וַאֲמַר לְהוֹן אַחְאָב פְּלַח יַת בְּעָלָא זְעֵיר יֵהוּא יִפְלְחִינֵיהּ סַגִי:
וּכְעַן כָּל נְבִיֵי בְעָלָא וְכָל פַּלְחוֹהִי וְכָל כּוּמְרוֹהִי זַמִינוּ לִי אֱנַשׁ לָא יִשְׁגֵי אֲרֵי דְבַח רַב לִי לִבְעָלָא כָּל דְיִשְׁגֵי לָא יִתְקַיֵם וְיֵהוּא עֲבַד בְּחוּכְמָא בְּדִיל לַאֲבָדָא יַת פַּלְחֵי בְעָלָא:
וַאֲמַר יֵהוּא זַמִינוּ כְנִשְׁתָּא לַבְּעָלָא וְזַמִינוּ:
וּשְׁלַח יֵהוּא בְּכָל יִשְׂרָאֵל וַאֲתוֹ כָּל פַּלְחֵי בְעָלָא וְלָא אִשְׁתָּאַר אֱנַשׁ דְלָא אֲתָא וַעֲלוּ לְבֵית בְּעָלָא וְאִתְמְלֵי בֵית בְּעָלָא סִפָּא בְּסִפָּא:
וַאֲמַר לְדִמְמַנָא עַל קוּמְטְרַיָא אַפֵיק לְבוּשָׁא לְכָל פַּלְחֵי בְעָלָא וְאַפֵּיק לְהוֹן לְבוּשָׁא:
וַאֲתָא יֵהוּא וִיהוֹנָדָב בַּר רֵכָב לְבֵית בְעָלָא וַאֲמַר לְפַלְחֵי בְעָלָא בְּלוּשׁוּ וַחֲזוֹ דִלְמָא אִית בָּא עִמְכוֹן מֵעַבְדַיָא דַייָ אֱלָהִין פַּלְחֵי בְעָלָא בִלְחוֹדֵיהוֹן:
וַאֲתוֹ לְמֶעְבַּד נִכְסַת קוּדְשִׁין וַעֲלַוָן וְיֵהוּא מַנֵי לֵיהּ בְּבָרָא תַּמְנִין גַבְרָא וַאֲמַר גַבְרָא דְיִשְׁתֵּיזֵיב מִן גַבְרָא דַאֲנָא מַעֵיל עַל יְדֵיכוֹן נַפְשֵׁיהּ חֲלַף נַפְשֵׁיהּ:
וַהֲוָה כַד שֵׁיצֵי לְמֶעְבַּד עֲלָתָא וַאֲמַר יֵהוּא לְרָהֲטַיָא וּלְגִבָּרַיָא עוּלוּ מְחוֹנוּן אֱנַשׁ לָא יִפּוֹק וּמְחוֹנוּן לְפִתְגַם דְחָרֶב וּרְמוֹ רָהֲטַיָא וְגִבָּרַיָא קְטִילִין וַאֲזַלוּ עַד קִרְיַת בֵּית בְּעָלָא:
וְאַפִּיקוּ יַת קָמַת בֵּית בְּעָלָא וְאוֹקִידוּהָ:
וּתְרָעוּ יַת קָמַת בְּעָלָא וּתְרָעוּ יַת בֵּית בַּעֲלָא וְשַׁוְיוּהָא לְבֵית מַפְקַת אֱנָשָׁא עַד יוֹמָא דֵין:
וְשֵׁיצֵי יֵהוּא יַת בְּעָלָא מִיִשְׂרָאֵל:
לְחוֹד בְּחוֹבֵי יְרָבְעָם בַּר נְבָט דְחַיֵיב יַת יִשְׂרָאֵל לָא סְטָא יֵהוּא מִבַּתְרֵיהוֹן אִשְׁתַּעְבֵיד לְעֶגְלֵי דַהֲבָא דִי בְבֵית אֵל וְדִי בְדָן:
וַאֲמַר יְיָ לְיֵהוּא חֲלַף דְאַתְקִינְתָּא לְמֶעְבַּד דְכָשַׁר קֳדָמַי כְּכֹל דְבִרְעוּתִי עַבַדְתָּא לְבֵית אַחְאָב בְּנִין רְבִיעִין יֵתְבוּן לָךְ עַל כּוּרְסֵי מַלְכוּתָא דְיִשְׂרָאֵל:
וְיֵהוּא לָא נְטַר לִמְהַךְ בְּאוֹרַיְתָא דַייָ אֱלָהָא דְיִשְׂרָאֵל בְּכָל לִבֵּיהּ לָא סְטָא מֵעַל חוֹבֵי יָרָבְעָם דְחַיֵב יַת יִשְׂרָאֵל:
בְּיוֹמַיָא הָאִינוּן שָׁרֵי רוּגְזָא דַייָ לְמִתְקַף בְּיִשְׂרָאֵל וּמְחָנוּן חֲזָאֵל בְּכָל תְּחוּם אַרְעָא דְיִשְׂרָאֵל:
מִן יַרְדְנָא מַדְנַח שִׁמְשָׁא יַת כָּל אַרְעָא גִלְעָד שִׁבְטָא גָד וְשִׁבְטָא רְאוּבֵן וְשִׁבְטָא דִמְנַשֶׁה מֵעֲרוֹעֵר דְעַל כֵּיף נַחֲלָא דְאַרְנוֹן וְאַרְעָא גִלְעָד וּמַתְנָן:
וּשְׁאַר פִּתְגָמֵי יֵהוּא וְכָל דַעֲבַד וּגְבוּרְתֵּיהּ הֲלָא אִינוּן כְּתִיבִין עַל סְפַר פִּתְגָמֵי יוֹמַיָא לְמַלְכֵי יִשְׂרָאֵל:
וּשְׁכַב יֵהוּא עִם אֲבָהָתוֹהִי וּקְבָרוּ יָתֵיהּ בְּשֹׁמְרוֹן וּמְלַךְ יְהוֹאָחָז בְּרֵיהּ תְּחוֹתוֹהִי:
וְיוֹמַיָא דִמְלַךְ יֵהוּא עַל יִשְׂרָאֵל עַסְרִין וְתַמְנֵי שְׁנִין בְּשֹׁמְרוֹן:
