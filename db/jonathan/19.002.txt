וְזַמִין יְיָ נוּנָא רַבָּא לְמִבְלַע יַת יוֹנָה וַהֲוָה יוֹנָה בִּמְעַיָא דְנוּנָא תְּלָתָא יוֹמִין וּתְלָתָא לֵילְוָן:
וְצַלֵי יוֹנָה קֳדָם יְיָ אֱלָהֵיהּ מִמְעַיָא דְנוּנָא:
וַאֲמַר צַלֵיתִי מִדְעָקַת לִי קֳדָם יְיָ וְקַבֵּל צְלוֹתִי מֵאַרְעִית תְּהוֹמָא בָּעִיתִי עֲבַדְתָּא בָּעוּתִי:
וּרְמִיתַנִי בְּעוּמְקַיָא בְּלִבָּא דְיַמָא וּנְהַר סְחוֹר סְחוֹר לִי כָּל נַחְשׁוֹלוֹהִי דְיַמָא וְגַלוֹהִי עֲלַי עָדוּ:
וַאֲנָא אֲמָרִית אִתְּרְכִית מִן קֳדָם מֵימְרָךְ בְּרַם אוֹסִיף לְאִסְתַּכְּלָא בְּהֵיכְלָא דְקוּדְשָׁךְ:
אַקְפוּנִי מַיָא עַד מוֹתָא תְּהוֹמָא סְחוֹר סְחוֹר לִי יַמָא דְסוּף תָּלֵי עֵיל מֵרֵישִׁי:
לְעִקָרֵי טוּרַיָא נְחָתֵית אַרְעָא נֶגְדַת בְּתוּקְפָהָא עֵיל מִנִי עַלְמִין וְאַתְּ קָרִיב קֳדָמָךְ לְאַיָקָא מֵחַבְּלָא חַיֵי יְיָ אֱלָהִי:
בְּאִשְׁתַּלְהָיוּת עֲלַי נַפְשִׁי פּוּלְחָנָא דַייָ אִידַכְּרֵית וְעָלַת לִקְדָמָךְ צְלוֹתִי לְהֵיכָלָא דְקוּדְשָׁךְ:
לָא כְּעַמְמַיָא פַּלְחֵי טַעֲוָתָא מֵאֲתַר דְאִתּוֹטַב לְהוֹן לֵית אִינוּן יָדְעוּן:
וַאֲנָא בְּתוּשְׁבְּחַת אוֹדָאָה קוּרְבָּנִי אַקְרִיב קֳדָמָךְ דִנְדָרִית אֲשַׁלֵם פּוּרְקַן נַפְשִׁי בִּצְלוֹ קֳדָם יְיָ:
וַאֲמַר יְיָ לְנוּנָא וּפְלֵט יַת יוֹנָה לְיַבֶּשְׁתָּא:
