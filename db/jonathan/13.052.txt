בַּר עַסְרִין וַחֲדָא שְׁנִין צִדְקִיָה כַּד מְלַךְ וַחֲדָא עַסְרֵי שְׁנִין מְלַךְ בִּירוּשְׁלֵם וְשׁוּם אִמֵיהּ חֲמוּטַל בַּת יִרְמְיָה מִלִבְנָה:
וַעֲבַד דְבִישׁ קֳדָם יְיָ כְּכֹל דִי עֲבַד יְהוֹיָקִים:
אֲרֵי עַל דְאַרְגִיזוּ קֳדָם יְיָ הֲוַת בִּירוּשְׁלֵם וּבִיהוּדָה עַד דְאַגְלֵי יַתְהוֹן מֵאַרְעָא בֵּית שְׁכִינְתֵּיהּ וּמְרִיד צִדְקִיָהוּ בְּמַלְכָּא דְבָבֶל:
וַהֲוָה בְשַׁתָּא תְּשִׁיעֵיתָא לְמִמְלְכֵהּ בְּיַרְחָא עֲשִׂירָאָה בְּעַשְׂרָא לְיַרְחָא אֲתָא נְבוּכַדְרֶאצַר מַלְכָּא דְבָבֶל הוּא וְכָל מַשְׁרִיָתֵיהּ עַל יְרוּשְׁלֵם וּשְׁרָא עֲלָהּ וּבְנוֹ עֲלָהּ כַּרְכּוּם סְחוֹר סְחוֹר:
וַעֲלַת קַרְתָּא בִּצְיָרָא עַד חֲדָא עַשְׂרֵי שְׁנִין לְמַלְכָּא צִדְקִיָהוּ:
בְּיַרְחָא רְבִיעָאָה בְּתִשְׁעָא לְיַרְחָא וּתְקֵיף כַּפְנָא בְּקַרְתָּא וְלָא הֲוַת מֵיכַל לְעַמָא דְאַרְעָא:
וְאִתְּרַעַת קַרְתָּא וְכָל גַבְרֵי עָבְדֵי קְרָבָא עֲרָקוּ וּנְפָקוּ מִן קַרְתָּא בְּלֵילְיָא בְּאוֹרַח תְּרַע בֵּין שׁוּרַיָא דִי עַל גִנְתָא דְמַלְכָּא וְכַסְדָאֵי שָׁרָן עַל קַרְתָּא סְחוֹר סְחוֹר וַאֲזַלוּ בְּאוֹרַח מֵישְׁרָא:
וּרְדָפָא מַשִׁרְיַת כַּסְדָאֵי בָּתַר מַלְכָּא וְאַדְבִּיקוּ יַת צִדְקִיָה בְּמֵישְׁרֵי יְרֵיחוֹ וְכָל מַשִׁרְיָתֵיהּ אִתְבַּדָרָא מִנֵיהּ:
וַאֲחִידוּ יַת מַלְכָּא וְאַסִיקוּ יָתֵיהּ לְוַת מַלְכָּא דְבָבֶל לְדִבְלָת בְּאַרְעָא דַחֲמָת וּמַלֵיל עִמֵיהּ פִּתְגָמֵי דִינִין:
וּנְכֵיס מַלְכָּא דְבָבֶל יַת בְּנֵי צִדְקִיָהוּ בְּרִבְלָתָה לְעֵינוֹהִי וְאַף יַת כָּל רַבְרְבֵי יְהוּדָה נְכַס בְּרִבְלָתָה:
וְיַת עֵינֵי צִדְקִיָהוּ עַוֵר וְאַסְרֵיהּ בְּשִׁלְשְׁלָן דִנְחַשׁ וְאוֹבִילֵיהּ מַלְכָּא דְבָבֶל לְבָבֶל וִיהָבֵיהּ בְּבֵית אֲסִירֵי עַד יוֹם מוֹתֵיהּ:
וּבְיַרְחָא חֲמִישָׁאָה בְּעַסְרָא לְיַרְחָא הִיא שְׁנַת תְּשַׁע עַסְרֵי שְׁנִין לְמַלְכָּא נְבוּכַדְרֶאצַר מַלְכָּא דְבָבֶל אֲתָא נְבוּזַרְאֲדָן רַב קָטוֹלַיָא קָם קֳדָם מַלְכָּא דְבָבֶל בִירוּשְׁלֵם:
וְאוֹקֵיד יַת בֵּית מַקְדְשָׁא דַייָ וְיַת בֵּית מַלְכָּא וְיַת כָּל בָּתֵּי יְרוּשְׁלֵם וְיַת כָּל בָּתֵּי רַבְרְבַיָא אוֹקִיד בְּנוּרָא:
וְיַת כָּל שׁוּרֵי יְרוּשְׁלֵם סְחוֹר סְחוֹר תְּרָעוּ כָּל מַשִׁרְיַת כַּסְדָאֵי דְעִם רַב קָטוֹלַיָא:
וּמַחֲשִׁיכֵי עַמָא וְיַת שְׁאָרָא דְעַמָא דְאִשְׁתַּאֲרוּ בְקַרְתָּא וְיַת שְׁמִיעַיָא דְאִשְׁתַּמָעוּ עַל מַלְכָּא דְבָבֶל וְיַת שְׁאָר הֲמוֹנָא אַגְלֵי נְבוּזַרְאֲדָן רַב קָטוֹלַיָא:
וּמֵחֲשִׁיכֵי עַמָא דְאַרְעָא אַשְׁאַר נְבוּזַרְאֲדָן רַב קָטוֹלַיָא לְמֶהֱוֵי מַפְלְחִין בְּחַקְלִין וּבְכַרְמִין:
וְיַת עַמוּדֵי נְחָשָׁא דִי בְבֵית מַקְדְשָׁא דַייָ וְיַת בְּסִיסַיָא וְיַת יַמָא דִנְחָשָׁא דִי בְּבֵית מַקְדְשָׁא דַייָ תְּבַרוּ כַסְדָאֵי וְאוֹבִילוּ יַת כָּל נְחָשֵׁיהוֹן לְבָבֶל:
וְיַת דוּדַיָא וְיַת מַגְרוֹפְיָתָא וְיַת מְזַמְרַיָא וְיַת מִזְרְקַיָא וְיַת בָּזִיכַיָא וְיַת כָּל מָנֵי נְחָשָׁא דַהֲווֹ מְשַׁמְשִׁין בְּהוֹן נְסִיבוּ:
וְיַת קוּלַיָא וְיַת מַחְתְּיָתָא וְיַת מִזְרְקַיָא וְיַת דוּדַיָא וְיַת מְנַרָתָא וְיַת בָּזִיכַיָא וְיַת מְכִילָתָא דִי דַהֲבָא דַהֲבָא וְדִי כַסְפָּא כַסְפָּא נְסִיב רַב קָטוֹלַיָא:
עֲמוּדַיָא תְּרֵין יַמָא חָד וְתוֹרֵי תְּרֵי עֲסַר דִנְחָשָׁא דִתְחוֹת בְּסִיסַיָא דַעֲבַד מַלְכָּא שְׁלֹמֹה בְּבֵית מַקְדָשָׁא דַייָ לָא הֲוָה מִתְקַל לִנְחָשְׁהוֹן דְכָל מָנַיָא הָאִלֵין:
וְעַמוּדַיָא תַּמְנָא עַסְרֵי אַמִין רוּמֵיהּ דְעַמוּדָא חָד וְחוּט מְשַׁח תַּרְתָּא עַסְרֵי אַמִין מַקֵף לֵיהּ סְחוֹר סְחוֹר וְסוּמְכֵיהּ אַרְבַּע אֶצְבָּעִין חֲלִיל:
וְקָרוּנְתְּהוֹן עֲלוֹהִי דִנְחָשָׁא וְרוּמָא דְקָרוּנְתְּהוֹן חֲדָא חֲמֵשׁ אַמִין וּסְרִיגְתָּא וְרִמוֹנַיָא עַל קְרוּנְתְּהוֹן סְחוֹר סְחוֹר כּוֹלָא נְחָשָׁא וּכְאִלֵין לְעַמוּדָא תִנְיָנָא וְרִמוֹנַיָא:
וַהֲווֹ רִמוֹנַיָא תִּשְׁעִין וְשִׁתָּא בְעֶבְרָתָא כָּל רִמוֹנַיָא מְאָה עַל סְרִיגְתָּא סְחוֹר סְחוֹר:
וּדְבַר רַב קָטוֹלַיָא יַת שְׂרָיָה כַּהֲנָא רַבָּא וְיַת צְפַנְיָה סְגַן כַּהֲנַיָא וְיַת תְּלָתָא אֲמַרְכְּלַיָא:
וּמִן קַרְתָּא דְבַר גֵוִזָא חַד דַהֲוָה מְמַנָא עַל גַבְרֵי עָבְדֵי קְרָבָא וְשִׁבְעָא גַבְרִין דְחָזָן אַפֵּי מַלְכָּא דְאִשְׁתְּכִחוּ בְקַרְתָּא וְיַת סַפְרָא רַב חֵילָא דְכָנֵשׁ יַת עַמָא דְאַרְעָא וְשִׁתִּין גַבְרָא מֵעַמָא דְאַרְעָא דְאִשְׁתְּכִחוּ בְגוֹ קַרְתָּא:
וּדְבַר יַתְהוֹן נְבוּזַרְאֲדָן רַב קָטוֹלַיָא וְאוֹבִיל יַתְהוֹן לְוַת מַלְכָּא דְבָבֶל לְרִבְלָת:
וּמְחָא יַתְהוֹן מַלְכָּא דְבָבֶל וְקַטְלִינוּן בְּרִבְלָה בְּאַרְעָא דַחֲמָת וְאִתְגְלִיאוּ דְבֵית יְהוּדָה מֵעַל אַרְעֲהוֹן:
דֵין עַמָא דְאַגְלֵי נְבוּכַדְרֶאצַר בִּשְׁנַת שְׁבַע יְהוּדָאֵי תִּלְתָא אַלְפִין וְעַסְרִין וּתְלָתָא:
בִּשְׁנַת תַּמְנֵי עַסְרֵי לִנְבוּכַדְרֶאצַר מִירוּשְׁלֵם נַפְשָׁתָא תַּמְנֵי מְאָה וּתְלָתִין וְתַרְתֵּין:
בִּשְׁנַת עַסְרִין וּתְלַת לִנְבוּכַדְרֶאצַר אֶגְלֵי נְבוּזַרְאֲדָן רַב קָטוֹלַיָא יְהוּדָאִי נַפְשָׁתָא שְׁבַע מְאָה אַרְבְּעִין וַחֲמֵשׁ כָּל נַפְשָׁתָא אַרְבְּעָא אַלְפִין וְשִׁית מְאָה:
וַהֲוָה בִּתְלָתִין וּשְׁבַע שְׁנִין לְגָלוּת יְהוֹיָכִין מְלַךְ שִׁבְטָא דְבֵית יְהוּדָה בִּתְרֵי עֲשַׂר יַרְחִין בְּעַשְׂרִין וְחַמְשָׁא לְיַרְחָא רַבֵי אֱוִיל מְרוֹדַךְ מַלְכָּא דְבָבֶל בִּשְׁנַת מַלְכוּתֵיהּ יַת רֵישׁ יְהוֹיָכִין מְלַךְ שִׁבְטָא דְבֵית יְהוּדָה וְאַפֵּיק יָתֵיהּ מִבֵּית אֲסִירֵי:
וּמַלֵיל עִמֵיהּ פִּתְגָמִין תַּקְנִין וִיהַב יַת כּוּרְסֵיהּ מֵעַיל לְכוּרְסֵי מַלְכַיָא דִי עִמֵיהּ בְּבָבֶל:
וְשַׁנִי יַת לְבוּשֵׁי אִסוּרֵיהּ וְאָכִיל לַחֲמָא קָדָמוֹהִי תְּדִירָא כָּל יוֹמָא חַיוֹהִי:
וְשֵׁרוּתֵהּ שֵׁרוּתָא תְּדִירָא מִתְיהֲבָא לֵיהּ מִן קֳדָם מַלְכָּא דְבָבֶל פִּתְגַם יוֹם בְּיוֹמֵיהּ עַד יוֹם מוֹתֵיהּ כֹּל יוֹמֵי חַיוֹהִי:
