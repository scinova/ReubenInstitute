בִּשְׁנַת עַסְרִין וּתְלַת שְׁנִין לְיוֹאָשׁ בַּר אֲחַזְיָה מֶלֶךְ שִׁבְטָא דְבֵית יְהוּדָה מְלַךְ יְהוֹאָחָז בַּר יֵהוּא עַל יִשְׂרָאֵל בְּשֹׁמְרוֹן שְׁבַע עַסְרֵי שְׁנִין:
וַעֲבַד דְבִישׁ קֳדָם יְיָ וַאֲזַל בָּתַר חוֹבֵי יָרָבְעָם בַּר נְבָט דְחַיֵיב יַת יִשְׂרָאֵל לָא סְטָא מִנְהוֹן:
וּתְקֵיף רוּגְזָא דַייָ בְּיִשְׂרָאֵל וּמַסְרִינוּן בְיַד חֲזָאֵל מַלְכָּא דַאֲרָם וּבְיַד בַּר הֲדַר בַּר חֲזָאֵל כָּל יוֹמַיָא:
וְצַלִי יְהוֹאָחָז קֳדָם יְיָ וְקַבֵּיל צְלוֹתֵיהּ יְיָ אֲרֵי גְלֵי קָדָמוֹהִי דוֹחֲקָא דְיִשְׂרָאֵל אֲרֵי דְחַק יַתְהוֹן מַלְכָּא דַאֲרָם:
וִיהַב יְיָ לְיִשְׂרָאֵל פָּרִיק וּנְפָקוּ מִתְּחוֹת יַת אֱנַשׁ אֲרָם וִיתִיבוּ בְנֵי יִשְׂרָאֵל בְּקִרְוֵיהוֹן כְּמַאִתְמָלֵי וּמִדְקָדָמוֹהִי:
בְּרַם לָא סְטוּ מֵחוֹבֵּי יָרָבְעָם דְחַיֵיב יַת יִשְׂרָאֵל בְּהוֹן אֲזָלוּ וְאַף אֲשֵׁירָתָא קָמַת בְּשֹׁמְרוֹן:
אֲרֵי לָא אַשְׁאַר לִיהוֹאָחָז עַם אֱלָהֵין חַמְשִׁין פְּרָשִׁין וַעֲסַר רְתִיכִין וְעַסְרָא אַלְפִין גְבַר רַגְלִין אֲרֵי אַבְּדִינוּן מַלְכָּא דַאֲרָם וְשַׁוִינוּן כְּעַפְרָא לְדָיֵשׁ:
וּשְׁאָר פִּתְגָמֵי יְהוֹאָחָז וְכָל דַעֲבַד וּגְבוּרְתֵּיהּ הֲלָא אִינוּן כְּתִיבִין עַל סְפָר פִּתְגָמֵי יוֹמַיָא לְמַלְכֵי יִשְׂרָאֵל:
וּשְׁכִיב יְהוֹאָחָז עִם אֲבָהָתוֹהִי וְקַבְרוֹהִי בְּשֹׁמְרוֹן וּמְלַךְ יוֹאָשׁ בְּרֵיהּ תְּחוֹתוֹהִי:
בִּשְׁנַת תְּלָתִין וּשְׁבַע שְׁנִין לְיוֹאָשׁ מֶלֶךְ שִׁבְטָא דְבֵית יְהוּדָה מְלַךְ יְהוֹאָשׁ בַּר יְהוֹאָחָז עַל יִשְׂרָאֵל בְּשֹׁמְרוֹן שִׁית עַסְרֵי שְׁנִין:
וַעֲבַד דְבִישׁ קֳדָם יְיָ לָא סְטָא מִכָּל חוֹבֵי יָרָבְעָם בַּר נְבָט דְחַיֵב יַת יִשְׂרָאֵל בְּהוֹן אֲזַל:
וּשְׁאַר פִּתְגָמֵי יוֹאָשׁ וְכָל דַעֲבַד וּגְבוּרָתֵיהּ דְאַגִיחַ עִם אֲמַצְיָה מֶלֶךְ שִׁבְטָא דְבֵית יְהוּדָה הֲלָא אִינוּן כְּתִיבִין עַל סֵפֶר פִּתְגָמֵי יוֹמָא לְמַלְכֵי יִשְׂרָאֵל:
וּשְׁכִיב יוֹאָשׁ עִם אֲבָהָתוֹהִי וְיָרָבְעָם יְתִיב עַל כּוּרְסֵיהּ וְאִתְקְבַר יוֹאָשׁ בְּשֹׁמְרוֹן עִם מַלְכֵי יִשְׂרָאֵל:
וֶאֱלִישָׁע מְרַע יַת מַרְעֵיהּ דִימוּת בֵּיהּ וּנְחַת לְוָתֵיהּ יוֹאָשׁ מַלְכָּא דְיִשְׂרָאֵל וּבְכָא עַל אַפּוֹהִי וַאֲמַר רַבִּי רַבִּי דְטַב לֵיהּ לְיִשְׂרָאֵל בִּצְלוֹתֵיהּ מֵרְתִיכִין וּמִפָּרָשִׁין:
וַאֲמַר לֵיהּ אֱלִישָׁע סַב קַשְׁתָּא וְגִרְרַיָא וּנְסֵיב לְוָתֵיהּ קַשְׁתָּא וְגִרְרַיָא:
וַאֲמַר לְמַלְכָּא דְיִשְׂרָאֵל אָחֵית יְדָךְ עַל קַשְׁתָּא וְאַחֵית יְדֵיהּ וְשַׁוִי אֱלִישָׁע יְדוֹהִי עַל יְדֵי מַלְכָּא:
וַאֲמַר פְּתַח כֵּוָן לְמַדִינְחָא וּפְתַח וַאֲמַר אֱלִישָׁע שְׁדֵי וּשְׁדָא וַאֲמַר הָדֵין גִירָא יִתְעֲבֵיד לָנָא פּוּרְקָנָא מִן קֳדָם יְיָ וְהָדֵין גִירָא יִתְעֲבֵיד לָנָא נִצְחָנָא בֶּאֱנַשׁ אֲרָם וְתִמְחֵי יַת אֱנַשׁ אֲרָם בַּאֲפֵיק עַד דְיִשְׁתֵּיצוּן:
וַאֲמַר סַב גִרְרַיָא וּנְסִיב וַאֲמַר לְמַלְכָּא דְיִשְׂרָאֵל מְחֵי לְאַרְעָא וּמְחָא תְּלַת זִמְנִין וּפְסַק:
וּרְגֵיז עֲלוֹהִי נְבִיָא דַייָ וַאֲמַר הֲוָה חָזֵי לָךְ לְמִמְחֵי חֲמֵשׁ אוֹ שִׁית זִמְנִין בְּכֵן פּוֹן מְחֵיתָא יַת אֱנַשׁ אֲרָם עַד שֵׁיצָיוּתָא וּכְעַן תְּלַת זִמְנִין תִּמְחֵי יַת אֱנַשׁ אֲרָם:
וּמִית אֱלִישָׁע וּקְבָרוּהִי וּמַשִׁרְיַת מוֹאָב עָלַת בְּאַרְעָא בְּמֵיעַל שַׁתָּא:
וַהֲוָה עַד דְאִינוּן קָבְרִין גַבְרָא וְהָא חֲזוֹ יַת מַשְׁרִיתָא וּרְמוֹ יַת גַבְרָא בְּקִבְרָא דֶאֱלִישָׁע וַאֲזַל וּקְרֵיב גַבְרָא בְּגַרְמֵי דֶאֱלִישָׁע וַחֲיָא וְקָם עַל רִגְלוֹהִי:
וַחֲזָאֵל מַלְכָּא דַאֲרָם דְחַק יַת יִשְׂרָאֵל כָּל יוֹמֵי יְהוֹאָחָז:
וְחֵס יְיָ עֲלֵיהוֹן וּרְחַם עֲלֵיהוֹן וְאִתְפְּנֵי בְּמֵימְרֵיהּ לְאוֹטָבָא לְהוֹן בְּדִיל קְיָמֵיהּ דְעִם אַבְרָהָם יִצְחָק וְיַעֲקֹב וְלָא אֲבָה לְחַבָּלוּתְהוֹן וְלָא אַגְלִינוּן מֵאֲרַע בֵּית שְׁכִינְתֵּיהּ עַד כְּעַן:
וּמִית חֲזָאֵל מַלְכָּא דַאֲרָם וּמְלַךְ בַּר הֲדַד בְּרֵיהּ תְּחוֹתוֹהִי:
וְתַב יְהוֹאָשׁ בַּר יְהוֹאָחָז וּנְסִיב יַת קִירְוַיָא מִיַד בַּר הֲדַד בַּר חֲזָאֵל דִי נְסִיב מִן יְהוֹאָחָז אֲבוּהִי בִּקְרָבָא תְּלַת זִמְנִין מְחָהִי יוֹאָשׁ וַאֲתֵיב יַת קִירְוֵי יִשְׂרָאֵל:
