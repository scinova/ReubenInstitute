וַהֲוָה פִּתְגָם נְבוּאָה מִן קֳדָם יְיָ עִמִי לְמֵימָר:
וְאַתְּ בַּר אָדָם כִּדְנַן אֲמַר יְיָ אֱלֹהִים לְאַרְעָא דְיִשְׂרָאֵל קִצָא מְטָא מְטַת פּוּרְעֲנוּת קִצָא לְמֵיתֵי עַל אַרְבַּע רוּחֵי אַרְעָא:
כְּעַן מְטַת פּוּרְעֲנוּת קִצָא לְמֵיתֵי עֲלָךְ וְאֶשְׁלַח רוּגְזִי בִּיךְ וְאִתְפְּרַע מִנִיךְ כְּאוֹרְחָתַיִךְ וְאֶתֵּן עֲלָךְ יַת כָּל תּוֹעֲבָתָיִךְ:
וְלָא יְחוּס מֵימְרִי עֲלָךְ וְלָא אֶרְחַם אֲרֵי חוֹב אוֹרְחָתָיִךְ עֲלָךְ אֶתֵּן וּפוּרְעֲנוּת תּוֹעֲבָתַיִךְ בְּגַוִיךְ יֶהֶוְיָן וְתֵידְעוּן אֲרֵי אֲנָא יְיָ:
כִּדְנַן אֲמַר יְיָ אֱלֹהִים בִּישְׁתָא בָּתַר בִּישְׁתָא הָא אַתְיָא:
קִצָא מְטָא מְטַת פּוּרְעֲנוּת קִיצָא לְמֵיתֵי עֲלַךְ הָא אַתְיָא:
אִתְגְלִיאַת מַלְכוּתָא עֲלָךְ יָתֵיב אַרְעָא מְטָא עִדַן תְּבִירַת קָרֵיב יוֹם אִתְרְגוּשָׁא וְלֵית לְאִשְׁתֵּיזָבָא בִּמְצָדֵי טוּרַיָא:
כְּעַן מִקָרֵיב אֶשְׁפּוֹךְ חֵימְתִי עֲלָךְ וְיֵחוֹל רוּגְזִי בִּיךְ וְאִתְפְּרַע מִנִיךְ כְּאוֹרְחָתָיִךְ וְאֶתֵּן עֲלָךְ יַת כָּל תּוֹעֲבָתָיִךְ:
וְלָא תְחוֹס מֵימְרִי וְלָא אֶרְחִים כְּחוֹבֵי אוֹרְחָתַיִךְ עֲלָךְ אֶתֵּן וּפוּרְעֲנוּת תּוֹעֲבָתַיִךְ בְּגַוִיךְ יֶהֱוְיָן וְתִדְעוּן אֲרֵי אֲנָא יְיָ אַיְתֵיתִי עֲלֵיכוֹן מְחָא:
הָא יוֹם פּוּרְעֲנוּתָא מְטָא הָא אַתְיָא אִתְגְלִיאַת מַלְכוּתָא צְמַח שׁוּלְטָנָא הוֹפַע רַשִׁיעָא:
חֲטוֹפִין קָמוּ לְמִסְעַר רַשִׁיעִין לָא מִנְהוֹן וְלָא מְאִתְרְגוּשְׁתְּהוֹן וְלָא מִבְּנֵיהוֹן וְלָא מִבְּנֵי בְּנֵיהוֹן:
מְטָא עִדָן תּוּשְׁלְמַת חוֹבִין קָרִיב יוֹם פּוּרְעֲנוּת חֶטְאִין זָבְנָא לָא יֶחֱדֵי וּמְזוֹבְנָא לָא יִדְוֵי אֲרֵי רְגַז מִן קֳדָמַי עַל כָּל אִתְרְגוּשַׁתְהוֹן:
אֲרֵי מְזוֹבְנָא לִזְבִינָא לָא יְתוּב וְעַד דְאִינוּן חַיָן יִרְדִינוּן בִּגְוִיָתְהוֹן אֲרֵי נְבִיַיָא מִתְנַבָּן עַל כָּל אִתְרְגוּשַׁתְהוֹן לְמֵיתַב בִּתְיוּבְתָּא וְלָא תָיְבִין גְבַר בְּחוֹבֵי נַפְשֵׁיהּ מִתְרְעָן וְעַד דְאִינוּן קַיָמִין בִּתְיוּבְתָּא לָא מִתַּקְפִין:
נַפְקִין בְּאַצְוָחַת קַרְנָא מִתַקְנִין בְּמָנֵי זְיָנָא וְלֵית דְאָזִיל לִקְרָבָא אֲרֵי רְגַז מִן קֳדָמַי עַל כָּל אִתְרְגוּשַׁתְהוֹן:
חַרְבָּא מִלְבָרָא וּמוֹתָא וְכַפְנָא מִלְגָיו דִבְחַקְלָא בְּחַרְבָּא יִתְקְטֵיל וְדִי בְקַרְתָּא כַּפְנָא וּמוֹתָא יְשֵׁיצִינֵיהּ:
וּמְשֵׁיזְבַיָא יִשְׁתֵּיזְבוּן מִנְהוֹן וִיהוֹן עַל טוּרַיָא כְּיוֹנֵי חֵילַיָא כּוּלְהוֹן מְנַהֲמִין גְבַר בְּחוּבֵי נַפְשֵׁיהּ:
כָּל יְדַיָא יִתְרַשְׁלָן וְכָל רְכִיבַיָא יִתְאַשְׁדָן מַיָא:
וְיֵסְרוּן סַקִין וְתַחְפֵי יַתְהוֹן אִתְכְּנָעוּתָא וְעַל כָּל אַפֵּיהוֹן בַּהֲתָא וּבְכָל רֵישֵׁיהוֹן מְרַט:
כַּסְפֵּהוֹן בְּשׁוּקִין יִרְמוֹן וּדְהַבְהוֹן לְבוּסְרָן יְהֵי כַּסְפֵּיהוֹן וּדְהַבְהוֹן לָא יִכּוּל לְשֵׁיזָבוּתְהוֹן בְּיוֹם רוּגְזָא דַייָ נַפְשֵׁיהוֹן לָא יִסְבְּעוּן וּמֵעֵיהוֹן לָא יִמְלוּן אֲרֵי לְתַקְלַת חוֹבִין יְהֵי:
וְחֶדְוַת תִּקוּנֵיהּ לִיקַר יְהַבְתֵּיהּ וְאִינוּן צַלְמֵי תוֹעֲבַתְהוֹן שִׁקוּצֵיהוֹן עֲבַדוּ בֵיהּ עַל כֵּן יְהַבְתֵּיהּ לְהוֹן לְבוּסְרָן:
וְאֶמְסְרִנֵיהּ בְּיַד רַשִׁיעִין לְמִבַז וּלְחַיָבֵי אַרְעָא לַעֲדִי וְיַחֲלוּנֵיהּ:
וַאֲסַלֵיק שְׁכִינְתִּי מִנְהוֹן עַל דְאָחִילוּ יַת אַרְעָא בֵּית שְׁכִינְתִּי וְיַעֲלוּן בָּהּ רַשִׁיעִין וְיַחֲלוּנָהּ:
עֲבֵיד שִׁלְשְׁלָן אֲרֵי אַרְעָא אִתְמְלִיאַת חַיָבֵי קְטוֹל וְקַרְתָּא אִתְמְלִיאַת חֲטוֹפִין:
וְאַיְתֵי מַבְאִשֵׁי עַמְמַיָא וְיַחְסְנוּן יַת בָּתֵּיהוֹן וַאֲבַטֵיל רְבוּת רַשִׁיעִין וְיִתָּחֲלוּן מִקְדָשֵׁיהוֹן:
אִתְקְפָדַת מְטַת וְיִבְעוֹן שְׁלָמָא וְלֵית:
תְּבַר עַל תְּבַר יֵיתִי וּכְסוֹרְתָּא עַל בְּסוֹרְתָּא תְּהֵי וְיִבְעוּן אוּלְפַן מִסָפֵר וְאוֹרָאָה תִּפְסוֹק מִכָּהֵן וּמְלַךְ מְחַכִּים:
מַלְכָּא יִתְאַבַּל וְרַבָּא יִלְבַּשׁ צָדוּ וִידֵי עַמָא דְאַרְעָא יִתְבַּהֲלָן מֵאָרְחֵיהוֹן אַעְבֵיד עִמְהוֹן וּבְנִימוּסֵיהוֹן אִתְפְּרַע מִנְהוֹן וְיֵדְעוּן אֲרֵי אֲנָא יְיָ:
