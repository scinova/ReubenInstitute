וַהֲוָה בַּחֲדָא עַסְרָא שְׁנִין בִּתְלִיתָאָה בְּחַד לְיַרְחָא הֲוָה פִּתְגָם נְבוּאָה מִן קֳדָם יְיָ עִמִי לְמֵימָר:
בַּר אָדָם אֲמַר לְפַרְעֹה מַלְכָּא דְמִצְרַיִם וּלְאִתְרְגוּשְׁתֵּהּ לְמָן דְמִיתָא בְּתוּקְפָךְ:
הָא אַתּוּרָאָה דַמְיָא לְאַרְזָא בְּלִבְנָן דְנוֹפֵיהּ שַׁפִּיר וְסוֹכֵיהּ מַטְלָן וְרָם בְקוֹמָה וְעַל נַגְדִין שָׁלַח שָׁרְשׁוֹהִי:
בְּעַמְמַיָא סַגִי בְּסוּמְכְוָתָא תַּקִיף יַת מַלְכַיָא שַׁעְבֵּיד לִתְחוֹת מַלְכוּתֵיהּ וְיַת שִׁלְטוֹנוֹהִי מָנֵי עַל כָּל מְדִינַת אַרְעָא:
עַל כֵּן רָאִים בְּתוּקְפֵיהּ מִכָּל מַלְכֵי אַרְעָא וּסְגִיאַת מַשִׁרְיָתֵיהּ וְאִתְנַצְחוּ סוּמְכְוָתֵיהּ עַל עַמְמִין סַגִיאִין בְּאִתְנַצְחוּתֵהּ:
בְּמַשִׁרְיָתֵיהּ כְּבֵישׁ כָּל כְּרַכִּין תַּקִיפִין וּתְחוֹת שִׁלְטוֹנוֹהִי שַׁעְבֵּיד כָּל מְדִינַת אַרְעָא וּבִטְלַל מַלְכוּתֵיהּ יְתִיבוּ כָּל עַמְמִין סַגִיאִין:
וְאִתְנְצַח בְּסוּמְכְוָתֵיהּ בִּסְגֵי גִבָּרוֹהִי אֲרֵי הֲוַת אֵימְתֵיהּ עַל עַמְמִין סַגִיאִין:
מַלְכִין תַּקִיפִין לָא אִתְנְצָחוּ לְקִבְלֵהּ בִּתְקוֹף גְבוּרָא דַהֲוַת עִמֵהּ מִן קֳדָם יְיָ שִׁלְטוֹנִין לָא קָמוּ קֳדָם מַשִׁרְיָתֵיהּ וְגִבָּרִין לָא יָכִילוּ כְּסִימְכְוָתֵיהּ בִּתְקוֹף גְבוּרָא דַהֲוַת עִמֵיהּ מִן קֳדָם יְיָ לָא דָמֵיהּ לֵיהּ בְּתוּקְפֵיהּ:
שַׁפִּיר עֲבִידְתֵּיהּ בִּסְגֵי גִבָּרוֹהִי וְזָעוּ מִן קֳדָמוֹהִי כָּל מַלְכֵי קֶדֶם בִּתְקוֹף גְבוּרָא דַהֲוַת עִמֵהּ מִן קֳדָם יְיָ:
בְּכֵן כִּדְנַן אֲמַר יְיָ אֱלֹהִים חֲלַף דְרָמְתָּא בְּקוֹמָה וִיהַב מַרְוָתֵיהּ עַל מַלְכְּוָתָא וְרָאֵים לִבֵּיהּ בְּתוּקְפֵיהּ:
וְאֶמְסְרִינֵיהּ בְּיַד תַּקִיף עַמְמַיָא אִתְפְּרַע מִנֵיהּ בְּחוֹבֵיהּ תָּרֵיכְתֵּיהּ:
וְשֵׁיצָיוֹהִי נוּכְרָאִין תַּקִיפֵי עַמְמַיָא וְרַטְשׁוּהִי עַל טוּרַיָא וּבְכָל חֵילַיָא נָפְלָא מַשִׁרְיָתֵיהּ וְאִתְבַּדָרוּ סוּמְכְוָתֵיהּ בְּכָל פַּצִידֵי אַרְעָא וּנְחָתוּ מִטְלַל מַלְכוּתֵיהּ כָּל עַמְמֵי אַרְעָא וְרַטְשׁוֹהִי:
עַל מַפָּלַת קְטִילוֹהִי שְׁרוֹ כָּל עוֹפָא דִשְׁמַיָא וְעַל פִּגְרֵי מַשִׁירְיָתֵיהּ נָחָא כָּל חַיַת בָּרָא:
בְּדִיל דְלָא יְרָאמוּן בְּתוּקְפֵיהוֹן כָּל מַלְכֵי קֶדֶם וְלָא יִתְּנוּן יַת מַרְוַתְהוֹן עַל מַלְכְּוָתָא וְלָא יִתְרַבְרְבוּן לְהוֹן בְּתוּקְפֵיהוֹן כָּל עַבְדֵי מַלְכוּ אֲרֵי כּוּלְהוֹן אִתְמְסָרוּ לְמוֹתָא לְאַרְעָא אַרְעִיתָא בְּגוֹ בְנֵי אֱנָשָׁא עִם נָחֲתֵי גוֹב בֵּית אַבְדָנָא:
כִּדְנַן אֲמַר יְיָ אֱלֹהִים בְּיוֹם אַחְתוֹתִי יָתֵיהּ לִשְׁאוֹל אִתְאַבָּלוּ עֲלוֹהִי חֲפַת עָקָא יַת עַלְמָא וְחָרוֹבָא מְדִינָתָא וְזָעוּ עַמְמִין סַגִיאִין וַחֲשׁוֹכוּ עֲלוֹהִי אַפֵּי מַלְכַיָא וְכָל מַלְכֵי עַמְמַיָא עֲלוֹהִי טְפָחוּ כַתְפָא:
מִקַל מַפַּלְתֵּיהּ אַזְעִית עַמְמַיָא בְּאַחֲתוֹתִי יָתֵיהּ לִשְׁאוֹל עִם נַחֲתֵי גוֹב בֵּית אַבְדָנָא וְאִתְנֵחֲמוּ בְּאַרְעָא אַרְעִיתָא כָּל מַלְכֵי קֶדֶם שִׁלְטוֹנִין וְעַתִּירֵי נִכְסַיָא כָּל עַבְדֵי מַלְכוּ:
אַף אִינוּן עִמֵיהּ אִתְחָתוּ לִשְׁאוֹל עִם קְטִילֵי חַרְבָּא וְשִׁלְטוֹנוֹהִי אִתְּבָרוּ מַתְקִיפֵיהוֹן בְּגוֹ מַלְכוּתָא:
לְמַן דַמְיֵתָא כְּדֵין בִּיקַר וּבִרְבוּ בְּמַלְכֵי קֶדֶם וְתִתְּחַת עִם מַלְכֵי קֶדֶם לְאַרְעָא אַרְעִיתָא בְּגוֹ חַיָבַיָא תִשְׁכַּב עִם קְטִילֵי חַרְבָּא הוּא פַרְעֹה וְכָל אִתְרְגוּשְׁתֵּיהּ אֲמַר יְיָ אֱלֹהִים:
