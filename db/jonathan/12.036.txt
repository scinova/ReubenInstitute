וַהֲוָה בְּאַרְבַּע עַסְרֵי שְׁנִין לְמַלְכָּא חִזְקִיָה סְלִיק סַנְחֶרִיב מַלְכָּא דְאַתּוּר עַל כָּל קִרְוַיָא דְבֵית יְהוּדָה כְּרִיכְתָּא וְאַחֲדִנוּן:
וּשְׁלַח מַלְכָּא דְאַתּוּר יַת רַבְשָׁקֵה מִלָכִישׁ לִירוּשְׁלֵם לְוַת מַלְכָּא חִזְקִיָה בְּמַשִׁרְיָת סַגִיאָן וְקָם בִּמְזִיקַת בְרֵיכְתָּא עִלִיתָא דִכְבִישׁ חֲקַל מִשְׁטַח קַצָרַיָא:
וּנְפַק לְוָתֵהּ אֱלְיָקִים בַּר חִלְקִיָה דִי מְמַנָא עַל בֵּיתָא וְשֶׁבְנָא סַפְרָא וְיוֹאָח בַּר אָסָף דִי מְמַנָא עַל דָכְרָנַיָא:
וַאֲמַר לְהוֹן רַבְשָׁקֵה אֲמָרוּ כְעַן לְחִזְקִיָה כִּדְנַן אֲמַר מַלְכָּא דְרַבְּתָנָא מַלְכָּא דְאַתּוּר מָה רוּחֲצָנָא הָדֵין דְאִיתְרַחֵיצְתָּא:
אֲמַרֵית בְּרַם בְּמַמְלֵל סִפְוָן בְּמֵלָךְ וּגְבוּרָא אַעְבֵּיד קְרָבָא כְּעַן עַל מַן אִיתְרַחֵיצְתָּא אֲרֵי מְרַדְתָּא בִּי:
הָא אִיצְרַחֵיצְתָּא עַל סֶמֶךְ קַנְיָא רְעִיעָא הָדֵין עַל פַּרְעֹה מַלְכָּא דְמִצְרַיִם דְאִם יִסְתְּמֵיךְ גַבְרָא עֲלוֹהִי וְיֵעוֹל בִּידֵהּ וִיבַזְעִינֵהּ כֵּן פַּרְעֹה מַלְכָּא דְמִצְרַיִם לְכָל דְמִתְרַחֲצִין עַלוֹהִי:
וַאֲרֵי תֵימְרוּן לִי עַל מֵימְרָא דַייָ אֱלָהָנָא אִתְרַחֵיצְנָא הֲלָא הוּא דְאַעְדִי חִזְקִיָה יַת בָמָתוֹהִי וְיַת אֱגְרוֹהִי וַאֲמַר לֶאֱנַשׁ יְהוּדָה וּלְיָתְבֵי יְרוּשְׁלֶם קֳדָם מַדְבְּחָא הָדֵין תִּסְגְדוּן:
וּכְעַן אִתְעָרֵב כְעַן עִם רִבּוֹנִי מַלְכָּא דְאַתּוּר וְאֶתֵּן לָךְ תְּרֵין אַלְפִין סוּסְוָן אִם תִּכּוּל לְמַנָאָה עֲלָךְ רָכְבִין עֲלֵיהוֹן:
וְאֵיכְדֵין אַתְּ מְדַמִי לַאֲתָבָא יַת אַפֵּי חַד מִשִׁלְטוֹנֵי עַבְדֵי רִיבּוֹנִי זְעֵרַיָא וְאִיתְרַחֵצְתָּא לָךְ עַל מִצְרָיִם לִרְתִיכִין וּלְפָרָשִׁין:
וּכְעַן הֲבַּר מִמֵימְרָא דַייָ סַלְקֵית לְאַרְעָא הָדָא לְחַבְּלוּתָא יְיָ אֲמַר לִי סַק עַל אַרְעָא הָדָא וְחַבְּלָהּ:
וַאֲמַר אֶלְיָקִים וְשֶׁבְנָא וְיוֹאָח לְרַבְשָׁקֵה מַלֵיל כְּעַן עִם עַבְדָךְ אֲרָמִית אֲרֵי שָׁמְעִין אֲנַחְנָא וְלָא תְמַלֵיל עִמָנָא יְהוּדִית קֳדָם עַמָא דִי עַל שׁוּרָא:
וַאֲמַר רַבְשָׁקֵה הַלְוַת רִבּוֹנָךְ וַעֲלָךְ שַׁלְחַנִי רִבּוֹנִי לְמַלָלָא יַת פִּתְגָמַיָא הָאִלֵין הֲלָא עַל גַבְרַיָא דְיַתְבִין עַל שׁוּרָא לְמֵיכַל יַת מַפְקַתְהוֹן וּלְמִשְׁתֵּי יַת מֵימֵי רִגְלֵיהוֹן בִּצְיָרָא עִמְכוֹן:
וְקָם רַבְשָׁקֵה וְאַכְלֵי בְקַל רַב יְהוּדִית וַאֲמַר שְׁמָעוּ יַת פִּתְגָמָא מַלְכָּא דְרַבְּתָנָא מַלְכָּא דְאַתּוּר:
כִּדְנַן אֲמַר מַלְכָּא לָא יִטְעֵי לְכוֹן חִזְקִיָה אֲרֵי לָא יִכוּל לְשֵׁיזָבָא יַתְכוֹן:
וְלָא יַרְחִיץ יַתְכוֹן חִזְקִיָה עַל מֵימְרָא דַייָ לְמֵימָר שֵׁיזָבָא יְשֵׁזְבָנָנָא יְיָ לָא תִּתְמְסַר קַרְתָּא הָדָא בְּיַד מַלְכָּא דְאַתּוּר:
לָא תְקַבְּלוּן מִן חִזְקִיָה אֲרֵי כִדְנַן אֲמַר מַלְכָּא דְאַתּוּר עַבִידוּ עִמִי שְׁלָמָא וּפוּקוּ לְוָתִי וֶאֱכוּלוּ גְבַר פֵּרֵי גוּפְנוֹהִי וּגְבַר פֵּרֵי תֵינוֹהִי וּשְׁתוּ גְבַר מֵי גוּבֵהּ:
עַד מֵיתָא וְאֶדְבַּר יַתְכוֹן לְאַרְעָא טָבָא כְּאַרְעָכוֹן לְאַרְעָא טָבָא כְּאַרְעָכוֹן אֲרַע עִבּוּר וַחֲמַר אֲרַע חֶקְלִין וְכַרְמִין:
דִלְמָא יַטְעֵי יַתְכוֹן חִזְקִיָה לְמֵימָר יְיָ יְשֵׁזְבָנָנָא הֲשֵׁיזְבָא דַחֲלַת עַמְמַיָא גְבַר יַת אַרְעֵהּ מִיַד מַלְכָּא דְאַתּוּר:
אָן דַחֲלַת חֲמָת וְאַרְפָּד אָן דַחֲלַת סְפַרְוָיִם וַאֲרֵי שֵׁיזָבוּ יַת שֹׁמְרוֹן מִן יְדִי:
מַן בְּכָל דַחֲלַת מְדִינָתָא הָאִלֵין דְשֵׁיזָבוּ יַת אַרְעֲהוֹן מִן יְדִי אֲרֵי יְשֵׁיזִיב יְיָ יַת יְרוּשְׁלֵם מִן יְדִי:
וּשְׁתִיקוּ וְלָא אָתִיבוּ יָתֵהּ פִּתְגָמָא אֲרֵי תַפְקִידַת מַלְכָּא הִיא לְמֵימָר לָא תֵתִיבוּנֵהּ:
וַאֲתָא אֶלְיָקִים בַּר חִלְקִיָה דִי מְמַנָא עַל בֵיתָא וְשֶׁבְנָא סַפְרָא וְיוֹאָח בַּר אָסָף דִי מְמַנָא עַל דָכְרָנַיָא לְוַת חִזְקִיָה כַּד מְבַזְעִין לְבוּשֵׁיהוֹן וְחַוִיאוּ לֵיהּ יַת פִּתְגָמֵי רַבְשָׁקֵה:
