וּנְפַקוּ כָּל בְּנֵי יִשְׂרָאֵל וְאִתְכְּנִישַׁת כְּנִשְׁתָּא כְגַבְרָא חָד לְמִדָן וְעַד בְּאֵר שֶׁבַע וְאַרְעָא גִלְעָד לָקֳדָם יְיָ לְמִצְפַּיָא:
וְאִתְעַתָּדוּ רֵישֵׁי כָל עַמָא כָּל שִׁבְטַיָא דְיִשְׂרָאֵל בִּקְהַל עַמָא דַייָ אַרְבַּע מְאָה אַלְפִין גְבַר רִגְלֵי שַׁלְפֵי סַיְפָא:
וּשְׁמָעוּ בְּנֵי בִנְיָמִין אֲרֵי סַלִיקוּ בְנֵי יִשְׂרָאֵל לְמִצְפַּיָא וַאֲמַרוּ בְּנֵי יִשְׂרָאֵל מַלִילוּ אֵיכְדֵין הֲוַת בִּישָׁתָא הָדָא:
וַאֲתֵיב גַבְרָא לֵוָאָה בַּעֲלָא דְאִתְּתָא קְטִלְתָּא וַאֲמַר לְגִבְעָתָא דִי לְשִׁבְטָא בִנְיָמִין עָלִית אֲנָא וּלְחֵינָתִי לִמְבָת:
וְקָמוּ עֲלַי יַתְבֵי גִבְעָתָא וְאַקִיפוּ עֲלַי יַת בֵּיתָא בְּלֵילְיָא עָלַי חֲשִׁיבוּ לְמִקְטַל וְיַת לְחֵינָתִי עַנְיוּהָא וּמִיתַת:
וַאֲחָדִית בִּלְחֵינָתִי וְהַדְמִיתָהּ וּשְׁלַחְתָּהּ בְּכָל תְּחוּם אַחְסָנַת אַרְעָא דְיִשְׂרָאֵל אֲרֵי עֲבָדוּ עֵיצַת חֶטְאִין וּדְלָא כָשַׁר בְּיִשְׂרָאֵל:
הָא כוּלְכוֹן בְּנֵי יִשְׂרָאֵל הָבוּ לְכוֹן פִּתְגָם וְעֵצָה הַלְכָא:
וְקָם כָּל עַמָא כְּגַבְרָא חָד לְמֵימָר לָא נֵיזִיל גְבַר לְמַשְׁכְּנֵיהּ וְלָא נְזוּר גְבַר לְבֵיתֵיהּ:
וּכְעַן דֵין פִּתְגָם דְנַעֲבֵיד לְגִבְעָתָא נִתְמַנֵי עֲלָהּ בְּעַדְבָא:
וּנְדַבֵּר עַשְׂרָא גַבְרִין לִמְאָה לְכָל שִׁבְטַיָא דְיִשְׂרָאֵל וּמְאָה לְאַלְפָא וְאַלְפָא לְרִבּוֹתָא לְמִיסַב זְוָדִין לְעַמָא לְמֶעֱבַּד לְאַיְתָיוּתְהוֹן לְגִבְעָתָא דְבֵית בִּנְיָמִין כְּכָל קְלָנָא דְאִתְעֲבִיד בְּיִשְׂרָאֵל:
וְאִתְכְּנִישׁ כָּל אֱנַשׁ יִשְׂרָאֵל לְקַרְתָּא כְּגַבְרָא חַד שָׁוָן:
וּשְׁלָחוּ שִׁבְטַיָא דְיִשְׂרָאֵל גַבְרִין בְּכָל שִׁבְטָא דְבִנְיָמִין לְמֵימָר מַה בִּישְׁתָא הָדָא דַהֲוָת בְּכוֹן:
וּכְעַן אַיְתוּ יַת גַּבְרַיָא בְּנֵי רִשְׁעָא דִי בְגִבְעָתָא וְנִקְטְלִינוּן וּנְפַלֵי עָבְדֵי בִּישְׁתָא מִיִשְׂרָאֵל וְלָא אָבוּ בְנֵי בִּנְיָמִין לְקַבָּלָא לִמֵימַר אֲחוּהוֹן בְּנֵי יִשְׂרָאֵל:
וְאִתְכְּנִישׁוּ בְנֵי בִנְיָמִין מִן קִרְוַיָא לְגִבְעָתָא לְמִפַּק לְאַגָחָא קְרָבָא עִם בְּנֵי יִשְׂרָאֵל:
וְאִתְמְנִיאוּ בְנֵי בִנְיָמִין בְּיוֹמָא הַהוּא מִן קִרְוַיָא עַסְרִין וְשִׁיתָּא אַלְפִין גַבְרָא שַׁלְפֵי סַיְפָא בַּר מִיַתְבֵי גִבְעֲתָא אִתְמְנִיאוּ שְׁבַע מְאָה גְבַּר עוּלֵים:
מִכָּל עַמָא הָדֵין שְׁבַע מְאָה גְבַר עוּלֵים גַבְרָא גְמִיד בִּידֵיהּ דְיַמִינָא כָּל אִלֵין דְשָׁדָן אַבְנָא בְקִלְעָא וּמַרְמָן בְּבִינַת שַׂעֲרָא וְלָא מְשַׁנָן:
וֶאֱנַשׁ יִשְׂרָאֵל אִתְמַנִיאוּ בַּר מִדְבֵית בִּנְיָמִין אַרְבַּע מְאָה אַלְפִין גַבְרָא שַׁלְפֵי סַיְפָא כָּל אִלֵין גַבְרֵי עָבְדֵי קְרָבָא:
וְקָמוּ וּסְלִיקוּ לְבֵית אֵל וּשְׁאִילוּ בְמֵימְרָא דַייָ וַאֲמַרוּ בְּנֵי יִשְׂרָאֵל מַן יִסַק לָנָא בְּקַדְמֵיתָא לְאַגָחָא קְרָבָא עִם בְּנֵי בִּנְיָמִין וַאֲמַר יְיָ יְהוּדָה בְּקַדְמֵיתָא:
וְקָמוּ בְּנֵי יִשְׂרָאֵל בְּצַפְרָא וּשְׁרוֹ עַל גִבְעָתָא:
וּנְפַק אֱנַשׁ יִשְׂרָאֵל לְאַגָחָא קְרָבָא עִם דְבֵית בִּנְיָמִין וְסַדַרוּ עִמְהוֹן אֱנַשׁ יִשְׂרָאֵל קְרָבָא עַל גִבְעָתָא:
וּנְפַקוּ בְנֵי בִנְיָמִין מִן גִבְעָתָא וְחַבִּילוּ בְּיִשְׂרָאֵל בְּיוֹמָא הַהוּא עַסְרִין וּתְרֵין אַלְפִין גַבְרָא קְטִילִין רְמָן לְאַרְעָא:
וְאִתְקֵף עַמָא אֱנַשׁ יִשְׂרָאֵל וְאוֹסִיפוּ לְסַדָרָא קְרָבָא בְּאַתְרָא דְסַדַרוּ תַּמָן בְּיוֹמָא קַדְמָאָה:
וּסְלִיקוּ בְנֵי יִשְׂרָאֵל וּבְכוֹ קֳדָם יְיָ עַד רַמְשָׁא וּשְׁאִילוּ בְמֵימְרָא דַייָ לְמֵימַר הַאוֹסִיף לְמִקְרַב לְאַגָחָא קְרָבָא עִם בְּנֵי בִנְיָמִין אֲחִי וַאֲמַר יְיָ סַקוּ עֲלֵיהוֹן:
וּקְרִיבוּ בְּנֵי יִשְׂרָאֵל לְוַת בְּנֵי בִנְיָמִין בְּיוֹמָא תִּנְיָנָא:
וּנְפַק בִּנְיָמִין לִקֳדָמוּתְהוֹן מִן גִבְעֲתָא בְּיוֹמָא תִּנְיָנָא וְחַבִּילוּ בִּבְנֵי יִשְׂרָאֵל עוֹד תַּמְנַת עֲסַר אַלְפִין גַבְרָא קְטִילִין רְמָן לְאַרְעָא כָּל אִלֵין שַׁלְפֵי סַיְפָא:
וּסְלִיקוּ כָּל בְּנֵי יִשְׂרָאֵל וְכָל עַמָא וַאֲתוֹ לְבֵית אֵל וּבְכוֹ וִיתִיבוּ תַמָן קֳדָם יְיָ וְצָמוּ בְּיוֹמָא הַהוּא עַד רַמְשָׁא וַאֲסִיקוּ עֲלַוָן וְנִכְסַת קוּדְשִׁין קֳדָם יְיָ:
וּשְׁאִילוּ בְנֵי יִשְׂרָאֵל בְּמֵימְרָא דַייָ וְתַמָן אֲרוֹן קְיָמָא דַייָ בְּיוֹמַיָא הָאִנוּן:
וּפִינְחָס בַּר אֶלְעָזָר בַּר אַהֲרֹן קָאִים קֳדָמוֹהִי בְּיוֹמַיָא הָאִנוּן לְמֵימָר הַאוֹסֵף עוֹד לְמִפַּק לְאַגָחָא קְרָבָא עִם בְּנֵי בִּנְיָמִין אָחִי אִם אִתְמְנַע וַאֲמַר יְיָ סַקוּ אֲרֵי מְחָר אֶמְסָרִינוּן בִּידָךְ:
וְשַׁוִי יִשְׂרָאֵל כְּמָנִין עַל גִבְעֲתָא סְחוֹר סְחוֹר:
וּסְלִיקוּ בְנֵי יִשְׂרָאֵל לְוַת בְּנֵי בִנְיָמִין בְּיוֹמָא תְּלִיתָאָה וְסַדָרוּ קְרָבָא עַל גִבְעֲתָא כִּזְמַן בִּזְמָן:
וּנְפָקוּ בְּנֵי בִּנְיָמִין לִקְדָמוּת עַמָא אִתְנְגִידוּ מִן קַרְתָּא וּשְׁרִיאוּ לְקַטָלָא מֵעַמָא קְטִילִין כִּזְמַן בִּזְמָן בִּכְבֵשׁ אָרְחָתָא דַחֲדָא סַלְקָא לְבֵית אֵל וַחֲדָא לְגִבְעֲתָא בְּחַקְלָא כִּתְלָתִין גַבְרָא בְּיִשְׂרָאֵל:
וַאֲמַרוּ בְּנֵי בִּנְיָמִין תְּבִירִין אִנוּן קֳדָמָנָא כִּדְבְּקַדְמֵיתָא וּבְנֵי יִשְׂרָאֵל אֲמַרוּ נְפַךְ וְנִגוֹדִנּוּן מִן קַרְתָּא לְכִבְשַׁיָא:
וְכָל אֱנַשׁ יִשְׂרָאֵל קָמוּ מֵאַתְרֵיהוֹן וְסַדָרוּ קְרָבָא בְּמֵישְׁרֵי יְרִיחוֹ וּכְמָנָא יִשְׂרָאֵל מַגִיחַ מֵאַתְרֵיהּ מִמֵישַׁר גִבְעָתָא:
וַאֲתוֹ מִדָרוֹם לְגִבְעָתָא עַסְרָא אַלְפִין גְבַר עוּלֵים מִכָּל יִשְׂרָאֵל וְעָבְדֵי קְרָבָא תַקִיפוּ וְאִנוּן לָא יָדְעִין אֲרֵי קְרִיבַת דְתֵיתֵי לְהוֹן בִּישְׁתָּא:
וְתַבֵּר יְיָ יַת בִּנְיָמִין קֳדָם יִשְׂרָאֵל וְחַבִּילוּ בְּנֵי יִשְׂרָאֵל בִּדְבֵית בִּנְיָמִין בְּיוֹמָא הַהוּא עַסְרִין וְחַמְשָׁא אַלְפִין וּמֵאָה גַבְרָא כָּל אִלֵין שַׁלְפֵי סַיְפָא:
וַחֲזוֹ בְנֵי בִנְיָמִין אֲרֵי אִתְבָּרוּ וִיהֲבוּ אֱנַשׁ יִשְׂרָאֵל אֲתַר לְבִנְיָמִין אֲרֵי אִתְרְחִיצוּ עַל כְּמָנָא דְשַׁוִיאוּ עַל גִבְעֲתָא:
וּכְמָנָא אוֹחִי וְאִתְנְגִיד עַל גִבְעֲתָא וְאִתְנְגִיד כְּמָנָא וּמְחָא יַת כָּל קַרְתָּא לְפִּתְּגָם דְחָרֶב:
וְזִמְנָא הֲוָה מְתַקֵן לֶאֱנַשׁ יִשְׂרָאֵל עִם כְּמָנָא דְיִסְגוּן לְאַסָקוּתְהוֹן יְטוּר דִתְנָן מִן קַרְתָּא:
וְאִתְחֲזַר אֱנַשׁ יִשְׂרָאֵל בִּקְרָבָא וּדְבֵית בִּנְיָמִין שְׁרִיאוּ לְקַטָלָא קְטִילַיָא בֶאֱנַשׁ יִשְׂרָאֵל כִּתְלָתִין גַבְרָא אֲרֵי אֲמַרוּ בְּרַם מִתְּבַר תְּבִירִין אִנוּן קֳדָמָנָא כִּקְרָבָא קַדְמָאָה:
וּתְנָנָא שְׁרֵי לְאַסָקָא מִן קַרְתָּא יְטוּר דִתְנָן וְאִתְפְּנִיאוּ דְּבֵית בִּנְיָמִין בַּתְרֵיהוֹן וְהָא סְלִיק תְּנָנָא דְקַרְתָּא לְצֵית שְׁמַיָא:
וֶאֱנַשׁ יִשְׂרָאֵל אִתְחַזֵר וְאִתְבְּהֵיל אֱנַשׁ בִּנְיָמִין אֲרֵי חֲזָא אֲרֵי קְרִיבַת דִתֵיתֵי לְהוֹן בִּישְׁתָּא:
וְאִתְפְּנִיאוּ קֳדָם אֱנָשׁ יִשְׂרָאֵל לְאוֹרַח מַדְבְּרָא וְעָבְדֵי קְרָבָא אַדְבִּיקוּנוּן וֶאֱנָשׁ כְּמָנָא קָמוּ מִקִרְוֵיהוֹן קָטְלִין בְּהוֹן מִכָּא וּמִכָּא:
טַלְטִילוּ יַת דְבֵית בִּנְיָמִין רְדָפוּנוּן מִבֵּית נְיַחֲהוֹן טְרָדוּנוּן עַד לָקֳבֵיל גִבְעֲתָא מִמַדְנַח שִׁמְשָׁא:
וְאִתְקְטָלוּ מִדְבֵית בִּנְיָמִין תַּמְנַת עֲסַר אַלְפִין גַבְרָא יַת כָּל אִלֵין גַבְרָא עַבְדֵי קְרָבָא:
וְאִתְפְּנִיאוּ וְאַפִּיכוּ לְמַדְבְּרָא לְכֵיף רִמוֹן וְאִטַמָרוּ בְּכִיבְשַׁיָא חַמְשָׁא אַלְפִין גַבְרָא וְאַדְבִּיקוּ בַתְרֵיהוֹן עַד גִדְעֹם וּקְטִילוּ מִנְהוֹן תְּרֵין אַלְפִין גַבְרָא:
וַהֲווֹ כָּל דְאִתְקְטִילוּ מִדְבֵית בִּנְיָמִין עַסְרִין וְחַמְשָׁא אַלְפִין גַבְרָא שַׁלְפֵי סַיְפָא בְּיוֹמָא הַהוּא יַת כָּל אִלֵין גַבְרֵי עָבְדֵי קְרָבָא:
וְאִתְפְּנִיאוּ וְאַפִּיכוּ לְמַדְבְּרָא לְכֵיף רִמוֹן שִׁית מְאָה גַבְרָא וִיתִיבוּ בְּכֵיף רִמוֹן אַרְבְּעָא יַרְחִין:
וֶאֱנַשׁ יִשְׂרָאֵל תָּבוּ לְוַת בְּנֵי בִנְיָמִין וּמְחוֹנוּן לְפִּתְגַם דְחָרֶב מִקִרְוֵיהוֹן גַמְרִינוּן עַד בְּעִירָא עַד כָּל דְאִשְׁתְּכַח אַף כָּל קִרְוַיָא דְאִשְׁתַּכָּחָא אוֹקִידוּ בְּנוּרָא:
