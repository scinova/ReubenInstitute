בִּשְׁנַת עַשְׂרִין וּשְׁבַע שְׁנִין לְיָרָבְעָם מַלְכָּא דְיִשְׂרָאֵל מְלַךְ עֲזַרְיָה בַר אֲמַצְיָה מֶלֶךְ שִׁבְטָא דְבֵית יְהוּדָה:
בַּר שִׁית עַסְרֵי שְׁנִין הֲוָה כַּד מְלַךְ וְחַמְשִׁין וְתַרְתֵּין שְׁנִין מְלַךְ בִּירוּשְׁלֵם וְשׁוּם אִמֵיהּ יְכָלְיָהּ מִירוּשְׁלֵם:
וַעֲבַד דְכָשַׁר קֳדָם יְיָ כְּכֹל דַעֲבַד אֲמַצְיָה אֲבוּהִי:
לְחוֹד בָּמָתָא לָא עָטָרָא עַד כְּעַן עַמָא מְדַבְּחִין וּמַסְקִין בּוּסְמָא עַל בָּמָתָא:
וְאַיְתֵי יְיָ מַכְתַּשׁ עַל מַלְכָּא וַהֲוָה סְגִיר עַל יוֹם מוֹתֵיהּ וִיתֵיב בַּר מִן יְרוּשְׁלֵם וְיוֹתָם בַּר מַלְכָּא מְמַנָא עַל בֵּיתָא דָאֵין יַת עַמָא דְאַרְעָא:
וּשְׁאַר פִּתְגָמֵי עֲזַרְיָהוּ וְכָל דִי עֲבַד הֲלָא אִינוּן כְּתִיבִין עַל סְפַר פִּתְגָמֵי יוֹמַיָא לְמַלְכֵי דְבֵית יְהוּדָה:
וּשְׁכִיב עֲזַרְיָה עִם אֲבָהָתוֹהִי וּקְבָרוּ יָתֵיהּ עִם אֲבָהָתוֹהִי בְּקַרְתָּא דְדָוִד וּמְלַךְ יוֹתָם בְּרֵיהּ תְּחוֹתוֹהִי:
בִּשְׁנַת תְּלָתִין וְתַמְנֵי שְׁנִין לַעֲזַרְיָה מֶלֶךְ שִׁבְטָא דְבֵית יְהוּדָה מְלַךְ זְכַרְיָהוּ בַר יָרָבְעָם עַל יִשְׂרָאֵל בְּשֹׁמְרוֹן שִׁתָּא יַרְחִין:
וַעֲבַד דְבִישׁ קֳדָם יְיָ כְּמָא דַעֲבַדוּ אֲבָהָתוֹהִי לָא סְטָא מֵחוֹבֵי יָרָבְעָם בַּר נְבָט דְחַיֵב יַת יִשְׂרָאֵל:
וּמְרַד עֲלוֹהִי שַׁלֻם בַּר יָבֵשׁ וּמְחָהִי קֳדָם עַמָא וְקַטְלֵיהּ וּמְלַךְ תְּחוֹתוֹהִי:
וּשְׁאָר פִּתְגָמֵי זְכַרְיָה הָא אִינוּן כְּתִיבִין עַל סְפַר פִּתְגָמֵי יוֹמַיָא לְמַלְכֵי יִשְׂרָאֵל:
הוּא פִּתְגָמָא דַייָ דִי מַלֵיל עִם יֵהוּא לְמֵימָר בְּנִין רְבִיעָאִין יֵתְבוּן לָךְ עַל כּוּרְסֵי מַלְכוּתָא דְיִשְׂרָאֵל וַהֲוָה כֵן:
שַׁלוּם בַּר יָבֵשׁ מְלַךְ בִּשְׁנַת תְּלָתִין וּתְשַׁע שְׁנִין לְעֻזִיָה מֶלֶךְ שִׁבְטָא דְבֵית יְהוּדָה וּמְלַךְ יְרַח יוֹמִין בְּשֹׁמְרוֹן:
וּסְלִיק מְנַחֵם בַּר גָדִי מִתִּרְצָה וַאֲתָא לְשֹׁמְרוֹן וּמְחָא יַת שַׁלוּם בַּר יָבֵשׁ בְּשֹׁמְרוֹן וְקַטְלֵיהּ וּמְלַךְ תְּחוֹתוֹהִי:
וּשְׁאָר פִּתְגָמֵי שַׁלוּם וּמֵירְדֵיהּ דִי מְרַד הָא אִינוּן כְּתִיבִין עַל סִפְרֵי פִתְגָמֵי יוֹמַיָא לְמַלְכֵי יִשְׂרָאֵל:
בְּכֵן מְחָא מְנַחֵם יַת תִפְסַח וְיַת כָּל דִי בָהּ וְיַת תְּחוּמָהָא מִתִּרְצָה אֲרֵי לָא פְתָחַת תַּרְעָהָא לְאַשְׁלָמָא וּמְחָא וְיַת כָּל מְעַדְיָתָהָא בְּזַע:
בִּשְׁנַת תְּלָתִין וּתְשַׁע שְׁנִין לַעֲזַרְיָה מֶלֶךְ שִׁבְטָא דְבֵית יְהוּדָה מְלַךְ מְנַחֵם בַּר גָדִי עַל יִשְׂרָאֵל עֲסַר שְׁנִין בְּשֹׁמְרוֹן:
וַעֲבַד דְבִישׁ קֳדָם יְיָ לָא סְטָא מֵעַל חוֹבֵי יָרָבְעָם בַּר נְבָט דְחַיֵב עַל יִשְׂרָאֵל כָּל יוֹמוֹהִי:
אֲתָא פוּל מַלְכָּא דְאַתּוּר עַל אַרְעָא וִיהַב מְנַחֵם לְפוּל אֶלֶף כִּכְּרִין דִכְסַף לְאַשְׁלָמָא יְדוֹהִי עִמֵיהּ לְאַתְקָפָא מַלְכוּתָא בִּידֵיהּ:
וּרְמָא מְנַחֵם יַת כַּסְפָּא עַל יִשְׂרָאֵל וְעַל כָּל גִבָּרֵי חֵילָא לְמִתַּן לְמַלְכָּא דְאַתּוּר חַמְשִׁין סַלְעִין דִכְסַף לְגַבְרָא חָד וְתַב מַלְכָּא דְאַתּוּר וְלָא אַנְזִיק תַּמָן בְּאַרְעָא:
וּשְׁאָר פִּתְגָמֵי מְנַחֵם וְכָל דִי עֲבַד הֲלָא אִינוּן כְּתִיבִין עַל סְפַר פִּתְגָמֵי יוֹמַיָא לְמַלְכֵי יִשְׂרָאֵל:
וּשְׁכִיב מְנַחֵם עִם אֲבָהָתוֹהִי וּמְלַךְ פְּקַחְיָה בְּרֵיהּ תְּחוֹתוֹהִי:
בִּשְׁנַת חַמְשִׁין שְׁנִין לַעֲזַרְיָה מֶלֶךְ שִׁבְטָא דְבֵית יְהוּדָה מְלַךְ פְּקַחְיָה בַּר מְנַחֵם עַל יִשְׂרָאֵל בְּשֹׁמְרוֹן תַּרְתֵּין שְׁנִין:
וַעֲבַד דְבִישׁ קֳדָם יְיָ לָא סְטָא מֵחוֹבֵי יָרָבְעָם בַּר נְבָט דְחַיֵיב יַת יִשְׂרָאֵל:
וּמְרַד עֲלוֹהִי פֶּקַח בַּר רְמַלְיָהוּ גִבָּרֵיהּ וּמְחָהִי בְשֹׁמְרוֹן בְּאִידְרוֹן בֵּית מַלְכָּא יַת אַרְגוֹב וְיַת אַרְיָא וְעִמֵיהּ חַמְשִׁין גַבְרָא מִבְּנֵי גִלְעָדָאֵי וְקַטְלֵיהּ וּמְלַךְ תְּחוֹתוֹהִי:
וּשְׁאַר פִּתְגָמֵי פְּקַחְיָה וְכָל דִי עֲבַד הָא אִינוּן כְּתִיבִין עַל סְפַר פִּתְגָמֵי יוֹמַיָא לְמַלְכֵי יִשְׂרָאֵל:
בִּשְׁנַת חַמְשִׁין וְתַרְתֵּין שְׁנִין לַעֲזַרְיָה מֶלֶךְ שִׁבְטָא דְבֵית יְהוּדָה מְלַךְ פֶּקַח בַּר רְמַלְיָהוּ עַל יִשְׂרָאֵל בְּשֹׁמְרוֹן עַסְרִין שְׁנִין:
וַעֲבַד דְבִישׁ קֳדָם יְיָ לָא סְטָא מִן חוֹבֵי יָרָבְעָם בַּר נְבָט דְחַיֵב יַת יִשְׂרָאֵל:
בְּיוֹמֵי פֶקַח מַלְכָּא דְיִשְׂרָאֵל אֲתָא תִּגְלַת פִּלְאֶסֶר מַלְכָּא דְאַתּוּר וּשְׁבָא יַת עִיוֹן וְיַת אָבֵל בֵּית מַעֲכָא וְיַת יָנוֹחַ וְיַת קֶדֶשׁ וְיַת חָצוֹר וְיַת גִלְעָדָא וְיַת גְלִילָא כֹּל אַרְעָא שֵׁבֶט נַפְתָּלִי וְאַגְלִינוּן לְאַתּוּר:
וּמְרֵד מֵירְדָא הוֹשֵׁעַ בַּר אֵלָה עַל פֶּקַח בַּר רְמַלְיָהוּ וּמְחָהִי וְקַטְלֵיהּ וּמְלַךְ תְּחוֹתוֹהִי בִּשְׁנַת עַסְרִין לְיוֹתָם בַּר עֻזִיָה:
וּשְׁאַר פִּתְגָמֵי פֶּקַח וְכֹל דִי עֲבַד הָא אִינוּן כְּתִיבִין עַל סֵפֶר פִּתְגָמֵי יוֹמַיָא לְמַלְכֵי יִשְׂרָאֵל:
בִּשְׁנַת תַּרְתֵּין לְפֶקַח בַּר רְמַלְיָהוּ מַלְכָּא דְיִשְׂרָאֵל מְלַךְ יוֹתָם בַּר עֻזִיָהוּ מֶלֶךְ שִׁבְטָא דְבֵית יְהוּדָה:
בַּר עַשְׂרִין וַחֲמֵשׁ שְׁנִין הֲוָה כַד מְלַךְ וְשִׁית עַשְׂרֵי שְׁנִין מְלַךְ בִּירוּשְׁלֵם וְשׁוּם אִמֵיהּ יְרוּשָׁא בַּת צָדוֹק:
וַעֲבַד דְכָשַׁר קֳדָם יְיָ כְּכָל דַעֲבַד עֻזִיָה אֲבוּהִי עֲבַד:
לְחוֹד בְּמָתָא לָא עָטָרָא עַד כְּעַן עַמָא מַדְבְּחִין וּמַסְקִין בּוּסְמַיָא עַל בָּמָתָא הוּא בְּנָא יַת תְּרַע בֵּית מַקְדְשָׁא דַייָ עִלָאָה:
וּשְׁאַר פִּתְגָמֵי יוֹתָם וְכָל דִי עֲבַד הֲלָא אִינוּן כְּתִיבִין עַל סְפַר פִּתְגָמֵי יוֹמַיָא לְמַלְכֵי דְבֵית יְהוּדָה:
בְּיוֹמַיָא הָאִינוּן שָׁרֵי יְיָ לְגָרָאָה בִּדְבֵית יְהוּדָה רְצִין מַלְכָּא דַאֲרָם וְיַת פֶּקַח בַּר רְמַלְיָהוּ:
וּשְׁכִיב יוֹתָם עִם אֲבָהָתוֹהִי וְאִתְקְבַר עִם אֲבָהָתוֹהִי בְּקַרְתָּא דְדָוִד אֲבוּהִי וּמְלַךְ אָחָז בְּרֵיהּ תְּחוֹתוֹהִי:
