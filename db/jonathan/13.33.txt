וַהֲוָה פִתְגַם נְבוּאָה מִן קֳדָם יְיָ עִם יִרְמְיָה תִּנְיָנוּת וְהוּא עוֹד כְּעַן כְּלֵי בְּדָרַת בֵּית אֲסִירַיָא לְמֵימָר:
כִּדְנַן אֲמַר יְיָ דַעֲבָדָהּ יְיָ בָּרָא יָתָהּ לְאַתְקְנוּתָהּ יְיָ שְׁמֵיהּ:
צַלִי קֳדָמַי וַאֲקַבֵּל צְלוֹתָךְ וַאֲחַוֵי לָךְ רַבְרְבָן וּנְטִירָן דְלָא יְדַעְתִּינוּן:
אֲרֵי כִדְנַן אֲמַר יְיָ אֱלָהָא דְיִשְׂרָאֵל עַל בָּתֵּי קַרְתָּא הָדָא וְעַל בָּתֵּי מַלְכַיָא דְבֵית יְהוּדָה דְמִתָרְעִין לְהוֹן וְצָבְרִין מִלֵיתָא לְתַקָפָא שׁוּרָא מִן קֳדָם דְקַטְלִין בְּחַרְבָּא:
אָתָן לְאַגָחָא קְרָבָא עִם כַּסְדָאֵי וּלְמַלְיוּתְהוֹן יַת פִּגְרֵי בְנֵי אֱנָשָׁא דְאַלְקֵיתִי בְרוּגְזִי וּבְחֵימְתִי וּדְסַלְקֵית שְׁכִנְתִּי מִן קַרְתָּא הָדָא עַל כָּל בִישַׁתְהוֹן:
הָא אֲנָא מַיְתִי לְהוֹן אָסוּ וּפוּרְקַן וְאַסִינוּן וְאֶגְלֵי לְהוֹן תְּרַע תְיוּבָא וֶאֱחַוִינוּן דִיהָכוּן בְּאוֹרַח שָׁלֵם וּקְשׁוֹט:
וַאֲתֵיב יַת גָלוּת יְהוּדָה וְיַת גָלוּת יִשְׂרָאֵל וַאֲקַיְמִינוּן כִּדְבְקַדְמֵיתָא:
וֶאֱדַכִּינוּן מִכָּל חוֹבֵיהוֹן דְחָבוּ קֳדָמַי וְאֶשְׁבּוֹק לְכָל חוֹבֵיהוֹן דְחָבוּ קֳדָמַי וְדִמְרָדוּ בְמֵימְרִי:
וִיהוֹן קֳדָמַי לְשׁוּם דְחֶדְוָא לְתוּשְׁבְּחָא וְלִיקַר לְכָל עַמְמֵי אַרְעָא דִי יִשְׁמְעוּן יַת כָּל טַבְתָּא דִי אֲנָא עָבֵיד עִמְהוֹן וִיזוּעוּן וִידַחֲלוּן עַל כָּל טַבְתָא וְעַל כָּל שְׁלָמָא דַאֲנָא עָבֵיד לְהוֹן:
כִּדְנַן אֲמַר יְיָ עוֹד יִשְׁתְּמַע בְּאַתְרָא הָדֵין דְאַתּוּן אָמְרִין חָרוֹב הוּא מִבְּלִי אֱנָשָׁא וּמִבְּלִי בְעִירָא בְּקִרְוַיָא דְבֵית יְהוּדָה וּבְשׁוּקֵי יְרוּשְׁלֵם דְצַדְיָן מִבְּלִי אֱנָשָׁא וּמִבְּלִי יָתֵיב וּמִבְּלִי בְעִירָא:
קַל בִּיעַ וְקַל חֶדְוָה קַל חַתְנִין וְקַל כַּלָן קַל דְאָמְרִין אוֹדוּ קֳדָם יְיָ צְבָאוֹת אֲרֵי טַב יְיָ אֲרֵי לְעַלְמָא טִיבוּתֵיהּ מַעֲלֵי תוֹדְתָא לְבֵית מַקְדְשָׁא דַייָ אֲרֵי אָתֵיב יַת גָלוּת אַרְעָא כִּדְבְקַדְמֵתָא אֲמַר יְיָ:
כִּדְנַן אֲמַר יְיָ צְבָאוֹת עוֹד יְהֵי בְּאַתְרָא הָדֵין דְחָרוּב מִבְּלִי אֱנָשָׁא וּמִבְּלִי בְעִירָא וְכָל קִרְוֵיהוֹן דִירְוַת בֵּית מֵישְׁרֵי רָעֲיָן וְחוֹטְרִין דְעָן:
בְּקִרְוֵי טוּרָא בְּקִרְוֵי שְׁפֶלְתָּא וּבְקִרְוֵי דָרוֹמָא וּבְאֲרַע שְׁבַט בִּנְיָמִן וּבְסַחְרָנֵי יְרוּשְׁלֵם וּבְקִרְוַיָא דְבֵית יְהוּדָה עוֹד יִתְנֵהּ עַמָא עַל יְדֵי מְשִׁיחָא אֲמַר יְיָ:
הָא יוֹמַיָא אָתָן אֲמַר יְיָ וַאֲקֵים יַת פִּתְגְמָא תַקְנָא דְמָלֵילִית עַל בֵּית יִשְׂרָאֵל וְעַל בֵּית יְהוּדָה:
בְּיוֹמַיָא הָאִינוּן וּבְעִדָנָא הַהִיא אַקֵים לְדָוִד מְשִׁיחַ דְצִדְקָא וְיַעֲבֵּיד דִין וּקְשׁוֹט וּזְכוּ בְּאַרְעָא:
בְּיוֹמַיָא הָאִינוּן יִתְפָּרְקָן דְבֵית יְהוּדָה וִירוּשְׁלֵם תִּשְׁרֵי לְרוֹחֲצָן וְדֵין שְׁמָהּ דִיהוֹן קָרָן לָהּ יִתְעַבְּדָן לָנָא זַכְוָן מִן קֳדָם יְיָ בְּגַוָהּ:
אֲרֵי כִדְנַן אֲמַר יְיָ לָא יִפְסוֹק לְדָוִד גְבַר יָתֵיב עַל כּוּרְסֵי מַלְכוּתָא דְבֵית יִשְׂרָאֵל:
וּלְכַהֲנַיָא לֵיוָאֵי לָא יִפְסוֹק גְבַר מִן קֳדָמַי מַסִיק עֲלָוָן וּמַקְרֵיב קוּרְבָּן וְעָבֵיד נִכְסַת קוּדְשִׁין כָּל יוֹמַיָא:
וַהֲוָה פִתְגַם נְבוּאָה מִן קֳדָם יְיָ עִם יִרְמְיָה לְמֵימָר:
כִּדְנַן אֲמַר יְיָ כְּמָא דְלֵית אֶפְשַׁר דְיִבְטֵיל קְיָמִי דַאֲקִימֵית עִם יְמָמָא וְעִם לֵילְיָא וּבְדִיל דְלָא לְמֶהֱוֵי יֵימָם וְלֵילֵי בְּעִדָנְהוֹן:
אַף קְיָמַי לָא יְבַטֵיל דַאֲקֵימִית עִם דָוִד עַבְדִי מִלְמֶהֱוֵי לֵיהּ בַּר מְלִיךְ עַל כּוּרְסֵיהּ וְיַת לֵוָאֵי כַהֲנַיָא דִמְשַׁמְשִׁין קֳדָמַי:
כְּמָה דְלֵית אֶפְשַׁר דְיִתְמְנוּן חֵילֵי שְׁמַיָא וְלָא יִתְכִיל חָלָא דְיַמָא כֵּן אַסְגֵי יַת זַרְעָא דָוִד עַבְדֵי וְיַת לֵוָאֵי דִמְשַׁמְשִׁין קֳדָמַי:
וַהֲוָה פִּתְגָם נְבוּאָה מִן קֳדָם יְיָ עִם יִרְמְיָה לְמֵימָר:
הֲלָא חֲזֵיתָא מָא עַמָא הָדֵין מְמַלְלִי לְמֵימָר תַּרְתֵּין זַרְעֲיָתָא דְאִתְרְעֵי יְיָ בְּהוֹן וְרָחִיקִינוּן וְיַת עַמִי מַרְגְזִין מִלְמֶהֱוֵי עוֹד עַם מְשַׁמֵשׁ קֳדָמַי כְּקִבְלֵיהוֹן:
כִּדְנַן אֲמַר יְיָ כְּמָא דְלֵית אֶפְשַׁר דִיבַטֵיל קְיָמִי דַאֲקִימֵית עִם יְמָמָא וְעִם לֵילְיָא כֵּן גְזֵירַת שְׁמַיָא וְאַרְעָא לָא שַׁוִיתוּן דְיֶעְדוֹן:
אַף זַרְעָא דְיַעֲקֹב וְדָוִד עַבְדִי לָא אַרְחִיק מִלְקָרָבָא מִבָּנֵיהוֹן עָבְדִין שׁוּלְטַן עַל זַרְעֵיהּ דְאַבְרָהָם יִצְחָק וְיַעֲקֹב אֲרֵי אֲתֵיב יַת גָלוּתְהוֹן וְאֶרְחַם עֲלֵיהוֹן:
