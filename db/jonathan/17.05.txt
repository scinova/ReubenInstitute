קַבִּילוּ יַת פִּתְגְמָא הָדֵין דִי אֲנָא נְטֵיל עֲלֵיכוֹן אִילְיָא בֵּית יִשְׂרָאֵל:
נָפְלָת לָא תוֹסִיף לְמֵיקָם שַׁתָּא הָדָא כְּנִשְׁתָּא דְיִשְׂרָאֵל אִתְרְטִישַׁת עַל אַרְעָא לֵית דִיקַיְמִינָהּ:
אֲרֵי כִדְנַן אֲמַר יְיָ אֱלֹהִים קַרְתָּא דְיִפְקוּן מִנָהּ אַלְפָא יִשְׁתָּאֲרוּן בָּה מְאָה וּדְיִפְקוּן מִנָה מְאָה יִשְׁתָּאֲרוּן בָּהּ עַד עַסְרָא לְבֵית יִשְׂרָאֵל:
אֲרֵי כִדְנַן אֲמַר יְיָ לְבֵית יִשְׂרָאֵל תְּבָעוּ דְחַלְתִּי וְאִתְקַיָמוּ:
וְלָא תִתְבְּעוּן לְמֵיהַךְ לְבֵית אֵל וּלְגִלְגָלָא לָא תֵיתוּן וְלִבְאֵר שָׁבַע לָא תֶעְבְּרוּן אֲרֵי דִי בְּגִלְגָלָא מִגְלָא יִגְלוּן וּדְפָלְחִין לְטַעֲוָתָא בְּבֵית אֵל יְהוֹן לִלְמָא:
תְּבָעוּ יַת דַחַלְתָּא דַייָ וְאִתְקַיָמוּ דִלְמָא יִדְלַק כְּאֶשְׁתָּא רוּגְזֵיהּ בְּבֵית יוֹסֵף וִישֵׁיצֵיהּ וְלָא יְהֵי חָיַס בְּחוֹבִין דַהֲוֵיתוּן פָּלְחִין לְטַעֲוָתָא בְּבֵית אֵל:
דְהָפְכִין לְגִידָה דִינָא וּזְכוּתָא לְאַרְעָא רָטָשׁוּ:
שְׁבָקוּ לְמִדְחַל מִן קֳדָם דַעֲבַד כִּימָה וּכְסִילָא וְהָפֵךְ לִנְהוֹר קָבֵל וִימָם וְלֵילֵי מַקְבֵּיל דַאֲמַר לְכַנָשָׁא מַשִׁרְיָן סַגִיאָן כְּמֵי יַמָא וּמְבַדַר לְהוֹן עַל אַפִּי אַרְעָא יְיָ שְׁמֵיהּ:
דְמִגַבַּר חֲלָשִׁין עַל תַּקִיפִין וּבְזִיזִין עַל כְּרַכֵּי תַּקִיפִין מַשְׁלִיט:
סְנָן לִדְמוֹכַח לְהוֹן בְּשַׁעֲרֵי בָתֵּי דִינָא פִּתְגָמֵי אוֹרַיְתָא וְלִדְמַלֵיל כֵּיוַנְתָּא מְעַקְמִין:
בְּכֵן חֲלַף בְּמִבִּזְכוֹן מִסְכֵּנַיָא וּמְנַסְכוֹן חֲשִׁיכַיָא בָתֵּי גְזִית בְּנִיתוּן וְלָא תֵיתְבוּן בְּהוֹן כַּרְמֵי חֶמְדָתָא נְצִבְתּוּן וְלָא תִשְׁתּוּן יַת חַמְרְהוֹן:
אֲרֵי גְלֵי קֳדָמַי דְסַגִיאִין מֵירְדֵיכוֹן וְתַקִיפָן חוֹבָתֵיכוֹן מְעִיקִין לֵיהּ לְזַכָּאָה בְּדִיל לְקַבָּלָא מָמוֹן דִשְׁקַר וְדִין חֲשִׁיכַיָא בְּבֵית כְּנִישְׁתֵּיהִוֹן מַסְטָן:
בְּכֵן מַלְפַיָא בְּעִדָנָא הַהִיא מִן קֳדָם רַשִׁיעַיָא שְׁתִיקוּ אֲרֵי עִדָן בִּישׁ הוּא:
תְּבָעוּ לְאוֹטָבָא וְלָא לְאַבְאָשָׁא בְּדִיל דְתִתְקַיְמוּן וִיהֵי כְעַן מֵימְרָא דַייָ אֱלֹהִים צְבָאוֹת בְּסַעְדְכוֹן כְּמָא דַאֲמַרְתּוּן:
סְגוּ לְאַבְאָשָׁא וְרִיחֲמוּ לְאַיְטָבָא וְאַתְקִינוּ בְּבֵית כְּנִישַׁתְכוֹן דִינָא מָאִים יִתְרַחֵם מִן קֳדָם יְיָ אֱלָהָא צְבָאוֹת עַל שְׁאָרָא דְיוֹסֵף:
בְּכֵן כִּדְנַן אֲמַר יְיָ אֱלֹהֵי צְבָאוֹת יְיָ בְּכָל רְחוֹבַיָא מִסְפְּדָא וּבְכָל שׁוּקַיָא יֵימְרוּן וַי וַי וִיעָרְעוּן אִכָּר לְאֵבָל וְעָבַד מִסְפְּדָא לִדְיָרְעֵי אִלְיָא:
וּבְכָל כַּרְמַיָא מִסְפְּדָא אֲרֵי אִתְגְלֵי לְמֶעְבַּד פּוּרְעֲנוּת דִין בְּגַוֵיךְ אֲמַר יְיָ:
וַי לִמְחַמְדִין לְיוֹמָא דַעֲתִיד לְמֵיתֵי מִן קֳדָם יְיָ לְמָא דְנַן לְכוֹן יוֹמָא דַעֲתִיד לְמֵיתֵי מִן קֳדָם יְיָ הוּא חֲשׁוֹךְ וְלָא נְהוֹר:
כְּמָא דִי יַעֲרוֹק גַבְרָא מִן קֳדָם אַרְיָא וִיעַרְעִנֵיהּ דוּבָא וְיֵיעוֹל לְבֵיתָא וְיִסְמוֹךְ יְדֵיהּ עַל כָּתְלָא וְיַכְתִנֵיהּ חִוְיָא:
הֲלָא חֲשׁוֹךְ יוֹמָא דְעָתִיד לְמֵיתֵי מִן קֳדָם יְיָ וְלָא נְהוֹר וּקְבַל וְלָא זֵהוֹר לֵיהּ:
סְגֵיתִי רְחִיקֵת חֲגֵיכוֹן וְלָא אֲקַבֵּל בְּרַעֲוָא קוּרְבַּן כְּנִשַׁתְכוֹן:
אֲרֵי אִם תִּסְקוּן קֳדָמַי עַלְוָן וְקוּרְבְּנֵיכוֹן לָא לְרַעֲוָא וְנִכְסַת קוּדְשֵׁיכוֹן לָא יִתְקַבְּלוּן קֳדָמַי:
אַעְדֵי מִן קֳדָמַי אִתְרְגוּשַׁת זִמְרָךְ וְנִגוּן כִּנָרָךְ לָא אֶשְׁמָע:
וְיִתְגְלֵי כְמַיָא דִינָא וּזְכוּתָא כְּנַחַל מַגְבַּר:
הֲנִכְסַת קוּדְשִׁין וְקוּרְבָּנִין קְרַבְתּוּן קֳדָמַי בְּמַדְבְּרָא אַרְבְּעִין שְׁנִין בֵּית יִשְׂרָאֵל:
וּנְטַלְתּוּן יַת סִכּוּת פַּת כּוּמָרֵכוֹן וְיַת כִּיוּן צַלְמֵיכוֹן כּוֹכַב טַעֲוַתְכוֹן דִי עֲבַדְתּוּן לְכוֹן:
וְאַגְלֵי יַתְכוֹן מֵהָלְאָה לְדַמֶשֶׂק אֲמַר יְיָ אֱלֹהִים צְבָאוֹת שְׁמֵיהּ:
