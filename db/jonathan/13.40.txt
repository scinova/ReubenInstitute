פִּתְגַם נְבוּאָה דַהֲוָה עִם יִרְמְיָה מִן קֳדָם יְיָ בָּתַר דִשְׁלַח יָתֵיהּ נְבוּזַרְאֲדָן רַב קָטוֹלַיָא מִן רָמָתָא כַּד דְבַר יָתֵיהּ וְהוּא אֲסִיר בְּעִזְקַיָא בְּגוֹ כָל גַלְוַת יְרוּשְׁלֵם וִיהוּדָה דְאִתְגְלִיאוּ לְבָבֶל:
וּדְבַר רַב קָטוֹלַיָא לְיִרְמְיָהוּ וַאֲמַר לֵיהּ יְיָ אֱלָהָךְ גְזַר יַת בִּישְׁתָא הָדָא עַל אַתְרָא הָדֵין:
וְאַיְתֵי וְקַיֵם יְיָ כְּמָא דְמַלֵיל אֲרֵי חַבְתּוּן קֳדָם יְיָ וְלָא קַבֵּילְתּוּן לְמֵימְרֵיהּ וַהֲוָה לְכוֹן פִּתְגָמָא הָדֵין:
וּכְעַן הָא שְׁרִיתָךְ יוֹמָא דֵין מִן עִזְקַיָא דִי עַל יְדָךְ אִם תַּקִין בְעֵינָיךְ לְמֵיתָא עִמִי לְבָבֶל אֲתָא וַאֲשַׁוֵי יַת עֵינִי עֲלָךְ וְאִם בִּישׁ בְּעֵינָךְ לְמֵיתֵי עִמִי לְבָבֶל אִתְמְנַע חֲזֵי כָּל אַרְעָא קָדְמָךְ לִדְתַקִין וְלִדְכָשַׁר בְּעֵינָךְ לְמֵיזַל תַּמָן אֱזֵיל:
וְאִם לֵית אַתְּ צָבֵי לְמֵיתַב וְתוּב לְוַת גְדַלְיָה בַּר אֲחִיקָם בַּר שָׁפָן דְמַנֵי מַלְכָּא דְבָבֶל בְּקִרְוַיָא דְבֵית יְהוּדָה וְתִיב עִמֵיהּ בְּגוֹ עַמָא אוֹ לְכָל דְכָשַׁר בְּעֵינָךְ לְמֵיזַל אֱזֵיל וּמָנֵי לֵיהּ רַב קָטוֹלַיָא מַתְּנַן וּנְבַזְבַּן וְשַׁלְחֵיהּ:
וַאֲתָא יִרְמְיָה לְוַת גְדַלְיָה בַּר אֲחִיקָם לְמִצְפְּיָא וִיתֵיב עִמֵיהּ בְּגוֹ עַמָא דְאִשְׁתְּאָרוּ בְּאַרְעָא:
וּשְׁמָעוּ כָּל רַבָּנֵי חֵילְוָתָא דִי בְּחַקְלָא אִינוּן וְגַבְרֵיהוֹן אֲרֵי מַנִי מַלְכָּא דְבָבֶל יַת גְדַלְיָה בַר אֲחִיקָם בְּאַרְעָא וַאֲרֵי מַנֵי עִמֵיהּ גַבְרַיָא וּנְשַׁיָא וְטַפְלָא וּמֵחֲשִׁיכֵי עַמָא דְאַרְעָא מִדְלָא אִתְגְלִיאוּ לְבָּבֶל:
וַאֲתוֹ לְוַת גְדַלְיָה לְמִצְפְּיָא וְיִשְׁמָעֵאל בַּר נְתַנְיָה וְיוֹנָתָן בַּר קָרֵחַ וּשְׂרָיָה בַר תַּנְחֻמֶת וּבְנֵי עֵיפִי דְמִנְטוֹפָא וִיזַנְיָה בָּר מַעֲכָת אִינוּן וְגַבְרֵיהוֹן:
וְקַיֵם לְהוֹן גְדַלְיָה בַּר אֲחִיקָם בַּר שָׁפָן וּלְגַבְרֵיהוֹן לְמֵימָר לָא תִדַחֲלוּן מִלְמִפְלַח לְכַסְדָאֵי תְּבוּ בְאַרְעָא וּפְלַחוּ יַת מַלְכָּא דְבָבֶל וְיִיטַב לְכוֹן:
וַאֲנָא הָא אֲנָא יָתֵיב בְּמִצְפְּיָא לְמֵיקַם קֳדָם כַּסְדָאֵי דְיֵתוֹן עֲלָנָא וְאַתּוּן כְּנוּשׁוּ חַמְרָא וּדְבֵילָתָא וּמִשְׁחָא וְשַׁווּ בְּמָנֵיכוֹן וּתְבוּ בְּקִרְוֵיכוֹן דַאֲחַדְתּוּן:
וְאַף כָּל יְהוּדָאֵי דִי בְמוֹאָב וּבִבְנֵי עַמוֹן וּבֶאֱדוֹם וְדִי בְכָל מְדִינָתָא שְׁמָעוּ אֲרֵי שְׁבַק מַלְכָּא דְבָבֶל שְׁאָרָא לִדְבֵית יְהוּדָה וַאֲרֵי מַנִי עֲלֵיהוֹן יַת גְדַלְיָה בַּר אֲחִיקָם בַּר שָׁפָן:
וְתָבוּ כָל יְהוּדָאֵי מִכָּל אַתְרַיָא דְאִתְבְּדָרוּ לְתַמָן וַאֲתוֹ לְאַרְעָא דְבֵית יְהוּדָה לְוַת גְדַלְיָהוּ לְמִצְפְּיָא וּכְנָשׁוּ חַמְרָא וּדְבֵילְתָּא סַגִי לַחֲדָא:
וְיוֹחָנָן בַּר קָרֵחַ וְכָל רַבָּנֵי חֵילְוָתָא דִי בְּחַקְלָא אֲתוֹ לְוַת גְדַלְיָהוּ לְמִצְפְּיָא:
וַאֲמָרוּ לֵיהּ הֲמִידַע תֵּדַע אֲרֵי בַעֲלִיס מַלְכָּא דִבְנֵי עַמוֹן שְׁלַח יַת יִשְׁמְעֵאל בַר נְתַנְיָה לְמִקְטְלָךְ נְפַשׁ וְלָא הֶאֱמִין לְהוֹן גְדַלְיָה בַּר אֲחִיקָם:
וְיוֹחָנָן בַּר קָרֵחַ אֲמַר לִגְדַלְיָה בְסִתְרָא בְּמִצְפְּיָא לְמֵימָר אֱזֵיל כְּעַן וְאֶמְחֵי יַת יִשְׁמָעֵאל בַּר נְתַנְיָה וֶאֱנַשׁ לָא יְדַע לְמָה יְקַטְלִינָךְ נְפַשׁ וְיִתְבַּדְרוּן כָּל דְבֵית יְהוּדָה דְאִתְכְּנִישׁוּ לְוָתָךְ וְיוֹבֵיד שְׁאָרָא דְבֵית יְהוּדָה:
וַאֲמַר גְדַלְיָהוּ בַּר אֲחִיקָם לְיוֹחָנָן בַּר קָרֵחַ לָא תַעֲבֵּיד יַת פִּתְגָמָא הָדֵין אֲרֵי שִׁקְרָא אַתְּ מְמַלֵל עַל יִשְׁמָעֵאל:
