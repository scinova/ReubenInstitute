הָא עַבְדִי +מְשִׁיחָא+ _אֶקְרְבִינֵהּ_, בְּחִירִי דְאִתְרְעֵי +בֵּיהּ+ _מֵימְרִי_, אֶתֵּן רוּחָא _דְקוּדְשִׁי_ עֲלוֹהִי, דִינִין לְעַמְמִין _יְגַלֵי_.
לָא יִצְוַח וְלָא יַכְלֵי וְלָא יְרִים לְבָרָא קָלֵהּ.
+עִנְוְתָנַיָא דְאִינוּן דָמָן לְ+קַנְיָא רְעִיעַ לָא יִתַּבְּרוּן, _וַחֲשִׁיכַיָא דִכְבוּצִין_ +עַמִי+ לָא יִטְפוּן; לְקוּשְׁטָא יַפֵיק דִינָא.
לָא יַלְהֵי וְלָא יִלְאֵי, עַד דְ_יַתְקֵן_ בְּאַרְעָא דֵינָא; וּלְאוֹרַיְתֵהּ נַגְוָן יְכַתְּרוּן.
כִּדְנַן אֲמַר +יְי,+ אֱלָהּ +עַלְמָא+ יְיָ, דִי בְרָא שְׁמַיָא וּתְלָנוּן, _שַׁכְלִיל_ אַרְעָא וְ_דַיְרָהָא_; יָהֵיב נִשְׁמְתָא לְעַמָא דִי עֲלָהּ, וְרוּחַ לְדִמְהַלְכִין בָּהּ.
“אֲנָא יְיָ, _רַבִּיתָךְ_ בִּקְשׁוֹט וְאַתְקְפֵית בִּידָךְ; וְ_אַתְקְנִינָךְ_, וְאֶתְּנִינָךְ לִקְיַם עַם לְנֵיהוֹר עַמְמִין.
לְפַתָּחָא עֵינֵי _בֵּית יִשְׂרָאֵל_ +- דְאִינוּן כְּסָמָן מִן אוֹרַיְתָא+; לְאַפָקָא +נַלְוַתְהוֹן+ _מִבֵּינֵי עַמְמַיָא_, דְאִינוּן דָמָן לַאֲסִירִין וּלְמִפְרַקְהוֹן _מִשִׁעְבּוּד מַלְכְּוָתָא_ +דְאִינוּן עֲנִינִין כַּ+אֲסִירֵי קְבָל.
אֲנָא - יְיָ הוּא שְׁמִי; וִיקָרִי +- דְאִתְגְלֵיתִי עֲלֵיכוֹן -+ לְ+עַם+ אוֹחֲרָן לָא אֶתֵּן, וְתוּשְׁבַּחְתִּי לְ_פַלְחֵי צַלְמַיָא_.
קַדְמָיָתָא הָא אֲתָאָה; וְחַדְתָּן אֲנָא מְחַוֵי, עַד לָא אַתְיַן אֶבְסַר יַתְכוֹן.:
שַׁבָּחוּ +קֳדָם+ יְיָ _תּוּשְׁבַּחְתָּא_ חַדְתָּא, +אֱמָרוּ+ _תּוּשְׁבַּחְתֵּהּ_ מִסְיָפֵי אַרְעָא; נַחֲתֵי יַמָא וּמְלָאָהּ, _נַגְוָן_ וְיָתְבֵיהוֹן.:
יְשַׁבַּח מַדְבְּרָא וְקִרְוִין דְיַתְבִין בֵּיהּ פְּצִיחִין יָתְבִין מִדְבַּר עַרְבָאִי; _יְשַׁבְּחוּן מֵיתַיָא_ +כַּד יִפְקוּן מִבָּתֵּי עַלְמֵיהוֹן+, מְרֵישֵׁי טוּרַיָא _יְרִימוּן קָלְהוֹן_.
יְשַׁווּן קֳדָם יְיָ יְקָרָא וְתוּשְׁבַּחְתֵּהּ בְּנַגְוָן יְחַווּן:
יְיָ לְמֶעְבַּד גְבוּרָן אִתְגְלֵי, לְמֶעְבַּד גְבוּרָא מִתְגְלֵי בִּרְגַז; בְּמִלוּל אַף בְּזִיעַ, עַל _בַּעֲלֵי דְבָבוֹהִי_ _מִתְגְלֵי בִּגְבוּרְתֵּהּ_.
_יְהָבִית לְהוֹן אַרְכָּא מֵעַלְמָא, דְאִם יְתוּבוּן לְאוֹרַיְתָא, וְלָא תָבוּ כְּחַבְלִין עַל יְלִידְתָּא - יִתְגְלֵי דִינִי עֲלֵיהוֹן, יִצְדוּן וִיסוּפוּן כַּחֲדָא._
אַחֲרִיב טוּרִין וְרָמָן, וְכָל עִסְבֵּיהוֹן אֲיַבֵּשׁ; וַאֲשַׁוֵי נַהֲרִין לְ_נַגְוָן_, וַאֲגַמִין אֲיַבֵּשׁ.
וָ_אֱדַבֵּר_ _לְבֵית יִשְׂרָאֵל_+, דְדָמִין כְּסָמָן+ - בְּאוֹרַח דְלָא יְדָעוּ, בִּשְׁבִילִין דְלָא אֲלִיפוּ אַדַרְכִינוּן; אֲשַׁוֵי קְבַל קֳדָמֵיהוֹן, לִנְהוֹר וְכַפְלָא לְמֵישְׁרָא- אִלֵין פִּתְגָמַיָא אַעְבְדִינוּן וְלָא אַרְחִיקִינוּן.
יִסְתַּחֲרוּן לַאֲחוֹרָא, _יִבַּהֲתוּן_ _בַּהֲתָא_, פַלְחֵי צַלְמַיָא! - דְאָמְרִין +לִצְלַם+ מַתְּכָא 'אַתּוּן טַעֲוָתָנָא'.
+רַשִׁיעַיָא - דְאִינוּן כְּ+חֵרְשִׁין+  הֲלָא אוּדְנִין לְכוֹן?!+ שְׁמָעוּ! וְחַיָבַיָא - +דְאִינוּן כְּסָמָן - הֲלָא עַיְנִין לְכוֹן?!+ _אִסְתַּכָּלוּ_ וַחֲזוֹ:
_הֲלָא אִם יְתוּבוּן רַשִׁיעַיָא, יִתְקְרוֹן 'עַבְדֵי'. וְחַיָבַיָא - דִנְבִיֵי שְׁלָחִית עֲלֵיהוֹן - אֶלָא רַשִׁיעַיָא עֲתִידִין לְאִשְׁתַּלָמָא פּוּרְעֲנוּת חוֹבֵיהוֹן. בְּרַם, אִם יְתוּבוּן - יִתְקְרוֹן 'עַבְדַיָא דַיָי'._
חֲזֵיתוּן סַגִיאָן - וְלָא נְטַרְתּוּן, אִתְפַּתָּחָא אוּדְנֵיכוּן - וְלָא _קַבֵּילְתּוּן_ +אוּלְפַן+.
יְיָ רָעֵי בְּדִיל לְזַכָּאוּתֵהּ +יִשְׂרָאֵל+, וְרַבֵּי +לְעָבְדֵי+ אוֹרַיְתֵהּ וְ_יַתְקֵיף_ +יַתְהוֹן+.
וְהוּא עַם בָּזִיז וַאֲנִיס, _אִתְחַפִיאוּ בַהֲתָא_ עוּלֵימִין כּוּלְהוֹן, וּבְבָתֵּי יְסוּרִין עֲנִינִין; הֲווֹ לַעֲדִי וְלֵית דִמְשֵׁיזֵיב, לְבִזָא וְלֵית דַאֲמַר אֲתֵיב.
מַן בְּכוֹן יָצֵית דָא, _יְקַבֵּל_ וִ_יסַבַּר_ לְסוֹפָא?!
מַן _מְסַר_ לַעֲדֵי יַעֲקֹב וְיִשְׂרָאֵל לִבְזוֹזִין - הֲלָא יְיָ?! _מִן קֳדָם_ דְחָבוּ _קֳדָמוֹהִי_, וְלָא אֲבוּ לְמֵיחָךְ בְּאוֹרְחָן +דְתַקְנָן קֳדָמוֹהִי+, וְלָא _קַבִּילוּ_ +אוּלְפָן+ אוֹרַיְתֵהּ.
וּשְׁפַךְ עֲלֵיהוֹן חֵימַת רוּגְזֵהּ, וּתְקוֹף עָבְדֵי קְרָבֵהּ; +אַיְתִי עֲלֵיהוֹן+ וּ_קְטָלוּ בְהוֹן_ מִסְחוֹר סְחוֹר וְלָא יְדַעוּ, וְ_שַׁלִיטוּ_ בְהוֹן וְלָא שַׁוִיאוּ +דְחַלְתֵּהּ+ עַל לִבָּא:
