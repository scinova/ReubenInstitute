בְּעִדָנָא הַהִיא אִתְמְרַע אֲבִיָה בַּר יָרָבְעָם:
וַאֲמַר יָרָבְעָם לְאִתְּתֵיהּ קוּמִי כְעַן וְתִשְׁתַּנִין וְלָא יֵדְעוּן דְאַתְּ אִתַּת יָרָבְעָם וְתַהֲכִין לְשִׁלֹה הָא תַמָן אַחִיָה נְבִיָא הוּא מַלֵיל עֲלַי לְמֶהֱוֵי מַלְכָּא עַל עַמָא הָדֵין:
וְתִסְבִין בִּידָךְ עֲסַר טוּלְמִין דִלְחֵם וְכִסְנִין וְזִלוּעַ דִדְבַשׁ וְתַהֲכִין לְוָתֵיהּ הוּא יְחַוֵי לִיךְ מָא יְהֵי לְרַבְיָא:
וַעֲבַדַת כֵּן אִתַּת יָרָבְעָם וְקָמַת וַאֲזַלַת לְשִׁלֹה וַאֲתַת לְבֵית אֲחִיָה וַאֲחִיָה לָא יָכִיל לְמֶחֱזֵי אֲרֵי כְהָאָה עֵינוֹהִי מִסֵיבֵיהּ:
וַיָי אֲמַר לַאֲחִיָה הָא אִתַּת יָרָבְעָם אַתְיָא לְמִתְבַּע פִּתְגַם מִנָךְ עַל בְּרָהּ אֲרֵי מְרַע הוּא כְּדֵין וּכְדֵין תְּמַלֵיל עִמָהּ וִיהִי כְּמֵיעֲלָהּ וְהִיא מִשְׁתַּנְיָא:
וַהֲוָה כַּד שְׁמַע אֲחִיָה יַת קַל רִגְלָאָה עָלַת בְתַרְעָא וַאֲמַר עוּלִי אִתַּת יָרָבְעָם לְמָא דְנַן אַתְּ מִשְׁתַּנְיָא וַאֲנָא שְׁלִיחַ לְאִתְנַבָּאָה עֲלָךְ פִּתְגָמִין קָשָׁן:
אֱזִילִי אֱמָרִי לְיָרְבָעָם כִּדְנַן אֲמַר יְיָ אֱלָהָא דְיִשְׂרָאֵל חֲלַף דְאַפְרִישְׁתָּךְ מִגוֹ עַמָא וּמַנִיתָךְ לְמֶהֱוֵי מַלְכָּא עַל עַמִי יִשְׂרָאֵל:
וְאַעְדֵתִי יַת מַלְכוּתָא מִבֵּית דָוִד וִיהַבְתָּא לָךְ וְלָא הֲוֵיתָא כְּעַבְדִי דָוִד דִי נְטַר פִּקוּדַי וּדְהַלֵיךְ בָתַר פּוּלְחָנִי בְּכָל לִבֵּיהּ לְמֶעְבַד לְחוּד דְכָשַׁר קָדָמַי:
וְאַבְאֵישְׁתָּא לְמֶעְבַּד מִכֹּל דַהֲווֹ קָדָמָךְ וַאֲזַלְתָּא וַעֲבַדְתָּא לָךְ טַעֲוַת עַמְמַיָא וּמַתְּכָן לְאַרְגָזָא קָדָמַי וּפוּלְחָנִי אַרְחֵיקְתָּא מִקָבֵיל עֵינָךְ:
בְּכֵן הָא אֲנָא מַיְתֵי בִישְׁתָּא לְבֵית יָרָבְעָם וֶאֱשֵׁיצֵי לְיָרְבְעָם יָדַע מַדַע וַאֲחִיד וּשְׁבִיק בְּיִשְׂרָאֵל וַאֲפַלֵי בָּתַר בֵּית יָרָבְעָם כְּמָא דִמְגַנְדְרִין בְּגוּלִילִין עַד דִמְסַיְפִין:
דִימוּת לְיָרְבְעָם בְּקַרְתָּא יֵיכְלוּן כַּלְבַיָא וְדִימוּת בְּחַקְלָא יֵיכְלוּן עוֹפָא דִשְׁמַיָא אֲרֵי מֵימְרָא דַייָ גְזַר כֵּן:
וְאַתְּ קוּמִי אֱזִילִי לְבֵיתָךְ בְּמֵיעַל רִגְלַיִךְ לְקַרְתָּא יְמוּת רַבְיָא:
וְיִסְפְּדוּן לֵיהּ כָּל יִשְׂרָאֵל וְיִקְבְּרוּן יָתֵיהּ אֲרֵי דֵין בִּלְחוֹדוֹהִי יִתְכְּנֵישׁ לְיָרָבְעָם לְקִבְרָא חֲלַף דְאִשְׁתְּכַח בֵּיהּ פִּתְגַם דְתַקֵין קֳדָם יְיָ אֱלָהָא דְיִשְׂרָאֵל בְּבֵית יָרָבְעָם:
וִיקִים יְיָ קֳדָמוֹהִי מַלְכָּא עַל יִשְׂרָאֵל דִישֵׁיצֵי יַת בֵּית יָרָבְעָם מִן דְקַיֵים יוֹמָא דֵין וְאַף דְאִתְיְלִיד מִבְּכֵן וּלְהָלָא:
וְיִמְחֵי יְיָ יַת יִשְׂרָאֵל כְּמָא דְנָאֵיד קַנְיָא בְּמַיָא וִיעַקֵר יַת יִשְׂרָאֵל מֵעַל אַרְעָא טַבְתָא הָדָא דִיהַב לַאֲבָהַתְהוֹן וַיַגְלִינוּן מֵעִיבַּר פְרָת חֲלַף דַעֲבַדוּ יַת אֲשֵׁרֵיהוֹן מַרְגְזִין קֳדָם יְיָ:
וְיִמְסַר יַת יִשְׂרָאֵל בְּדִיל חוֹבֵי יָרָבְעָם דְחַב וּדְחַיֵיב יַת יִשְׂרָאֵל:
וְקָמַת אִתַּת יָרָבְעָם וַאֲזַלַת וַאֲתַת לְתִרְצָת הִיא עֲלַת בְּסִיפָא דְבֵיתָא וְדַבְיָא מִית:
וּקְבָרוּ יָתֵיהּ וּסְפָדוּ עֲלוֹהִי כָּל יִשְׂרָאֵל כְּפִתְגָמָא דַייָ דְמַלֵיל בְּיַד עַבְדֵיהּ אֲחִיָה נְבִיָא:
וּשְׁאָר פִּתְגָמֵי יָרָבְעָם דְאַגִיחַ קְרָבָא וְדִמְלָךְ הֲלָא אִינוּן כְּתִיבִין עַל סְפַר פִּתְגָמֵי יוֹמַיָא לְמַלְכַיָא דְיִשְׂרָאֵל:
וְיוֹמַיָא דִמְלַךְ יָרָבְעָם עַסְרִין וְתַרְתֵּין שְׁנִין וּשְׁכִיב עִם אֲבָהָתוֹהִי וּמְלַךְ נָדָב בְּרֵיהּ תְּחוֹתוֹהִי:
וּרְחַבְעָם בַּר שְׁלֹמֹה מְלַךְ עַל דְבֵית יְהוּדָה בַּר אַרְבְּעִין וַחֲדָא שְׁנָא רְחַבְעָם כַּד מְלַךְ וּשְׁבַע עַשְׂרֵי שְׁנִין מְלַךְ בִּירוּשְׁלֵם קַרְתָּא דְאִתְרְעֵי יְיָ לְאַשְׁרָאָה יַת שְׁכִנְתֵּיהּ תַּמָן מִכָּל שִׁבְטַיָא דְיִשְׂרָאֵל וְשׁוּם אִמֵיהּ נַעֲמָה מִבְּנֵי עַמוֹן:
וַעֲבַד דְבֵית יְהוּדָה דְבִישׁ קֳדָם יְיָ וְאַרְגִיזוּ קָדָמוֹהִי מִכָּל דַעֲבַדוּ אֲבָהַתְהוֹן בְּחוֹבֵיהוֹן דִי חָבוּ:
וּבְנוֹ אַף אִינוּן לְהוֹן כָּמָן וְקָמָן וַאֲשֵׁרִין עַל כָּל רְמָא מְנַטְלָא וּתְחוֹת כָּל אִילָן עַבּוּף:
וְאַף נַפְקַת בָּרָא הֲוָה בְּאַרְעָא עֲבַדוּ כְּכֹל תּוּעֲבַת עַמְמַיָא דִי תָרִיךְ יְיָ מִן קֳדָם בְּנֵי יִשְׂרָאֵל:
וַהֲוָה בְּשַׁתָּא חֲמִישָׁתָא לְמַלְכָּא רְחַבְעָם סְלִיק שִׁישַׁק מַלְכָּא דְמִצְרַיִם עַל יְרוּשְׁלֵם:
וּנְסִיב יַת אוֹצָרֵי בֵּית מַקְדְשָׁא דַייָ וְיַת אוֹצָרֵי בֵּית מַלְכָּא וְיַת כּוֹלָא נְסִיב וּנְסִיב יַת כָּל מְגִנֵי דַהֲבָא דַעֲבַד שְׁלֹמֹה:
וַעֲבַד מַלְכָּא רְחַבְעָם חִלוּפֵיהוֹן מְגִנֵי נְחָשָׁא וּמַנֵי עַל רַבָּנֵי רָהֲטַיָא דְנַטְרִין תְּרַע בֵּית מַלְכָּא:
וַהֲוָה בִּזְמַן דַעֲלֵיל מַלְכָּא לְבֵית מַקְדְשָׁא דַייָ נַסְבִין לְהוֹן רָהֲטַיָא וּמְתִיבִין לְהוֹן לְדוּבָא דְרָהֲטַיָא:
וּשְׁאַר פִּתְגָמֵי רְחַבְעָם וְכָל דַעֲבַד הֲלָא אִינוּן כְּתִיבִין עַל סְפַר פִּתְגָמֵי יוֹמַיָא לְמַלְכֵי דְבֵית יְהוּדָה:
וּקְרָבָא הֲוַת בֵּין רְחַבְעָם וּבֵין יָרָבְעָם כָּל יוֹמַיָא:
וּשְׁכִיב רְחַבְעָם עִם אֲבָהָתוֹהִי וְאִתְקְבַר עִם אֲבָהָתוֹהִי בְּקַרְתָּא דְדָוִד וְשׁוּם אִמֵיהּ נַעֲמָה דְמִבְנֵי עַמוֹן וּמְלַךְ אֲבִיָם בְּרֵיהּ תְּחוֹתוֹהִי:
