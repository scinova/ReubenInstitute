חוּתִי וְתִיבִי עַל עַפְרָא מַלְכּוּ כְּנִשְׁתָּא דְבָבֶל תִּיבִי לְאַרְעָא לֵית כּוּרְסֵי יְקָר מַלְכוּת כַּשְׂדָאֵי אֲרֵי לָא תוֹסִיפִין דְיִקְרוּן לָךְ רַכִּיבָא וּמְפַנְקָא:
קַבִּילִי מְרוֹעָא וְעוּלִי בְשִׁעְבּוּד גְלָא יְקַר מַלְכוּתִיךְ אִתְּבַּרוּ שִׁלְטוֹנִיךְ אִתְבַּדְרוּ עַם מַשִׁרְיָתֵיךְ גְלוֹ כְמֵי נַהֲרָא:
תִּתְגְלֵי בַּהֲתָתִיךְ אַף יִתְחֲזֵי קְלָנִיךְ פּוּרְעֲנוּת גְמִירָא אִתְפְּרַע מִינִיךְ וַאֲשַׁנֵי דִינִיךְ מִבְּנֵי אֱנָשָׁא:
פָּרְקָנָא יְיָ צְבָאוֹת שְׁמֵהּ קַדִישָׁא דְיִשְׂרָאֵל:
תְּבִי שַׁתְקָא וְעוּלֵי בְקִבְלָא יְקַר מַלְכוּת כַּסְדָאֵי אֲרֵי לָא תּוֹסִיפִין דְיִקְרוֹן לִיךְ תְּקֵיפַת מַלְכְּוָן:
רְגֵיזִית עַל עַמִי אַפֵיסִית אַחְסַנְתִּי וּמְסַרְתִּינוּן בִידָךְ לָא אִתְמַלֵית עֲלֵיהוֹן רַחֲמִין עַל סָבָא אַתְקֵיפְתָּא מַרְוָתִיךְ לַחְדָא:
וַאֲמַרְתְּ לַעֲלָם אֱהֵי תַקִיפַת מַלְכְּוָן עַד לָא שַׁוִית אִלֵין עַל לִבִּיךְ לָא אִדְכַּרְתְּ לְסוֹפָא:
וּכְעַן שְׁמַעִי דָא מְפַנֶקְתָּא דְיָתְבָא לְרָחֲצָן דְאָמְרָא בְּלִבָּהּ אֲנָא וְלֵית בַּר מִינִי עוֹד לָא אֵיתֵיב אַרְמְלוּ וְלָא אֵדַע תִּיכְלוּ:
וְיָתְיָן לִיךְ תַּרְתֵּין אִלֵין זְמַן בְּיוֹמָא חַד תְּכַל וְאַרְמְלוּ כִּדְשָׁלְמָן יֵיתוּן עֲלָךְ בִּסְגִיאוּת חָרָשָׁךְ בִּתְקוֹף קְסָמָךְ לַחֲדָא:
וְאִתְרְחֵיצְתְּ בְּבִישְׁתִּיךְ אֲמַרְתְּ לֵית דְחָזֵי לִי חוּכְמְתִיךְ וּמַדְעִיךְ הִיא קַלְקַלְתִיךְ וַאֲמַרְתְּ בְּלִבֵּיךְ אֲנָא וְלֵית בַּר מִינִי עוֹד:
וְתֵיתֵי עֲלָךְ בִּישְׁתָּא לָא תֵדְעִין לְמִבְעֵי עֲלָהּ וְתִיפּוֹל עֲלָךְ עָקָא לָא תִיכְּלִין לְאַעְדָיוּתָהּ וְתֵיתֵי עֲלָךְ בִּתְכֵּיף אִתְרְגוּשָׁא לָא תִידְעִין:
קוּמִי כְעַן בְּקִסוּמִיךְ וּבְסַגְיוּת חָרָשִׁיךְ דַהֲוֵית מִתְעַסְקָא בְּהוֹן מִיַנְקוּתִיךְ מָאִים תִּכְלִין לַהֲנָאָה מָאִים תִּכְלִין לְמִתְקַף:
לָאֵית בִּסְגִיאוּת מַלְכַּךְ יְקוּמוּן כְּעַן וְיִפְרְקוּנִיךְ דְמַסְכִּין לְמַזָלַת שְׁמַיָא וּדְחָזָן בְּכוֹכְבַיָא מְהוֹדְעִין זִמְנַיָא מַטְעָן לִיךְ לְמֵימָר כֵּן עֲתִיד לְמֵיתֵי עֲלָךְ יְרַח בִּירָח:
הָא הֲווֹ חֲלָשִׁין כְּקַשָׁא עַמְמַיָא דְתַקִיפִין כְּאֶשְׁתָּא שֵׁיצִיאוּנוּן לָא יְשֵׁיזְבוּן יַת נַפְשְׁהוֹן מִיַד קָטוֹלִין לֵית שְׁאָר וּמְשֵׁיזֵיב אַף לָא אֲתַר לְאִשְׁתֵּיזָבָא בֵיהּ:
כֵּן הֲווֹ בִיךְ עוֹבָדֵי שְׁקָרַיִךְ דַהֲוֵית מִתְעַסְקָא בְּהוֹן מִיַנְקוּתִיךְ שִׁלְטוֹנַי תּוּקְפִיךְ גְבַר לָקֳבֵיל אַפּוֹהִי גְלוּ לֵית דְפָרִיק לִיךְ:
