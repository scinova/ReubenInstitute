וַהֲוָה מַלְכָּא שְׁלֹמֹה מַלְכָּא עַל כָּל יִשְׂרָאֵל:
וְאִלֵין רַבְרְבַיָא דִי לֵיהּ עֲזַרְיָהוּ בַר צָדוֹק כַּהֲנָא:
אֱלִיחוֹרֶף וַאֲחִיָה בְּנֵי שִׁישָׁא סַפְרַיָא יְהוֹשָׁפָט בַּר אֲחִילוּד מְמַנָא עַל דָכְרָנַיָא:
וּבְנָיָהוּ בַר יְהוֹיָדָע מְמַנָא עַל חֵילָא וְצָדוֹק וְאֶבְיָתָר כַּהֲנַיָא:
וַעֲזַרְיָהוּ בַר נָתָן מְמַנָא עַל אִסְטַרְטִיגַיָא וְזָבוּד בַּר נָתָן רַבָּא שׁוֹשְׁבִינָא דְמַלְכָּא:
וַאֲחִישָׁר מְמַנָא עַל בֵּיתָא וַאֲדֹנִירָם בַּר עַבְרָא מְמַנָא עַל מַסְקֵי מִסִין:
וְלִשְׁלֹמֹה תְּרֵי עֲסַר אִסְטַרְטִיגַיָא מְמַנָן עַל כָּל יִשְׂרָאֵל וּמְסוֹבְרִין יַת מַלְכָּא וְיַת אֱנַשׁ בֵּיתֵיהּ יַרְחָא בְּשַׁתָּא מָטֵי לְחָד לְסוֹבָרָא:
וְאִלֵין שְׁמָהַתְהוֹן בַּר חוּר בְּטוּרָא דְבֵית אֶפְרָיִם:
בַּר דֶקֶר בְּמָקַץ וּבְשַׁעֲלַבִים וּבֵית שָׁמֶשׁ וְאֵלוֹן דְמִבֵּית חָנָן:
בַּר חֶסֶד בַּאֲרֻבּוֹת דִילֵיהּ סוֹכוֹ וְכָל אֲרַע חֵפֶר:
בַּר אֲבִינָדָב כָּל פִּלְכֵי דֹאר טָפַת בַּת שְׁלֹמֹה הֲוַת לֵיהּ לְאִתּוּ:
בַּעֲנָא בַּר אֲחִילוּד תַּעֲנָךְ וּמְגִידוֹ וְכָל בֵּית שְׁאָן דִי בִסְטַר צָרְתָנָה מִלְרַע לְיִזְרְעֶאל מִבֵּית שְׁאָן עַד מֵישַׁר מְחוֹלָה עַד מֵעִבַּר לְיָקְמְעָם:
בַּר גֶבֶר בְּרָמוֹת גִלְעָד דִילֵיהּ כַּפְרָנֵי יָאִיר בַּר מְנַשֶׁה דִי בְגִלְעָד דִילֵיהּ בֵּית פְלַךְ טַרְכוֹנָא דִבְמַתְנָן שִׁתִּין קִרְוִין רַבְרְבָן מַקְפַן שׁוּר רָם דִילְהֶן עַבְּרִין דִנְחָשָׁא:
אֲחִינָדָב בַּר עִדוֹא מִדְבֵית מַחְנָיִם:
אֲחִימַעַץ בְּשִׁבְטָא דְבֵית נַפְתָּלִי אַף הוּא נְסִיב יַת בָּשְׂמַת בַּת שְׁלֹמֹה לְאִתּוּ:
בַּעֲנָא בַּר חוּשָׁי בְּשִׁבְטָא דְאָשֵׁר וּבְעָלוֹת:
יְהוֹשָׁפָט בַּר פָּרוּחַ בְּשִׁבְטָא דְיִשָׂשכָר:
שִׁמְעִי בַר אֵלָא בְּשִׁבְטָא דְבֵית בִנְיָמִן:
גֶבֶר בַּר אֻרִי בְּאַרְעָא גִלְעָד אַרְעָא סִיחוֹן מַלְכָּא אֱמוֹרָאָה וְעוֹג מַלְכָּא דְמַתְנָן וְאִסְטַרְטִיגָא חָד דִי מְמַנָא בְּאַרְעָא:
יְהוּדָה וְיִשְׂרָאֵל סַגִיאִין כְחָלָא דְעַל יַמָא לִסְגֵי אָכְלִין וְשָׁתָן וְחָדָן:
