פִּתְגַם נְבוּאָה דַהֲוָה עִם יִרְמְיָה מִן קֳדָם יְיָ לְמֵימָר:
קוּם בִּתְרַע בֵּית מַקְדְּשָׁא דַייָ וְתִתְנַבֵּי תַמָן יַת פִּתְגָמָא הָדֵין וְתֵימַר קַבִּילוּ פִּתְגָמָא דַייָ כָּל דְבֵית יְהוּדָה דְעָלִין בְּתַרְעַיָא הָאִילֵין לְמִסְגַד קֳדָם יְיָ:
כִּדְנַן אֲמַר יְיָ צְבָאוֹת אֱלָהָא דְיִשְׂרָאֵל אַתְקִינוּ אוֹרְחָתְכוֹן וְעוֹבָדֵיכוֹן וְאַשְׁרֵי יַתְכוֹן בְּאַתְרָא הָדֵין:
לָא תִתְרַחֲצוּן לְכוֹן עַל פִּתְגָמֵי נְבִיֵי שִׁקְרָא דְאָמְרִין קֳדָם הֵיכְלָא דַייָ אַתּוּן פָּלְחִין קֳדָם הֵיכְלָא דַייָ אַתּוּן דָבְחִין קֳדָם הֵיכְלָא דַייָ אַתּוּן סָגְדִין תְּלַת זִמְנִין בְּשַׁתָּא אַתּוּן מִתְחַזָן קֳדָמוֹהִי:
אֲרֵי אִם אַתְקְנָא תְּתַקְנוּן יַת אוֹרְחַתְכוֹן וְיַת עוֹבָדֵיכוֹן אִם מֶעְבַּד תַּעְבְּדוּן דִין דִקְשׁוֹט בֵּין גַבְרָא וּבֵין חַבְרֵיהּ:
גִיוֹר יָתוֹם וְאַרְמְלָא לָא תוֹנִיסוּן וְדַם זַכַּאי לָא תִשְׁדוּן בְּאַתְרָא הָדֵין וּבָתַר פַּלְחֵי כּוֹכְבַיָא לָא תְּהָכוֹן לְאַבְאָשָׁא לְכוֹן:
וְאַשְׁרֵי יַתְכוֹן בְּאַתְרָא הָדֵין בְּאַרְעָא דִיהָבִית לַאֲבָהַתְכוֹן לְמִן עַלְמָא וְעַד עַלְמָא:
הָא אַתּוּן מִתְרַחְצִין לְכוֹן עַל פִּתְגָמֵי נְבִיֵי שִׁקְרָא בְּדִיל דְלָא לַהֲנָאָה:
גַנָבִין קְטוֹלֵי נַפְשָׁן וְגַיְפִין וּמִשְׁתַּבְּעִין לִשְׁקַר וּמַסְקִין בּוּסְמִין לְבַעֲלָא וְאָזְלִין בָּתַר פַּלְחֵי כּוֹכְבַיָא דְלָא יְדַעְתּוּן:
וְאָתָן אַתּוּן וְקַיְימִין קֳדָמַי בְּבֵיתָא הָדֵין דְאִתְקְרֵי שְׁמִי עֲלוֹהִי וְאָמְרִין אַתּוּן שֵׁיזֵיבְנָא בְּדִיל לְמֶעְבַּד יַת כָּל תּוֹעֲבָתָא הָאִלֵין:
הַכְּבֵית כְּנִישַׁת רַשִׁיעִין הֲוָה בֵיתָא הָדֵין דְאִתְקְרֵי שְׁמִי עֲלוֹהִי בְּעֵינֵיכוֹן אַף קֳדָמֵי כֵּן גְלֵי אֲמַר יְיָ:
אֲרֵי אֱזִילוּ כְעַן לַאֲתַר בֵּית שְׁכִנְתִּי דִי בְּשִׁלּוֹ דְאַשְׁרֵיתִי שְׁכִנְתִּי תַּמָן בְּקַדְמֵיתָא וַחֲזוֹ יַת דַעֲבָדִית לֵיהּ מִן קֳדָם בִּישַׁת עַמִי יִשְׂרָאֵל:
וּכְעַן חֲלַף דַעֲבַדְתּוּן יַת כָּל עוֹבָדַיָא הָאִלֵין אֲמַר יְיָ וּשְׁלָחֵית לְוַתְכוֹן יַת כָּל עַבְדֵי נְבִיַיָא מַקְדִים וּמְמַלֵיל וְלָא קַבֵּילְתּוּן וְאִתְנַבִּיאוּ לְכוֹן וְלָא תַבְתּוּן:
וְאַעְבֵּיד לְבֵיתָא דְאִתְקְרֵי שְׁמִי עֲלוֹהִי דְאַתּוּן רְחִיצִין בֵּיהּ וּלְאַתְרָא דִיהָבֵית לְכוֹן וּלְאֲבָהַתְכוֹן כְּמָא דַעֲבָדֵית לְשִׁילוֹ:
וְאַגְלֵי יַתְכוֹן מֵאַרְעָא בֵּית שְׁכִנְתִּי כְּמָא דְאַגְלֵיתִי יַת כָּל אֲחֵיכוֹן יַת כָּל זַרְעָא דְיִשְׂרָאֵל:
וְאַתְּ לָא תְצַלֵי עַל עַמָא הָדֵין וְלָא תִתְחַנֵן עֲלֵיהוֹן בְּבָעוּ וּצְלוֹ וְלָא תִבְעֵי מִן קָדָמַי אֲרֵי לָא רַעֲוָא קֳדָמַי לְקַבָּלָא צְלוֹתָךְ:
הֲלֵית אַתְּ חָזֵי מָא דְאִינוּן עָבְדִין בְּקִרְוַיָא דְבֵית יְהוּדָה וּבְשׁוּקֵי יְרוּשְׁלֵם:
בְּנַיָא מְלַקְטִין אָעָא וְאַבְהָתָא מַדְלְקִין יַת נוּרָא וּנְשַׁיָא לִישָׁן לִישָׁא לְמֶעְבַּד כַּרְדוֹטִין לְכוֹכְבַת שְׁמַיָא וּלְנַסְכָא נַסְכִין לְפַלְחֵי כּוֹכְבַיָא בְּדִיל לְאַרְגַזָא קֳדָמַי:
הַמְדַמַי דִי קָדָמַי אִינוּן מְרַגְזִין אֲמַר יְיָ הֲלָא לְאַבְאָשָׁא לְהוֹן בְּדִיל דְיִבְהַּתוּן בְּעוֹבָדֵיהוֹן:
בְּכֵן כִּדְנַן אֲמַר יְיָ אֱלֹהִים הָא רוּגְזִי וַחֲמָתִי חָל בְּאַתְרָא הָדֵין עַל אֱנָשָׁא וְעַל בְּעִירָא וְעַל אִילַן חַקְלָא וְעַל אִבָּא דְאַרְעָא וִישֵׁיצֵי וְלָא יְהֵי חָיָס:
כִּדְנַן אֲמַר יְיָ צְבָאוֹת אֱלָהָא דְיִשְׂרָאֵל עֲלָוָתְכוֹן אוֹסִיפוּ עַל נִכְסַת קוּדְשֵׁיכוֹן וַאֲכוּלוּ בִסְרָא:
אֲרֵי לָא מַלְלִית עִם אֲבָהַתְכוֹן וְלָא פַּקֵדְתִּינוּן בְּיוֹמָא דְאַפְקֵית יַתְהוֹן מֵאַרְעָא דְמִצְרַיִם עַל עִסְקֵי עַלְוָן וְנִכְסַת קוּדְשִׁין:
אֱלָהֵין יַת פִּתְגָמָא הָדֵין פַּקְדֵית יַתְהוֹן לְמֵימַר קַבִּילוּ לְמֵימְרִי וְאֶהֱוֵי לְכוֹן לֶאֱלָהָא וְאַתּוּן תְּהוֹן קֳדָמַי לְעָם וּתְהָכוּן בְּכָל אוֹרְחָא דַאֲפַקֵיד יַתְכוֹן בְּדִיל דְיִיטַב לְכוֹן:
וְלָא קַבִּילוּ וְלָא אַרְכִינוּ יַת אוּדְנֵיהוֹן וַאֲזָלוּ בַּעֲצָתְהוֹן בְּהִרְהוּר לִבְּכוֹן בִּישָׁא וְאַחְזָרוּ בְּפוּלְחָנִי קְדַל וְלָא שַׁוִיאוּ דְחַלְתִּי לָקֳבֵיל אַפֵּיהוֹן:
לְמִן יוֹמָא דִנְפָקוּ אֲבָהַתְכוֹן מֵאַרְעָא דְמִצְרַיִם עַד יוֹמָא הָדֵין וּשְׁלָחֵית לְוַתְכוֹן יַת כָּל עֲבָדַי נְבִיַיָא יוֹם מַקְדִים וּמְשַׁלַח:
וְלָא קַבִּילוּ לְמֵימְרִי וְלָא אַרְכִינוּ יַת אוּדְנֵיהוֹן וְקַשִׁיאוּ יַת קְדַלְהוֹן אַבְאִישׁוּ מֵאֲבָהַתְהוֹן:
וּתְמַלֵיל עִמְהוֹן יַת כָּל פִּתְגָמַיָא הָאִלֵין וְלָא יְקַבְּלוּן מִנָךְ וְתִתְנַבֵּי לְהוֹן וְלָא יְתִיבוּנָךְ:
וְתֵימַר לְהוֹן דֵין עַמָא דִי לָא שְׁמָעוּ לְמֵימְרָא דַייָ אֱלָהָהוֹן וְלָא קַבִּילוּ אוּלְפַן פִּסְקַת הֵימָנוּתָא וְאִתְמַנָעַת מִפּוּמְהוֹן:
תְּלִישִׁי עַל רַבְרְבַיִךְ דְיִגְלוּן וְטוּלִי עַל נַגְרִין אִלְיָא אֲרֵי רָחֵיק יְיָ וְרַטֵישׁ יַת דָרָא דַעֲבָרוּ עַל מֵימְרֵיהּ:
אֲרֵי עֲבָדוּ בְנֵי יְהוּדָה דְבִישׁ קֳדָמַי אֲמַר יְיָ שַׁוִיאוּ שִׁקוּצֵיהוֹן בְּבֵיתָא דְאִתְקְרֵי שְׁמִי עֲלוֹהִי לְסַאֲבוּתֵיהּ:
וּבְנוֹ בָּמַת תּוֹפֶת דִי בְחֵילַת בַּר הִנוֹם לְאוֹקָדָא יַת בְּנֵיהוֹן וְיַת בְּנָתֵיהוֹן בְּנוּרָא דְלָא פַקְדֵית בְּאוֹרַיְתִי וְלָא רַעֲוָא קָדְמַי:
בְּכֵן הָא יוֹמַיָא אָתָן אֲמַר יְיָ וְלָא יִתְאַמָר עוֹד תּוֹפֶת וְחֵילַת בַּר הִנוֹם אֱלָהֵין חֵילַת קְטִילַיָא וִיקַבְּרוּן בְּתוֹפֶת מִבְּלִי אֲתַר:
וּתְהֵי נְבֵילְתָּא דְעַמָא הָדֵין מְשַׁגְרָא לְמֵיכַל לְעוֹפָא דִשְׁמַיָא וְלִבְעִירָא דְאַרְעָא וְלֵית דְמָנִיד:
וָאֱבַטֵיל מִקִרְוַיָא דְבֵּית יְהוּדָה וּמִשׁוֹקֵי יְרוּשְׁלֵם קַל בִּיעַ וְקַל חֶדְוָא קַל חַתְנוּן וְקַל כַּלָן אֲרֵי לְחוּרְבָּא תְּהֵי אַרְעָא:
