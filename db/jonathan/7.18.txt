בְּיוֹמַיָא הָאִינוּן לֵית מַלְכָּא בְּיִשְׂרָאֵל וּבְיוֹמַיָא הָאִינוּן שִׁבְטָא דְבֵית דָן בָּעָן לְהוֹן אַחֲסַנְתָּא לְמִתַּב אֲרֵי לָא אִתְפַּלְגַת לְהוֹן עַד יוֹמָא הַהוּא מִגוֹ שִׁבְטַיָא דְיִשְׂרָאֵל בְּאַחְסָנָא:
וּשְׁלֵחוּ בְנֵי דָן מִזַרְעֲיַתְהוֹן חַמְשָׁא גוּבְרִין מִקְצַתְהוֹן גַבְרִין בְּנֵי חֵילָא מִצָרְעָה וּמֵאֶשְׁתָּאוֹל לְאַלָלָא יַת אַרְעָא וּלְמִבְדְקָא וַאֲמַרוּ לְהוֹן אֶזִילוּ בְּדוּקוּ יַת אַרְעָא וַאֲתוֹ לְטוּרָא דְבֵית אֶפְרַיִם עַד בֵּית מִיכָה וּבָתוּ תַמָן:
אִינוּן עִם בִּיתָא דְמִיכָה וִאִינוּן אִשְׁתְּמוֹדָעוּ יַת קַל עוּלֵימָא לִיוָאָה וְזַרוּ לְתַמָן וַאֲמַרוּ לֵיהּ מַן אַיְתָךְ הַלִכָא וּמָא אַתְּ עָבֵיד הָכָא וּמַה לָךְ כְּדֵין:
וַאֲמַר לְהוֹן כְּדֵין וּכְדֵין עֲבַד לִי מִיכָה וְאַגִרַנִי וַהֲוֵית לֵיהּ לְכוּמְרָא:
וַאֲמַרוּ לֵיהּ שְׁאֵיל כְּעַל בְמֵימְרָא דַייָ וְנֵידַע הַתַצְלַח אוֹרְחָנָא דִי אֲנַחְנָא אָזְלִין עֲלָהּ:
וַאֲמַר לְהוֹן כַּהֲנָא אֱזִילוּ לִשְׁלָם אַתְקִין יְיָ אוֹרְחַתְכוֹן דִי תְהָכוּן בָּהּ:
וַאֲזַלוּ חַמְשָׁא גוּבְרִין וַאֲתוֹ לְלָיִשׁ וַחֲזוֹ יַת עַמָא דִי בְגַוָהּ יַתְבִין לְרוֹחֲצָן כְּהִלְכַת צִידוֹנָאֵי דְשָׁקְטִין וּשְׁרָן לְרוֹחֲצָן וְלֵית דְמַנְזַק פִּתְגָמָא בְּאַרְעָא יְרוּתִין זְעִירִין וּרְחִיקִין אִינוּן מִצִידוֹנָאֵי וּפִתְגַם לֵית לְהוֹן עִם אֱנָשׁ:
וַאֲתוֹ לְוַת אֲחוּהוֹן צָרְעָה וְאֶשְׁתָּאוֹל וַאֲמַרוּ לְהוֹן אֲחוּהוֹן מָה אַתּוּן:
וַאֲמַרוּ קוּמוּ וְנִיסַק עֲלֵיהוֹן אֲרֵי חֲזֵינָא יַת אַרְעָא וְהָא טָבָא לַחֲדָא וְאַתּוּן שַׁתְקִין לָא תִתְרַשְׁלוּן לְמֵיזַל לְמֵיעַל לְמֵירַת יַת אַרְעָא:
כְּמֵיתֵיכוֹן תַּעֲלוּן לְוַת עַמָא דְשָׁרָן לְרוֹחֲצָן וְאַרְעָא פַּתְיוּת יְדִין אֲרֵי מַסְרָא יְיָ בִידֵיכוֹן אַתְרָא דְלֵית תַּמָן חֻסְרָן כָּל מִדַעַם דִי בְאַרְעָא:
וּנְטָלוּ מִתַּמָן מִזַרְעִית דָן מִצָרְעָה וּמֵאֶשְׁתָּאֹל שִׁית מְאָה גַבְרִין מְזָרְזִין בְמָנֵי קְרָבָא:
וּסְלִיקוּ וּשְׁרוֹ בְּקִרְיַת יְעָרִים בְּאַרְעָא דְבֵית יְהוּדָה עַל כֵּן קְרוֹ לְאַתְרָא הַהוּא מַשִׁרְיַת דָן עַד יוֹמָא הָדֵין הָא הוּא אֲחוֹרֵי קִרְיַת יְעָרִים:
וַעֲבָרוּ מִתַּמָן לְטוּרָא דְבֵית אֶפְרַיִם וַאֲתוֹ עַד בֵּית מִיכָה:
וַאֲתִיבוּ חַמְשָׁא גוּבְרַיָא דַאֲזַלוּ לְאַלָלָא יַת אַרְעָא לַיִשׁ וַאֲמַרוּ לַאֲחוּהוֹן הַיְדַעְתּוּן אֲרֵי אִית בְּבָתַּיָא הָאִלֵין אֵפוֹד וּדְמָאִין וּצְלֵים וּמַתְּכָא וּכְעַן דְעוּ מַה תַעְבְּדוּן:
וְזָרוּ לְתַמָן וְעָלוּ לְבֵית עוּלֵימָא לְוָאָה בֵּית מִיכָה וּשְׁאִילוּ לֵיהּ לִשְׁלָם:
וְשִׁית מְאָה גַבְרִין מְזָרְזִין בְּמָנֵי קְרָבֵיהוֹן קַיְמִין בְּמַעֲלָנָא דִתְרַע דְמִשֵׁבֶט בְּנֵי דָן:
וּסְלִיקוּ חַמְשָׁא גַבְרַיָא דַאֲזָלוּ לְאַלָלָא יַת אַרְעָא עָלוּ לְתַמָן וּנְסִיבוּ יַת צַלְמָנָא וְיַת אֵפוֹדָא וְיַת דְמָאִין וְיַת מַתְּכָא וְכַהֲנָא קָאִים בְּמַעֲלָנָא דְתַרְעָא וְשִׁית מְאָה גַבְרֵי דִמְזָרְזֵי בְּמָנֵי קְרָבָא:
וְאִלֵין עָלוּ לְבֵית מִיכָה וּנְסִיבוּ יַת צְלַם אֵפוֹדָא וְיַת דְמָאִין וְיַת מַתְּכָא וַאֲמַר לְהוֹן כַּהֲנָא מָה אַתּוּן עָבְדִין:
וַאֲמַרוּ לֵיהּ שְׁתוֹק שַׁוֵי יְדָךְ עַל פּוּמָךְ וֶאֱזֵיל עִמָּנָא וֶהֱוֵי לָנָא לְאָב וּלְכַהֲנָא הֲטַב דִתְהֵי כָהִין לְבֵית גַבְרָא חָד אוֹ דִתְהֵי כָהִין לְשִׁבְטָא וּלְזַרְעִית בְּיִשְׂרָאֵל:
וּשְׁפַר לִבָּא דְכַהֲנָא וּנְסִיב יַת אֵפוֹדָא וְיַת דְמָאִין וְיַת צַלְמָא וְעַל בְּגוֹ עַמָא:
וְאִתְפְּנִיאוּ וַאֲזַלוּ וְשַׁוִיאוּ יַת טַפְלַיָא וְיַת בְּעִירָא וְיַת קִנְיָנָא קֳדָמֵיהוֹן:
אִינוּן אַרְחִיקוּ מִבֵּית מִיכָה וְגַבְרַיָא דִבְבָתַּיָא דְעִם בֵּית מִיכָה אִתְכְּנִישׁוּ וְאַדְבִּיקוּ יַת בְּנֵי דָן:
וּקְרוֹ לִבְנֵי דָן וְאַסְחֲרוּ אַפֵּיהוֹן וַאֲמַרוּ לְמִיכָה מַה לָךְ אֲרֵי אִתְכְּנִישְׁתָּא:
וַאֲמַר יַת דַחַלְתִּי דִי עֲבָדִית נְסֵיבְתּוּן וְיַת כַּהֲנָא וַאֲזַלְתּוּן וּמַה לִי עוֹד וּמַה דֵין תֵּימְרוּן לִי מַה לָךְ:
וַאֲמַרוּ לֵיהּ בְּנֵי דָן לָא תַשְׁמַע קָלָךְ עִמָנָא דִלְמָא יִשְׁלְטוּן בְּכוֹן גַבְרִין מְרִירַן נַפְשָׁן וְתָסֵיף נַפְשָׁךְ וּנְפַשׁ אֱנָשׁ בֵיתָךְ:
וַאֲזַלוּ בְּנֵי דָן לְאוֹרְחֵיהוֹן וַחֲזָא מִיכָה אֲרֵי תַקִיפִין אִינוּן מִינֵיהּ וְאִתְפְּנֵי וְתַב לְבֵיתֵיהּ:
וְאִינוּן נְסִיבוּ יַת דַעֲבַד מִיכָה וְדַבֵּרוּ יַת כַּהֲנָא דִי הֲוָה לֵיהּ וַאֲתוֹ לְלַיִשׁ לְעַמָא דְשָׁקְטִין וְשָׁרָן לְרוֹחֲצָן וּמְחוֹ יַתְהוֹן לְפִּתְגָם דְחָרֶב וְיַת קַרְתָּא אוֹקִידוּ בְּנוּרָא:
וְלֵית דִמְשֵׁיזִיב אֲרֵי רְחִיקָא הִיא מִצִידוֹן וּפִתְגַם לֵית לְהוֹן עִם אֱנַשׁ וְאִינוּן בְּמֵשַׁר דִלְבֵית רְחוֹב וּבְנוֹ יַת קַרְתָּא וִיתִיבוּ בָהּ:
וּקְרוֹ שְׁמָא דְקַרְתָּא דָן בְּשׁוּם דָן אֲבוּהוֹן דְאִתְיְלִיד לְיִשְׂרָאֵל וּבְרַם לַיִשׁ שְׁמָא דְקַרְתָּא מִלְקַדְמִין:
וַאֲקִימוּ לְהוֹן בְּנֵי דָן יַת צַלְמָא וִיהוֹנָתָן בַּר גֵרְשֹׁם בַּר מְנַשֶׁה הוּא וּבְנוֹהִי הֲווֹ כוּמָרִין לְשִׁבְטָא דְבֵית דָן עַד יוֹמָא דִי גְלוֹ יַתְבֵי אַרְעָא:
וְשַׁוִיאוּ לְהוֹן יַת צְלָם מִיכָה דִי עֲבַד כָּל יוֹמַיָא דַהֲוָה קַיָם בֵּית מַקְדְשָׁא דַייָ בְּשִׁלֹה:
