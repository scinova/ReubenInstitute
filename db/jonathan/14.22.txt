וַהֲוָה פִּתְגַם נְבוּאָה מִן קֳדָם יְיָ עִמִי לְמֵימָר:
וְאַתְּ בַּר אָדָם הֲתוֹכַח הֲתוֹכַח יַת קַרְתָּא דְאִשְׁדֵי דַם זַכַּאי בְּגַוָהּ וְחַוֵי לָהּ יַת כָּל תּוֹעֲבָתָהָא:
וְתֵימַר כִּדְנַן אֲמַר יְיָ אֱלֹהִים קַרְתָּא דְאַשְׁדֵי דַם זַכַּאי בְּגַוָהּ מְטָא עִדַן תַּבְרָהּ וּדְפָלְחִין לְטַעֲוָתָא בְגַוָהּ בְּדִיל לְסַאֲבוּתָא:
בְּחוֹבַת דַם זַכַּאי דִאַשַׁדְתְּ חַבְתְּ וּבְּפוּלְחָן טַעֲוָתֵךְ דִפְלַחְתְּ אִסְתַּאַבְתְּ וְקָרֵיב יוֹם תַּבְרִיךְ וּמְטָא עִדַן בִּישְׁתִּיךְ עַל כֵּן יְהַבְתִּיךְ חִסוּדִין לְעַמְמַיָא וּלְעֵיב לְכָל מִדִינָתָא:
דְקָרִיבִין וְדִרְחִיקִין מִנִיךְ יִתְלַעֲבוּן בִּיךְ סוֹאֲבַת שְׁמָא דְסַגִיאִין אִתְרְגוּשְׁתָּהָא:
הָא רַבְרְבֵי יִשְׂרָאֵל גְבַר לְתוּקְפֵיהּ הֲווֹ בִיךְ בְּדִיל לְמִשְׁפַּךְ דַם זַכַּאי:
אַבָּא וְאִמָא אָקִילוּ בָךְ לְגִיוֹרָא אֲנִיסוּ אוֹנְסָא בְּגַוִיךְ יַתְמִין וְאַרְמְלָן אוֹנִיאוּ בִיךְ:
עַל קוּדְשֵׁי בִּסַרְתְּ וְיַת יוֹמֵי שַׁבַּיָא דִילִי אָחֵילְתְּ:
גַבְרֵי אָכְלֵי קוּרְצָא הֲיוֹ בִיךְ בְּדִיל לְמִשְׁפַּךְ דַם זַכַּאי וּבְטוּרַיָא פְלָחוּ בִיךְ לְטַעֲוָתָא אֲרֵי עֵיצַת חֶטְאִין עֲבָדוּ בְגַוִיךְ:
עֶרְיַת אַבָּא גַלִיאוּ בִּיךְ אִתָּא טוּמְאָה עֵנִיאוּ בִיךְ:
וּגְבַר יַת אִתַּת חַבְרֵיהּ עֲבַד תּוֹעֵיבָא וּגְבַר יַת כַּלָתֵיהּ סָאֵיב בְּעֵיצַת חֶטְאִין וּגְבַר יַת אֲחָתֵיהּ בַת אֲבוּהִי עַנֵי בִיךְ:
שׁוֹחֲדָא קַבִּילוּ בִיךְ בְּדִיל לְמִשְׁפַּךְ דַם זַכַּאי חִבּוּלְיָא וְרִבִּיתָא נְסֵיבְתְּ וְאַהֲנֵית רְחִימָךְ בְּאוּנְסָא וּפּוּלְחָנִי שְׁבַקְתְּ אֲמַר יְיָ אֱלֹהִים:
וְהָא אַיְתֵיתִי פּוּרְעֲנוּתִי עֲלָךְ בְּחוֹבֵי מָמוֹן אוֹנְסִיךְ דַאֲנַסְתְּ וְעַל דַם זַכַּאי דְאִשְׁתְּפִיךְ בְּגַוִיךְ:
הֲיִתְקַף לִבִּיךְ אִם יִתְחַסְנָן יְדֵיךְ לְיוֹמַיָא דַאֲנָא עֲתִיד לְמֶעְבַד עִמָךְ אֲנָא יְיָ גְזָרֵית בְּמֵימְרִי וַאֲקַיֵם:
וְאַגְלֵי יָתִיךְ לְבֵינֵי עַמְמַיָא וְאַבַדְרִינָךְ בִּמְדִינָתָא וַאֲסִיף סְאוֹבְתִּיךְ מִנֵיךְ:
וְאֶתְקַדֵשׁ בִּיךְ לְעֵינֵי עַמְמַיָא וְתֵידְעוּן אֲרֵי אֲנָא יְיָ:
וַהֲוָה פִּתְגַם נְבוּאָה מִן קֳדָם יְיָ עִמִי לְמֵימָר:
בַּר אָדָם הֲווֹ קֳדָמַי בֵּית יִשְׂרָאֵל לִפְסוּלָא כּוּלְהוֹן כְּמָא דִמְעָרַב נְחַשׁ וּבְעִיץ וּבַרְזֶל וַאֲבַר בְּגוֹ כוּרָא לִפְּסוֹלַת כַּסְפָּא הֲווֹ:
בְּכֵן כִּדְנַן אֲמַר יְיָ אֱלֹהִים חֲלַף דַהֲוֵיתוּן כּוּלְכוֹן לִפְסוּלָא בְּכֵן הָא אֲנָא כָנֵישׁ יַתְכוֹן לְגוֹ יְרוּשְׁלֵם:
כְּמַכְנֵשׁ כְּסַף וּנְחַשׁ וּבַרְזְלָא וַאֲבַר וּבְעִיץ לְגוֹ כוּרָא לְאַתְקָפָא עֲלוֹהִי נוּרָא לְאַמְסָיוּתֵיהּ כֵּן אַכְנֵשׁ בְּרוּגְזִי וּבְחֵימְתִי וְאָסֵיף וֶאֱשֵׁיצֵי יַתְכוֹן:
וְאַכְנֵישׁ יַתְכוֹן וְאַתְקֵיף עֲלֵיכוֹן בְּאֶשְׁתָּא פּוּרְעֲנוּתִי וְתִתְמְסוּן בְגַוָהּ:
כְּאִתְמְסָאָה כְסַף בְּגוֹ כוּרָא כֵּן תִּתְמְסוּן בְּגַוָהּ וְתִידְעוּן אֲרֵי אֲנָא יְיָ שְׁפִיכֵית חֵימְתִי עֲלֵיכוֹן:
וַהֲוָה פִּתְגַם נְבוּאָה מִן קֳדָם יְיָ עִמִי לְמֵימָר:
בַּר אָדָם אֲמַר לָהּ אַתְּ אַרְעָא דְיִשְׂרָאֵל אַרְעָא לָא מִדַכְיָא הִיא וְלָא אִתְעֲבִידוּ בָּהּ עוֹבָדִין טָבִין דְיִגְנוּן עֲלָהּ בְּיוֹם לְוָט:
סִיעַת סַפְרָהָא בְּגַוָהּ כְּאַרְיְוָן נָהֲמִין לְמִקְטַל קְטוֹל אֱנָשָׁא קְטָלוּ עוֹתֵר וִיקָר נְסִיבוּ אַרְמַלְתָּהָא אַסְגִיאוּ בְּגַוָהּ:
כַּהֲנָהָא חָטְפִין בְּאוֹרַיְתִי וּמַפְסִין קֳדָשַׁי בֵּין קוּדְשָׁא לְחוֹלָא לָא אַפְרִישׁוּ וּבֵין מְסַאֲבָא לְדַכְיָא לָא הוֹדִיעוּ וּמִיוֹמֵי שַׁבַּיָא דִילִי כְבָשׁוּ עֵינֵיהוֹן וְחַלַת רְעוּתִי בֵּינֵיהוֹן:
רַבְרְבָהָא בְגַוָהּ כַדִיבִין חָטְפֵי חֲטוֹף לְמִשְׁפַּךְ דַם זַכַּאי לַאֲבָדָא נַפְשָׁן בְּדִיל לְקַבָּלָא מָמוֹן דִשְׁקַר:
וּנְבִיֵי שִׁקְרָא דִי בְגַוָה דָמָן לִדְבָנֵי מְחִיצָא וְשָׁעֵיעַ לֵיהּ טִין פַטִיר דְלָא תֶּבֶן מִתְנַבַּן שְׁקַר וּמַלְפִין לְהוֹן כַּדְבִין אָמְרִין כִּדְנַן אֲמַר יְיָ אֱלֹהִים וּמִן קֳדָם יְיָ לָא אִתְמַלָל:
עַמָא דְאַרְעָא עֲשָׁקוּ עוּשְׁקָא וּגְזָלוּ גְזֵלָה וּמִסְכֵּן וַחֲשִׁיךְ אוֹנִיאוּ וּלְגִיוֹרָא אֵנָסוּ בְּלָא כִדְחֲזֵי:
וְאֶתְבַּע מִנְהוֹן קֳדָמַי גְבַר דְלֵיהּ עוֹבָדִין טָבִין דִיקוּם בְּתִרְעֲתָא קֳדָמַי וְיִבְעֵי רַחֲמִין עַל עַמָא דְאַרְעָא בְּדִיל דְלָא לְחַבָּלוּתָא וְלָא אַשְׁכְּחֵית:
וּשְׁפִיכֵית עֲלֵיהוֹן חֵימְתִי בְּאֶשַׁת פּוּרְעָנוּתִי שֵׁיצִיתִינוּן פּוּרְעֲנוּת אוֹרְחֵיהוֹן בְּרֵישֵׁיהוֹן יְהָבֵית אֲמַר יְיָ אֱלֹהִים:
