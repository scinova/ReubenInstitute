מְטַל נְבוּאֲתָא דְאִתְנַבֵּי חֲבַקוּק נְבִיָא:
עַד אֵימָתַי יְיָ אֲנָא מְצַלֵי הֲלָא גְלֵי קֳדָמָךְ קָבֵילְנָא קֳדָמָךְ עַל חֲטוֹפִין הֲלָא יוּכְלָא קֳדָמָךְ לְמִפְרַק:
לְמָא אֲנָא חָזֵי אֲנוֹסִין וּמִסְתַּכֵּיל כְּעָבְדֵי לֵיאוּת שְׁקַר בְּזוֹזִין וַחֲטוֹפִין לְקִבְלִי דִבְתַחֲרוּתָא וּבִזְדוֹנָא אִינוּן מִתְגַבְּרִין:
עַל כֵּן פָּגַת אוֹרַיְתָא וְלָא נָפֵיק לְאַפְרַשׁ דִינָא אֲרֵי רְשִׁיעַיָא מְסַלְעֲמִין לְצַדִיקַיָא עַל כֵּן לָא נָפֵיק דִינָא כֵיוָן:
חֲזוֹ בְּעַמְמַיָא וְאִסְתַּכָּלוּ אִתְרְפוֹ שְׁוָהוּ אֲרֵי עוֹבָדָא אִתְעֲבֵיד בְּיוֹמֵיכוֹן לָא תְהֵימְנוּן אֲרֵי יִשְׁתָּעוּן לְכוֹן:
אֲרֵי הָא אֲנָא מֵקִים יַת כַּסְדָאֵי עַמָא מוֹחֲיָא וְקַלִילָא דְאָזֵיל לִפְתָאֵי אַרְעָא לְאַחְסָנָא קִרְוִין דְלָא דִילֵיהּ:
אֵימְתָנִין וּדְחִילִין אִינוּן מִנֵיהּ דִינֵיהּ וּגְזֵרָתֵיהּ נָפְקִין:
וְקַלִילִין מִנְמָרַיָא סוּסְוָתֵיהּ וַחֲרִיפִין מִדֵיבֵי רַמְשָׁא וְיִרְבוּן פָּרָשׁוֹהִי וּפָרָשׁוֹהִי מֵרָחִיק דְיֵיתוּן מִשְׁתְּדָן כְּנִשְׁרָא דְטָאֵיס לְמֵיכַל:
כּוּלְהוֹן לַחֲטוֹף אָתָן מִקֳבֵיל אַפֵּיהוֹן דָמָן כְּרוּחַ קָדִים קִידוּמָא וּכְנַשׁ כְּחָלָא שִׁבְיָא:
וְהוּא עַל מַלְכַיָא מִתְלַעַב וְעַל שִׁלְטוֹנַיָא מְחַיָךְ הוּא לְכָל כְּרַךְ תַּקִיף מֵעִיק וּצְבַר מִלְיָתָא וְכַּבְשָׁהּ:
בְּכֵן עַל דְרָמַת רוּחֵיהּ עֲלוֹהִי עֲדָא מִמַלְכוּתֵיהּ וְחַב עַל דְאַסְגֵי יְקָרָא לְטַעֲוָתֵיהּ:
הֲלָא אַתְּ יְיָ בָּרֵיתָ עַלְמָא מִבְּרֵאשִׁית אַתְּ אֱלָהָא דַיָן קְשׁוֹט עַל כָּל בִּרְיָתָךְ קַדִישׁ בְּעוֹבָדֵי הֵימָנוּתָא מֵימְרָךְ קַיָם לְעַלְמִין יְיָ לְמֶעְבַּד דִינָא בְרֵיתֵיהּ וְתַקִיף לְאִתְפְּרָעָא אַתְקִינְתֵּיהּ:
בְּרִיר מֵימְרָךְ מִלְמֶחֱזֵי בְּעוֹבָדֵי בִּישׁ וּמִלְאִסְתַּכָּלָא בְּעָבְדֵי לֵיאוּת שְׁקַר הֲלָא גְלֵי קֳדָמָךְ וּלְמָא אַתְּ מִסְתַּכֵּל בְּאָנְסִין וְאַתְּ יָהֵיב אַרְכָּא לְרַשִׁיעַיָא וּמְסַלְעֲמִין לִדְטָבִין מִנְהוֹן:
וַעֲבַדְתָּא אֱנָשָׁא כְּנוּנֵי יַמָא כְּרִיחֲשָׁא דְלֵית שׁוּלְטַן עֲלוֹהִי:
כּוּלְהוֹן דָמָן לְצַיָדִין דְאָחֵיד נוּנָא בְּחַכְּתָא כָנֵישׁ לֵיהּ בְּחָרְמֵיהּ וְרָמֵי לֵיהּ בִּמְצַדְתֵּיהּ עַל כֵּן חָדַי וְדָאִיץ:
עַל כֵּן מַדְבַּח לִזְיָנֵיהּ וּמַסִיק בּוּסְמִין לְסַמְוָתֵיהּ אֲרֵי בְהוֹן טַב מֵיכְלֵיהּ וְשֵׁירוּתֵיהּ מַטְעֲמָא:
הַעַל אִילֵין יִשְׁלַח מַשִׁרְיָתֵיהּ בִּתְדִירָא לְאוֹסָפָא עַמְמִין וְלָא בַּחַיָס:
