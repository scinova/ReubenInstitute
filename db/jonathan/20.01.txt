פִּתְגַם נְבוּאָה מִן קֳדָם יְיָ דִי הֲוָה עִם מִיכָה מִמָרֵשָׁה בְּיוֹמֵי יוֹתָם אָחָז יְחִזְקִיָה מַלְכַיָא דְבֵית יְהוּדָה דְאִתְנַבֵּי עַל שֹׁמְרוֹן וִירוּשְׁלֵם:
שְׁמָעוּ עַמְמַיָא כוּלְהוֹן אֲצִיתֵי אַרְעָא וּמְלָהָא וִיהֵי מֵימְרָא דַייָ אֱלֹהִים בְּכוֹן לְסָהִיד יְיָ מֵהֵיכְלָא דְקוּדְשֵׁיהּ:
אֲרֵי הָא יְיָ מִתְגְלֵי מֵאֲתַר בֵּית שְׁכִינְתֵּיהּ וְיִתְגְלֵי וְיִרְמוֹס יַת תּוּקְפֵי אַרְעָא:
וְיִתְמְסוּן טוּרַיָא תְּחוֹתוֹהִי וּמֵשְׁרַיָא יִתְבַּזְעוּן כְּשָׁעִיתָא מִן קֳדָם נוּרָא כְּמַיָא דְאַשִׁידִין בְּמוֹדְרוֹן:
בְּחוֹבֵי בֵּית יַעֲקֹב כָּל דָא וּבְחֵטְא דְבֵית יִשְׂרָאֵל הֵיכָא חָבוּ דְבֵית יַעֲקֹב הֲלָא בְשֹׁמְרוֹן וְהֵיכָא חֲטוֹ דְבֵית יְהוּדָה הֲלָא בִּירוּשְׁלֵם:
וַאֲשַׁוֵי שֹׁמְרוֹן לְיַגְרֵי חַקְלָן בֵּית צָדוּת מַצְבַת כְּרַם וְאֶרְמֵי לְחֵילָתָא אַבְנָהָא וְאַיְשָׁהָא אֶגְלֵי:
וְכָל צַלְמָנָהָא יִדְקְרוּן וְכָל טַעֲוָתָהָא יִתּוֹקְדוּן בְּנוּרָא וְכָל בָּתֵּי דַחֲלָתָהָא אֲשַׁוֵי לְצָדוּ אֲרֵי מַאֲגַר זַנְיָתָא אִתְכְּנָשׁוּ וּלְבֵית פָּלְחֵי טַעֲוָתָהָא יִתְמַסְרוּן:
עַל דָא יִסְפְּדוּן וְיֵלִילוּן יְהָכוּן בְּשׁוּלְלַיָא עַרְטִילָאִין יַעַבְּדוּן מִסְפְּדָא כִּירוֹדִין וְאֵיבְלָא כִּבְנַת נַעֲמִין:
אֲרֵי מְמָרְעָא מֵחֲתָהָא אֲרֵי מְטַת עַד דְבֵית יְהוּדָה קְרִיבַת לְמֵיעַל בְּתַרְעֵי עַמִי עַד יְרוּשְׁלֵם:
בְגַת לָא תְחַווּן מִבְּכָא לָא תִבְכּוּן דְיָתְבִין בְּבָתַּיָא בְּעַפְרָא יִתְפַּלְשׁוּן בְקִיטְמָא חֲפוֹ רֵישֵׁיהוֹן:
עֲבָרוּ לְכוֹן דְיַתְבִין בְּשָׁפִיר גְלָן עֶרְיָא עַרְטִילָאִין בָחֲתִין לָא יִפְקוּן מְשֵׁיזְבִין יַתְבֵי צַאֲנָן עֱבִידוּ לְכוֹן מִסְפַּד בְּמֶרֶר עַל קְטִילֵי גִבָּרֵיכוֹן בֵּית צוּלָאֵי בָּתֵּי חֶמְדַתְכוֹן דַהֲוֵיתוּן אַנְסִין וּמְקַרְבִין דֵין לִסְטַר דֵין בְּכֵן יִתְנְסֵיב מִנְכוֹן שְׁפַר אַרְעֲכוֹן:
אֲרֵי דַהֲוַת יַתְבָא עַל שְׁפַר אַרְעָא וּמְסַבְּרָא לְטַב וּמְסַבְּרָא לְמֵיסַב לְאוֹרַיְתָא מַה תַעַבְּדוּן אֲרֵי נַחֲתַת בִּישְׁתָּא מִן קֳדָם יְיָ לְמֵיעַל בְּתַרְעֵי יְרוּשְׁלֵם:
טְקִיסוּ רְתִיכַיָא רְמוֹ עַל רִכְשָׁא דְיָתְבִין בְּלָכִישׁ רֵישׁ חוֹבִין הִיא לִכְנִשְׁתָּא דְצִיוֹן אֲרֵי בִּיךְ אִשְׁתְּכָחוּ מְרוֹדֵי יִשְׂרָאֵל:
בְּכֵן תְּשַׁדְרִין קוּרְבָּנִין לְמַחֲסִינֵי גַת בָּתֵּי אַכְזִיב יִתְמַסְרוּן לְעַמְמַיָא בְּחוֹבִין דִי פְלָחוּ בְּהוֹן לְטַעֲוָתָא מַלְכֵי יִשְׂרָאֵל:
עוֹד מַחְסְנִין אַיְתֵי עֲלֵיכוֹן דְיַתְבִין בְּמָרֵשָׁה עַד עֲדֻלָם יִסְקוּן וְיַעֲלוּן בִּתְחוּם אַרְעָא דְיִשְׂרָאֵל:
תְּלוּשִׁי וּרְמָא עַל בְּנֵי תַפְנוּקָיִךְ הָא כְּנִשְׁרָא דִנְתָרוּ גַדְפוֹהִי כֵּן אַסְגֵי מְרַט בְּרֵישַׁךְ עַל בָּנַיִךְ אֲרֵי גְלוֹ מִנָךְ:
