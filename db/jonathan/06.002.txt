וּשְׁלַח יְהוֹשֻׁעַ בַּר נוּן מִן שִׁטִין תְּרֵין גוּבְרִין מְאַלְלִין בְּרָז לְמֵימָר אֲזִילוּ חֲזוּ יַת אַרְעָא וְיַת יְרִיחוֹ וַאֲזַלוּ וְעַלוּ לְבֵית אִתְּתָא פּוּנְדְקִיתָא וּשְׁמָה רָחָב וּשְׁכִיבוּ תַמָן:
וְאִתְאַמַר לְמַלְכָּא דִירִיחוֹ לְמֵימָר הָא גוּבְרַיָא אֲתוֹ הָכָא בְּלֵילְיָא מִבְּנֵי יִשְׂרָאֵל לְאַלָלָא יַת אַרְעָא:
וּשְׁלַח מַלְכָּא דִירִיחוֹ לְוַת רָחָב לְמֵימָר אַפִיקִי גֻבְרַיָא דַאֲתוֹ לְוָתִיךְ דַעֲלוּ לְבֵיתֵיךְ אֲרֵי לְאַלָלָא יַת כָּל אַרְעָא אֲתוֹ:
וּדְבִירַת אִתְּתָא יַת תְּרֵין גֻבְרַיָא וְאַטְמַרַתְנוּן וַאֲמָרַת בְּקוּשְׁטָא אֲתוֹ לְוָתִי גֻבְרַיָא וְלָא יִדְעֵית אֵי מִנָן אִינוּן:
וַהֲוָה עִדָן לְמֵיחַד תַּרְעָא בְּקִבְלָה וְגֻבְרַיָא נְפַקוּ לָא יַדְעֵית לְאָן אֲזַלוּ גֻבְרַיָא רִדְפוּ בִּפְרִיעַ בַּתְרֵיהוֹן אֲרֵי תַדְבְּקוּנוּן:
וְהִיא אַסְקַתְנוּן לְאִגְרָא וְאַטְמַרַתְנוּן בִּטְעוּנֵי כִתָּנָא דִסְדִירִין לָהּ עַל אִגְרָא:
וְגֻבְרַיָא רְדָפוּ בַּתְרֵיהוֹן אוֹרַח יַרְדֵנָא עַד מְגִזָתָא וְתַרְעָא אֲחָדוּ בָּתַר דִנְפַקוּ רַדְפַיָא בַּתְרֵיהוֹן:
וְאִינוּן עַד לָא שְׁכִיבוּ וְהִיא סְלֵיקַת לְוַתְהוֹן לְאִגְרָא:
וַאֲמַרַת לְגֻבְרַיָא יְדַעְנָא אֲרֵי יְהַב יְיָ לְכוֹן יַת אַרְעָא וַאֲרֵי נְפָלַת אֵמַתְכוֹן עֲלָנָא וַאֲרֵי אִתְּבָּרוּ כָּל יַתְבֵי אַרְעָא מִן קֳדָמֵיכוֹן:
אֲרֵי שְׁמַעְנָא יַת דְיַבֵּשׁ יְיָ יַת מֵי יַמָא דְסוּף מִן קֳדָמֵיכוֹן בְּמִפַּקְכוֹן מִמִצְרַיִם וְדִי עֲבַדְתּוּן לִתְרֵין מַלְכֵי אֱמוֹרָאָה דִי בְּעַבְרָא דְיַרְדְנָא לְסִיחוֹן וּלְעוֹג דִי גְמַרְתּוּן יַתְהוֹן:
וּשְׁמַעְנָא וְאִתְמְסֵי לִבָּנָא וְלָא אִשְׁתְּאָרַת עוֹד רוּחָא בֶּאֱנַשׁ מִן קֳדָמֵיכוֹן אֲרֵי יְיָ אֱלָהָכוֹן הוּא אֱלָהָא דִשְׁכִינְתֵּיהּ בִּשְׁמַיָא מִלְעֵילָא וְשַׁלִיט עַל אַרְעָא מִלְרָע:
וּכְעַן קַיְימוּ כְעַן לִי בְּמֵימְרָא דַייָ אֲרֵי עֲבָדִית עִמְכוֹן טִיבוּ וְתַעְבְּדוּן אַף אַתּוּן עִם בֵּית אַבָּא טִיבוּ וְתִתְּנוּן לִי אָת דִקְשׁוֹט:
וְתֵיחוּן יַת אַבָּא וְיַת אִמָא וְיַת אֲחַי וְיַת אֲחָתֵי וְיַת כָּל דִילְהוֹן וּתְשֵׁזְבוּן יַת נַפְשָׁתָנָא מִמוֹתָא:
וַאֲמַרוּ לָהּ גֻבְרַיָא נַפְשָׁנָא מְסִירָן חֲלַף נַפְשָׁתֵיכוֹן לְמֵמָת אִם לָא תְחַווּן יַת פִּתְגָמָנָא דֵין וִיהֵי כַּד יִתֵּן יְיָ לָנָא יַת אַרְעָא וְנַעֲבֵד עִמָךְ טֵיבוּ וּקְשׁוֹט:
וְשַׁלְשְׁלַתְנוּן בְּאַטוּנָא מִן חֲרַכָּא אֲרֵי בֵיתָהּ בְּכוֹתֵל שׁוּרָא וּבְשׁוּרָא הִיא יָתְבָה:
וַאֲמַרַת לְהוֹן לְטוּרָא אֱזִילוּ דִלְמָא יְעַרְעוּן בְּכוֹן רָדְפַיָא וְתִטַמְרוּן תַּמָן תְּלָתָא יוֹמִין עַד דִיתוּבוּן רָדְפַיָא וּבָתַר כֵּן תְּהָכוּן לְאוֹרְחַתְכוֹן:
וַאֲמַרוּ לָהּ גֻבְרַיָא זַכָּאִין אֲנַחְנָא מִמוֹמָתֵךְ הָדָא דְקַיֵמְתָּ עֲלָנָא:
הָא אֲנַחְנָא עָלִין בְּאַרְעָא יַת תּוֹרָא דְחוּט זְהוֹרִיתָא הָדֵין תִקְטְרִי בַּחֲרַכָּא דְשַׁלְשַלְתָּנָא בֵּיהּ וְיַת אֲבוּךְ וְיַת אִמֵךְ וְיַת אֲחָיְכִי וְיַת כָּל בֵּית אָבוּךְ תִּכְנְשִׁין לְוָתֵיךְ לְבֵיתָא:
וִיהֵי כָּל דְיִפּוֹק בַּר מִדָשֵׁי בֵּיתֵךְ לְבָרָא חוֹבַת קְטוֹלֵיהּ בְּרֵישֵׁיהּ וַאֲנַחְנָא זַכָּאִין וְכֹל דִי יְהֵי עִמָךְ בְּבֵיתָא חוֹבַת קְטוֹלֵיהּ בְּרֵישָׁנָא אִם יַד אֱנָשׁ תְהֵי בֵיהּ:
וְאִם תְּחַוִין יַת פִּתְגָמָנָא דֵין וּנְהֵי זַכָּאִין מִמוֹמָתִיךְ דְקַיֵימְתְּ עֲלָנָא:
וַאֲמָרַת כְּפִתְגָמֵיכוֹן כָּן הוּא וּשְׁלַחַתְנוּן וַאֲזָלוּ וְקִטְרַת יַת תּוֹרָא דִזְהוֹרִיתָא בַּחֲרַכָּא:
וַאֲזַלוּ וְעָלוּ לְטוֹרָא וִיתָבוּ תַמָן תְּלָתָא יוֹמִין עַד דְתָבוּ רָדְפַיָא וּבְעוֹ רָדְפַיָא בְּכָל אָרְחָא וְלָא אַשְׁכָּחוּ:
וְתָבוּ תְּרֵין גֻבְרָין וּנְחִיתוּ מִן טוּרָא וַעֲבָרוּ וַאֲתוֹ לְוַת יְהוֹשֻׁעַ בַּר נוּן וְאִשְׁתְּעִיאוּ לֵיהּ יַת כָּל דְעַרָעָא יַתְהוֹן:
וַאֲמַרוּ לִיהוֹשֻׁעַ אֲרֵי מְסַר יְיָ בִּידָנָא יַת כָּל אַרְעָא וְאַף אִתְּבַּרוּ כָּל יָתְבֵי אַרְעָא מִן קֳדָמָנָא:
