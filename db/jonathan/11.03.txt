וִיהוֹרָם בַּר אַחְאָב מְלַךְ עַל יִשְׂרָאֵל בְּשֹׁמְרוֹן בִּשְׁנַת תַּמְנֵי עַסְרֵי לִיהוֹשָׁפָט מֶלֶךְ שִׁבְטָא דְבֵית יְהוּדָה וּמְלַךְ תַּרְתָּא עַסְרֵי שְׁנִין:
וַעֲבַד דְבִישׁ קֳדָם יְיָ לְחוֹד לָא כַאֲבוּהִי וּכְאִמֵיהּ וְאַעְדֵי יַת קָמַת בְּעָלָא דַעֲבַד אֲבוּהִי:
לְחוֹד בְּחוֹבֵי יְרָבְעָם בַּר נְבָט דְחַיֵב יַת יִשְׂרָאֵל אִתְרְעֵי לָא סְטָא מִנָהּ:
וּמֵישַׁע מַלְכָּא דְמוֹאָב הֲוָה מָרֵי גֵיתִין וּמְתִיב תּוּקְרָבָא לְמַלְכָּא דְיִשְׂרָאֵל שְׁנָא בִּשְׁנָא מְאָה אַלְפִין תּוֹרִין דְפִטְמָא וּמְאָה אַלְפִין דִכְרִין דְרַעְיָא:
וַהֲוָה כַּד מִית אַחְאָב וּמְרִיד מַלְכָּא דְמוֹאָב בְּמַלְכָּא דְיִשְׂרָאֵל:
וּנְפַק מַלְכָּא יְהוֹרָם בְּיוֹמָא הַהוּא מִשֹׁמְרוֹן וּמְנָא יַת כָּל יִשְׂרָאֵל:
וַאֲזַל וּשְׁלַח לְוַת יְהוֹשָׁפָט מַלְכָּא דְבֵית יְהוּדָה לְמֵימַר מַלְכָּא דְמוֹאָב מְרִיד בִּי הֲתֵיזֵיל עִמִי לְוַת מוֹאָב בִּקְרָבָא וַאֲמַר אֶסַק אֲנָא כְוָתָךְ עַמִי כְעַמָךְ סוּסְוָי כְּסוּסְוָךְ:
וַאֲמַר בְּאֵידָא אוֹרְחָא נִיסַק וַאֲמַר אוֹרַח מַדְבְּרָא דֶאֱדוֹם:
וַאֲזַל מַלְכָּא דְיִשְׂרָאֵל וּמֶלֶךְ שִׁבְטָא דְבֵית יְהוּדָה וּמַלְכָּא דֶאֱדוֹם וְאַסְחָרוּ מַהֲלַךְ שַׁבְעָא יוֹמִין וְלָא הֲוָה מַיָא לְמַשְׁרִיתָא וְלִבְעִירָא דְעִמְהוֹן:
וַאֲמַר מַלְכָּא דְיִשְׂרָאֵל וַי אֲרֵי זַמִין יְיָ לִתְלָתָא מַלְכַיָא הָאִלֵין לְמִמְסַר יַתְהוֹן בִּידָא דְמוֹאָב:
וַאֲמַר יְהוֹשָׁפָט הֲלֵית כָּא נְבִיָא דַייָ וְנִשְׁאַל יַת פִּתְגָמָא דַייָ מִנֵיהּ וַאֲתֵב חַד מֵעַבְדֵי מַלְכָּא דְיִשְׂרָאֵל וַאֲמַר הָכָא אֱלִישָׁע בַּר שָׁפָט דְשַׁמֵשׁ יַת אֵלִיָהוּ:
וַאֲמַר יְהוֹשָׁפָט אִית עִמֵיהּ פִּתְגָמָא דַייָ וּנְחָתוּ לְוָתֵיהּ מַלְכָּא דְיִשְׂרָאֵל וִיהוֹשָׁפָט וּמַלְכָּא דֶאֱדוֹם:
וַאֲמַר אֱלִישָׁע לְמַלְכָּא דְיִשְׂרָאֵל מַה לִי וָלָךְ אֱזֵיל לְוַת נְבִיֵי אֲבוּךְ וּלְוַת נְבִיֵי אִמָךְ וַאֲמַר לֵיהּ מַלְכָּא דְיִשְׂרָאֵל בְּבָעוּ לָא תִדְכַּר חוֹבִין רִשְׁעָתָא הַהִיא בָּעֵי עֲלָנָא רַחֲמִין אֲרֵי זַמִין יְיָ לִתְלָתָא מַלְכַיָא הָאִלֵין לְמִמְסַר יַתְהוֹן בִּידָא מוֹאֲבָאֵי:
וַאֲמַר אֱלִישָׁע קַיָם הוּא יְיָ צְבָאוֹת דִמְשַׁמֵשְׁנָא קֳדָמוֹהִי אֲרֵי אִלוּלָא פוֹן אַפֵּי יְהוֹשָׁפָט מֶלֶךְ שִׁבְטָא דְבֵית יְהוּדָה אֲנָא נָסֵיב אִם אִסְתַּכַּל בָּךְ וְאִם אַחֲזִינָךְ:
וּכְעַן אַיְתוֹ לִי גַבְרָא דְיָדַע לְנַגָנָא בְּכִנוֹרָא וַהֲוָה כַּד נַגֵן דִמְנַגֵן וּשַׁרְיָא עֲלוֹהִי רוּחַ נְבוּאָה מִן קֳדָם יְיָ:
וַאֲמַר כִּדְנַן אֲמַר יְיָ יִתְעֲבֵיד נַחֲלָא הָדֵין פַּצִידִין פַּצִידִין:
אֲרֵי כִּדְנַן אֲמַר יְיָ לָא תֶחֱזוּן רוּחָא וְלָא תֶחֱזוּן מִטְרָא וְנַחֲלָא הַהוּא יִתְמְלֵי מַיָא וְתִשְׁתּוּן אַתּוּן וְגֵיתֵיכוֹן וּבְעִירְכוֹן:
וּקְלִילָא דָא קֳדָם יְיָ וְיִמְסַר יַת מוֹאֲבָאֵי בְּיֶדְכוֹן:
וְתִמְחוּן כָּל כְּרַךְ תַּקִיף וְכָל קִרְוִין כְּרִיכַן וְכָל אִילָן דְיָאֵי תְּעַקְרִין וְכָל מַבּוּעַי מַיָא תִּסְתְּמוּן וְכָל אַחֲסַנְתָּא שַׁפֶּרְתָּא תְטוּמוּן בְּאַבְנַיָא:
וַהֲוָה בְּצַפְרָא כְּמִיסַק מִנְחָתָא וְהָא מַיָא אָתָן בְּאוֹרַח אֱדוֹם וְאִתְמְלִיאַת אַרְעָא יַת מַיָא:
וְכָל מוֹאֲבָאֵי שְׁמָעוּ אֲרֵי סְלִיקוּ מַלְכַיָא לְאַגָחָא קְרָבָא בְּהוֹן וְאִתְכְּנִישׁוּ מִכֹּל אֲסִירֵי סַיְפָא וּלְעֵילָא וְקָמוּ עַל תְּחוּמָא:
וְאַקְדִימוּ בְּצַפְרָא וְשִׁמְשָׁא נְחַת עַל מַיָא וַחֲזוֹ מוֹאֲבָאֵי מִקֳבֵל יַת מַיָא סוּמָקִין כִּדְמָא:
וַאֲמַרוּ דַם דֵין אִתְגְרָאָה אִתְגְרִיאוּ מַלְכַיָא וּקְטַלוּ גְבַר יַת חַבְרֵיהּ וּכְעַן פּוּקוּ לְבִּיזְתָא מוֹאֲבָאֵי:
וַאֲתוֹ לְמַשְׁרִיתָא דְיִשְׂרָאֵל וְקָמוּ יִשְׂרָאֵל וּמְחוֹ יַת מוֹאֲבָאֵי וַאֲפָּכוּ מִן קֳדָמֵיהוֹן וּמְחוֹ בְהוֹן וּקְטַלוּ יַת מוֹאֲבָאֵי:
וְקִרְוַיָא פַּגָרוּ וְכָל אַחֲסַנְתָּא שַׁפִּירָא רְמוֹ גְבַר אַבְנֵיהּ וּמַלְאוּהָא וְכָל מַבּוּעֵי מַיָא סְתָמוּ וְכָל אִילָן דְיָאֵי עֲקָרוּ עַד דְלָא אִשְׁתַּאֲרַת אַבְנָא בְּקַרְתָּא דְלָא פַּגְרוּהָא וְאִסְתַּחֲרוּ קְלָעַיָא וּמְחוּהָא:
וַחֲזָא מַלְכָּא דְמוֹאָב אֲרֵי תַקִיפוּ מִנֵיהּ עָבְדֵי קְרָבָא וּדְבַר עִמֵיהּ שְׁבַע מְאָה גַבְרָא שַׁלְפֵי סַיְפָא לְאַפְנָאָה לְוַת מַלְכָּא דֶאֱדוֹם וְלָא יָכִילוּ:
וּנְסִיב יַת בְּרֵיהּ בּוּכְרָא דַעֲתִיד דְיִמְלוֹךְ תְּחוֹתוֹהִי וְאַסְקֵיהּ עֲלָתָא עַל שׁוּרָא וַהֲוָה רְגַז רַב עַל יִשְׂרָאֵל וְאִסְתַּלָקוּ מִנֵיהּ וְתָבוּ לְאַרְעָא:
