וְצַלִיאַת חַנָה בְּרוּחַ נְבוּאָה וַאֲמַרַת כְּבַר שְׁמוּאֵל בְּרִי עֲתִיד לְמֶהֱוֵי נְבִיָא עַל יִשְׂרָאֵל בְּיוֹמוֹהִי יִתְפָּרְקוּן מִידָא דִפְלִשְׁתָּאֵי וְעַל יְדוֹהִי יִתְעַבְּדַן לְהוֹן נִסִין וּגְבוּרָן בְּכֵן תְּקוֹף לִבִּי בְּחוּלְקָא דִיהַב לִי יְיָ וְאַף הֵימָן בַּר יוֹאֵל בַּר בְּרִי שְׁמוּאֵל עָתִיד דִיקוּם הוּא וְאַרְבַּעַת עֲשַׂר בְּנוֹהִי לְמֶהֱוֵי אָמְרִין בְּשִׁירָה עַל יְדֵי נִבְלִין וְכִנוֹרִין עִם אֲחֵיהוֹן לֵיוָאֵי לְשַׁבָּחָא בְּבֵית מַקְדְשָׁא בְּכֵן רָמַת קַרְנִי בְּמַתַּנְתָּא דְמַנִי לִי יְיָ וְאַף עַל פּוּרְעָנוּת נִסָא דַעֲתִיד לְמֶהֱוֵי בִּפְלִשְׁתָּאֵי דַעֲתִידִין דְיֵיתוּן יַת אֲרוֹנָא דַייָ בַּעֲגַלְתָּא חֲדַתָּא וְעִמֵיהּ קוּרְבַּן אֲשָׁמָא בְּכֵן תֵּימַר כְּנִשְׁתָּא דְיִשְׂרָאֵל אֶפְתָּח פּוּמִי לְמַלָלָא רַבְרְבָן עַל בַּעֲלֵי דְבָבַי אֲרֵי חֲדִיתִי בְּפוּרְקָנָךְ:
עַל סַנְחֵרִיב מַלְכָּא דְאַתּוּב אִתְנַבִּיאַת וַאֲמַרַת עֲתִיד דִיקוּם הוּא וְכָל חֲיָלוּתֵיהּ עַל יְרוּשְׁלֵם וְנֵס סַגִי יִתְעֲבֵיד בֵּיהּ תַּמָן יִפְלוּן פִּגְרֵי מַשִׁרְיָתֵיהּ בְּכֵן יוֹדוּן כָּל עַמְמַיָא אוּמַיָא וְלִשָׁנַיָא וְיֵימְרוּן לֵית דְקַדִישׁ אֶלָא יְיָ אֲרֵי לֵית בַּר מִנָךְ וַעֲמָךְ יֵימְרוּן לֵית תַּקִיף אֶלָא אֱלָהָנָא:
עַל נְבוּכַדְנֶצַר מַלְכָּא דְבָבֶל אִתְנַבִּיאַת וַאֲמַרַת אַתּוּן כַּסְדָאֵי וְכָל עַמְמַיָא דַעֲתִידִין לְמִשְׁלַט בְּיִשְׂרָאֵל לָא תִסְגוּן לְמַלָלָא רַבְרְבָן לָא יִפְקוּן גִידוּפִין מִפּוּמְכוֹן אֲרֵי אֱלָהָא יְדַע כוֹלָא וְעַל כָּל עוֹבָדוֹהִי מְתִיחַ דִינֵיהּ אַף לְכוֹן עֲתִיד לְשַׁלָמָא פּוּרְעָנוּת חוֹבֵיכוֹן:
עַל מַלְכוּת מַקְדוֹן אִתְנַבִּיאַת וַאֲמַרַת קַשְׁתַּת דְגִבָּרֵי מִקְדוּנָאֵי יִתַּבְּרִין וּדְבֵית חַשְׁמוֹנָאֵי דַהֲווֹ חַלָשִׁין יִתְעַבְּדָן לְהוֹן נִסִין וּגְבוּרָן:
עַל בְּנוֹהִי דְהָמָן אִתְנַבִּיאַת וַאֲמַרַת דַהֲווֹ שָׂבְעִין בְּלַחְמָא וְגָאָן בְּעוּתְרָא וְסַגִיאִין בְמָמוֹנָא אִתְמַסְכְּנוּ תָּבוּ לְאִתַּגָרָא בִּלְחֵם וּמָזוֹן פּוּמְהוֹן מָרְדְכַי וְאֶסְתֵּר דַהֲווֹ חֲשִׁיכִין וּמִסְכְּנִין עֲתָרוּ וְאִיתְנְשִׁיאוּ יַת מִסְכְּנוּתְהוֹן תָּבוּ לְמֶהֱוֵי בְנֵי חוֹרִין כֵּן יְרוּשְׁלֵם דַהֲוַת כְּאִתְּתָא עֲקְרָא עֲתִידָא דְתִתְמְלֵי מֵעַם גַלְוָתָהָא וְרוֹמֵי דְמַלְיָא סַגִי עַמְמַיָא יְסוּפוּן מַשִׁרְיָתָהָא תְּצְדֵי וְתֵיחֲרוּב:
כָּל אִלֵין גְבוּרָתָא דַייָ דְהוּא שַׁלִיט בְעַלְמָא מֵמִית וַאֲמַר לְאַחָאָה וּמָחִית לִשְׁאוֹל וְאַף עֲתִיד לְאַסָקָא בְּחַיֵי עַלְמָא:
יְיָ מְמַסְכֵּין וּמְעַתֵּיר מַשְׁפִּיל אַף מֵרִים:
מֵקִים מֵעַפְרָא מִסְכֵּינָא מִקִלְקַלְתָּא מֵרִים חֲשִׁיכָא לְאוֹתָבוּתְהוֹן עִם צַדִיקַיָא רַבְרְבֵי עַלְמָא וְכוּרְסֵי יְקָרָא מַחֲסִין לְהוֹן אֲרֵי קֳדָם יְיָ גְלָן עוֹבָדֵי בְּנֵי אֱנָשָׁא מִלְרַע אַתְקֵין גֵהִנָם לְרַשִׁיעַיָא עָבְרֵי עַל מֵימְרֵיהּ וְצַדִיקַיָא עַבְדֵי רְעוּתֵיהּ שַׁכְלֵיל לְהוֹן תֵּבֵל:
וּגְוִיַת עַבְדוֹהִי צַדִיקַיָא יִטַר מִגֵהִנָם וְרַשִׁיעַיָא בְגֵהִנָם בַּחֲשׁוֹכָא יִתְדְנוּן לְהוֹדָעָא אֲרֵי לָא מַן דְאִית בֵּיהּ חֵילָא זַכֵּי לְיוֹם דִינָא:
יְיָ יִתַבַּר בַּעֲלֵי דְבָבָא דְקַיְמִין לְאַבְאָשָׁא לְעַמֵיהּ עֲלֵיהוֹן מִן שְׁמַיָא בְּקַל רָם יַשְׁקֵיף יְיָ יַעְבֵּיד פּוּרְעֲנוּת מִן מָגוֹג וּמִמַשִׁרְיַת עַמְמִין חֲטוּפִין דְאָתָן עִמֵיהּ מִסְיָפֵי אַרְעָא וְיִתֵּן תּוּקְפָא לְמַלְכֵּיהּ וִירַבֵּי מַלְכוּת מְשִׁיחֵיהּ:
וַאֲזַל אֶלְקָנָה לְרָמָתָא לְבֵיתֵיהּ וְרַבְיָא הֲוָה מְשַׁמֵשׁ קֳדָם יְיָ בְּחַיֵי עֵלִי כַהֲנָא:
וּבְנֵי עֵלִי גַבְרִין רַשִׁעַיָא לָא הֲווֹ יָדְעִין לְמִדְחַל קֳדָם יְיָ:
וְנִימוּסָא דְכַהֲנַיָא מִן עַמָא כָּל גְבַר דְנָכֵיס נִכְסְתָא וְאָתֵי עוּלֵימָא דְכַהֲנָא כִּמְבַשֵׁל בִּסְרָא וּמַשִׁילְיָא דִי לֵיהּ תְּלַת שְׁנִין בִּידֵיהּ:
וּקְבַע לֵיהּ בְּכִיוֹרָא אוֹ בְדוּדָא אוֹ בִקְדֵירָא אוֹ בְמִלֵיסָא כֹּל דְמַסִיק מַשִׁלְיָא נְסִיב כַּהֲנָא לְנַפְשֵׁיהּ כְּדֵין עָבְדִין לְכָל יִשְׂרָאֵל דְאָתָן לְדַבָּחָא תַּמָן בְּשִׁלֹה:
אַף עַד דְלָא יִתַּסְקוּן יַת תַּרְבָּא לְמַדְבְּחָא וְיֵתֵי עוּלֵימָא דְכַהֲנָא וַאֲמַר לְגַבְרָא דְדַבַּח הַב בִּסְרָא לְמִטְוֵי לְכַהֲנָא וְלָא יִסַב מִנָךְ בְּשַׂר מְבַשֵׁל אֱלָהֵן כַּד חָי:
וַאֲמַר לֵיהּ גַבְרָא אוֹרִיךְ עַד דְיִתַּסְקוּן תַּרְבַּיָא וְסַב לָךְ כְּמָא דְרַעֲוָא נַפְשָׁךְ וַאֲמַר לֵיהּ אֲרֵי כְעַן תִּתֵּן וְאִם לָא יִסְבּוּן מִנָךְ בְּעַל כּוֹרְחָךְ:
וַהֲוַת חוֹב עוּלֵמַיָא סַגִי לַחֲדָא קֳדָם יְיָ אֲרֵי בָזוּ גַבְרַיָא יַת קוּרְבָּנַיָא דַיְיָ:
וּשְׁמוּאֵל מְשַׁמֵשׁ קֳדָם יְיָ עוּלֵימָא אֲסִיר כַּרְדוּט דְבוּץ:
וּמְעִיל זְעֵיר עָבְדָא לֵיהּ אִמֵיהּ וּמַסְקָא לֵיהּ מִזְמַן מוֹעֵד לְמוֹעֵד בְּמִסְקָהּ עִם בַּעֲלָהּ לְדַבָּחָא יַת דְבַח מוֹעֲדַיָא:
וּבָרִיךְ עֵלִי יַת אֶלְקָנָה וְיַת אִתְּתֵיהּ וַאֲמַר יְקַיֵים יְיָ לָךְ בְּנִין כַּשְׁרִין מִן אִתְּתָא הָדָא חֲלַף שְׁאֶלְתָּא דְשָׁאֵל קֳדָם יְיָ וַאֲזַלוּ לְאַתְרֵיהוֹן:
אֲרֵי עַל דָכְרָנָא דְחַנָה קֳדָם יְיָ וְעַדִיאַת וִילֵידַת תְּלָתָא בְנִין וְתַרְתֵּין בְּנָן וּרְבָא רַבְיָא שְׁמוּאֵל מְשַׁמֵשׁ קֳדָם יְיָ:
וְעֵלִי סִיב לַחֲדָא וּשְׁמַע יַת כָּל דְעָבְדִין בְּנוֹהִי לְכָל יִשְׂרָאֵל וְיַת דִי שָׁכְבִין יַת נְשַׁיָא דְאַתְיָן לְצַלָאָה בִּתְרַע מַשְׁכַּן זִמְנָא:
וַאֲמַר לְהוֹן לְמָא תַעְבְּדוּן כְּפִתְגָמַיָא הָאֵלִין דִי אֲנָא שְׁמַע יַת פִּתְגָמֵיכוֹן בִּישִׁין מִן כָּל עַמָא אִלֵין:
לָא בְּנַי אֲרֵי לָא תַקְנָא שְׁמוּעֲתָא דִי אֲנָא שְׁמַע דְרַנְנִין עַמָא דַיָי:
אִם יֵחוֹב גְבַר לִגְבַר הֲלָא יֵיתוּן לָקֳדָם דַיָנָא וְיִשְׁמַע מִלֵיהוֹן וְיַפְלֵי בֵינֵיהוֹן וְאִם קֳדָם יְיָ יֵיחוֹב גַבְרָא מִמָן יִבְעֵי וְיִשְׁתְּבֵיק לֵיהּ וְלָא קַבִּילוּ לְמֵימַר אֲבוּהוֹן אֲרֵי רַעֲוָא קֳדָם יְיָ לְקַטָלוּתְהוֹן:
וְרַבְיָא שְׁמוּאֵל אָזִיל וְסַגִי וּשְׁמֵיהּ טָב וְתַקְנָן אוֹרְחָתֵיהּ קֳדָם יְיָ וּתְרִיצִין עוֹבָדוֹהִי בְּגוֹ בְּנֵי אֱנָשָׁא:
וַאֲתָא נְבִיָא דַייָ לְוַת עֵלִי וַאֲמַר לֵיהּ כִּדְנַן אֲמַר יְיָ אִתְגְלָאָה אִתְגְלֵיתִי לְבֵית אֲבוּךְ כַּד הֲווֹ בְמִצְרַיִם וּמְשַׁעְבְּדִין לְבֵית פַּרְעֹה:
וְאִתְרְעֵיתִי בֵיהּ מִכָּל שִׁבְטַיָא דְיִשְׂרָאֵל קֳדָמַי לְמֶהֱוֵי כָהֵין לְאַסָקָא קוּרְבַּנָא עַל מַדְבְּחִי לְאַקְטָרָא קְטוֹרֶת בּוּסְמִין לְמִלְבַּשׁ אֵפוֹד לְשַׁמָשָׁא קֳדָמַי וִיהָבִית לְבֵית אָבוּךְ יַת כָּל קוּרְבָּנַיָא דִבְנֵי יִשְׂרָאֵל:
לְמָא אַתּוּן אַנְסִין בְּנִכְסַת קוּדְשַׁיָא וּבְקוּרְבָּנִי דְפַקְדִית לִקָרָבָא קֳדָמַי בְּבֵית מַקְדְשִׁי וְיָקַרְתָּ יַת בְּנָךְ קֳדָמֵיתָא לְאוֹכָלוּתְהוֹן מֵרֵישׁ כָּל קוּרְבַּן מִנְחַת יִשְׂרָאֵל דְעַמִי:
בְּכֵן אֲמַר יְיָ אֱלָהָא דְיִשְׂרָאֵל מֵימַר אֲמַרִית בֵּיתָךְ וּבֵית אָבוּךְ יְשַׁמְשׁוּן קֳדָמַי עַד עַלְמָא וּכְעַן אֲמַר יְיָ קוּשְׁטָא אִינוּן דִינַי אֲרֵי לְדִמְיַקְרִין קֳדָמַי אַיְקַר וְדִמְבַסְרִין עַל שְׁמִי יִתְשָׁטוּן:
הָא יוֹמַיָא אָתָאן וְאֶקְטוֹף יַת תְּקוֹף דַרְעָךְ וְיַת תְּקוֹף דְרַע בֵּית אָבוּךְ מִלְמֶהֱוֵי סִיב בְּבֵיתָךְ:
וּתְהֵי מִתְתַּכֵּל וּתְהֵי חָזֵי בְעַקְתָא דְתֵיתֵי עַל אֱנַשׁ בֵּיתֵךְ בְּחוֹבִין דְחַבְתּוּן קֳדָמַי בְּבֵית מַקְדְשִׁי וּבָתַר כֵּן אַיְיתִי טַבְתָא עַל יִשְׂרָאֵל וְלָא יְהֵי סִיב בְּבֵיתָךְ כָּל יוֹמַיָא:
וָאֱנַשׁ לָא אַפְסִיק לָךְ מִן מַדְבְּחִי לְחַשָׁכָא יַת עֵינָךְ וּלְאַפָּחָא יַת נַפְשָׁךְ וְכָל סַגְיוֹת בֵּיתָךְ יְמוּתוּן עוּלֵימִין:
וְדֵין לָךְ אָתָא דְתֵיתֵי לְוַת תְּרֵין בְּנָךְ לְוַת חָפְנִי וּפִנְחָס בְּיוֹמָא חַד יִתְקַטְלוּן תַּרְוֵיהוֹן:
וַאֲקֵים קֳדָמַי כָּהֵין מְהֵימָן דִכְמֵימְרִי וְכִרְעוּתִי יַעְבֵּד וַאֲקֵים לֵיהּ מַלְכוּ קַיָמָא וִישַׁמֵשׁ קֳדָם מְשִׁיחִי כָּל יוֹמַיָא:
וִיהֵי כָּל דְיִשְׁתָּאֵר בְּבֵיתָךְ יֵיתֵי לְאִשְׁתַּפָּלָא לֵיהּ לְמָעָא דִכְסַף וּפִתָּא דְלַחְמָא וְיֵימַר מְנִינִי כְעַן לַחֲדָא מִמַטְרַת כַּהֲנַיָא לְמֵיכַל פִּתָּא דְלַחְמָא:
