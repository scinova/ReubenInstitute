אֲמַר נְבִיָא וַי לִי אֲרֵי הֲוֵיתִי כְּמֵיסַף טָבַיָא בְּעִדַן דַאֲבָדוּ חֲסִידַיָא מִן אַרְעָא הָא כִסְיוּפֵי קַיְטָא כְּעוֹלְלָן בָּתַר קְטוֹף לֵית גְבַר דְבֵיהּ עוֹבָדִין טָבִין לְטָבַיָא חֲמָדַת נַפְשִׁי:
אָבָדוּ חֲסִידַיָא מִן אַרְעָא וּדְקָשִׁיט בֶּאֱנָשָׁא לֵית כּוּלְהוֹן לְמִשְׁפַּךְ דַם זַכַּאי כַּמְנִין גְבַר יַת אֲחוֹהִי מָסְרִין לִגְמִירָא:
מַבְאֲשִׁין בִּידֵיהוֹן וְלָא מוֹטִיבִין שִׁלְטוֹנָא שְׁאֵיל וְדַיָנָא אֲמַר עֱבֵיד לִי דְאֶשְׁלַם לָךְ וְרַבָּא מְמַלֵיל תַּחֲמוּדֵי נַפְשֵׁיהּ וַי עֲלֵיהוֹן עַל דְקַלְקְלוּתָא:
טָבָא דִבְהוֹן קָשֵׁי לְמִפַּק מִן יְדֵיהּ כַּד מִן חִדְקָא וּדְקָשֵׁיט בְּהוֹן מִסְיוּגַת סְיַג בִּישׁ יוֹם סִבּוּרָךְ לְטַב עִדַן סְעוּרַת בִּישְׁתָךְ מְטַת כְּעַן יְהֵי עֲרָבוּ לְהוֹן:
לָא תְהֵימְנוּן בִּרְחֵים וְלָא תִתְרַחֲצוּן בְּקָרִיב מֵאִתַּת קְיָמָךְ טַר מְלֵי פוּמָךְ:
אֲרֵי בְרָא מְנַבֵּל אַבָּא בְּרַתָּא מְנַצְיָא עִם אִמָהּ כַּלְתָא מְקִילָא בַּחֲמָתָהּ סָנְאוֹהִי דְגַבְרָא אֱנָשֵׁי בֵיתֵיהּ:
וַאֲנָא בְּמֵימְרָא דַייָ אֲבוּעַ אֲדוּץ לֶאֱלָהָא עָבֵיד פּוּרְקָנִי יְקַבֵּיל צְלוֹתִי אֱלָהִי:
לָא תֶחְדוּן בְּעֵיל דְבָבוּתִי לִי דְנָפְלֵית אָקוּם אֲרֵי יְתִיבֵית כִדְבְקִבְלָא יְיָ יַנְהַר עֲלָי:
אֲמָרֵית יְרוּשְׁלֵם לְוָט מִן קֳדָם יְיָ קַבְּלֵית אֲרֵי חָבִית קֳדָמוֹהִי עַד דְיָדִין דִינִי וְיִתְפְּרַע עוּלְבָּנִי יַפְקִינַנִי לִנְהוֹר אֶחְזֵי בִּזְכוּתֵיהּ:
וְתֶחֱזֵי בְּעֵיל דְבָבוּתִי וְתַחֲפִינָהּ בַּהֲתָא דַהֲוַת אָמְרָה לִי אָן דְאִתְפְּרֵיקְתָּא בְּמֵימְרָא דַייָ אֱלָהָךְ עֵינַי יֶחֶזְיָן בְּמַפַּלְתְּהוֹן כְּעַן תְּהֵי לְדָיִשׁ כְּטִין שׁוּקִין:
בְּעִדָנָא הַהִיא יִתְבַּנְיָן כְּנִשְׁתָּא דְיִשְׂרָאֵל בְּעִדָנָא הַהִיא יִתְבַּטְלָן גְזֵירַת עַמְמַיָא:
בְּעִדָנָא הַהִיא יִתְכַּנְשׁוּן גַלְוָתָא דְמִן אַתּוּר וְקִרְוֵי תוּקְפָא וּדְמִן חוּרְמִינִי רַבְּתָא וְקִרְוֵי צְיָרָא וְעַד פְּרָת וְיַמָא מַעֲרָבָא וְטוּרֵי טוּרָא:
וּתְהֵי אַרְעָא לְצָדוּ עַל יַתְבָהָא מִפְּרִי עוֹבְדֵיהוֹן:
פַּרְנֵס עַמָךְ בְּמֵימְרָךְ עַמָא דְאַחֲסַנְתָּךְ בְּעָלְמָא דְהוּא עָתִיד לְאִיתְחַדְתָּא יִשְׁרוֹן בִּלְחוֹדֵיהוֹן דַהֲווֹ צָדְיָן כְּחוּרְשָׁא יִתְיַתְבוּן כְּכַרְמָלָא וְיִתְפַּרְנְסוּן בְּאַרְעָא מַתְּנַן וְגִלְעָד כְּיוֹמֵי עַלְמִין:
כְּיוֹמֵי מִפַקְהוֹן מֵאַרְעָא דְמִצְרָיִם אַחְזִינוּן פְּרִישָׁן:
יֶחֱזוּן עַמְמַיָא וְיִבְהֲתוּן מִכָּל גְבוּרְתְּהוּן יְשַׁווּן יְדֵיהוֹן עַל פּוּמְהוֹן וְאוּדְנֵיהוֹן יִתְחַרְשָׁן:
יִשְׁתַּטְחוּן עַל אַפֵּיהוֹן עַל אַרְעָא כְּחִוְיָא כְּזַחֲלֵי עַפְרָא יְזוּעוּן מִבִּרְנְיָתֵיהוֹן מִן קֳדָם יְיָ אֱלָהָנָא יִתַּבְרוּן וְיִדַחֲלוּן מִן קֳדָמָךְ:
לֵית בַּר מִנָךְ אַתְּ הוּא אֱלָהָא שָׁבֵיק לְעַוְיָן וּמַעֲבַר עַל חוֹבִין לִשְׁאָרָא דְאַחֲסַנְתֵּיהּ לָא מוֹרִיךְ לְעַלְמִין רוּגְזֵיהּ אֲרֵי רָעַי לְאַיְטָבָא הוּא:
יְתוּב מֵימְרִי לְרָחֲמָא עֲלָנָא וְיִכְבּוֹשׁ עַל חוֹבָנָא בִּרְחִמְתֵּיהּ וְיִרְמֵי לְעוּמְקֵי יַמָא כָּל חֲטָאֵי יִשְׂרָאֵל:
תִּתְיְהֵיב קוּשְׁטֵיהּ דְיַעֲקֹב לִבְנוֹהִי כְּמָא דְקַיֵמְתָּא לֵיהּ בְּבֵית אֵל טַבְוָת אַבְרָהָם לְזַרְעֵיהּ בַּתְרוֹהִי כְּמָא דְקַיֵמְתָּא לֵיהּ בֵּין בַּתְרַיָא תִּדְכַּר לָנָא עֲקֵידַת יִצְחָק דְאִתְעֲקֵד עַל גַבֵּי מַדְבְּחָא קֳדָמָךְ תַּעֲבֵיד עִמָנָא טַבְוָן דְקַיֵמְתָּא לְאַבְהָתָנָא מִיוֹמֵי קֶדֶם:
