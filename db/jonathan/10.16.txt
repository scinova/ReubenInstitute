וַהֲוָה פִּתְגַם נְבוּאָה מִן קֳדָם יְיָ עִם יֵהוּא בַר חֲנָנִי עַל בַּעְשָׁא לְמֵימָר:
חֲלַף דְאַפְרֵישְׁתָּךְ מִן עַפְרָא וּמַנִיתָךְ מַלְכָּא עַל עַמִי יִשְׂרָאֵל וַאֲזַלְתָּא בְּאוֹרַח יָרָבְעָם וְחַיֵבְתָּא יַת עַמִי יִשְׂרָאֵל לְאַרְגָזָא קֳדָמַי בְּחוֹבֵיהוֹן:
הָא אֲנָא מְפַלֵי בָּתַר בַּעְשָׁא וּבָתַר בֵּיתֵיהּ וְאֶתֵּן יַת בֵּיתָךְ כְּבֵית יָרָבְעָם בַּר נְבָט:
דִימוּת לְבַעְשָׁא בְּקַרְתָּא יֵכְלוּן כַּלְבַיָא וְדִימוּת לֵיהּ בְּחַקְלָא יֵכְלוּן עוֹפָא דִשְׁמַיָא:
וּשְׁאַר פִּתְגָמֵי בַעְשָׁא וְדִעֲבַד וּגְבוּרְתֵּיהּ הֲלָא אִינוּן כְּתִיבִין עַל סֵפֶר פִּתְגָמֵי יוֹמַיָא לְמַלְכֵי יִשְׂרָאֵל:
וּשְׁכִיב בַּעְשָׁא עִם אֲבָהָתוֹהִי וְאִתְקְבַר בְּתִרְצָה וּמְלַךְ אֵלָה בְּרֵיהּ תְּחוֹתוֹהִי:
וְאַף עִם יֵהוּא בַּר חֲנָנִי נָבִיא פִּתְגַם נְבוּאָה מִן קֳדָם יְיָ הֲוָה עַל בַּעְשָׁא וְעַל בֵּיתֵיהּ וְעַל כָּל בִּישְׁתָּא דַעֲבַד קֳדָם יְיָ לְאַרְגָזָא קֳדָמוֹהִי בְּעוֹבָדֵי יְדוֹהִי לְמֶהֱוֵי כְּבֵית יָרָבְעָם וְעַל דִקְטַל יָתֵיהּ:
בִשְׁנַת עַסְרִין וְשִׁית שְׁנִין לְאָסָא מֶלֶךְ שִׁבְטָא דְבֵית יְהוּדָה מְלַךְ אֵלָה בַר בַּעְשָׁא עַל יִשְׂרָאֵל בְּתִרְצָה תַּרְתֵּין שְׁנִין:
וּמְרַד עֲלוֹהִי עַבְדֵיהּ זִמְרִי חַד מִתְרֵין רַבָּנֵי רְתִיכַיָא וְהוּא בְתִרְצָה שָׁתֵי רַוֵי בֵּית אַרְצָא טַעֲוָתָא דִי בְבֵיתָא בְּתִרְצָה:
וַאֲתָא זִמְרִי וּמְחָהִי וְקַטְלֵיהּ בִּשְׁנַת עַסְרִין וּשְׁבַע לְאָסָא מֶלֶךְ שִׁבְטָא דְבֵית יְהוּדָה וּמְלַךְ תְּחוֹתוֹהִי:
וַהֲוָה בְּמִמְלְכֵיהּ כְּמִתְבֵי עַל כּוּרְסֵיהּ קְטַל יַת כָּל בֵּית בַּעְשָׁא לָא אַשְׁאַר לֵיהּ יָדַע מַדָע וְקָרִיבוֹהִי וְחַבְרוֹהִי:
וְשֵׁיצֵי זִמְרִי יַת כָּל בֵּית בַּעְשָׁא כְּפִתְגָמָא דַייָ דְמַלֵיל עַל בַּעְשָׁא בְּיַד יֵהוּא נְבִיָא:
עַל כָּל חוֹבֵי בַעְשָׁא וְחוֹבֵי אֵלָה בְרֵיהּ דִי חָבוּ וְדִי חַיִבוּ יַת יִשְׂרָאֵל לְאַרְגָזָא קֳדָם יְיָ אֱלָהָא דְיִשְׂרָאֵל בְּטַעֲוַתְהוֹן:
וּשְׁאַר פִּתְגָמֵי אֵלָה וְכָל דִי עֲבַד הֲלָא אִינוּן כְּתִיבִין עַל סְפַר פִּתְגָמֵי יוֹמַיָא לְמַלְכֵי יִשְׂרָאֵל:
בִּשְׁנַת עַסְרִין וּשְׁבַע שְׁנִין לְאָסָא מֶלֶךְ שִׁבְטָא דְבֵית יְהוּדָה מְלַךְ זִמְרִי שַׁבְעָא יוֹמִין בְּתִרְצָה וְעַמָא שָׁרָן בְּגִבְּתוֹן דִי לִפְלִשְׁתָּאֵי:
וּשְׁמַע עַמָא דְשָׁרָן לְמֵימַר מְרַד זִמְרִי וְאַף קְטַל יַת מַלְכָּא וְאַמְלִיכוּ כָּל יִשְׂרָאֵל יַת עָמְרִי רַב חֵילָא עַל יִשְׂרָאֵל בְּיוֹמָא הַהוּא בְּמַשְׁרִיתָא:
וּסְלִיק עָמְרִי וְכָל יִשְׂרָאֵל עִמֵיהּ מִגִבְּתוֹן וְצָרוּ עַל תִּרְצָה:
וַהֲוָה כַּד חַזָא זִמְרִי אֲרֵי אִתְכְּבִישַׁת קַרְתָּא וְעַל לְאִידְרוֹן בֵּית מַלְכָּא וְאוֹקִיד עֲלוֹהִי יַת בֵּית מַלְכָּא בְּנוּרָא וּמִית:
עַל כָּל חוֹבַיָא דִי חַב לְמֶעְבַד דְבִישׁ קֳדָם יְיָ לְמֵיזַל בְּאוֹרַח יָרָבְעָם וּבְחוֹבֵיהּ דִי עֲבַד לְחַיָבָא יַת יִשְׂרָאֵל:
וּשְׁאַר פִּתְגָמֵי זִמְרִי וּמֵירְדֵיהּ דִמְרַד הֲלָא אִינוּן כְּתִיבִין עַל סְפַר פִּתְגָמֵי יוֹמִין לְמַלְכֵי יִשְׂרָאֵל:
בְּכֵן אִתְפְּלַג עַמָא יִשְׂרָאֵל לְתַרְתֵּי פַּלְגְוָן פַלְגוּת עַמָא הֲוָה בָּתַר תִּבְנִי בַר גִינַת לְאַמְלְכָיוּתֵיהּ וּפַלְגוּת בָּתַר עָמְרִי:
וּתְקֵיף עַמָא דְבָתַר עָמְרִי מִן עַמָא דְבָתַר תִּבְנֵי בַר גִינַת וּמִית תִּבְנִי וּמְלַךְ עָמְרִי:
בִּשְׁנַת תְּלָתִין וַחֲדָא שְׁנִין לְמַלְכָא אָסָא מֶלֶךְ שִׁבְטָא דְבֵית יְהוּדָה מְלַךְ עָמְרִי עַל יִשְׂרָאֵל תַּרְתָּא עַסְרֵי שְׁנִין בְּתִרְצָה מְלַךְ שִׁית שְׁנִין:
וּזְבַן יַת כְּרַכָּא דִלְשׁוֹמְרוֹן מִן שֶׁמֶר בְּתַרְתֵּין כִּכְרִין דִכְסַף וּבְנָא יַת כְּרַכָּא וּקְרָא יַת שְׁמָא דְקַרְתָּא דִבְנָא עַל שׁוּם שֶׁמֶר מָרֵיהּ דְטוּרָא שׁוֹמְרוֹן:
וַעֲבַד עָמְרִי דְבִישׁ קֳדָם יְיָ וּבְאִישׁ מִכֹּל דִי קָדָמוֹהִי:
וַאֲזַל בְּכָל אוֹרְחָא יָרָבְעָם בַּר נְבָט וּבְחוֹבוֹהִי דִי חַיַב יַת יִשְׂרָאֵל לְאַרְגָזָא קֳדָם יְיָ אֱלָהָא דְיִשְׂרָאֵל בְּטַעֲוַתְהוֹן:
וּשְׁאַר פִּתְגָמֵי עָמְרִי דִי עֲבַד וּגְבוּרְתֵּיהּ דִי עֲבַד הֲלָא אִינוּן כְּתִיבִין עַל סְפַר פִּתְגָמֵי יוֹמַיָא לְמַלְכֵי יִשְׂרָאֵל:
וּשְׁכִיב עָמְרִי עִם אֲבָהָתוֹהִי וְאִתְקְבַר בְּשׁוֹמְרוֹן וּמְלַךְ אַחְאָב בְּרֵיהּ תְּחוֹתֵיהּ:
וְאַחְאָב בַּר עָמְרִי מְלַךְ עַל יִשְׂרָאֵל בִּשְׁנַת תְּלָתִין וְתַמְנֵי שְׁנִין לְאָסָא מַלְכָּא שִׁבְטָא דְבֵית יְהוּדָה וּמְלַךְ אַחְאָב בַּר עָמְרִי עַל יִשְׂרָאֵל בְּשׁוֹמְרוֹן עַסְרִין וְתַרְתֵּין שְׁנִין:
וַעֲבַד אַחְאָב בַּר עָמְרִי דְבִישׁ קֳדָם יְיָ מִכֹּל דִי קֳדָמוֹהִי:
וַהֲוָה שִׁיט בְּעֵינוֹהִי לְמֵיזַל בְּחוֹבֵי יָרָבְעָם בַּר נְבָט וּנְסִיב אִתְּתָא יַת אִיזֶבֶל בַּת אֶתְבַּעַל מַלְכָּא דְצִידוֹן וַאֲזַל וּפְלַח יַת בַּעֲלָא וּסְגִיד לֵיהּ:
וַאֲקֵם אֶגוֹרָא לְבָעְלָא בֵּית בְּעָלָא דִבְנָא בְּשׁוֹמְרוֹן:
וַעֲבַד אַחְאָב יַת אֲשֵׁרָתָא וְאוֹסִיף אַחְאָב לְמֶעְבַּד לְאַרְגָזָא קֳדָם יְיָ אֱלָהָא דְיִשְׂרָאֵל מִכֹּל מַלְכֵי יִשְׂרָאֵל דַהֲווֹ קֳדָמוֹהִי:
בְּיוֹמוֹהִי בְּנָא חִיאֵל בֵּית מוֹמֵי יַת יְרִיחֹה קְטַל אֲבִירָם בּוּכְרֵיהּ כַּד שָׁרֵי לְשַׁכְלֵיל יָתָהּ וּשְׂגוּב זְעֵיר בְּנוֹהִי כַּד אָקֵים דְשָׁהָא כְּפִתְגָמָא דַייָ דִי מַלֵיל בִּידָא דִיהוֹשֻׁעַ בַּר נוּן:
