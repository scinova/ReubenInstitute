וַהֲוָה יוֹמָא וַאֲמַר יוֹנָתָן בַּר שָׁאוּל לְעוּלֵימָא נָטֵיל זְיָנֵיהּ אִיתָא וְנֶעְבַד לְאִסְטַרְטִיג פְלִשְׁתָּאֵי דְמֵעֶבְרָא דֵיכִי וְלַאֲבוּהִי לָא חַוֵי:
וְשָׁאוּל יָתֵיב בִּסְיָפֵי גִבְעֲתָא בְּשִׁפּוּלֵי רִמוֹן דִי בְמִגְרוֹן וְעַמָא דְעִמֵיהּ כְּשִׁית מְאָה גַבְרָא:
וַאֲחִיָה בַּר אֲחִיטוּב אֲחוּהִי דְאִיכָבוֹד בַּר פִּינְחָס בַּר עֵלִי כָּהֵין מְשַׁמֵשׁ קֳדָם יְיָ בְּשִׁילֹה לָבֵישׁ אֵיפוֹדָא וְעַמָא לָא יְדַע אֲרֵי אֲזַל יוֹנָתָן:
וּבֵין מְגִיזָתָא דִבְעָא יוֹנָתָן לְמֶעְבַּר עַל אִסְטַרְטִיג פְּלִשְׁתָּאֵי שִׁנָא דְכֵיפָא מֵעֶבְרָא מִכָּא וְשִׁנָא דְכֵיפָא מֵעֶבְרָא מִכָּא וְשׁוּם חֲדָא מַשְׁרוּעִיתָא וְשׁוּם חֲדָא מַדְרוֹכִיתָא:
שִׁנָא חֲדָא מִסְתַּכְיָא מִצִפוּנָא לְקָבֵל מִכְמָשׁ וַחֲדָא מִדָרוֹמָא לְקָבֵיל גִבְעֲתָא:
וַאֲמַר יְהוֹנָתָן לְעוּלֵימָא נָטֵל זְיָנֵיהּ אִיתָא וְנֶעְבַּר לְאִסְטַרְטִיג עֲרֵלַיָא הָאִלֵין מָאִים יַעְבֵּיד יְיָ לָנָא נִסָא אֲרֵי לֵית קֳדָם יְיָ מַעְצוֹר לְמִפְרַק בְּסַגִיאֵי אוֹ בִזְעֵירֵי:
וַאֲמַר לֵיהּ נָטֵל זְיָנֵיהּ עֲבֵיד כָּל דִי בְּלִבָּךְ אִתְפְּנֵי לָךְ הָא אֲנָא עִמָךְ כִּרְעוּתָךְ:
וַאֲמַר יְהוֹנָתָן הָא אֲנַחְנָא עָבְרִין לְוַת גַבְרַיָא וְנִתְגְלֵי לְהוֹן:
אִם כְּדֵין יֵימְרוּן לָנָא אוֹרִיכוּ עַד דְנִתְמְטֵי לְוַתְכוֹן וּנְקוּם בְּאַתְרָנָא וְלָא נִיסַק לְוַתְהוֹן:
וְאִם כְּדֵין יֵימְרוּן סָקוּ עֲלָנָא וְנִיסַק אֲרֵי מַסְרִינוּן יְיָ בִּידָנָא וְדֵין לָנָא אָתָא:
וְאִתְגְלִיאוּ תַרְוֵיהוֹן לְאִסְטַרְטְגֵי פְלִשְׁתָּאֵי וַאֲמָרוּ פְלִשְׁתָּאֵי הָא יְהוּדָאֵי נַפְקִין מִן חוֹרָאֵי דְאִטַמָרוּ תַמָן:
וַאֲתִיבוּ אֱנָשֵׁי מַטַרְתָּא יַת יוֹנָתָן וְיַת נְטַל זְיָנֵיהּ וַאֲמָרוּ סָקוּ עֲלָנָא וּנְהוֹדַע יַתְכוֹן פִּתְגָמָא וַאֲמַר יוֹנָתָן לִנְטַל זְיָנֵיהּ סַק בַּתְרַי אֲרֵי מַסְרִינוּן יְיָ בִּידָא דְיִשְׂרָאֵל:
וּסְלִיק יוֹנָתָן עַל יְדוֹהִי וְעַל רַגְלוֹהִי וּנָטַל זְיָנֵיהּ בַּתְרוֹהִי וּנְפָלוּ מְטַעֲנִין קֳדָם יוֹנָתָן וְנָטַל זְיָנֵיהּ מֵמִית בַּתְרוֹהִי:
וַהֲוַת מְחָתָא קַדְמֵיתָא דִמְחָא יוֹנָתָן וְנָטַל זְיָנֵיהּ כְּעַסְרִין גַבְרָא כְּבֵית פַּלְגוּת מַהֲלַךְ פַּדַן תּוֹרַיָא בְּחַקְלָא:
וַהֲוַת זִיעָא בְּמַשְׁרִיתָא בְּחַקְלָא וּבְכָל עַמָא אִסְטַרְטִיגָא וּמְחַבְּלָא זָעוּ אַף אִינוּן וְזָעַת אַרְעָא וַהֲוַת לְזִיעַ מִן קֳדָם יְיָ:
וַחֲזוֹ סִכְוָאַיָא לְשָׁאוּל בְּגִבְעֲתָא דְבֵית בִּנְיָמִין וְהָא הֲמוֹן מַשִׁרְיַת פְּלִשְׁתָּאֵי אִתְּבַר אָזֵיל תַּבְרֵיהּ וְסַגִי:
וַאֲמַר שָׁאוּל לְעַמָא דְעִמֵיהּ סַעֲרוּ כְעַן וַחֲזוֹ מַן שְׁגָא מִנָנָא וּמְנוֹ וְהָא לֵית יוֹנָתָן וְנָטַל זְיָנֵיהּ:
וַאֲמַר שָׁאוּל לַאֲחִיָהּ קָרֵיב אֲרוֹנָא דַייָ אֲרֵי הֲוָה אֲרוֹנָא דַייָ בְּיוֹמָא הַהוּא עִם בְּנֵי יִשְׂרָאֵל:
וַהֲוָה עַד דְמַלֵיל שָׁאוּל עִם כַּהֲנָא וְהָמוֹנָא דִבְמַשִׁרְיַת פְּלִשְׁתָּאֵי אָזֵיל מֵיזַל וְסָגֵי וַאֲמַר שָׁאוּל לְכַהֲנָא קָרֵיב אֵיפוֹדָא:
וְאִתְכְּנִישׁ שָׁאוּל וְכָל עַמָא דְעִמֵיהּ וְאָתוֹ עַד קְרָבָא וְהָא הֲוַת חַרְבָּא גְבַר בְּחַבְרֵיהּ שִׁגוּשָׁא רַב לַחֲדָא:
וִיהוּדָאֵי הֲווֹ לִפְלִשְׁתָּאֵי כְּמֵאִתְמָלֵי וּמִדְקָדָמוֹהִי דִסְלִיקוּ עִמְהוֹן בְּמַשְׁרִיתָא סְחוֹר סְחוֹר וְאַף אִינוּן תָּבוּ לְמֶהֱוֵי עִם יִשְׂרָאֵל דְעִם שָׁאוּל וְיוֹנָתָן:
וְכָל אֱנַשׁ יִשְׂרָאֵל דְאִטְמָרוּ בְּטוּרָא דְבֵית אֶפְרַיִם שְׁמָעוּ אֲרִי אַפִיכוּ פְלִשְׁתָּאֵי וְאַדְבִּיקוּ אַף אִינוּן בַּתְרֵיהוֹן בִּקְרָבָא:
וּפְרַק יְיָ בְּיוֹמָא הַהוּא יַת יִשְׂרָאֵל וְעָבְדֵי קְרָבָא מְטוֹ עַד בֵּית אָוֶן:
וֶאֱנָשׁ יִשְׂרָאֵל אִידְחִיק בְּיוֹמָא הַהוּא וְאוֹמֵי שָׁאוּל יַת עַמָא לְמֵימַר לִיט גַבְרָא דְיֵכוּל לַחְמָא עַד רַמְשָׁא עַד דְאִתְפְּרַע מִבַּעֲלֵי דְבָבַי וְלָא טְעַם כָּל עַמָא לַחְמָא:
וְכָל דַיָרֵי אַרְעָא עָלוּ בְחוֹרְשָׁא וַהֲוָה דוּבְשָׁא עַל אַפֵּי חַקְלָא:
וְאָתָא עַמָא לְחוֹרְשָׁא וְהָא בְּרֵיז דוּבְשָׁא וְלֵית דְמֵיתִיב יְדֵיהּ לְפוּמֵיהּ אֲרֵי דְחֵיל עַמָא מִשְׁבוּעֲתָא:
וְיוֹנָתָן לָא שְׁמַע כַּד אוֹמֵי אֲבוּהִי יַת עַמָא וְאוֹשִׁיט יַת רֵישׁ שׁוֹטִיתָא דְבִידֵיהּ וּטְבָל יָתָהּ בְּקִינָא דְדוּבְשָׁא וַאֲתֵיב יְדֵיהּ לְפוּמֵיהּ וְנַהֲרָא עֵינוֹהִי:
וַאֲתֵיב גַבְרָא חַד מֵעַמָא וַאֲמַר אוֹמָאָה אוֹמֵי אֲבוּךְ יַת עַמָא לְמֵימַר לִיט גַבְרָא דְיֵכוּל לַחְמָא יוֹמָא דֵין וְאִשְׁתַּלְהֵי עַמָא:
וַאֲמַר יוֹנָתָן עֲכַר אַבָּא יַת עַמָא דְאַרְעָא חֲזוֹ כְעַן אֲרֵי נַהֲרָא עֵינֵי אֲרֵי טְעֵימִית זְעֵיר דוּבְשָׁא הָדֵין:
בְּרַם אִלוּ מֵיכַל אָכִיל יוֹמָא דֵין עַמָא מִבִּזַת בַּעֲלֵי דְבָבוֹהִי דְאִשְׁכַּח אֲרֵי כְעַן לָא סַגִיאַת מְחָתָא בִּפְלִשְׁתָּאֵי:
וּקְטַלוּ בְּיוֹמָא הַהוּא בִּפְלִשְׁתָּאֵי מִמִכְמָשׁ עַד מֵישַׁר אֲיָלוֹן וְאִשְׁתַּלְהֵי עַמָא לַחֲדָא:
וְאִתְפְּנֵי עַמָא עַל בִּיזְתָא וּנְסִיבוּ עָאן וְתוֹרִין וּבְנֵי תוֹרִין וּנְכִיסוּ עַל אַרְעָא וַאֲכַל עַמָא עַל דְמָא:
וְחַוִיאוּ לְשָׁאוּל לְמֵימַר הָא עַמָא חַיְבִין קֳדָם יְיָ לְמֵיכַל עַל דְמָא וַאֲמַר שְׁקַרְתּוּן קְרִיבוּ לְוָתִי יוֹמָא דֵין אַבְנָא רַבְּתָא:
וַאֲמַר שָׁאוּל אִתְבַּדָרוּ בְעַמָא וְתֵימְרוּן לְהוֹן קְרִיבוּ לְוָתִי גְבַר תּוֹרֵיהּ וּגְבַר אִימְרֵיהּ וְתִיכְסוּן הָכָא וְתֵיכְלוּן וְלָא תְחוֹבוּן קֳדָם יְיָ לְמֵיכַל עַל דְמָא וְקָרִיבוּ כָל עַמָא גְבַר תּוֹרֵיהּ בִּידֵיהּ בְּלֵילְיָא וּנְכִיסוּ תַמָן:
וּבְנָא שָׁאוּל מַדְבְּחָא קֲדָם יְיָ יָתֵיהּ שָׁרֵי לְמִבְנֵי מַדְבְּחָא קֳדָם יְיָ:
וַאֲמַר שָׁאוּל נֵחוֹת בָּתַר פְּלִשְׁתָּאֵי בְּלֵילְיָא וְנִקְטוֹל בְּהוֹן עַד מַנְהַר צַפְרָא וְלָא נִשְׁאַר בְּהוֹן אֱנַשׁ וַאֲמַרוּ כָּל דְתַקִין בְּעֵינָךְ עֲבִיד וַאֲמַר כַּהֲנָא נִתְקְרַב הַלְכָא וְנִשְׁאַל בְּמֵימְרָא דַיָי:
וּשְׁאַל שָׁאוּל בְּמֵימְרָא דַייָ הַאֵיחוֹת בָּתַר פְּלִשְׁתָּאֵי הֲתִמְסְרִינוּן בְּיַד יִשְׂרָאֵל וְלָא קַבֵּיל צְלוֹתֵיהּ בְּיוֹמָא הַהוּא:
וַאֲמַר שָׁאוּל אִתְקְרִיבוּ הַלְכָא כָּל רֵישֵׁי עַמָא וּדְעוֹ וַחֲזוֹ בְּמָן הֲוַת חוֹבָא הָדֵין יוֹמָא דֵין:
אֲרֵי קַיָם הוּא יְיָ דְפָרִיק יַת יִשְׂרָאֵל אֲרֵי אִלּוּ אִיתוֹהִי בְּיוֹנָתָן בְּרִי אֲרֵי אִתְקְטָלָא יִתְקְטֵל וְלֵית דְמֵתִיב לֵיהּ מִכָּל עַמָא:
וַאֲמַר לְכָל יִשְׂרָאֵל אַתּוּן תְּהוֹן לְעֶבְרָא חָד וַאֲנָא וְיוֹנָתָן בְּרִי נְהֵי לְעֶבְרָא חָד וַאֲמַרוּ עַמָא לְשָׁאוּל דְתַקִין בְּעֵינָךְ עֲבֵיד:
וַאֲמַר שָׁאוּל קֳדָם יְיָ אֱלָהָא דְיִשְׂרָאֵל אַיְתִנֵיהּ בִּקְשׁוֹט וְאִיתְאָחַד יוֹנָתָן וְשָׁאוּל וְעַמָא נְפָקוּ:
וַאֲמַר שָׁאוּל רְמוֹ עַדְבִין בֵּינָא וּבֵין יוֹנָתָן בְּרִי וְאִיתְאָחַד יוֹנָתָן:
וַאֲמַר שָׁאוּל לְיוֹנָתָן חַוֵי לִי מֶה עֲבַדְתָּא וְחַוֵי לֵיהּ יוֹנָתָן וַאֲמַר מַטְעַם טְעֵימַת בְּרֵישׁ שׁוֹטִיתָא דְבִידִי זְעֵיר דוּבְשָׁא הָא אֲנָא חַיָב לְמִמָת:
וַאֲמַר שָׁאוּל כְּדֵין יַעֲבֵיד לִי יְיָ וּכְדֵין יוֹסִיף אֲרֵי מְמַת תְּמוּת יוֹנָתָן:
וַאֲמַר עַמָא לְשָׁאוּל הַיוֹנָתָן יִתְקְטֵיל דִי עֲבַד פּוּרְקָנָא רַבָּא הָדֵין בְּיִשְׂרָאֵל חָס קַיָם הוּא יְיָ אִם יִפּוֹל מִסְעַר רֵישֵׁיהּ לְאַרְעָא אֲרֵי קֳדָם יְיָ גְלֵי דִבְשָׁלוּ עֲבַד יוֹמָא דֵין וּפְרָקוּ עַמָא יַת יוֹנָתָן וְלָא מִית:
וּסְלֵיק שָׁאוּל מִבָּתַר פְּלִשְׁתָּאֵי וּפְלִשְׁתָּאֵי אֲזָלוּ לְאַתְרֵיהוֹן:
וְשָׁאוּל אַצְלַח בְּמַלְכוּתָא עַל יִשְׂרָאֵל וְאַגִיחַ קְרָבָא סְחוֹר סְחוֹר בְּכָל בַּעֲלֵי דְבָבוֹהִי בְּמוֹאָב וּבִבְנֵי עַמוֹן וּבֶאֱדוֹם וּבְמַלְכֵי צוֹבָה וּבִפְלִשְׁתָּאֵי וּבְכָל אֲתַר דְמִתְפְּנֵי מְחַיָב:
וּכְנַשׁ מַשִׁרְיָן וּמְחָא יַת דְבֵית עֲמָלֵק וְשֵׁיזֵיב יַת יִשְׂרָאֵל מִיַד בָּזְזֵיהוֹן:
וַהֲווֹ בְּנֵי שָׁאוּל יוֹנָתָן וְיִשְׁוִי וּמַלְכִּישׁוּעַ וְשׁוּם תַּרְתֵּין בְּנָתֵיהּ שׁוּם רַבְּתָא מֵרַב וְשׁוּם זְעֶרְתָּא מִיכָל:
וְשׁוּם אִתַּת שָׁאוּל אֲחִינוֹעַם בַּת אֲחִימָעַץ וְשׁוּם רַב חֵילֵיהּ אֲבִינֵר בַּר נֵר אָח אֲבוּהִי דְשָׁאוּל:
וְקִישׁ אֲבוּהִי דְשָׁאוּל וְנֵר אֲבוּהִי דְאַבְנֵר בַּר אֲבִיאֵל:
וַהֲוַת קְרָבָא תַּקִיפָא עַל פְּלִישְׁתָּאֵי כָּל יוֹמֵי שָׁאוּל וַחֲזָא שָׁאוּל כָּל גְבַר גִבַּר וְכָל גְבַר עָבֵיד קְרָב וּכְנַשׁ יָתֵיהּ לְוָתֵיהּ:
