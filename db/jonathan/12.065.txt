אִשְׁתְּאֵילִית בְּמֵימְרִי לִדְלָא שְׁאִילוּ מִן קֳדָמַי תְּבָעֵית אוּלְפַן אוֹרַיְתִי לִדְלָא תְּבָעוּ דְחַלְתִּי אֲמָרֵית הָא אֲנָא מִשְׁתָּאֵיל תְּדִירָא כָּל יוֹמָא לְעַם דְלָא מְצַלֵי בִשְׁמִי:
שְׁלָחִית נְבִיאֵי כָּל יוֹמָא לְעַמָא מְסָרְבָנָא דְאָזְלִין בְּאוֹרַח דְלָא תַקְנָא בָּתַר עֶשְׁתּוֹנֵיהוֹן:
עַמָא דְמַרְגְזִין עַל מֵימְרִי קֳדָמַי תְּדִירָא דָבְחִין בְּגִנָן טַעֲוָתָא וּמַסְקִין בּוּסְמַיָא עַל לִבְנַיָא:
דְיָתְבִין בְּבָתַּיָא דְבָנָן מֵעֲפַר קִבְרַיָא וְעִם פִּגְרֵי בְּנֵי אֱנָשָׁא דַיְרִין דְאָכְלִין בְּשַׂר חֲזִירָא וּרְטַב פִּגוּל בְּמַנֵיהוֹן:
דְאָמְרִין רָחֵיק לְהָלָא לָא תִקְרַב בִּי אֲרֵי דָכֵינָא מִנָךְ אִלֵין רוּגְזֵיהוֹן כִּתְנָנָא קֳדָמַי פּוּרְעֲנוּתְהוֹן בְּגֵיהִנָם דְאֶשְׁתָּא דְלָקָא בָהּ כָּל יוֹמָא:
הָא כְתִיבָא קֳדָמַי לָא אֶתֵּן לְהוֹן אַרְכָּא בְּחַיַיָא אֱלָהֵן אֲשַׁלֵם לְהוֹן פּוּרְעֲנוּת חוֹבֵיהוֹן וְאֶמְסוֹר לְמוֹתָא תִּנְיָנָא יַת גְוִיַתְהוֹן:
חוֹבֵיכוֹן וְחוֹבֵי אֲבָהַתְכוֹן גְלָן קֳדָמַי כַּחֲדָא אֲמַר יְיָ דַאֲסִיקוּ בוּסְמִין עַל טוּרַיָא וְעַל רְמָתָא חֲסִידוּ קֳדָמַי וְאֶתֵּן אֲגַר עוֹבָדֵיהוֹן בְּקַדְמֵיתָא עַל חֵיקְהוֹן:
כִּדְנַן אֲמַר יְיָ כְּמָא דְאִשְׁתְּכַח נֹחַ זַכַּאי בְּדָרָא דְטוֹפָנָא וַאֲמָרֵית דְלָא לְחַבְּלוּתֵהּ בְּדִיל לְקַיָמָא עַלְמָא מִנֵיהּ כֵּן אַעְבֵּיד בְּדִיל עַבְדֵי צַדִיקַיָא בְּדִיל דְלָא לְחַבָּלָא כֻּלָא:
וְאַפֵיק מִיַעֲקֹב זַרְעָא וּמִיהוּדָה יָרֵית טוּרֵי וְיַחְסְנוּנָהּ בְּחִירִי וְעַבְדֵי צַדִיקַיָא יֵשְׁרוּן תַּמָן:
וִיהֵי שָׁרוֹנָא לְבֵית מֵישְׁרֵי עֶדְרִין דְעָן וּמֵישַׁר עָכוּר לְבֵיתָא רַבָּעָה בַּקְרָן דְתוֹרִין לְעַמִי דִתְבָעוּ דְחַלְתִּי:
וְאַתּוּן בֵּית יִשְׂרָאֵל שְׁבַקְתּוּן פּוּלְחָנָא דַייָ אִתְנְשֵׁיתוּן יַת פּוּלְחַן טוּרָא דְקוּדְשִׁי דִמְסַדְרִין לְטַעֲוָן פְּתוֹרִין וּמְמַזְגִין לִדְחַלְתְּהוֹן אֲגָנִין:
וְאֶמְסַר יַתְכוֹן לְחַרְבָּא וְכוּלְכוֹן לְקַטָלָא תִתְמַסְרוּן חֲלַף דִשְׁלָחֵית נְבִיֵי וְלָא תַבְתּוּן אִתְנַבִּיאוּ וְלָא קַבֵּילְתּוּן וַעֲבַדְתּוּן דְבִישׁ קֳדָמַי וּבִדְלָא צְבִינָא אִתְרְעֵיתוּן:
בְכֵן כִּדְנַן אֲמַר יְיָ אֱלֹהִים הָא עַבְדֵי צַדִיקַיָא יֵכְלוּן וְאַתּוּן רַשִׁיעַיָא תִּכְפְּנוּן הָא עַבְדֵי צַדִיקַיָא יִשְׁתּוּן וְאַתּוּן רַשִׁיעַיָא תִּצְחוֹן הָא עַבְדֵי צַדִיקַיָא יֶחְדוֹן וְאַתּוּן רַשִׁיעַיָא תִּבְהֲתוּן:
הָא עַבְדֵי צַדִיקַיָא יְשַׁבְּחוּן מִטוּב לֵב וְאַתּוּן תִּצַוְחוּן מִכְּאֵב לֵב וּמִתְּבַר רוּחַ תְּיַלְלוּן:
וְתִשְׁבְּקוּן שׁוּמְכוֹן לְקַיָמָא לִבְחִירִי וִימִיתְכוֹן יְיָ אֱלֹהִים מוֹתָא תִנְיָנָא וּלְעַבְדוֹהִי צַדִיקַיָא יִקְרֵי שְׁמָא אוֹחֲרָנָא:
דִמְבָרֵךְ בְּאַרְעָא יְבָרֵךְ בֶּאֱלָהָא קַיָמָא וְדִמְקַיֵם בְּאַרְעָא יְקַיֵם בֶּאֱלָהָא קַיָמָא אֲרֵי יִתְנַשְׁיָן עָקָתָא קַדְמְיָתָא וַאֲרֵי מְסַתְּרָן מִן קֳדָמַי:
אֲרֵי הָא אֲנָא בְרָא שְׁמַיָא חַדְתִּין וְאַרְעָא חַדְתָּא וְלָא יְדַכְּרוּן קַדְמְיָתָא וְלָא יִתַּסְקוּן עַל לֵב:
אֱלָהֵין יֶחְדוּן וִיבוּעוּן בַּעֲלַם עָלְמַיָא דַאֲנָא בָרֵי אֲרֵי הָא אֲנָא בָרֵי יַת יְרוּשְׁלֵם בִּיעָא וְעַמָהּ חָדֵי:
וַאֲבוּעַ בִּירוּשְׁלֵם וְיֶחְדוּן בָהּ עַמִי וְלָא יִשְׁתְּמַע בָּהּ עוֹד קַל דְבָכָן וְקַל דִמְצַוְחִין:
וְלָא יְהֵי מִתַּמָן עוֹד יָנִיק יוֹמִין וְסָבָא דְלָא יַשְׁלַם יַת יוֹמוֹהִי אֲרֵי דְחַיֵיב עוֹלֵים בַּר מְאָה שְׁנִין יְהֵי מָאֵית וּדְחָטֵי בַּר מְאָה שְׁנִין יִתְּרָךְ:
וְיִבְנוֹן בָּתִּין וְיִתְבוּן וְיִצְבוּן כַּרְמִין וְיֵכְלוּן אִבְּהוֹן:
לָא יִבְנוֹן וְאָחְרִין יָתְבוּן לָא יִצְבוּן וְאוֹחֲרָנִין יֵכְלוּן אֲרֵי כְיוֹמֵי אִילָן חַיַיָא יוֹמֵי עַמִי וְעוֹבֵד יְדֵיהוֹן יְבַלוּן בְּחִירָי:
לָא יֶחְלוּן לְרֵיקָן וְלָא יִרְבּוּן לְמוֹתָא אֲרֵי זַרְעָא דְבָרְכֵיהּ יְיָ אִינוּן וּבְנֵי בְנֵיהוֹן עִמְהוֹן:
וִיהֵי עַד לָא יִצְלוּן קֳדָמַי וַאֲנָא אֲקַבֵּל צְלוֹתְהוֹן עַד לָא יִבְעוֹן מִן קֳדָמַי אַעְבֵּיד בָּעוּתְהוֹן:
דֵיבָא וְאִמְרָא יִרְעוּן כַּחֲדָא וְאַרְיָא כְּתוֹרָא יֵיכוּל תִבְנָא וְחִוְיָא עַפְרָא מְזוֹנֵיהּ לָא יִבְאֲשׁוּן וְלָא יְחַבְּלוּן בְּכָל טוּרָא דְקוּדְשִׁי אֲמַר יְיָ:
