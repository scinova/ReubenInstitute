קַבִּילוּ לְמֵימְרִי רָדְפֵי קוּשְׁטָא דְבָעָן אוּלְפַן מִן קֳדָם יְיָ אִסְתַּכָּלוּ דִי בַּחֲצִיבָא מִטִינָרָא אִתְחָצַבְתּוּן וְכִפְסוּלָא דְמִגוּב רֵיקָן אִתְפְסִלְתּוּן:
אִסְתַּכָּלוּ בְּאַבְרָהָם אֲבוּכוֹן וּבְשָׂרָה דְאַעְדִיאַתְכוֹן אֲרֵי חַד הֲוָה אַבְרָהָם יְחִידִי בְּעַלְמָא קָרֵיבְתֵּהּ לְפוּלְחָנִי וּבֵרַכְתֵּהּ וְאַסְגֵיתֵהּ:
אֲרֵי עֲתִיד יְיָ לְנַחֲמָא צִיוֹן לְנַחֲמָא כָּל חָרְבָתָהָא וִישַׁוֵי מַדְבְּרָהָא כְּעֵדֶן וּמֵישְׁרָאָה כְּגִינְתָא דַייָ בִּיעַ וְחֶדְוָא יִשְׁתְּכַח בָּהּ מַעֲלֵי תוֹדָתָא וְקַל דִמְשַׁבְּחִין:
קַבִּילוּ לְמֵימְרִי עַמִי וּכְנִשְׁתִּי לְפוּלְחָנִי אֲצִיתָא אֲרֵי אוֹרַיְתָא מִן קֳדָמַי תִּיפּוֹק וְדִינַי כְּנֵיהוֹר לֵהּ עַמְמִין אֲגַלֵי יִשְׁתַּבְּחוּן:
קְרִיבָא זְכוּתִי נְפַק פּוּרְקָנִי וּבִתְקוֹף דְרַע גְבוּרְתִּי עַמְמַיָא יִתְדְנוּן לְמֵימְרִי נַגְוָן יְסַבְּרוּן וְלִיתְקוֹף דְרַע גְבוּרְתִּי יִכְתְּרוּן:
זְקִיפוּ לִשְׁמַיָא עֵינֵיכוֹן וְאִסְתַּכָּלוּ בְאַרְעָא מִלְרַע אֲרֵי שְׁמַיָא כִּתְנָנָא דְעָדִי כֵן יֶעְדוּ וְאַרְעָא כִּכְסוּתָא דְבָלְיָא כֵּן תִּבְלֵי וְיָתְבָהָא אַף אִינוּן הָכֵן יְמוּתוּן וּפוּרְקָנִי לַעֲלַם יְהֵי וְזָכוּתִי לָא תִתְעַכָּב:
קַבִּילוּ לְמֵימְרִי יָדְעֵי קוּשְׁטָא עַמָא דְאוּלְפַן אוֹרַיְתִי בְּלִבְּהוֹן לָא תִדְחֲלוּן מֵחִיסוּדֵי בְנֵי אֱנָשָׁא וּמֵאִתְרַבְרְבוּתְהוֹן לָא תִתַבְרוּן:
אֲרֵי בִלְבוּשָׁא דְאָכִיל לֵהּ עָשָׁא וּכְעִמְרָא דְאָחֵיד בֵהּ רוּקְבָא וְזָכוּתִי לַעֲלַם תְהֵי וּפוּרְקָנִי לְדָר דָרִין:
אִתְגְלָא לִבְשִׁי תְקוֹף גְבוּרָא מִן קֳדָם יְיָ אִתְגְלָא כְּיוֹמֵי קֳדָם דָרַיָא דְמִלְקָדָמִין הֲלָא בְדִילָךְ כְּנִשְׁתָּא דְיִשְׂרָאֵל תְּבָרֵית גִבָרַיָא שֵׁיצֵיתִי פַּרְעֹה וּמַשִׁרְיָתֵהּ דַהֲווּ תַּקִיפִין כְּתַנִינָא:
הֲלָא בְדִילָךְ כְּנִשְׁתָּא דְיִשְׂרָאֵל אַחְרְבֵית יַמָא מֵי תְהוֹם רַבָּה שַׁוֵיתִי עוּמְקֵי יַמָא אוֹרַח לְמֵיגָז מְשֵׁיזְבַיָא:
וּפְרִיקַיָא דַייָ יִתְכַּנְשׁוּן מִבֵּינֵי גַלְוַתְהוֹן וְיַעֲלוּן לְצִיוֹן בְּתוּשְׁבַּחְתָּא וְחֶדְוַת עֲלָם תְּהֵי לְהוֹן דְלָא פָסְקָה וַעֲנַן יְקַר יְהֵי מַטֵיל עַל רֵישֵׁיהוֹן בִּיעַ וְחֶדְוָה יִשְׁכְּחוּן וְיֵסוּף מִנְהוֹן מִבֵּית יִשְׂרָאֵל דָווֹנָא וְתֵינַחְתָּא:
אֲנָא אֲנָא הוּא מְנַחֶמְכוֹן מִמַן אַתּוּן דַחֲלִין מֵאֱנָשָׁא דְמָאֵית וּמִבַּר אֱנָשָׁא דִי כְעִסְבָּא חֲשִׁיב:
וְאִתְנְשֵׁיתָא פּוּלְחָנָא דַייָ דְעָבְדָךְ דִתְלָא שְׁמַיָא וְשַׁכְלֵיל אַרְעָא וּדְחַלְתָּא תְדִירָא כָּל יוֹמָא מִן קֳדָם חֲמַת מְעִיקָא כְּמָא דִמְתַקְנִין לְחַבָּלָא וּכְעַן אָן הִיא חֲמַת דִמְעִיקָא:
מוֹחִי פּוּרְעֲנוּתָא לְאִתְגְלָאָה וְלָא יְמוּתוּן צַדִיקַיָא לְחַבֵּל וְלָא יַחְסְרוּן מְזוֹנֵיהוֹן:
וַאֲנָא יְיָ אֱלָהָךְ דִנְזֵיף בְּיַמָא וְהָמָן גַלוֹהִי יְיָ צְבָאוֹת שְׁמֵהּ:
וְשַׁוֵיתִי פִּתְגָמֵי נְבוּאָתִי בְּפוּמָךְ וּבִטְלַל גְבוּרְתִּי אַגְנֵית עֲלָךְ לְקַיָמָא עַמָא דַאֲמִיר עֲלֵיהוֹן דְיִסְגוּן כְּכוֹכְבֵי שְׁמַיָא וּלְשַׁכְלָלָא כְנִשְׁתָּא דַאֲמִיר עֲלֵיהוֹן דְיִסְגוּן כְּעַפְרָא דְאַרְעָא וּלְמֵימַר לִדְדָיְרִין בְּצִיוֹן עַמִי אַתּוּן:
אִתְרַבָּא אִתְרַבָּא קוּמִי יְרוּשְׁלֵם דְקַבֵּילְתְּ מִן קֳדָם יְיָ יַת כָּסָא דְחֵימְתֵהּ יַת פַּיְלֵי כָּסָא דִלְוָטָא שְׁתֵית אַעְרֵית:
לֵית דִמְנַחֵם לָהּ מִכָּל בְּנִין דִילֵידַת וְלֵית דְמַתְקִיף בִּידָהּ מִכָּל בְּנִין דְרַבִּיאַת:
תַּרְתֵּין עָקָן אָתָאָה עֲלָךְ יְרוּשְׁלֵם לָא יְכֵילֵית לְמֵיקָם כַּד אַרְבַּע יֵיתוּן עֲלָךְ בִּיזָא וּתְבָרָא וְכַפְנָא וְחַרְבָּא לֵית דִי מְנַחֲמֵיךְ אֱלָהֵין אֲנָא:
בְּנַיִךְ יְהוֹן מִטָרְפִין רָמָן בְּרֵישׁ כָּל שׁוּקַיָא כְּמִזְרָקֵי מְצַדָן מְלַן חֵימְתָא מִן קֳדָם יְיָ מְזוֹפִיתָא מִן קֳדָם אֱלָהָיִךְ:
בְּכֵן שִׁמְעִי כְעַן דָא מְטַלְטַלְתָּא דְרַוְיָא מִן עָקָא וְלָא מִן חֲמָר:
כִּדְנַן אֲמַר רִבּוֹנִיךְ יְיָ וֵאלָהָיִךְ דַעֲתִיד לְמֶעְבַּד פּוּרְעֲנוּת דִין עַמֵהּ הָא קַבֵּלִית מִן יְדִיךְ יַת כָּסָא דִלְוָטָא יַת פַּיְלֵי כָסָא דְחֵימְתִי לָא תוֹסִיפִין לְמִשְׁתָּהּ עוֹד:
וְאֶמְסְרִינָהּ בְּיַד דַהֲווֹ מוֹנָן לִיךְ דַהֲווֹ אָמְרִין לְנַפְשֵׁיךְ אַמְאִיכִי וְנֶעְדֵי וְאַמְאֵיכְתְּ כְּאַרְעָא יְקָרִיךְ וַהֲוֵית כְּשׁוּק לְעָדָן:
