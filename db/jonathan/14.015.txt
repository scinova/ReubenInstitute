וַהֲוָה פִּתְגָם נְבוּאָה מִן קֳדָם יְיָ עִמִי לְמֵימָר:
בַּר אָדָם מַה שְׁנָא אִילַן גוּפְנָא מִכָּל אִילָנָא שַׁבֶּשְׁתָּא דַהֲוַת בְּאָעֵי חוּרְשָׁא:
הֲיִסְבוּן מִנֵיהּ אָעָא לְמֶעְבַּד עֲבִידָא אִם יִסְבוּן מִנֵיהּ סִיכָתָא לְמִתְלֵי בָּהּ כָּל מָאן:
הָא לְאֶשְׁתָּא אִתְמְסַר לְאִשְׁתֵּיצָאָה יַת תְּרֵין סִטְרוֹהִי אָכְלָא אַשִׁתָּא וְגַוֵיהּ חַר הַיִכְשַׁר לַעֲבִידָא:
הָא כַּד הֲוָה שְׁלִים לָא כְשַׁר לַעֲבִידָא אַף אֱלָהֵין דְאֶשְׁתָּא אֲכַלְתֵּיהּ וְחַד הֲיִכְשַׁר עוֹד לַעֲבִידָא:
בְּכֵן כִּדְנַן אֲמַר יְיָ אֱלֹהִים כְּמָא דְאִילָן גוּפְנָא בַּאֲעֵי חוּרְשָׁא דִמְסַרְתֵּיהּ לְאֶשְׁתָּא לְאִשְׁתֵּיצָאָהּ כֵּן מְסָרֵית יַת יַתְבֵי יְרוּשְׁלֵם:
וְאֶתֵּן יַת פּוּרְעֲנוּתִי בְּהוֹן עַל פִּתְגָמַי אוֹרַיְתָא דִיהִיבִין מִגוֹ אֶשְׁתָּא עֲבַרוּ וְעַמְמַיָא דִתְקִיפִין כְּאֶשְׁתָּא יְשֵׁיצִינוּן וְתִידְעוּן אֲרֵי אֲנָא יְיָ כַּד יָחוּל רוּגְזִי בְּהוֹן:
וְאֶתֵּן יַת אַרְעָא לְצָדוּ חֲלַף דִשְׁקָרוּ שְׁקַר אֲמַר יְיָ אֱלֹהִים:
