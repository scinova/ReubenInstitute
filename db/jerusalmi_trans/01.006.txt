וְהָיָה כַּאֲשֶׁר הֶחֱלוּ +בְּנֵי+ אֱנוֹשׁ לִרְבּוֹת עַל אֵפִי הָאָרֶץ, וּבָנוֹת +יָפוֹת+ נוֹלְדוּ לָהֶם.
וְרָאוּ בְּנֵי _מְכֻובָּדִים_ אֶת בְּנוֹת הָאֱנוֹשׁ, כִּי _יָפוֹת_ הֵן; +וּכְחֻולּוֹת וּפְקָסוֹת, וּמְהַלְּכוֹת בְּגִילּוּי בָּשָׂר, וְהַרְהִירוֹ לִזְנוּת,+ וְלָקְחוּ לָהֶם נָשִׁים, מִכָּל שֶׁפָּגְשׁוּ.
וְאָמַר ה׳ +בְּמַאֲמָרוֹ:+ "לֹא יִידּוֹנוּ(~יִהְיוּ נִדּוֹנִים) _כֹּל הַדּוֹרוֹת הָרָעִים_+, שֶׁעֲתִידִים לָקוּם בְּסֵדֶר דַּיָּינֵי דּוֹר הַמַּבּוּל, לְמוֹבְדָה** וְלַהֲכָלָה מִתּוֹךְ הָעוֹלָם. הַלֹּא נָתַתִּי +[אֶת]+ רוּחַ קָודְשִׁי בָּהֶם, מִן־בִּגְלַל+(~כְּדֵי)+ שֶׁיַּעֲשׂוּ מַעֲשִׂים טוֹבִים, וְהֵן הִרְעוּ+(~מִלָּשׁוֹן 'רַע')+ +[אֶת]+ מַעֲשֵׂיהֶם+; הִנֵּה _נָתַתִּי לָהֶם הַאֲרָכָה_ - מֵאָה וְעֶשְׂרִים שָׁנִים+, מִן־בִּגְלַל שֶׁיַּעֲשׂוּ תְּשׁוּבָה, וְלֹא עָשׂוּ".
+שַׁמְחַזָאֵי וְעֻוזִּיאֵל הֵם+ נוֹפְלִים +מִן הַשָּׁמַיִם וְ+הָיוּ בָאָרֶץ בַּיָּמִים הָהֵם, וְאַף אַחֵר כֵּן, שֶׁיִּכָּנְסוּ בְּנֵי _גְּדוֹלִים־גְּדוֹלִים_ אֶל בְּנוֹת הָאֱנוֹשׁ, וְיָלְדוּ לָהֶם; הֵם +נִקְרָאִים+ גְּבָרִים, שֶׁמֵּעוֹלָם אַנְשֵׁי הַשֵּׁמוֹת.
וְרָאָה ה׳, כִּי רַבָּה רָעַת הָאָדָם בָּאָרֶץ; וְכָל יֵצֶר מַחְשֶׁבֶת לִיבּוֹ - רַק רַע, כֹּל הַיּוֹם.
וְ_שָׁב_ ה׳ +בְּמַאֲמָרוֹ+, כִּי עָשָׂה אֶת הָאָדָם בָּאָרֶץ; _וְהִתְדַּיֵּין עֲלֵיהֶם בְּמַאֲמָרוֹ_.
וְאָמַר ה׳: "_אֲבַטֵּל_ הָאָדָם +בְּמַאֲמָרִי+ שֶׁבָּרָאתִי מֵעַל פְּנֵי הָאָרֶץ - מֵאָדָם עַד בְּהֵמָה, עַד _רוֹחֵשׁ_, וְעַד עוֹף הַשָּׁמָיִם; כִּי _שַׁבְתִּי בְּמַאֲמָרִי_ כִּי עֲשִׂיתִיכֶם".
וְנֹחַ+, שֶׁהָיָה צַדִּיק+ - מָצָא חֵן לִפְנֵי ה׳.
אֵלֶּה [ה]_יוּחֲסִין_ +שֶׁל־מִשְׁפַּחַת+ נֹחַ: נֹחַ +הָיָה+ אִישׁ צַדִּיק, _שָׁלֵם_ +בְּמַעֲשִׂים טוֹבִים+ הָיָה בְּדוֹרוֹתָיו; _בְּיִרְאַת ה׳_ הָלַךְ נֹחַ.
וְהוֹלִיד נֹחַ שְׁלוֹשָׁה בָּנִים: אֶת שֵׁם, אֶת חָם, וְאֶת יֶפֶת.
_וְנֶחְבְּלָה_ הָאָרֶץ +בִּגְלַל דַּיָּירֶיהָ, שֶׁסָּטוּ מִן +[ה]+דְּרָכִים שֶׁתקינות+ לִפְנֵי ה׳, וְנִתְמַלְּאָה הָאָרֶץ _חוֹטְפִים_.
וְרָאָה ה׳ אֶת הָאָרֶץ, וְהֵנָּה _נֶחְבְּלָה_, כִּי _חִיבְּלוֹ_ כָּל בָּשָׂר+, כָּל אֶחָד וְאֶחָד,+ אֶת דַּרְכּוֹ עַל הָאָרֶץ.
וְאָמַר ה׳ לְנֹחַ: _סוֹף_ כָּל בָּשָׂר _הִגִּיעַ_ מִן פָּנָי, כִּי נִתְמַלְּאָה הָאָרֶץ _חוֹטְפִים_ מִן +מַעֲשֵׂיהֶם הָרָעִים+, וְהֵנָּה אֲנִי _מֵחַבְלָם_ _עִם_ הָאָרֶץ.
עֲשֵׂה לָךְ תֵּיבָה שֶׁל עֲצֵי־קַדְרוֹנִים(~סוּג עֲצֵי אֶרֶז), +מֵאָה וַחֲמִישִּׁים+ _חֲדָרִים_ תַּעֲשֶׂה לְתֵיבָה +בִּשְׂמֹאלָהּ, וּשְׁלוֹשִׁים וְשֵׁשׁ בְּרָוחְבָּהּ, וַעֲשָׂרָה בָּתִּים בָּאֶמְצַע, לְהַצְנִיעַ בָּהֶם+ [אֶת] +הַמָּזוֹן, וְחָמֵשׁ ?פְּתָחִים? בְּיָמִין, וְחָמֵשׁ בִּשְׂמֹאלָהּ+, וּ_תְטַיֵּיחַ_ אוֹתָהּ _מִן תּוֹךְ_ וּמִחוּץ בְּ_חֵימָר_”.
[חָסֵר תַּרְגּוּם לְפָסוּק זֶה]
“+לֵךְ לְפִישׁוֹן, וְקַח מִשֵּׁם+ ?תְּאוּרָה?(~אֶבֶן טוֹבָה)+, וּ+_תְשִׂימָהּ_(=תָּשִׂים אוֹתָהּ) בְּתֵיבָה +לְהָאִיר לָכֶם+, וְלָאַמָּה תִּכְלֶהָ מִלְּמַעְלָה, וְשַׁעַר הַתֵּיבָה בְּצִידָּהּ תָּשִׂים; +מְגוּרִים+ תַּחְתּוֹנִים, שְׁנִייִּם וּשְׁלִישִׁים תַּעֲשֶׂהָ”.
וַאֲנִי הִנְנִי מֵבִיא אֶת הַמַּבּוּל מַיִם עַל הָאָרֶץ, _לְכַלּוֹת_ כָּל בָּשָׂר שֶׁבּוֹ רוּחַ חַיִּים, מִן תַּחַת הַשָּׁמַיִים; כֹּל שֶׁבָּאָרֶץ _יִתְנַגֵּד_ (יִגְוָע).
וְאָקִים אֶת בְּרִיתִי עִמְּךָ, וְ_תִכָּנֵס_ לַתֵּיבָה אַתָּה, וּבִנְךָ, וְאִשְׁתְּךָ, וּנְשׁוֹת בִּנְךָ עִמְּךָ”.
וּמִן כֹּל שֶׁחַי, מִכֹּל בָּשָׂר - שְׁנַיִים מֵהַכֹּל _תַּכְנִיס_ לַתֵּיבָה, לַעֲמוֹד עִמְּךָ, זָכָר וּנְקֵבָה יִהְיוּ.
מֵהָעוֹף לְמִינֵהוּ, וּמֵהַבְּהֵמָה לְמִינָהּ, וּמִכָּל רָחַשׁ הָאָרֶץ לְמִינֵהוּ - שְׁנַיִים מֵהַכֹּל _יִכָּנְסוּ_ אֵלֶיךָ +עַל יַד מַלְאָךְ הָאֶחָד, וּמַכְנִיס אוֹתָם+ לָךְ לְקַיֵּים”.
“וְאַתָּה - קַח לְךָ מִכֹּל מַאֲכָל שֶׁנֶּאֱכַל, וְיִהְיֶה לְךָ וְלַהֶם לְמַאֲכָל”.
וְעָשָׂה נֹחַ כְּכֹל שֶׁצִּיוָּוהוּ ה׳.
