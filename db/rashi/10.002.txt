



אֶת אֲשֶׁר עָשָׂה לִי. שֶׁהֶרְאָה הַסֵּפֶר שֶׁשָּׁלַח דָּוִד בְּיַד אוּרִיָּה. (במדרש רבי תנחומא שם) דְּמֵי מִלְחָמָה בְּשָׁלֹם. שֶׁהָיָה לָהֶם שָׁלוֹם עִמּוֹ, וְלֹא הָיוּ נִשְׁמָרִים מִמֶּנּוּ. בַּחֲגֹרָתוֹ אֲשֶׁר בְּמָתְנָיו. שֶׁחָגַר חַרְבּוֹ מְצֻמֶּדֶת עַל מָתְנָיו, שֶׁלֹּא כְּדֶרֶךְ הַחוֹגְרִים, בְּצֵאתוֹ לִקְרַאת עֲמָשָׂא, וּפִיהָ לְמַטָּה כְּדֵי שֶׁתִּפֹּל, כְּמָה שֶׁנֶּאֱמַר: וְהוּא יָצָא וַתִּפֹּל, וּכְשֶׁנָּפְלָה וּנְטָלָהּ, כְּסָבוּר עֲמָשָׂא שֶׁלֹּא נְטָלָהּ אֶלָּא לְהַגְבִּיהָהּ מִן הָאָרֶץ, וְלֹא נִשְׁמַר מֵהַחֶרֶב אֲשֶׁר בְּיַד יוֹאָב. וּבְנַעֲלוֹ אֲשֶׁר בְּרַגְלָיו. אֶת אַבְנֵר הָרַג בְּעָרְמָה, שֶׁשְּׁאָלוֹ גִּדֶּמֶת הֵיאָךְ חוֹלֶצֶת, כְּמָה שֶׁנֶּאֱמַר: וַיַּטֵּהוּ יוֹאָב אֶל תּוֹךְ הַשַּׁעַר לְדַבֵּר אִתּוֹ 'בַשֶּׁלִי', לְשׁוֹן 'שַׁל' נְעָלֶיךָ.
וְלֹא תוֹרֵד שֵׂיבָתוֹ בְּשָׁלֹם שְׁאֹל. אַל תַּנִּיחֵהוּ לָמוּת מִיתַת עַצְמוֹ וְלִפֹּל בְּגֵיהִנֹּם.

קְלָלָה נִמְרֶצֶת. מְפֹרֶשֶׁת, כְּמוֹ: מַה יַּמְרִיצְךָ; (שם ו כה): וּמַה נִּמְרְצוּ, כְּמוֹ מַה נִּמְלְצוּ.
כִּי אִישׁ חָכָם אָתָּה. תֵּן לִבְּךָ לִמְצֹא לוֹ עֲוֹן מָוֶת בִּשְׁבִיל דָּבָר אַחֵר, וְזֶהוּ: וְיָדַעְתָּ אֵת אֲשֶׁר תַּעֲשֶׂה לּוֹ.


וַתִּכֹּן מַלְכֻתוֹ מְאֹד. אַף עַל הָעֶלְיוֹנִים.






לְדַבֶּר לוֹ. לְדַבֵּר בִּשְׁבִילוֹ, בִּשְׁבִיל אֲדוֹנִיָּהוּ. לְאֵם הַמֶּלֶךְ. אָמְרוּ רַבּוֹתֵינוּ (בבא בתרא צא ב): לְאִמָּהּ שֶׁל מַלְכוּת, הִיא רוּת.


וְשַׁאֲלִי לוֹ אֶת הַמְּלוּכָה. מִשֶּׁיִּשְׁתַּמֵּשׁ בְּשַׁרְבִיטוֹ שֶׁל מֶלֶךְ, הוּא תְּחִלַּת הַשְּׂרָרָה.

עָשָׂה לִי בַּיִת. בֵּית הַמְּלוּכָה, כַּאֲשֶׁר דִּבֵּר בְּיַד נָתָן.

עֲנָתֹת לֵךְ. מִשָּׁם הָיָה. כִּי אִישׁ מָוֶת אָתָּה. שֶׁמָּרַדְתָּ בְּמַלְכוּת אָבִי בִּהְיוֹתְךָ עִם אֲדוֹנִיָּהוּ, לְהַמְלִיכוֹ שֶׁלֹּא מִדַּעְתּוֹ. כִּי נָשָׂאתָ אֶת אֲרוֹן. בִּבְרֹחַ אָבִי מִפְּנֵי אַבְשָׁלוֹם. וְכִי הִתְעַנִּיתָ. עִמּוֹ בְּאוֹתָהּ הַצָּרָה, שֶׁנֶּאֱמַר: כִּי אָמְרוּ הָעָם רָעֵב וְעָיֵף וְצָמֵא בַּמִּדְבָּר.
אֶת אֲשֶׁר דִּבֶּר עַל בֵּית עֵלִי. וַהֲקִימֹתִי לִי כֹּהֵן נֶאֱמָן וְגוֹ', וְהִבַּטְתָּ צַר מָעוֹן וְגוֹ', וּמִשֶּׁחָרְבָה שִׁילֹה, לֹא הָיָה בַּיִת חָשׁוּב לַעֲבוֹדָה לְהַקָּבָּ"ה, וְעַתָּה כְּשֶׁיִּבָּנֶה בֵּית הַבְּחִירָה, מִלֵּא הַקָּבָּ"ה אֶת דְּבָרוֹ לְגָרֵשׁ אֶת בְּנֵי עֵלִי.
וְהַשְּׁמֻעָה בָּאָה עַד יוֹאָב. שְׁמוּעַת מִצְוַת דָּוִד שֶׁצִּוָּה לְהָרְגוֹ, וְיָדַע שֶׁשְּׁלֹמֹה שׂוֹנְאוֹ. כִּי יוֹאָב נָטָה אַחֲרֵי אֲדֹנִיָּה וְאַחֲרֵי אַבְשָׁלוֹם לֹא נָטָה. סוֹף דָּבָר, אֲבָל בִּקֵשׁ לִנְטוֹת, אֶלָּא נִתְיָרֵא שֶׁעֲדַיִן לַחְלוּחִית שֶׁל דָּוִד קַיֶּמֶת.

וַיֹּאמֶר לֹא. שְׁתַּיִם לֹא אֲקַבֵּל עָלַי, קְלָלוֹת שֶׁקִּלְלַנִי דָּוִד אָבִיךָ בְּמִיתַת אַבְנֵר: יָחוּלוּ עַל רֹאשׁ יוֹאָב, יְקַבְּלֵם שְׁלֹמֹה עָלָיו, וְאַחַר כָּךְ אֵצֵא.
עֲשֵׂה כַּאֲשֶׁר דִּבֶּר. אֲנִי מְקַבְּלָם עָלַי, לְכָךְ נִדְבְּקוּ כָּל אֵלֶּה בְּזֶרַע דָּוִד, עֻזִּיָּה הָיָה מְצֹרָע, אָסָא מַחֲזִיק בַּפֶּלֶךְ, יְהוֹיָקִים נָפַל בַּחֶרֶב, וַחֲסַר לֶחֶם צִדְקִיָּהוּ.


בְּבֵיתוֹ בַּמִּדְבָּר. שֶׁהָיָה בֵּיתוֹ הֶפְקֵר כַּמִּדְבָּר, לַעֲנִיִּים.

וְלֹא תֵצֵא מִשָּׁם. כְּדֵי שֶׁיְּהֵא מָצוּי לוֹ, וְלִמְצֹא לוֹ חֵטְא מָוֶת.





וָאָעִד בְּךָ. לְשׁוֹן הַתְרָאָה.
אֵת שְׁבֻעַת ה'. אֲשֶׁר הִשְׁבַּעְתִּיךָ בַּה'.


וְהַמַּמְלָכָה נָכוֹנָה וְגוֹ'. כְּלוֹמַר, לֹא נֶעֱנַשׁ עַל כָּל אֵלֶּה.
