


מֵימֵינוּ בְּכֶסֶף שָׁתִינוּ. שֶׁהָיִינוּ יְרֵאִים לִשְׁאוֹב מַיִם מִן הַנָּהָר מִפְּנֵי הָאוֹיְבִים, וְהָיִינוּ קוֹנִים מֵהֶם בְּכֶסֶף:
עַל צַוָּארֵנוּ נִרְדָּפְנוּ. מֵעוֹל עֲבוֹדָה קָשָׁה: יָגַעְנוּ. לֶאֱסוֹף מָמוֹן וּנְכָסִים: וְלֹא הוּנַח לָנוּ. יְגִיעֵנוּ בְּיָדֵינוּ, כִּי הָאוֹיְבִים הָיוּ גוֹבִים וְחוֹטְפִים הַכֹּל בְּמִסִּים וְגֻלְגָּלִיּוֹת וְאַרְנוּנִיּוֹת:
מִצְרַיִם נָתַנּוּ יָד. דֶּרֶךְ אָדָם הַנּוֹפֵל וְרוֹצֶה לַעֲמוֹד, מוֹשִׁיט יָד לְמִי שֶׁאֶצְלוֹ לַעֲזוֹר לוֹ, וְאַף כַּאן, לְמִצְרַיִם הוֹשַׁטְנוּ יָד שֶׁיַּעְזְרוּנוּ: אַשּׁוּר. שֶׁיַּשְׂבְּעוּנוּ בְלַחְמָם: נָתַנּוּ. כְּמוֹ "נָתַנְנוּ". דַּגְשׁוּת הַנּוּ"ן מְשַׁמֶּשֶׁת בִּמְקוֹם נוּ"ן שְׁנִיָּה, וְכֵן, "כִּי מִמְּךָ הַכֹּל וּמִיָּדְךָ נָתַנּוּ לָךְ". וְכֵן, "וְנָתַנּוּ אֶת בְּנֹתֵינוּ לָכֶם":


בְּנַפְשֵׁנוּ נָבִיא לַחְמֵנוּ. בְּסַכָּנַת נַפְשֵׁנוּ. הָיִינוּ מְסֻּכָּנִים, כְּשֶׁהָיִינוּ מְבִיאִים מְזוֹנוֹתֵינוּ מִן הַשָּׂדֶה, מִפְּנֵי חֶרֶב הַמִּדְבָּר:
נִכְמָרוּ. נִתְחַמְּמוּ, וְכֵן, "כִּי נִכְמְרוּ רַחֲמָיו". וּבִלְשׁוֹן גְּמָרָא יֵשׁ הַרְבֵּה: "עַל הַכּוֹמֶר שֶׁל עֲנָבִים". "מִכְמָר בְּישְׂרָאֵל": זַלְעֲפוֹת רָעָב. כְּמוֹ "וְרוּחַ זִלְעָפוֹת", לְשׁוֹן שְׂרֵפָה:


טְחוֹן נָשָׂאוּ. כְּשֶׁהָיוּ הָאוֹיְבִים מוֹלִיכִין אוֹתָם בְּקוֹלָרִין, הָיוּ נוֹתְנִים עַל כְּתֵפֵיהֶם רֵחַיִם וּמַשָׂאוֹת כְּדֵי לְיַגְּעָם. וְכֵן, "בָּעֵץ כָּשָׁלוּ", כָּשַׁל כֹּחָם. וּלְשׁוֹן כִּשְׁלוֹן נוֹפֵל בְּתַשּׁוּת כֹּחַ, כְּמוֹ שֶׁנֶּאֱמַר בְּעֶזְרָא "וַיֹּאמֶר יְהוּדָה כָּשַׁל כֹּחַ הַסַּבָּל". וְכֵן "הִכְשִׁיל כֹּחִי":



עַל זֶה הָיָה דָוֶה לִבֵּנוּ וְגוֹ'. עַל הַמְפֹרָשׁ בַּמִּקְרָא שֶׁל אַחֲרָיו: "עַל הַר צִיּוֹן שֶׁשָּׁמֵם". וְ"שׁוּעָלִים הִלְּכוּ בוֹ":

אַתָּה ה'. יָדַעְנוּ כִּי לְעוֹלָם תֵּשֵׁב וְהוֹאִיל וְכֵן הוּא:
לָמָּה לָנֶצַח תִּשְׁכָּחֵנוּ. הֲלֹא נִשְׁבַּעְתָּ לָנוּ בְּךָ כְּשֵׁם שֶׁאַתָּה קַיָּם כַּךְ שְׁבוּעָתְךָ קַיֶּמֶת:

כִּי אִם מָאֹס מְאַסְתָּנוּ. בִּשְׁבִיל שֶׁחָטָאנוּ לֹא הָיָה לְךָ לְהַרְבּוֹת קֶצֶף עַד מְאֹד כַּאֲשֶׁר קָצַפְתָּ: הֲשִׁיבֵנוּ ה'. מִפְּנֵי שֶׁמְּסַיֵּם בְּדִבְרֵי תוֹכֵחָה הֻצְרַךְ לִכְפֹּל מִקְרָא שֶׁלְּפָנָיו פַּעַם אַחֶרֶת, וְכֵן יְשַׁעְיָה וּתְרֵי עָשָׂר וְקֹהֶלֶת:
