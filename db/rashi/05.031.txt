וילך משה. וגו':
וילך משה וגו', אנכי היום. הַיּוֹם מָלאוּ יָמַי וּשְׁנוֹתַי, בְּיוֹם זֶה נוֹלַדְתִּי וּבְיוֹם זֶה אָמוּת (סוטה י"ג): לא אוכל עוד לצאת ולבוא. יָכוֹל שֶׁתָּשַׁשׁ כֹּחוֹ, תַּלְמוּד לוֹמַר "לֹא כָהֲתָה עֵינוֹ וְלא נָס לֵחֹה", אֶלָּא מַהוּ לא אוכל? אֵינִי רַשַּׁאי, שֶׁנִּטְּלָה מִמֶּנִּי הָרְשׁוּת וְנִתְּנָה לִיהוֹשֻׁעַ. דָּ"אַ — לצאת ולבוא. בְּדִבְרֵי תוֹרָה, מְלַמֵּד שֶׁנִּסְתְּמוּ מִמֶּנּוּ מָסוֹרוֹת וּמַעְיְנוֹת הַחָכְמָה (סוטה י"ג): וה' אמר אלי. זֶהוּ פֵּרוּשׁ לא אוכל עוד לצאת ולבוא — לְפִי שֶׁה' אמר אלי :



לא ירפך. לֹא יִתֵּן לְךָ רִפְיוֹן לִהְיוֹת נֶעֱזָב מִמֶּנּוּ:
כי אתה תבוא את העם הזה. "אֲרֵי אַתְּ תֵּעוּל עִם עַמָּא הָדֵין": מֹשֶׁה אָמַר לוֹ לִיהוֹשֻׁעַ זְקֵנִים שֶׁבַּדּוֹר יִהְיוּ עִמְּךָ, הַכֹּל לְפִי דַעְתָּן וַעֲצָתָן, אֲבָל הַקָּבָּ"ה אָמַר לִיהוֹשֻׁעַ "כִּי אַתָּה תָּבִיא אֶת בְּנֵי יִשְׂרָאֵל אֶל הָאָרֶץ אֲשֶׁר נִשְׁבַּעְתִּי לָהֶם" — תָּבִיא עַל כָּרְחָם, הַכֹּל תָּלוּי בְּךָ, טֹל מַקֵּל וְהַךְ עַל קָדְקֳדָן, דַּבָּר אֶחָד לְדוֹר וְלֹא שְׁנֵי דַבָּרִים לְדוֹר (סנהדרין ח'):

ויכתב משה, ויתנה. כְּשֶׁנִּגְמְרָה כֻּלָּהּ נְתָנָהּ לִבְנֵי שִׁבְטוֹ (עי' ספרי דברים ל"ד):
מקץ שבע שנים. בְּשָׁנָה רִאשׁוֹנָה שֶׁל שְׁמִטָּה — בַּשְּׁמִינִית, וְלָמָּה קוֹרֵא אוֹתָהּ שנת השמטה? שֶׁעֲדַיִּן שְׁבִיעִית נוֹהֶגֶת בָּהּ, בַּקָּצִיר שֶׁל שְׁבִיעִית הַיּוֹצֵא לְמוֹצָאֵי שְׁבִיעִית:
תקרא את התורה הזאת. הַמֶּלֶךְ הָיָה קוֹרֵא מִתְּחִלַּת "אֵלֶּה הַדְּבָרִים", כִּדְאִיתָא בְמַסֶּכֶת סוֹטָה (דף מ"א) עַל בִּימָה שֶׁל עֵץ שֶׁהָיוּ עוֹשִׂין בָּעֲזָרָה:
האנשים. לִלְמֹד: והנשים. לִשְׁמֹעַ: והטף. לָמָּה בָא? לָתֵת שָׂכָר לִמְבִיאֵיהֶם (חגיגה ג'):

ואצונו. וַאֲזָרְזֶנּוּ:

נכר הארץ. גּוֹיֵי הָאָרֶץ:
והסתרתי פני. כְּמוֹ שֶׁאֵינִי רוֹאֶה בְּצָרָתָם:

את השירה הזאת. "הַאֲזִינוּ הַשָּׁמַיִם" עַד "וְכִפֶּר אַדְמָתוֹ עַמּוֹ":
ונאצוני. וְהִכְעִיסוּנִי, וְכֵן כָּל נִאוּץ לְשׁוֹן כַּעַס:
וענתה השירה הזאת לפניו לעד. שֶׁהִתְרֵיתִי בוֹ בְתוֹכָהּ עַל כָּל הַמּוֹצְאוֹת אוֹתוֹ: כי לא תשכח מפי זרעו. הֲרֵי זוֹ הַבְטָחָה לְיִשְׂרָאֵל שֶׁאֵין תּוֹרָה מִשְׁתַּכַּחַת מִזַּרְעָם לְגַמְרֵי:

ויצו את יהושע בן נון. מוּסָב לְמַעְלָה כְּלַפֵּי שְׁכִינָה, כְּמוֹ שֶׁמְּפֹרָשׁ "אֶל הָאָרֶץ אֲשֶׁר נִשְׁבַּעְתִּי לָהֶם":


לקח. כְּמוֹ זָכוֹר (שמות כ') שָׁמוֹר (דברים ה'), הָלוֹךְ (ירמיהו ב'): מצד ארון ברית ה'. נֶחְלְקוּ בוֹ חַכְמֵי יִשְׂרָאֵל בְּבָבָא בָּתְרָא (דף י"ד), יֵשׁ מֵהֶם אוֹמְרִים דַּף הָיָה בוֹלֵט מִן הָאָרוֹן מִבַּחוּץ וְשָׁם הָיָה מֻנָּח, וְיֵשׁ אוֹמְרִים מִצַּד הַלּוּחוֹת הָיָה מֻנָּח בְּתוֹךְ הָאָרוֹן:

הקהילו אלי. וְלֹא תָקְעוּ אוֹתוֹ הַיּוֹם בַּחֲצוֹצְרוֹת לְהַקְהִיל אֶת הַקָּהָל, לְפִי שֶׁנֶּאֱמַר (במדבר י') "עֲשֵׂה לְךָ", וְלֹא הִשְׁלִיט יְהוֹשֻׁעַ עֲלֵיהֶם, וְאַף בְּחַיָּיו נִגְנְזוּ — בְּיוֹם מוֹתוֹ — לְקַיֵּם מַה שֶּׁנֶּאֱמַר (קהלת ח') "וְאֵין שִׁלְטוֹן בְּיוֹם הַמָּוֶת" (תנחומא ויחי ובהעלותך): ואעידה בם את השמים ואת הארץ. וְאִם תֹּאמַר הֲרֵי כְּבָר הֵעִיד לְמַעְלָה — "הַעִדֹתִי בָכֶם הַיּוֹם וְגוֹ'"? הָתָם לְיִשְׂרָאֵל אָמַר, אֲבָל לַשָּׁמַיִם וְלָאָרֶץ לֹא אָמַר, וְעַכְשָׁו בָּא לוֹמַר "הַאֲזִינוּ הַשָּׁמַיִם וְגוֹ'"(דברים ל"ב):
אחרי מותי כי השחת תשחיתון. וַהֲרֵי כָּל יְמוֹת יְהוֹשֻׁעַ לֹא הִשְׁחִיתוּ, שֶׁנֶּאֱמַר "וַיַּעַבְדוּ הָעָם אֶת ה' כֹּל יְמֵי יְהוֹשֻׁעַ" (שופטים ב')? מִכָּאן שֶׁתַּלְמִידוֹ שֶׁל אָדָם חָבִיב עָלָיו כְּגוּפוֹ, כָּל זְמַן שֶׁיְּהוֹשֻׁעַ חַי הָיָה נִרְאֶה לְמֹשֶׁה כְּאִלּוּ הוּא חַי:
