פְּלֹנִי אַלְמֹנִי. וְלֹא נִכְתַּב שְׁמוֹ לְפִי שֶׁלֹּא אָבָה לִגְאֹל: פְּלֹנִי אַלְמֹנִי. מְתֻרְגָּם בַּנְּבִיאִים "כַּסִּי וְטָמִיר": פְּלֹנִי. מְכֻסֶּה וְנֶעְלָם לְשׁוֹן "כִּי יִפָּלֵא", "הֲיִפָּלֵא מֵהַ' דָּבָר": אַלְמֹנִי. אַלְמוֹן מִבְּלִי שֵׁם. (סְפָרִים אֲחֵרִים, אַלְמֹנִי שֶׁהָיָה אַלְמָן מִדִּבְרֵי תוֹרָה, שֶׁהָיָה לוֹ לִדְרשׁ, עַמּוֹנִי וְלֹא עַמּוֹנִית, מוֹאָבִי וְלֹא מוֹאָבִית וְהוּא אָמַר, "פֶּן אַשְׁחִית אֶת נַחֲלָתִי":)


וְאֵדְעָה כִּי אֵין זוּלָתְךָ. קָרוֹב לִגְאוֹל:
וּמֵאֵת רוּת הַמּוֹאֲבִיָּה. אַתָּה צָרִיךְ לִקְנוֹת, וְהִיא אֵינָהּ מִתְרַצָּה אֶלָּא אִם כֵּן תִּשָּׂאֶנָּה:
פֶּן אַשְׁחִית אֶת נַחֲלָתִי. זַרְעִי. כְּמוֹ "נַחֲלַת ה' בָּנִים". לָתֵת פְּגַם בְּזַרְעִי, שֶׁנֶּאֱמַר, "לֹא יָבֹא עַמּוֹנִי וּמוֹאָבִי". וְטָעָה בְּ"עַמּוֹנִי וְלֹא עַמּוֹנִית":
עַל הַגְּאוּלָּה. זוּ מְכִירָה: תְּמוּרָה. זוּ חֲלִיפִין: שָׁלַף אִישׁ נַעֲלוֹ. זֶהוּ קִנְיָן. כְּמוֹ שֶׁאָנוּ קוֹנִין בְּסוּדָר בִּמְקוֹם נַעַל. וְרַבּוֹתֵינוּ זִכְרוֹנָם לִבְרָכָה נֶחְלְקוּ בַדָּבָר, מִי נָתַן לְמִי. יֵשׁ אוֹמְרִים קוֹנִין בְּכֶלְיוֹ שֶׁל קוֹנֶה וּבֹעַז נָתַן לַגּוֹאֵל, וְיֵשׁ אוֹמְרִים קוֹנִין בְּכֶלְיוֹ שֶׁל מַקְנֶה וְגוֹאֵל נָתַן לְבֹעַז: וְזֹאת הַתְּעוּדָה בְּיִשְׂרָאֵל. מִשְׁפַּט הָעֵדוּת:


לְהָקִים שֵׁם הַמֵּת עַל נַחֲלָתוֹ. מִתּוֹךְ שֶׁאִשְׁתּוֹ יוֹצְאָה וּבָאָה בַּנַּחֲלָה, וּמַכְנֶסֶת וּמוֹצִיאָה, אוֹמְרִים "זֹאת הָיְתָה אֵשֶׁת מַחְלוֹן", וּשְׁמוֹ נִזְכָּר עָלֶיהָ:
כְּרָחֵל וּכְלֵאָה. אַף עַל פִּי שֶׁהָיוּ מִשֵּׁבֶט יְהוּדָה וּמִבְּנֵי לֵאָה, מוֹדִים הֵם עַל רָחֵל שֶׁהָיְתָה עִקָּרָהּ שֶׁל בַּיִת, וְהִקְדִּימוּ רָחֵל לְלֵאָה: וּקְרָא שֵׁם. כְּלוֹמַר, "יִגְדַּל שְׁמֶךָ":
כְּבֵית פֶּרֶץ. שֶׁיָּצְאוּ מִמֶּנּוּ:
