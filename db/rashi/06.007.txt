
עִם בֵּית אָוֶן אֵצֶל בֵּית אָוֶן. מִקֶּדֶם לְבֵית אֵל מִמִּזְרַח לְבֵית אֵל.


עַד הַשְּׁבָרִים (תַּרְגּוּם:) עַד דְּתַבְרוּנוֹן.*

וְלוּ הוֹאַלְנוּ הַלְוַאי נִמְלַכְנוּ לָשֶׁבֶת בְּעֵבֶר הַיַּרְדֵּן מִזְרָחָה, בְּאֶרֶץ סִיחוֹן וְעוֹג שֶׁנִכְבְּשָׁה כְּבָר.

וּמַה תַּעֲשֶׂה לְשִׁמְךָ הַמְשׁוּתָּף בִּשְׁמֵנוּ, זֶהוּ מִדְרָשׁוֹ. וּפְשׁוּטוֹ וּמַה תַּעֲשֶׂה לְשֵׁם גְּבוּרָתְךָ שֶׁיָּצָא כְּבָר, וְעַתָּה יֹאמְרוּ תָּשַׁשׁ כֹּחוֹ.
קֻם לָךְ קָם לָךְ כְּתִיב, עָמַד לְךָ מַה שֶּׁהִתְפַּלַּלְתָּ לְפָנַי וְהִזְכַּרְתָּ. דָּבָר אַחֵר: עָמַדְתָּ לְךָ בַּמַּחֲנֶה וְלֹא יָצָאתָ עִמָּהֶם, וַאֲנִי אָמַרְתִּי: (במדבר כז יז), אֲשֶׁר יוֹצִיאֵם וַאֲשֶׁר יְבִיאֵם, (דברים לא כג): כִּי אַתָּה תָּבִיא אֶת בְּנֵי יִשְׂרָאֵל... וְאָנֹכִי אֶהְיֶה עִמְּךָ: (שם ג כח), כִּי הוּא יַעֲבֹר לִפְנֵי הָעָם הַזֶּה וְהוּא יַנְחִיל אוֹתָם אִם תֵּלֵךְ אַתָּה לִפְנֵיהֶם יַצְלִיחוּ, וְאִם לָאו לֹא יַצְלִיחוּ. דָּבָר אַחֵר: קֻם לָךְ בִּשְׁבִילְךָ זֹאת לָהֶם, לֹא אָמַרְתִּי לְךָ לְהַקְדִּישׁ שְׁלַל הָעִיר. 



לַבָּתִּים יֵשׁ בְּמִשְׁפָּחָה אַחַת הַרְבֵּה בָּתֵּי אָבוֹת. לַגְּבָרִים לְרָאשֵׁי גּוּלְגּוֹלֶת, כָּל אַנְשֵׁי הַבַּיִת יָבֹאוּ לַגּוֹרָל.
(תוספת) יִשָּׂרֵף בָּאֵשׁ הָאֹהֶל וְהַמִּטַּלְטְלִין. אֹתוֹ וְאֶת כָּל אֲשֶׁר לוֹ כַּדִּין הַמְפוֹרָשׁ לְּמַטָּה (להלן פסוק כה): אוֹתוֹ וְהַבְּהֵמָה בִּסְקִילָה, וְהַזָּקֵף שֶׁהוּא עַל 'בָּאֵשׁ' מוֹכִיחַ שֶׁהוּא נִפְרָד 'מֵאוֹתוֹ', שֶׁכֵּן מָצִינוּ מִקְרָאוֹת שֶׁהַטַּעַם מְחַלְּקָן, כְּמוֹ (דברים יא ל): אַחֲרֵי דֶּרֶךְ מְבוֹא הַשֶּׁמֶשׁ, שֶׁ'אַחֲרֵי' מוּפְלָג 'מִדֶּרֶךְ' עַל יְדֵי הַטַּעַם, וְכֵן פִּתְרוֹנוֹ: הַנִּלְכָּד בַּחֵרֶם יִשָּׂרֵף בָּאֵשׁ, הָרָאוּי לִישָּׂרֵף כְּמוֹ שֶׁמְּפֹרָשׁ לְּמַטָּה. אֹתוֹ וְאֶת כָּל אֲשֶׁר לוֹ כַּדִּין הָאָמוּר לְּמַטָּה, וְזֶה מִקְרָא קָצָר, וְדוֹמֶה לוֹ (שמואל ב ה ח): כָּל מַכֵּה יְבֻסִי וְיִגַּע בַּצִּנּוֹר, לֹא פֵּירַשׁ מַה יִּהְיֶה לוֹ. וּבְדִבְרֵי הַיָּמִים (א יא ו) פֵּירַשׁ: יִהְיֶה לְרֹאשׁ, אַף כָּאן לֹא פֵּירַשׁ מַה יִּהְיֶה בְּ'אוֹתוֹ וְכָל אֲשֶׁר לוֹ' וּלְמַטָּה פֵּירַשׁ.
וַיַּקְרֵב אֶת יִשְׂרָאֵל לִפְנֵי הַחֹשֶׁן, מָקוֹם שֶׁהַשְּׁבָטִים כְּתוּבִים, וּמָסַר לוֹ הַקָּדוֹשׁ בָּרוּךְ הוּא סִימָן, שֶׁהַשֵּׁבֶט שֶׁחָטָא אַבְנוֹ כָּהָה, וְכָהֲתָה אַבְנוֹ שֶׁל יְהוּדָה.
וַיַּקְרֵב אֶת מִשְׁפַּחַת יְהוּדָה לְגוֹרָל אֶחָד קֵרַב רָאשֵׁי הַמִּשְׁפָּחוֹת, אָדָם אֶחָד מִכָּל הַמִּשְׁפָּחָה וְהִטִּילוּ גּוֹרָל עַל מִי יִפּוֹל, וְאַחַר כָּךְ עַל כָּל בָּתֵּי אָבוֹת שֶׁל אוֹתָהּ מִשְׁפָּחָה, וּבָא אָדָם אֶחָד מִכָּל בֵּית אָב לְגוֹרָל, וְאַחַר כֵּן בָּאוּ כָּל אַנְשֵׁי אוֹתוֹ הַבַּיִת.

שִׂים־נָא כָבוֹד הִתְחִיל לְהוֹצִיא לַעַז עַל הַגּוֹרָל, אָמַר לוֹ: בְּגוֹרָל אַתָּה בָּא עָלַי, אַתָּה וְאֶלְעָזָר הַכֹּהֵן גְּדוֹלֵי הַדּוֹר אַתֶּם; הַטֵּל הַגּוֹרָל עַל שְׁנֵיכֶם, וְיִפּוֹל עַל הָאֶחָד. אָמַר לוֹ: שִׂים־נָא כָבוֹד, בְּבַקָּשָׁה מִמְּךָ, אַל תּוֹצִיא לַעַז עַל הַגּוֹרָלוֹת שֶׁבּוֹ עֲתִידָה הָאָרֶץ לֵיחָלֵק. 
וַיַּעַן עָכָן רָאָה בְּנֵי יְהוּדָה נֶאֱסָפִים לְמִלְחָמָה, אָמַר: מוּטָב שֶׁאָמוּת אֲנִי לְבַדִּי וְאַל יֵהָרְגוּ כַּמָּה אֲלָפִים מִיִּשְׂרָאֵל. וְכָזֹאת וְכָזֹאת עָשִׂיתִי גַּם בַּחֲרָמִים אֲחֵרִים בִּימֵי משֶׁה, שֶׁנֶּאֱמַר (במדבר כא ב): וְהַחֲרַמְתִּי אֶת עָרֵיהֶם.
וָאֵרֶא בַשָּׁלָל הִתְבּוֹנַנְתִּי בְּמַה שֶּׁכָּתוּב בַּתּוֹרָה (דברים כ יד): וְאָכַלְתָּ [אֶת־שְׁלַל] וְגוֹ'. אַדֶּרֶת שִׁנְעָר תַּרְגּוּמוֹ: אִיצְטְלָא (דְּמֵילָתָא) בַּבְלִי, שֶׁכָּל מֶלֶךְ שֶׁלֹּא קָנָה לוֹ פַּלְטֵרִין בְּאֶרֶץ יִשְׂרָאֵל, לֹא נִתְקָרְרָה דַּעְתּוֹ בְּמַלְכוּתוֹ, שֶׁנֶּאֱמַר (ירמיהו ג יט): וְאֶתֵּן־לָךְ אֶרֶץ חֶמְדָּה נַחֲלַת צְבִי צִבְאוֹת גּוֹיִם, וְהָיָה לְמֶלֶךְ בָּבֶל פַּלְטֵרִין בִּירִיחוֹ, וּכְשֶׁהָיָה בָּא לְכָאן לוֹבֵשׁ אוֹתָם.
וַיָּרֻצוּ הָאֹהֱלָה שֶׁלֹּא יַקְדִּימוּ שֵׁבֶט יְהוּדָה וְיִטְּלוּהָ מִשָּׁם לְהַכְחִישׁ אֶת הַגּוֹרָל.
וַיַּצִּקֻם (תַּרְגּוּם:) וְאַתִּיכוּנוּן, וְרַבּוֹתֵינוּ אָמְרוּ (סנהדרין מד א): בָּא וַחֲבָטָן לִפְנֵי הַמָּקוֹם, אָמַר לְפָנָיו: רִבּוֹנוֹ שֶׁל עוֹלָם, עַל אֵלּוּ יִפְּלוּ רוּבָּן שֶׁל סַנְהֶדְרִין.
וְאֶת בָּנָיו וְגוֹ' וְכָל יִשְׂרָאֵל לִרְאוֹת בְּרִדּוּיוֹ וְיִוַּסְּרוּ מִלַּעֲשׂוֹת כָּמוֹהוּ. וְאֶת שׁוֹרוֹ וְאֶת חֲמֹרוֹ לְאַבְּדָם, כְּמָה שֶׁנֶּאֱמַר (לעיל פסוק טו): וְהָיָה הַנִּלְכָּד בַּחֵרֶם יִשָּׂרֵף בָּאֵשׁ אֹתוֹ וְכָל אֲשֶׁר לוֹ.
וַיִּרְגְּמוּ אֹתוֹ... אֶבֶן שֶׁחִלֵּל אֶת הַשַּׁבָּת. וַיִּשְׂרְפוּ אֹתָם הָאֹהֶל וְהַמִּטַּלְטְלִין. וַיִּסְקְלוּ אֹתָם הַשּׁוֹר וְהַבְּהֵמָה.
