וּבִשְׁנַת שְׁמוֹנֶה עֶשְׂרֵה לַמֶּלֶךְ יָרָבְעָם. שָׁנָה אַחַת אַחַר מוֹת רְחַבְעָם, וּבִשְׁנַת עֶשְׂרִים לְיָרָבְעָם מָלַךְ אָסָא, שָׁנָה הַשְּׁלִישִׁית שֶׁל נָדָב (אֲבִיָּם) נִמְנֵית לַמֶּלֶךְ אָסָא שָׁנָה, וְלוֹ נַמֵּי נִמְנֵית.










וַיַּעֲבֵר הַקְּדֵשִׁים. תִּרְגֵּם יוֹנָתָן: וּפַלֵּי נָפְקַת בְּרָא.
וַיְסִרֶה מִגְּבִירָה. מִהְיוֹת גְּבִירָה. אֲשֶׁר עָשְׂתָה מִפְלֶצֶת. אָמְרוּ רַבּוֹתֵינוּ מַפְלִיא לֵיצָנוּתָא, כְּמִין זַכְרוּת עָשְׂתָה לָּהּ, וְהִיא הָיְתָה נִבְעֶלֶת לוֹ בְּכָל יוֹם.
וְהַבָּמוֹת לֹא סָרוּ. בָּמוֹת הַיָּחִיד, שֶׁהֻרְגְּלוּ לִקָּרֵב עֲלֵיהֶן לַשָּׁמַיִם מִשֶּׁחָרְבָה שִׁילֹה עַד שֶׁנִּבְנָה הַבַּיִת שֶׁהָיוּ הַבָּמוֹת מֻתָּרוֹת, לֹא סָרוּ עַתָּה, וְאַף עַל פִּי שֶׁנֶּאֶסְרוּ מִשֶּׁנִּבְנָה הַבַּיִת, וְהָיוּ עֲנוּשִׁים עָלֶיהָ כָּרֵת.







אֵין נָקִי. אֲפִלּוּ חָתָן מֵחֶדְרוֹ, שֶׁנֶּאֱמַר בּוֹ: נָקִי יִהְיֶה לְבֵיתוֹ שָׁנָה אֶחָת.


שְׁנָתַיִם. הַשְּׁנִיָּה וְהַשְּׁלִישִׁית לְאָסָא.







בִּשְׁנַת שָׁלשׁ לְאָסָא מֶלֶךְ יְהוּדָה מָלַךְ בַּעְשָׁא. שָׁנָה הַשְּׁלִישִׁית לְאָסָא, נִמְנֵית לְנָדָב וּלְבַּעְשָׁא.
