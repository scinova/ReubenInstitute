
שְׁנֵים עָשָׂר אֲנָשִׁים הֵם שֶׁנִּזְכָּרִים לְמַעְלָה לִהְיוֹת נְכוֹנִים.
וְהַעֲבַרְתֶּם אוֹתָם עִמָּכֶם כְּמִצְוַת משֶׁה (דברים כז ד) לִבְנוֹת לָהֶם מִזְבֵּחַ בְּהַר עֵיבָל וְלִכְתּוֹב עֲלֵיהֶם אֶת דִּבְרֵי הַתּוֹרָה, וּבוֹ בַּיּוֹם בָּאוּ אֶל הַר עֵיבָל וּבָנוּ בָּהֶם אֶת הַמִּזְבֵּחַ וְהֶעֱלוּ עוֹלוֹת וּשְׁלָמִים, וְאָכְלוּ וְשָׁתוּ, וְקִפְּלוּ אוֹתָם וּבָאוּ וְלָנוּ בַּגִּלְגָּל.

עִבְרוּ לִפְנֵי אֲרוֹן וְגוֹ' הִכָּנְסוּ עַתָּה בַּיַּרְדֵּן וְעִבְרוּ עַד לִפְנֵי הַכֹּהֲנִים.

וְהָיוּ הָאֲבָנִים הָאֵלֶּה בַּגִּלְגָּל לְזִכְרוֹן זֶה.

וּשְׁתֵּים עֶשְׂרֵה אֲבָנִים אֲחֵרוֹת הֵקִים יְהוֹשֻׁעַ בְּתוֹךְ הַיַּרְדֵּן.
עַד תֹּם כָּל־הַדָּבָר כְּמוֹ שֶׁשָּׁנִינוּ בְּמַסֶּכֶת סוֹטָה (לד א): עוֹדָם בַּיַּרְדֵּן, אָמַר לָהֶם יְהוֹשֻׁעַ: דְּעוּ לָמָּה אַתֶּם עוֹבְרִים אֶת הַיַּרְדֵּן, עַל מְנָת שֶׁתּוֹרִישׁוּ אֶת כָּל יוֹשְׁבֵי הָאָרֶץ וְכוּ'.
וַיְהִי כַּאֲשֶׁר תַּם כָּל הָעָם לַעֲבוֹר וַיַּעֲבֹר אֲרוֹן וְגוֹ' לֹא כְּדֶרֶךְ שֶׁעָבְרוּ הָאֲחֵרִים, אֶלָּא כְּמוֹ שֶׁמְּפוֹרָשׁ לְמַטָּה בָּעִנְיָן (להלן פסוק יח): נִתְּקוּ כַּפּוֹת רַגְלֵי הַכֹּהֲנִים לַאֲחוֹרֵיהֶם אֶל הֶחָרָבָה עַל הַשָּׂפָה שֶׁנִּכְנְסוּ בָּהּ, שֶׁהָיוּ עוֹמְדִים אֶצְלָהּ, וְחָזְרוּ הַמַּיִם לַאֲחוֹרֵיהֶם לֵילֵךְ כִּתְמוֹל שִׁלְשׁוֹם; נִמְצָא, אֲרוֹן וְנוֹשְׂאָיו מִצַּד זֶה וְכָל יִשְׂרָאֵל מִצַּד זֶה, נָשָׂא הָאֲרוֹן אֶת נוֹשְׂאָיו וְעָבַר. לִפְנֵי הָעָם לְעֵינֵי הָעָם.




צַוֵּה אֶת הַכֹּהֲנִים כָּאן פֵּירַשׁ הַכָּתוּב אֶת הָאָמוּר לְמַעְלָה, הֵיאָךְ עָבַר אֲרוֹן ה' וְהַכֹּהֲנִים לִפְנֵי הָעָם וְגוֹ'. וְיַעֲלוּ מִן הַיַּרְדֵּן 'וְיַעַבְרוּ' אֵין כְּתִיב כָּאן, אֶלָּא 'וְיַעֲלוּ', לָמַדְנוּ שֶׁעַל הַשָּׂפָה שֶׁהָיוּ עוֹמְדִין אֶצְלָהּ עָלוּ, וְאִי אֶפְשָׁר לוֹמַר שֶׁהָיוּ עוֹמְדִין אֵצֶל שְׂפַת הַיַּרְדֵּן שֶׁבַּמַּעֲרָב, שֶׁהֲרֵי נֶאֱמַר לְמַעְלָה (ג טו) נִטְבְּלוּ רַגְלֵי הַכֹּהֲנִים בִּקְצֵה הַמַּיִם, וְשָׁם נֶאֱמַר וַיַּעַמְדוּ הַכֹּהֲנִים וְכָל יִשְׂרָאֵל עוֹבְרִים.

נִתְּקוּ כַּפּוֹת רַגְלֵי הַכֹּהֲנִים מִן הַמַּיִם אֶל הֶחָרָבָה שֶׁאֶצְלָם, וַיָּשׁוּבוּ הַמַּיִם לִמְקוֹמָם; נִמְצָא, אֲרוֹן מִצַּד זֶה וְיִשְׂרָאֵל מִצַּד זֶה, נָשָׂא אֲרוֹן אֶת נוֹשְׂאָיו וְעָבַר. וְעַל דָּבָר זֶה נֶעֱנַשׁ עוּזָּא (שמואל ב ו ז) כְּשֶׁאָחַז בָּאָרוֹן, נוֹשְׂאָיו נָשָׂא, עַצְמוֹ לֹא כָּל שֶׁכֵּן.

הֵקִים יְהוֹשֻׁעַ בַּגִּלְגָּל הוּא הַמָּלוֹן אֲשֶׁר לָנוּ בּוֹ הַלַּיְלָה.
