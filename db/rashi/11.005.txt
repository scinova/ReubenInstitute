וְנַעֲמָן שַׂר צְבָא. נִסִּים שֶׁנַּעֲשׂוּ עַל יְדֵי אֱלִישָׁע, מְסַדֵּר וְהוֹלֵךְ. תְּשׁוּעָה לַאֲרָם. הוּא מָשַׁךְ בַּקֶּשֶׁת לְתֻמּוֹ, וְהָרַג אֶת אַחְאָב.
יָצְאוּ גְדוּדִים. כְּשֶׁהוֹלְכִין מֵאָה אוֹ מָאתַיִם מֵעַצְמָן לִשְׁלֹל כַּאֲשֶׁר יִמְצְאוּן, הוּא קָרוּי גְּדוּד. נַעֲרָה קְטַנָּה. רִיבָה קְטַנָּה, מֵעִיר נַעֲרוֹן. וַתְּהִי לִפְנֵי. וַהֲוַת מְשַׁמְּשָׁא קֳדָם אִתַּת נַעֲמָן.
אַחֲלֵי אֲדֹנִי לִפְנֵי הַנָּבִיא. לְשׁוֹן וַיְחַל משֶׁה, בַּקָּשׁוֹת כָּל הַמִּתְפַּלְּלִים עָלָיו יִהְיוּ, שֶׁיָּבוֹא הַנֵּס לִפְנֵי הַנָּבִיא. אַחֲלֵי. שוהיידמנ"ס בְּלַעַ"ז, כְּלוֹמַר זוֹ הִיא בַּקָּשָׁה שֶׁהוּא צָרִיךְ לָהּ.
וַיָּבֹא. נַעֲמָן. וַיַּגֵּד לַאדֹנָיו. מֶלֶךְ אֲרָם.
וַיִּקַּח בְּיָדוֹ. מִנְחָה לַנָּבִיא. וְשֵׁשֶׁת אֲלָפִים זָהָב. זְהוּבִים.
לֵאמֹר וְעַתָּה כְּבוֹא הַסֵּפֶר הַזֶּה וְגוֹ'. הַסֵּפֶר אָמַר לוֹ, וְעַתָּה כְּבוֹא הַסֵּפֶר הַזֶּה וְגוֹ'. וַאֲסַפְתּוֹ מִצָּרַעְתּוֹ. 'אֲסֵפָה' בִּמְצֹרָע הִיא לְשׁוֹן רְפוּאָתוֹ, כִּי בְּהִתְרַפּאוֹתוֹ הוּא נֶאֱסָף אֶל תּוֹךְ בְּנֵי אָדָם, וּבְחָלְיוֹ הַכֹּל בְּדֵלִין הֵימֶנּוּ.
הַאֱלֹהִים אָנִי. הֵ"א נָקוּד פַּתָּ"ח, לְלַמֵּד שֶׁהוּא בִּלְשׁוֹן תְּמִיהָה. מִתְאַנֶּה. מְבַקֵּשׁ תּוֹאֲנָה לְהִתְגָּרוֹת בִּי. מִתְאַנֶּה. לְשׁוֹן עֲלִילוֹת דְּבָרִים.



הִנֵּה אָמַרְתִּי. סָבוּר הָיִיתִי, אֵלַי יֵצֵא הַנָּבִיא וִידַבֵּר עִמִּי וְיִרְאֶה אֶת הַחֹלִי. וְהֵנִיף יָדוֹ אֶל הַמָּקוֹם. הַצָּרַעַת.

אָבִי. כְּמוֹ אֲדוֹנִי. הֲלוֹא תַעֲשֶׂה. וְכִי לֹא תַעֲשֵׂהוּ, אֲפִלּוּ אָמַר לְךָ דְּבַר טֹרַח. וְאַף כִּי. קַל וָחֹמֶר שֶׁאָמַר לְךָ דָּבָר קַל, רְחַץ וּטְהָר.

בְרָכָה. מִנְחַת שָׁלוֹם שֶל הַקְבָּלַת פָּנִים, שֶׁשּׁוֹאֵל תַּלְמִיד אוֹ עֶבֶד בִּשְׁלוֹם הָרַב, שולד"ו בְּלַעַ"ז.
אִם אֶקָּח. שֶׁדְּמֵי עֲבוֹדָה זָרָה מְעֹרָבִין בּוֹ.
וָלֹא. לְשׁוֹן בַּקָּשָׁה, לְשׁוֹן הַלְוַאי. יֻתַּן נָא. מֵאֲדָמָה זוֹ מֵאֶרֶץ יִשְׂרָאֵל שֶׁהִיא קְדוֹשָׁה, מַשָּׂא שְׁנֵי פְּרָדִים, וְאֶשָּׂאֶנָּה לְעִירִי וְאֶעֱשֶׂה אוֹתָהּ מִזְבֵּחַ.
בֵית רִמּוֹן. שֵׁם עֲבוֹדָה זָרָה. וְהִשְׁתַּחֲוֵיתִי. עַל כָּרְחִי, כְּשֶׁשּׁוֹחֶה אֲדוֹנִי וְהוּא נִשְׁעָן עַל יָדִי.
כִּבְרַת אָרֶץ. שֵׁם מִדָּה שֶל קַרְקַע, כְּמוֹ: צִמְדֵי כֶרֶם, ארפינ"ט בְּלַעַ"ז.
מְאוּמָה. חָסֵר אָלֶ"ף, לְפִי שֶׁהָיְתָה לְקִיחָה זוֹ לְמוּם.
וַיִּפֹּל. וְאִתְרְכִין.

הוֹאֵל. הִשָּׁבַע שֶׁשְּׁלַחֲךָ. בִּשְׁנֵי חֲרִטִים. מִינֵי בְּגָדִים וְסוּדָרִין, כְּמוֹ: הַמִּטְפָּחוֹת וְהָחֲרִיטִים, כָּךְ חִבְּרוֹ מְנַחֵם. אֲבָל יוֹנָתָן תִּרְגֵּם זֶה 'פַּלְדָּסִין', וְאֶת שֶׁבְּסֵפֶר יְשַׁעְיָהוּ תִּרְגֵּם 'מְחַכַּיָּא'. וְהַפּוֹתְרִין אוֹמְרִין: בִּשְׁנֵי חֲרִיטֵי, בִּשְׁנֵי כִּיסִין אֲרֻכִּים, פירדיי"ש בְּלַעַ"ז. אֶל שְׁנֵי נְעָרָיו. שֶׁל נַעֲמָן. וַיִּשְׂאוּ לְפָנָיו. לִפְנֵי גֵּחֲזִי.
וַיָּבֹא אֶל הָעֹפֶל. תִּרְגֵּם יוֹנָתָן: לַאֲתַר כָּסֵי. וַיִּקַּח מִיָּדָם. מִיַּד שְׁנֵי נַעֲרֵי נַעֲמָן. וַיְשַׁלַּח אֶת הָאֲנָשִׁים. שֶׁלֹּא יִרְאֶה אוֹתָם אֱלִישָׁע.

הַעֵת לָקַחַת אֶת הַכֶּסֶף. לְהִתְעַשֵּׁר, וְלָקַחַת מִמֶּנּוּ בְּגָדִים וְזֵיתִים וְגוֹ'.
