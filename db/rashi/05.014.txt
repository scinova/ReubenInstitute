לא תתגדדו. לֹא תִתְּנוּ גְדִידָה וְשֶׂרֶט בִּבְשַׂרְכֶם עַל מֵת כְּדֶרֶךְ שֶׁהָאֱמוֹרִיִּים עוֹשִׂין, לְפִי שֶׁאַתֶּם בָּנָיו שֶׁל מָקוֹם וְאַתֶּם רְאוּיִין לִהְיוֹת נָאִים וְלֹא גְּדוּדִים וּמְקֹרָחִים: בין עיניכם. אֵצֶל הַפַּדַּחַת, וּבְמָקוֹם אַחֵר הוּא אוֹמֵר (ויקרא כ"א) לֹא יִקְרְחֻה בְּרֹאשָׁם – לַעֲשׂוֹת כָּל הָרֹאשׁ כְּבֵין הָעֵינַיִם (ספרי):
כי עם קדוש אתה. קְדֻשַּׁת עַצְמְךָ מֵאֲבוֹתֶיךָ, וְעוֹד ובך בחר ה' :
כל תועבה. כָּל שֶׁתִּעַבְתִּי לְךָ, כְּגוֹן צָרַם אֹזֶן בְּכוֹר כְּדֵי לְשָׁחֳטוֹ בַמְּדִינָה הֲרֵי דָּבָר שֶׁתִּעַבְתִּי לְךָ, כָּל מוּם לֹא יִהְיֶה בּוֹ, בָּא וְלִמֵּד כָּאן שֶׁלֹּא יִשְׁחַט וְיֹאכַל עַל אוֹתוֹ הַמּוּם. בִּשֵּׁל בָּשָׂר בְּחָלָב הֲרֵי דָּבָר שֶׁתִּעַבְתִּי לְךָ, וְהִזְהִיר כָּאן עַל אֲכִילָתוֹ (חולין קי"ד):
זאת הבהמה, איל וצבי ויחמור. לָמַדְנוּ שֶׁהַחַיָּה בִכְלַל בְּהֵמָה, וְלָמַדְנוּ שֶׁהַבְּהֵמָה וְחַיָּה טְמֵאָה מְרֻבָּה מִן הַטְּהוֹרָה, שֶׁבְּכָל מָקוֹם פּוֹרֵט אֶת הַמֻּעָט (ספרי; חולין י"ג):
ואקו. מְתֻרְגָּם יַעֲלָא (איוב ל"ט) "יַעֲלֵי סָלַע", הוּא אשטנב"וק: ותאו. "תּוֹרְבָּלָא", תּוֹר הַיַּעַר, "בָּלָא" — יַעַר בִּלְשׁוֹן אֲרַמִּי:
מפרסת. סְדוּקָה, כְּתַרְגּוּמוֹ: פרסה. פלאנט"א: ושסעת. חֲלוּקָה בִשְׁתֵּי צִפָּרְנַיִם, שֶׁיֵּשׁ סְדוּקָה וְאֵינָהּ חֲלוּקָה בַּצִּפָּרְנַיִם, וְהִיא טְמֵאָה: בבהמה. מַשְׁמַע מַה שֶּׁנִּמְצָא בַבְּהֵמָה אֱכֹל, מִכָּאן אָמְרוּ שֶׁהַשְּׁלִיל נִתָּר בִּשְׁחִיטַת אִמּוֹ (חולין ס"ט):
השסועה. בְּרִיָּה הִיא שֶׁיֵּשׁ לָהּ שְׁנֵי גַּבִּין וּשְׁתֵּי שִׁדְרָאוֹת (שם ס'); אָמְרוּ רַבּוֹתֵינוּ לָמָּה נִשְׁנוּ? בַּבְּהֵמוֹת מִפְּנֵי הַשְּׁסוּעָה וּבָעוֹפוֹת מִפְּנֵי הָרָאָה, שֶׁלֹּא נֶאֱמְרוּ בְּת"כֹּ (שם ס"ג):
ובנבלתם לא תגעו. רַבּוֹתֵינוּ פֵּרְשׁוּ — בָּרֶגֶל, שֶׁאָדָם חַיָּב לְטַהֵר אֶת עַצְמוֹ בָּרֶגֶל, יָכוֹל יִהְיוּ מֻזְהָרִים בְּכָל הַשָּׁנָה, תַּ"לֹ "אֱמֹר אֶל הַכֹּהֲנִים וְגוֹ'", וּמַה טֻּמְאַת הַמֵּת חֲמוּרָה, הַכֹּהֲנִים מֻזְהָרִים וְאֵין יִשְׂרָאֵל מֻזְהָרִים, טֻמְאַת נְבֵלָה קַלָּה, לֹא כָּל שֶׁכֵּן:


כל צפור טהרה תאכלו. לְהַתִּיר מְשֻׁלַּחַת שֶׁבַּמְּצֹרָע (ספרי; קידושין נ"ז):
וזה אשר לא תאכלו מהם. לֶאֱסֹר אֶת הַשְּׁחוּטָה (שם):
הראה והאיה. הִיא רָאָה הִיא אַיָּה הִיא דַיָּה, וְלָמָּה נִקְרָא שְׁמָהּ רָאָה? שֶׁרוֹאָה בְּיוֹתֵר, וְלָמָּה הִזְהִיר בְּכָל שְׁמוֹתֶיהָ? שֶׁלֹּא לִתֵּן פִּתְחוֹן פֶּה לְבַעַל דִּין לַחֲלֹק – שֶׁלֹּא יְהֵא הָאוֹסְרָהּ קוֹרֵא אוֹתָהּ רָאָה, וְהַבָּא לְהַתִּיר אוֹמֵר זוֹ דַּיָּה שְׁמָהּ אוֹ אַיָּה שְׁמָהּ, וְזוֹ לֹא אָסַר הַכָּתוּב. וּבָעוֹפוֹת פֵּרֵט לְךָ הַטְּמֵאִים לְלַמֵּד שֶׁהָעוֹפוֹת הַטְּהוֹרִים מְרֻבִּים עַל הַטְּמֵאִים, לְפִיכָךְ פֵּרֵט אֶת הַמֻּעָט (חולין ס"ג):


התנשמת. קלב"א שורי"ץ:
שלך. הַשּׁוֹלֶה דָּגִים מִן הַיָּם (שם):
דוכיפת. הוּא תַּרְנְגוֹל הַבָּר, וּבְלַעַז הרו"פא, וְכַרְבָּלְתּוֹ כְפוּלָה (גיט' ס"ח):
שרץ העוף. הֵם הַנְּמוּכִים, הָרוֹחֲשִׁים עַל הָאָרֶץ. זְבוּבִין וּצְרָעִים וַחֲגָבִים טְמֵאִים קְרוּיִים שֶׁרֶץ:
כל עוף טהור תאכלו. וְלֹא אֶת הַטָּמֵא, בָּא לִתֵּן עֲשֵׂה עַל לֹא תַעֲשֶׂה; וְכֵן בַּבְּהֵמָה "אֹתָהּ תֹּאכֵלוּ", וְלֹא בְהֵמָה טְמֵאָה, וְלָאו הַבָּא מִכְּלַל עֲשֵׂה עֲשֵׂה, לַעֲבֹר עֲלֵיהֶם בַּעֲשֵׂה וְלֹא תַעֲשֶׂה:
לגר אשר בשעריך. גֵּר תּוֹשָׁב שֶׁקִּבֵּל עָלָיו שֶׁלֹּא לַעֲבֹד עֲ"זָ וְאוֹכֵל נְבֵלוֹת (ספרי): כי עם קדוש אתה לה'. קַדֵּשׁ אֶת עַצְמְךָ בַּמֻּתָּר לְךָ – דְּבָרִים הַמֻּתָּרִים וַאֲחֵרִים נוֹהֲגִים בָּהֶם אִסּוּר, אַל תַּתִּירֵם בִּפְנֵיהֶם (שם): לא תבשל גדי. שָׁלוֹשׁ פְּעָמִים, פְּרָט לְחַיָּה וּלְעוֹפוֹת וְלִבְהֵמָה טְמֵאָה (חולין קי"ג):
לא תבשל גדי‎ … עשר תעשר. מָה עִנְיַן זֶה אֵצֶל זֶה? אָמַר לָהֶם הַקָּבָּ"ה לְיִשְׂרָאֵל: אַל תִּגְרְמוּ לִי לְבַשֵּׁל גְּדָיִים שֶׁל תְּבוּאָה עַד שֶׁהֵן בִּמְעֵי אִמּוֹתֵיהֶן, שֶׁאִם אֵין אַתֶּם מְעַשְּׂרִים מַעַשְׂרוֹת כָּרָאוּי כְּשֶׁהוּא סָמוּךְ לְהִתְבַּשֵּׁל אֲנִי מוֹצִיא רוּחַ קָדִים וְהִיא מְשַׁדַּפְתָּן, שֶׁנֶּאֱמַר (מלכים ב י"ט) "וּשְׁדֵפָה לִפְנֵי קָמָה", וְכֵן לְעִנְיַן בִּכּוּרִים (פסיק' ר'; תנחומא): שנה שנה. מִכָּאן שֶׁאֵין מְעַשְּׂרִין מִן הֶחָדָשׁ עַל הַיָּשָׁן (ספרי):
ואכלת וגו'. זֶה מַעֲשֵׂר שֵׁנִי. שֶׁכְּבָר לִמְּדָנוּ לִתֵּן מַעֲשֵׂר רִאשׁוֹן לַלְוִיִם, שֶׁנֶּאֱמַר "כִּי תִקְחוּ מֵאֵת בְּנֵי יִשְׂרָאֵל וְגוֹ'" (במדבר י"ח) וְנָתַן לָהֶם רְשׁוּת לְאָכְלוֹ בְּכָל מָקוֹם, שֶׁנֶּאֱמַר "וַאֲכַלְתֶּם אֹתוֹ בְּכָל מָקוֹם" (שם י"ח), עַל כָּרְחֲךָ זֶה מַעֲשֵׂר אַחֵר הוּא:
כי יברכך. שֶׁתְּהֵא הַתְּבוּאָה מְרֻבָּה לָשֵׂאת:

בכל אשר תאוה נפשך – כְּלָל: בבקר ובצאון וביין ובשכר – פְּרָט: ובכל אשר תשאלך נפשך – חָזַר וְכָלַל, מַה הַפְּרָט מְפֹרָשׁ וְלַד וַלְדוֹת הָאָרֶץ וְרָאוּי לְמַאֲכַל אָדָם וְכוּ' (עירובין כ"ז):
והלוי, לא תעזבנו. מִלִּתֵּן לוֹ מַעֲשֵׂר רִאשׁוֹן: כי אין לו חלק ונחלה עמך. יָצְאוּ לֶקֶט שִׁכְחָה וּפֵאָה וְהֶפְקֵר, שֶׁאַף הוּא יֵשׁ לוֹ חֵלֶק עִמְּךָ בָּהֶן כָּמוֹךָ, וְאֵינָן חַיָּבִין בְּמַעֲשֵׂר:
מקצה שלש שנים. בָּא וְלִמֵּד שֶׁאִם הִשְׁהָה מַעַשְׂרוֹתָיו שֶׁל שָׁנָה רִאשׁוֹנָה וּשְׁנִיָּה לִשְׁמִטָּה, שֶׁיְּבַעֲרֵם מִן הַבַּיִת בַּשְּׁלִישִׁית:
ובא הלוי. וְיִטֹּל מַעֲשֵׂר רִאשׁוֹן: והגר והיתום. וְיִטְּלוּ מַעֲשֵׂר שֵׁנִי, שֶׁהוּא שֶׁל עָנִי שֶׁל שָׁנָה זוֹ, וְלֹא תֹאכְלֶנּוּ אַתָּה בִּירוּשָׁלַיִם כְּדֶרֶךְ שֶׁנִּזְקַקְתָּ לֶאֱכֹל מַעֲשֵׂר שֵׁנִי שֶׁל שְׁתֵּי שָׁנִים: ואכלו ושבעו. תֵּן לָהֶם כְּדֵי שָׂבְעָן, מִכָּאן אָמְרוּ אֵין פּוֹחֲתִין לֶעָנִי בַּגֹּרֶן וְכוּ' (ספרי קי). וְאַתָּה הוֹלֵךְ לִירוּשָׁלַיִם בַּמַּעֲשֵׂר שֶׁל שָׁנָה רִאשׁוֹנָה וּשְׁנִיָּה שֶׁהִשְׁהֵיתָ, וּמִתְוַדֶּה "בִּעַרְתִּי הַקֹּדֶשׁ מִן הַבַּיִת" כְּמוֹ שֶׁמְּפֹרָשׁ בְּכִי תְכַלֶּה לַעְשֵׂר (דברים כ"ו):
