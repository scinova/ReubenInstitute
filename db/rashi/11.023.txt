

עַל הָעַמּוּד. מְקוֹם מַעֲמַד הַמֶּלֶךְ בְּבֵית הַמִּקְדָּשׁ.
וְאֶת כֹּהֲנֵי הַמִּשְׁנֶה. סְגָנֵי כְּהֻנָּה, שֶׁהֵם שְׁנִיִּים לַכֹּהֲנִים גְּדוֹלִים. שֹׁמְרֵי הַסַּף. אֲמַרְכָּלִים הַמְמֻנִּים עַל כָּל צָרְכֵי הַבַּיִת, וּמַפְתְּחוֹת הָעֲזָרָה בְּיָדָם. בְּשַׁדְמוֹת קִדְרוֹן. (תַּרְגּוּם:) בְּמֵישָׁר קִדְרוֹן. וְנָשָׂא אֶת עֲפָרָם בֵּית אֵל. אֶל מָקוֹם טָמֵא, אֲשֶׁר טִמֵּא שָׁם אֶת הַמִּזְבֵּחַ שֶׁעָשָׂה יָרָבְעָם, וְשָׂרַף עָלָיו אֶת כֹּהֲנֵי הַבָּמוֹת.
וַיְקַטֵּר בַּבָּמוֹת. וְאוֹתָן שֶׁקִּטְּרוּ בַּבָּמוֹת בְּעָרֵי יְהוּדָה.

אֶת בָּתֵּי הַקְּדֵשִׁים. בָּתֵּי הַזִּמָּה. אֹרְגוֹת שָׁם בָּתִּים. יְרִיעוֹת.

לֹא יַעֲלוּ. הַכֹּהֲנִים שֶׁקִּטְּרוּ בַּבָּמוֹת לַעֲבוֹדָה זָרָה, לֹא יַעֲלוּ עוֹד לְשָׁרֵת עַל מִזְבַּח ה'. כִּי אִם אָכְלוּ מַצּוֹת. כְּלוֹמַר כִּי אִם בְּדָבָר זֶה לֹא יָצְאוּ מִכְּהֻנָּתָם, שֶׁמֻּתָּרִין לֶאֱכֹל בַּקֳּדָשִׁים, הֲרֵי הֵם כְּבַעֲלֵי מוּמִין, חוֹלְקִין וְאוֹכְלִין אֲבָל לֹא מַקְרִיבִין. אָכְלוּ מַצּוֹת. מְנָחוֹת, וְהוּא הַדִּין לְכָל הַקֳּדָשִׁים.
הַתֹּפֶת. הוּא הַמֹּלֶךְ, וְעַל שֵׁם שֶׁהָיוּ הַכֹּהֲנִים מַקִּישִׁים בְּתֻפִּים, שֶׁלֹּא יִשְׁמַע הָאָב אֶת נַהֲמַת הַבֵּן כְּשֶׁהוּא נִכְוָה בִּידֵי הַצּוּרָה שֶל עֲבוֹדָה זָרָה מֹלֶךְ, קְרָאוּהוּ תֹּפֶת.
אֶת הַסּוּסִים. אוֹתָם הַמִּשְׁתַּחֲוִים לַחַמָּה, הָיוּ לָהֶם סוּסִים מוּכָנִים לָצֵאת לִקְרַאת הַחַמָּה בַּבֹּקֶר. נְתַן מֶלֶךְ. כָּךְ שְׁמוֹ. אֲשֶׁר בַּפַּרְוָרִים. לֹא יָדַעְתִּי מָה הֵם.
וַיָּרָץ מִשָּׁם. וּבִעֲרָם מִשָּׁם.
לְהַר הַמַּשְׁחִית. הַר הַזֵּיתִים, הוּא הַר הַמִּשְׁחָה, וְכֵיוָן שֶׁהִזְכִּיר אֶת שְׁמוֹ עַל יְדֵי עֲבוֹדָה זָרָה, שִׁנָּה שְׁמוֹ לִגְנַאי.



מַה הַצִּיּוּן הַלָּז. רָאָה קֶבֶר, מִצִּדּוֹ אֶחָד עָלָה קִמְּשׁוֹנִים וַחֲרוּלִים, וְצִדּוֹ אֶחָד הֲדַסִּים וּבְשָׂמִים, וְתָמַהּ עָלָיו וְאָמַר מַה זֶּה. וְהֵשִׁיבוּ: הַקֶּבֶר אִישׁ הָאֱלֹהִים. וּנְבִיא הַשֶּׁקֶר שֶׁצִּוָּה לְבָנָיו שֶׁיִּקְבְּרוּהוּ אֶצְלוֹ.

אֲשֶׁר בְּעָרֵי שֹׁמְרוֹן. שֶׁהָיָה יֹאשִׁיָּהוּ מֶלֶךְ עַל כָּל יִשְׂרָאֵל, מַה שֶּׁחָזְרוּ מֵעֲשֶׂרֶת הַשְּׁבָטִים עַל יְדֵי יִרְמְיָהוּ.


לֹא נַעֲשָׂה כַּפֶּסַח הַזֶּה. לֹא נִתְקַבְּצוּ רֹב עַם בְּכָל פֶּסַח כְּמוֹ שֶׁנִּתְקַבְּצוּ בָּזֶה. מִימֵי הַשֹּׁפְטִים. מִימֵי שְׁמוּאֵל, כָּתוּב בְּדִבְרֵי הַיָּמִים (דה"ב לה יח), לְפִי שֶׁאַף שְׁמוּאֵל הֶחֱזִיר יִשְׂרָאֵל לְמוּטָב, וְקִבְּצָם הַמִּצְפָּתָה. דָּבָר אַחֵר: לֹא נַעֲשָׂה כַּפֶּסַח הַזֶּה בִּירוּשָׁלַיִם, (אֲבָל) מִשֶּׁמָּלְכוּ מַלְכֵי יִשְׂרָאֵל וּמַלְכֵי יְהוּדָה, לֹא עָשׂוּ פֶּסַח כָּל יִשְׂרָאֵל בִּירוּשָׁלַיִם, לְפִי שֶׁנֶּחְלְקָה הַמַּלְכוּת לִשְׁנַיִם מִיָּרָבְעָם, וְהָיוּ הוֹלְכִים לָעֵגֶל שֶׁבְּבֵית אֵל וּבְדָן, עַד עַכְשָׁיו שֶׁגָּלוּ עֲשֶׂרֶת הַשְּׁבָטִים, וְהֶחֱזִירָם יִרְמְיָהוּ וּמָלַךְ עֲלֵיהֶם יֹאשִׁיָּהוּ, וּבָאוּ כֻּלָּם לִירוּשָׁלַיִם.

הַתְּרָפִים. צוּרוֹת שֶׁמְּדַבְּרִים עַל יְדֵי כְשָׁפִים, וְהָעוֹשֶׂה אוֹתָן צָרִיךְ לְכַוֵּן שָׁעָה אַחַת בְּשָׁנָה, וְשָׁנָה אַחַת רְאוּיָה לְכָךְ.





וַיִּמְשְׁחוּ. לֹא הָיָה צָרִיךְ מְשִׁיחָה, אֶלָּא מִפְּנֵי שֶׁהָיָה יְהוֹיָקִים אָחִיו גָּדוֹל מִמֶּנּוּ שְׁתֵּי שָׁנִים, כְּמוֹ שֶׁכָּתוּב בָּעִנְיָן.




אִישׁ כְּעֶרְכּוֹ. כְּפִי עָשְׁרוֹ.
