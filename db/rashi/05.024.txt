כי מצא בה ערות דבר. מִצְוָה עָלָיו לְגָרְשָׁהּ, שֶׁלֹּא תִמְצָא חֵן בְּעֵינָיו (גיט' צ'):
לאיש אחר. אֵין זֶה בֶּן זוּגוֹ שֶׁל רִאשׁוֹן, הוּא הוֹצִיא רְשָׁעָה מִתּוֹךְ בֵּיתוֹ וְזֶה הִכְנִיסָהּ (שם):
ושנאה האיש האחרון. הַכָּתוּב מְבַשְּׂרוֹ שֶׁסּוֹפוֹ לִשְׂנֹאתָהּ, וְאִם לָאו קוֹבַרְתּוֹ, שֶׁנֶּאֱמַר או כי ימות (ספרי):
אחרי אשר הטמאה. לְרַבּוֹת סוֹטָה שֶׁנִּסְתְּרָה (ספרי; יבמות י"א):
אשה חדשה. שֶׁהִיא חֲדָשָׁה לוֹ וַאֲפִלּוּ אַלְמָנָה, פְּרָט לְמַחֲזִיר גְּרוּשָׁתוֹ (ספרי; סוטה מ"ד): ולא יעבור עליו. דְּבַר הַצָּבָא: לכל דבר שֶׁהוּא צֹרֶךְ הַצָּבָא, לֹא לְסַפֵּק מַיִם וּמָזוֹן וְלֹא לְתַקֵּן הַדְּרָכִים, אֲבָל הַחוֹזְרִים מֵעוֹרְכֵי הַמִּלְחָמָה עַל פִּי כֹהֵן, כְּגוֹן בָּנָה בַיִת וְלֹא חֲנָכוֹ אוֹ אֵרַשׂ אִשָּׁה וְלֹא לְקָחָהּ, מַסְפִּיקִין מַיִם וּמָזוֹן וּמְתַקְּנִין אֶת הַדְּרָכִים (שם): יהיה לביתו. אַף בִּשְׁבִיל בֵּיתוֹ, אִם בָּנָה בַיִת וַחֲנָכוֹ וְאִם נָטַע כֶּרֶם וְחִלְּלוֹ אֵינוֹ זָז מִבֵּיתוֹ בִּשְׁבִיל צָרְכֵי הַמִּלְחָמָה: יהיה. לְרַבּוֹת אֶת כַּרְמוֹ: לביתו. זֶה בֵיתוֹ: ושמח. יְשַׂמֵּחַ אֶת אִשְׁתּוֹ, וְתַרְגּוּמוֹ "וְיַחְדֵי יָת אִתְּתֵהּ", וְהַמְתַרְגֵּם וְיֶחְדֵּי עִם אִתְּתֵהּ טוֹעֶה הוּא, שֶׁאֵין זֶה תַּרְגּוּם שֶׁל וְשִׂמַּח אֶלָּא שֶׁל וְשָׂמַח:
לא יחבל. אִם בָּא לְמַשְׁכְּנוֹ עַל חוֹבוֹ בְּבֵית דִּין, לֹא יְמַשְׁכְּנֶנּוּ בִּדְבָרִים שֶׁעוֹשִׂים בָּהֶן אֹכֶל נֶפֶשׁ (ב"מ קי"ג): רחים. הִיא הַתַּחְתּוֹנָה: ורכב. הִיא הָעֶלְיוֹנָה:
כי ימצא. בְּעֵדִים וְהַתְרָאָה, וְכֵן כָּל "יִמָּצֵא" שֶׁבַּתּוֹרָה: והתעמר בו. אֵינוֹ חַיָּב עַד שֶׁיִּשְׁתַּמֵּשׁ בּוֹ:
השמר בנגע הצרעת. שֶׁלֹּא תִתְלֹשׁ סִימָנֵי טֻמְאָה וְלֹא תָקֹץ אֶת הַבַּהֶרֶת (מכות כ"ב): ככל אשר יורו אתכם. אִם לְהַסְגִּיר אִם לְהַחֲלִיט אִם לְטַהֵר:
זכור את אשר עשה ה' אלהיך למרים. אִם בָּאתָ לְהִזָּהֵר שֶׁלֹּא תִלְקֶה בְּצָרַעַת אַל תְּסַפֵּר לָשׁוֹן הָרָע, זְכֹר הֶעָשׂוּי לְמִרְיָם שֶׁדִּבְּרָה בְאָחִיהָ וְלָקְתָה בִנְגָעִים (עי' ספרי):
כי תשה ברעך. תָחֹב בַּחֲבֵרְךָ: משאת מאומה. חוֹב שֶׁל כְּלוּם:

לא תשכב בעבטו. לֹא תִשְׁכַּב וַעֲבוֹטוֹ אֶצְלְךָ (ספרי; בבא מציעא קי"ד):
כבוא השמש. אִם כְּסוּת לַיְלָה הוּא, וְאִם כְּסוּת יוֹם — הַחֲזִירֵהוּ בַבֹּקֶר, וּכְבָר כָּתוּב בִּוְאֵלֶּה הַמִּשְׁפָּטִים (כ"ב) "עַד בֹּא הַשֶּׁמֶשׁ תְּשִׁיבֶנּוּ לוֹ" — כָּל הַיּוֹם תְּשִׁיבֶנּוּ לוֹ, וּכְבֹא הַשֶּׁמֶשׁ תִּקָּחֶנּוּ: וברכך. וְאִם אֵינוֹ מְבָרֶכְךָ, מִכָּל מָקוֹם ולך תהיה צדקה (ספרי):
לא תעשק שכיר. וַהֲלֹא כְבָר כָּתוּב? אֶלָּא לַעֲבֹר עַל הָאֶבְיוֹן בִּשְׁנֵי לָאוין, "לֹא תַעֲשֹׁק שְׂכַר שָׂכִיר" שֶׁהוּא עָנִי וְאֶבְיוֹן, וְעַל הֶעָשִׁיר כְּבָר הֻזְהַר (ויקרא י"ט) "לֹא תַעֲשֹׁק אֶת רֵעֲךָ" (ב"מ ס"א): אביון. הַתָּאֵב לְכָל דָּבָר: מגרך. זֶה גֵּר צֶדֶק: בשעריך. זֶה גֵּר תּוֹשָׁב הָאוֹכֵל נְבֵלוֹת: אשר בארצך. לְרַבּוֹת שְׂכַר בְּהֵמָה וְכֵלִים (בבא מציעא קי"א):
ואליו הוא נשא את נפשו. אֶל הַשָּׂכָר הַזֶּה הוּא נוֹשֵׂא אֶת נַפְשׁוֹ לָמוּת, עָלָה בַכֶּבֶשׁ וְנִתְלָה בְאִילָן (ספרי; ב"מ קי"ב): והיה בך חטא. מִכָּל מָקוֹם, אֶלָּא שֶׁמְּמַהֲרִין לִפָּרֵעַ עַל יְדֵי הַקּוֹרֵא (ספרי):
לא יומתו אבות על בנים. בְּעֵדוּת בָּנִים, וְאִם תֹּאמַר בַּעֲוֹן בָּנִים, כְּבָר נֶאֱמַר איש בחטאו יומתו (ספרי; סנהדרין כ"ז): איש בחטאו יומתו. אֲבָל מִי שֶׁאֵינוֹ אִישׁ מֵת בַּעֲוֹן אָבִיו, וְהַקְּטַנִּים מֵתִים בַּעֲוֹן אֲבוֹתָם בִּידֵי שָׁמַיִם (ספרי):
לא תטה משפט גר יתום. וְעַל הֶעָשִׁיר כְּבָר הֻזְהַר (דברים ט"ז) "לֹא תַטֶּה מִשְׁפָּט", וְשָׁנָה בֶעָנִי לַעֲבֹר עָלָיו בִּשְׁנֵי לָאוין, לְפִי שֶׁנָּקֵל לְהַטּוֹת מִשְׁפַּט עָנִי יוֹתֵר מִשֶּׁל עָשִׁיר, לְכָךְ הִזְהִיר וְשָׁנָה עָלָיו (ספרי): ולא תחבל. שֶׁלֹּא בִשְׁעַת הַלְוָאָה:
וזכרת. עַל מְנָת כֵּן פְּדִיתִיךָ, לִשְׁמֹר חֻקּוֹתַי אֲפִלּוּ יֵשׁ חֶסְרוֹן כִּיס בַּדָּבָר:
ושכחת עמר. וְלֹא גָדִישׁ, מִכָּאן אָמְרוּ, עֹמֶר שֶׁיֵּשׁ בּוֹ סָאתַיִם וּשְׁכָחוֹ אֵינוֹ שִׁכְחָה (פאה ו' מ"ב): בשדה. לְרַבּוֹת שִׁכְחַת קָמָה שֶׁשָּׁכַח מִקְצָתָהּ מִלִּקְצֹר: לא תשוב לקחתו. מִכָּאן אָמְרוּ, שֶׁלְּאַחֲרָיו שִׁכְחָה, שֶׁלְּפָנָיו אֵינוֹ שִׁכְחָה, שֶׁאֵינוֹ בְּבַל תָּשׁוּב (פאה מ"ד; בבא מציעא י"א): למען יברכך. וְאַעַ"פִּ שֶׁבָּאת לְיָדוֹ שֶׁלֹּא בְמִתְכַּוֵּן, קַ"וָ לָעוֹשֶׂה בְמִתְכַּוֵּן. אֱמֹר מֵעַתָּה נָפְלָה סֶלַע מִיָּדוֹ וּמְצָאָהּ עָנִי וְנִתְפַּרְנֵס בָּהּ — הֲרֵי הוּא מִתְבָּרֵךְ עָלֶיהָ (ספרי):
לא תפאר. לֹא תִטֹּל תִּפְאַרְתּוֹ מִמֶּנּוּ (חולין קל"א), מִכָּאן שֶׁמַּנִּיחִין פֵּאָה לָאִילָן: אחריך. זוֹ שִׁכְחָה:
לא תעולל. אִם מָצָאתָ בוֹ עוֹלֵלוֹת, לֹא תִקָּחֶנָּה. וְאֵי זוֹ הִיא עוֹלֵלוֹת? כָּל שֶׁאֵין לָהּ לֹא כָּתֵף וְלֹא נָטֵף. יֵשׁ לָהּ אֶחָד מֵהֶם — הֲרֵי הִיא לְבַעַל הַבַּיִת (ספרא); וְרָאִיתִי בְתַלְמוּד יְרוּשַׁלְמִי אֵי זוֹ הִיא כָּתֵף? פְּסִיגִין זֶה עַל גַּב זֶה, נָטֵף — אֵלּוּ הַתְּלוּיוֹת בַּשִּׁדְרָה וְיוֹרְדוֹת:
