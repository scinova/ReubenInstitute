וְהָאָרֶץ נִשְׁאֲרָה הַרְבֵּה מְאֹד מִמַּה שֶּׁאָמַרְתִּי לְאַבְרָהָם נִשְׁאֲרָה לְרִשְׁתָּהּ, שֶׁלֹּא נִכְבְּשָׁה.
גְּלִילוֹת מרקא"ש בְּלַעַז (אֵזוֹרֵי הַגְבוּל, הַסְפַר).
מִן הַשִּׁיחוֹר הוּא נִילוּס, הוּא נַחַל מִצְרַיִם, הוּא סָמוּךְ לִתְחוּם מִקְצוֹעַ דְּרוֹמִית מַעֲרָבִית שֶׁל אֶרֶץ יִשְׂרָאֵל, כְּמוֹ שֶׁאָמַר בְּ'אֵלֶּה מַסְעֵי' (במדבר לד ה), וְלָמַדְנוּ כָּאן שֶׁלֹּא כָּבַשׁ יְהוֹשֻׁעַ בְּחַיָּיו כָּל מֶצֶר הַדְּרוֹמִי אֶלָּא מִמִּדְבַּר צִין, הוּא הָהָר הֶחָלָק הָעוֹלֶה שֵׂעִירָה עַד הַנִּילוּס, הוּא תְּחוּם עַזָּה הָאָמוּר לְמַעְלָה, מִקָּדֵשׁ בַּרְנֵע עַד עַזָּה. וְעַד גְּבוּל עֶקְרוֹן צָפוֹנָה הָעֶקְרוֹנִי יוֹשֵׁב עַל הַיָּם וּמִתְפַּשֵּׁט לְצַד הַצָּפוֹן יוֹתֵר מִן הָאֲחֵרִים. לַכְּנַעֲנִי תֵּחָשֵׁב מֵאֶרֶץ הַכְּנַעֲנִי שֶׁנָּתַתִּי לְאַבְרָהָם הוּא. חֲמֵשֶׁת סַרְנֵי פְּלִשְׁתִּים וְגוֹ' הִיא סוֹף גְּמַר הַמֶּצֶר עַד הַיָּם הַגָּדוֹל שֶׁבַּמַּעֲרָב. הָעַזָּתִי וְהָאַשְׁדּוֹדִי הָאֶשְׁקְלוֹנִי וְגוֹ' שִׁשָּׁה סְרָנִים הוּא מוֹנֶה, וּמִתְּחִלָתָן קְרָאָן 'חֲמֵשֶׁת'. אָמַר רַבִּי יוֹנָתָן (חולין ס ב), אִידוֹנְרִיקֵי שֶׁלָּהֶם חַמִשָּׁה; הַחֲשׁוּבִים חֲמִשָּׁה הָיוּ, שֶׁהָעַוִּים אֵינוֹ מוֹנֶה מִן הַסְּרָנִים הַחֲשׁוּבִים. וְעוֹד יֵשׁ לְפָרֵשׁ, חֲמֵשֶׁת סַרְנֵי פְּלִשְׁתִּים: הָעַזָּתִי, וְהָאַשְׁדּוֹדִי, הָאֶשְׁקְלוֹנִי, הַגִּתִּי וְהָעֶקְרוֹנִי הֵם חֲמִשָּׁה, וְעוֹד נִשְׁאַר לִכְבּוֹשׁ אֶרֶץ הָעַוִּים שֶׁאֵינָם מִן הַפְּלִשְׁתִּים.
מִתֵּימָן כָּל אֶרֶץ הַכְּנַעֲנִי פָּסוּק רִאשׁוֹן מָנָה מִצַּד הַדְּרוֹמִי מִן הַמִּזְרָח לַמַּעֲרָב, וְזֶה מוֹנֶה אֶת רוֹחַב אוֹתוֹ הַמִּקְצוֹעַ הַנּוֹתָר לִיכָּבֵשׁ, כַּמָּה הוּא מִן הַדָּרוֹם לַצָּפוֹן. מִתֵּימָן כָּל אֶרֶץ הַכְּנַעֲנִי מִתֵּימָן זוֹ מְעָרָה לַצִּידוֹנִים, עַד אֲפֵקָה עַד הָאֱמוֹרִי.
וְהָאָרֶץ הַגִּבְלִי וְכָל הַלְּבָנוֹן מִזְרַח הַשֶּׁמֶשׁ וּבַמֶּצֶר מִזְרָחִי נִשְׁאָר לִכְבּוֹשׁ בְּמִקְצוֹעַ הַצָּפוֹן כָּל הַלְּבָנוֹן, מִבַּעַל גָּד עַד סוֹף הַמֶּצֶר, וּבָרוֹחַב מִן הַמִּזְרָח לַמַּעֲרָב עַד לְבוֹא חֲמָת, זֶהוּ כָּל מֶצֶר הַצְּפוֹנִי שֶׁלְּבוֹא חֲמָת בְּמִקְצוֹעַ צְפוֹנִית מַעֲרָבִית בְּ'אֵלֶּה מַסְעֵי' (במדבר לד ח).
אָנֹכִי אוֹרִישֵׁם אַחֲרֵי מוֹתְךָ. הַפִּלֶהָ לְיִשְׂרָאֵל בְּנַחֲלָה וְיִכְבּוֹשׁ לְאַחַר זְמָן כָּל שֵׁבֶט וְשֵׁבֶט מַה שֶּׁנָּפַל בְּגוֹרָלוֹ.

עִמּוֹ הָראוּבֵנִי וְהַגָּדִי עִם חֲצִי הָרִאשׁוֹן שֶׁל מְנַשֶּׁה, הָראוּבֵנִי וְהַגָּדִי לָקְחוּ נַחֲלָתָם.
מֵעֲרוֹעֵר אֲשֶׁר עַל שְׂפַת וְגוֹ' מוֹנֶה וְהוֹלֵךְ כָּל אֶרֶץ עֵבֶר הַיַּרְדֵּן, וְאַחַר כָּךְ מְפָרֵשׁ גְּבוּל כָּל שֵׁבֶט וְשֵׁבֶט וְיִתֵּן משֶׁה לְמַטֶּה פְּלוֹנִי וּפְלוֹנִי.


מִיֶּתֶר הָרְפָאִים שֶׁהָרַג כְּדָרְלָעֹמֶר וְהַמְּלָכִים אֲשֶׁר אִתּוֹ, כְּמָה שֶׁנֶּאֱמַר (בראשית יד ה): וַיַּכּוּ אֶת רְפָאִים בְּעַשְׁתְּרוֹת קַרְנַיִם.






בְּהַר הָעֵמֶק (תַּרְגּוּם:) בְּטוּרָא דְמֵישָׁרָא.





וַחֲצִי אֶרֶץ בְּנֵי עַמּוֹן חֲצִי מַה שֶּׁכָּבְשׁוּ מֵאֶרֶץ בְּנֵי עַמּוֹן וּמִסִּיחוֹן.

הַיַּרְדֵּן וּגְבֻל יַרְדְּנָא וּתְחוּמֶיהָ הֶעָרִים עַל שְׂפָתוֹ.
הֶעָרִים וְחַצְרֵיהֶם הֶעָרִים הַמּוּקָּפוֹת חוֹמָה. וְחַצְרֵיהֶם עָרֵי הַפְּרָזִי בְּלֹא חוֹמָה.
