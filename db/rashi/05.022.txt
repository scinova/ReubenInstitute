והתעלמת. כּוֹבֵשׁ עַיִן כְּאִלּוּ אֵינוֹ רוֹאֵהוּ: לא תראה … והתעלמת. לֹא תִרְאֶה אוֹתוֹ שֶׁתִּתְעַלֵּם מִמֶּנּוּ, זֶהוּ פְּשׁוּטוֹ. וְרַבּוֹתֵינוּ אָמְרוּ פְּעָמִים שֶׁאַתָּה מִתְעַלֵּם וְכוּ' (ספרי; בבא מציעא ל'):
עד דרש אחיך. וְכִי תַעֲלֶה עַל דַּעְתְּךָ שֶׁיִּתְּנֵהוּ לוֹ קֹדֶם שֶׁיִּדְרְשֵׁהוּ? אֶלָּא דָּרְשֵׁהוּ שֶׁלֹּא יְהֵא רַמַּאי (בבא מציעא כ"ז; עי' ספרי): והשבתו לו. שֶׁתְּהֵא בוֹ הֲשָׁבָה, שֶׁלֹּא יֹאכַל בְּבֵיתְךָ כְּדֵי דָּמָיו וְתִתְבָּעֵם מִמֶּנּוּ. מִכָּאן אָמְרוּ, כָּל דָּבָר שֶׁעוֹשֶׂה וְאוֹכֵל יַעֲשֶׂה וְיֹאכַל וְשֶׁאֵינוֹ עוֹשֶׂה וְאוֹכֵל יִמָּכֵר (בבא מציעא כ"ח):
לא תוכל להתעלם. לִכְבֹּשׁ עֵינְךָ כְּאִלּוּ אֵינְךָ רוֹאֶה אוֹתוֹ:
הקם תקים. זוֹ טְעִינָה – לְהַטְעִין מַשּׂאוֹי שֶׁנָּפַל מֵעָלָיו: עמו. עִם בְּעָלָיו, אֲבָל אִם הָלַךְ וְיָשַׁב לוֹ וְאָמַר לוֹ הוֹאִיל וְעָלֶיךָ מִצְוָה אִם רָצִיתָ לִטְעֹן טְעֹן, פָּטוּר (שם ל"ב):
לא יהיה כלי גבר על אשה. שֶׁתְּהֵא דּוֹמָה לְאִישׁ, כְּדֵי שֶׁתֵּלֵךְ בֵּין הָאֲנָשִׁים, שֶׁאֵין זוֹ אֶלָּא לְשֵׁם נִאוּף (עי' ספרי): ולא ילבש גבר שמלת אשה. לֵילֵךְ לֵישֵׁב בֵּין הַנָּשִׁים. דָּ"אַ — שֶׁלֹּא יַשִּׁיר שְׂעַר הָעֶרְוָה וְשֵׂעָר שֶׁל בֵּית הַשֶּׁחִי (נזיר נ"ט): כי תועבת. לֹא אָסְרָה תוֹרָה אֶלָּא לְבוּשׁ הַמֵּבִיא לִידֵי תוֹעֵבָה (עי' ספרי):
כי יקרא. פְּרָט לִמְזֻמָּן (חולין קל"ט): לא תקח האם. בְּעוֹדָהּ עַל בָּנֶיהָ:
למען ייטב לך וגו'. אִם מִצְוָה קַלָּה שֶׁאֵין בָּהּ חֶסְרוֹן כִּיס אָמְרָה תוֹרָה "לְמַעַן יִיטַב לָךְ וְהַאֲרַכְתָּ יָמִים", קַל וָחֹמֶר לְמַתַּן שְׂכָרָן שֶׁל מִצְווֹת חֲמוּרוֹת (שם קמ"ב):
כי תבנה בית חדש. אִם קִיַּמְתָּ מִצְוַת שִׁלּוּחַ הַקֵּן, סוֹפְךָ לִבְנוֹת בַּיִת חָדָשׁ, וּתְקַיֵּם מִצְוַת מַעֲקֶה, שֶׁמִּצְוָה גּוֹרֶרֶת מִצְוָה, וְתַגִּיעַ לְכֶרֶם וְשָׂדֶה וְלִבְגָדִים נָאִים, לְכָךְ נִסְמְכוּ פָּרָשִׁיּוֹת הַלָּלוּ (תנחומא): מעקה. גָּדֵר סָבִיב לַגַּג, וְאֻנְקְלוֹס תִּרְגֵּם "תְּיָקָא", כְּגוֹן תִּיק שֶׁמְּשַׁמֵּר מַה שֶּׁבְּתוֹכוֹ: כי יפל הנפל. רָאוּי זֶה לִפֹּל, וְאַעַ"פִּ כֵן לֹא תִתְגַּלְגֵּל מִיתָתוֹ עַל יָדְךָ, שֶׁמְּגַלְגְּלִין זְכוּת עַל יְדֵי זַכַּאי וְחוֹבָה עַל יְדֵי חַיָּב (ספרי):
כלאים. חִטָּה וּשְׂעוֹרָה וְחַרְצָן בְּמַפֹּלֶת יָד (ברכות כ"ב): פן תקדש. כְּתַרְגוּמוֹ "תִּסְתָּאַב", כָּל דָּבָר הַנִּתְעָב עַל הָאָדָם, בֵּין לְשֶׁבַח כְּגוֹן הֶקְדֵּשׁ, בֵּין לִגְנַאי כְּגוֹן אִסּוּר, נוֹפֵל בּוֹ לְשׁוֹן קִדּוּשׁ, כְּמוֹ "אַל תִּגַּשׁ בִּי כִּי קִדַּשְׁתִּיךָ" (ישעיהו ס"ה): ( המלאה. זֶה מִלּוּי וְתוֹסֶפֶת שֶׁהַזֶּרַע מוֹסִיף):
לא תחרש בשור ובחמור. הוּא הַדִּין לְכָל שְׁנֵי מִינִים שֶׁבָּעוֹלָם (בבא קמא נ"ד), הוּא הַדִּין לְהַנְהִיגָם יַחַד קְשׁוּרִים זוּגִים בְּהוֹלָכַת שׁוּם מַשָּׂא (כלאים פ"ח מ"ב):
שעטנז. לְשׁוֹן עֵרוּב. וְרַבּוֹתֵינוּ פֵּרְשׁוּ — שׁוּעַ טָווּי וְנוּז (ספרי):
גדלים תעשה לך. אַף מִן הַכִּלְאַיִם, לְכָךְ סְמָכָן הַכָּתוּב (יבמות ד'):
ובא אליה ושנאה. סוֹפוֹ,
ושם לה עלילת דברים – עֲבֵרָה גּוֹרֶרֶת עֲבֵרָה, עָבַר עַל לֹא תִשְׂנָא, סוֹפוֹ לָבֹא לִידֵי לָשׁוֹן הָרָע (עי' ספרי): את האשה הזאת. מִכָּאן שֶׁאֵין אוֹמֵר דָּבָר אֶלָּא בִּפְנֵי בַעַל דִּין (שם):
אבי הנערה ואמה. מִי שֶׁגִּדְּלוּ גִּדּוּלִים הָרָעִים יִתְבַּזּוּ עָלֶיהָ (עי' שם):
ואמר אבי הנערה. מְלַמֵּד שֶׁאֵין רְשׁוּת לָאִשָּׁה לְדַבֵּר בִּפְנֵי הָאִישׁ (שם):
ופרשו השמלה. הֲרֵי זֶה מָשָׁל, מְחֻוָּרִין הַדְּבָרִים כַּשִּׂמְלָה (שם; כתובות שם):
ויסרו אתו. מַלְקוּת (שם):

ואם אמת היה הדבר. בְּעֵדִים וְהַתְרָאָה, שֶׁזִּנְּתָה לְאַחַר אֵרוּסִין (כתובות מ"ד):
אל פתח בית אביה. רְאוּ גִּדּוּלִים שֶׁגִּדַּלְתֶּם (שם מ"ה): אנשי עירה. בְּמַעֲמַד כָּל אַנְשֵׁי עִירָהּ (ספרי): לזנות בית אביה. כְּמוֹ בְּבֵית אָבִיהָ:
ומתו גם שניהם. לְהוֹצִיא מַעֲשֵׂה חִדּוּדִים שֶׁאֵין הָאִשָּׁה נֶהֱנֵית מֵהֶם (ספרי; סנהדרין ס"ו): גם. לְרַבּוֹת הַבָּאִים אַחֲרֵיהֶם (ספרי); דָּבָר אַחֵר, גם שניהם לְרַבּוֹת אֶת הַוָּלָד, שֶׁאִם הָיְתָה מְעֻבֶּרֶת אֵין מַמְתִּינִין לָהּ עַד שֶׁתֵּלֵד (ערכין ז'):
ומצאה איש בעיר. לְפִיכָךְ שָׁכַב עִמָּהּ – פִּרְצָה קוֹרְאָה לַגַּנָּב, הָא אִלּוּ יָשְׁבָה בְּבֵיתָהּ, לֹא אֵרַע לָהּ (ספרי):


כי כאשר יקום וגו'. לְפִי פְשׁוּטוֹ זֶהוּ מַשְׁמָעוֹ: כִּי אֲנוּסָה הִיא וּבְחָזְקָה עָמַד עָלֶיהָ, כְּאָדָם הָעוֹמֵד עַל חֲבֵרוֹ לְהָרְגוֹ. וְרַבּוֹתֵינוּ דָּרְשׁוּ בוֹ הֲרֵי זֶה בָא לְלַמֵּד וְנִמְצָא לָמֵד וְכוּ' (סנהדרין ע"ג):
