בַּיָּמִים הָהֵם. שְׁלשָׁה יָמִים לִפְנֵי מַפַּלְתּוֹ שֶל סַנְחֵרִיב. חָלָה חִזְקִיָּהוּ. וְיוֹם שְׁלִישִׁי כְּשֶׁעָלָה בֵּית ה', יוֹם מַפַּלְתּוֹ שֶל סַנְחֵרִיב, וְיוֹם טוֹב שֶל פֶּסַח הָיָה. כִּי מֵת אַתָּה. בָּעוֹלָם הַזֶּה. וְלֹא תִחְיֶה. לָעוֹלָם הַבָּא, מִשּׁוּם דְּלָא נָסַבְתְּ אִשָּׁה, כִּדְאִיתָא בִּבְרָכוֹת.


לֹא יָצָא חָצֵר וְגוֹ'. מִהֵר הַקָּבָּ"ה לְהוֹדִיעַ רְפוּאָתוֹ, בְּטֶרֶם יֵצֵא קוֹל הֲבָרָה בָּעִיר שֶׁנִּקְנְסָה עָלָיו מִיתָה, כִּדְאִיתָא בִּירוּשַׁלְמִי.

וּמִכַּף מֶלֶךְ אַשּׁוּר אַצִּילְךָ. לָמַדְנוּ שֶׁלִּפְנֵי מַפָּלָתּוֹ שֶל סַנְחֵרִיב חָלָה.
דְּבֶלֶת תְּאֵנִים. כְּשֶׁהֵם לַחִין קְרוּיִין תְּאֵנִים, וּכְשֶׁנִּדְרָסִים בְּעִגּוּל קְרוּיִין דְּבֵלָה. וַיָּשִׂימוּ עַל הַשְּׁחִין וַיֶּחִי. וְנֵס בְּתוֹךְ נֵס, שֶׁהֲרֵי אַף בָּשָׂר חַי כְּשֶׁאַתָּה נוֹתֵן עָלָיו דְּבֵלָה, נִסְרָח, וְהַקָּבָּ"ה נוֹתֵן דָּבָר הַמְחַבֵּל לְתוֹךְ דָּבָר שֶׁנִּתְחַבֵּל, וּמִתְרַפֵּא.

הָלַךְ הַצֵּל עֶשֶׂר מַעֲלוֹת אִם יָשׁוּב עֶשֶׂר מַעֲלוֹת. הֲרֵי זֶה מִקְרָא קָצָר, כְּבָר יָרַד הַצֵּל לְמַעֲלוֹת הָעֲשׂוּיוֹת לִשְׁעוֹת הַיּוֹם, כְּמוֹ אורלויי"ן בְּלַעַ"ז (שָׁעוֹן), שֶׁעוֹשִׂין הָאֲנָשִׁים לִבְחֹן שְׁעוֹת הַיּוֹם, חָפֵץ אַתָּה שֶׁיַּמְתִּין בְּמָקוֹם שֶׁהוּא שָׁם כְּדֵי מַהֲלַךְ עֶשֶׂר שָׁעוֹת, אוֹ יָשׁוּב לְמַעְלָה עֶשֶׂר מַעֲלוֹת, וִיהֵא נִרְאֶה כְּאִלּוּ הוּא שַׁחֲרִית.
וַיֹּאמֶר יְחִזְקִיָּהוּ נָקֵל לַצֵּל לִנְטוֹת עֶשֶׂר מַעֲלוֹת. דָּבָר קַל הוּא לִנְטוֹת בִּמְקוֹמוֹ, וְנוֹטֶה וְעוֹמֵד עֶשֶׂר מַעֲלוֹת אֵינוֹ דָּבָר הַנִּכָּר כָּל כָּךְ, לֹא כֵן אֶלָּא יָשׁוּב הַצֵּל, וְיִהְיֶה הַדָּבָר תֵּמַהּ, שֶׁרָאוּ שֶׁפָּנָה הַיּוֹם לַעֲרֹב, וְחוֹזֵר לִהְיוֹת בֹּקֶר.
אֲשֶׁר יָרְדָה בְּמַעֲלוֹת אָחָז. יוֹם שֶׁמֵּת אָחָז, נִתְקַצֵּר הַיּוֹם וּמִהֵר הַצֵּל לֵירֵד עֶשֶׂר מַעֲלוֹת, כְּדֵי שֶׁלֹּא יְהֵא נִסְפָּד, וְאוֹתָן עֶשֶׂר שָׁעוֹת חָזְרוּ עַכְשָׁיו לְחִזְקִיָּהוּ.
כִּי שָׁמַע כִּי חָלָה. וְעַד אוֹתוֹ הַיּוֹם לֹא הָיָה אָדָם חוֹלֶה וְחָיָה, וְאֵיךְ שָׁמַע, לָמוּד הָיָה לֶאֱכֹל בְּשָׁלשׁ שָׁעוֹת בַּיּוֹם, וְיָשַׁן עַד תֵּשַׁע, כֵּיוָן שֶׁחָזַר גַּלְגַּל חַמָּה לְחִזְקִיָּהוּ, נֵעוֹר בְּתֶשַׁע שָׁעוֹת וּמָצָא שֶׁהוּא שַׁחֲרִית, בִּקֵּשׁ לַהֲרֹג אֶת עֲבָדָיו, אָמַר: אַתֶּם הִנַּחְתֶּם אוֹתִי לִישֹׁן יוֹם וָלַיְלָה, אָמְרוּ: גַּלְגַּל חַמָּה חָזַר, אָמַר לָהֶם: מִי הֶחֱזִירוֹ, אָמְרוּ לוֹ: אֱלֹהָיו שֶל חִזְקִיָּהוּ, כִּדְאִיתָא בִּפְסִיקְתָּא.
בֵּית נְכֹתֹה. בֵּית גִּנְזֵי בְּשָׂמָיו, כְּמוֹ: נְכֹאת וּצְרִי וָלֹט. וְאֵת שֶׁמֶן הַטּוֹב. יֵשׁ פּוֹתְרִין שֶׁמֶן הַמִּשְׁחָה, וְיֵשׁ פּוֹתְרִים שֶׁמֶן אֲפַרְסְמוֹן (בלסמון), שֶׁהוּא מָצוּי בְּאֶרֶץ יִשְׂרָאֵל, שֶׁנֶּאֱמַר: יְהוּדָה וְיִשְׂרָאֵל הֵמָּה רֹכְלָיִךְ בְּחִטֵּי מִנִּית וּפַנַּג. וְרָאִיתִי בְסֵפֶר יוֹסִיפוּן: פַּנַּג הוּא אֲפַרְסְמוֹן, וְגָדֵל בִּירִיחוֹ, וְעַל שֵׁם הָרֵיחַ נִקְרָא יְרִיחוֹ. לֹא הָיָה דָבָר. אֶת הָאָרוֹן וְהַלּוּחוֹת וְסֵפֶר הַתּוֹרָה.
מֵאֶרֶץ רְחוֹקָה בָּאוּ. זֶה אֶחָד מִשְּׁלשָׁה שֶׁבְּדָקָן הַקָּבָּ"ה וּמְצָאָן עָבִיט שֶל מַיִם עֲכוּרִים, קַיִן, יְחִזְקִיָּהוּ, וּבִלְעָם. קַיִן: לֹא יָדַעְתִּי הֲשׁוֹמֵר אָחִי אָנֹכִי, הָיָה לוֹ לוֹמַר רִבּוֹנוֹ שֶל עוֹלָם הֲלֹא כָל הַנִּסְתָּרוֹת גְּלוּיוֹת לְךָ, כִּדְאִיתָא בְּתַנְחוּמָא; בִּלְעָם, שֶׁאָמַר לוֹ הַמָּקוֹם: מִי הָאֲנָשִׁים הָאֵלֶּה עִמָּךְ, הָיָה לוֹ לוֹמַר ה' אֱלֹהִים אַתָּה יָדָעְתָּ, כְּמוֹ שֶׁנֶּאֱמַר בִּיחֶזְקֵאל, וְהוּא הֵשִׁיב בְּגַאֲוָה: בָּלָק בֶּן צִפּוֹר מֶלֶךְ מוֹאָב שָׁלַח אֵלַי, הַרְבֵּה מְבַקְּשִׁים יֵשׁ לִי; חִזְקִיָּהוּ, הָיָה לוֹ לְהָשִׁיב לִישַׁעְיָהוּ, אַתָּה נְבִיאוֹ שֶל מָקוֹם, וְאַתָּה שׁוֹאֵל אֵלַי, הִתְחִיל חִזְקִיָּה מִתְגָּאֶה וְאָמַר: מֵאֶרֶץ רְחוֹקָה בָּאוּ אֵלַי, לְכָךְ נֶעֱנַשׁ, וְעַל שֶׁשָּׂמַח עֲלֵיהֶם וְהֶאֱכִילָם עַל שֻׁלְחָנוֹ.


לֹא יִוָּתֵר דָּבָר. מִדָּה כְּנֶגֶד מִדָּה, לֹא הָיָה דָּבָר.
וּמִבָּנֶיךָ. חֲנַנְיָה מִישָׁאֵל וַעֲזַרְיָה.
טוֹב דְּבַר ה'. מֵאַחַר שֶׁבְּיָמַי יִהְיֶה שָׁלוֹם.
וַאֲשֶׁר עָשָׂה אֶת הַבְּרֵכָה וְגוֹ'. בְּדִבְרֵי הַיָּמִים.
