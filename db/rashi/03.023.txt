
דבר אל בני ישראל, מועדי ה'. עֲשֵׂה מוֹעֲדוֹת שֶׁיִּהְיוּ יִשְֹרָאֵל מְלֻמָּדִין בָּהֶם, שֶׁמְּעַבְּרִים אֶת הַשָּׁנָה עַל גָּלֻיּוֹת שֶׁנֶּעֶקְרוּ מִמְּקוֹמָן לַעֲלוֹת לָרֶגֶל וַעֲדַיִן לֹא הִגִּיעוּ לִירוּשָׁלַיִם (ספרא):
ששת ימים. מָה עִנְיַן שַׁבָּת אֵצֶל מוֹעֲדוֹת? לְלַמֶּדְךָ שֶׁכָּל הַמְחַלֵּל אֶת הַמּוֹעֲדוֹת מַעֲלִין עָלָיו כְּאִלּוּ חִלֵּל אֶת הַשַּׁבָּתוֹת, וְכָל הַמְקַיֵּם אֶת הַמּוֹעֲדוֹת מַעֲלִין עָלָיו כְּאִלּוּ קִיֵּם אֶת הַשַּׁבָּתוֹת (שם):
אלה מועדי ה'. לְמַעְלָה מְדַבֵּר בְּעִבּוּר שָׁנָה וְכָאן מְדַבֵּר בְּקִדּוּשׁ הַחֹדֶשׁ:
בין הערבים. מִשֵּׁשׁ שָׁעוֹת וּלְמַעְלָה (שם): פסח לה'. הַקְרָבַת קָרְבָּן שֶׁשְּׁמוֹ פֶּסַח:


והקרבתם אשה וגו'. הֵם הַמּוּסָפִין הָאֲמוּרִים בְּפָרָשַׁת פִּינְחָס, וְלָמָּה נֶאֶמְרוּ כָאן? לוֹמַר שֶׁאֵין הַמּוּסָפִין מְעַכְּבִין זֶה אֶת זֶה: והקרבתם אשה לה'. מִכָּל מָקוֹם — אִם אֵין פָּרִים הָבֵא אֵילִים, וְאִם אֵין פָּרִים וְאֵילִים הָבֵא כְבָשִׂים (שם): שבעת ימים. כָּל מָקוֹם שֶׁנֶּאֱמַר שִׁבְעַת שֵׁם דָּבָר הוּא — שָׁבוּעַ שֶׁל יָמִים, שטי"נא בְּלַעַז, וְכֵן כָּל לְשׁוֹן שְׁמוֹנַת שֵׁשֶׁת חֲמֵשֶׁת שְׁלוֹשֶׁת: מלאכת עבדה. אֲפִלּוּ מְלָאכוֹת הַחֲשׁוּבוֹת לָכֶם עֲבוֹדָה וְצֹרֶךְ — שֶׁיֵּשׁ חֶסְרוֹן כִּיס בְּבַטָּלָה שֶׁלָּהֶן — כְּגוֹן דָּבָר הָאָבֵד, כָּךְ הֵבַנְתִּי מִתֹּ"כֹּ, דְּקָתָנֵי יָכוֹל אַף חֻלּוֹ שֶׁל מוֹעֵד יְהֵא אָסוּר בִּמְלֶאכֶת עֲבוֹדָה וְכוּ':

ראשית קצירכם. שֶׁתְּהֵא רִאשׁוֹנָה לַקָּצִיר (ספרא): עמר. עֲשִׂירִית הָאֵיפָה כָּךְ הָיְתָה שְׁמָהּ, כְּמוֹ וַיָּמֹדּוּ בָעֹמֶר (שמות ט"ז):
והניף. כָּל תְּנוּפָה מוֹלִיךְ וּמֵבִיא מַעֲלֶה וּמוֹרִיד, מוֹלִיךְ וּמֵבִיא לַעֲצֹר רוּחוֹת רָעוֹת, מַעֲלֶה וּמוֹרִיד לַעֲצֹר טְלָלִים רָעִים (מנחות ס"ב): לרצנכם. אִם תַּקְרִיבוּ כְּמִשְׁפָּט זֶה יִהְיֶה לְרָצוֹן לָכֶם: ממחרת השבת. מִמָּחֳרַת יוֹם טוֹב הָרִאשׁוֹן שֶׁל פֶּסַח, שֶׁאִם אַתָּה אוֹמֵר שַׁבַּת בְּרֵאשִׁית אֵי אַתָּה יוֹדֵעַ אֵיזֶהוּ (ספרא; מנחות ס"ו):
ועשיתם … כבש. חוֹבָה לָעֹמֶר הוּא בָא:
ומנחתו. מִנְחַת נְסָכָיו: שני עשרנים. כְּפוּלָה הָיְתָה: ונסכו יין רביעת ההין. אַעַ"פִּ שֶׁמִּנְחָתוֹ כְפוּלָה אֵין נְסָכָיו כְּפוּלִים:
וקלי. קֶמַח עָשׂוּי מִכַּרְמֶל רַךְ שֶׁמְּיַבְּשִׁין אוֹתוֹ בַּתַּנּוּר: וכרמל. הֵן קְלָיוֹת, שֶׁקּוֹרִין גרניי"ליש: בכל משבתיכם. נֶחְלְקוּ בוֹ חַכְמֵי יִשְֹרָאֵל, יֵשׁ שֶׁלָּמְדוּ מִכָּאן שֶׁהֶחָדָשׁ נוֹהֵג בְּחוּצָה לָאָרֶץ, וְיֵ"אֹ לֹא בָא אֶלָּא לְלַמֵּד שֶׁלֹּא נִצְטַוּוּ עַל הֶחָדָשׁ אֶלָּא לְאַחַר יְרֻשָּׁה וִישִׁיבָה, מִשֶּׁכִּבְּשׁוּ וְחִלְּקוּ (קידושין ל"ז):
ממחרת השבת. מִמָּחֳרַת יוֹם טוֹב: תמימת תהיינה. מְלַמֵּד שֶׁמַּתְחִיל וּמוֹנֶה מִבָּעֶרֶב, שֶׁאִם לֹא כֵן אֵינָן תְּמִימוֹת (ספרא; מנחות ס"ו):
השבת השביעת. כְּתַרְגּוּמוֹ שְׁבוּעֲתָא שְׁבִיעֵתָא: עד ממחרת השבת השביעת תספרו. וְלֹא עַד בִּכְלָל, וְהֵן אַרְבָּעִים וְתִשְׁעָה יוֹם: חמשים יום והקרבתם מנחה חדשה לה'. בַּיּוֹם הַחֲמִשִּׁים תַּקְרִיבוּהָ, וְאוֹמֵר אֲנִי זֶה מִדְרָשׁוֹ, אֲבָל פְּשׁוּטוֹ עַד מִמָּחֳרַת הַשַּׁבָּת הַשְּׁבִיעִת שֶׁהוּא יוֹם חֲמִשִּׁים תִּסְפְּרוּ, וּמִקְרָא מְסֹרָס הוּא: מנחה חדשה. הִיא הַמִּנְחָה הָרִאשׁוֹנָה שֶׁהוּבְאָה מִן הֶחָדָשׁ, וְאִ"תֹּ הֲרֵי קָרְבָה מִנְחַת הָעֹמֶר אֵינָהּ כִּשְׁאָר כָּל הַמְּנָחוֹת, שֶׁהִיא בָאָה מִן הַשְּׂעוֹרִים (מנחות י"ד):
ממושבתיכם. וְלֹא מִחוּצָה לָאָרֶץ (ספרא): לחם תנופה. לֶחֶם תְּרוּמָה, הַמּוּרָם לְשֵׁם גָּבוֹהַּ, וְזוֹ הִיא הַמִּנְחָה הַחֲדָשָׁה הָאֲמוּרָה לְמַעְלָה: בכורים. רִאשׁוֹנָה לְכָל הַמְּנָחוֹת, אַף לְמִנְחַת קְנָאוֹת הַבָּאָה מִן שְׂעוֹרִים לֹא תִּקְרַב מִן הֶחָדָשׁ קֹדֶם לִשְׁתֵּי הַלֶּחֶם (שם):
על הלחם. בִּגְלַל הַלֶּחֶם, חוֹבָה לַלֶּחֶם (ספרא; מנחות מ"ה): ומנחתם ונסכיהם. כְּמִשְׁפַּט מִנְחָה וּנְסָכִים הַמְפֹרָשִׁים בְּכָל בְּהֵמָה בְּפָרָשַׁת נְסָכִים, שְׁלוֹשָׁה עֶשְׂרוֹנִים לַפָּר וּשְׁנֵי עֶשְׂרוֹנִים לָאַיִל וְעִשָּׂרוֹן לַכֶּבֶשׂ, זוֹ הִיא הַמִּנְחָה, וְהַנְּסָכִים חֲצִי הַהִין לַפָּר וּשְׁלִישִׁית הַהִין לָאַיִל וּרְבִיעִית הַהִין לַכֶּבֶשׂ:
ועשיתם שעיר עזים. יָכוֹל שִׁבְעַת הַכְּבָשִׂים וְהַשָּׂעִיר הָאֲמוּרִים כָּאן הֵם ז' הַכְּבָשִׂים וְהַשָּׂעִיר הָאֲמוּרִים בְּחֻמַּשׁ הַפְּקוּדִים, כְּשֶׁאַתָּה מַגִּיעַ אֵצֶל פָּרִים וְאֵילִים אֵינָן הֵם, אֱמֹר מֵעַתָּה אֵלּוּ לְעַצְמָן וְאֵלּוּ לְעַצְמָן — אֵלּוּ קָרְבוּ בִּגְלַל הַלֶּחֶם וְאֵלּוּ לַמּוּסָפִין (שם):
והניף הכהן אתם, תנופה. מְלַמֵּד שֶׁטְּעוּנִין תְּנוּפָה חַיִּים, יָכוֹל כֻּלָּם, תַּ"לֹ "עַל שְׁנֵי כְבָשִֹים" (עי' מנחות ס"ב): קדש יהיו. לְפִי שֶׁשַּׁלְמֵי יָחִיד קָדָשִׁים קַלִּים, הֻזְקַק לוֹמַר בְּשַׁלְמֵי צִבּוּר שֶׁהֵן קָדְשֵׁי קָדָשִׁים:

ובקצרכם. חָזַר וְשָׁנָה, לַעֲבֹר עֲלֵיהֶם בִּשְׁנֵי לָאוין; אָמַר ר' אַבְדִּימֵי בְּרַבִּי יוֹסֵף, מָה רָאָה הַכָּתוּב לִתְּנָהּ בְאֶמְצַע הָרְגָלִים — פֶּסַח וַעֲצֶרֶת מִכָּאן וְרֹאשׁ הַשָּׁנָה וְיוֹם הַכִּפּוּרִים וְחַג מִכָּאן —? לְלַמֶּדְךָ שֶׁכָּל הַנּוֹתֵן לֶקֶט שִׁכְחָה וּפֵאָה לֶעָנִי כָּרָאוּי, מַעֲלִין עָלָיו כְּאִלּוּ בָּנָה בֵּית הַמִּקְדָּשׁ וְהִקְרִיב קָרְבְּנוֹתָיו בְּתוֹכוֹ (ספרא): תעזב. הַנַּח לִפְנֵיהֶם וְהֵם יִלְקְטוּ, וְאֵין לְךָ לְסַיֵּעַ לְאֶחָד מֵהֶם (פאה פ"ה): אני ה' אלהיכם. נֶאֱמָן לְשַׁלֵּם שָׂכָר:

זכרון תרועה. זִכְרוֹן פְּסוּקֵי זִכְרוֹנוֹת וּפְסוּקֵי שׁוֹפָרוֹת (ספרא: ראש השנה ל"ב) לִזְכֹּר לָכֶם עֲקֵדַת יִצְחָק שֶׁקָּרַב תַּחְתָּיו אַיִל:
והקרבתם אשה. הַמּוּסָפִין הָאֲמוּרִין בְּחֻמַּשׁ הַפְּקוּדִים:

אך. כָּל אַכִין וְרַקִּין שֶׁבַּתּוֹרָה מִעוּטִים — מְכַפֵּר הוּא לַשָּׁבִים וְאֵינוֹ מְכַפֵּר עַל שֶׁאֵינָם שָׁבִים (ספרא):


והאבדתי. לְפִי שֶׁהוּא אוֹמֵר כָּרֵת בְּכָל מָקוֹם וְאֵינִי יוֹדֵעַ מַה הוּא, כְּשֶׁהוּא אוֹמֵר וְהַאֲבַדְתִּי, לִמֵּד עַל הַכָּרֵת שֶׁאֵינוֹ אֶלָּא אֲבַדּוֹן:
כל מלאכה וגו'. לַעֲבֹר עָלָיו בְּלָאוִין הַרְבֵּה, אוֹ לְהַזְהִיר עַל מְלֶאכֶת לַיְלָה כִּמְלֶאכֶת יוֹם:



מקרא קדשׁ. קַדְּשֵׁהוּ בִכְסוּת נְקִיָּה וּבִתְפִלָּה, וּבִשְׁאָר יָמִים טוֹבִים בְּמַאֲכָל וּבְמִשְׁתֶּה וּבִכְסוּת נְקִיָּה וּבִתְפִלָּה (עי' ספרא):
עצרת הוא. עָצַרְתִּי אֶתְכֶם אֶצְלִי; כְּמֶלֶךְ שֶׁזִּמֵּן אֶת בָּנָיו לִסְעוּדָה לְכָךְ וְכָךְ יָמִים, כֵּיוָן שֶׁהִגִּיעַ זְמַנָּן לִפָּטֵר, אָמַר, בָּנַי בְּבַקָּשָׁה מִכֶּם עַכְּבוּ עִמִּי עוֹד יוֹם אֶחָד, קָשָׁה עָלַי פְּרֵדַתְכֶם: כל מלאכת עבדה. אֲפִלּוּ מְלָאכָה שֶׁהִיא עֲבוֹדָה לָכֶם, שֶׁאִם לֹא תַעֲשׂוּהָ יֵשׁ חֶסְרוֹן כִּיס בַּדָּבָר. לא תעשו. יָכוֹל אַף חֻלּוֹ שֶׁל מוֹעֵד יְהֵא אָסוּר בִּמְלֶאכֶת עֲבוֹדָה, תַּ"לֹ "הִוא" (שם):
עלה ומנחה. מִנְחַת נְסָכִים הַקְּרֵבָה עִם הָעוֹלָה: דבר יום ביומו. חֹק הַקָּצוּב בְּחֻמַּשׁ הַפְּקוּדִים: דבר יום ביומו. הָא אִם עָבַר יוֹמוֹ בָּטֵל קָרְבָּנוֹ (עי' ספרא):

אך בחמשה עשר יום, תחגו. קָרְבַּן שְׁלָמִים לַחֲגִיגָה, יָכוֹל תִּדְחֶה אֶת הַשַּׁבָּת, תַּ"לֹ אַךְ, הוֹאִיל וְיֵשׁ לָהּ תַּשְׁלוּמִין כָּל שִׁבְעָה: באספכם את תבואת הארץ. שֶׁיְּהֵא חֹדֶשׁ שְׁבִיעִי זֶה בָּא בִּזְמַן אֲסִיפָה, מִכָּאן שֶׁנִּצְטַוּוּ לְעַבֵּר אֶת הַשָּׁנִים, שֶׁאִם אֵין הָעִבּוּר, פְּעָמִים שֶׁהוּא בְאֶמְצַע הַקַּיִץ אוֹ הַחֹרֶף: תחגו. שַׁלְמֵי חֲגִיגָה: שבעת ימים. אִם לֹא הֵבִיא בָזֶה, יָבִיא בָזֶה, יָכוֹל יְהֵא מְבִיאָן כָּל שִׁבְעָה, תַּלְמוּד לוֹמַר וחגתם אתו — יוֹם אֶחָד בְּמַשְׁמָע וְלֹא יוֹתֵר, וְלָמָּה נֶאֱמַר שִׁבְעָה? לְתַשְׁלוּמִין (חגיגה ט'):
פרי עץ הדר. עֵץ שֶׁטַּעַם עֵצוֹ וּפִרְיוֹ שָׁוֶה (סכה ל"ה): הדר. הַדָּר בְּאִילָנוֹ מִשָּׁנָה לְשָׁנָה, וְזֶהוּ אֶתְרוֹג (שם): כפת תמרים. חָסֵר וי"ו, לִמֵּד שֶׁאֵינָהּ אֶלָּא אַחַת: וענף עץ עבת. שֶׁעֲנָפָיו קְלוּעִים כַּעֲבוֹתוֹת וְכַחֲבָלִים, וְזֶהוּ הֲדַס, הֶעָשׂוּי כְּמִין קְלִיעָה (שם ל"ב):
