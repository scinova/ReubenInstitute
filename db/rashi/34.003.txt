אַחַר הַדְּבָרִים הָאֵלֶּה. שֶׁנִבְרֵאת רְפוּאָה זוּ לִהְיוֹת לִתְשׁוּעָה לְיִשְׂרָאֵל: גִּדַּל הַמֶּלֶךְ וְגוֹ' אֶת הָמָן. שֶׁהַקָּדוֹשׁ בָּרוּךְ הוּא בּוֹרֵא רְפוּאָה לְמַכָּתָן שֶׁל יִשְׂרָאֵל קוֹדֶם שֶׁיָבִיא הַמַּכָּה עֲלֵיהֶם:
כֹּרְעִים וּמִשְׁתַּחֲוִים. שֶׁעָשָׂה עַצְמוֹ אֱלוֹהַּ, לְפִיכָךְ, וּמָרְדְּכַי לֹא יִכְרַע וְלֹא יִשְׁתַּחֲוֶה:

הֲיַעַמְדוּ דִּבְרֵי מָרְדְּכַי. הָאוֹמֵר שֶׁלֹּא יִשְׁתַּחֲוֶה עוֹלָמִית, כִּי הוּא יְהוּדִי, וְהוּזְהַר עַל עֲבוֹדַת אֱלִילִים:


הִפִּיל פּוּר. הִפִּיל מִי שֶׁהִפִּיל, וְלֹא פִּירֵשׁ מִי. וּמִקְרָא קָצֵר הוּא: הוּא הַגּוֹרָל. הַכָּתוּב מְפָרֵשׁ וּמַהוּ הַפּוּר הוּא הַגּוֹרָל. הִפִּיל הַגּוֹרָל בְּאֵיזֶה חֹדֶשׁ יַצְלִיחַ: מִיּוֹם לְיוֹם. בְּאֵיזֶה יוֹם בַּחֹדֶשׁ יַצְלִיחַ:
וְאֶת דָּתֵי הַמֶּלֶךְ. לָתֵת מַס לַעֲבוֹדַת הַמֶּלֶךְ: אֵין שֹׁוֶה. אֵין חֲשַׁשׁ. כְּלוֹמַר אֵין בֶּצַע:
יִכָּתֵב לְאַבְּדָם. יִכָּתֵב סְפָרִים לִשְׁלֹחַ לְשָׂרֵי הַמְּדִינוֹת לְאַבְּדָם:
וַיָּסַר הַמֶּלֶךְ אֶת טַבַּעְתּוֹ. הוּא מַתַּן כָּל דָּבָר גָּדוֹל שֶׁיִּשְׁאֲלוּ מֵאֵת הַמֶּלֶךְ, לִהְיוֹת מִי שֶׁהַטַּבַּעַת בְּיָדוֹ שַׁלִּיט בְּכָל דְּבַר הַמֶּלֶךְ:


וְנִשְׁלוֹחַ סְפָרִים. וְיִהְיוּ נִשְׁלָחִים אישטר"א טרמי"ש בלע"ז. וְהוּא מִגִּזְרַת "אִם נִלְחוֹם נִלְחַם" נִלְחוֹם, "הַנִגְלֹה נִגְלֵיתִי" נִגְלֹה, "נִדְמֹה נִדְמֵיתִי": נִדְמֹה: בִּשְׁלוֹשָׁה עָשָׂר לְחֹדֶשׁ שְׁנֵים עָשָׂר. בְּיוֹם י"ג לְאוֹתוֹ חֹדֶשׁ שֶׁהוּא י"ב לְחָדְשֵׁי הַשָנָה:
פַּתְשֶׁגֶן. לָשׁוֹן אֲרַמִּי פַּתְשֶׁגֶן סִפּוּר הַכְּתָב דרישמאנ"ט בְּלַעַ"ז: לְהִנָּתֵן דָּת. צִוּוּי הַכְּתָב שֶׁהָיָה אוֹמֵר לְהִנָּתֵן חֹק גְּזֵרַת הַמֶּלֶךְ: גָּלוּי לְכָל הָעַמִּים. דָּבָר זֶה:
וְהַדָּת נִתְּנָה בְּשׁוּשַׁן. מָקוֹם שֶׁהָיָה הַמֶּלֶךְ שָׁם נִתַּן הַחֹק בּוֹ בַיּוֹם, לִהְיוֹת עֲתִידִים לְיוֹם י"ג לְחֹדֶשׁ אֲדָר. לְכַךְ, וְהָעִיר שׁוּשָׁן נָבוֹכָה. הַיְּהוּדִים שֶׁבָּהּ:
