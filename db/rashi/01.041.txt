ויהי מקץ. כְּתַרְגּוּמוֹ מִסּוֹף, וְכָל לְשׁוֹן קֵץ סוֹף הוּא: על היאר. כָּל שְׁאָר נְהָרוֹת אֵינָם קְרוּיִין יְאוֹרִים חוּץ מִנִּילוּס, מִפְּנֵי שֶׁכָּל הָאָרֶץ עֲשׂוּיִם יְאוֹרִים יְאוֹרִים בִּידֵי אָדָם וְנִילוּס עוֹלֶה בְּתוֹכָם וּמַשְׁקֶה אוֹתָם, לְפִי שֶׁאֵין גְּשָׁמִים יוֹרְדִין בְּמִצְרַיִם תָּדִיר כִּשְׁאָר אֲרָצוֹת:
יפות מראה. סִימָן הוּא לִימֵי שֹׂבַע, שֶׁהַבְּרִיּוֹת נִרְאוֹת יָפוֹת זוֹ לָזוֹ, שֶׁאֵין עֵין בְּרִיָּה צָרָה בַחֲבֶרְתָּהּ: באחו. בָּאֲגַם, מריש"ק בְּלַעַז, כְּמוֹ יִשְׂגֶּא אָחוּ (איוב ח'):
ודקות בשר. טינב"ש בְּלַעַז, לְשׁוֹן דַּק:
ותאכלנה. סִימָן שֶׁתְּהֵא כָל שִׂמְחַת הַשָּׂבָע נִשְׁכַּחַת בִּימֵי הָרָעָב:
בקנה אחת. טוד"ל בְּלַעַז: בריאות. שיינ"ש בְּלַעַז:
ושדופת. הלייא"ש בְּלַעַז, שְׁקִיפָן קִדּוּם, חֲבוּטוֹת, לְשׁוֹן מַשְׁקוֹף הֶחָבוּט תָּמִיד עַל יְדֵי הַדֶּלֶת הַמַּכָּה עָלָיו: קדים. רוּחַ מִזְרָחִית, שֶׁקּוֹרִין בי"סא בְּלַעַז:
הבריאות. שיינ"ש בְּלַעַז: והנה חלום. וְהִנֵּה נִשְׁלַם חֲלוֹם שָׁלֵם לְפָנָיו וְהֻצְרַךְ לְפוֹתְרִים:
ותפעם רוחו. וּמִטַּרְפָּא רוּחֵיהּ, מְקַשְׁקֶשֶׁת בְּתוֹכוֹ כְּפַעֲמוֹן; וּבִנְבוּכַדְנֶצַּר אוֹמֵר, וַתִּתְפָּעֶם רוּחוֹ (דניאל ב'), לְפִי שֶׁהָיוּ שָׁם שְׁתֵּי פְעִימוֹת, שִׁכְחַת הַחֲלוֹם וְהַעֲלָמַת פִּתְרוֹנוֹ (בראשית רבה): חרטמי. הַנֵּחָרִים בְּטִימֵי מֵתִים, שֶׁשּׁוֹאֲלִין בַּעֲצָמוֹת. טִימֵי הֵן עֲצָמוֹת בִּלְשׁוֹן אֲרַמִּי, וּבַמִּשְׁנָה בַּיִת שֶׁהוּא מָלֵא טִימַיָּא (אהלות פי"ז), מָלֵא עֲצָמוֹת: ואין פותר אותם לפרעה. פּוֹתְרִים הָיוּ אוֹתָם, אֲבָל לֹא לְפַרְעֹה, שֶׁלֹּא הָיָה קוֹלָן נִכְנָס בְּאָזְנָיו, וְלֹא הָיָה לוֹ קוֹרַת רוּחַ בְּפִתְרוֹנָם, שֶׁהָיוּ אוֹמְרִים שֶׁבַע בָּנוֹת אַתָּה מוֹלִיד וְשֶׁבַע בָּנוֹת אַתָּה קוֹבֵר (בראשית רבה):


איש כפתרון חלמו. חֲלוֹם הָרָאוּי לַפִּתְרוֹן שֶׁנִּפְתַּר לָנוּ וְדוֹמֶה לוֹ:
נער עברי עבד. אֲרוּרִים הָרְשָׁעִים, שֶׁאֵין טוֹבָתָם שְׁלֵמָה, מַזְכִּירוֹ בִּלְשׁוֹן בִּזָּיוֹן: נער. שׁוֹטֶה וְאֵין רָאוּי לִגְדֻלָּה: עברי. אֲפִלּוּ לְשׁוֹנֵנוּ אֵינוֹ מַכִּיר: עבד. וְכָתוּב בְּנִמּוּסֵי מִצְרַיִם שֶׁאֵין עֶבֶד מוֹלֵךְ וְלֹא לוֹבֵשׁ בִּגְדֵי שָׂרִים: איש כחלמו. לְפִי הַחֲלוֹם וְקָרוֹב לְעִנְיָנוֹ:
השיב על כני. פַּרְעֹה, הַנִּזְכָּר לְמַעְלָה, כְּמוֹ שֶׁאָמַר פַּרְעֹה קָצַף עַל עֲבָדָיו; הֲרֵי מִקְרָא קְצַר לָשׁוֹן, וְלֹא פֵּרֵשׁ מִי הֵשִׁיב, לְפִי שֶׁאֵין צָרִיךְ לְפָרֵשׁ; מִי הֵשִׁיב? מִי שֶׁבְּיָדוֹ לְהָשִׁיב, וְהוּא פַּרְעֹה; וְכֵן דֶּרֶךְ כָּל מִקְרָאוֹת קְצָרִים, עַל מִי שֶׁעָלָיו לַעֲשׂוֹת הֵם סוֹתְמִים אֶת הַדָּבָר:
מן הבור. מִן בֵּית הַסֹּהַר שֶׁהוּא עָשׂוּי כְּמִין גֻּמָּא, וְכֵן כָּל בּוֹר שֶׁבַּמִּקְרָא לְשׁוֹן גֻּמָּא הוּא, וְאַף אִם אֵין בּוֹ מַיִם קָרוּי בּוֹר, פוש"א בְּלַעַז: ויגלח. מִפְּנֵי כְבוֹד הַמַּלְכוּת (בראשית רבה):
תשמע חלום לפתר אתו. תַּאֲזִין וְתָבִין חֲלוֹם לִפְתֹּר אוֹתוֹ: תשמע. לְשׁוֹן הֲבָנָה וְהַאֲזָנָה, כְּמוֹ שׁוֹמֵעַ יוֹסֵף, אֲשֶׁר לֹא תִשְׁמַע לְשֹׁנוֹ, אנטינ"דרא בְּלַעַז:
בלעדי. אֵין הַחָכְמָה מִשֶּׁלִּי, אֶלָּא אֱלֹהִים יַעֲנֶה – יִתֵּן עֲנִיָּה בְּפִי – לִשְׁלוֹם פַּרְעֹה:


דלות. כְּחוּשׁוֹת, כְּמוֹ מַדּוּעַ אַתָּה כָּכָה דַּל, דְּאַמְנוֹן (שמואל ב י"ג): ורקות בשר. כָּל לְשׁוֹן רַקּוֹת שֶׁבַּמִּקְרָא חַסְרֵי בָשָׂר, וּבְלַעַז בלוא"ש:



צנמות. צוּנְמָא בִּלְשׁוֹן אֲרַמִּי סֶלַע, הֲרֵי הֵן כְּעֵץ בְּלִי לִחְלוּחַ וְקָשׁוֹת כְּסֶלַע, וְתַרְגּוּמוֹ נָצָן לָקְיָן, נָצָן – אֵין בָּהֶם אֶלָּא הַנֵּץ, לְפִי שֶׁנִּתְרוֹקְנוּ מִן הַזֶּרַע:


שבע שנים ושבע שנים. כֻּלָּן אֵינָן אֶלָּא שֶׁבַע, וַאֲשֶׁר נִשְׁנָה הַחֲלוֹם פַּעֲמַיִם, לְפִי שֶׁהַדָּבָר מְזֻמָּן כְּמוֹ שֶׁפֵּרֵשׁ לוֹ בַּסּוֹף ועל השנות החלום וגו' בְּשֶׁבַע שָׁנִים הַטּוֹבוֹת נֶאֱמַר הגיד לפרעה לְפִי שֶׁהָיָה סָמוּךְ וּבְשֶׁבַע שְׁנֵי רָעָב נֶאֱמַר הראה את פרעה, לְפִי שֶׁהָיָה הַדָּבָר מֻפְלָג וְרָחוֹק, נוֹפֵל בּוֹ לְשׁוֹן מַרְאֶה:



ונשכח כל השבע. הוּא פִתְרוֹן הַבְּלִיעָה:
ולא יודע השבע. הוּא פִּתְרוֹן "ולא נודע כי באו אל קרבנה" :
נכון. מְזֻמָּן:

וחמש. כְּתַרְגּוּמוֹ וִיזָרְזוּן; וְכֵן וַחֲמֻשִׁים (שמות י"ג):
את כל אכל. שֵׁם דָּבָר הוּא, לְפִיכָךְ טַעֲמוֹ בָּאָלֶ"ף וְנָקוּד בְּפַתָּח קָטָן, וְאוֹכֵל שֶׁהוּא פֹּעַל, כְּגוֹן כִּי כָּל אֹכֵל חֵלֶב, טַעֲמוֹ לְמַטָּה בַּכַּ"ף, וְנָקוּד קָמָץ קָטָן: תחת יד פרעה. בִּרְשׁוּתוֹ וּבְאוֹצְרוֹתָיו:
והיה האכל. הַצָּבוּר כִּשְׁאָר פִּקָּדוֹן הַגָּנוּז לְקִיּוּם הָאָרֶץ:

הנמצא כזה. הֲנִשְׁכַּח כְּדֵין? אִם נֵלֵךְ וּנְבַקְשֶׁנּוּ, הֲנִמְצָא כָמוֹהוּ? הֲנִמְצָא לְשׁוֹן תְּמִיהָה; וְכֵן כָּל הֵ"א הַמְשַׁמֶּשֶׁת בְּרֹאשׁ תֵּבָה וּנְקוּדָה בַחֲטַף פַּתָּח:
אין נבון וחכם כמוך. לְבַקֵּשׁ אִישׁ נָבוֹן וְחָכָם שֶׁאָמַרְתָּ, לֹא נִמְצָא כָמוֹךָ:
ישק. יִתְּזַן, יִתְפַּרְנֵס – כָּל צָרְכֵי עַמִּי יִהְיוּ נַעֲשִׂים עַל יָדְךָ, כְּמוֹ נַשְּׁקוּ בַר (תהילים ב'), גרני"שון בְּלַעַז: רק הכסא. שֶׁיִּהְיוּ קוֹרִין לִי מֶלֶךְ: כסא. לְשׁוֹן שֵׁם הַמְּלוּכָה, כְּמוֹ וִיגַדֵּל אֶת כִּסְאוֹ מִכִּסֵּא אֲדֹנִי הַמֶּלֶךְ (מלכים א א'):
נתתי אתך. מַנֵּיתִי יָתָךְ, וְאַעַ"פִּ כֵּן לְשׁוֹן נְתִינָה הוּא, כְּמוֹ וּלְתִתְּךָ עֶלְיוֹן (דברים כ"ו); בֵּין לִגְדֻלָּה בֵּין לְשִׁפְלוּת נוֹפֵל לְשׁוֹן נְתִינָה עָלָיו, כְּמוֹ נָתַתִּי אֶתְכֶם נִבְזִים וּשְׁפָלִים (מלאכי ב'):
ויסר פרעה את טבעתו. נְתִינַת טַבַּעַת הַמֶּלֶךְ הִיא אוֹת לְמִי שֶׁנּוֹתְנָהּ לוֹ לִהְיוֹת שֵׁנִי לוֹ לִגְדֻלָּה: בגדי שש. דְּבַר חֲשִׁיבוּת הוּא בְּמִצְרַיִם: רביד. עֲנָק, וְעַל שֶׁהוּא רָצוּף בְּטַבָּעוֹת קָרוּי רְבִיד וְכֵן מַרְבַדִּים רָבַדְתִּי עַרְשִׂי (משלי ז') – רִצַּפְתִּי עַרְשִׂי מַרְצָפוֹת, בִּלְשׁוֹן מִשְׁנָה מֻקָּף רוֹבְדִין שֶׁל אֶבֶן, עַל הָרֹבֶד שֶׁבָּעֲזָרָה, וְהִיא רִצְפָּה:
במרכבת המשנה. הַשְּׁנִיָּה לְמֶרְכַּבְתּוֹ, הַמְהַלֶּכֶת אֵצֶל שֶׁלּוֹ: אברך. כְּתַרְגּוּמוֹ "דֵּין אַבָּא לְמַלְכָּא", רֵךְ בִּלְשׁוֹן אֲרַמִּי (נ"א רוֹמִי) מֶלֶךְ; בְּהַשֻּׁתָּפִין לָא רֵיכָא וְלֹא בַּר רֵיכָא, וּבְדִבְרֵי אַגָּדָה דָּרַשׁ רַבִּי יְהוּדָה אַבְרֵךְ זֶה יוֹסֵף שֶׁהוּא אָב בְּחָכְמָה וְרַךְ בַּשָּׁנִים, אָמַר לוֹ רַ' יוֹסֵי בֶּן דֻּרְמַסְקִית עַד מָתַי אַתָּה מְעַוֵּת עָלֵינוּ אֶת הַכְּתוּבִים? אֵין אַבְרֵךְ אֶלָּא לְשׁוֹן בִּרְכַּיִם, שֶׁהַכֹּל הָיוּ נִכְנָסִין וְיוֹצְאִין תַּחַת יָדוֹ, כָּעִנְיָן שֶׁנֶּאֱמַר, וְנָתוֹן אוֹתוֹ וְגוֹ':
אני פרעה. שֶׁיֵּשׁ יְכֹלֶת בְּיָדִי לִגְזֹר גְּזֵרָה עַל מַלְכוּתִי, וַאֲנִי גּוֹזֵר שֶׁלֹּא יָרִים אִישׁ יָדוֹ בִלְעָדֶיךָ, – שֶׁלֹּא בִּרְשׁוּתְךָ. דָּ"אַ אֲנִי פַרְעֹה, אֲנִי אֶהְיֶה מֶלֶךְ, וּבִלְעָדֶיךָ וְגוֹ' וְזֶהוּ דֻּגְמַת רַק הַכִּסֵּא: את ידו ואת רגלו. כְּתַרְגּוּמוֹ:
צפנת פענח. מְפָרֵשׁ הַצְּפוּנוֹת; וְאֵין לְפַעְנֵחַ דִּמְיוֹן בַּמִּקְרָא: פוטיפרע. הוּא פּוֹטִיפַר, וְנִקְרָא פּוֹטִיפֶרַע עַל שֶׁנִּסְתָּרֵס מֵאֵלָיו, לְפִי שֶׁחָמַד אֶת יוֹסֵף לְמִשְׁכַּב זָכָר (סוטה י"ג):

ותעש הארץ. כְּתַרְגּוּמוֹ, וְאֵין הַלָּשׁוֹן נֶעֱקָר מִלְּשׁוֹן עֲשִׂיָּה: לקמצים. קֹמֶץ עַל קֹמֶץ, יָד עַל יָד, הָיוּ אוֹצְרִים:
אכל שדה העיר אשר סביבתיה נתן בתוכה. שֶׁכָּל אֶרֶץ וְאֶרֶץ מַעֲמֶדֶת פֵּרוֹתֶיהָ, וְנוֹתְנִין בַּתְּבוּאָה מֵעֲפַר הַמָּקוֹם, וּמַעֲמִיד אֶת הַתְּבוּאָה מִלִּרְקֹב:
עד כי חדל לספר. עַד כִּי חָדַל לוֹ הַסּוֹפֵר לִסְפֹּר, וַהֲרֵי זֶה מִקְרָא קָצָר: כי אין מספר. לְפִי שֶׁאֵין מִסְפָּר, וַהֲרֵי כִי מְשַׁמֵּשׁ בְּלְשׁוֹן דְּהָא:
בטרם תבא שנת הרעב. מִכָּאן שֶׁאָדָם אָסוּר לְשַׁמֵּשׁ מִטָּתוֹ בִּשְׁנֵי רְעָבוֹן (תענית י"א):




ותרעב כל ארץ מצרים. שֶׁהִרְקִיבָה תְבוּאָתָם שֶׁאָצְרוּ חוּץ מִשֶּׁל יוֹסֵף: אשר יאמר לכם תעשו. לְפִי שֶׁהָיָה יוֹסֵף אוֹמֵר לָהֶם שֶׁיִּמּוֹלוּ, וּכְשֶׁבָּאוּ אֵצֶל פַּרְעֹה וְאוֹמְרִים: כָּךְ הוּא אוֹמֵר לָנוּ, אָמַר לָהֶם: וְלָמָּה לֹא צְבַרְתֶּם בָּר, וַהֲלֹא הִכְרִיז לָכֶם שֶׁשְּׁנֵי הָרָעָב בָּאִים? אָמְרוּ לוֹ: אָסַפְנוּ הַרְבֵּה וְהִרְקִיבָה, אָמַר לָהֶם: אִם כֵּן, כָּל אֲשֶׁר יֹאמַר לָכֶם תַּעֲשׂוּ; הֲרֵי גָּזַר עַל הַתְּבוּאָה וְהִרְקִיבָה, מַה אִם יִגְזֹר עָלֵינוּ וְנָמוּת!
על כל פני הארץ. מִי הֵם פְּנֵי הָאָרֶץ אֵלּוּ הָעֲשִׁירִים: את כל אשר בהם. כְּתַרְגּוּמוֹ דִּי בְהוֹן עִיבוּרָא: וישבר למצרים. שבר לְשׁוֹן מֶכֶר וּלְשׁוֹן קִנְיָן הוּא; כָּאן מְשַׁמֵּשׁ לְשׁוֹן מֶכֶר; שִׁבְרוּ לָנוּ מְעַט אֹכֶל, לְשׁוֹן קִנְיָן. וְאַל תֹּאמַר אֵינוֹ כִּי אִם בִּתְבוּאָה, שֶׁאַף בְּיַיִן וְחָלָב מָצִינוּ וּלְכוּ שִׁבְרוּ בְּלוֹא כֶסֶף וּבְלוֹא מְחִיר יַיִן וְחָלָב (ישעיהו נ"ה):
