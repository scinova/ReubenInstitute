
אני ה' אלהיכם. אֲנִי הוּא שֶׁאָמַרְתִּי בְסִינַי אָנֹכִי ה' אֱלֹהֶיךָ וְקִבַּלְתֶּם עֲלֵיכֶם מַלְכוּתִי, מֵעַתָּה קַבְּלוּ גְזֵרוֹתָי; רַבִּי אוֹמֵר גָּלוּי וְיָדוּעַ לְפָנָיו שֶׁסּוֹפָן לִנָּתֵק בַּעֲרָיוֹת בִּימֵי עֶזְרָא, לְפִיכָךְ בָּא עֲלֵיהֶם בִּגְזֵרָה אני ה' אלהיכם — דְּעוּ מִי גוֹזֵר עֲלֵיכֶם, דַּיָּן לִפָּרַע וְנֶאֱמָן לְשַׁלֵּם שָׂכָר (עי' ספרא):
כמעשה ארץ מצרים. מַגִּיד שֶׁמַּעֲשֵׂיהֶם שֶׁל מִצְרִיִּים וְשֶׁל כְּנַעֲנִיִּים מְקֻלְקָלִים מִכָּל הָאֻמּוֹת, וְאוֹתוֹ מָקוֹם שֶׁיָּשְׁבוּ בוֹ יִשְׂרָאֵל מְקֻלְקָל מִן הַכֹּל: אשר אני מביא אתכם שמה. מַגִּיד שֶׁאוֹתָן עֲמָמִין שֶׁכִּבְּשׁוּ יִשְֹרָאֵל מְקֻלְקָלִים מִכֻּלָּם (שם): ובחקתיהם לא תלכו. מַה הִנִּיחַ הַכָּתוּב שֶׁלֹּא אָמַר? אֶלָּא אֵלּוּ נִימוֹסוֹת שֶׁלָּהֶן — דְּבָרִים הַחֲקוּקִין לָהֶם — כְּגוֹן טַרְטִיָּאוֹת וְאִצְטַדִיָּאוֹת, רַבִּי מֵאִיר אוֹמֵר, אֵלּוּ דַּרְכֵי הָאֱמוֹרִי שֶׁמָּנוּ חֲכָמִים (שם):
את משפטי תעשו. אֵלּוּ דְּבָרִים הָאֲמוּרִים בַּתּוֹרָה בְּמִשְׁפָּט, שֶׁאִלּוּ לֹא נֶאֶמְרוּ הָיוּ כְדַאי לְאָמְרָן: ואת חקתי תשמרו. דְּבָרִים שֶׁהֵן גְּזֵרַת הַמֶּלֶךְ, שֶׁיֵּצֶר הָרַע מֵשִׁיב עֲלֵיהֶם לָמָּה לָנוּ לְשָׁמְרָן? וְאֻמּוֹת הָעוֹלָם מְשִׁיבִין עֲלֵיהֶם, כְּגוֹן אֲכִילַת חֲזִיר וּלְבִישַׁת שַׁעַטְנֵז וְטָהֳרַת מֵי חַטָּאת, לְכָךְ נֶאֱמַר אני ה', גָּזַרְתִּי עֲלֵיכֶם, אֵי אַתָּה רַשַּׁאי לִפָּטֵר (יומא ס"ז): ללכת בהם. אַל תִּפָּטֵר מִתּוֹכָם, שֶׁלֹּא תֹאמַר לָמַדְתִּי חָכְמַת יִשְׂרָאֵל אֵלֵךְ וְאֶלְמַד חָכְמַת הָאֻמּוֹת (ספרא):
ושמרתם את חקתי וגו'. לְרַבּוֹת שְׁאָר דִּקְדּוּקֵי הַפָּרָשָׁה, שֶׁלֹּא פֵּרֵט הַכָּתוּב בָּהֶם; דָּבָר אַחֵר לִתֵּן שְׁמִירָה וַעֲשִׂיָּה לַחֻקִּים וּשְׁמִירָה וַעֲשִׂיָּה לַמִּשְׁפָּטִים, לְפִי שֶׁלֹּא נָתַן אֶלָּא עֲשִׂיָּה לַמִּשְׁפָּטִים וּשְׁמִירָה לַחֻקִּים (ספרא): וחי בהם. לָעוֹלָם הַבָּא, שֶׁאִם תֹּאמַר בָּעוֹלָם הַזֶּה וַהֲלֹא סוֹפוֹ הוּא מֵת (שם): אני ה'. נֶאֱמָן לְשַׁלֵּם שָׂכָר (שם):
לא תקרבו. לְהַזְהִיר הַנְּקֵבָה כַּזָּכָר, לְכָךְ נֶאֱמַר לְשׁוֹן רַבִּים: אני ה'. נֶאֱמָן לְשַׁלֵּם שָׂכָר:
ערות אביך. זוֹ אֵשֶׁת אָבִיךָ, אוֹ אֵינוֹ אֶלָּא כְמַשְׁמָעוֹ? נֶאֱמַר כָּאן עֶרְוַת אָבִיךָ וְנֶאֱמַר לְהַלָּן עֶרְוַת אָבִיו גִּלָּה, מַה לְּהַלָּן אֵשֶׁת אָבִיו אַף כָּאן אֵשֶׁת אָבִיו (סנהדרין נ"ד): וערות אמך. לְהָבִיא אִמּוֹ שֶׁאֵינָהּ אֵשֶׁת אָבִיו (שם):
ערות אשת אביך. לְרַבּוֹת אַחַר מִיתָה (שם):
בת אביך. אַף בַּת אֲנוּסָה בְמַשְׁמָע: מולדת בית או מולדת חוץ. בֵּין שֶׁאוֹמְרִים לוֹ לְאָבִיךָ קַיֵּם אֶת אִמָּהּ, וּבֵין שֶׁאוֹמְרִים לוֹ הוֹצֵא אֶת אִמָּהּ, כְּגוֹן מַמְזֶרֶת אוֹ נְתִינָה (יבמות כ"ג):
ערות בת בנך וגו'. בְּבִתּוֹ מֵאֲנוּסָתוֹ הַכָּתוּב מְדַבֵּר, וּבִתּוֹ וּבַת בִּתּוֹ מֵאִשְׁתּוֹ אָנוּ לְמֵדִין מֵ"עֶרְוַת אִשָּׁה וּבִתָּהּ", שֶׁנֶּאֱמַר בָּהֶן "לֹא תְגַלֵּה", בֵּין שֶׁהִיא מִמֶּנּוּ וּבֵין שֶׁהִיא מֵאִישׁ אַחֵר (שם כב): ערות בת בנך. קַל וָחֹמֶר לְבִתְּךָ, אֶלָּא לְפִי שֶׁאֵין מַזְהִירִין מִן הַדִּין, לְמָדוּהָ מִגְּזֵרָה שָׁוָה בְּמַסֶּכֶת יְבָמוֹת (דף ג'):
ערות בת אשת אביך. לִמֵּד שֶׁאֵינוֹ חַיָּב עַל אֲחוֹתוֹ מִשִּׁפְחָה וְנָכְרִית, לְכָךְ נֶאֱמַר "בַּת אֵשֶׁת אָבִיךָ" — בִּרְאוּיָה לְקִדּוּשִׁין (שם כ"ג):


ערות אחי אביך לא תגלה. וּמַה הִיא עֶרְוָתוֹ? אל אשתו לא תקרב :
אשת בנך. לֹא אָמַרְתִּי אֶלָּא בְּשֶׁיֵּשׁ לְבִנְךָ אִישׁוּת בָּהּ, פְּרָט לַאֲנוּסָה וְשִׁפְחָה וְנָכְרִית (עי' ספרא):

ערות אשה ובתה. לֹא אָסַר הַכָּתוּב אֶלָּא עַל יְדֵי נִשּׂוּאֵי הָרִאשׁוֹנָה, לְכָךְ נֶאֱמַר לֹא "תִקַּח" — לְשׁוֹן קִיחָה, וְכֵן לְעִנְיַן הָעֹנֶשׁ, אֲשֶׁר "יִקַּח" אֶת אִשָּׁה וְאֶת אִמָּה — לְשׁוֹן קִיחָה, אֲבָל אָנַס אִשָּׁה מֻתָּר לִשָּׂא בִּתָּהּ (יבמות צ"ז): שארה הנה. קְרוֹבוֹת זוֹ לָזוֹ: זמה. עֵצָה, כְּתַרְגּוּמוֹ עֲצַת חֶטְאִין, שֶׁיִּצְרְךָ יוֹעֶצְךָ לַחֲטֹא:
אל אחתה. שְׁתֵּיהֶן כְּאַחַת (קידושין נ'): לצרר. לְשׁוֹן צָרָה — לַעֲשׂוֹת אֶת זוֹ צָרָה לָזוֹ: בחייה. לִמֶּדְךָ שֶׁאִם גֵרְשָׁהּ לֹא יִשָּׂא אֶת אֲחוֹתָהּ כָּל זְמַן שֶׁהִיא בַחַיִּים (יבמות ח'):


למלך. עֲ"זָ הִיא שֶׁשְּׁמָהּ מֹלֶךְ, וְזוֹ הִיא עֲבוֹדָתָהּ, שֶׁמּוֹסֵר בְּנוֹ לַכּוֹמְרִים וְעוֹשִׂין שְׁתֵּי מְדוּרוֹת גְּדוֹלוֹת וּמַעֲבִירִין אֶת הַבֵּן בְּרַגְלָיו בֵּין שְׁתֵּי מְדוּרוֹת הָאֵשׁ (סנהדרין ס"ד): לא תתן. זוֹ הִיא מְסִירָתוֹ לַכּוֹמְרִים: להעביר למלך. זוֹ הַעֲבָרַת הָאֵשׁ:

תבל הוא. לְשׁוֹן קָדֵשׁ וְעֶרְוָה וְנִאוּף, וְכֵן וְאַפִּי עַל תַּבְלִיתָם (ישעיהו י'); דָּבָר אַחֵר תֶּבֶל הוּא לְשׁוֹן בְּלִילָה וְעִרְבּוּב זֶרַע אָדָם וְזֶרַע בְּהֵמָה:




ולא תקיא הארץ אתכם. מָשָׁל לְבֶן מֶלֶךְ שֶׁהֶאֱכִילוּהוּ דָּבָר מָאוּס, שֶׁאֵין עוֹמֵד בְּמֵעָיו אֶלָּא מְקִיאוֹ, כָּךְ אֶרֶץ יִשְׂרָאֵל אֵינָהּ מְקַיֶּמֶת עוֹבְרֵי עֲבֵרָה (ספרא סוף קדושים), וְתַרְגּוּמוֹ וְלָא תְרוֹקֵן, לְשׁוֹן רִקּוּן — מְרִיקָה עַצְמָהּ מֵהֶם:
הנפשות העשת. הַזָּכָר וְהַנְּקֵבָה בְמַשְׁמָע (ספרא):
ושמרתם את משמרתי. לְהַזְהִיר בֵּית דִּין עַל כָּךְ (שם): ולא תטמאו בהם אני ה' אלהיכם. הָא אִם תִּטַּמְּאוּ אֵינִי אֱלֹהֵיכֶם, וְאַתֶּם נִפְסָלִים מֵאַחֲרַי, וּמַה הֲנָאָה יֵשׁ לִי בָּכֶם וְאַתֶּם מִתְחַיְּבִים כְּלָיָה? לְכָךְ נֶאֱמַר אֲנִי ה' אֱלֹהֵיכֶם (שם):
