
קח את אהרן. פָּרָשָׁה זוֹ נֶאֶמְרָה שִׁבְעַת יָמִים קֹדֶם הֲקָמַת הַמִּשְׁכָּן, שֶׁאֵין מֻקְדָּם וּמְאֻחָר בַּתּוֹרָה: קח את אהרן. קָחֶנּוּ בִדְבָרִים וּמָשְׁכֵהוּ: ואת פר החטאת וגו'. אֵלּוּ הָאֲמוּרִים בְּעִנְיַן צַוָּאַת הַמִּלּוּאִים בִּוְאַתָּה תְּצַוֶּה, וְעַכְשָׁיו בְּיוֹם רִאשׁוֹן לַמִּלּוּאִים חָזַר וְזֵרְזוֹ בִשְׁעַת מַעֲשֶֹה:
הקהל אל פתח אהל מועד. זֶה אֶחָד מִן הַמְּקוֹמוֹת שֶׁהֶחֱזִיק מֻעָט אֶת הַמְרֻבֶּה (ויקרא רבה י'):

זה הדבר. דְּבָרִים שֶׁתִּרְאוּ שֶׁאֲנִי עוֹשֶׂה לִפְנֵיכֶם, צִוַּנִי הַקָּבָּ"ה לַעֲשׂוֹת, וְאַל תֹּאמְרוּ לִכְבוֹדִי וְלִכְבוֹד אָחִי אֲנִי עוֹשֶֹׁה; כָּל הָעִנְיָן הַזֶּה פֵּרַשְׁתִּי בְּ"וְאַתָּה תְּצַוֶּה":


את האורים. כְּתָב שֶׁל שֵׁם הַמְפֹרָשׁ:
וישם על המצנפת. פְּתִילֵי תְכֵלֶת הַקְּבוּעִים בַּצִּיץ נָתַן עַל הַמִּצְנֶפֶת, נִמְצָא הַצִּיץ תָּלוּי בַּמִּצְנֶפֶת:

ויז ממנו על המזבח. לֹא יָדַעְתִּי הֵיכָן נִצְטַוָּה בַּהַזָּאוֹת הַלָּלוּ:
ויצק, וימשח. בַּתְּחִלָּה יוֹצֵק עַל רֹאשׁוֹ וְאַחַר כָּךְ נוֹתֵן בֵּין רִיסֵי עֵינָיו וּמוֹשֵׁךְ בְּאֶצְבָּעוֹ מִזֶּה לָזֶה (הוריות י"ב):
ויחבש. לְשׁוֹן קְשִׁירָה:

ויחטא את המזבח. חִטְּאוֹ וְטִהֲרוֹ מִזָּרוּת לִכָּנֵס לִקְדֻשָּׁה: ויקדשהו. בַּעֲבוֹדָה זוֹ: לכפר עליו. מֵעַתָּה כָּל הַכַּפָּרוֹת:
הכבד. לְבַד הַכָּבֵד, שֶׁהָיָה נוֹטֵל מְעַט מִן הַכָּבֵד עִמָּהּ:





איל המלאים. אֵיל הַשְּׁלָמִים, שֶׁמִּלּוּאִים לְשׁוֹן שְׁלָמִים, שֶׁמְּמַלְּאִין וּמַשְׁלִימִין אֶת הַכֹּהֲנִים בִּכְהֻנָּתָם:



וחלת לחם שמן. הִיא רְבוּכָה שֶׁהָיָה מַרְבֶּה בָּהּ שֶׁמֶן כְּנֶגֶד הַחַלּוֹת וְהָרְקִיקִין, כָּךְ מְפֹרָשׁ בִּמְנָחוֹת (דף פ"ט):

ויקטר המזבחה. מֹשֶׁה שִׁמֵּשׁ כָּל שִׁבְעַת יְמֵי הַמִּלּוּאִים בְּחָלוּק לָבָן (עבודה זרה ל"ד): על העלה. אַחַר הָעוֹלָה, וְלֹא מָצִינוּ שׁוֹק שֶׁל שְׁלָמִים קָרֵב בְּכָל מָקוֹם חוּץ מִזֶּה:





צוה ה' לעשת. כָּל שִׁבְעַת הַיָּמִים; וְרַבּוֹתֵינוּ דָרְשׁוּ, לעשת — זֶה מַעֲשֵׂה פָרָה, לכפר — זֶה מַעֲשֵׂה יוֹם הַכִּפּוּרִים, וּלְלַמֵּד שֶׁכֹּהֵן גָּדוֹל טָעוּן פְּרִישָׁה קֹדֶם יוֹם הַכִּפּוּרִים שִׁבְעַת יָמִים, וְכֵן הַכֹּהֵן הַשּׂוֹרֵף אֶת הַפָּרָה (ספרא):
ולא תמותו. הָא אִם לֹא תַעֲשׂוּ כֵן הֲרֵי אַתֶּם חַיָּבִים מִיתָה:
ויעש אהרן ובניו. לְהַגִּיד שִׁבְחָן, שֶׁלֹא הִטּוּ יָמִין וּשְֹמֹאל:
