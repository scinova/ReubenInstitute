ראיתי צדיק. וְלֹא נֶאֱמַר צַדִּיק תָּמִים, מִכָּאן שֶׁאוֹמְרִים מִקְצַת שִׁבְחוֹ שֶׁל אָדָם בְּפָנָיו וְכֻלּוֹ שֶׁלֹּא בְּפָנָיו (ב"ר):
הטהורה. הָעֲתִידָה לִהְיוֹת טְהוֹרָה לְיִשְׂרָאֵל; לָמַדְנוּ שֶׁלָּמַד נֹחַ תּוֹרָה (שם פ' כ"ו): שבעה שבעה. כְּדֵי שֶׁיַּקְרִיב מֵהֶם קָרְבָּן בְּצֵאתוֹ:
גם מעוף השמים וגו'. בִּטְהוֹרִים הַכָּתוּב מְדַבֵּר, וְלָמַד סָתוּם מִן הַמְפֹרָשׁ:
כי לימים עוד שבעה. אֵלּוּ ז' יְמֵי אֶבְלוֹ שֶׁל מְתוּשֶׁלַח הַצַּדִּיק שֶׁחָס הַקָּבָּ"ה עַל כְּבוֹדוֹ וְעִכֵּב אֶת הַפֻּרְעָנוּת. צֵא וַחֲשֹׁב שְׁנוֹתָיו שֶׁל מְתוּשֶׁלַח וְתִמְצָא שֶׁהֵם כָּלִים בִּשְנַת ת"ר שָׁנָה לְחַיֵּי נֹחַ (סנה' ק"ח): כי לימים עוד. מַהוּ עוֹד? זְמַן אַחַר זְמַן זֶה נוֹסָף עַל ק"כ שָׁנָה: ארבעים יום. כְּנֶגֶד יְצִירַת הַוָּלָד, שֶׁקִּלְקְלוּ לְהַטְרִיחַ לְיוֹצְרָם לָצוּר צוּרַת מַמְזֵרִים:
ויעש נח. זֶה בִיאָתוֹ לַתֵּבָה:

נח ובניו. הָאֲנָשִׁים לְבַד וְהַנָּשִׁים לְבַד, לְפִי שֶׁנֶּאֱסְרוּ בְּתַשְׁמִישׁ הַמִּטָּה מִפְּנֵי שֶׁהָעוֹלָם שָׁרוּי בְּצַעַר: מפני מי המבול. אַף נֹחַ מִקְּטַנֵּי אֲמָנָה הָיָה, מַאֲמִין וְאֵינוֹ מַאֲמִין שֶׁיָבֹא הַמַּבּוּל, וְלֹא נִכְנַס לַתֵּבָה עַד שְׁדְּחָקוּהוּ הַמַּיִם:

שנים שנים. כֻּלָם הֻשְׁווּ בְּמִנְיָן זֶה – מִן הַפָּחוּת הָיוּ שְׁנַיִם: באו אל נח. מֵאֵלֵיהֶן:

בחודש השני. רַבִּי אֱלִיעֶזֶר אוֹמֵר זֶה מַרְחֶשְׁוָן; רַבִּי יְהוֹשֻׁעַ אוֹמֵר זֶה אִיָּר (ר"ה י"א): נבקעו. לְהוֹצִיא מֵימֵיהֶן: תהום רבה. מִדָּה כְנֶגֶד מִדָּה, הֵם חָטְאוּ בְּרַבָּה רָעַת הָאָדָם וְלָקוּ בִּתְהוֹם רַבָּה:
ויהי הגשם על הארץ. וּלְהַלָּן הוּא אוֹמֵר וַיְהִי הַמַּבּוּל? אֶלָּא כְּשֶׁהוֹרִידָן הוֹרִידָן בְּרַחֲמִים, שֶׁאִם יַחְזְרוּ יִהְיוּ גִּשְׁמֵי בְרָכָה; כְּשֶׁלֹּא חָזְרוּ הָיוּ לְמַבּוּל: ארבעים יום וגו'. אֵין יוֹם רִאשׁוֹן מִן הַמִּנְיָן, לְפִי שֶׁאֵין לֵילוֹ עִמּוֹ שֶׁהֲרֵי כְתִיב בַּיּוֹם הַזֶּה נִבְקְעוּ כָּל מַעְיְנֹת, נִמְצְאוּ אַרְבָּעִים יוֹם כָּלִים בְּכ"ח בְּכִסְלֵו לְרַבִּי אֱלִיעֱזֶר, שֶׁהַחֳדָשִׁים נִמְנִין כְּסִדְרָן אֶחָד מָלֵא וְאֶחָד חָסֵר, הֲרֵי י"ב מִמַּרְחֶשְׁוָן וְכ"ח מִכִּסְלֵו:
בעצם היום הזה. לִמֶּדְךָ הַכָּתוּב שֶׁהָיוּ בְנֵי דוֹרוֹ אוֹמְרִים אִלּוּ אָנוּ רוֹאִים אוֹתוֹ נִכְנָס לַתֵּבָה, אָנוּ שׁוֹבְרִין אוֹתָהּ וְהוֹרְגִין אוֹתוֹ, אָמַר הַקָּבָּ"ה אֲנִי מַכְנִיסוֹ לְעֵינֵי כֻלָּם וְנִרְאֶה דְבַר מִי יָקוּם:
צפור כל כנף. דָבוּק הוּא, צִפּוֹר שֶׁל כָּל מִין כָּנָף, לְרַבּוֹת חֲגָבִים (כָּנָף לְשׁוֹן נוֹצָה, כְּמוֹ וְשִׁסַּע אֹתוֹ בִכְנָפָיו, שֶׁאֲפִלּוּ נוֹצָתָהּ עוֹלָה, אַף כָּאן צִפּוֹר כָּל מִין מַרְאִית נוֹצָה):

ויסגור ה' בעדו. הֵגֵן עָלָיו שֶׁלֹּא יִשְׁבְּרוּהָ, הִקִּיף הַתֵּבָה דֻּבִּים וַאֲרָיוֹת וְהָיוּ הוֹרְגִים בָּהֶם; וּפְשׁוּטוֹ שֶׁל מִקְרָא סָגַר כְּנֶגֶדוֹ מִן הַמַּיִם, וְכֵן כָּל בְּעַד שֶׁבַּמִּקְרָא לְשׁוֹן כְּנֶגֶד הוּא, בְּעַד כָּל רֶחֶם (בראשית כ׳:י״ח), בַּעֲדֵךְ וּבְעַד בָּנַיִךְ (מ"ב ד'), עוֹר בְּעַד עוֹר (איוב ב'), מָגֵן בַּעֲדִי (תהלים ג'), הִתְפַּלֵּל בְּעַד עֲבָדֶיךָ (שמואל א' י"ב) – כְּנֶגֶד עֲבָדֶיךָ:
ותרם מעל הארץ. מְשֻׁקַּעַת הָיְתָה בַּמַּיִם י"א אַמָּה כִסְפִינָה טְעוּנָה הַמְשֻׁקַּעַת מִקְצָתָהּ בַּמַּיִם; וּמִקְרָאוֹת שֶׁלְּפָנֵינוּ יוֹכִיחוּ:
ויגברו. מֵאֵלֵיהֶן:

חמש עשרה אמה מלמעלה. לְמַעְלָה שֶׁל גֹּבַהּ כָּל הֶהָרִים, לְאַחַר שֶׁהֻשְׁווּ הַמַּיִם לְרָאשֵׁי הֶהָרִים:

נשמת רוח חיים. נְשָׁמָה שֶׁל רוּחַ חַיִּים: אשר בחרבה. וְלֹא דָגִים שֶׁבַּיָּם (סנה' ק"ח):
וימח. לְשׁוֹן וַיִּפְעַל הוּא וְאֵינוֹ לָשׁוֹן וַיִפָּעֵל, וְהוּא מִגִּזְרַת וַיִּפֶן וַיִּבֶן כָּל תֵּבָה שֶׁסּוֹפָהּ הֵ"א, כְּגוֹן בָּנָה, מָחָה, קָנָה, כְּשֶׁהוּא נוֹתֵן וָי"ו יוֹ"ד בְּרֹאשָׁהּ, נָקוּד בְּחִירִק תַּחַת הַיּוֹ"ד: אך נח. לְבַד נֹחַ, זֶהוּ פְּשׁוּטוֹ; וּמִ"אַ גוֹנֵחַ וְכוֹהֶה דָם מִטֹּרַח הַבְּהֵמוֹת וְהַחַיּוֹת. וְיֵ"אֹ שֶׁאִחֵר מְזוֹנוֹת לָאֲרִי וְהִכִּישׁוֹ, וְעָלָיו נֶאֱמַר הֵן צַדִּיק בָּאָרֶץ יְשֻׁלָּם (משלי יא):
