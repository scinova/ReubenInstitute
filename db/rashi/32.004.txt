אֵיכָה יוּעַם זָהָב. קִינָה זוּ נֶאֶמְרָה עַל יֹאשִׁיָּהוּ, כְּמוֹ שֶׁנֶּאֱמַר בְּדִבְרֵי הַיָּמִים "וְהִנָּם כְּתוּבִים עַל הַקִּינוֹת". וְעִמָּהּ חִבֵּר בְּתוֹךְ הַקִּינָה אֶת שְׁאָר בְּנֵי צִיּוֹן: יוּעַם. יִכְהֶה, כְּמוֹ דְאַתְּ מְתַּרְגֵּם "כֵּהָה" עַמְיָא. זָהָב. מַרְאִית פָּנִים הַמַּצְהִיב כְּזָהָב: יִשְׁנֶא. יִשְׁתַּנֶּה מִמַּרְאִיתוֹ: הַכֶּתֶם. קְבוּצַת כְּלֵי נוֹי הַזָּהָב שֶׁהֵם לַעֲדִי, קָרוּי כֶּתֶם: אַבְנֵי קֹדֶשׁ. בָּנִים הַמְאִירִים כַּאֲבָנִים טוֹבוֹת וּמִדְרַשׁ אַגָּדָה: כָּל רְבִיעִית דָּם שֶׁיָּצָא מִיֹּאשִׁיָּהוּ בְּכָל חֵץ וְחֵץ שֶׁנָּעֲצוּ בוֹ הָיָה יִרְמְיָה קוֹבְרָהּ בִּמְקוֹמָהּ, וְעָלֶיהָ הוּא קוֹרֵא: "תִּשְׁתַּפֵּכְנָה אַבְנֵי קֹדֶשׁ":
הַמְסֻלָּאִים בַּפָּז. הַמְהֻלָּלִים וְהַנֶּעֱרָכִים כְּפָז. הָרוֹאֶה אוֹתָם אוֹמֵר: "רְאוּ תָאֳרָם שֶׁל אֵלּוּ כְּמַרְאִית פָּז". וְכֵן "לֹא תְסֻלֶּה בְּכֶתֶם אוֹפִיר", "בְּכֶתֶם טָהוֹר לֹא תְסֻלֶּה" "סֹלּוּ לָרֹכֵב בָּעֲרָבוֹת", לְשׁוֹן הִלּוּל וְעֵרֶךְ הֵן: לְנִבְלֵי חֶרֶשׂ. כַּדֵּי חֶרֶשׂ שֶׁנּוֹתְנִים בָּהֶם יַיִן, כְּגוֹן נִבְלֵי יַיִן "וְנִבְלֵיהֶם יְנַפֵּצוּ":
גַּם תַּנִּין. אַף עַל פִּי שֶׁאַכְזָרִי הוּא, חָלְצוּ שַׁד. כְּשֶׁרוֹאֶה אֶת בְּנוֹ בָּא מֵרָחוֹק רָעֵב, חוֹלֵץ שָׁדָיו מִתּוֹךְ נַרְתִּיקוֹ, שֶׁיֵשׁ לוֹ כִסּוּי עַל דַּדָּיו וּמוֹצִיאוֹ מִתּוֹכוֹ, כְּדֵי שֶׁלֹּא יִרְאֵהוּ בְנוֹ מְכֻסֶּה וַיַחֲזוֹר לַאֲחוֹרָיו, וְהֵינִיקוּ גּוּרֵיהֶן: בַּת עַמִּי לְאַכְזָר. רוֹאִים אֶת בְּנֵיהֶם צוֹעֲקִים לְלֶחֶם, וְאֵין פּוֹרֵשׂ לָהֶם, שֶׁחַיֵּיהֶם קוֹדְמִים לָהֶם לְחַיֵּי בְנֵיהֶם מֵחֲמַת הָרָעָב:

הָאֱמֻנִים עֲלֵי תוֹלָע. עַל בִּגְדֵי צִבְעוֹנִין: הָאֱמֻנִים. לְשׁוֹן "וַיְהִי אֹמֵן אֶת הֲדַסָּה. אַשְׁפַּתּוֹת. אַשְׁפּוֹת שֶׁל זֶבֶל. שׁוֹכְבִים עַל הָאַשְׁפּוֹת בַּחוּץ:
וַיִּגְדַּל עֲוֹן וְגוֹ'. פֻּרְעֲנוּתָם מוֹכִיחַ עֲלֵיהֶם שֶׁגָּדוֹל עֲוֹנָם מִשֶּׁל סְדוֹם: הַהֲפוּכָה כְמוֹ רָגַע. שֶׁהֲרֵי סְדוֹם לֹא נִמְשְׁכָה צָרָתָם. כְּרֶגַע אֶחָד נֶהֶפְכָה: וְלֹא חָלוּ בָהּ יָדָיִם. יְדֵי הָאוֹיֵב, כִּי עַל יְדֵי הַמַּלְאָכִים נֶהֶפְכָה. וְיֵשׁ מִדְרְשֵׁי אַגָּדָה, וְאֵינוֹ מְיֻשָּׁב עַל סֵדֶר הַמִּקְרָאוֹת:
זַכּוּ נְזִירֶיהָ. שָׂרֶיהָ, כְּמוֹ נֵזֶר וְכֶתֶר. וַאֲנִי אוֹמֵר, נְזִירֶיהָ מַמָּשׁ, שֶׁהָיוּ מְגֻדְּלֵי שֵׂעָר וְנָאִים בְּיוֹתֵר, וּמוּסָב עַל "בַּת עַמִּי": אָדְמוּ עֶצֶם מִפְּנִינִים. אוֹתָם שֶׁהָיוּ מַרְאֵיהֶם אָדוֹם מִפְּנִינִים וְגִזְרָתָם כְּמוֹ סַפִּיר, חָשַׁךְ תָּאֳרָם מִפֶּחָמִים:
מִשְּׁחוֹר. הוּא פֶחָם: עֶצֶם. לְשׁוֹן מַרְאֶה, כְּמוֹ "וּכְעֶצֶם הַשָּׁמַיִם לָטֹהַר", קול"ר בְּלַעַ"ז: צָפַד. נִקְמַט וְנִתְחַבֵּר, וְאֵין לוֹ דִמְיוֹן:
שֶׁהֵם יָזֻבוּ מְדֻקָּרִים וְגוֹ'. חַלְלֵי הָרָעָב הָיוּ נְפוּחִים מֵרֵיחַ תְּנוּבוֹת הַשָּׂדֶה, שֶׁהָיוּ הָאוֹיְבִים צוֹלִים בָּשָּׂר עַל הָעֲשָׂבִים חוּץ לַחוֹמָה, וְהָרֵיחַ נִכְנָס בְּתוֹךְ נְפוּחֵי הָרָעָב וּכְרֵסָן נִבְקַעַת, וּפִרְשָׁם זָב, וַהֲרֵי זֶה מִיתַת נִוּוּל יוֹתֵר מֵהֲרוּגֵי חָרֶב: מְדֻקָּרִים. מְבֻקָּעִים. בֵּין נְפִיחַת רָעָב בֵּין בִּקּוּעַ חֶרֶב קָרוּי דְּקִירָה: מִתְּנוּבֹת שָׂדָי. מִשָׁרָשִׁים וַעֲשָׂבִים שֶׁמְּלַקְּטִים וְאוֹכְלִים, רַב פִּרְשָׁם וּמִאוּסָם:
לְבָרוֹת. לְמַאֲכָל, כְּמוֹ "וְלֹא בָרָה אִתָּם לָחֶם", "לְהַבְרוֹת אֶת דָּוִד:
כִּלָּה ה' אֶת חֲמָתוֹ. אֲשֶׁר בָּעֲרָה בוֹ כַּמָּה שָׁנִים, עַתָּה כִלָּה אוֹתָהּ בְּהִנָּקְמוֹ מֵהֶם:

מֵחַטֹּאת נְבִיאֶיהָ. שֶׁל שֶׁקֶר, הָיְתָה לָהּ הָרָעָה הַזֹּאת:
נָעוּ עִוְרִים בַּחוּצוֹת. כְּשֶׁהָיוּ הָעִוְרִים הוֹלְכִים בַּשּׁוּק, הָיוּ נָעִים, וְרַגְלֵיהֶם נִשְׁמָטִים בְּדַם הַהֲרוּגִים שֶׁהָיוּ הָרְשָׁעִים הוֹרְגִים בְּתוֹכָהּ: נְגֹאֲלוּ בַּדָּם. נִתְלַכְלְכוּ בַּדָּם, עַד אֲשֶׁר לֹא יָכְלוּ הַקְּרוֹבִים אֲלֵיהֶם לִגַּע בִּלְבוּשֵׁיהֶם, וְהָיוּ קוֹרְאִים אֲלֵיהֶם:
סוּרוּ. מֵעָלֵינוּ, אַתֶּם הַטְּמֵאִים הַמְלֻכְלָכִים בַּדָּם: כִּי נָצוּ. לְשׁוֹן סִרְחוֹן וְלִכְלוּךְ, כְּמוֹ "מֻרְאָתוֹ בְּנֹצָתָהּ", דִּמְתַרְגְּמִינָן "בְּאוּכְלֵיהּ". כַּךְ חִבְּרוֹ מְנַחֵם: גַּם נָעוּ. נִשְׁמְטוּ בַדָּם:
פְּנֵי ה' חִלְּקָם. פָּנִים שֶׁל זַעַם מֵאֵת הַקָּדוֹשׁ בָּרוּךְ הוּא חִלְּקָם וְהִפְרִידָם בָּעַכּוּ"ם לְמַעַן אֲשֶׁר פְּנֵי הַכֹּהֲנִים לֹא נָשָׂאוּ, בִּהְיוֹתָם בְּשַׁלְוָתָם:
עוֹדֵינוּ תִּכְלֶינָה עֵינֵינוּ אֶל עֶזְרָתֵנוּ הָבֶל. כְּשֶׁבָּאָה עָלֵינוּ הָרָעָה, עֲדַיִן הָיוּ עֵינֵינוּ צוֹפוֹת אֶל חֵיל פַּרְעֹה, שֶׁנֶּאֱמַר בָּהֶם "וּמִצְרַיִם הֶבֶל וָרִיֹק יַעְזֹרוּ", שֶׁהָיוּ מַבְטִיחִים אוֹתָנוּ לְעֶזְרָה וְלֹא בָאוּ, כְּמוֹ שֶׁנֶּאֱמַר בָּהֶם "הִנֵּה חֵיל פַּרְעֹה הַיֹּצֵא לָכֶם לְעֶזְרָה שָׁב לְאַרְצוֹ מִצְרָיִם". מָצִינוּ בְמִדְרַשׁ קִינוֹת שֶׁהָיוּ בָאִים בִּסְפִינוֹת. רָמַז הַקָּדוֹשׁ בָּרוּךְ הוּא לַיָּם וְהֵצִיף לִפְנֵיהֶם נוֹדוֹת נְפוּחִים כְּמִין מֵעֵי אָדָם נִנְעָרִים בַּמָּיִם. אָמְרוּ זֶה לָזֶה, "הַנּוֹדוֹת הַלָּלוּ הֵם אֲבוֹתֵינוּ אַנְשֵׁי מִצְרַיִם שֶׁטֻּבְּעוּ בַיָּם מֵחֲמַת הַיְּהוּדִים הַלָּלוּ, וַאֲנַחְנוּ יוֹצְאִים לְעֶזְרָתָם?" עָמְדוּ וְחָזְרוּ לַאֲחוֹרֵיהֶם: צִפִּינוּ. חִכִּינוּ:
צָדוּ צְעָדֵינוּ. אוֹיְבֵינוּ אָרְבוּ אֶת צְעָדֵינוּ מִלֶּכֶת בִּרְחוֹבוֹתֵינוּ, כְּמוֹ "וַאֲשֶׁר לֹא צָדָה", "וְאַתָּה צֹדֶה אֶת נַפְשִׁי":

מְשִׁיחַ ה'. הוּא יֹאשִׁיָּהוּ, כְּמוֹ שֶׁנֶּאֱמַר בְּדִבְרֵי הַיָּמִים "וַיְקוֹנֵן יִרְמְיָהוּ עַל יֹאשִׁיָּהוּ": בִּשְׁחִיתוֹתָם. בַּגּוּמוֹת שֶׁחָפְרוּ:
שִׂישִׂי וְשִׂמְחִי בַּת אֱדוֹם. נִתְנַבֵּא יִרְמְיָהוּ עַל חֻרְבַּן בַּיִת שֵׁנִי, שֶׁיַּחֲרִיבוּהוּ רוֹמִיִּים: שִׂישִׂי וְשִׂמְחִי. לְפִי שָׁעָה, אֲבָל סוֹפֵךְ שֶׁגַּם עָלַיִךְ תַּעֲבָר כּוֹס הַפֻּרְעָנוּת וְתִשְׁכְּרִי מִמֶּנּוּ: וְתִתְעָרִי. וְתָקִיאִי, כְּמוֹ "וַתְּעַר כַּדָּהּ":
תַּם עֲו‍ֹנֵךְ בַּת צִיּוֹן. לָקִית עַל כָּל חַטֹּאתָיִךְ: לֹא יוֹסִיף לְהַגְלוֹתֵךְ. מִגָּלוּת אֱדוֹם וּלְהַלָּן עוֹד:
