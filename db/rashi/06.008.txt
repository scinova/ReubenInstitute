
תָּבֹזּוּ לָכֶם וְאַל תַּחֲרִימוּ הַשָּׁלָל עוֹד.



הַתִּיקֵנוּ אוֹתָם לְשׁוֹן תִּיק, שֶׁנּוֹצִיאֵם מִתִּיק הָעִיר, דישפורי"ר בְּלַעַז (לְהוֹצִיא, לִשְׁלוֹף). וְיֵשׁ לְפוֹתְרוֹ לְשׁוֹן נְתִיקָה, כְּמוֹ (ירמיהו יב ג): הַתִּיקֵם כְּצֹאן לְטִבְחָה.
וְהוֹרַשְׁתֶּם (תַּרְגּוּם:) וּתְתָרְכוּן.

מִיָּם לָעַי שֶׁהָעַי מִקֶּדֶם לְבֵית אֵל, וּבֵית אֵל מִיָּם לָעַי. וַיָּלֶן יְהוֹשֻׁעַ בְּתוֹךְ הָעָם לִהְיוֹת נָכוֹן בְּהַשְׁכָּמַת הַבֹּקֶר.
וַיַּעַל הוּא כְּמוֹ שֶׁאָמַר לוֹ הַמָּקוֹם, אִם הוּא עוֹבֵר לִפְנֵיהֶם עוֹבְרִין, וְאִם לָאו אֵין עוֹבְרִין.

וַיִּקַּח כַּחֲמֵשֶׁת אֲלָפִים אִישׁ אוֹרֵב אַחַר אוֹרֵב, אֶחָד קָרוֹב לָעִיר מֵחֲבֵירוֹ.
וַיָּשִׂימוּ הָעָם שִׂימָה זוֹ לְשׁוֹן הַזְמָנָה, סָמוּךְ לַחוֹמָה לְהִלָּחֵם, כְּמוֹ שֶׁנֶּאֱמַר בְּבֶן הֲדַד (מלכים א כ ל): שִׂימוּ עַל הָעִיר, וַיָּשִׂימוּ. וְאֶת עֲקֵבוֹ וְאֶת אָרְבּוֹ, כְּמוֹ (בראשית כז לו): וַיַּעְקְבֵנִי. וַיֵּלֶךְ יְהוֹשֻׁעַ בַּלַּיְלָה הַהוּא בְּתוֹךְ הָעֵמֶק אָמְרוּ רַבּוֹתֵינוּ (סנהדרין מד ב) שֶׁלָּן בְּעוֹמְקָהּ שֶׁל הֲלָכָה.
לַמּוֹעֵד לִזְמַן הַיּוֹם, שֶׁנּוֹעֲצוּ יַחַד מֵאֶתְמוֹל בְּאוֹתוֹ שָׁעָה נֵצֵא, שֶׁהָיוּ מְנַחֲשִׁים וּמְעוֹנְנִים.
וַיִּנָּגְעוּ לְשׁוֹן נֶגַע (כְּתַרְגּוּמוֹ:) וְאִתְּבַרוּ, הֶרְאוּ עַצְמָן כְּאִלּוּ הֵם נִגָּפִים לִפְנֵיהֶם.


נְטֵה בַּכִּידוֹן הוּא הָיָה סִימָן לָאוֹרֵב לָצֵאת מִן הַמַּאֲרָב, בִּרְאוֹתוֹ הַכִּידוֹן נָטוּי עַל הָעִיר. כִּידוֹן שפיד"ו בְּלַעַז (רוֹמַח, חַנִית).

יָדַיִם כֹּחַ. וְהָעָם הַנָּס הַמִּדְבָּר יִשְׂרָאֵל שֶׁנָּסוּ אֶל הַמִּדְבָּר כְּמוֹ שֶׁאָמַר לְמַעְלָה, נֶהְפַּךְ לְהִלָּחֵם אֶל הָרוֹדֵף.

וְאֵלֶּה יָצְאוּ מִן הָעִיר הָאוֹרֵב שֶׁהִצִּית אֶת הָעִיר.







אָז יִבְנֶה וְגוֹ' פָּרָשָׁה זוֹ כְּתוּבָה מוּקְדָּם וּמְאוּחָר, שֶׁמִּיּוֹם שֶׁעָבְרוּ אֶת הַיַּרְדֵּן עָשָׂה כֵּן.

וַיִּכְתָּב־שָׁם עַל הָאֲבָנִים הֵן הֵנָּה הָאֲבָנִים הָאֲמוּרִים לְמַעְלָה, לְאַחַר מַעֲשֶׂה זֶה קִפְּלוּ (קלפו) הַסִּיד מֵעֲלֵיהֶם וֶהֱבִיאוּם הַגִּלְגָּל.
לְבָרֵךְ אֶת הָעָם יִשְׂרָאֵל בָּרִאשׁוֹנָה לְהַקְדִּים בְּרָכוֹת לִקְלָלוֹת: בָּרוּךְ הָאִישׁ אֲשֶׁר לֹא יַעֲשֶׂה פֶסֶל וּמַסֵּכָה.
