והיה כי תבוא … וירשתה וישבת בה. מַגִּיד שֶׁלֹּא נִתְחַיְּבוּ בְּבִכּוּרִים עַד שֶׁכָּבְשׁוּ אֶת הָאָרֶץ וְחִלְּקוּהָ (עי' קידושין ל"ז ע"ב):
מראשית. וְלֹא כָל רֵאשִׁית, שֶׁאֵין כָּל הַפֵּרוֹת חַיָּבִים בְּבִכּוּרִים אֶלָּא שִׁבְעַת הַמִּינִין בִּלְבַד, נֶאֱמַר כָּאן "אֶרֶץ" וְנֶאֱמַר לְהַלָּן (דברים ח') "אֶרֶץ חִטָּה וּשְׂעֹרָה וְגוֹ'", מַה לְּהַלָּן מִשִׁבְעַת הַמִּינִים שֶׁנִּשְׁתַּבְּחָה בָהֶן אֶרֶץ יִשְׂרָאֵל, אַף כָּאן שֶׁבַח אֶרֶץ יִשְׂרָאֵל שֶׁהֵן שִׁבְעַת מִינִין (ספרי; מנחות פ"ד): זית שמן. זַיִת אֱגוֹרִי, שֶׁשַּׁמְנוֹ אָגוּר בְּתוֹכוֹ (ספרי; ברכות ל"ט): ודבש. הוּא דְּבַשׁ תְּמָרִים (ספרי): מראשית. אָדָם יוֹרֵד לְתוֹךְ שָׂדֵהוּ וְרוֹאֶה תְאֵנָה שֶׁבִּכְּרָה, כוֹרֵךְ עָלֶיהָ גְּמִי לְסִימָן וְאוֹמֵר "הֲרֵי זוֹ בִכּוּרִים" (ספרי; בכורים ג'):
אשר יהיה בימים ההם. אֵין לְךָ אֶלָּא כֹּהֵן שֶׁבְּיָמֶיךָ כְּמוֹ שֶׁהוּא (ראש השנה כ"ה): ואמרת אליו. שֶׁאֵינְךָ כְּפוּי טוֹבָה: הגדתי היום. פַּעַם אַחַת בַּשָּׁנָה וְלֹא שְׁתֵּי פְעָמִים (ספרי):
ולקח הכהן הטנא מידך. לְהָנִיף אוֹתוֹ, כֹּהֵן מַנִיחַ יָדוֹ תַּחַת יַד הַבְּעָלִים וּמֵנִיף (סוכה מ"ז):
וענית. לְשׁוֹן הֲרָמַת קוֹל (סוטה ל"ב): ארמי אבד אבי. מַזְכִּיר חַסְדֵי הַמָּקוֹם, אֲרַמִּי אֹבֵד אָבִי — לָבָן בִּקֵּשׁ לַעֲקֹר אֶת הַכֹּל כְּשֶׁרָדַף אַחַר יַעֲקֹב, וּבִשְׁבִיל שֶׁחָשַׁב לַעֲשׂוֹת חָשַׁב לוֹ הַמָּקוֹם כְּאִלּוּ עָשָׂה, שֶׁאֻמּוֹת הָעוֹלָם חוֹשֵׁב לָהֶם הַקָּבָּ"ה מַחֲשָׁבָה כְּמַעֲשֶׂה (עי' ספרי): וירד מצרימה. וְעוֹד אֲחֵרִים בָּאוּ עָלֵינוּ לְכַלּוֹתֵנוּ, שֶׁאַחֲרֵי זֹאת יָרַד יַעֲקֹב לְמִצְרַיִם: במתי מעט. בְּשִׁבְעִים נֶפֶשׁ (ספרי):



אל המקום הזה. זֶה בֵּית הַמִּקְדָּשׁ: ויתן לנו את הארץ. כְּמַשְׁמָעוֹ:
והנחתו. מַגִּיד שֶׁנּוֹטְלוֹ אַחַר הֲנָפַת הַכֹּהֵן וְאוֹחֲזוֹ בְיָדוֹ כְּשֶׁהוּא קוֹרֵא וְחוֹזֵר וּמֵנִיף (ספרי; סוכה מ"ז):
ושמחת בכל הטוב. מִכָּאן אָמְרוּ אֵין קוֹרִין מִקְרָא בִכּוּרִים אֶלָּא בִזְמַן שִׂמְחָה, מֵעֲצֶרֶת וְעַד הֶחָג, שֶׁאָדָם מְלַקֵּט תְּבוּאָתוֹ וּפֵרוֹתָיו וְיֵינוֹ וְשַׁמְנוֹ, אֲבָל מֵהֶחָג וָאֵילָךְ מֵבִיא וְאֵינוֹ קוֹרֵא (פסחים ל"ו): אתה והלוי. אַף הַלֵּוִי חַיָּב בְּבִכּוּרִים אִם נָטְעוּ בְּתוֹךְ עָרֵיהֶם: והגר אשר בקרבך. מֵבִיא וְאֵינוֹ קוֹרֵא, שֶׁאֵינוֹ יָכוֹל לוֹמַר לַאֲבוֹתֵינוּ (תלמוד ירושלמי מ"ש פ"ה ה"ה):
כי תכלה לעשר את כל מעשר תבואתך בשנה השלישת. כְּשֶׁתִּגְמֹר לְהַפְרִישׁ מַעַשְׂרוֹת שֶׁל שָׁנָה הַשְּׁלִישִׁית; קָבַע זְמַן הַבִּעוּר וְהַוִּדּוּי בְּעֶרֶב הַפֶּסַח שֶׁל שָׁנָה הָרְבִיעִית, שֶׁנֶּאֱמַר (דברים י"ד) "מִקְצֵה שָׁלֹשׁ שָׁנִים תּוֹצִיא וְגוֹ'", נֶאֱמַר כָּאן "מִקֵּץ", וְנֶאֱמַר לְהַלָּן "מִקֵּץ שֶׁבַע שָׁנִים" לְעִנְיַן הַקְהֵל (דברים ל"א) מַה לְּהַלָן רֶגֶל אַף כָּאן רֶגֶל, אִי מַה לְּהַלָּן חַג הַסֻּכּוֹת אַף כָּאן חַג הַסֻּכּוֹת, תַּלְמוּד לוֹמַר כי תכלה לעשר מַעַשְׂרוֹת שֶׁל שָׁנָה הַשְּׁלִישִׁית, רֶגֶל שֶׁהַמַּעַשְׂרוֹת כָּלִין בּוֹ וְזֶהוּ פֶּסַח, שֶׁהַרְבֵּה אִילָנוֹת יֵשׁ שֶׁנִּלְקָטִין אַחַר הַסֻּכּוֹת; נִמְצְאוּ מַעְשְׂרוֹת שֶׁל שְׁלִישִׁית כָּלִין בְּפֶסַח שֶׁל רְבִיעִית, וְכָל מִי שֶׁשִּׁהָה מַעַשְׂרוֹתָיו הִצְרִיכוֹ הַכָּתוּב לְבַעֲרוֹ מִן הַבַּיִת (שם ה"ו): שנת המעשר. שָׁנָה שֶׁאֵין נוֹהֵג בָּהּ אֶלָּא מַעֲשֵׂר אֶחָד מִשְּׁנֵי מַעַשְׂרוֹת שֶׁנָּהֲגוּ בִשְׁתֵּי שָׁנִים שֶׁלְּפָנֶיהָ, שֶׁשָּׁנָה רִאשׁוֹנָה שֶׁל שְׁמִטָּה נוֹהֵג בָּה מַעֲשֵׂר רִאשׁוֹן, כְּמוֹ שֶׁנֶּאֱמַר (במדבר י"ח) "כִּי תִקְחוּ מֵאֵת בְּנֵי יִשְׂרָאֵל אֶת הַמַּעֲשֵׂר", וּמַעֲשֵׂר שֵׁנִי, שֶׁנֶּאֱמַר (דברים י"ד) "וְאָכַלְתָּ לִפְנֵי ה' אֱלֹהֶיךָ … מַעְשַׂר דְּגָנְךָ תִּירֹשְׁךָ וְיִצְהָרֶךָ", הֲרֵי שְׁנֵי מַעַשְׂרוֹת, וּבָא וְלִמֶּדְךָ כָאן בַּשָּׁנָה הַשְּׁלִישִׁית שֶׁאֵין נוֹהֵג מֵאוֹתָן שְׁנֵי מַעַשְׂרוֹת אֶלָּא הָאֶחָד, וְאֵי זֶה? זֶה מַעֲשֵׂר רִאשׁוֹן, וְתַחַת מַעֲשֵׂר שֵׁנִי יִתֵּן מַעְשַׂר עָנִי, שֶׁנֶּאֱמַר כָּאן ונתתה ללוי אֶת אֲשֶׁר לוֹ — הֲרֵי מַעֲשֵׂר רִאשׁוֹן, לגר ליתום ולאלמנה — זֶה מַעְשַׂר עָנִי (עי' ספרי): ואכלו בשעריך ושבעו. תֵּן לָהֶם כְּדֵי שָׂבְעָן, מִכָּאן אָמְרוּ אֵין פּוֹחֲתִין לֶעָנִי בַּגֹּרֶן פָּחוֹת מֵחֲצִי קַב חִטִּים וְכוּ' (ספרי; תלמוד ירושלמי פאה פ"ח ה"ה):
ואמרת לפני ה' אלהיך. הִתְוַדֵּה שֶׁנָּתַתָּ מַעַשְׁרוֹתֶיךָ: בערתי הקדש מן הבית. זֶה מַעֲשֵׂר שֵׁנִי וְנֶטַע רְבָעִי (ספרי; מע"ש ה"י) וְלִמֶּדְךָ שֶׁאִם שִׁהָה מַעַשְׂרוֹתָיו שֶׁל שְׁתֵּי שָׁנִים וְלֹא הֶעֱלָם לִירוּשָׁלַיִם שֶׁצָּרִיךְ לְהַעֲלוֹתָם עַכְשָׁו: וגם נתתיו ללוי. מַעֲשֵׂר רִאשׁוֹן: וגם. לְרַבּוֹת תְּרוּמָה וּבִכּוּרִים: ולגר ליתום ולאלמנה. זֶה מַעְשַׂר עָנִי (שם): ככל מצותך. נְתַתִּים כְּסִדְרָם, לֹא הִקְדַּמְתִּי תְרוּמָה לְבִכּוּרִים וְלֹא מַעֲשֵׂר לִתְרוּמָה וְלֹא שֵׁנִי לָרִאשׁוֹן, שֶׁהַתְּרוּמָה קְרוּיָה רֵאשִׁית, שֶׁהִיא רִאשׁוֹנָה מִשֶּׁנַעֲשָׂה דָגָן, וּכְתִיב (שמות כ"ב) "מְלֵאָתְךָ וְדִמְעֲךָ לֹא תְאַחֵר" — לֹא תְשַׁנֶּה אֶת הַסֵּדֶר (ספרי): לא עברתי ממצותיך. לֹא הִפְרַשְׁתִּי מִמִּין עַל שֶׁאֵינוֹ מִינוֹ וּמִן הֶחָדָשׁ עַל הַיָּשָׁן (שם): ולא שכחתי. מִלְּבָרֶכְךָ עַל הַפְרָשַׁת מַעַשְׂרוֹת (ברכות מ'):
לא אכלתי באוני ממנו. מִכָּאן שֶׁאָסוּר לְאוֹנֵן (עי' ביכורים פ"ב משנה ב'): ולא בערתי ממנו בטמא. בֵּין שֶׁאֲנִי טָמֵא וְהוּא טָהוֹר, בֵּין שֶׁאֲנִי טָהוֹר וְהוּא טָמֵא (ספרי); וְהֵיכָן הֻזְהַר עַל כָּךְ? "לֹא תוּכַל לֶאֱכֹל בִּשְׁעָרֶיךָ" (דברים י"ב) זוֹ אֲכִילַת טֻמְאָה, כְּמוֹ שֶׁנֶּאֱמַר בִּפְסוּלֵי הַמֻּקְדָּשִׁים (שם ט"ו) "בִּשְׁעָרֶיךָ תֹּאכְלֶנּוּ הַטָּמֵא וְהַטָּהוֹר וְגוֹ'", אֲבָל זֶה לֹא תוּכַל לֶאֱכֹל דֶּרֶךְ אֲכִילַת שְׁעָרֶיךָ הָאָמוּר בְּמָקוֹם אַחֵר (יבמות ע"ג): ולא נתתי ממנו למת. לַעֲשׂוֹת לוֹ אָרוֹן וְתַכְרִיכִין: שמעתי בקול ה' אלהי. הֲבִיאוֹתִיו לְבֵית הַבְּחִירָה: עשיתי ככל אשר צויתני. שָׂמַחְתִּי וְשִׂמַּחְתִּי בוֹ (ספרי; מעש"ש פ"ה י"ב):
השקיפה ממעון קדשך. עָשִׂינוּ מַה שֶּׁגָּזַרְתָּ עָלֵינוּ, עֲשֵׂה אַתָּה מַה שֶּׁעָלֶיךָ לַעֲשׂוֹת (ספרי), שֶׁאָמַרְתָּ (ויקרא כ"ו) "אִם בְּחֻקֹּתַי תֵּלֵכוּ … וְנָתַתִּי גִשְׁמֵיכֶם בְּעִתָּם": אשר נתתה לנו כאשר נשבעת לאבותינו. לָתֵת לָנוּ וְקִיַּמְתָּ — ארץ זבת חלב ודבש :
היום הזה ה' אלהיך מצוך. בְּכָל יוֹם יִהְיוּ בְעֵינֶיךָ חֲדָשִׁים כְּאִלּוּ בוֹ בַיּוֹם נִצְטַוֵּיתָ עֲלֵיהֶם (תנחומא): ושמרת ועשית אותם. בַּת קוֹל מְבָרַכְתּוֹ — הֵבֵאתָ בִכּוּרִים הַיּוֹם, תִּזְכֶּה לְשָׁנָה הַבָּאָה (שם):
האמרת … האמירך. אֵין לָהֶם עֵד מוֹכִיחַ בַּמִּקְרָא, וְלִי נִרְאֶה שֶׁהוּא לְשׁוֹן הַפְרָשָׁה וְהַבְדָּלָה — הִבְדַּלְתָּ לְךָ מֵאֱלֹהֵי הַנֵּכָר לִהְיוֹת לְךָ לֵאלֹהִים וְהוּא הִפְרִישְׁךָ אֵלָיו מֵעַמֵּי הָאָרֶץ לִהְיוֹת לוֹ לְעַם סְגֻלָּה, וּמָצָאתִי לָהֶם עֵד וְהוּא לְשׁוֹן תִּפְאֶרֶת כְּמוֹ (תהלים צ"ד) "יִתְאַמְּרוּ כָּל פֹּעֲלֵי אָוֶן":
כאשר דבר לך. וִהְיִיתֶם לִי סְגֻלָּה:
ולהיותך עם קדש … כאשר דבר. וִהְיִיתֶם לִי קְדֹשִׁים (שמות י״ט:ה׳):
