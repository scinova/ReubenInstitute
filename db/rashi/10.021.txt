וַיְהִי אַחַר הַדְּבָרִים הָאֵלֶּה. עוֹד אַחֶרֶת לְקָרֵב מִיתָתוֹ.
מְחִיר זֶה. הַכֶּרֶם הַזֶּה.





אֶל הַזְּקֵנִים. זִקְנֵי אַשְׁמַאי, סָבֵי דְבַהֲתָא.
קִרְאוּ צוֹם. וּבְיוֹם צוֹם דַּרְכָּם לִבְדֹּק בַּעֲבֵרוֹת שֶׁבְּיָדָם.



בֵּרַךְ נָבוֹת. חִלּוּף שֶׁל קְלָלָה, וְתַרְגּוּם: גָּדִיף נָבוֹת קֳדָם ה' וּמַלְכָּא לָט.

קוּם רֵשׁ. יֵשׁ מֵחֲכָמִים אוֹמְרִים: הֲרוּגֵי מַלְכוּת נִכְסֵיהֶם לַמֶּלֶךְ, וְיֵשׁ מֵהֶן אוֹמְרִים: בֶּן אֲחִי אָבִיו הָיָה, וְאוֹתוֹ וְאֶת בָּנָיו הָרַג, וְרָאוּי לְיָרְשׁוֹ.




הִתְמַכֶּרְךָ. מָכַרְתָּ אֶת עַצְמְךָ לְהַכְעִיס אֶת בּוֹרַאֲךָ. הִתְמַכֶּרְךָ כְּמוֹ: וְהִתְמַכַּרְתֶּם שָׁם לְאוֹיְבֶיךָ.


בְּחֵל יִזְרְעֶאל. בְּבִקְעַת יִזְרְעֶאל, בְּחֵל, תַּרְגּוּמוֹ שֶׁל גַּיְא.

רַק לֹא הָיָה כְאַחְאָב. הַכָּתוּב מֵעִיד עָלָיו, שֶׁלֹּא הָיָה בַּמְּלָכִים כְּמוֹתוֹ, שֶׁיָּרָבְעָם וְכָל אֲשֶׁר אַחֲרָיו, עָבְדוּ אֶת הָעֲגָלִים מִיִּרְאָה, אִם יַעֲלֶה הָעָם יְרוּשָׁלַיִם, תָּשׁוּב הַמַּמְלָכָה לְבֵית דָּוִד, וְזֶה הוֹסִיף הַבַּעַל וְהָאֲשֵׁרָה לְהַכְעִיס. הִתְמַכֵּר. נִמְכַּר לַעֲבוֹדָה זָרָה. רָאִיתִי בִּירוּשַׁלְמִי חִיאֵל בֵּית הָאֱלִי שׁוֹשְׁבִינוֹ הָיָה, וּבְכָל יוֹם שָׁם דָּמָיו וְנָתַן לַעֲבוֹדָה זָרָה.

וַיְהַלֵּךְ אַט. יָחֵף.

לֹא אָבִיא הָרָעָה. רָעַת בֵּיתוֹ לֹא אָבִיא בְּיָמָיו, אֲבָל גְּזֵרַת יָלֹקּוּ הַכְּלָבִים אֶת דָּמוֹ, אִי אֶפְשָׁר לָבֹא בִּימֵי בְּנוֹ, אֶלָּא עָלָיו.
