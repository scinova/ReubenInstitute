וַיֹּאמְרוּ בְנֵי הַנְּבִיאִים. נִסִּים שֶׁנַּעֲשׂוּ לוֹ מְסַדֵּר וְהוֹלֵךְ. רַבּוֹתֵינוּ אָמְרוּ: מִכָּאן שֶׁהָיָה גֵּיחֲזִי דּוֹחֶה הַתַּלְמִידִים מִלְּפָנָיו, וּכְשֶׁנִּטְרַד בָּאוּ תַּלְמִידִים רַבִּים, וְנִדְחַק הַמָּקוֹם מֵהֶם.

הוֹאֶל. הִתְרַצֶּה.

וְאֶת הַבַּרְזֶל. הַנְּסֹרֶת. וְהוּא שָׁאוּל. שְׁאִלְתִּיו, וְאֵין לִי מִמַּה לְשַׁלֵּם.
וַיָּצֶף. צָף עַל הַמַּיִם.

פְּלֹנִי אַלְמֹנִי. כָּסֵי וְטָמִיר, פְּלוֹנִי, לְשׁוֹן: כִּי יִפָּלֵא 'יִתְכַּסֵּי'. אַלְמוֹנִי. מִבְּלִי שֵׁם, שֶׁלֹּא הָיָה רוֹצֶה לְהִתְגַּלּוֹת. תַּחֲנֹתִי. שָׁם אֶחֱנֶה, וְאֶאֱרֹב אֶת מֶלֶךְ יִשְׂרָאֵל אוֹ אֶת גְּדוּדָיו הָעוֹבְרִים לִשְׁלֹל בְּאַרְצִי דֶּרֶךְ אוֹתוֹ הַמָּקוֹם.
מֵעֲבֹר הַמָּקוֹם הַזֶּה. מִלֶּכֶת שָׁם. נְחִתִּים. חוֹנִים.
וַיִּשְׁלַח מֶלֶךְ יִשְׂרָאֵל. הָיָה שׁוֹלֵחַ וְרוֹאֶה אִם אֱמֶת הוּא. וְהִזְהִירֹה. שֶׁהִזְהִיר אִישׁ הָאֱלֹהִים מֵעֲבֹר שָׁם. לֹא אַחַת וְלֹא שְׁתָּיִם. עָשָׂה לוֹ כָּךְ, כִּי אִם פְּעָמִים רַבּוֹת.
עַל הַדָּבָר הַזֶּה. מִי הוּא מְגַלֶּה סוֹדוֹ.

אֵיכֹה. הֵיכָן הוּא.


אֲשֶׁר אִתָּנוּ. בְּעֶזְרָתֵנוּ.

וַיֹּאמַר הַךְ נָא וְגוֹ'. עַל הַקְּלָלָה לֹא הִזְכִּיר אֶת הַשֵּׁם, אֲבָל עַל הַפְּקִיחָה הִזְכִּיר אֶת הַשֵּׁם עַל שְׁתֵּיהֶן, עַל שֶל נַעַר וְעַל שֶל גְּדוּד. סַנְוֵרִים. חֹלִי שֶׁל שִׁמָּמוֹן, רוֹאֶה וְאֵינוֹ יוֹדֵעַ מַה הוּא רוֹאֶה.
וְלֹא זֹה הָעִיר. שֶׁהַנָּבִיא בְּתוֹכָהּ, וֶאֱמֶת אָמַר לָהֶם, שֶׁכְּבָר יָצָא מִמֶּנָּה.


הַאֲשֶׁר שָׁבִיתָ וְגוֹ'. וְכִי דַּרְכְּךָ לַהֲרֹג אוֹתָם שֶׁאַתָּה מֵבִיא שִׁבְיָה.
וַיִּכְרֶה. לְשׁוֹן תִּקּוּן סְעוּדָה.

דִּבְיוֹנִים. זֶבֶל הַזָּב מִן הַיּוֹנִים.
הוֹשִׁיעָה אֲדֹנִי הַמֶּלֶךְ. כְּסָבוּר שֶׁשּׁוֹאֶלֶת מְזוֹנוֹת.


וַתַּחְבִּא אֶת בְּנָהּ. מֵת, וְרוֹצָה לְאָכְלוֹ לְבַדָּהּ.
וַיַּרְא הָעָם. דֶּרֶךְ הַקֶּרַע, אֶת הַשַּׂק תַּחַת בְּגָדָיו.
אִם יַעֲמֹד רֹאשׁ אֱלִישָׁע. שֶׁהַיְכֹלֶת בְּיָדוֹ לְבַקֵּשׁ רַחֲמִים.
וַיִּשְׁלַח אִישׁ מִלְּפָנָיו. הַמֶּלֶךְ שָׁלַח לְהָרְגוֹ, וּבְטֶרֶם יָבֹא אוֹתוֹ הַמַּלְאָךְ אֶל אֱלִישָׁע, וְהוּא נִגְלָה לוֹ בְּרוּחַ הַקֹּדֶשׁ, וְאָמַר אֶל הַזְּקֵנִים: וְהִנֵּה הַמַּלְאָךְ יוֹרֵד אֵלָיו וְהַמֶּלֶךְ אַחֲרָיו.
וַיֹּאמֶר. הַמֶּלֶךְ. הִנֵּה זֹאת הָרָעָה מֵאֵת ה'. זֹאת אַחַת מִן הַקְּלָלוֹת שֶׁקִּלֵּל עַל יְדֵי משֶׁה: וְאָכַלְתָּ פְּרִי בִטְנְךָ בְּמָצוֹר וּבְמָצוֹק וְגוֹ'. מָה אוֹחִיל לַה' עוֹד. לְהוֹשִׁיעַ, הֲלֹא לֹא יוֹעִיל.
