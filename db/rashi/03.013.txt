
שאת או ספחת וגו'. שְׁמוֹת נְגָעִים הֵם וּלְבָנוֹת זוֹ מִזּוֹ: בהרת. חֲבַרְבּוּרוֹת, טיי"א בְּלַעַז, וְכֵן בָּהִיר הוּא בַּשְּׁחָקִים (איוב ל"ז): אל אהרן וגו'. גְּזֵרַת הַכָּתוּב הִיא שֶׁאֵין טֻמְאַת נְגָעִים וְטָהֳרָתָן אֶלָּא עַל פִּי כֹהֵן (ספרא):
הפך לבן. מִתְּחִלָּה שָׁחֹר וְהָפַךְ לְלָבָן בְּתוֹךְ הַנֶּגַע; וּמִעוּט שֵׂעָר שְׁנַיִם: עמק מעור בשרו. כָּל מַרְאֵה לָבָן עָמֹק הוּא, כְּמַרְאֵה חַמָּה עֲמֻקָּה מִן הַצֵּל: וטמא אתו. יֹאמַר לוֹ טָמֵא אַתָּה, שֶׁשֵּׂעָר לָבָן סִימָן טֻמְאָה הוּא גְּזֵרַת הַכָּתוּב:
ועמק אין מראה. לֹא יָדַעְתִּי פֵרוּשׁוֹ: והסגיר. יַסְגִּירֶנּוּ בְּבַיִת אֶחָד וְלֹא יִרְאֶה עַד סוֹף הַשָּׁבוּעַ וְיוֹכִיחוּ סִימָנִים עָלָיו:
בעניו. בְּמַרְאֵהוּ וּבְשִׁעוּרוֹ הָרִאשׁוֹן: והסגירו … שנית. הָא אִם פָּשָׂה בְּשָׁבוּעַ רִאשׁוֹן, טָמֵא מֻחְלָט:
כהה. הָכְהָה מִמַּרְאִיתוֹ, הָא אִם עָמַד בְּמַרְאִיתוֹ אוֹ פָשָׂה טָמֵא: מספחת. שֵׁם נֶגַע טָהוֹר: וכבס בגדיו וטהר. הוֹאִיל וְנִזְקַק לְהִסָּגֵר נִקְרָא טָמֵא וְצָרִיךְ טְבִילָה:

וטמאו הכהן. וּמִשֶּׁטִּמְּאוֹ הֲרֵי הוּא מֻחְלָט, וְזָקוּק לְצִפֳּרִים וּלְתִגְלַחַת וּלְקָרְבָּן הָאָמוּר בְּפָרָשַׁת זֹאת תִּהְיֶה: צרעת הוא. הַמִּסְפַּחַת הַזֹּאת: צרעת. לְשׁוֹן נְקֵבָה. נגע. לְשׁוֹן זָכָר:

ומחית. שינמי"נט בְּלַעַז, שֶׁנֶּהְפַּךְ מִקְצַת הַלֹּבֶן שֶׁבְּתוֹךְ הַשְּׂאֵת לְמַרְאֵה בָשָׂר, אַף הִיא סִימָן טֻמְאָה — שֵׂעָר לָבָן בְּלֹא מִחְיָה וּמִחְיָה בְּלֹא שֵׂעָר לָבָן — וְאַף עַל פִּי שֶׁלֹּא נֶאֶמְרָה מִחְיָה אֶלָּא בַּשְּׂאֵת, אַף בְּכָל הַמַּרְאוֹת וְתוֹלְדוֹתֵיהֶן הוּא סִימָן טֻמְאָה:
צרעת נושנת הוא. מַכָּה יְשָׁנָה הִיא תַּחַת הַמִּחְיָה, וְחַבּוּרָה זוֹ נִרְאֵית בְּרִיאָה לְמַעְלָה וְתַחְתֶּיהָ מְלֵאָה לֵחָה — שֶׁלֹּא תֹאמַר הוֹאִיל וְעָלְתָה מִחְיָה אֲטַהֲרֶנָּה:
מראשו. שֶׁל אָדָם ועד רגליו : לכל מראה עיני הכהן. פְּרָט לְכֹהֵן שֶׁחָשַׁךְ מְאוֹרוֹ (ספרא):

וביום הראות בו בשר חי. אִם צָמְחָה בוֹ מִחְיָה הֲרֵי כְבָר פֵּרֵשׁ שֶׁהַמִּחְיָה סִימָן טֻמְאָה? אֶלָּא הֲרֵי שֶׁהָיָה הַנֶּגַע בְּאֶחָד מֵעֶשְׂרִים וְאַרְבָּעָה רָאשֵׁי אֵבָרִים שֶׁאֵין מִטַּמְּאִין מִשּׁוּם מִחְיָה — לְפִי שֶׁאֵין נִרְאֶה הַנֶּגַע כֻּלּוֹ כְאֶחָד, שֶׁשּׁוֹפְעִין אֵילָךְ וָאֵילָךְ (נגעים פ"ו; קידושין כ"ה) — וְחָזַר רֹאשׁ הָאֵבֶר וְנִתְגַּלָּה שִׁפּוּעוֹ עַל יְדֵי שֻׁמָּן, כְּגוֹן שֶׁהִבְרִיא וְנַעֲשָֹה רָחָב וְנִרְאֵית בּוֹ הַמִּחְיָה, לִמְּדָנוּ הַכָּתוּב שֶׁתְּטַמֵּא (ספרא): וביום. מַה תַּ"ל? לְלַמֵּד יֵשׁ יוֹם שֶׁאַתָּה רוֹאֶה בוֹ וְיֵשׁ יוֹם שֶׁאֵין אַתָּה רוֹאֶה בוֹ, מִכָּאן אָמְרוּ: חָתָן נוֹתְנִין לוֹ כָּל שִׁבְעַת יְמֵי הַמִּשְׁתֶּה, לוֹ וְלִכְסוּתוֹ וּלְבֵיתוֹ, וְכֵן בָּרֶגֶל נוֹתְנִין לוֹ כָּל יְמֵי הָרֶגֶל (נגעים פ"ג, מועד קטן ז'):
צרעת הוא. הַבָּשָׂר הַהוּא: בָּשָׂר לְשׁוֹן זָכָר:


שחין. לְשׁוֹן חִמּוּם, שֶׁנִּתְחַמֵּם הַבָּשָֹר בְּלִקּוּי הַבָּא לוֹ מֵחֲמַת מַכָּה שֶׁלֹּא מֵחֲמַת הָאוּר (חולין ח'): ונרפא. הַשְּׁחִין הֶעֱלָה אֲרוּכָה וּבִמְקוֹמוֹ הֶעֱלָה נֶגַע אַחֵר:
או בהרת לבנה אדמדמת. שֶׁאֵין הַנֶּגַע לָבָן חָלָק, אֶלָּא פָּתוּךְ וּמְעֹרָב בִּשְׁתֵּי מַרְאוֹת — לֹבֶן וְאֹדֶם:
מראה שפל. וְאֵין מַמָּשָׁהּ שָׁפָל, אֶלָּא מִתּוֹךְ לַבְנוּנִיתוֹ הוּא נִרְאֶה שָׁפָל וְעָמֹק, כְּמַרְאֵה חַמָּה עֲמֻקָּה מִן הַצֵּל:

נגע הוא. הַשְּׂאֵת הַזֹּאת אוֹ הַבַּהֶרֶת:
תחתיה. בִּמְקוֹמָהּ: צרבת השחין. כְּתַרְגּוּמוֹ רֹשֶׁם שִׁחֲנָא; — אֵינוֹ אֶלָּא רֹשֶׁם הַחִמּוּם הַנִּכָּר בַּבָּשָֹר; כָּל צָרֶבֶת לְשׁוֹן רְגִיעַת עוֹר הַנִּרְגָּע מֵחֲמַת חִמּוּם, כְּמוֹ וְנִצְרְבוּ בָהּ כָּל פָּנִים (יחזקאל כ"א), רייטרו"ר בְּלַעַז: צרבת. רייטרי"שמנט בְּלַעַז:
מחית המכוה. שנימני"ט; כְּשֶׁחָיְתָה הַמִּכְוָה הָפְכָה לְבַהֶרֶת פְּתוּכָה אוֹ לְבָנָה חֲלָקָה; וְסִימָנֵי מִכְוָה וְסִימָנֵי שְׁחִין שָׁוִים הֵם, וְלָמָּה חִלְּקָן הַכָּתוּב? לוֹמַר שֶׁאֵין מִצְטָרְפִין זֶה עִם זֶה — נוֹלַד חֲצִי גְרִיס בַּשְּׁחִין וַחֲצִי גְרִיס בַּמִּכְוָה, לֹא יִדּוֹנוּ כִגְרִיס (חולין ח'):




בראש או בזקן. בָּא הַכָּתוּב לְחַלֵּק בֵּין נֶגַע שֶׁבִּמְקוֹם שֵׂעָר לְנֶגַע שֶׁבִּמְקוֹם בָּשָׂר, שֶׁזֶּה סִימָנוֹ בְּשֵׂעָר לָבָן וְזֶה סִימָנוֹ בְּשֵׂעָר צָהֹב:
ובו שער צהב. שֶׁנֶּהְפַּךְ שֵׂעָר שָׁחֹר שֶׁבּוֹ לְצָהֹב: נתק הוא. כָּךְ שְׁמוֹ שֶׁל נֶגַע שֶׁבִּמְקוֹם שֵׂעָר:
ושער שחר אין בו. הָא אִם הָיָה בוֹ שֵׂעָר שָׁחֹר טָהוֹר וְאֵין צָרִיךְ לְהֶסְגֵּר, שֶׁהַשֵּׂעָר שָׁחֹר סִימָן טָהֳרָה הוּא בִנְתָקִים, כְּמוֹ שֶׁנֶּאֱמַר (פסוק ל"ז), וְשֵׂעָר שָׁחֹר צָמַח בּוֹ וְגוֹ':
והנה לא פשה וגו'. הָא אִם פָּשָׂה אוֹ הָיָה בוֹ שֵׂעָר צָהֹב טָמֵא:
והתגלח. סְבִיבוֹת הַנֶּתֶק, ואת הנתק לא יגלח. מַנִּיחַ שְׁתֵּי שְׂעָרוֹת סָמוּךְ לוֹ סָבִיב, כְּדֵי שֶׁיְּהֵא נִכָּר אִם פָּשָׂה, שֶׁאִם יִפְשֶׂה יַעֲבֹר הַשְּׂעָרוֹת וְיֵצֵא לִמְקוֹם הַגִּלּוּחַ (נגעים פ"י, ספרא):

אחרי טהרתו. אֵין לִי אֶלָּא פוֹשֶׂה לְאַחַר הַפְּטוֹר, מִנַּיִן אַף בְּסוֹף שָׁבוּעַ רִאשׁוֹן וּבְסוֹף שָׁבוּעַ שֵׁנִי? תַּלְמוּד לוֹמַר פָּשֹׂה יִפְשֶׂה (ספרא):

ושער שחר. מִנַּיִן אַף הַיָּרֹק וְהָאָדֹם, שֶׁאֵינוֹ צָהֹב? תַּלְמוּד לוֹמַר "וְשֵׂעָר"; וּלְשׁוֹן צָהֹב דּוֹמֶה לְתַבְנִית הַזָּהָב — צָהֹב כְּמוֹ זָהֹב, אור"פלא בְּלַעַז: טהור הוא וטהרו הכהן. הָא טָמֵא שֶׁטִּהֲרוֹ הַכֹּהֵן לֹא טָהוֹר:
בהרת. חֲבַרְבּוּרוֹת:
כהות לבנת. שֶׁאֵין לֹבֶן שֶׁלָּהֶן עַז אֶלָּא כּוֹהֶה: בהק. כְּמוֹ לֹבֶן הַנִּרְאֶה בִּבְשַׂר אָדָם אָדֹם, שֶׁקּוֹרִין רוש"ו, בֵּין חֲבַרְבּוּרוֹת אַדְמִימוּתוֹ, קָרוּי בֹּהַק, כְּאִישׁ עַדְשָׁן שֶׁבֵּין עֲדָשָׁה לַעֲדָשָׁה מַבְהִיק הַבָּשָֹר בְּלֹבֶן צַח:
קרח הוא טהור הוא. טָהוֹר מִטֻּמְאַת נְתָקִין, שֶׁאֵינוֹ נִדּוֹן בְּסִימָנֵי רֹאשׁ וְזָקָן שֶׁהֵם מְקוֹם שֵׂעָר, אֶלָּא בְסִימָנֵי נֶגַע עוֹר בָּשָׂר — (בְּשֵֹעָר לָבָן) וּמִחְיָה וּפִשְׂיוֹן (נגעים פ"י):
ואם מפאת פניו. מִשִּׁפּוּעַ קָדְקֹד כְּלַפֵּי פָנָיו קָרוּי גַּבַּחַת — וְאַף הַצְּדָעִין שֶׁמִּכָּאן וּמִכָּאן בִּכְלָל — וּמִשִּׁפּוּעַ קָדְקֹד כְּלַפֵּי אֲחוֹרָיו קָרוּי קָרַחַת:
נגע לבן אדמדם. פָּתוּךְ; מִנַּיִן שְׁאָר הַמַּרְאוֹת? תַּלְמוּד לוֹמַר נֶגַע:
כמראה צרעת עור בשר. כְּמַרְאֵה הַצָּרַעַת הָאָמוּר בְּפָרָשַׁת "עוֹר בָּשָׂר" — אָדָם כִּי יִהְיֶה בְעוֹר בְּשָׂרוֹ, וּמָה אָמוּר בּוֹ? שֶׁמְּטַמֵּא בְאַרְבַּע מַרְאוֹת, וְנִדּוֹן בִּשְׁנֵי שָׁבוּעוֹת; וְלֹא כְמַרְאֵה צָרַעַת הָאָמוּר בִּשְׁחִין וּמִכְוָה שֶׁהוּא נִדּוֹן בְּשָׁבוּעַ אֶחָד, וְלֹא כְמַרְאֵה נְתָקִין שֶׁל מְקוֹם שֵׂעָר, שֶׁאֵין מְטַמְּאִין בְּאַרְבַּע מַרְאוֹת — שְׂאֵת וְתוֹלַדְתָּהּ, בַּהֶרֶת וְתוֹלַדְתָּהּ:
בראשו נגעו. אֵין לִי אֶלָּא נְתָקִין, מִנַּיִן לְרַבּוֹת שְׁאָר הַמְנֻגָּעִים? תַּ"ל "טַמֵּא יְטַמְּאֶנּוּ", לְרַבּוֹת אֶת כֻּלָּן — עַל כֻּלָּן הוּא אוֹמֵר בְּגָדָיו יִהְיוּ פְרֻמִים וְגוֹ':
פרמים. קְרוּעִים: פרוע. מְגֻדַּל שֵׂעָר: ועל שפם יעטה. כְּאָבֵל: שפם. שְׂעַר הַשְּׂפָתַיִם, גרנו"ן בְּלַעַז: וטמא טמא יקרא. מַשְׁמִיעַ שֶׁהוּא טָמֵא וְיִפְרְשׁוּ מִמֶּנּוּ (ספרא, מועד קטן ה'):
בדד ישב. שֶׁלֹּא יִהְיוּ טְמֵאִים יוֹשְׁבִין עִמּוֹ; וְאָמְרוּ רַבּוֹתֵינוּ מַה נִּשְׁתַּנָּה מִשְּׁאָר טְמֵאִים לֵישֵׁב בָּדָד? הוֹאִיל וְהוּא הִבְדִּיל בְּלָשׁוֹן הָרָע בֵּין אִישׁ לְאִשְׁתּוֹ וּבֵין אִישׁ לְרֵעֵהוּ, אַף הוּא יִבָּדֵל (ערכין ט"ז): מחוץ למחנה. חוּץ לְשָׁלוֹשׁ מַחֲנוֹת:

לפשתים ולצמר. שֶׁל פִּשְׁתִּים אוֹ שֶׁל צֶמֶר: או בעור. זֶה עוֹר שֶׁלֹא נַעֲשָׂה בוֹ מְלָאכָה: או בכל מלאכת עור. זֶה עוֹר שֶׁנַּעֲשָׂה בוֹ מְלָאכָה:
ירקרק. יָרֹק שֶׁבַּיְרֻקִּין: אדמדם. אָדֹם שֶׁבָּאֲדֻמִּים:

צרעת ממארת. לְשׁוֹן סִלּוֹן מַמְאִיר (יחזקאל כ"ח), פויי"נט בְּלַעַז; וּמִדְרָשׁוֹ: תֵּן בּוֹ מְאֵרָה שֶׁלֹא תֵהָנֶה הֵימֶנּוּ (ספרא):
בצמר או בפשתים. שֶׁל צֶמֶר אוֹ שֶׁל פִּשְׁתִּים, זֶהוּ פְשׁוּטוֹ; וּמִדְרָשׁוֹ: יָכוֹל יָבִיא גִּזֵּי צֶמֶר וַאֲנִיצֵי פִשְׁתָּן וְיִשְׂרְפֵם עִמּוֹ? תַּ"ל "הוּא" באש תשרף — אֵינָהּ צְרִיכָה דָבָר אַחֵר עִמָּהּ, אִם כֵּן מַה תַּ"ל "בַּצֶּמֶר אוֹ בַפִּשְׁתִּים"? לְהוֹצִיא אֶת הָאֻמְרִיּוֹת שֶׁבּוֹ שֶׁהֵן מִמִּין אַחֵר; אֻמְרִיּוֹת לְשׁוֹן שָׂפָה, כְּמוֹ אִימְרָא (ספרא):

את אשר בו הנגע. יָכוֹל מְקוֹם הַנֶּגַע בִּלְבַד, תַּ"ל "אֵת אֲשֶׁר בּוֹ הַנָּגַע", יָכוֹל כָּל הַבֶּגֶד כֻּלּוֹ טָעוּן כִּבּוּס, תַּ"ל "הַנָּגַע", הָא כֵיצַד? יְכַבֵּס מִן הַבֶּגֶד עִמּוֹ:
אחרי הכבס. לְשׁוֹן הֵעָשׂוֹת: לא הפך הנגע את עינו. לֹא הָכְהָה מִמַּרְאִיתוֹ: והנגע לא פשה. שָׁמַעְנוּ שֶׁאִם לֹא הָפַךְ וְלֹא פָשָׂה טָמֵא — וְאֵין צָרִיךְ לוֹמַר לֹא הָפַךְ וּפָשָׂה — הָפַךְ וְלֹא פָשָׂה אֵינִי יוֹדֵעַ מַה יֵּעָשֶׂה לוֹ, תַּ"ל "וְהִסְגִּיר אֶת הַנֶּגַע" — מִכָּל מָקוֹם, דִּבְרֵי רַ' יְהוּדָה, וַחֲכָמִים אוֹמְרִים וְכוּ', כִּדְאִיתָא בְת"כֹּ, וּרְמַזְתִּיהָ כָּאן לְיַשֵּׁב הַמִּקְרָא עַל אָפְנָיו: פחתת הוא. לְשׁוֹן גֻּמָּא, כְּמוֹ בְּאַחַת הַפְּחָתִים (שמואל ב י"ז), כְּלוֹמַר שְׁפֵלָה הִיא — נֶגַע שֶׁמַּרְאָיו שׁוֹקְעִין (ספרא): בקרחתו או בגבחתו. כְּתַרְגּוּמוֹ בִּשְׁחִיקוּתֵהּ אוֹ בְחַדָּתוּתֵהּ: קרחתו. שְׁחָקִים, יְשָׁנִים; וּמִפְּנֵי הַמִּדְרָשׁ שֶׁהֻצְרַךְ לִגְזֵרָה שָׁוָה — מִנַּיִן לִפְרִיחָה בַבְּגָדִים שֶׁהִיא טְהוֹרָה? נֶאֶמְרָה קָרַחַת וְגַבַּחַת בָּאָדָם, וְנֶאֶמְרָה קָרַחַת וְגַבַּחַת בַּבְּגָדִים, מַה לְּהַלָּן פָּרַח בְּכֻלּוֹ טָהוֹר אַף כָּאן פָּרַח בְּכֻלּוֹ טָהוֹר — לְכָךְ אָחַז הַכָּתוּב לְשׁוֹן קָרַחַת וְגַבַּחַת; וּלְעִנְיַן פֵּרוּשׁוֹ וְתַרְגּוּמוֹ זֶהוּ מַשְׁמָעוֹ: קָרַחַת לְשׁוֹן יְשָׁנִים וְגַבַּחַת לְשׁוֹן חֲדָשִׁים — כְּאִלּוּ נִכְתַּב בְּאַחֲרִיתוֹ אוֹ בְקַדְמוּתוֹ — שֶׁהַקָּרַחַת לְשׁוֹן אֲחוֹרַיִם וְגַּבַּחַת לְשׁוֹן פָּנִים, כְּמוֹ שֶׁכָּתוּב וְאִם מִפְּאַת פָּנָיו וְגוֹ', וְהַקָּרַחַת כָּל שֶׁשּׁוֹפֵעַ וְיוֹרֵד מִן הַקָּדְקֹד וּלְאַחֲרָיו, כָּךְ מְפֹרָשׁ בְּתֹ"כֹּ:
וקרע אתו. יִקְרַע מְקוֹם הַנֶּגַע מִן הַבֶּגֶד וְיִשְׂרְפֶנּוּ:
פרחת הוא. דָּבָר הַחוֹזֵר וְצוֹמֵחַ: באש תשרפנו. אֶת כָּל הַבֶּגֶד:
וסר מהם הנגע. אִם כְּשֶׁכִּבְּסוּהוּ בַתְּחִלָּה עַל פִּי כֹהֵן סָר מִמֶּנּוּ הַנֶּגַע לְגַמְרֵי. וכבס שנית. לְשׁוֹן טְבִילָה; תַּרְגּוּם שֶׁל כִּבּוּסִין שֶׁבְּפָרָשָׁה זוֹ לְשׁוֹן לִבּוּן, "וְיִתְחַוַּר", חוּץ מִזֶּה, שֶׁאֵינוֹ לְלִבּוּן אֶלָּא לִטְבֹּל, לְכָךְ תַּרְגּוּמוֹ "וְיִצְטַבַּע", וְכֵן כָּל כִּבּוּסֵי בְגָדִים שֶׁהֵן לִטְבִילָה מְתֻרְגָּמִין וְיִצְטַבַּע:
