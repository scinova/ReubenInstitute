בראשית, רַב המנונה הזקן אמר:

- מָצָאנוּ אוֹתִיּוֹת בְּהִפּוּךְ! אָלֶ״ף בַּהַתְחָלָה וְאַחַר כָּךְ. אָלֶ״ף בַּהַתְחָלָה - הַיְנוּ 'אֱלֹהִים', 'אֶת' לְאַחַר מִכֵּן.
אֶלָּא כַּד בָּעָא קוּדְשָׁא בְּרִיךְ־הוּא לְמֶעְבַּד עָלְמָא, כׇּל אָתְוָון הֲווּ סְתִימִין, וּתְרֵין־אַלְפִין שְׁנִין עַד דְּלָא בְּרָא עָלְמָא, הֲוָה מִסְתַּכֵּל קוּדְשָׁא בְּרִיךְ־הוּא וְאִשְׁתַּעְשַׁע בְּהוּ.

- כאשר רָצָה לִבְרֹא אֶת הָעוֹלָם, בָּאוּ לְפָנָיו כָּל הָאוֹתִיּוֹת מהסוף לְרֹאשָׁם.


- הִתְחִילָה הָאוֹת ת׳ לְהִכָּנֵס בָּרֹאשׁ. אָמְרָה:
- רִבּוֹן הָעוֹלָמִים! נוֹחַ לְפָנֶיךְ לִבְרֹא בִּי אֶת הָעוֹלָם? שֶׁאֲנִי חותמת החוֹתָם שֶׁלְּךְ, 'אֱמֶת', וְאַתָּה נִקְרָא 'אֱמֶת', נאות לְמֶלֶךְ לְהַתְחִיל בְּאוֹת 'אֱמֶת', וְלִבְרֹא בִּי אֶת הָעוֹלָם.
אָמַר לָהּ הַקָּדוֹשׁ בָּרוּךְ־הוּא:
- נאותה אַתְּ, וְזַכָּאִית אַתְּ! אֶלָּא _שֶׁ_אֵין את כְּדָאִית לִבְרֹא בָּךְ אֶת הָעוֹלָם, הוֹאִיל וְאַתְּ עֲתִידָה לִהְיוֹת רְשׁוּמָה עַל מִצְחֵי אֲנָשִׁים נֶּאֱמָנִים, שֶׁקִּיְּמוּ אֶת הַתּוֹרָה מֵאָלֶ״ף וְעַד תָּי״ו, וברישום שֶׁלָּךְ יָמוּתוּ! וְעוֹד, _מִשּׁוּם_ שֶׁאַתְּ חותמת הַ'מָּוֶת', וְהוֹאִיל וְאַתְּ כָּךְ, אֵין את כְּדָאִית לִבְרֹא בָּךְ אֶת הָעוֹלָם.
מִיָּד יָצְאָה.


- נִכְנְסָה הָאוֹת שֶׁ׳ לְפָנָיו. אָמְרָה לְפָנָיו:
- רִבּוֹן עוֹלָמִים! נוֹחַ לְפָנֶיךְ לִבְרֹא בִּי אֶת הָעוֹלָם? מִשּׁוּם שֶׁבִּי נִקְרַא שִׁמְךָ ‘שַׁדַּי’, וְיָאוּת לִבְרֹא אֶת הָעוֹלָם בּשֵׁם הַקָדוֹשׁ.
אָמַר לָהּ:
- נאותה אַתְּ, וְטוֹבָה אַתְּ, וֶאֱמֶת אַתְּ! אֲבָל הוֹאִיל וְאוֹתִיּוֹת הַזִּיּוּף נוטלות אוֹתָךְ לִהְיוֹת עִמָּהֶן, לֹא רוּצֵנִי לִבְרֹא בָּךְ אֶת הָעוֹלָם, מִשּׁוּם שֶׁלֹּא יִתְקַיֵּם הַשֶּׁקֶר, אֶלָּא אִם יִטְלוּ אוֹתָךְ ק׳ _ו־_ר׳.


מִכָּאן, מִּי שֶׁרוֹצֶה לוֹמַר שֶׁקֶר, יִטֹּל יְסוֹד שֶׁל אֱמֶת ~בַּהַתְחָלָה~, ולאחר מִכֵּן ~יְקַיֵּם~ לוֹ אֶת הַשֶּׁקֶר. שֶׁהֲרֵי הָאוֹת שֶׁ׳ - אוֹת אֱמֶת הִיא, אוֹת אֱמֶת שֶל אָבוֹתֶינוּ, שֶׁהִתְיַחֲדוּ בָהּ. ק׳ ו־ר׳ - אוֹתִיּוֹת שֶׁנִּרְאוּ עַל הַצַּד הָרָע הֵם, וּכְדֵי לְהִתְקַיֵּם, נָטְלוּ אֶת הָאוֹת שֶׁ׳ בְּתוֹכָן וְנִהְיָה 'קֶשֶׁר'.
כֵּיוָן שֶׁרָאֲתָה כָּךְ, יָצְאָה מִלְּפָנָיו.


- נִכְנְסָה הָאוֹת צ׳. אָמְרָה לְפָנָיו:
- רִבּוֹן הָעוֹלָם! נוֹחַ לְפָנֶיךְ לִבְרֹא בִּי _אֶת_ הָעוֹלָם? _מִשּׁוּם_ שֶׁאֲנִי - בִּי חֲתוּמִים הַצַּדִּיקִים, וְאַתָּה, שֶׁנִּקְרֵאתָ 'צַדִּיק', בִּי רָשׁוּם, שֶׁכָּתוּב: “כִּי צַדִּיק יְהֹוָה צְדָקוֹת אָהֵב” (תהלים יא ז), וּבִי יָאוּת לִבְרֹא אֶת הָעוֹלָם.
אָמַר לָהּ:
- צָדִ״י, 'צָדִיק' אַת, ו'צַּדּיּק' אַת נקראת, אֲבָל אַתְּ צְרִיכָה לִהְיוֹת טְמוּנָה, אֵינֵךְ צְרִיכָה לְהִתְגַּלּוֹת כָּל כָּךְ, כְּדֵי שֶׁלֹּא לָתֵת פִּתְחוֹן פֶּה לָעוֹלָם.
מַהוּ הַטַּעַם? נ׳ הִיא, בָּאָה הַ־י׳ שֶׁל הַשֵּׁם שֶׁל הַבְּרִית הַקָּדוֹשׁ, וְרוֹכֵב עָלֶיהָ, ונאחָז עִמָּהּ. וְהַסּוֹד הוּא זֶה: כְּשֶׁבָּרָא הַקָּדוֹשׁ בָּרוּךְ־הוּא אֶת הַאָדָם הָרִאשׁוֹן, דּוּ פַּרְצוּפִים בָּרָאוֹ. וּמִשּׁוּם כָּךְ פָּנָיו שֶׁל יוֹ״ד חוֹזֵר לְאָחוֹר כְּמוֹ זֶה: צ׳; וְלֹא חוֹזְרִים פָּנִים בְּפָנִים כְּמוֹ זֶה: צ׳; מִסְתַּכֵּל לְמַעְלָה כְּמוֹ זֶה: צ׳; _אוֹ_ מִסְתַּכֵּל לְמַטָּה כְּמוֹ זֶה: צ׳.
אָמַר לָהּ הַקָּדוֹשׁ בָּרוּךְ־הוּא:
- עוֹד, שֶׁאֲנִי עָתִיד לְנַסֵּר אוֹתָךְ וְלַעֲשׂוֹת אוֹתָךְ פָּנִים בְּפָנִים, אֲבָל בְּמָקוֹם אַחֵר תִּתְעַלִּי.
יָצְאָה מִלְּפָנָיו וְהָלְכָה.


- נִכְנְסָה הָאוֹת פ׳. אָמְרָה לְפָנָיו:
- רִבּוֹן עוֹלָמִים! נוֹחַ לְפָנֶיךְ לִבְרֹא בִּי אֶת הָעוֹלָם? שֶׁהֲרֵי הגאולה שֶׁאַתָּה עָתִיד לַעֲשׂוֹת בָּעוֹלָם - בִּי רְשׁוּמָה, וְזוֹ הִיא 'פְדוּת', וּבִי יָאוּת לִבְרֹא אֶת הָעוֹלָם.
אָמַר לָהּ:
- נאותה אַתְּ, אֲבָל בָּךְ רָשׁוּם 'פֶּשַׁע' בְּהֶסְתֵּר, כְּמוֹ שֶׁהַנָּחָשׁ שֶׁמַּכֶּה וּמַכְנִיס _אֶת_ רֹאשׁוֹ בֵּין גּוּפוֹ, כָּךְ מִי שֶׁחוֹטֵא, כּוֹפֵף רֹאשׁוֹ וּמוֹצִיא יָדָיו. וְכֵן ע׳ - _הִיא_ 'עָוֹן', אַף עַל פִּי שֶׁאָמְרָה שֶׁיֵּשׁ בָּהּ 'עֲנָוָה'.
אָמַר לָהּ הַקָּדוֹשׁ בָּרוּךְ־הוּא:
- לֹא אֶבְרָא בָּךְ אֶת הָעוֹלָם.
יָצְאָה מִלְּפָנָיו.


נִכְנְסָה הָאוֹת ס׳. אָמְרָה לְפָנָיו:
רִבּוֹן {3a} הָעוֹלָמִים! נוֹחַ לְךְ לִבְרֹא בִּי אֶת הָעוֹלָם? שֶׁיֵּשׁ בִּי סְמִיכָה לַנּוֹפְלִים, שֶׁכָּתוּב: “סוֹמֵךְ יְהֹוִה לְכָל הַנּוֹפְלִים” (תהלים קמה יָד).
אָמַר לָהּ:
עַל כֵּן, אַתְּ נצרכת לִמְקוֹמֵךְ, וְאַל תָּזוּזִי מִמֶּנּוּ! אִם אַתְּ יוֹצֵאת מִמְּקוֹמֵךְ, מַה יִּהְיֶה עֲלֵיהֶם הַאוֹתָם הַנּוֹפְלִים?! הוֹאִיל וְהֵם סוֹמְכִים עָלַיִךְ!
מִיָּד יָצְאָה מִלְּפָנָיו.


נִכְנְסָה הָאוֹת נ׳. אָמְרָה לְפָנָיו:
רִבּוֹן הָעוֹלָם! נוֹחַ לְפָנֶיךְ לִבְרֹא בִּי אֶת הָעוֹלָם? שֶׁבִּי כָּתוּב “נוֹרָא תְהִלֹּת” (שמות טו יא), וּתְהִלָּה לַצַּדִּיקִים “נָאוָה תְהִלָּה” (תהילים לג א).
אָמַר לָהּ:
נוּ״ן, שׁוּבִי לִמְקוֹמֵךְ! שֶׁהֲרֵי בגללך שָׁבָה סָמֶ״ךְ לִמְקוֹמָהּ, וְהְיִי סְמוּכָה עָלֶיהָ!
מִיָּד שָׁבָה לִמְקוֹמָהּ וְיָצְאָה מִלְּפָנָיו.


נִכְנְסָה הָאוֹת מ׳. אָמְרָה לְפָנָיו:
רִבּוֹן הָעוֹלָם! נוֹחַ לְפָנֶיךְ לִבְרֹא בִּי אֶת הָעוֹלָם? שֶׁבִּי נִקְרֵאתָ ‘מֶלֶךְ’.
אָמַר לָהּ:
כָּךְ הוּא וַדַּאי! אֲבָל לֹא אֶבְרָא בָּךְ אֶת הָעוֹלָם, מִשּׁוּם שֶׁהָעוֹלָם נצטרך למֶלֶךְ. שׁוּבִי לִמְקוֹמֵךְ! אַתְּ וְ-ל׳ וְ-ךָּ׳! שֶׁהֲרֵי לֹא יָאֶה לָעוֹלָם לַעֲמֹד בְּלִי מֶלֶךְ.


בְּאוֹתָהּ שָׁעָה יָרְדָה מִלְּפָנָיו הָאוֹת כ׳ מעַל כּסּא כבודו, הִזְדַּעְזְעָה וְאָמְרָה לְפָנָיו:
רִבּוֹן הָעוֹלָם! נוֹחַ לְפָנֶיךְ לִבְרֹא בִּי אֶת הָעוֹלָם? שֶׁאֲנִי כְּבוֹדְך.
וכְשֶׁיָּרְדָה כ׳ מֵעַל כִּסֵּא כְבוֹדוֹ, הִזְדַּעְזְעוּ מָאתַיִם־אֶלֶף עוֹלָמוֹת, וְהִזְדַּעְזַע הַכִּסֵּא, וְכָל הָעוֹלָמוֹת הִזְדַּעְזְעוּ לִפֹּל.
אָמַר לָהּ הַקָּדוֹשׁ בָּרוּךְ־הוּא:
כָּ״ף, כָּ״ף, מָה אַתְּ עוֹשָׂה כָּאן? שֶׁלֹּא אֶבְרָא בָּךְ אֶת הָעוֹלָם. שׁוּבִי לִמְקוֹמֵךְ! שֶׁהֲרֵי בָּךְ 'כְּלָיָה' - “כָּלָה וְנֶחֱרָצָה” (ישעיה כח כב) נִשְׁמָע. שׁוּבִי לְכִסְאֵךְ וְהְיִי שָׁם!
בְּאוֹתָהּ שָׁעָה יָצְאָה מִלְּפָנָיו, וְשָׁבָה לִמְקוֹמָהּ.


נִכְנְסָה הָאוֹת י׳. אָמְרָה לְפָנָיו:
רִבּוֹן הָעוֹלָם! נוֹחַ לְפָנֶיךְ לִבְרֹא בִּי אֶת הָעוֹלָם? מִשּׁוּם שִּאַניּ ראשּיּת הַשּם הַקָּדוֹשׁ, וְרָאוּי לְךְ לִבְרֹא בִּי אֶת הָעוֹלָם.
אָמַר לָהּ:
דַּי לָךְ שֶׁאַתְּ חֲקוּקָה בִּי, וְאַתְּ רְשׁוּמָה בִּי, וְכָל רְצוֹנִי+ בָּךְ עוֹלֶה, אֵין את נאותה לְהֵעָקֵר מִשְּׁמִי!


נִכְנְסָה הָאוֹת ט׳. אָמְרָה לְפָנָיו:
רִבּוֹן הָעוֹלָם! נוֹחַ לְפָנֶיךְ לִבְרֹא בִּי אֶת הָעוֹלָם? מִשּׁוּם שֶׁאַתָּה - בִּי נִקְרֵאתָ ‘טוֹב וְיָשָׁר’.
אָמַר לָהּ:
לֹא אֶבְרָא בָּךְ אֶת הָעוֹלָם, שֶׁהֲרֵי טוּבֵךְ נִסְתָּר בְּתוֹכֵךְ, וְצָפוּן בְּתוֹכֵךְ, זֶהוּ שֶׁכָּתוּב: ”מָה רַב טוּבְךְ אֲשֶׁר צָפַנְתָּ לִּירֵאֶיךְ” (תהלים לא ב). הוֹאִיל וְגָנוּז בְּתוֹכֵךְ, אֵין בּוֹ חֵלֶק לָעוֹלָם הַזֶּה שֶׁאֲנִי רוֹצֶה לִבְרֹא, אֶלָּא בָּעוֹלָם הַבָּא.
וְעוֹד, שֶׁעַל שֶׁטּוּבֵךְ גָּנוּז בְּתוֹכֵךְ, יִטְבְּעוּ שַׁעֲרֵי הַהֵיכָל, זֶהוּ שֶׁכָּתוּב: “טָבְעוּ בָאָרֶץ שְׁעָרֶיהָ” (איכה ב ט).
וְעוֹד, שֶׁח׳ כְּנֶגְדֵּךְ, וּכְשֶׁתִּתְחַבְּרוּ יַחַד - הִנֵּה ח״ט! וְעַל זה, הָאוֹתִיּוֹת הַלָּלוּ לֹא רְשׁוּמוֹת בַּשְּׁבָטִים הַקְּדוֹשִׁים.
מִיָּד יָצְאָה מִלְּפָנָיו.


נִכְנְסָה הָאוֹת ז׳. אָמְרָה לוֹ:
רִבּוֹן הָעוֹלָם! נוֹחַ לְפָנֶיךְ לִבְרֹא בִּי אֶת הָעוֹלָם? שֶׁבִּי שׁוֹמְרִים בָּנֶיךְ אֶת הַשַּׁבָּת, שֶׁכָּתוּב: “זָכוֹר אֶת יוֹם הַשַּׁבָּת לְקַדְּשׁוֹ” (שמות כ ח).
אָמַר לָהּ:
לֹא אֶבְרָא בָּךְ אֶת הָעוֹלָם, שֶׁאת - יֵשׁ בָּךְ קְרָב, וְחֶרֶב שֶׁשְׁנוּנָה, וְרֹמַח שֶׁל קְרָב, כְּמוֹ שֶׁלַּנּוּ״ן.
מִיָּד יָצְאָה מִלְּפָנָיו.


נִכְנְסָה הָאוֹת ו׳. אָמְרָה לְפָנָיו:
רִבּוֹן הָעוֹלָם! נוֹחַ לְפָנֶיךְ לִבְרֹא בִּי אֶת הָעוֹלָם? מִשּׁוּם שֶׁאֲנִי אוֹת מִשִּׁמְךְ.
אָמַר לָהּ:
וָא״ו, אַתְּ וְה״א, דַּי לָכֶם שֶׁאַתֶּן אוֹתִיּוֹת שֶׁל שְׁמִי, שֶׁאַתֶּן בַּסּוֹד שֶׁל שְׁמִי, וַחֲקוּקוֹת וּגְלוּפוֹת בִּשְׁמִי, וְלֹא אֶבְרָא בָכֶן אֶת הָעוֹלָם.


נִכְנְסָה הָאוֹת ד׳ וְהָאוֹת ג׳, אָמְרוּ אַף הֵן כָּךְ. אָמַר אַף לָהֶן:
דַּי לָכֶן לִהְיוֹת זוֹ עִם זוֹ, שֶׁהֲרֵי עֲנִיִּים לֹא יִתְבַּטְּלוּ מִן הָעוֹלָם, וּצְרִיכִים לִגְמֹל עִמָּהֶם חֶסֶד. דָּלֶ״ת הוּא הֶעָנִי, גִּימֶ״ל גּוֹמֶלֶת לָהּ חֶסֶד, אַל תִּפָּרְדוּ זוֹ מִזּוֹ, וְדַי, לָכֶן לָזוּן זוֹ אֶת זוֹ!


נִכְנְסָה הָאוֹת ב׳. אָמְרָה לוֹ:
רִבּוֹן הָעוֹלָם! נוֹחַ לְפָנֶיךְ לִבְרֹא בִּי אֶת הָעוֹלָם? שֶׁבִּי מְבָרְכִים אוֹתְךְ לְמַעְלָה וּלְמַטָּה.
אָמַר לָהּ הַקָּדוֹשׁ בָּרוּךְ־הוּא:
הֲרֵי וַדַּאי שֶׁבָּךְ אֶבְרָא אֶת הָעוֹלָם, וְאַתְּ תִּהְיִי ההתחלה לִבְרֹא אֶת הָעוֹלָם.


עָמְדָה הָאוֹת א׳, וְלֹא נִכְנְסָה. אָמַר לָהּ הַקָּדוֹשׁ בָּרוּךְ־הוּא:
- אָלֶ״ף, אָלֶ״ף! לָמָּה אֵין את נִכְנֶסֶת לְפָנַי כִּשְׁאָר כָּל הָאוֹתִיּוֹת?
אָמְרָה לְפָנָיו:
- רִבּוֹן הָעוֹלָם! מִשּׁוּם שֶׁרָאִיתִי שֶׁכָּל הָאוֹתִיּוֹת יָצְאוּ מִלְּפָנֶיךְ בְּלִי תוֹעֶלֶת, מָה אֲנִי אֶעֱשֶׂה שָׁם?
- וְעוֹד, שֶׁהֲרֵי נָתַתָּ לָאוֹת בֵּי״ת אוֹצָר גָּדוֹל זֶה, וְלֹא יָאֶה לְמֶלֶךְ עֶלְיוֹן לְהַעֲבִיר אוֹצָר שֶׁנָּתַן לְעַבְדּוֹ וְלָתֵת לְאַחֵר.
אָמַר לָהּ הַקָּדוֹשׁ בָּרוּךְ־הוּא:
- אָלֶ״ף, אָלֶ״ף, אַף עַל פִּי שֶׁאוֹת בֵּי״ת -  בה אֶבְרָא _אֶת_ הָעוֹלָם, אַתְּ תִּהְיִי הָרֹאשׁ לְכָל הָאוֹתִיּוֹת. אֵין בִּי יִחוּד אֶלָּא בָּךְ. בָּךְ יַתְחִילוּ כָּל חֶשְׁבּוֹנוֹת וְכָל מַעֲשֵׂי הָעוֹלָם, וְכָל יִחוּד לֹא נהיה אֶלָּא בָּאוֹת אָלֶ״ף.


וְעָשָׂה הַקָּדוֹשׁ בָּרוּךְ־הוּא אוֹתִיּוֹת עֶלְיוֹנוֹת גְּדוֹלוֹת, וְאוֹתִיּוֹת תַּחְתּוֹנוֹת קְטַנּוֹת. וּמִשּׁוּם כָּךְ:
בֵּי״ת, בֵּי״ת - "בְּרֵאשִׁית בָּרָא".
אָלֶ״ף, אָלֶ״ף - "אֱלֹהִים אֵת".
אוֹתִיּוֹת מִלְמַעְלָה וְאוֹתִיּוֹת מִלְּמַטָּה, וְכֻלָּן כְּאֶחָד היו, מֵהָעוֹלָם הָעֶלְיוֹן וּמֵהָעוֹלָם הַתַּחְתּוֹן.
