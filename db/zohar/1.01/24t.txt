מִצְוָה רְבִיעִית: לָדַעַת שֶׁיְהֹוָה הוּא הָאֱלֹהִים, כְּמוֹ שֶׁאתה אוֹמֵר: "וְיָדַעְתָּ הַיּוֹם וַהֲשֵׁבֹתָ אֶל־לְבָבֶךְ כִּי יְהֹוָה הוּא הָאֱלֹהִים" (דְּבָרִים ד לט). וְשֶׁיִּכָּלֵל הַשֵּׁם שֶׁל 'אֱלֹהִים' בַּשֵּׁם שֶׁל 'יְהֹוָה', לָדַעַת שֶׁהֵם אֶחָד, וְאֵין בָּהֶם פֵּרוּד.

וְהַיְנוּ, סוֹד הַכָּתוּב "יְהִי מְאֹרֹת בִּרְקִיעַ הַשָּׁמַיִם לְהָאִיר עַל הָאָרֶץ": שֶׁיִּהְיוּ שְׁנֵי הַשֵּׁמוֹת אֶחָד, בְּלִי פֵרוּד כְּלָל, שֶׁיִּכָּלֵל מְאֹרֹת חָסֵר בַּשֵּׁם שֶׁל 'שָׁמַיִם', שֶׁהֵם אֶחָד, וְאֵין בָּהֶם פֵּרוּד. אוֹר שָׁחֹר בְּאוֹר לָבָן - אֵין בָּהֶם פֵּרוּד, וְהַכֹּל אֶחָד.
וְזֶה הוּא עָנָן הַלָּבָן שֶׁל הַיּוֹם, וְעָנָן שֶׁל אֵשׁ בַּלַּיְלָה; 'מִדַּת יוֹם', וְ'מִדַּת לַיְלָה'; ~וְיִתְתַּקְנוּ~ זֶה עִם זֶה לְהָאִיר, כְּמוֹ שֶׁנֶּאֱמַר: "לְהָאִיר עַל הָאָרֶץ".

וְזְהִוּ הַחֵטְא שֶׁל הַנָּחָשׁ הַקַּדְמוֹנִי הַהוּא: חִבֵּר לְמַטָּה, וְהִפְרִיד לְמַעְלָה. וּמִשּׁוּם כָּךְ, גָּרַם מַה שֶּׁגָּרַם לָעוֹלָם, מִשּׁוּם שֶׁצָּרִיךְ לְהַפְרִיד לְמַטָּה, וּלְחַבֵּר לְמַעְלָה.
וְאוֹר שָׁחֹר צָרִיךְ לְהֵאָחֵז לְמַעְלָה בְּחִבּוּר אֶחָד, וּלְהֵאָחֵז אַחַר כָּךְ בְּאוּכְלוּסֶיהָ בְּיִחוּדָהּ, וּלְהַפְרִיד אוֹתָהּ מהַצַּד הָרָע.

וְעִם כָּל זֶה, צָרִיךְ לָדַעַת שֶׁאֱלֹהִים, יְהֹוָה - הַכֹּל אֶחָד, בְּלִי פֵרוּד. יְהֹוָה הוּא הָאֱלֹהִים! וְכַאֲשֶׁר יֵּדַע בן־אָדָם שֶׁהַכֹּל אֶחָד, וְלֹא יָשִׂים פֵּרוּד, אֲפִלּוּ הַצַּד הָאַחֵר הַהוּא יִסְתַּלֵּק מעל הָעוֹלָם, וְלֹא יִמָּשֵׁךְ לְמַטָּה.

וְהַיְנוּ, סוֹד הַכָּתוּב "וְהָיוּ לִמְאֹרוֹת": הֲרֵי קְּלִפָּה אַחַר הַמֹּחַ עוֹלָה. הַמֹּחַ - 'אוֹר', הַצַּד הָאַחֵר - 'מָוֶת'; 'אוֹר' בְּחִבּוּר שֶׁל אוֹתִיּוֹת, 'מָוֶת' בְּפֵרוּד. וְכַאֲשֶׁר הָ'אוֹר' הַזֶּה נסְתַּלֵּק מִשָּׁם, מִתְחַבֵּר בְּאוֹתִיּוֹת שֶׁל פֵּרוּד 'מָוֶת'.

מֵהָאוֹתִיּוֹת הַלָּלוּ הִתְחִילָה חַוָּה, וְגָרְמָה רַע עַל הָעוֹלָם, כְּמוֹ שכָּתוּב: "וַתֵּרֶא הָאִשָּׁה כִּי טוֹב" (בראשית ג ו). הֶחֱזִירָה את הָאוֹתִיּוֹת לְמַפְרֵעַ, וְנִשְׁאַר מֵ״ם וי״ו, וְהֵם הָלְכוּ וְנָטְלוּ את הָאוֹת תָּי״ו עִמָּם, וְגָרְמָה מָוֶת עַל הָעוֹלָם, כְּמוֹ שֶׁכָּתוּב: "וַתֵּרֶא".

אָמַר רַבִּי אֶלְעָזָר:
- אַבָּא, הֲרֵי לָמַדְנוּ: מֵ״ם נִשְׁאֲרָה יְחִידָה, וָוי״ו - שֶׁהִיא חַיִּים תָּמִיד - מִתְהַפֶּכֶת, וְהוֹלֶכֶת, וְנוֹטֶלֶת תָּי״ו, שֶׁכָּתוּב: "וַתִּקַּח וַתִּתֵּן", וְנִשְׁלְמָה תֵּבָה זּוֹ, וְהִתְחַבְּרוּ הָאוֹתִיּוֹת.

אָמַר לוֹ:
- בָּרוּךְ אַתָּה בְּנִי! וַהנה הֶעֱמַדְנוּ את הַדָּבָר הַזֶּה.
