מִצְוָה עֲשִׂירִית

מִצְוָה עֲשִׂירִית: לְהַנִּיחַ תְּפִלִּין וּלְהַשְׁלִים אֶת עַצְמוֹ, בַּדְּמוּת הָעֶלְיוֹנָה, שֶׁכָּתוּב: "וַיִּבְרָא אֱלֹהִים אֶת־הָאָדָם בְּצַלְמוֹ" (בראשית א כז).
פָּתַח וְאָמַר: "רֹאשֵׁךְ עָלַיִךְ כַּכַּרְמֶל" (שיר השירים ז ו). את הַפָּסוּק הַזֶּה הֶעֱמַדְנוּ, וְנאמר. אֲבָל "רֹאשֵׁךְ עָלַיִךְ כַּכַּרְמֶל" - זֶה הָרֹאשׁ הָעֶלְיוֹן, תְּפִלִּין שֶׁל רֹאשׁ, השם שֶׁל הַמֶּלֶךְ הַקָּדוֹשׁ העליון - יְהֹוָה - בְּאוֹתִיּוֹת רְשׁוּמוֹת, וְכָל אוֹת וְאוֹת פָּרָשָׁה אַחַת, הַשֵּׁם הַקָּדוֹשׁ גלוף בְּסִדּוּר שֶׁל הָאוֹתִיּוֹת כיאות.
וְשָׁנִינוּ: "כִּי שֵׁם יְהֹוָה נִקְרָא עָלֶיךְ וְיָרְאוּ מִמֶּךָּ" (דברים כח י) - אֵלּוּ תְפִלִּין שֶׁל רֹאשׁ, שֶׁהֵם השֵׁם קָדוֹשׁ בְּסִדּוּר שֶׁל האוֹתִיּוֹתָיו.

הַפָּרָשָׁה הַרִאשׁוֹנָה: "קַדֶּשׁ־לִי כָל־בְּכוֹר" (שמות יג ב), זוֹ יו״ד שֶׁהִיא קֹדֶשׁ, בְּכוֹר כָּל הַקְּדָשִׁים הָעֶלְיוֹנִים. "פֶּטֶר כָּל־רֶחֶם" - בַּשְּׁבִיל הַדַּק הַהוּא שֶׁיּוֹרֵד מִן יּוֹ״ד, שֶׁהוּא פּוֹתֵחַ אֶת רַחְמָהּ, כדי לַעֲשׂוֹת פֵּרוֹת וְאִבִּים כיאות, וְהוּא קֹדֶשׁ עֶלְיוֹן.

הַפָּרָשָׁה הַשְּׁנִיָּה: "וְהָיָה כִּי־יְבִאֲךְ" (שמות יג ה) - זוֹ ה״א הַהֵיכָל, שֶׁנִּפְתָּח הרחם שלה מִתּוֹךְ יוֹ״ד בַּחֲמִשִּׁים פְּתָחִים, אַכְסַדְרָאוֹת וַחֲדָרִים סתומים שֶׁבּוֹ. שֶׁהַפֶּטֶר הַהוּא שֶׁעָשָׂה יוֹ״ד בַּהֵיכָל הַזֶּה, לִשְׁמֹעַ בָּהּ קוֹל אשר יּוֹצֵא מִתּוֹךְ הַשּׁוֹפָר הַזֶּה, מִשּׁוּם שֶׁהַשּׁוֹפָר הַזֶּה הוּא סתום בְּכָל הַצְּדָדִים, וּבָא יוֹ״ד וּפָתַח אוֹתוֹ כדי לְהוֹצִיא מִמֶּנּוּ קוֹל. וְכֵיוָן שֶׁפָּתַח אוֹתוֹ, תָּקַע לוֹ וְהוֹצִיא מִמֶּנּוּ קוֹל, כדי לְהוֹצִיא עֲבָדִים לַחֵרוּת.

וּבִתְקִיעַת הַשּׁוֹפָר הַזֶּה, יָצְאוּ יִשְׂרָאֵל מִמִּצְרַיִם, וְכָךְ עָתִיד בְּפַעַם אַחֶרֶת בְּסוֹף הַיָּמִים. וְכָל גְּאֻלָּה - מהַשּׁוֹפָר הַזֶּה בָּאָה. וּמִשּׁוּם כָּךְ, יֵשׁ בה יְצִיאַת מִצְרַיִם בַּפָּרָשָׁה הַזּוֹ, שֶׁהֲרֵי מֵהַשּׁוֹפָר הַזֶּה בָּא בַּכֹּחַ שֶׁל יוֹ״ד, שֶׁפּוֹתֵחַ את הרחם שלה, וּמוֹצִיא קוֹלוֹ לִגְאֻלַּת הָעֲבָדִים. וְזוֹ ה״א, הָאוֹת הַשְּׁנִיָּה שֶׁל הַשֵּׁם הַקָּדוֹשׁ.

הַפָּרָשָׁה הַשְׁלִישִׁית: סוֹד הַיִּחוּד שֶׁל שְׁמַע יִשְׂרָאֵל (דברים ו ד) - זוֹ וָא״ו שֶׁכּוֹלֶלֶת את הַכֹּל, וּבָהּ הַיִּחוּד שֶׁל הַכֹּל, וּבוֹ מִתְיַחֲדִים, וְהוּא נוֹטֵל את הַכֹּל.

הַפָּרָשָׁה הָרְבִיעִית: "וְהָיָה אִם־שָׁמוֹעַ" (דברים יא יג) - כְּלָלוּת שֶׁל שְׁנֵי צְדָדִים, {14a} שֶׁנֶּאֱחֶזֶת בָּהֶם כְּנֶסֶת יִשְׂרָאֵל, הַגְּבוּרָה שֶׁלְּמַטָּה. וְזוֹ ה״א הָאַחֲרוֹנָה, שֶׁנּוֹטֶלֶת אוֹתָם, וְנִכְלֶלֶת מֵהֶם.

וְהַתְּפִלִּין- אוֹתִיּוֹת הַשֵּׁם הַקָּדוֹשׁ הֵם מַמָּשׁ. וְעַל זה, "רֹאשֵׁךְ עָלַיִךְ כַּכַּרְמֶל" - אֵלּוּ תְּפִלִּין שֶׁל רֹאשׁ. "וְדַלַּת רֹאשֵׁךְ" (שיר השירים ז ו) - הַתְּפִלָּה שֶׁל יָד הַהִיא, שֶׁהִיא עֲנִיָּה כְּלַפֵּי מַעְלָה, אַף כָּךְ, שְׁלֵמוּת יֵשׁ לָהּ כְּמוֹ שֶׁלְּמַעְלָה.

"מֶלֶךָ אָסוּר בָּרְהָטִים" (שיר השירים ז ו) - הוּא קָשׁוּר וְאָחוּז בְּאוֹתָם הַבָּתִּים, לְהֵאָחֵז בַּשֵּׁם הַקָּדוֹשׁ הַהוּא כיאות. וְעַל זה, מִי שֶׁמִּתְתַּקֵּן בָּהֶם, הוּא הוה בְּצֶלֶם אֱלֹהִים. מָה בֵּאלֹהִים מִתְיַחֵד בּוֹ הַשֵּׁם הַקָּדוֹשׁ, אַף הוּא מִתְיַחֵד בּוֹ הַשֵּׁם הַקָּדוֹשׁ כיאות. "זָכָר וּנְקֵבָה בָּרָא אֹתָם". תְּפִלִּין שֶׁל רֹאשׁ וּתְפִלָּה שֶׁל יָד, וְהַכֹּל אֶחָד.
