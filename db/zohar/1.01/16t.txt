כִּי בְכָל־חַכְמֵי הַגּוֹיִם - אֵין כָּמֽוֹךָ

אָמַר לוֹ רַבִּי אֶלְעָזָר:
פָּסוּק זֶה שֶׁכָּתוּב: "מִי לֹא יִרָאֲךְ מֶלֶךְ הַגּוֹיִם כִּי לְךְ יָאָתָה" (ירמיה י ז) - אֵיזֶה שֶׁבַח הוּא?
אָמַר לוֹ:
אֶלְעָזָר בְּנִי, פָּסוּק זֶה נֶאֱמַר בְּכַמָּה מְקוֹמוֹת, אֲבָל וַדַּאי אין הוּא כך, שֶׁכָּתוּב "כִּי בְכָל־חַכְמֵי הַגּוֹיִם וּבְכָל־מַלְכוּתָם" (ירמיה י ז). שֶׁהֲרֵי בָּא לִפְתֹּחַ פֵּה הָרְשָׁעִים, שֶׁחוֹשְׁבִים, שֶׁהַקָּדוֹשׁ בָּרוּךְ־הוּא לֹא יוֹדֵעַ את ההִרְהוּרים וּמַחְשְׁבוֹתֵ שלהם?! ובגלל כך, יֵשׁ לְהוֹדִיעַ אֶת השְׁטוּת שלהם! 

שֶׁפַּעַם אַחַת, בָּא פִּילוֹסוֹף אֶחָד מֵאֻמּוֹת הָעוֹלָם אֵלַי, אָמַר לִי:
אַתֶּם אוֹמְרִים שֶׁאֱלֹהֵיכֶם שׁוֹלֵט בְּכָל רוּמֵי הַשָּׁמַיִם - כָּל הַצְּבָאוֹת וְהַמֶּרְכָּבוֹת, לֹא מַשִּׂיגִים, וְלֹא יוֹדְעִים אֶת מְקוֹמוֹ?! פָּסוּק זֶה לֹא מרבה אֶת כְּבוֹדוֹ כָּל כָּךְ, שֶׁכָּתוּב: "כִּי בְכָל־חַכְמֵי הַגּוֹיִם וּבְכָל־מַלְכוּתָם מֵאֵין כָּמוֹךְ". מַה שִּׁקּוּל זֶה לִבְנֵי־אָדָם שֶׁאֵין {1a} לָהֶם קִיּוּם?


וְעוֹד, שֶׁאַתֶּם אוֹמְרִים "וְלֹא־קָם נָבִיא עוֹד בְּיִשְׂרָאֵל כְּמֹשֶׁה" (דברים לד י) - בְּיִשְׂרָאֵל, לֹא קָם, אֲבָל בְּאֻמּוֹת הָעוֹלָם, כן קָם! גַּם כָּךְ, אֲנִי אוֹמֵר: בְּכָל חַכְמֵי הַגּוֹיִם אֵין כָּמוֹהוּ, אֲבָל בְּחַכְמֵי יִשְׂרָאֵל - יֵשׁ. אִם כָּךְ, אֱלוֹהַּ שֶׁיֵּשׁ בְּחַכְמֵי יִשְׂרָאֵל כמותו, אֵין הוא העליון הַשַּׁלִּיט. הסתכל  בַּפָּסוּק, וְתִמְצָא שֶׁדִּיַּקְתִּי כיאות!


אָמַרְתִּי לוֹ:
וַדַּאי יָפֶה אָמַרְתָּ!
מִי מְחַיֵּה מֵתִים? אֶלָּא הַקָּדוֹשׁ בָּרוּךְ־הוּא לְבַדּוֹ?! - בָּא אֵלִיָּהוּ וֶאֱלִישָׁע, וְהֶחֱיוּ מֵתִים.
מִי מוֹרִיד גְּשָׁמִים? אֶלָּא הַקָּדוֹשׁ בָּרוּךְ־הוּא לְבַדּוֹ?! - בָּא אֵלִיָּהוּ וּמָנַע אוֹתָם, וְהוֹרִידָם בִּתְפִלָּתוֹ.
מִי עָשָׂה שָׁמַיִם וָאָרֶץ? אֶלָּא הַקָּדוֹשׁ בָּרוּךְ־הוּא לְבַדּוֹ?! - בָּא אַבְרָהָם, וְהִתְקַיְּמוּ בְעָמְדָם בִּגְלָלוֹ.


מִי מַנְהִיג אֶת הַשֶּׁמֶשׁ? אֶלָּא הַקָּדוֹשׁ בָּרוּךְ־הוּא?! - בָּא יְהוֹשֻׁעַ וְהִשְׁתִּיקוֹ וְצִוָּהוּ שֶׁיַּעֲמֹד עַל עָמְדוֹ וְשָׁקַט, וְכָתוּב: "וַיִּדּם הַשֶּׁמֶשׁ וְיָרֵחַ עָמָד" (יהושע י יג).
הַקָּדוֹשׁ בָּרוּךְ־הוּא גּוֹזֵר גְּזַר דִּין - גַּם כָּךְ מֹשֶׁה גָּזַר גְּזַר דִּין, וְהִתְקַיֵּם. וְעוֹד, שֶׁהַקָּדוֹשׁ בָּרוּךְ־הוּא גּוֹזֵר גְּזֵרוֹת, וְצַדִּיקֵי יִשְׂרָאֵל מְבַטְּלִים אוֹתָם, שֶׁכָּתוּב: "צַדִּיק מוֹשֵׁל יִרְאַת אֱלֹהִים" (שמואל ב כג). וְעוֹד, שֶׁהוּא מְצַוֶּה אוֹתָם לָלֶכֶת בִּדְרָכָיו מַמָּשׁ, לְהִתְדַּמּוֹת לוֹ בַּכֹּל.
הָלַךְ אוֹתוֹ פִּילוֹסוֹף, וְהִתְגַּיֵּר בִּכְפַר שַׁחֲלַיִם, וְקָרְאוּ לוֹ יוֹסֵי קְטִינָאָה. וְלָמַד תּוֹרָה הַרְבֵּה, וְהוּא בֵּין חַכְמֵי וְצַדִּיקֵי אוֹתוֹ מָקוֹם.


כָּעֵת, יֵשׁ לְהִסְתַּכֵּל בַּפָּסוּק, וַהֲרֵי כָּתוּב: "כָּל־הַגּוֹיִם כְּאַיִן נֶגְדּוֹ" (ישעיה מ יז). מָה הָרִבּוּי כָּאן? אֶלָּא, "מִי לֹא יִרָאֲךְ מֶלֶךְ הַגּוֹיִם" - וְכִי "מֶלֶךְ הַגּוֹיִם" הוּא, וְלֹא "מֶלֶךְ יִשְׂרָאֵל". אֶלָּא בְּכָל מָקוֹם, הַקָּדוֹשׁ בָּרוּךְ־הוּא רָצָה לְהִשְׁתַּבֵּחַ בְּיִשְׂרָאֵל, וְלֹא נִקְרָא, אלא עַל יִשְׂרָאֵל לְבַדָּם, שֶׁכָּתוּב: "אֱלֹהֵי יִשְׂרָאֵל" (שמות ה א), "אֱלֹהֵי הָעִבְרִים" (שמות ה ג), וְכָתוּב: "כֹּה־אָמַר יְהֹוָה מֶלֶךְ־יִשְׂרָאֵל" (ישעיה מד ו). "מֶלֶךְ יִשְׂרָאֵל" בְּוַדַּאי.
אָמְרוּ אֻמּוֹת הָעוֹלָם:
פַּטְרוֹן אַחֵר לָנוּ בַּשָּׁמַיִם, שֶׁהֲרֵי מַלְכְּכֶם לֹא שׁוֹלֵט, אֶלָּא עֲלֵיכֶם בלְבַדְּכֶם, וְעָלֵינוּ לא שׁוֹלֵט!


בָּא הַפָּסוּק וְאָמַר: "מִי לֹא יִרָאֲךְ מֶלֶךְ הַגּוֹיִם" - המלך העליון, לרדות בהם, ולהלקות להם, ולעשות בהם רצונו. "כי לך יאתה" - לירא ממך למעלה ולמטה. "כי בכל־חכמי הגוים" - אֵלֶּה שַׁלִּיטִים גְּדוֹלִים הַמְמֻנִּים עֲלֵיהֶם. "וּבְכָל־מַלְכוּתָם" - בַּמַּלְכוּת הַהִיא שֶׁלְּמַעְלָה, שֶׁהֲרֵי אַרְבַּע מַלְכֻיּוֹת שׁוֹלְטוֹת יֵשׁ לְמַעְלָה, וְשׁוֹלְטוֹת בִּרְצוֹנוֹ עַל כָּל שְׁאָר הָעַמִּים. וְעִם כָּל זֶה, אֵין בָּהֶם מי שֶׁיַּעֲשֶׂה אֲפִלּוּ דָּבָר קָטָן, אֶלָּא כְּמוֹ שֶׁצִּוָּה אוֹתָם, כְּמוֹ שֶׁכָּתוּב: "~וְכִרְצוֹנוֹ~ עוֹשֶׂה בִּצְבָא הַשָּׁמַיִם וּבְדַיָּרֵי הָאָרֶץ" (דניאל ד לב). "חַכְמֵי הַגּוֹיִם" - אוֹתָם הַמְמֻנִּים וְהַגְּדוֹלִים שֶׁלְּמַעְלָה, שהחכמה שלהם מֵהֶם הִיא. "וּבְכָל־מַלְכוּתָם" - הַמַּלְכוּת הַשַּׁלֶּטֶת, כְּמוֹ שֶׁנֶּאֱמַר. וְזֶה הַמִּקְרָא כִּפְשׁוּטוֹ.


אֲבָל "בְּכָל־חַכְמֵי הַגּוֹיִם וּבְכָל־מַלְכוּתָם" - אֶת זֶה מָצָאנוּ בְּסִפְרֵי הָרִאשׁוֹנִים, שֶׁאוֹתָן מֶרְכָּבוֹת וּצְבָאוֹת, אַף עַל פי שֶׁנִּתְמַנּוּ עַל דִּבְרֵי הָעוֹלָם, וּמִנָּה לְכָל אֶחָד לַעֲשׂוֹת פְּעֻלָּתוֹ. מִיהוּ, נסתר קדוש ולא כרצונו, שֶׁיַּעֲשֶׂה שׁוּם אֶחָד מֵהֶם כָּמוֹךְ? בִּגְלַל שֶׁאַתָּה רָשׁוּם בְּעִלּוּי, וְאַתָּה רָשׁוּם בְּמַעַשְׂךְ מִכֻּלָּם. וְזֶהוּ "מֵאֵין כָּמוֹךְ יְהֹוָה". מִיהוּ נִסְתָּר קָדוֹשׁ, שֶׁיַּעֲשֶׂה וְיִהְיֶה כָמוֹךְ לְמַעְלָה וּלְמַטָּה, וְיִהְיֶה דוֹמֶה לְךְ בַּכֹּל? מַעֲשֵׂה הַמֶּלֶךְ הַקָּדוֹשׁ, שָׁמַיִם וָאָרֶץ. אֲבָל הֵם "תֹּהוּ וַחֲמוּדֵיהֶם בַּל־יוֹעִילוּ" (ישעיה מד ט). בַּקָּדוֹשׁ בָּרוּךְ־הוּא כָּתוּב "בְּרֵאשִׁית בָּרָא אֱלֹהִים …", בְּמַלְכוּתָם כָּתוּב "וְהָאָרֶץ הָיְתָה תֹהוּ וָבֹהוּ".


אָמַר רַבִּי שִׁמְעוֹן לַחֲבֵרִים:
בְּנֵי הִלּוּלָא זוֹ! כָּל אֶחָד מִכֶּם יְקַשֵּׁט קִשּׁוּט אֶחָד לַכַּלָּה.
אָמַר לְרַבִּי אֶלְעָזָר בְּנוֹ:
אֶלְעָזָר! תֵּן מַתָּנָה אַחַת לַכַּלָּה, שֶׁהֲרֵי מָחָר יִסְתַּכֵּל הקדוש ברוך הוא בה, כְּשֶׁיִּכָּנֵס לַחֻפָּה בְּאוֹתָם שִׁירִים וּשְׁבָחִים, שנתנו לָהּ בְּנֵי הַהֵיכָל לַעֲמֹד לְפָנָיו.
