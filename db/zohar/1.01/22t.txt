מִצְוָה שְׁנִיָּה

מִצְוָה שְׁנִיָּה: זוֹהִי מִצְוָה, שֶׁמִּצְוָה שֶׁל 'יִרְאָה' נֶאֶחְזָה בָּהּ, וְלֹא יוֹצֵאת מִמֶּנָּה לְעוֹלָמִים, וְהִיא 'אַהֲבָה', שֶׁאָדָם יאהַב אֶת רִבּוֹנוֹ אַהֲבָה שְׁלֵמָה. וּמַה הִיא אַהֲבָה שְׁלֵמָה? זוֹ 'אַהֲבָה רַבָּה', שֶׁכָּתוּב: "הִתְהַלֵּךְ לְפָנַי וֶהְיֵה תָמִים", שָׁלֵם בְּאַהֲבָה. וְזֶהוּ שֶׁכָּתוּב: "וַיֹּאמֶר אֱלֹהִים יְהִי אוֹר" - זוֹ אַהֲבָה שְׁלֵמָה שֶׁנִּקְרֵאת 'אַהֲבָה רַבָּה', וְכָאן הִיא הַמִּצְוָה, שֶׁאָדָם יֹאהַב אֶת רִבּוֹנוֹ כָּרָאוּי.

אָמַר רַבִּי אֶלְעָזָר:
אַבָּא, אַהֲבָה בִּשְׁלֵמוּת, אֲנִי שָׁמַעְתִּי בָּזֶה.
אָמַר לוֹ:
אֱמֹר, בְּנִי, לִפְנֵי רַבִּי פִּינְחָס! שֶׁהֲרֵי הוּא עוֹמֵד בַּדַּרְגָּה הַזּוֹ.
אָמַר רַבִּי אֶלְעָזָר:
'אַהֲבָה רַבָּה', הַיְנוּ 'אַהֲבָה' שְׁלֵמָה, בִּשְׁלֵמוּת שֶׁל שְׁנֵי צְדָדִים, וְאִם אֵין נִכְלֶלֶת בִּשְׁנֵי צְדָדִים - אֵינָהּ 'אַהֲבָה' כָּרָאוּי בִּשְׁלֵמוּת.


וְעַל זה שָׁנִינוּ, לִשְׁנֵי צְדָדִים נִפְרֶדֶת הָ'אַהֲבָה', אַהֲבַת הַקָּדוֹשׁ בָּרוּךְ הוּא.
יֵשׁ מִי שֶׁאוֹהֵב אוֹתוֹ מִתּוֹךְ שֶׁיֵּשׁ לוֹ עֹשֶׁר, אֹרֶךְ יָמִים, בָּנִים סְבִיבוֹ, שׁוֹלֵט עַל שׂוֹנְאָיו, דְּרָכָיו מְתֻקָּנוֹת לוֹ, וּמִתּוֹךְ כָּךְ אוֹהֵב אוֹתוֹ.
וְאִם לָזֶה יִתְהַפֵּךְ, וְיַחֲזִיר עָלָיו הַקָּדוֹשׁ בָּרוּךְ הוּא גִּלְגּוּל שֶׁל דִּין קָשֶׁה - יִשְׂנָא אוֹתוֹ, וְלֹא יֹאהַב אוֹתוֹ כְּלָל, וּבִגְלַל כָּךְ, אַהֲבָה זוֹ אֵינָהּ 'אַהֲבָה' שֶׁיֵּשׁ לָהּ עִקָּר.

אַהֲבָה שֶׁנִּקְרֵאת שְׁלֵמָה, הַהוּא שהיה בִּשְׁנֵי הַצְּדָדִים - בֵּין בְּדִין, בֵּין בְּחֶסֶד וְתִקּוּן דַּרְכּוֹ, שֶׁאוֹהֵב אֶת רִבּוֹנוֹ - כְּמוֹ שֶׁשָּׁנִינוּ - אֲפִילּוּ שֶׁנּוֹטֵל אֶת נִשְׁמָתְךְ מִמְּךְ - זוֹ הִיא 'אַהֲבָה' שְׁלֵמָה בִּשְׁנֵי צְדָדִים.
וְעַל כֵּן, הָאוֹר שֶׁל מַעֲשֵׂה בְרֵאשִׁית יָצָא וְאַחַר כָּךְ נִגְנַז. כְּשֶׁנִּגְנַז, יָצָא דִין קָשֶׁה, וְנִכְלְלוּ שְׁנֵי הַצְּדָדִים יַחַד, שֶׁתִּהְיֶה שְׁלֵמוּת. זֹאת אַהֲבָה כָּרָאוּי.

נְטָלוֹ רַבִּי שִׁמְעוֹן וּנְשָׁקוֹ. בָּא רַבִּי פִּינְחָס וּנְשָׁקוֹ וּבֵרְכוֹ, וְאָמַר:
בְּוַדַּאי הַקָּדוֹשׁ בָּרוּךְ הוּא שְׁלָחַנִי לְכָאן. זֶה הוּא אוֹר קָטָן, שֶׁאָמְרוּ לִי שֶׁנִּכְלָל בְּבֵיתִי, וְאַחַר כָּךְ מֵאִיר לְכָל הָעוֹלָם.
אָמַר רַבִּי אֶלְעָזָר:
בְּוַדַּאי לֹא צָרִיךְ לִשְׁכֹּחַ 'יִרְאָה' בְּכָל הַמִּצְווֹת, כָּל שֶׁכֵּן בַּמִּצְוָה הַזּוֹ צָרִיךְ 'יִרְאָה' לְהִתְדַּבֵּק בָּזֶה. אֵיךְ מִתְדַּבֵּק? הָ'אַהֲבָה' הִיא בְּצַד אֶחָד טוֹב - כְּמוֹ שֶׁנִּתְבָּאֵר - שֶׁנּוֹתֵן עֹשֶׁר וְטוֹב, אֹרֶךְ שֶׁל חַיִּים, בָּנִים וּמְזוֹנוֹת - אָז צָרִיךְ לְעוֹרֵר 'יִרְאָה', וְלִפְחֹד שֶׁלֹּא יִגְרֹם הַחֵטְא. וְעַל זֶה כָּתוּב: "אַשְׁרֵי אָדָם מְפַחֵד תָּמִיד" (משלי כח יָד), מִשּׁוּם שֶׁזֶּה כּוֹלֵל 'יִרְאָה' בְּ'אַהֲבָה'.


וְכָךָ צָרִיךְ בַּצַּד הָאַחֵר שֶׁל הַדִּין הַקָּשֶׁה, לְעוֹרֵר בּוֹ 'יִרְאָה'. כְּשֶׁרוֹאֶה שֶׁהַדִּין הַקָּשֶׁה שׁוֹרֶה עָלָיו, אָז יְעוֹרֵר 'יִרְאָה', וְיִפְחַד מֵרִבּוֹנוֹ כָּרָאוּי, וְלֹא יַקְשֶׁה לִבּוֹ. וְעַל זֶה כָּתוּב: "וּמַקְשֶׁה לִבּוֹ יִפּוֹל בְּרָעָה" (משלי כח יָד), בְּאוֹתוֹ הַצַּד הָאַחֵר שֶׁנִּקְרָא 'רָעָה'. נִמְצֵאת 'יִּרְאָה' שֶׁנֶּאֶחְזָה בִּשְׁנֵי צְדָדִים וְנִכְלֶלֶת מֵהֶם, וְזוֹהִי 'אַהֲבָה' שְׁלֵמָה כָּרָאוּי.
