=שמיים וארקא
פָּתַח רַבִּי שִׁמְעוֹן וְאָמַר:
- "בְּרֵאשִׁית בָּרָא אֱלֹהִים"[… את השמים והארץ] (בראשית א א)
פָסוּק זֶה - יֵשׁ לְהִסְתַּכֵּל בו, שֶׁכָּל מִי שֶׁאוֹמֵר 'יֵשׁ אֱלוֹהַּ אַחֵר' (~כי כתוב 'אלהים' בלשון רבים) - מושְׁמָד מֵהָעוֹלָמוֹת, כְּמוֹ שֶׁנֶּאֱמַר: (ירמיה י יא) "כִּדְנָה תֵמְרוּן לְהוֹם:"(=כָּךְ תֹּאמְרוּ לָהֶם) "אֱלָהַיָּא דִי־שְׁמַיָּא וְאַרְקָא לָא עֲבַדוּ"(=אֱלֹהֵי הַשָּׁמַיִם וְה'אַרְקָא' לֹא עָשׂוּ) - [לכן] "יֵאבַדוּ מֵאַרְעָא וּמִן־תְּחוֹת שְׁמַיָּא אֵֽלֶּה!" (=יאבְדוּ מֵהָאָרֶץ וּמִתַּחַת שַׁמַיִם אֵלֶּה!) - בִּגְלַל שֶׁאֵין אֱלוֹהַּ אַחֵר, חוּץ מִהַקָּדוֹשׁ בָּרוּךְ־הוּא בִּלְבַדּוֹ!

וּפָסוּק זֶה הוּא [כולו] תַרְגּוּם, חוּץ מִהמילה של סוֹף הַפָּסוּק (~המילה 'אלא', היחידה שלא כתובה בארמית). אִם תֹּאמַר, [ש]בִּגְלַל שֶׁמַּלְאָכִים קְדוֹשִׁים לֹא נִזְקָקִים לְתַרְגּוּם וְלֹא מודעים אליו (=לא מכירים אותו) - מִלָּה זוֹ ראויה הִיא לְהֵאָמֵר בִּלְשׁוֹן הַקֹּודֶשׁ, כְּדֵי שֶׁיִּשְׁמְעוּ מַלְאָכִים קְדוֹשִׁים, וְיִהְיוּ זְקוּקִים לְהוֹדוֹת עַל זֶה.

אֶלָּא וַדַּאי בגלל כָּךְ כָּתוּב תַּרְגּוּם, כי לֹּא נִזְקָקִים לוֹ הַמַּלְאָכִים הַקְּדוֹשִׁים, וְלֹא יְקַנְּאוּ בְּבֶן־אָדָם (~שאומר את הפסוק שמקלל אותם), [ויבואו] לְהַרע לוֹ, בִּגְלַל שֶׁבְּפָסוּק זֶה, בתוך הכלל [נכללים] אותם המַלְאָכִים הקְדוֹשִׁים, כי [גם] הֵם 'אֱלֹהִים' נִקְרָאִים, בִּכְלָל שֶׁל 'אֱלֹהִים' היו - וְהֵם לֹא עָשׂוּ שָׁמַיִם וָאָרֶץ.

[המילה] "וְאַרְקָא" - 'וְאַרְעָא' מתבקש לוֹ [שיהיה כתוב]! אֶלָּא בִּגְלַל שֶׁ"אַרְקָא" הִיא אַחַת מֵאוֹתָן שֶׁבַע אֲרָצוֹת שֶׁלְּמַטָּה, וּבְאוֹתוֹ מָקוֹם יֶשְׁנָם בְּנֵי־בָנָיו שֶׁל קַיִן. לְאַחַר שֶׁגּורַשׁ מֵעַל פְּנֵי הָאָרֶץ, יָרַד לשָׁם וְעָשָׂה תוֹלָדוֹת(=צאצאים), וְנשְׁתַּבֵּשׁ שָׁם (~בדעתו) [עד] שֶׁלֹּא יָדַע כְּלוּם.

וְהִיא אֶרֶץ כְּפוּלָה, שֶׁנִּכְפְּלת מֵחֹושֶׁךְ וְאוֹר, וְיש שָׁם שְׁנֵי מְמונִּים שַׁלִּיטִים אשר שּׁוֹלְטִים, זֶה בַּחֹשֶׁךְ וְזֶה בָּאוֹר, וְשָׁם [יש] קִטְרוּג (~של הממונים) זֶה עם זֶה.

וּבְשָׁעָה שֶׁיָּרַד לְשָׁם קַיִן, נשְׁתַּתְּפוּ (~שני הממונים) זֶה עִם זֶה, ונשתלמו כְּאֶחָד, וְכולם רוֹאִים שֶׁהֵם תּוֹלְדוֹת (=צאצאים) של קַיִן. וְעַל כן, הֵם בִּשְׁנֵי רָאשִׁים, כְּשְׁני נחשים, רק כְּאשר אותו הַאוֹר שׁוֹלֵט הנצחון שֶׁלּוֹ וּמְנַצֵּחַ את הָאַחֵר, וְעַל זֶה, נִכְלְלוּ [אֵלּה] אשר בַּחֹשֶׁךְ [עם אלה] בָּאוֹר, וְהָיוּ אֶחָד.

אותם שְׁנֵי מְמונִּים [נקראים] 'עַפְרִירָא' (~מצד החושך, כפי שרומז שמו) וְ'קַסְטִימוֹן' (~מצד האור) - והדמות שֶׁלָּהֶם [היא] כִּדְמוּת מַלְאָכִים קְדוֹשִׁים בְּשֵׁשׁ כְּנָפַיִם. אֶחָד - הדְּמוּת [היא] כְּשׁוֹר, וְאֶחָד - הדְּמוּת [היא] כְּנֶשֶׁר, וּכְאשר מִּתְחַבְּרִים - נַעֲשִׂים דְּמוּת אָדָם.

כאשר הֵם בַּחֹשֶׁךְ, מִתְהַפְּכִים לִדְמוּת שֶׁל נָחָשׁ בִּשְׁנֵי רָאשִׁים, וְהוֹלְכִים כְּנָחָשׁ, וְעפים לְתוֹךְ הַתְּהוֹם(=הפתח של הצד האחר), ונרחצים בַּיָּם הַגָּדוֹל. כאשר מַּגִּיעִים לְשַׁלְשֶׁלֶת של 'עֲזָא' וַ'עֲזָאֵל' (~מלאכים שחטאו, וקשורים בשלשלאות) - מַרְגִּיזִים אותם, וּמְעוֹרְרִים אוֹתָם, וְהֵם מְדַלְּגִים לְתוֹךְ ההרים החשוכים (~שלרגלם היו קשורים), וְחוֹשְׁבִים שֶׁהַקָּדוֹשׁ בָּרוּךְ־הוּא רוֹצֶה לִתְבֹּועַ מהם [את] הדִּין.

וּשְׁנֵי מְמונִּים אֵלּוּ שָׁטִים בַּיָּם הַגָּדוֹל, וּפוֹרְחִים מִשָּׁם, וְהוֹלְכִים בַּלַּיְלָה אל נַעֲמָה - אימם של השֵּׁדִים - שֶׁטָּעוּ אַחֲרֶיהָ יְרֵאִים (~יראי שמיים) רִאשׁוֹנִים, וְחָשְׁבוּ לְהִתְקָרֵב אֵלֶיהָ. וְהִיא מְדַלֶּגֶת שִׁשִּׁים־אֶלֶף פַּרְסָאוֹת, וְנַעֲשֵׂית(=נראית) בְּכַמָּה צִיּוּרִים(=דמויות) אל בְּנֵי־אָדָם, בשביל שֶׁיִּטְעוּ בְּנֵי־אָדָם אַחֲרֶיהָ.

וּשְׁנֵי מְמונִּים הַאלה פּוֹרְחִים וּמְשׁוֹטְטִים בְּכָל הָעוֹלָם, וְחוֹזְרִים לִמְקוֹמָם, וְהֵם מְעוֹרְרִים לְאוֹתָם בְּנֵי־בָנָיו שֶׁל קַיִן את רוּחַ היְצָרִים הרָעִים, [כדי] לַעֲשׂוֹת תּוֹלָדוֹת(=צאצאים).

הַשָּׁמַיִם שֶׁשׁוֹלְטִים שָּׁם (~בארץ 'ארקא') - לא כמו אֵלּה [שלנו], וְלֹא מוֹלִידָה הָאָרֶץ [שם] בְּכֹח שֶׁלָּהֶם (~של השמיים שלהם) זריעה וְקָצִיר כְּמוֹ אֵלֶּה [שלנו], וְלֹא מוחזרים (~מחזורי השמיים והעונות שמביאות גשם), אֶלָּא בְּכַמָּה שָׁנִים וּזְמַנִּים.

וְהֵם (~שני הממונים) - [הם] 'אֱלֹהֵי הַשָּׁמַיִם וְהאַרְקָא לֹא עָשׂוּ'. יֹאבְדוּ מֵהאֶרֶץ העֶלְיוֹנָה [הנקראת] 'תֵּבֵל', שֶׁלֹּא יִשְׁלְטוּ בָהּ וְלֹא יְשׁוֹטְטוּ בָהּ, וְלֹא יהיה גוֹרְמִים לִבְנֵי אָדָם לְהִטָּמֵא מִ'מִּקְרֵה לַיְלָה'. וְעַל כן, 'יֹאבְדוּ מֵהָאָרֶץ וּמִתַּחַת הַשָּׁמַיִם', שֶׁנַּעֲשׂוּ בַשֵּׁם 'אֵלֶּה', כְּמוֹ שנאמר.

ועל כן, הפָּסוּק הזֶה [הוּא] תַרְגּוּם, שֶׁלֹּא יַחְשְׁבוּ מַלְאָכִים עֶלְיוֹנִים שֶׁעֲלֵיהֶם אוֹמְרִים [אותו], וְלֹא יְקַטְרְגוּ לָנוּ. וְעַל כן, הַסּוֹד שֶׁל [המילה] "אֵלֶּה", כְּמוֹ שנאמר, [הוא זה:] היא (~המילה 'אלא') מילה קדושׁה שֶׁלֹּא מִתְחַלֵּפת בַּתַּרְגּוּם.
