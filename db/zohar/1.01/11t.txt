בְּרֵאשִׁית, רַבִּי שִׁמְעוֹן פָּתַח:

- "וָאָשִׂם דְּבָרַי בְּפִיךְ" (ישעיה נא טז). כַּמָּה יֵשׁ לבן־אָדָם לְהִשְׁתַּדֵּל בַּתּוֹרָה? יוֹמָם וָלַיְלָה! מִשּׁוּם שֶׁהַקָּדוֹשׁ בָּרוּךְ־הוּא מַקְשִׁיב לְקוֹלוֹתיהם של אוֹתָם שמשתדלים בַּתּוֹרָה, וּבְכָל דָּבָר שֶׁמִּתְחַדֵּשׁ בַּתּוֹרָה עַל יְדֵי אוֹתוֹ שֶׁמִּשְׁתַּדֵּל בַּתּוֹרָה, עוֹשֶׂה רָקִיעַ אֶחָד.

- שָׁנִינוּ, שבַּאותה השָּׁעָה שֶׁדְּבַר תּוֹרָה מִתְחַדֵּשׁ מִפִּיו של בן־אָדָם, אותו הַדָּבָר עוֹלֶה, ~וּמִזְדַּמֵּן~ לפניו של הַקָּדוֹשׁ בָּרוּךְ־הוּא. וְהַקָּדוֹשׁ בָּרוּךְ־הוּא נוֹטֵל אֶת אוֹתוֹ הַדָּבָר, וּמְנַשֵּׁק אוֹתוֹ, וּמְעַטֵּר אוֹתוֹ בְּשִׁבְעִים עֲטָרוֹת גְּלוּפוֹת וּמְחֻקָּקוֹת. וּדְבַר חָכְמָה שֶׁהִתְחַדֵּשׁ, עוֹלֶה וְיוֹשֵׁב עַל רֹאשׁ הַצַּדִּיק חַי הָעוֹלָמִים .וְטָס מִשָּׁם, וּמְשׁוֹטֵט בְּשִׁבְעִים אֶלֶף עוֹלָמוֹת, וְעוֹלֶה אֶל עַתִּיק־יָּמִים. וְכָל הַדְּבָרִים שֶׁל עַתִּיק־יָּמִים - דִּבְרֵי חָכְמָה הֵם, בְּסוֹדוֹת סתומים עֶלְיוֹנִים.

- וְאוֹתוֹ דָּבָר סתום שֶׁל חָכְמָה שֶׁהִתְחַדֵּשׁ כָּאן, כאשר הוּא עוֹלֶה, הוא מִתְחַבֵּר עִם אוֹתָם הַדְּבָרִים שֶׁל עַתִּיק־יָּמִים, וְעוֹלֶה וְיוֹרֵד עִמָּהֶם, וְנִכְנָס בִּשְׁמוֹנָה־עָשָׂר עוֹלָמוֹת גְּנוּזִים שֶׁ"עַיִן לֹא רָאָתָה אֱלֹהִים זוּלָתְךְ" (ישעיה סד ג). יוֹצְאִים מִשָּׁם, וּמְשׁוֹטְטִים וְהוֹלְכִיםף מְלֵאִים וּשְׁלֵמִים, ~וּמִזְדַּמְּנִים~ לִפְנֵי עַתִּיק־יָּמִים.

- בְּאוֹתָהּ הַשָּׁעָה הריח עַתִּיק־הַיָּמִים אֶת הַדָּבָר הַזֶּה, וְזֶה נוֹחַ לְפָנָיו מִן הַכֹּל. נוֹטֵל אֶת אוֹתוֹ הַדָּבָר, וּמְעַטֵּר אוֹתוֹ בִּשְׁלֹשׁ־מֵאוֹת וְשִׁבְעִים־אֶלֶף עֲטָרוֹת. אוֹתוֹ הַדָּבָר טָס, וְעוֹלֶה וְיוֹרֵד, וְנַעֲשֶׂה רָקִיעַ אֶחָד. וְכֵן, כָּל דָּבָר וְדָבָר שֶׁל חָכְמָה - נעשים רקיעים, עוֹמְדִים בְּקִיּוּם שָׁלֵם לִפְנֵי עַתִּיק־יָּמִים. וְהוּא קוֹרֵא לָהֶם 'שָׁמַיִם חֲדָשִׁים', 'שָׁמַיִם מְחֻדָּשִׁים', סתומים שֶׁל סוֹדוֹת שֶׁל חָכְמָה עֶלְיוֹנָה. וְכָל אוֹתָם שְׁאָר דִּבְרֵי הַתּוֹרָה שֶׁמִּתְחַדְּשִׁים, עוֹמְדִים לִפְנֵי הַקָּדוֹשׁ בָּרוּךְ־הוּא, וְעוֹלִים וְנַעֲשִׂים 'אַרְצוֹת הַחַיִּים', וְיוֹרְדִים וּמִתְעַטְּרִים לְ'אֶרֶץ' אַחַת, וּמִתְחַדֵּשׁ וְנַעֲשֶׂה הַכֹּל 'אֶרֶץ חֲדָשָׁה' מֵאוֹתוֹ הַדָּבָר שֶׁהִתְחַדֵּשׁ בַּתּוֹרָה.


- וְעַל זֶה כָּתוּב: "כִּי כַאֲשֶׁר הַשָּׁמַיִם הַחֲדָשִׁים וְהָאָרֶץ הַחֲדָשָׁה אֲשֶׁר אֲנִי עֹשֶׂה עֹמְדִים לְפָנַי" (ישעיה סו כב). 'עָשִׂיתִי' לֹא כָתוּב, אֶלָּא "עוֹשֶׂה"! משום שֶׁהוא עוֹשֶׂה תָמִיד מֵאוֹתָם הַחִדּוּשִׁים וְהַסּוֹדוֹת שֶׁל הַתּוֹרָה. וְעַל זֶה כָּתוּב: "וָאָשִׂם דְּבָרַי בְּפִיךְ וּבְצֵל יָדִי כִּסִּיתִיךְ לִנְטֹעַ שָׁמַיִם וְלִיסֹד אָרֶץ" (ישעיה נא טז). 'הַשָּׁמַיִם' לֹא כָתוּב, אֶלָּא "שָׁמַיִם"!


אָמַר רַבִּי אֶלְעָזָר:

- מַה הוא פירוש "וּבְצֵל יָדִי כִּסִּיתִיךְ"?

אָמַר לוֹ:

- בְּשָׁעָה שֶׁנִּמְסְרָה הַתּוֹרָה לְמֹשֶׁה, בָּאוּ כַּמָּה רבבות מַלְאָכִים עֶלְיוֹנִים כדי לִשְׂרֹף אוֹתוֹ בְּשַׁלְהֶבֶת פִּיהֶם, עַד שֶׁכִּסָּה עָלָיו הַקָּדוֹשׁ בָּרוּךְ־הוּא. וְכָעֵת שֶׁהַדָּבָר הַזֶּה עוֹלֶה וּמִתְעַטֵּר וְעוֹמֵד לִפְנֵי הַקָּדוֹשׁ בָּרוּךְ־הוּא, הוּא מחפה עַל אוֹתוֹ הַדָּבָר, וּמְכַסֶּה עַל אוֹתוֹ בן־אָדָם, כדי שֶׁלֹּא יוודע אֲלֵיהֶם אֶלָּא הַקָּדוֹשׁ בָּרוּךְ־הוּא, וְלֹא יְקַנְּאוּ לוֹ, עַד שֶׁנַּעֲשֶׂה מֵהַדָּבָר הַהוּא 'שָׁמַיִם חֲדָשִׁים' וְ'אֶרֶץ חֲדָשָׁה', זֶהוּ שֶׁכָּתוּב: "וּבְצֵל יָדִי כִּסִּיתִיךְ לִנְטֹעַ שָׁמַיִם וְלִיסֹד אָרֶץ".
מִכָּאן, שֶׁכָּל דָּבָר שֶׁסתום מִן הָעַיִן, עוֹלֶה לְתוֹעֶלֶת עֶלְיוֹנָה, זֶהוּ שֶׁכָּתוּב: "וּבְצֵל יָדִי כִּסִּיתִיךְ". ומדוע מתחפה וּמִתְכַּסֶּה מִן הָעַיִן? בִּשְׁבִיל תּוֹעֶלֶת עֶלְיוֹנָה, זֶהוּ שֶׁכָּתוּב: "לִנְטֹעַ שָׁמַיִם וְלִיסֹד אָרֶץ", כְּמוֹ שנאמר.

- וְלֵאמֹר לְצִיּוֹן: 'עַמִּי אָתָּה'. וְלֵאמֹר לְאוֹתָם הַשְּׁעָרִים וְהַדְּבָרִים הַמְצֻיָּנִים אֵלּוּ עַל אֵלּוּ: 'עַמִּי אָתָּה'. אַל תִּקְרא 'עַמִּי אָתָּה' אֶלָּא 'עִמִּי אָתָּה', לִהְיוֹת שֻׁתָּף עִמִּי. מָה אֲנִי בַּדִּבּוּר שֶׁלִּי עָשִׂיתִי 'שָׁמַיִם וָאָרֶץ', כְּמוֹ שאתה אומר: "בִּדְבַר יְהֹוָה שָׁמַיִם נַעֲשׂוּ" (תהלים לג ו), אַף כָּךְ אַתָּה. זכאית אוֹתָם שֶׁמִּשְׁתַּדְּלִים בַּתּוֹרָה!

- וְאִם תֹּאמַר שֶׁדִּבּוּר שֶׁל כָּל אָדָם שֶׁלֹּא יוֹדֵעַ גם עוֹשֶׂה זאת, בֹּא רְאֵה! אוֹתוֹ אחד שֶׁאֵין דַּרְכּוֹ בְּסוֹדוֹת הַתּוֹרָה, וּמְחַדֵּשׁ דְּבָרִים שֶׁאֵינוֹ יוֹדֵעַ עַל בֻּרְיָם כָּנאות - אוֹתוֹ הַדָּבָר עוֹלֶה, וְיוֹצֵא לְאוֹתוֹ הַדָּבָר "אִישׁ תַּהְפּוּכוֹת לְשׁוֹן שָׁקֶר" (משלי טז כח), מִתּוֹךְ נֶקֶב הַתְּהוֹם הַגָּדוֹל, וּמְדַלֵּג חֲמֵשׁ־מֵאוֹת פַּרְסָאוֹת לְקַבֵּל אֶת אוֹתוֹ הַדָּבָר,. וְנוֹטֵל אוֹתוֹ, וְהוֹלֵךְ עִם אוֹתוֹ הַדָּבָר לְתוֹךְ הַנֶּקֶב שֶׁלּוֹ, וְעוֹשֶׂה בּוֹ רָקִיעַ שֶׁל שָׁוְא שֶׁנִּקְרָא 'תֹהוּ'. וְטָס בְּאוֹתוֹ הָרָקִיעַ אוֹתוֹ "אִישׁ תַּהְפּוּכוֹת" שֵׁשֶׁת־אֲלָפִים פַּרְסָאוֹת בְּפַעַם אַחַת. וְכֵיוָן שֶׁהָרָקִיעַ הַזֶּה שֶׁל שָׁוְא עוֹמֵד, מִיָּד יוֹצֵאת "אֵשֶׁת זְנוּנִים" וּמִתְחַזֶּקֶת בְּאוֹתוֹ הָרָקִיעַ שֶׁל שָׁוְא, היא ו"איש תהפוכות", וּמִשְׁתַּתֶּפֶת עִמּוֹ, וּמִשָּׁם יוֹצֵאת וְהוֹרֶגֶת כַּמָּה אֲלָפִים וּרְבָבוֹת. מִשּׁוּם שֶׁכַּאֲשֶׁר הִיא עוֹמֶדֶת בָּרָקִיעַ הַהוּא, יֵשׁ לָהּ רְשׁוּת וִיכֹלֶת לִהְיוֹת טָסָה אֶת כָּל הָעוֹלָם בְּרֶגַע אֶחָד.

- וְעַל זֶה כָּתוּב: "הוֹי מֹשְׁכֵי הֶעָון בְּחַבְלֵי הַשָּׁוְא" (ישעיה ה יח). 'הֶעָון' - זֶה הַזָּכָר. "וְכַעֲבוֹת הָעֲגָלָה חַטָּאָה", מִי 'הַחַטָּאָה'? זוֹ הַנְּקֵבָה שֶׁנִּקְרֵאת 'חַטָּאָה'. הוּא מוֹשֵׁךְ אֶת אוֹתוֹ שֶׁנִּקְרָא עָון בְּאוֹתָם 'חַבְלֵי הַשָּׁוְא'. וְאַחַר כָּךְ 'כַּעֲבוֹת הָעֲגָלָה חַטָּאָה', אֶת אוֹתָהּ הַנְּקֵבָה שֶׁנִּקְרֵאת 'חַטָּאָה', שֶׁשָּׁם הִתְגַּבְּרָה לִהְיוֹת טָסָה לַהֲרֹג בְּנֵי־אָדָם. וְעַל זֶה, "כִּי־רַבִּים חֲלָלִים הִפִּילָה" (משלי ז כו). מִי הִפִּילָה? זוֹ אוֹתָהּ הַחַטָּאָה שֶׁהוֹרֶגֶת בְּנֵי־אָדָם. מִי גָרַם אֶת זֶה? תַּלְמִיד הֶחָכָם שֶׁלֹּא הִגִּיעַ לְהוֹרָאָה וּמוֹרֶה, הָרַחֲמָן יַצִּילֵנוּ.


אָמַר רַבִּי שִׁמְעוֹן לַחֲבֵרִים:

- בְּבַקָּשָׁה מִכֶּם, שֶׁלֹּא תוֹצִיאוּ מִפִּיכֶם דְּבַר תּוֹרָה שֶׁלֹּא יְדַעְתֶּם וְלֹא שְׁמַעְתֶּם מֵהעֵץ הגָּדוֹל כמו שנאות, כְּדֵי שֶׁלֹּא תִהְיוּ גוֹרְמִים לְאוֹתָהּ 'חַטָּאָה' לַהֲרֹג הֲמוֹנֵי בְּנֵי־אָדָם לְחִנָּם.

פָּתְחוּ כֻלָּם וְאָמְרוּ:

- הָרַחֲמָן יַצִּילֵנוּ, הָרַחֲמָן יַצִּילֵנוּ!


- בֹּא רְאֵה! בַּתּוֹרָה בָּרָא הַקָּדוֹשׁ בָּרוּךְ־הוּא אֶת הָעוֹלָם, והרי העמידוה, שֶׁכָּתוּב: "וָאֶהְיֶה אֶצְלוֹ אָמוֹן וָאֶהְיֶה שַׁעֲשׁוּעִים יוֹם יוֹם" (משלי ח ל). וְהוּא הִסְתַּכֵּל בָּהּ פַּעַם, וּשְׁתַּיִם, וְשָׁלֹשׁ, וְאַרְבַּע פְּעָמִים, וְאַחַר כָּךְ אָמַר להם, וְאַחַר־כָּךְ עָשָׂה בָּהּ מַעֲשֶׂה, כדי לְלַמֵּד אֶת בְּנֵי־האָדָם שֶׁלֹּא יָבאוּ לִטְעוֹת בָּהּ, כְּמוֹ שאתה אומר: "אָז רָאָהּ וַיְסַפְּרָהּ הֱכִינָהּ וְגַם חֲקָרָהּ וַיֹּאמֶר לָאָדָם" (איוב כח כז).
וּכְנֶגֶד אַרְבַּע פְּעָמִים האלו שֶׁכָּתוּב: "אָז רָאָהּ, וַיְסַפְּרָהּ, הֱכִינָהּ, וְגַם חֲקָרָהּ" - בָּרָא הַקָּדוֹשׁ בָּרוּךְ הוּא אֶת מַה שֶּׁבָּרָא. וְטֶרֶם שֶׁהוֹצִיא אֶת הַמַּעֲשֶׂה שֶׁלּוֹ, הִכְנִיס בָּרִאשׁוֹנָה אַרְבַּע תֵּבוֹת, שֶׁכָּתוּב: "בְּרֵאשִׁית בָּרָא אֱלֹהִים אֵת", הִנֵּה אַרְבַּע, וְאַחַר כָּךְ "הַשָּׁמַיִם" - הֵם כְּנֶגֶד אַרְבַּע הַפְּעָמִים שֶׁהִסְתַּכֵּל הַקָּדוֹשׁ בָּרוּךְ־הוּא בַּתּוֹרָה טֶרֶם שיוֹצִיא אֶת מַעֲשֵׂהוּ ~לִמְלַאכְתּוֹ~.
