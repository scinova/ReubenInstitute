בַּלַּיְלָה שֶׁל הַכַּלָּה

רַבִּי שִׁמְעוֹן הָיָה יוֹשֵׁב וְעוֹסֵק בַּתּוֹרָה, בַּלַּיְלָה שֶׁהַכַּלָּה מִתְחַבֶּרֶת בְּבַעְלָהּ. שֶׁשָּׁנִינוּ, שכָּל אוֹתָם הַחֲבֵרִים בְּנֵי הֵיכַל הַכַּלָּה, הִצְטָרְכוּ בּלַיְלָה הַהוּא שֶׁהַכַּלָּה ~מתעתדת~ לִהְיוֹת לַמָּחֳרָת בְּתוֹךְ הַחֻפָּה עִם בַּעְלָהּ, לִהְיוֹת עִמָּהּ כָּל הַלַּיְלָה הַהוּא, וְלִשְׂמֹחַ עִמָּהּ בְּתִקּוּנֶיהָ שֶׁהִיא מִתְתַּקֶּנֶת, ~לַעֲסֹק~ בַּתּוֹרָה, מִתּוֹרָה לִנְבִיאִים, וּמִנְּבִיאִים לִכְתוּבִים, וּבִדְרָשׁוֹת הַפְּסוּקִים, וּבְסוֹדוֹת הַחָכְמָה, בִּגְלַל שֶׁאֵלּוּ הֵם התיקונים שלה וְתַכְשִׁיטֶיהָ. וְהִיא וְעַלְמוֹתֶיהָ נִכְנֶסֶת וְעוֹמֶדֶת עַל רָאשֵׁיהֶם, וּמִתְתַּקֶּנֶת בָּהֶם, וּשְׂמֵחָה בָהֶם, כָּל הַלַּיְלָה הַהוּא. וְלַמָּחֳרָת לֹא נִכְנֶסֶת לַחֻפָּה אֶלָּא יַחַד אִתָּם, וְאֵלֶּה נִקְרָאִים 'בְּנֵי הַחֻפָּה'. וְכֵיוָן שֶׁנִּכְנֶסֶת לַחֻפָּה, הַקָּדוֹשׁ בָּרוּךְ־הוּא שׁוֹאֵל עֲלֵיהֶם, וּמְבָרֵךְ אוֹתָם, וּמְעַטֵּר אוֹתָם בַּעֲטֶרֶתה של הַכַּלָּה, זכאית חֶלְקָם!

וְהָיָה רַבִּי שִׁמְעוֹן וְכָל הַחֲבֵרִים מְרַנְּנִים בְּרִנַּת הַתּוֹרָה, וּמְחַדְּשִׁים דִּבְרֵי תוֹרָה כָּל אֶחָד וְאֶחָד מֵהֶם, וְהָיָה שָׂמֵחַ רַבִּי שִׁמְעוֹן וְכָל שְׁאָר הַחֲבֵרִים.
אָמַר לָהֶם רַבִּי שִׁמְעוֹן:
בָּנַי, זכאית חֶלְקְכֶם! בִּגְלַל שֶׁמָּחָר לֹא תִכָּנֵס הַכַּלָּה לַחֻפָּה, אֶלָּא אִתְּכֶם! בִּגְלַל שֶׁכֻּלָּם שֶׁמְּתַקְנִים את תִּקּוּנֶיהָ בַּלַּיְלָה הַזֶּה וּשְׂמֵחִים בָּהּ, כֻּלָּם יִהְיוּ רְשׁוּמִים וּכְתוּבִים בְּסֵפֶר הַזִּכְרוֹנוֹת, וְהַקָּדוֹשׁ בָּרוּךְ־הוּא מברך אותם בְּשִׁבְעִים בְּרָכוֹת וַעֲטָרוֹת שֶׁל הָעוֹלָם הָעֶלְיוֹן.


פָּתַח רַבִּי שִׁמְעוֹן וְאָמַר:
"הַשָּׁמַיִם מְסַפְּרִים כְּבוֹד־אֵל …" (תהלים יט ב). את הפָּסוּק זֶה הֲרֵי הֶעֱמַדְנוּהוּ. אֲבָל בִּזְמַן זֶה, שֶׁהַכַּלָּה מִתְעוֹרֶרֶת לְהִכָּנֵס לַחֻפָּה לַמָּחֳרָת, מִתְתַּקֶּנֶת וּמְאִירָה בְּקִשּׁוּטֶיהָ יַחַד עִם הַחֲבֵרִים שֶׁשָּׂמְחוּ עִמָּהּ כָּל אוֹתוֹ הַלַּיְלָה, וְהִיא שְׂמֵחָה אִתָּם. וְלַמָּחֳרָת כַּמָּה המונים, חֲיָלוֹת ~וּמַחֲנוֹת~ מתאספים אִתָּהּ! וְהִיא וְכֻלָּם מְחַכִּים לְכָל אֶחָד וְאֶחָד שֶׁתִּקְּנוּ אוֹתָהּ בַּלַּיְלָה הַזֶּה, כֵּיוָן שֶׁמִּתְחַבְּרִים כְּאֶחָד וְהִיא רוֹאָה אֶת בַּעְלָהּ, מַה כָּתוּב? "הַשָּׁמַיִם מְסַפְּרִים כְּבוֹד־אֵל".
"הַשָּׁמַיִם" - זֶה הֶחָתָן שֶׁנִּכְנָס לַחֻפָּה.
"מְסַפְּרִים" - מְאִירִים כְּזֹהַר הַסַּפִּיר, שֶׁמֵּאִיר וְזוֹהֵר מִסּוֹפי הָעוֹלָם וְעַד סוֹפי הָעוֹלָם.
"כְּבוֹד־אֵל" - זֶה כְּבוֹד כַּלָּה שֶׁנִּקְרָא "אֵל", שֶׁכָּתוּב "וְאֵל זוֹעֵם בְּכָל יוֹם" (תהלים ז יב). בְּכָל יְמֵי הַשָּׁנָה נִקְרָא "אֵל", וְכָעֵת, כְּשֶׁהֲרֵי נִכְנֶסֶת לַחֻפָּה, נִקְרָא "כָּבוֹד", וְנִקְרָא "אֵל", כָּבוֹד עַל כָּבוֹד, אוֹר עַל אוֹר, שִׁלְטוֹן עַל שִׁלְטוֹן.

וְאָז, בּשָׁעָה הַהִיא שֶׁהַשָּׁמַיִם נִכְנָס לַחֻפָּה וּבָא וּמֵאִיר לָהּ, כָּל אוֹתָם הַחֲבֵרִים שֶׁהִתְקִינוּ אוֹתָהּ, כֻּלָּם ~מִתְפָּרְשִׁים~ בִּשְׁמוֹתָם שָׁם, וְזֶה הוא שֶׁכָּתוּב: "וּמַעֲשֵׂה יָדָיו מַגִּיד הָרָקִיעַ".
"מַעֲשֵׂה יָדָיו" - אֵלֶּה הם בַּעֲלֵי קִיּוּם הַבְּרִית עִם הַכַּלָּה. וְאוֹתָם בַּעֲלֵי קִיּוּם הַבְּרִית נִקְרָאִים "מַעֲשֵׂה יָדָיו", כְּמוֹ שֶׁאתה אומר: "וּמַעֲשֵׂה יָדֵינוּ כּוֹנְנֵהוּ" (תהלים צ יז), זוֹ בְּרִית הַקַּיֶּמֶת שֶׁחֲתוּמָה בִּבְשַׂר בֶּן־הָאָדָם.

רַב הַמְנוּנָא הַזָּקֵן אָמַר כָּךְ:
"אַל־תִּתֵּן אֶת־פִּיךְ לַחֲטִיא אֶת־בְּשָׂרֶךְ" (קהלת ה ה) - כדי שֶׁלֹּא יִתֵּן בֶּן־אָדָם אֶת פִּיו להגיע לְהִרְהוּר רַע, ויהיה גורם לְהַחֲטִיא ל'בְּשַׂר קֹדֶשׁ' הַהוּא שֶׁחתוּמה בּוֹ בְּרִית הַקֹּדֶשׁ, שֶׁאִֽילּוּ עוֹשֶׂה כֵּן - מוֹשְׁכִים אוֹתוֹ לַגֵּיהִנֹּם.
וְהַמְמֻנֶּה הַהוּא עַל הַגֵּיהִנֹּם, 'דּוּמָה' שְׁמוֹ, וְכַמָּה רִבּוֹאוֹת מַלְאֲכֵי חַבָּלָה אִתּוֹ, וְעוֹמֵד עַל פֶּתַח הַגֵּיהִנֹּם. וְכָל אוֹתָם שֶׁשָּׁמְרוּ בְּרִית הַקֹּדֶשׁ בָּעוֹלָם הַזֶּה - אֵין לוֹ רְשׁוּת לִקְרַב אֲלֵיהֶם!

דָּוִד הַמֶּלֶךְ - בְּשָׁעָה שֶׁאֵרֵעַ לוֹ הַמַעֲשֶׂה הַהוּא - פָּחַד. בּשָׁעָה הַהִיא עָלָה דוּמָה לִפְנֵי הַקָּדוֹשׁ בָּרוּךְ־הוּא, וְאָמַר לוֹ:
רִבּוֹן הָעוֹלָם! {8b} כָּתוּב בַּתּוֹרָה: "וְאִישׁ אֲשֶׁר יִנְאַף אֶת־אֵשֶׁת אִישׁ" (ויקרא כ י), וְכָתוּב: "וְאֶל־אֵשֶׁת עֲמִיתְךְ …" (ויקרא יח כ). דָּוִד, שֶׁקִּלְקֵל בְּרִית בְּעֶרְוָה, מַהוּ דינו?
אָמַר לוֹ הַקָּדוֹשׁ בָּרוּךְ־הוּא:
דָּוִד זכאי הוּא, וּבְרִית הַקֹּדֶשׁ עַל תִּקּוּנוֹ עוֹמֵד, שֶׁהֲרֵי גָּלוּי לְפָנַי ~שֶׁמְּזֻמֶּנֶת~ לוֹ בַּת־שֶׁבַע מִיּוֹם שֶׁנִּבְרָא הָעוֹלָם.

אָמַר לוֹ:
אִם לְפָנֶיךְ גָּלוּי - לְפָנָיו לֹא גָלוּי!
אָמַר לוֹ:	;
וְעוֹד, בְּהֶתֵּר הָיָה מַה שֶּׁהָיָה, שֶׁהֲרֵי כָּל אֵלּוּ שעלו לַקְּרָב, לֹא עלה אף אֶחָד מֵהֶם, עַד ~שֶׁפתר~ את אִשְׁתּוֹ בְּגֵט.
אָמַר לוֹ:
אִם כָּךְ, הָיָה לוֹ ~לְחַכּוֹת~ שְׁלֹשָׁה חֳדָשִׁים, וְלֹא ~חִכָּה~.
אָמַר לוֹ:
בַּמָּה ~עוֹמֵד~ הַדָּבָר? בְּמָקוֹם שֶׁחוֹשְׁשִׁים שֶׁהִיא מְעֻבֶּרֶת, וְגָלוּי לְפָנַי, שֶׁאוּרִיָּה לֹא קָרַב אֵלֶיהָ מֵעוֹלָם, שֶׁהֲרֵי שְׁמִי חָתוּם בְּתוֹכוֹ לְעֵדוּת:
כָּתוּב "אוּרִיָּה" (שמואל ב יא ח),
וְכָתוּב "אוּרִיָּהוּ" (ירמיה כו כ).
שְׁמִי חָתוּם עִמּוֹ, שֶׁלֹּא שִׁמֵּשׁ בָּהּ לְעוֹלָמִים.
אָמַר לוֹ:
רִבּוֹן הָעוֹלָם! זֶה מַה שֶּׁאָמַרְתָּ - אִם לְפָנֶיךְ גָּלוּי, שֶׁלֹא שָׁכַב עִמָּהּ אוּרִיָּה, לפני מי גָּלוּי? הָיָה לוֹ לְחַכּוֹת לָהּ שְׁלֹשָׁה חֳדָשִׁים!
וְעוֹד, אִם יָדַע שֶׁמֵּעוֹלָם לֹא שָׁכַב עִמָּהּ - מדוע שָׁלַח לוֹ דָּוִד וְצִוָּה עָלָיו לְשַׁמֵּשׁ ~עִם~ אִשְׁתּוֹ? שֶׁכָּתוּב: "רֵד לְבֵיתְךְ וּרְחַץ רַגְלֶיךְ" (שמואל ב יא ח).

אָמַר לוֹ:
וַדַּאי לֹא יָדַע, אֲבָל יוֹתֵר מִשְּׁלֹשָׁה חֳדָשִׁים חִכָּה, שֶׁהֲרֵי אַרְבָּעָה חֳדָשִׁים הָיוּ. שֶׁכָּךְ לָמַדְנוּ: בְּעֶשְׂרִים וַחֲמִשָּׁה בְּנִיסָן הֶעֱבִיר דָּוִד כְּרוּז בְּכָל יִשְׂרָאֵל, וְהָיוּ עִם יוֹאָב בְּשִׁבְעָה יָמִים שֶׁבְּסִיוָן, וְהָלְכוּ וְהִשְׁחִיתוּ אֶרֶץ בְּנֵי עַמּוֹן. סִיוָן, תַּמּוּז, אָב וֶאֱלוּל שָׁהוּ שָׁם, וּבְעֶשְׂרִים וְאַרְבָּעָה בֶּאֱלוּל הָיָה מַה שֶּׁהָיָה עִם בַּת־שֶׁבַע, וּבְיוֹם הַכִּפּוּרִים מָחַל לוֹ הַקָּדוֹשׁ בָּרוּךְ־הוּא את הַחֵטְא הַהוּא.
וְיֵשׁ שֶּׁאוֹמְרִים: בְּשִׁבְעָה בַּאֲדָר הָעֳבַר כְּרוּז, וְהִתְכַּנְּסוּ בַּחֲמִשָּׁה־עָשָׂר בְּאִיָּר, וּבַחֲמִשָּׁה־עָשָׂר בֶּאֱלוּל הָיָה מַה שֶּׁהָיָה עִם בַּת־שֶׁבַע, וּבְיוֹם הַכִּפּוּרִים הִתְבַּשֵּׂר: "גַּם־יְהֹוָה הֶעֱבִיר חַטָּאתְךְ לֹא תָמוּת" (שמואל ב יב יג). מַה זֶּה "לֹא תָמוּת"? לֹא תָמוּת בְּיַד דּוּמָה.

אָמַר דּוּמָה:
רִבּוֹן הָעוֹלָם! הנה דָּבָר אֶחָד יֵשׁ לִי אֵלָיו, שֶׁהוּא פָּתַח פִּיו, וְאָמַר: "חַי־יְהֹוָה כִּי בֶן־מָוֶת הָאִישׁ הָעֹשֶׂה זֹאת" (שמואל ב יב ה), וְהוּא דָן אֶת נַפְשׁוֹ. ~טְרוּנְיָא~ יֵשׁ לִי עָלָיו.
אָמַר לוֹ:
אֵין לְךְ רְשׁוּת! שֶׁהֲרֵי הוא הוֹדָה לְפָנַי, וְאָמַר: "חָטָאתִי לַיְהֹוָה" (שמואל-ב יב יג), אַף עַל פִּי שֶׁלֹּא חָטָא. אֲבָל בְּמַה שֶּׁחָטָא בְּאוּרִיָּה - הַעֹנֶשׁ כָּתַבְתִּי עָלָיו, וְהוא קִבֵּל אותו. 
מִיָּד חָזַר דּוּמָה לִמְקוֹמוֹ בְּפַחֵי נֶפֶשׁ.
ועַל זֶה אָמַר דָּוִד:
"לוּלֵי יְהֹוָה עֶזְרָתָה לִּי כִּמְעַט שָׁכְנָה דוּמָה נַפְשִׁי" (תהלים צד יז). "לוּלֵי יְהֹוָה עֶזְרָתָה לִּי" - שֶׁהָיָה אֶפּוֹטְרוֹפּוֹס שֶׁלִּי, "כִּמְעַט שָׁכְנָה …" - מַהו "כִּמְעַט"? כְּחוּט דַּק כּשִּׁעוּר שֶׁיֵּשׁ בֵּינִי וּבֵין הַצַּד הָאַחֵר, כּשִׁעוּר הַהוּא הָיָה, שֶׁלֹּא "שָׁכְנָה דוּמָה נַפְשִׁי".

וְלָכֵן רצוי לְהִשָּׁמֵר בֶּן־אָדָם כדי שֶׁלֹּא יֹאמַר דָּבָר כדָוִד, בִּגְלַל שֶׁלֹּא יוּכַל לוֹמַר לְדוּמָה "כִּי שְׁגָגָה הִיא" (קהלת ה ה), כְּמוֹ שֶׁהָיָה לְדָוִד, וְנִצַּח אוֹתוֹ הַקָּדוֹשׁ בָּרוּךְ־הוּא בַּדִּין.
"לָמָּה יִקְצֹף הָאֱלֹהִים עַל־קוֹלֶךְ" - עַל קוֹל ההוא שֶׁהוּא אָמַר.
"וְחִבֵּל אֶת־מַעֲשֵׂה יָדֶיךְ" - זֶה בְּשַׂר קֹדֶשׁ בְּרִית הַקָּדוֹשׁ שֶׁפָּגַם וְנִמְשָׁךְ בַּגֵּיהִנֹּם עַל יְדֵי דוּמָה, וּבְגִין כָּךְ: "וּמַעֲשֵׂה יָדָיו מַגִּיד הָרָקִיעַ" (תהלים יט ב) - אֵלּוּ הם הַחֲבֵרִים שֶׁהִתְחַבְּרוּ עִם הַכַּלָּה הַזּוֹ, וּבַעֲלֵי בְרִיתָ שלה, מַגִּיד וְרוֹשֵׁם כָּל אֶחָד וְאֶחָד. מִי "הָרָקִיעַ"? זֶהוּ הָרָקִיעַ שֶׁבּוֹ חַמָּה וּלְבָנָה, וְכוֹכָבִים וּמַזָּלוֹת, וְזֶהוּ 'סֵפֶר זִכָּרוֹן'. הוּא מַגִּיד וְרוֹשֵׁם להם, וְכוֹתֵב להם לִהְיוֹת בְּנֵי הַהֵיכָל, וְלַעֲשׂוֹת רְצוֹנָם תָּמִיד.
"יוֹם לְיוֹם יַבִּיעַ אֹמֶר" - יוֹם קָדוֹשׁ מֵאוֹתָם יָמִים עֶלְיוֹנִים שֶׁל הַמֶּלֶךְ, מְשַׁבְּחִים להם לחֲבֵרִים, וְאוֹמְרִים את הדָבָר ההוא שֶׁאָמַר כָּל אֶחָד לַחֲבֵרוֹ. יוֹם לְיוֹם "יַבִּיעַ" הַ"אֹמֶר" הַהוּא וּמְשַׁבְּח לו,
"וְלַיְלָה לְּלַיְלָה" - כָּל דַּרְגָּה שֶׁנשְׁלֶמֶת בַּלַּיְלָה, מְשַׁבֵּחַ זֶה לָזֶה הַ'דַעַת' הַהוּא שֶׁל כָּל אֶחָד מֵחֲבֵרוֹ, ובאהבה רַבָּה נַעֲשִׂים לָהֶם חֲבֵרִים וְאוֹהֲבִים.

"אֵין־אֹמֶר וְאֵין דְּבָרִים" (תהלים יט ד) - בִּשְׁאָר דִּבְרֵי הָעוֹלָם, שֶׁלֹּא נִשְׁמָעִים לִפְנֵי הַמֶּלֶךְ הַקָּדוֹשׁ וְלֹא רצוי לשמוע להם. אֲבָל הַדְּבָרִים הַלָּלוּ - "בְּכָל־הָאָרֶץ יָצָא קַוָּם" (תהלים יט ה). ~עוֹשִׂים {9a} מְדִידָה~ אוֹתָם דְּבָרִים, מָדוֹרִים עֶלְיוֹנִים וּמָדוֹרִים תַּחְתּוֹנִים, מֵאֵלֶּה נַעֲשִׂים רְקִיעִים, וּמֵאֵלֶּה הָאָרֶץ, מֵהַשֶּׁבַח הַהוּא.
וְאִם תֹּאמַר שֶׁאוֹתָם דְּבָרִים בְּמָקוֹם אֶחָד מְשׁוֹטְטִים בָּעוֹלָם - "בִּקְצֵה תֵבֵל מִלֵּיהֶם" (תהלים יט ה). וְכֵיוָן שֶׁנַּעֲשִׂים רְקִיעִים מֵהֶם, מִי שׁוֹרֶה בָהֶם? חָזַר וְאָמַר: "לַשֶּׁמֶשׁ שָׂם־אֹהֶל בָּהֶם" (תהלים יט ה). השמש הקדושה ההיא, שָׂם מְדוֹרוֹ וּמִשְׁכָּנוֹ בָהֶם, וּמִתְעַטֵּר בָּהֶם.

כֵּיוָן שֶׁשּׁוֹרֶה בְּאוֹתָם רְקִיעִים וּמִתְעַטֵּר בָּהֶם, אֲזַי - "וְהוּא כְּחָתָן יֹצֵא מֵחֻפָּתוֹ" (תהלים יט ו), שָׂמֵחַ וְרָץ בְּאוֹתָם רְקִיעִים, יוֹצֵא מֵהֶם, וְנכנס ורָץ לְתוֹךְ מִגְדָּל אֶחָד אַחֵר, בְּמָקוֹם אַחֵר. "מִקְצֵה הַשָּׁמַיִם מוֹצָאוֹ" (תהלים יט ז) - וַדַּאי מֵהעוֹלָם העֶלְיוֹן יוֹצֵא וּבָא, שֶׁהוּא "קְצֵה הַשָּׁמַיִם" לְמַעְלָה. "וּתְקוּפָתוֹ" (תהלים יט ז) - מִי זֶה "תְּקוּפָתוֹ"? זֶה "קְצֵה הַשָּׁמַיִם" לְמַטָּה, שֶׁהִיא 'תְקוּפַת הַשָּׁנָה' שֶׁסּוֹבֶבֶת אֶת כָּל ~הָאֵיבָרִים~, וּמִתְקַשֶּׁרֶת מִן הַשָּׁמַיִם עַד רָקִיעַ זֶה.

"וְאֵין נִסְתָּר מֵחַמָּתוֹ" (תהלים יט ז) - שֶׁהתְּקוּפָה הַהִיא זוֹ, וּתְקוּפַת הַשֶּׁמֶשׁ שֶׁמְּסוֹבֵב בְּכָל צַד. "וְאֵין נִסְתָּר" - אֵין מִי שֶׁמִּתְכַּסֶּה מִמֶּנּוּ מִכָּל הַדְּרָגוֹת הָעֶלְיוֹנוֹת, שֶׁהָיוּ כֻלָּם מַקִּיפוֹת וּבָאוֹת אֵלָיו, וְכָל אֶחָד וְאֶחָד - אֵין מִי שֶׁמִּתְכַּסֶּה מִמֶּנּוּ. "מֵחַמָּתוֹ" - בְּשָׁעָה שֶׁמִּתְחַמֵּם, וְמשתוקק ~לְגַבֵּיהֶם~ בתשוקה שְׁלֵמָה. כָּל הַשֶּׁבַח הַזֶּה, וְכָל הָעִלּוּי הַזֶּה - ~בִּגְלַל~ הַתּוֹרָה הוּא, שֶׁכָּתוּב: "תּוֹרַת יְהֹוָה תְּמִימָה" (תהלים יט ח).

שֵׁשׁ פְּעָמִים כָּתוּב כָּאן "יְהֹוָה", וְיש שִׁשָּׁה פְסוּקִים מִן "הַשָּׁמַיִם מְסַפְּרִים" עַד "תּוֹרַת  יְהֹוָה תְּמִימָה". וְעַל סוֹד זֶה כָּתוּב "בְּרֵאשִׁית" - הנה שֵׁשׁ אוֹתִיּוֹת. "בָּרָא אֱלֹהִים אֵת הַשָּׁמַיִם וְאֵת הָאָרֶץ" - הנה שֵׁשׁ תֵּבוֹת. פְּסוּקִים אֲחֵרִים כְּנֶגֶד שֵׁשׁ פְּעָמִים "יְהֹוָה", שִׁשָּׁה פְסוּקִים בִּגְלַל שֵׁשׁ אוֹתִיּוֹת שֶׁכָּאן, שִׁשָּׁה שֵׁמוֹת בִּגְלַל שֵׁשׁ תֵּבוֹת שֶׁכָּאן.

בְּעוֹדָם יוֹשְׁבִים, נִכְנְסוּ רַבִּי אֶלְעָזָר בְּנוֹ וְרַבִּי אַבָּא. אָמַר לָהֶם:
וַדַּאי פְּנֵי הַשְּׁכִינָה בָּאוּ, וְלָכֵן 'פְּנִיאֵל' קָרָאנוּ לָכֶם, שֶׁהֲרֵי רְאִיתֶם אֶת פני הַשְּׁכִינָה פָּנִים בְּפָנִים. וְעַכְשָׁו, שֶׁיְּדַעְתֶּם, וְגִלָּה לָכֶם ~הַכָּתוּב~ שֶׁל "וּבְנָיָהוּ בֶן־יְהוֹיָדָע" - וַדַּאי שֶׁדָּבָר של ~עַתִּיק קָדוֹשׁ~ הוּא, וְהַפָּסוּק שֶׁלְּאַחֲרָיו, וְאוֹתוֹ שסָּתוּם מֵהַכֹּל אֲמָרו.

וּפָסוּק זֶה הוּא בְּמָקוֹם אַחֵר, כְּגוֹן זֶה. פָּתַח וְאָמַר:
"וְהוּא הִכָּה אֶת־הָאִישׁ הַמִּצְרִי אִישׁ מִדָּה חָמֵשׁ בָּאַמָּה" (דברי הימים א יא כג), וְהַכֹּל סוֹד אֶחָד הוּא. מִי הַמִּצְרִי? הַהוּא שנודע, "גָּדוֹל מְאֹד בְּאֶרֶץ מִצְרַיִם בְּעֵינֵי עַבְדֵי …" (שמות יא ג), גָּדוֹל וְנִכְבָּד, כְּמוֹ שֶׁגִּלָּה הַזָקֵן הַהוּא.

וּפָסוּק זֶה - בַּיְשִׁיבָה הָעֶלְיוֹנָה נאמר. "אִישׁ מִדָּה" - הַכֹּל אֶחָד. "אִישׁ מַרְאֶה" (שמואל ב כג) וְ"אִישׁ מִדָּה" - הַכֹּל אֶחָד. בִּגְלַל שֶׁהוּא שַׁבָּת וּתְחוּם, שֶׁכָּתוּב: "וּמַדּתֶם מִחוּץ לָעִיר" (במדבר לה ה), וְכָתוּב: "לֹא־תַעֲשׂוּ עָוֶל בַּמִּשְׁפָּט בַּמִּדָּה" (ויקרא יט לה). וְעַל זֶה, אִישׁ מִדָּה הוּא, וְהוּא מַמָּשׁ אִישׁ מִדָּה הוּא, אָרְכּוֹ מִסּוֹפי הָעוֹלָם וְעַד סוֹפי הָעוֹלָם. אָדָם הָרִאשׁוֹן כָּךְ הָיָה. וְאִם תֹּאמַר: הֲרֵי כָּתוּב: "חָמֵשׁ בָּאַמָּה", אוֹתָם חָמֵשׁ בָּאַמָּה - מִסּוֹפי הָעוֹלָם וְעַד סוֹפי הָעוֹלָם הָיוּ.

"וּבְיַד הַמִּצְרִי חֲנִית" (דברי הימים א יא) - כְּמוֹ שאתה אומר: "כִּמְנוֹר אֹרְגִים" - זֶה "מַטֵּה הָאֱלֹהִים" שֶׁהָיָה בְּיָדוֹ, חָקוּק בְּשֵׁם גָּלוּף מְפֹרָשׁ, ~בְּאוֹר~ שֶׁל צֵרוּפֵי אוֹתִיּוֹת, שֶׁהָיָה ~גולף~ בְּצַלְאֵל וִישִׁיבָתוֹ, שֶׁנִּקְרָא 'אוֹרֵג', שֶׁכָּתוּב: "מִלֵּא אֹתָם ... חָרָשׁ וְחֹשֵׁב וְרֹקֵם …" (שמות לה לה). וְאוֹתוֹ מַטֶּה הָיָה מֵאִיר שֵׁם גלוף בְּכָל הַצְּדָדִים בָּאוֹר שֶׁהַחֲכָמִים שֶׁהָיוּ מגלים שֵׁם הַמְפֹרָשׁ בְּאַרְבָּעִים וּשְׁנַיִם אֳפָנִים. וְהַפָּסוּק מִכָּאן וָהָלְאָה, כְּמוֹ שֶׁאָמַר, זכאית חֶלְקוֹ.

שְׁבוּ, נִכְבָּדִים! שְׁבוּ, וּנְחַדֵּשׁ את תִּקּוּן הַכַּלָּה בַּלַּיְלָה הַזֶּה. משום שֶׁכָּל מִי שֶׁמִּשְׁתַּתֵּף אִתָּהּ בַּלַּיְלָה הַזֶּה, יִהְיֶה שָׁמוּר לְמַעְלָה וּלְמַטָּה כָּל הַשָּׁנָה הַהִיא, וְיוֹצִיא שנה בְשָׁלוֹם. עֲלֵיהֶם כָּתוּב: "חוֹנֶה מַלְאַךְ־יְהֹוָה סָבִיב לִירֵאָיו וַיְחַלְּצֵם" (תהלים לד ח). טַעֲמוּ, וּרְאוּ כִּי טוֹב יְהֹוָה.
