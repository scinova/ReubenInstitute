התיקון השני של הזקן


הַתִּיקּוּן השני: מִתַּתְקָן הַשֵּׂיעָר, מתחילת הַפֶּה, עַד תְּחִילָּה אַחֶרֶת של הַפֶּה, בְּתִיקּוּן שקול.


קוּם רבי חִזְקִיָּה, ועמוד בקיומך, והוקר יָקָר תִּיקּוּן זה של הזקן הקדוש! קם רבי חִזְקִיָּה, הִתְחִיל וְאָמַר: "אֲנִי לְדוֹדִי וְעָלַי תְּשׁוּקָתוֹ" (שִׁיר השירים ז יא). מי גרם שֶׁ"אֲנִי לְדוֹדִי"? מִשּׁוּם שֶׁ"עָלַי תְּשׁוּקָתוֹ".


מִסְתַּכֵּל הָיִיתִי, וַהנה רָאִיתִי אוֹר יָקָר של נֵר עֶלְיוֹן, מֵאִיר, וְעוֹלֶה לשלוש מאות ועשרים וחמישה עיברים. וְאֶחָד חשוך הָיָה מתרחץ בְּאוֹתוֹ אוֹר, כְּמִי שמתרחץ בְּאוֹתוֹ נָהָר עמוק, שמימיו מתחלקים, וּמְאִירִים, וזורמים לכל עיבר, מִמַּה שעליו. וְעוֹלֶה אוֹתוֹ האור, בשפת הים העמוק העליון, שכל פְּתָחִים טובים ויקרים, בְּאוֹתוֹ הַפֶּתַח נפתחים.


אֲנִי שאלתי מֵהֶם, פשר שֶׁרָאִיתִי. פָּתְחוּ וְאָמְרוּ: "נושא עון" ראית. אמר: זה הוּא הַתִּיקּוּן השני. יָשַׁב. אמר רבי שִׁמְעוֹן: עכשיו נתקשר העולם. בָּרוּךְ אתה רבי חִזְקִיָּה, לְעַתִּיק העתיקים.


אמר רבי שִׁמְעוֹן: כל הַנֵּרוֹת - חֲבֵרִים שֶׁבָּאִים בזאת הטבעת הַקְּדוֹשָׁה! אֲנִי מעיד עַל הַשָּׁמַיִים, הָעֶלְיוֹנִים מֵעֶלְיוֹנִים, וְהָאָרֶץ הַקְּדוֹשָׁה, הָעֶלְיוֹנָה מעליונה, דאני רוֹאָה עכשיו, מה שלא ראה בן אדם, מֵהַיּוֹם שעלה מֹשֶׁה בְּפַעַם השנייה לְהַר סיני, שאני רָאִיתִי פְּנֵי מְאִירִים, כְּאוֹר שִׁמֵּשׁ חֲזָקָה, שעתיד לצאת ברפואה לעולם, שכתוב, "וְזָרְחָה לָכֶם יִרְאֵי שְׁמִי, שֶׁמֶשׁ צְדָקָה וּמַרְפֵּא בִּכְנָפֶיהָ" (מלאכי ג כ). וְעוֹד שאני יָדַעְתִּי שפני מְאִירִים, ומשה לא ידע, וְלֹא הִסְתַּכֵּל, זֶהוּ שכתוב, "וּמֹשֶׁה לָא־יָדַע כִּי קָרַן עוֹר פָּנָיו" (שמות לד כט).


וְעוֹד, שאני רוֹאָה בעיני, שְׁלוֹשׁ עֶשְׂרֵה מידות מחוקקות לפני, וּמְאִירִים כנרות. וכשמתפרש כל אֶחָד ממנן מִפִּיכֶם, עולה ומתתקן, ומתעטר ומתסתתר בנסתרות תיקוני הזקן, וְכָל הָאֲחֵרִים נִשְׁאָרִים. וּבְעוֹד שכל אֶחָד מתפרש מִפִּיכֶם, מֵאִיר ומתעטר וְיוֹשֵׁב כמלך בתוך בצבאותיו. וכשמסתיים להתפרש, עולה ומתעטר בעטרת קְדוֹשָׁה, ומתתקן מסתתר, וְיוֹשֵׁב בְּתִיקּוּנָיו של הזקן הקדוש, וְכֵן לכל אֶחָד וְאֶחָד. הִזְדָּרְזוּ חֲבֵרִים קְדוֹשִׁים, שֶׁהֲרֵי בְּקִיּוּם זה לא יהיה, עַד שיבוא הַמֶּלֶךְ הַמָּשִׁיחַ.


קוּם, רבי חִזְקִיָּה, בשנית! והוקר הַתִּיקּוּן הַשְּׁלִישִׁי של הזקן הקדוש.  לָמַדְנוּ, שֶׁעַד שלא קם רבי חִזְקִיָּה, קוֹל יָצָא וְאָמַר: 'אֵין מַלְאָךְ אֶחָד עוֹשֶׂה שְׁתֵּי שְׁלִיחוּיוֹת'. התרגש רבי שִׁמְעוֹן וְאָמַר: וַדַּאי כל אֶחָד וְאֶחָד במקומו. ואני, וְרַבִּי אֶלְעָזָר בני, וְרַבִּי אבא, נשלמת השלמות העליונה
