תִּקּוּנֵי דִּזְעֵיר אַפִּין


עכשיו הִתְתַּקְּנוּ, וכוונו את הדעת, לְתַקֵּן את תִּיקּוּנָיו של זְעֵיר אַנְפִּין, אֵיךְ יתתקן, ואיך יתלבש בְּתִיקּוּנָיו מתיקוני עַתִּיק יָמִים, קדוש הַקְּדוֹשִׁים, נִסְתָּר הַנִּסְתָּרִים, נִסְתָּר מֵהַכֹּל. שעכשיו החובה עֲלֵיכֶם, לגזור דין אֱמֶת נאה וְיָפֶה, ולתקן את כל התיקונים עַל בוריים.


תיקוני זְעֵיר אַנְפִּין, מתיקוני אריך אַנְפִּין הִתְתַּקְּנוּ. והתפשטו תִּיקּוּנָיו מִכָּאן וּמִכָּאן, כמראה בן אדם, ושולף בו רוּחַ שנסתרת מכל הַנִּסְתָּרִים, כְּדֵי לשבת עַל הַכִּסֵּא, שכתוב, "וְעַל דְּמוּת הַכִּסֵּא דְּמוּת כְּמַרְאֵה אָדָם עָלָיו מִלְמָעְלָה" (יחזקאל א כו). "כְּמַרְאֵה אָדָם" - שכולל את כל הדמויות; "כְּמַרְאֵה אָדָם" - שכולל את כל הַשֵּׁמוֹת; "כְּמַרְאֵה אָדָם" - שבו סְתוּמִים כל הָעוֹלָמוֹת, עֶלְיוֹנִים וְתַחְתּוֹנִים; "כְּמַרְאֵה אָדָם" - שכולל את כל הַסּוֹדוֹת שנאמרו ונתתקנו בְּטֶרֶם נברא העולם, וְאַף עַל פִּי שלא התקיימו.


לָמַדְנוּ בנסתרות הַסֵּפֶר, עַתִּיק העתיקים בְּטֶרֶם הכין תִּיקּוּנָיו, בְּנָהּ מְלָכִים, חָקַק מְלָכִים, ושיער מְלָכִים, וְלֹא היו מתקיימים, עַד שדחה אוֹתָם, והצניע אוֹתָם לאחר זְמַן, זֶהוּ שכתוב, "וְאֵלֶּה הַמְּלָכִים אֲשֶׁר מָלְכוּ בְּאֶרֶץ אֱדוֹם" (בראשית לוֹ לא). "בְּאֶרֶץ אֱדוֹם" - במקום שכל הדינים מתקיימים שם, וכולם לא התקיימו עַד שהראש הַלָּבָן, עַתִּיק העתיקים, התתקן. וכשהתתקן - תיקן את כל התיקונים שֶׁלְּמַטָּה, תיקן את כל התיקונים של עֶלְיוֹנִים וְתַחְתּוֹנִים. מִכָּאן לימדנו, שכל רֹאשׁ לעם, שלא התתקן תְּחִילָּה - אֵין הָעָם מִתַּתְקָן. וְאִם הוּא כֵּן מִתַּתְקָן - כולם מתתקנים. וְאִם הוּא לא מִתַּתְקָן - לא יְכוֹלִים הָעָם לְהִתְתַּקֵּן.


מִנַּיִן לָנוּ? מעתיק יָמִים. שבטרם התתקן הוּא בְּתִיקּוּנָיו, לא הִתְתַּקְּנוּ כל אֵלּוּ שרצו לְהִתְתַּקֵּן, וְכָל הָעוֹלָמוֹת נחרבו. זֶהוּ שכתוב, "וַיִּמְלוֹךְ בֶּאֱדוֹם בֶּלַע בֶּן־בְּעוֹר" (בראשית לוֹ לב). "וַיִּמְלוֹךְ בֶּאֱדוֹם" - סוֹד יָקָר, הוּא מקום שכל הדינים נִקְשָׁרִים שם, ותלויים מִשָּׁם.


"בֶּלַע בֶּן־בְּעוֹר" - לָמַדְנוּ, שהוא גזר הדין, חָזָק החזקים, שבגללו מתקשרים אֶלֶף אלפי בַּעֲלֵי יבבה ויללה. "וְשֵׁם עִירוֹ דִּנְהָבָה" (בראשית לוֹ לב), מהו "דִּנְהָבָה"? כְּלוֹמַר, דין הָבֵא, כְּמוֹ שנאמר, "לַעֲלוּקָה שְׁתֵּי בָּנוֹת הַב הַב" (משלי ל טו).


כֵּיוָן שעלה להתיישב בו, לא התקיים, וְלֹא הָיָה יָכוֹל להתקיים, וְכָל עוֹלָמוֹת נחרבו, מה הַטַּעַם? מִשּׁוּם שהאדם לא התתקן, שתיקון האדם בִּדְמוּתוֹ, כּוֹלֵל הַכֹּל, וְיָכוֹל הַכֹּל להתיישב בו.


וּמִשּׁוּם שתיקון זה של אדם, לא נִמְצָא, לא יָכְלוּ לקום ולהתיישב, והתבטלו. והתבטלו, עָלָה בדעתך, וַהֲרֵי כולם באדם נִכְלְלוּ? אֶלָּא, התבטלו והסתלקו מֵאוֹתוֹ הַתִּיקּוּן, עַד שתבוא הַדְּמוּת של אדם. וכשבאה זֹאת הַדְּמוּת, נחקקו כולם, וחזרו לקיום אחר. מֵהֶם התבסמו, וּמֵהֶם לא התבסמו כלל.


וְאִם תֹּאמַר, וַהֲרֵי כתוב "וַיָּמָת". "וַיָּמָת" - שהתבטלו לגמרי? אֵין זה כך. אֶלָּא, כל מי שֶׁיּוֹרֵד מִדַּרְגָּה קודמת שהיה בה, נקראת בו 'מִיתָה', כְּמוֹ שנאמר, "וַיָּמָת מֶלֶךְ מִצְרַיִם" (שמות ב כג), שנחת מִדַּרְגָּה קודמת שהיה עוֹמֵד בה. וְכֵיוָן שהתתקן אדם, נִקְרָאִים בשמות אֲחֵרִים, והתבסמו בְּקִיּוּם בו, וקיימים בִמְקוֹמָם.


וכולם נִקְרָאִים בשמות אֲחֵרִים מהקודמים, חוּץ מֵהַהוּא שכתוב בו, "וְשֵׁם אִשְׁתּוֹ מְהֵיטַבְאֵל בַּת־מַטְרֵד בַּת מֵי זָהָב" (בראשית לוֹ לט). מה הַטַּעַם? מִשּׁוּם שֶׁאֵלּוּ לא התבטלו כִּשְׁאָר הָאֲחֵרִים, מִשּׁוּם שהיה זָכָר וּנְקֵבָה, כזה התמר, שלא עָלָה אֶלָּא זָכָר וּנְקֵבָה. וּמִשּׁוּם כך, עכשיו שנמצאו זָכָר וּנְקֵבָה, לא כתוב בהם 'מִיתָה' כאחרים, והתקיימו. אֲבָל לא התיישבו, עַד שהתתקנה הַדְּמוּת של אדם, וְכֵיוָן שהתתקנה הַדְּמוּת של אדם, חָזְרוּ והתקיימו בְּקִיּוּם אחר, והתיישבו
