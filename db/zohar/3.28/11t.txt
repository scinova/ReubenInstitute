התיקון הָרִאשׁוֹן של הזקן


הַתִּיקּוּן הָרִאשׁוֹן: הֲרֵי לָמַדְנוּ, שכל שערה ושערה, וְכָל נימה ונימה, לא נדבקת לחברתה. ומתחילים נימי הזקן לְהִתְתַּקֵּן, מהתיקון של שֵׂיעָר הָרֹאשׁ.


כָּאן יֵשׁ לְהִסְתַּכֵּל, אִם כל נימי שיערות הָרֹאשׁ, ונימי הזקן הַיָּקָר העליון, במשקל אֶחָד נשקלים, מַדּוּעַ אֵלּוּ ארוכים, וְאֵלּוּ לא ארוכים? מַדּוּעַ נימי הזקן לא ארוכים כל כך, וקשים, וְאֵלּוּ של הָרֹאשׁ לא קשים, אֶלָּא חלקים?


אֶלָּא, כל הנימים שקולים, של הָרֹאשׁ ושל הזקן. של הָרֹאשׁ ארוכים עַל כתפיים, להמשיך מהראש לראש של זְעֵיר אַנְפִּין, מאותה המשכת המוח, למוח שֶׁלּוֹ. וּמִשּׁוּם כך, לא היו קשים, ועל זה, ראוים לִהְיוֹת רכים.


לָמַדְנוּ, מה שכתוב, "חָכְמוֹת בַּחוּץ תָּרֹנָּה" (משלי א כ), וּלְבַסּוֹף כתוב, "בָּרְחוֹבוֹת תִּתֵּן קוֹלָהּ". פָּסוּק זה, אֵין תחילתו כסופו, אֵין סופו כתחילתו. אֶלָּא, "חָכְמוֹת בַּחוּץ תָּרֹנָּה" - כשנמשך מהמוח הסתום של זְעֵיר אַנְפִּין, בְּאוֹתָם נימים, כאילו מתחברים בחוץ שני מוחות, וְנַעֲשִׂים מוח אֶחָד. מִשּׁוּם שֶׁאֵין קִיּוּם למוח הַתַּחְתּוֹן, אֶלָּא בְּקִיּוּם של המוח העליון. וכשנמשך מִזֶּה לזה, כתוב, "תִּתֵּן קוֹלָהּ" - אֶחָד.


וּמִשּׁוּם שנמשך ממוח למוח בְּאוֹתָם נימים, הֵם לא נמצאים קשים. מה הַטַּעַם? מִשּׁוּם שֶׁאִם נמצאים קשים, לא נמשכת החוכמה למוח בהם. מִשּׁוּם כך, אֵין החוכמה יוצאת מבן אדם שהוא קשה ובעל רוגז. שכתוב, "דִּבְרֵי חֲכָמִים בְּנַחַת נִשְׁמָעִים" (קהלת ט יז). וּמִכָּאן לָמַדְנוּ, מי ששערות רֹאשׁוֹ קשות, אֵין החוכמה מתישבת איתו.


ועל זה הֵן ארוכות, לְהָבִיא תועלת לכל. מהו 'לכל'? לְהִכָּנֵס עַל חוט השדרה, שמושקים מהמוח. וּמִשּׁוּם זה, לא תָּלוּי שֵׂיעָר הָרֹאשׁ עַל שֵׂיעָר הזקן. ששיער הָרֹאשׁ תָּלוּי וְעוֹלֶה עַל אוזניו לאחורו, וְלֹא תָּלוּי עַל הזקן, מִשּׁוּם שלא צָרִיךְ להערב אֵלּוּ בְּאֵלּוּ, שכולם מתחלקים בדרכם.


לָמַדְנוּ, שכל השערות - בֵּין של הָרֹאשׁ, וּבֵין של הזקן - כולן לבנות כַּשֶּׁלֶג. ולמדנו, אֵלֶּה של הזקן - קשות כולן. מה הַטַּעַם? מִשּׁוּם שֶׁהֵן חזקות מכל חוזק, להוריד אוֹתָן שְׁלוֹשׁ עֶשְׂרֵה מידות, מעתיק העתיקים. וְאֵלֶּה המידות מִלִּפְנֵי הָאָוזְנַיִים מתחילות, וְאֵלֶּה המידות - סתומות הֵן. שלא תתערבנה באחרות.


וְאִם תֹּאמַר שֶׁאֵין אֲחֵרוֹת כמוהן - לא. שלמדנו שְׁלוֹשׁ עֶשְׂרֵה מידות של רַחֲמִים מהעתיק הקדוש: "מִי־אֵל כָּמוֹךָ" - אַחַת. "נוֹשֵׂא עָוֹן" - שתיים. "וְעוֹבֵר עַל־פֶּשַׁע" - שְׁלוֹשׁ. "לִשְׁאֵרִית נַחֲלָתוֹ" - אַרְבַּע. "לֹא־הֶחֱזִיק לָעַד אַפּוֹ" - חָמֵשׁ. "כִּי־חָפֵץ חֶסֶד הוּא" - שֵׁשׁ (מיכה ז יח). "יָשׁוּב יְרַחֲמֵנוּ" - שֶׁבַע. "יִכְבּוֹש עֲוֹנוֹתֵינוּ" - שמונה. "וְתַשְׁלִיךְ בִּמְצוּלוֹת יָם כָּל־חַטֹּאתָם" - תשע (מיכה ז יט). "תִּתֵּן אֱמֶת לְיַעֲקֹב" - עשר. "חֶסֶד לְאַבְרָהָם" - אַחַת עֶשְׂרֵה. "אֲשֶׁר־נִשְׁבַּעְתָּ לַאֲבוֹתֵינוּ" - שתיים עֶשְׂרֵה. "מֵימֵי קֶדֶם" - שְׁלוֹשׁ עֶשְׂרֵה (מיכה ז כ). כְּנֶגֶד זה, "אֵל רַחוּם וְחַנּוּן ..." (שמות לד ו) - הֵן לְמַטָּה.


וְאִם תֹּאמַר, מֹשֶׁה אֵיךְ לא אמר אֵלֶּה עֶלְיוֹנוֹת? אֶלָּא, מֹשֶׁה לא הצטרך, אֶלָּא לַמָּקוֹם שֶׁהַדִּין נִמְצָא, ובמקום שֶׁהַדִּין נִמְצָא, לא צָרִיךְ כך לוֹמַר. ומשה לא אמר, אֶלָּא בִּזְמַן שֶׁיִּשְׂרָאֵל חָטְאוּ, וְהַדִּין הָיָה תָּלוּי, וּמִשּׁוּם כך לא אמר מֹשֶׁה, אֶלָּא במקום שֶׁהַדִּין נִמְצָא. אֲבָל בזה הַמָּקוֹם - סדור שֶׁבַח עַתִּיק הַיָּמִים - מסדר הנביא.


וְאוֹתָם שלושה עשר תיקוני הזקן עֶלְיוֹן הקדוש, נִסְתָּר הַנִּסְתָּרִים - חֲזָקִים לשבור ולאכוף כל גזרי הדינים. מי שרואה הזקן העליון הקדוש, טמון הטמונים, שלא יתבייש מִמֶּנּוּ?! וּמִשּׁוּם כך, כל שערותיו קשות, וחזקות בְּתִיקּוּנָיו.


וְאִם תֹּאמַר, אִם כך, הֲרֵי שְׂעָרוֹת שֶׁלְּמַטָּה, הֵן שחורות, לָמָּה לא היו זֹאת כזאת? שלמדנו שכתוב,  "קְווּצּוֹתָיו תַּלְתַּלִּים שְׁחוֹרוֹת כָּעוֹרֵב" (שִׁיר השירים הַ יא). וכתוב: "וּשְׂעַר רֵישֵׁיהּ כַּעֲמַר נְקֵא" (דניאל ז ט). לא קשה - זה בַּזָּקָן העליון, זה בַּזָּקָן הַתַּחְתּוֹן. ועל זה, כשניתנה הַתּוֹרָה לְיִשְׂרָאֵל, ניתנה בָּאֵשׁ שחורה עַל גַּבֵּי אֵשׁ לבנה.


ועיקר הדבר, מִשּׁוּם שְׁאֵלָה השערות, מִשּׁוּם שמהמוח נמצאים, להמשיך למוח שֶׁלְּמַטָּה, טהן לְמַעְלָה מן הזקן. הזקן בפני עַצְמוֹ הוּא, וְכָל תִּיקּוּנָיו בפני עצמם הֵם, הזקן בפני עַצְמוֹ, והשערות בפני עצמן.


הַתִּיקּוּן הָרִאשׁוֹן הוּא תִּיקּוּן שמתחיל מתחילת שְׂעָרוֹת הָרֹאשׁ. ולמדנו, כל תיקוני הזקן, לא נמצאים אֶלָּא ממוח הָרֹאשׁ. וְכָאן לא מפורש כך, שֶׁהֲרֵי לא נִמְצָא אֶלָּא תִּיקּוּן זה, שֶׁיּוֹרֵד מתחילת שְׂעָרוֹת הָרֹאשׁ, כך נִמְצָא.


ומזה הזקן, נודע כל מה שבראש, שאלף עוֹלָמוֹת חתומות בטבעת שטהורה, טבעת שֶׁכּוֹלֶלֶת את כל הטבעות.


האורך של כל שערה, שֶׁיּוֹרֶדֶת מִלִּפְנֵי אוזניו, אֵינָהּ ארוכה, וְלֹא נדבקת זֹאת בזאת, וְלֹא יורדות אֵלֶּה השערות, מכשנמשכות, נמשכות ותלויות.


והתחלת הַתִּיקּוּן הָרִאשׁוֹן, שלושלים וְאֶחָד קוצים שווים, נמשכים עַד תחילת הַפֶּה. ושלוש מאות וְתִשְׁעִים נימים נמצאים בכל קוץ וקוץ.


שלושים וְאֶחָד קוצים שווים, שֶׁהָיוּ בְּתִיקּוּן הָרִאשׁוֹן, כוחם לאכוף לְמַטָּה, כחשבון 'אל'. מהו 'אל'? חָזָק, יָכוֹל. וּבְכָל קוץ וקוץ, מתחלקים שלושים וְאֶחָד עוֹלָמוֹת, חֲזָקִים, שולטים, לעיין, שלושים וְאֶחָד בצד זה, ושלושים וְאֶחָד בצד זה. וְכָל עוֹלָם ועולם מִמֶּנּוּ, מתחלק לאלף עוֹלָמוֹת שמשתוקקים לעידון רַב. וְהַכֹּל סתום בָּרֹאשׁ הזקן שכלול וְחָזָק, וכלולים בזה הַ'אל'.וְאִם כל זה, זה הַ'אל' אוכף לרחמים, שרחמי עַתִּיק הַיָּמִים, מתבסם, ונכלל, ומתפשט בו.


מַדּוּעַ עַד הַפֶּה? מִשּׁוּם שכתוב, "דִּינָא יְתִיב וְסִפְרִין …(הדין יָשׁוּב וספרים ... )" (דניאל ז י). מהו "דִּינָא יְתִיב (הדין יָשׁוּב)"?. יָשׁוּב לִמְקוֹמוֹ שלא ישלוט. זֶהוּ שכתוב, "פֶּלֶא יוֹעֵץ אֵל גִּבּוֹר" (ישעיה ט הַ) - "אֵל" שהוא "גִּבּוֹר", ומתבשם בַּזָּקָן הקדוש של עַתִּיק הַיָּמִים. והסוד שכתוב, "מִי אֵל כָּמוֹךָ" (מיכה ז יח), עַל עַתִּיק הַיָּמִים נֶאֱמַר, בְּתִיקּוּן הָרִאשׁוֹן של הזקן הקדוש העליון.


העולם הָרִאשׁוֹן, שֶׁיּוֹצֵא מהתיקון הָרִאשׁוֹן, שׁוֹלֵט וְיוֹרֵד, וְעוֹלֶה לאלף אלפי רבבי רבבות בַּעֲלֵי מגינים. וּמִמֶּנּוּ מתאחדים, במידה בטבעת גְּדוֹלָה.


העולם השני, שֶׁיּוֹצֵא מִזֶּה הַתִּיקּוּן, שׁוֹלֵט, וְיוֹצֵא, וְיוֹרֵד, וְעוֹלֶה לחמישים ושבעה אלפי דְרָגוֹת, בַּעֲלֵי יבבה, ומתאחדות מִמֶּנּוּ, לאכוף בעורף בלבן.


העולם הַשְּׁלִישִׁי, שֶׁיּוֹצֵא מִזֶּה הַתִּיקּוּן, שׁוֹלֵט, וְיוֹרֵד, וְעוֹלֶה לתשעים ושישה אלפי בַּעֲלֵי יללה, ומתאחדים מִמֶּנּוּ כבוסיטא לבדיל. ומזה הַתִּיקּוּן, נאכפים כולם, ומתבסמים במרירות הַדְּמָעוֹת, שמתבסמות בַּיָּם הגדול.


מי שרואה תִּיקּוּן זה, של הזקן הקדוש, העליון, הַיָּקָר, שלא יתבייש מִמֶּנּוּ?! מי רוֹאָה נִסְתָּרוֹת קוצי השערות שֶׁתְּלוּיוֹת מִזֶּה הזקן. יוֹשֵׁב בעטרת העטרות, עֲטָרוֹת כל העטרות, עֲטָרוֹת שלא נכללות בַּעֲטָרוֹת, עֲטָרוֹת שלא כִּשְׁאָר העטרות, עֲטָרוֹת, שעטרות שֶׁלְּמַטָּה נאחזות מֵהֶן. וּמִשּׁוּם כך, אֵלּוּ התיקונים, הֵם תִּיקּוּנִים שֶׁלְּמַטָּה, מֵהֶם נאחזים.


תִּיקּוּנִים שמתתקנים, שֶׁצָּרִיךְ להתברך, מי שֶׁצָּרִיךְ בְּרָכָה. שכל התיקונים שמתתקנים, כְּנֶגְדָּם בְּרָכוֹת נמצאות, ונעשה מה שֶׁנַּעֲשֶׂה. הַכֹּל כלול בְּאֵלֶּה התיקונים, הַכֹּל זוקפים כְּנֶגֶד תיקוני הַמֶּלֶךְ הֶחָזָק, הָעַתִּיק הסתום מֵהַכֹּל, וכולם מתבסמים מתיקונים אֵלֶּה.


לָמַדְנוּ, אִם עַתִּיק העתיקים, קדוש הַקְּדוֹשִׁים, לא מִתַּתְקָן בְּאֵלֶּה התיקונים - לא נמצאים עֶלְיוֹנִים ותחותומים, וְהַכֹּל היו כאילו לא היו. ולמדנו, עַד כַּמָּה זוהרים אֵלֶּה תיקוני הזקן? שלושה עשר. וְכָל זְמַן ששלושה עשר אךה נמצאים, זוהרים אֵלֶּה שֶׁלְּמַטָּה, וְהַכֹּל בחשבון של אֵלֶּה שלושה עשר, נמצק זקן הַמֶּלֶךְ הָעַתִּיק, הַיָּקָר מכל. הַכֹּל כְּאֶחָד הוּא, נִסְתָּר וְיָקָר.


וּמִשּׁוּם שהוא יָקָר ונסתר מֵהַכֹּל, לא נִזְכָּר בתורה, וְלֹא מתגלה. ואיזה זקן מתגלה? הזקן של הכהן הגדול העליון. ומזה הזקן, יורד לַזָּקָן הכהן הגדול שֶׁלְּמַטָּה. מהו זקן הכהן הגדול? זקן הכהן הגדול השמונה תִּיקּוּנִים מִתַּתְקָן. וּמִשּׁוּם כך, שמונה תִּיקּוּנִים לכהן הגדול, כשמשחה יורדת עַל זקנו, זֶהוּ שכתוב, "כַּשֶׁמֶן הַטּוֹב עַל־הָרֹאשׁ יוֹרֵד עַל־הַזָּקָן" (תהלים קלג ב).


ומאין לָנוּ? שכתוב, "שֶׁבֶת אַחִים גַּם יָחַד" (תהלים קלג א). "גַּם" - לְרַבּוֹת כֹּהֵן גָּדוֹל שֶׁלְּמַטָּה, שכל זְמַן שהכהן הגדול שֶׁלְּמַטָּה, מְשַׁמֵּשׁ בכהונה גְּדוֹלָה, כִּבְיָכוֹל כֹּהֵן גָּדוֹל שֶׁלְּמַעְלָה, מְשַׁמֵּשׁ בכהונה גְּדוֹלָה.


זה תִּיקּוּן אֶחָד של זקן הָעַתִּיק, הסתום מֵהַכֹּל. אמר לוֹ רבי שִׁמְעוֹן: נאה לך רבי יִצְחָק, לִרְאוֹת בִּיקָר תיקוני הזקן, וסבר פְּנֵי עַתִּיק הַיָּמִים, עַתִּיק העתיקים. אַשְׁרֵי חֶלְקְךְ, ואשרי חֶלְקִי עמכם בעולם הבא
