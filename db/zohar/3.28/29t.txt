עַיְינִין


עֵינָיו של הָרֹאשׁ, משתנים משאר העיניים. הצבע שבגב שֶׁעַל ריסי העיניים, צבועים בשחורות, תלויים תלים עַל תלים של שְׂעָרוֹת, וְהֵם הַתִּיקּוּן שֶׁעַל העיניים, בתחילת המצח, וְנֶאֱחָזִים משניהם שֶׁבַע מאות אלפי בַּעֲלֵי ההשגחה.


בכסות העיניים, לוהטים אֶלֶף וְאַרְבַּע מאות רבבות, שנאחזים בגבינים שֶׁהֵם הכסות. וההשגחת עַיִן עַתִּיק יָמִים עָלָיו. ובשעה שֶׁעוֹלִים אוֹתָם הכסות, נראה כְּמִי שהעתעורר משנתו, ונפקחים עֵינָיו, ורואות לעין הפקוחה, ורוחצות בלבן אֶחָד של הָעַיִן הטובה,  זֶהוּ שכתוב, "רוֹחֲצוֹת בֶּחָלָב" (שִׁיר השירים הַ יב). מהו "בֶּחָלָב"? בלבן שֶׁלְּמַעְלָה הָרִאשׁוֹן. וּבְאוֹתָהּ שָׁעָה נִמְצֵאת ההשגחה של רַחֲמִים.


ועל זה התפלל דָּוִד: "עוּרָה לָמָּה תִישַׁן אֲדנָי, הָקִיצָה" (תהלים מד כד) - שיפקח עֵינָיו, ויתרחצו בְּאוֹתוֹ הַלָּבָן. וְכָל הַזְּמַן שעיניו לא נפקחות - כל  בעליהם של דינים, כופים לָהֶם לְיִשְׂרָאֵל, וּשְׁאָר עמים שולטים עֲלֵיהֶם. וּבִזְמַן שיפקח עֵינָיו, יתרחץ בעין טוֹבָה, וְרַחֲמִים עַל ישראל, ומסתובבת הָעַיִן, וְעוֹשֶׂה נקמות בִּשְׁאָר הָעַמִּים. זֶהוּ שכתוב, "הָעִירָה וְהָקִיצָה" (תהלים לָהּ כג). "הָעִירָה" - להתרחץ בְּאוֹתוֹ הַלָּבָן; "הָקִיצָה" - לַעֲשׂוֹת נקמות בְּאֵלּוּ שכופים לָהֶם.


עֵינָיו, כשנפקחות, נִרְאוּ יפות כְּאִלּוּ היונים, באדום ושחור וירוק, לָבָן לא מתגלה, אֶלָּא בִּזְמַן שמסתכל בעין טוֹבָה, ומתרחצים כל אוֹתָם גוונים, בְּאוֹתוֹ לָבָן.


מֵאוֹתָם גוונים שמתגלים, יוֹצְאוֹת שֶׁבַע עיניים של השגחה, שיוצאות מהשחור של הָעַיִן, זֶהוּ שכתוב, "עַל־אֶבֶן אַחַת שִׁבְעָה עֵינָיִם" (זכריה ג ט). מי "אֶבֶן אַחַת"? השחורות של הָעַיִן.


מהשחור, יוצאים שבעה רצים, שסומכים לצד השמאל, ומתלהטים בָּאֵשׁ שלצד צפון, וְנֶאֱחָזִים להתפשט בעולם, לְגַלּוֹת דַּרְכִּי הָחוטאים, זֶהוּ שכתוב, "שִׁבְעָה־אֵלֶּה עֵינֵי יְהֹוָה הֵמָּה מְשׁוֹטְטִים בְּכָל־הָאָרֶץ" (זכריה ד י).


מהירוק, יוצאים שבעה צוהרים של קֶשֶׁר שלצד הדרום, וְנֶאֱחָזִים להתפשט בעולם, לְגַלּוֹת דרכים ומעשים של בני אדם, בֵּין טוב, בֵּין רע, שכתוב, "כִּי־עֵינָיו עַל־דַּרְכֵי־אִיש" (איוב לד כא).


וכשמתרחצים בלבן, נמצאים כולם להשגיח לכל בַּעֲלֵי אֱמֶת, להטיב עִם העולם בגללם. וְכָל ההשגחה של אוֹתוֹ הַלָּבָן, הינה לטוב עַל ישראל, ומשגיח באדום לְמִי שמעיקים לָהֶם. זֶהוּ שכתוב, "רָאֹה רָאִיתִי" (שמות ג ז). "רָאֹה" - להטיב לָהֶם; "רָאִיתִי" - לנקום לָהֶם, משמעיקים לָהֶם. וּמִשּׁוּם כך כתוב, "עוּרָה לָמָּה תִישַׁן אֲדֹנָי, הָקִיצָה אַל־תִּזְנַח לָנֶצַח" (תהלים מד כד). "עוּרָה", וְ"הָקִיצָה" - שְׁתֵּי השגחות, שְׁתֵּי פקוחות, שְׁתֵּי טוֹבוֹת, רַחֲמִים ונקמות.


הגוון הָרִאשׁוֹן: אדום בתוך אדום, כּוֹלֵל וסותם את כל האדומים, מִלְּפָנָיו לא נִרְאוּ. מסביבו של אוֹתוֹ האדום, סוֹבֵב חוט אֶחָד שחור, ומקיף אוֹתוֹ.


הגוון שני: שחור, כאבן אַחַת שֶׁיּוֹצֵאת מהתהום, פַּעַם אַחַת לאלף שנים, בַּיָּם הגדול. וכשיוצאת זֹאת האבן, בא רעש וזעף עַל הים. וקולו של הים וגליו הוֹלְכִים, ונשמעים לדג הגדול, שֶׁנִּקְרָא 'לִוְיָתָן', וְיוֹצֵא מהתהום. וזאת האבן מתגלגלת בזעף הים, ויוצאת הַחוּצָה. והיא שחורה, שכל השחורים סְתוּמִים לְפָנֶיהָ. וכך הִיא השחורות של הָעַיִן השחורה, שכולל וסותם את כל שְׁאָר השחורים. וסביבו של אוֹתוֹ השחור, סוֹבֵב חוט אֶחָד אדום, ומקיף את אוֹתוֹ האדום.


הגוון הַשְּׁלִישִׁי: ירוק הירוקים, שכולל וסותם את כל הירוקים. ובסביבו של אוֹתוֹ הירוק, סובבים שני חוטים - חוט אדום לצד אֶחָד, וחוט אֶחָד אדום לצד אֶחָד, ומקיפים את אוֹתוֹ הירוק.


וכשמתגלה הַלָּבָן, ונרחצת הָעַיִן, כל אוֹתָם הגוונים לא נמצאים, ומשתקעים לְמַטָּה (האדום, הירוק, והשחור), וְלֹא נראה חוּץ מֵאוֹתוֹ הַלָּבָן, שֶׁמֵּאִיר מעתיק יָמִים, וּמְאִירִים מִמֶּנּוּ כל אוֹתָם שֶׁלְּמַטָּה.


וְאֵין גוון נראה, חוּץ מֵאוֹתוֹ הַלָּבָן לְבַדּוֹ. וּמִשּׁוּם כך, הסתלקו כל בעליהם של האדום ושל הירוק, שֶׁהֵם תאומים כְּאֶחָד. זֶהוּ שכתוב, "שִׁנַּיִךְ כְּעֵדֶר הַקְּצוּבוֹת שֶׁעָלוּ מִן־הָרַחְצָה שֶׁכֻּלָּם מַתְאִימוֹת" (שִׁיר השירים ד ב). מהו "מִן־הָרַחְצָה"? מאותה הרחיצה של הָעַיִן הַקְּדוֹשָׁה הָעֶלְיוֹנָה. "שֶׁכֻּלָּם מַתְאִימוֹת" - מתערבים זה בזה, ונדבקים זה בזה. ומה שנאמר "שִׁנַּיִךְ כְּעֵדֶר הַקְּצוּבוֹת", וְנאמר "שֶׁכֻּלָּם מַתְאִימוֹת"? כְּלוֹמַר, הַלָּבָן שֶׁלָּהֶם, כאותו הַלָּבָן של העיניים, כשרוחצים בלבנות של הָעַיִן הָעֶלְיוֹנָה.


ואת זה עתידים לדעת צַדִּיקִים, לִרְאוֹת ברוח החוכמה, כְּמוֹ שנאמר, "כִּי עַיִן בְּעַיִן יִרְאוּ" (ישעיה נב ח). מָתַי? "בְּשׁוּב יְהֹוָה צִיּוֹן" (ישעיה נב ח). וכתוב: "אֲשֶׁר־עַיִן בְּעַיִן נִרְאָה אַתָּה יְהֹוָה" (במדבר יָד יָד), וְאָז הפקיחות העיניים לטוב.


ויש פקיחות של עיניים לטוב, ויש פקיחות של עיניים לרע ולטוב, כְּמוֹ שכתוב, "פְּקַח עֵינֶיךָ וּרְאֵה שׁוֹמְמוֹתֵינוּ" (דניאל ט יח). זה כָּאן לטוב, ולרע. וכתוב, "עֵינֶיךָ תִרְאֶינָה יְרוּשָׁלִַם נָוֶה שַׁאֲנָן אֹהֶל בַּל־יִצְעָן בַּל־יִסַּע יְתֵדֹתָיו לָנֶצַח" (ישעיה לג כ). הֲרֵי כָּאן לטוב ולרע, שלא נעשה זה בְּלִי זה.


לָמַדְנוּ בנסתרות הַסֵּפֶר, מהו "עֵינֶיךָ תִרְאֶנָה יְרוּשָׁלַם נָוְה שַׁאֲנָן". וְכִי "יְרוּשָׁלַם" - "נָוְה שַׁאֲנָן" הוּא, וַהֲרֵי כתוב, "צֶדֶק יָלִין בָּה" (ישעיה א כא). ובמקום שנמצא צדק - אֵינוֹ שקט, וְאֵין "שַׁאֲנָן" הוּא? אֶלָּא, "עֵינֶיךָ תִרְאֶנָה יְרוּשָׁלַם נָוְה שַׁאֲנָן": "נָוְה שַׁאֲנָן" - לְעַתִּיק יָמִים נֶאֱמַר, שֶׁאוֹתָהּ הָעַיִן שקטה ושאננה, עַיִן של רַחֲמִים, עיו שלא נוטלת מההשגחה זֹאת, להשגחה אַחֶרֶת. וּמִשּׁוּם כך כתוב, "עֵינֶךָ תִּרְאֶינָּה" (ישעיה א כא), חָסֵר יוֹ״ד, וְלֹא 'עֵינֶיךָ' .וּמַה שֶׁאָמַר "יְרוּשָׁלַם" וְלֹא 'צִיּוֹן', כך צָרִיךְ, לכפות את הדין שנמצא בה, ולרחם עָלֶיהָ, שכתוב, "צֶדֶק יָלִין בָּה", שָׁבָה נמצאו גְּזֵרוֹת דינים יוֹתֵר מכל שְׁאָר המקומות.


ולמדנו, שכתוב, "עֵינֵי יְהֹוָה אֱלֹהֶיךָ בָּהּ מֵרֵשִׁית הַשָּׁנָה וְעַד אַחֲרִית שָׁנָה" (דְּבָרִים יא יב). ולזמן שיבוא, יימצא בה עַיִן אַחַת של רַחֲמִים, עַיִן של עַתִּיק העתיקים, זֶהוּ שכתוב, "וּבְרַחֲמִים גְּדוֹלִים אֲקַבְּצֵךְ" (ישעיה נד ז). כֵּיוָן שֶׁאָמַר, "רַחֲמִים", מַהוּ "גְּדוֹלִים"? אֶלָּא, יֵשׁ רַחֲמִים, ויש רַחֲמִים. רַחֲמִים של עַתִּיק העתיקים, הֵם נִקְרָאִים 'רַחֲמִים גְּדוֹלִים'; רַחֲמִים של זְעֵיר אַנְפִּין, נִקְרָאִים 'רַחֲמִים סְתָם'. וּמִשּׁוּם כך, "וּבְרַחֲמִים גְּדוֹלִים אֲקַבְּצֵךְ" - של עַתִּיק יָמִים.


לָמַדְנוּ, שבאלו העיניים, בִּשְׁנֵי גוונים מֵהֶן, באדום ובשחור, שורות שְׁתֵּי דְּמָעוֹת. וכשרוצה קדוש הַקְּדוֹשִׁים לְרַחֵם עַל ישראל, מוֹרִיד שְׁתֵּי דְּמָעוֹת, להתבשם בַּיָּם הגדול. מהו הים הגדול? הים של החוכמה הָעֶלְיוֹנָה. כְּלוֹמַר, שיתרחצו בלבן, במבועה שֶׁיּוֹצֵאת מהחכמה הגדולה, ומרחם עַל ישראל.
