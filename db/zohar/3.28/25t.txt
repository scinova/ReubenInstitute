גולגלתא דִּזְעֵיר אַנְפִּין


לָמַדְנוּ, שכשעלה ברצון הָרֹאשׁ הַלָּבָן, לַעֲשׂוֹת יוקרה ליוקרתו, תיקן, והכין, וְהוֹצִיא מנר של אפלה, ניצוץ אֶחָד, נֵשֵׁב בו, התקרר, וְעָלָה רְצוֹנוֹ, והתפשט לשלוש מאות וְשִׁבְעִים עיברים, והניצוץ עוֹמֵד. וּמַתְחִיל לצאת האויר הטהור, ומתגלגל, נֵשֵׁב בו, התתקן והתפשט, וְיָצְאָה גולגולת אַחַת חֲזָקָה, והתפשט לְאַרְבָּעָה צְדָדִים.


וּבָזֶה האויר הטהור, נשאב הניצוץ והתאחד, ונכלל בו. 'בו', עָלָה בדעתך? אֶלָּא, נִסְתָּר 'בו'. וּמִשּׁוּם כך, זֹאת הגולגולת מתפשטת בצדדיה, וְזֶה האויר הוּא נִסְתָּר הַנִּסְתָּרִים של עַתִּיק יָמִים, ברוח שֶׁגָּנַז.


בזאת הגולגולת התפשטו הָאֵשׁ מִצַּד אֶחָד, והאויר מִצַּד אֶחָד. והאויר הטהור עוֹמֵד עָלֶיהָ מִזֶּה הַצַּד; והאש הטהורה עוֹמֶדֶת מִזֶּה הַצַּד. מהו הָאֵשׁ כָּאן? אֶלָּא, אֵין הִיא אֵשׁ, אֲבָל הניזוץ הזה, שנכלל בָּאֲוִיר הטהור, מֵאִיר למאתיים וְשִׁבְעִים עוֹלָמוֹת, וְהַדִּין מצדדיו נִמְצָא, וּבִגְלַל זה, זֹאת הגולגולת נקראת 'גולגולת חֲזָקָה'.


בגולגולת זֹאת, יוֹשְׁבִים תשעת אלפי רבבות עוֹלָמוֹת, שנוטלים עָלֶיהָ, וסומכים עָלֶיהָ. בזאת הגולגולת, נוטף טל מהראש הַלָּבָן, שמתמלא מִמֶּנָּה תמיד. ומזה הטל שמתנער מראשו, עתידים המתים להחיות.


והוא טל שנכלל בִּשְׁנֵי גוונים, מִצַּד הָרֹאשׁ הַלָּבָן, לָבָן בְּתוֹכוֹ, שכולל כל הלבנים. אֲבָל כשמתיישבים בזה הָרֹאשׁ של זְעֵיר אַנְפִּין, מתגלה בו האדום, כזה הבדולח שהוא לָבָן, ומתגלה הגוון האדום הגוון הַלָּבָן.


וּמִשּׁוּם כך כתוב, "וְרַבִּים מִיְּשֵׁנֵי אַדְמַת־עָפָר יָקִיצוּ, אֵלֶּה לְחַיֵּי עוֹלָם, וְאֵלֶּה לַחֲרָפוֹת לְדִרְאוֹן עוֹלָם" (דניאל יב ב). "לְחַיֵּי עוֹלָם" - מִשּׁוּם שראוים לְאוֹתוֹ הַלָּבָן, שֶׁבָּא מהצד של עַתִּיק יָמִים, ארוך הפנים. "לַחֲרָפוֹת לְדִרְאוֹן עוֹלָם" - מִשּׁוּם שראוים לְאוֹתוֹ האדום של זְעֵיר אַנְפִּין, וְהַכֹּל כלול בְּאוֹתוֹ הטל, זֶהוּ שכתוב, "כִּי טַל אוֹרוֹת טַלֶּךָ" (ישעיה כו יט). "אוֹרוֹת" - שניים. וְאוֹתוֹ הטל שנוטף, נוטף כל יוֹם לשדה תַּפּוּחִים, כגווני הַלָּבָן והאדום.


זֹאת הגולגולת מֵאִיר בִּשְׁנֵי גוונים, לזה הַצַּד ולזה הַצַּד. ומזה האויר הטהור, מתפשט מהגולגולת לְפָנָיו מאה וחמישים רבבות עוֹלָמוֹת. וּמִשּׁוּם כך נקרא 'זְעֵיר אַנְפִּין'. ובשעה שֶׁצָּרִיךְ, התפשטו פָּנָיו, וארוכים בְּאוֹתוֹ הַזְּמַן, מִשּׁוּם שהבחין בפניו של עַתִּיק העתיקים, וְחָס עַל העולם.


ומזאת הגולגולת, יוצא עיבר אֶחָד, לכל אֵלּוּ שֶׁלְּמַטָּה, וְנוֹתְנִים שכר הלבנה לְעַתִּיק יָמִים. כשעולים בחשבון תַּחַת השרביט. וּכְנֶגֶד זה, "בֶּקַע לַגוּלְגֹּלֶת" (שמות לח כו) לְמַטָּה, כשעולים בחשבון. וְזֶה הַ"בֶּקַע" הוּא שכר הלבנה, נִמְצָא מִמֶּנּוּ לְעַתִּיק ימים
