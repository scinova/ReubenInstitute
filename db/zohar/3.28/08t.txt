

הָעַיִן של ארוך פָּנִים


לָמַדְנוּ: המצח הזה מתפשט במאתיים וְשִׁבְעִים אלפי אוֹרוֹת הַנֵּרוֹת (המאירים) מ'עדן' העליון. שלמדנו: יֵשׁ 'עֵדֶן' שֶׁמֵּאִיר ל'עדן'. 'עֵדֶן' עֶלְיוֹן לא מתגלה, והוא סתום בסתימה, וְלֹא מתחלק לדרכים, כְּמוֹ שֶׁאָמַרְנוּ. וה'עדן' הזה שֶׁלְּמַטָּה, מתחלק בשביליו, לשלושים ושניים שבילים.


ואף־על־פי שמתחלק גן־עֵדֶן הזה בשביליו - אין מי שידע לו, חוץ מקטן־פנים הזה. ןגן־עדן שלמעלה - אין מי שידע לו, ולא את שביליו, חוץ מארוך־פנים ההוא.

זה הוּא שכתוב: "אֱלֹהִים הֵבִין דַּרְכָּהּ וְהוּא יָדַע אֶת מְקוֹמָהּ" (איוב כח כג).
"אֱלֹהִים הֵבִין דַּרְכָּהּ" - זה גן עדן שלמטה, שידע זעיר־אנפין.
"וְהוּא יָדַע אֶת מְקוֹמָהּ" - זה גַּן עֵדֶן שֶׁלְּמַעְלָה, שידע עַתִּיק־יָּמִים, הסתום מֵהַכֹּל.











עינים


עֵינָיו של הָרֹאשׁ הַלָּבָן, משתנות הֵן משער העיניים - אֵין כסות עַל הָעַיִן, וְאֵין לָהֶן גבות עַל הָעַיִן. מה הַטַּעַם? שכתוב: "הִנֵּה לֹא יָנוּם וְלֹא יִישָׁן שׁוֹמֵר יִשְׂרָאֵל" (תהלים קכא ד). ישראל שֶׁלְּמַעְלָה. וכתוב: "אֲשֶׁר עֵינֶיךָ פְּקוּחוֹת" (ירמיה לב יט). ושנינו: שכל מה שֶׁבָּא ברחמים - אֵין לוֹ כסות עַל הָעַיִן, וְאֵין לוֹ גבות עַל הָעַיִן. כל שֶׁכֵּן הָרֹאשׁ הַלָּבָן, שלא צָרִיךְ כְּלוּם.


אָמַר רִבִּי שִׁמְעוֹן לְרִבִּי אַבָּא: לָמָּה זה רמוז? אמר לוֹ: לדגי הים, שֶׁאֵין לָהֶם כסות עַל הָעַיִן, וְאֵין לָהֶם גבות עַל הָעַיִן, וְלֹא יְשֵׁנִים, וְלֹא צריכים שמירה עַל הָעַיִן. כל שֶׁכֵּן עַתִּיק מכל עַתִּיק, שלא צָרִיךְ שמירה. וְכָל שֶׁכֵּן שהוא משגיח עַל הַכֹּל, וְהַכֹּל ניזון מִמֶּנּוּ וְלֹא ישן, זֶהוּ שכתוב, "הִנֵּה לֹא יָנוּם וְלֹא יִישָׁן שׁוֹמֵר יִשְׂרָאֵל", יִשְׂרָאֵל שֶׁלְּמַעְלָה.


כתוב: "הִנֵּה עֵין יְהֹוָה אֶל יְרֵאָיו" (תהלים לג יח), וכתוב: "עֵינֵי יְהֹוָה הֵמָּה מְשׁוֹטְטִים בְּכָל הָאָרֶץ" (זכריה ד י). לא קשה - זה בזעיר אַנְפִּין, זה בארוך הפנים. וְעִם כל זה, שְׁתֵּי עיניים הֵן, וחזרו לאחד.

עַיִן שֶׁהִיא לָבָן בתוך לָבָן, וְלָבָן שכולל כל לָבָן.


לָבָן הָרִאשׁוֹן, מֵאִיר וְעוֹלֶה, וְיוֹרֵד לְהִסְתַּכֵּל, שצרור בצרור. לָמַדְנוּ, שֶׁהִכָּה זה הַלָּבָן, והדליק שְׁלוֹשׁ נֵרוֹת, שנראים 'הוֹד', 'הָדָר', וְ'חֶדְוָה', ולוהטים בשמחה בִּשְׁלֵמוּת.


לָבָן השני, מֵאִיר וְעוֹלֶה וְיוֹרֵד, וְמכה ומוציא (ומדליק) שלושה נֵרוֹת אחרים, שֶׁנִּקְרָאִים 'נֶצַח', 'חֶסֶד' וְ'תִפְאֶרֶת', ולוהטים בשלימות בשמחה.


לָבָן הַשְּׁלִישִׁי, לוהט וּמֵאִיר, וְיוֹרֵד וְעוֹלֶה וְיוֹצֵא מסתימת המוח. וּמַכָּה בנר האמצעי השביעי, וּמוֹצִיא דֶּרֶךְ למוח הַתַּחְתּוֹן, ומתלהטים כולם הַנֵּרוֹת שֶׁלְּמַטָּה. אמר רבי שִׁמְעוֹן: נאה הוּא, וְעַתִּיק הַיָּמִים יפקח עַיִן זֹאת עליך, בשעה שתצטרך לָהּ.


לָמַדְנוּ, לָבָן בתוך לָבָן, וְלָבָן שכולל כל לָבָן. לָבָן הָרִאשׁוֹן, מֵאִיר, וְעוֹלֶה, וְיוֹרֵד לְמַטָּה, לשלושת הַנֵּרוֹת שלצד שְׂמֹאל, ולוהטים וְרוֹחֲצִים בזה הַלָּבָן, כְּמִי שרוחץ בבשמים טובים, ובריחות, עַל מה שֶׁהָיוּ עָלָיו בַּהַתְחָלָה.


לָבָן השני יורד, וְעוֹלֶה, וּמֵאִיר לשלושת הַנֵּרוֹת שלצד יָמִין, ולוהטים, וְרוֹחֲצִים בזה הַלָּבָן, כְּמִי שרוחץ בבשמים טובים ובריחות, עַל מה שהיה עָלָיו בַּהַתְחָלָה.


לָבָן הַשְּׁלִישִׁי מֵאִיר, וְעוֹלֶה, וְיוֹרֵד, וְיוֹצֵא הארת הַלָּבָן שלתוֹךְ תוכו של המוח, וּמַכָּה בשער שחור, כשמצטרך, ובראש, ובמוח הָרֹאשׁ, וּמֵאִיר לשלושת הכתרים שנשארו, כְּמוֹ שֶׁצָּרִיךְ לְגַלּוֹת, אִם נוֹחַ לפני עַתִּיק הסתום מֵהַכֹּל.


ושנינו, שלא סתומה זֹאת הָעַיִן, והן שתיים, ומתלבנות וחוזרות לאחת. הַכֹּל הוּא יָמִין - אֵין בו שְׂמֹאל. לא נם וְלֹא ישן, וְלֹא צָרִיךְ שמירה. אֵין מי שמגן עָלָיו, הוּא מגן עַל הַכֹּל, הוא משגיח על הכל, ומההשגחה של הָעַיִן הזו ניזונים כולם.


שנינו, אִם הָעַיִן הזאת נסתמת רֶגַע אֶחָד, לא יְכוֹלִים להתקיים כולם, מִשּׁוּם כך נקראת 'עַיִן פקוחה', 'עַיִן עֶלְיוֹנָה', 'עַיִן קְדוֹשָׁה', 'עַיִן ההשגחה', 'עַיִן שלא {130a} ישנה וְלֹא נמה', 'עַיִן שֶׁהִיא השמירה של הַכֹּל', 'עַיִן שֶׁהִיא קִיּוּם הַכֹּל'. ועל זה כתוב: "טוֹב עַיִן הוּא יְבוֹרָךְ" (משלי כב ט), אַל תִּקְרא 'יְבוֹרָךְ', אֶלָּא 'יְבָרֵךְ'. שֶׁזֶּה נקרא "טוֹב עַיִן", וּמִמֶּנּוּ מְבָרֵךְ לכל.


ושנינו, [שֶׁ]אֵין הארה לעין הַתַּחְתּוֹנָה, להתרחץ מאדמימות ומשחורות, חוּץ מכשרואה מִזֶּה האור הַלָּבָן של הָעַיִן הָעֶלְיוֹנָה, שנקראת "טוֹב עַיִן". וְאֵין שיודע כשמאירה הָעַיִן הָעֶלְיוֹנָה הַקְּדוֹשָׁה הזאת, ורוחצת לעין הַתַּחְתּוֹנָה הזאת, חוּץ מִמֶּנּוּ.


ועתידים צַדִּיקִים, זַכָּאִים עֶלְיוֹנִים, לִרְאוֹת זֹאת ברוח הַחָכְמָה, זֶהוּ שכתוב: "כִּי עַיִן בְּעַיִן יִרְאוּ" (ישעיה נב ח). מָתַי? "בְּשׁוּב יהוה צִיּוֹן". וכתוב: "אֲשֶׁר־עַיִן בְּעַיִן נִרְאָה אַתָּה יהוה" (במדבר יד יד). וְאִם לא הָעַיִן הָעֶלְיוֹנָה הטובה, שמשגיחה ורוחצת את הָעַיִן הַתַּחְתּוֹנָה, לא יָכוֹל העולם להתקיים רֶגַע אֶחָד.


שנינו בצנעת הַסֵּפֶר: שההשגחה של הָעַיִן הַתַּחְתּוֹנָה, כאשר משגיח האור העליון בו, ונכנס אוֹתוֹ האור של הָעֶלְיוֹנָה בתחתונה, שממנו מֵאִיר הַכֹּל, זה הוא שכתוב, "אֲשֶׁר־עַיִן בְּעַיִן נִרְאָה אַתָּה יְהֹוָה" (במדבר יד יד).


כתוב: "הִנֵּה עֵין יהוה אֶל־יְרֵאָיו" (תהלים לג יח), וכתוב: "עֵינֵי יהוה הֵמָּה מְשׁוֹטְטִים בְּכָל־הָאָרֶץ" (זכריה ד י). זָכוּ - "עֵין יהוה אֶל־יְרֵאָיו", הָעַיִן שֶׁלְּמַעְלָה. לא זָכוּ - "עֵינֵי יהוה הֵמָּה מְשׁוֹטְטוֹת", הָעַיִן שֶׁלְּמַטָּה.


שלמדנו: מפני מה זָכָה יוֹסֵף, שלא שלטה בו עַיִן הָרָע? מפני שֶׁזָּכָה לִהְיוֹת מושגח בעין טוֹבָה עֶלְיוֹנָה, זֶה הוא שכתוב: "בֵּן פּוֹרָת יוֹסֵף בֵּן פּוֹרָת עֲלֵי־עָיִן" (בראשית מט כב). מַדּוּעַ הוּא "בֵּן פּוֹרָת"? "עֲלֵי־עָיִן", כְּלוֹמַר עַל סִבַּת עַיִן שמשגיחה בו.


וכתוב: "טוֹב־עַיִן הוּא יְבוֹרָךְ" (משלי כב ט), מה הַטַּעַם? "כִּי־נָתַן מִלַּחְמוֹ לַדָּל". מה הַטַּעַם שֶׁנִּקְרָא 'אַחַת'? בוא ראה, בעינו שֶׁלְּמַטָּה, יֵשׁ עַיִן ימנית, ויש עַיִן שמאלית, והן שתיים, בִּשְׁנֵי אופנים. אֲבָל כָּאן, אֵין עַיִן שמאלית. ושתיהן בַּדַּרְגָּה אַחַת עולות, וְהַכֹּל יָמִין. וּמִשּׁוּם כך, עַיִן אַחַת, וְלֹא שתיים.


ולמדנו: העין הזאת - שֶׁהִיא עַיִן ההשגחה - פקוחה תמיד, מחייכת תמיד, וּשְׂמֵחָה תמיד, שֶׁאֵין כך לְמַטָּה, שכלולים באדום ובשחור ובלבן, בשלוש גוונים, ואינה תמיד פקוחה, שֶׁלָּהּ גבות כיסוי עַל הָעַיִן. ועל זה כתוב: "עוּרָה לָמָּה תִישַׁן יְהֹוָה" (תהלים מד כד), "פְּקַח יְהֹוָה עֵינֶיךָ" (מלכים־ב יט טז).


כשמתפקח - יֵשׁ מי שמתפקח לטוב, וּמִי שלא מתפקח לטוב. אוֹי לְמִי שמתפקח והעין מתערבצ באדום, ואדום נקרא לנגדו, וּמְכַסֶּה עַל הָעַיִן. מי יינצל מִמֶּנּוּ?


אֲבָל עַתִּיק יָמִים, טוב הָעַיִן, לָבָן בתוך לָבָן, לָבָן שכולל כל הלבנים. אַשְׁרֵי חֶלְקוֹ, לְמִי שֶׁיַּשְׁגִּיחַ עָלָיו, לָבָן אֶחָד מֵהֶם. ועל זה וַדַּאי כתוב: "טוֹב עַיִן הוּא יְבוֹרָךְ". וכתוב: "בֵּית יַעֲקֹב לְכוּ וְנֵלְכָה בְּאוֹר יְהֹוָה" (ישעיה ב ה).


שנינו: שמו של הָעַתִּיק סתום מֵהַכֹּל, וְלֹא מפורש בתורה, חוּץ ממקום אֶחָד, שֶׁנִּשְׁבַּע קצר־פנים לאברהם, שכתוב: "בִּי נִשְׁבַּעְתִּי נְאֻם יהוה" (בראשית כב טז), 'נאום' של זְעֵיר אַנְפִּין, וכתוב: "בְּךָ יְבָרֵךְ יִשְׂרָאֵל" (בראשית מח כ), ישראל שֶׁלְּמַעְלָה, וכתוב: "יִשְׂרָאֵל אֲשֶׁר בְּךָ אֶתְפָּאָר" (ישעיה מט ג), לְיִשְׂרָאֵל אמר זֹאת. ושנינו, עתיק יָמִים אמרו, וְזֶה וְזֶה יָפֶה.


שנינו, [ש]כתוב: "חָזֵה הֲוִית עַד דִּי כָּרְסָוָון רְמִיו, וְעַתִּיק יוֹמִין יְתִיב (רוֹאָה הָיִיתִי עַד שכסאות הושלכו, וְעַתִּיק יָמִים יוֹשֵׁב)" (דניאל ז ט). "כָּרְסָוָון רְמִיו (כסאות מושלכים)", מי הוּא? אמר לרבי יְהוּדָה קוּם בקיומך והתקן כיסא זה!


אמר רבי יְהוּדָה:

- כתוב: "כּוֹרְסְיֵיהּ שְׁבִיבִין דִּי נוּר (כיסאו שביבים של אֵשׁ)" (דניאל ז ט), וְעַתִּיק־יָמִים יוֹשֵׁב עַל הכיסא הזה. מה הַטַּעַם? ששנינו, אִם עַתִּיק־יָמִים לא יוֹשֵׁב עַל הכיסא הזה, לא יָכוֹל להתקיים העולם, מִלִּפְנֵי אוֹתוֹ הכיסא. כאשר יוֹשֵׁב עַתִּיק־יָמִים עָלָיו, לכפות לְאוֹתוֹ הכיסא, וּמִי שרוכב שׁוֹלֵט.


בִּזְמַן שֶׁנּוֹטֵל מהכיסא הזה, וְיוֹשֵׁב עַל כיסא אחר, הכיסא הָרִאשׁוֹן מושלך, שלא [יהיה] שׁוֹלֵט אֶלָּא הוּא שרוכב בו עַתִּיק יָמִים. אמר רבי שִׁמְעוֹן לרבי יְהוּדָה: יתתקן דרכך, ויבוא בך מעתיק ימים.
