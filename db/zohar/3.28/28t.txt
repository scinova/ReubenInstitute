מִצְחָא דִּזְעֵיר אַנְפִּין


מֶצַח הגולגולת הוּא ההשגחה של ההשגחה, וְלֹא מתגלה, חוּץ מֵאוֹתוֹ הַזְּמַן, שבו צריכים חוטאים להתפקד, ולעיין במעשיהם. ולמדנו, שכשמתגלה זה המצח, מִתְעוֹרְרִים כל בעליהם של הדין, וְכָל העולם בדין נִּמְסַר, חוּץ מבאותה השעה, כשעלו תפילותיהם של ישראל לפני עַתִּיק יָמִים, וְרוֹצֶה לְרַחֵם עַל בָנָיו, מגלה המצח של רָצוֹן הָרְצוֹנוֹת, וּמֵאִיר לזה של זְעֵיר אַנְפִּין, ונשקט הדין.


בזה המצח, יוצאת שערה אַחַת, שמתפשטת בו מהמוח שֶׁהוֹצִיא חמישים שערים. וכשמתפשט, נעשה המצח של ההשגחה, לחוטאי העולם, לאלו שלא מתביישים במעשיהם, זֶהוּ שכתוב, "וּמֵצַח אִשָּׁה זוֹנָה הָיָה לָךְ מֵאַנְתְּ הִכָּלֵם" (ירמיה ג ג).


ולמדנו, שהשיער לא עוֹמֵד בזה הַמָּקוֹם של המצח, מִשּׁוּם שמתגלה לאלו שחצופים בַּחֲטָאֵיהֶם. ובשעה שמתעורר הקדוש בָּרוּךְ הוּא לְהִשְׁתַּעֲשֵׁעַ עִם הַצַּדִּיקִים, מְאִירִים פָּנָיו של עַתִּיק יָמִים, בפניו של זְעֵיר אַנְפִּין, ומתגלה מצחו, וּמֵאִיר לזה המצח, וְאָז נקרא 'עֵת רָצוֹן'. וְכָל שָׁעָה ושעה שֶׁהַדִּין תָּלוּי, וְזֶה המצח של זְעֵיר אַנְפִּין מתגלה, מתגלה המצח של עַתִּיק העתיקים, ונשקט הדין, וְלֹא נעשה.


לָמַדְנוּ, שֶׁזֶּה המצח, מתפשט במאתיים אֶלֶף אדומי אדומים, שנאחזים בו, וכלולים בו. וכש מתגלה המצח של זְעֵיר אַנְפִּין, יֵשׁ רְשׁוּת לכולם לחרב, וכשמתגלה המצח של רָצוֹן הָרְצוֹנוֹת, שֶׁמֵּאִיר לזה המצח, אָז כולם נשקטים.


ולמדנו, שעשרים וְאַרְבָּעָה בִּתִּי דינים נמצאים בזה המצח, וכולם נִקְרָאִים 'נֶצַח'. וּבָאוֹתִיּוֹת רצופות, זה 'מֶצַח'. ויש 'נֶצַח' ויש 'מֶצַח', שֶׁהֵם 'נְצָחִים', וְהַיְינוּ שלמדנו, 'נֶצַח נְצָחִים'. וְהֵם במצח, ומתפשטים מֵהֶם בַּגּוּף, במקומות ידועים.


לָמַדְנוּ, מה שכתוב, "וְגַם נֵצַח יִשְׂרָאֵל לֹא יְשַׁקֵּר וְלֹא יִנָּחֵם כִּי לֹא אָדָם הוּא לְהִנָּחֵם" (שמואל א טו כט). את זה הַסּוֹד ביארנו, כל אוֹתוֹ הַ'נֶצַח' שהתפשט בַּגּוּף, לפעמים שתולה עַל העולם לדון, ושב ומתחרט וְלֹא עושה דין, אִם שָׁבִים. מה הַטַּעַם? מִשּׁוּם שֶׁעוֹמֵד במקום שֶׁנִּקְרָא 'אָדָם', וְיָכוֹל להתחרט. אֲבָל אִם במקום שֶׁנִּקְרָא 'רֹאשׁ', נראה ומתגלה זה הַ'נֶצַח', אֵין זה הַזְּמַן והמקום להתחרט. מה הַטַּעַם? מִשּׁוּם שלא הָיָה ממקום שֶׁנִּקְרָא 'אָדָם', שֶׁהֲרֵי לא מתגלה הפרצוף והחוטם, אֶלָּא המצח לְבַדּוֹ. ובמקום שלא נִמְצָא הפרצוף, לא נקרא 'אָדָם'. וּמִשּׁוּם כך, "כִּי לֹא אָדָם הוּא לְהִנָּחֵם", כְּנֶצַח שבשאר תיקוני הגוף
