אוזני קָטָן פָּנִים


לָמַדְנוּ, שכתוב, "הַטֵּה אֱלֹהַי אָזְנְךָ וּשֲׁמַע" (דניאל ט יח), זֹאת האוזן שנעשית תַּחַת השערות, ושערות תלויות עָלֶיהָ. וזאת האוזן נעשית ברישומי רְשׁוּמוֹת לפנימיות, כְּמוֹ שעשה דרגה בעקומה. מה הַטַּעַם בעקומה? מִשּׁוּם שיתעכב הקול לְהִכָּנֵס במוח, ויבחין בו המוח, וְלֹא בבהילות, כְּדֵי לשמוע טוב וָרָע. ולמדנו, שמזאת העקומה שבתוך הָאָוזְנַיִים, תלויים כל אוֹתָם בַּעֲלֵי הכנפיים, שכתוב בהם, "כִּי עוֹף הַשָּׁמַיִם יוֹלִיךְ אֶת־הַקּוֹל וּבַעַל כְּנָפַיִם יַגֵּיד דָּבָר" (קהלת י ב).


בתוך האוזן, נוטף משלושת חללי המוח, לזה נֶקֶב הָאָוזְנַיִים. וּמִתּוֹךְ אוֹתוֹ הנטיף, נכנס קוֹל בְּאוֹתָהּ העקומה, ומצתרף בְּאוֹתוֹ הנטיף, בֵּין טוב וּבֵין רע. טוב - שכתוב, "כִּי־שֹׁמֵעַ אֶל־אֶבְיוֹנִים יְהֹוָה" (תהלים סט לד). רע - שכתוב, "וַיִּשְׁמַע יְהֹוָה וַיִּחַר אַפּוֹ וַתִּבְעַר־בָּם אֵשׁ יְהֹוָה" (במדבר יא א).


וְזֶה האוזן סתום לבחוץ, והעקומה נִכְנֶסֶת לְתוֹךְ, לְאוֹתוֹ הַנֶּקֶב של הנטיף מהמוח, בשביל לאסוף את הקול פנימה, שלא יָצָא הַחוּצָה, וְיִהְיֶה נִסְתָּר וְסָתוּם מכל צדדיו. מִשּׁוּם כך הוּא סוֹד. אוֹי להוא שמגלה סוֹדוֹת! דמי שמגלה סוֹדוֹת - כאילו מכחיש את הַתִּיקּוּן שֶׁלְּמַעְלָה, שֶׁמִּתְתַּקֵּן כְּדֵי לְהַכְנִיס סוֹדוֹת, ושלא יצאו הַחוּצָה.


לָמַדְנוּ, שֶׁבְּשָׁעָה שצועקים ישראל במועקה, ושערות מתגלות מעל הָאָוזְנַיִים, אָז נכנס הקול באוזניים, בְּאוֹתוֹ הַנֶּקֶב שנוטף מהמוח, ונאסף במוח, וְיוֹצֵא בנקבי הַחוֹטֶם. ומתקטן הַחוֹטֶם, ומתחמם, וְיוֹצֵא אֵשׁ ועשן מֵאוֹתָם הנקבים, ומתעוררות כל הגבורות, וְעוֹשֶׂה נקמות.


ובטרם יוצאים מֵאוֹתָם נקבים אֵשׁ ועשן, עולה אוֹתוֹ הקול לְמַעְלָה, וּמַכָּה בָּרֹאשׁ המוח, ונוזלות שְׁתֵּי דְּמָעוֹת מהעניים, וְיוֹצֵא מנחיריו עשן ואש, בקול הַהוּא שזורם לָהֶם בחוץ.


בהוא הקול שנכנס באוזניים, נמשכים ומתעוררים כל אֵלּוּ, מִשּׁוּם כך כתוב, "וַיִּשְׁמַע יְהֹוָה וַיִּחַר אַפּוֹ וַתִּבְעַר־בָּם אֵשׁ יְהֹוָה". בהיא השמיעה של הַהוּא קוֹל, מִתְעוֹרֵר המוח. לָמַדְנוּ, שכתוב, "הַטֵּה אֱלֹהַי אָזְנְךָ", כְּלוֹמַר מרכין אוזניים. שֵׁשׁ מאות אֶלֶף רבבות הֵן בעליהם של כנפיים, שתלויים בְּאֵלּוּ הָאָוזְנַיִים. וְהַכֹּל נִקְרָאִים 'אָזְנֵי יְהֹוָה'. ומה שנאמר "הַטֵּה יְהֹוָה אָזְנְךָ", "אָזְנְךָ" - עַל קָטָן פָּנִים נֶאֱמַר.


מהצד של חלל אֶחָד של המוח תלויות אוזניים. ומחמישים שערים שֶׁיּוֹצְאִים מֵהַהוּא חלל, יֵשׁ שַׁעַר אֶחָד, שֶׁשּׁוֹפֵעַ וְיוֹצֵא ומתפתח בההוא נֶקֶב של האוזן. וּמִצַּד ההתפשטות של הַהוּא הֶחָלָל של חמישים שערים, מתפשט בַּגּוּף במקום שהלב שורה, ועל זה באוזן קוֹרֵא בו 'בְּחִינָה', ובלב קוֹרֵא בו 'בְּחִינָה', שכתוב, "כִּי־אֹזֶן מִלִּין תִּבְחָן" (איוב לד ג), וכתוב, "וּבֹחֵן לִבּוֹת וּכְלָיוֹת" (תהלים ז י), מִשּׁוּם שממקום אֶחָד מתפשטים.


לָמַדְנוּ בִּצְנִיעוּת הַסֵּפֶר, גמו שהאוזן הזאת מבחינה בֵּין טוב וּבֵין רע, כך הַכֹּל. שבזעיר אַנְפִּין יֵשׁ צַד של טוב וָרָע, יָמִין ושמאל, רַחֲמִים ודין. וזאת האוזן כלולה במוח. וּמִשּׁוּם שנכללת במוח ובחלל אֶחָד, נִכְלֶלֶת בקול שנכנס בה, ובאוזן נקראת בה 'שְׁמִיעָה'. וּבִ'שְׁמִיעָה' אִתְכְּלִיל 'בִּינָה'. "שְׁמַע" - כְּלוֹמַר, 'הָבֵן', נִמְצָא שהכל באחד משקל נשקל. ודברים אֵלּוּ לבעליהם של הבעלים ניתן, לשמוע, ולהסתכל, וְלָדַעַת.


בוא ראה, כתוב, "יְהֹוָה שָׁמַעְתִּי שִׁמְעֲךָ יָרֵאתִי" (חבקוק ג ב). זה הפסוק נודע, שכשהנביא הקדוש שמע, והסתכל, וידע, וְעָמַד עַל תִּיקּוּנִים אֵלּוּ, כתוב "יָרֵאתִי", שם נאה הוּא לפחד ולהישבר לְפָנָיו, זה בזעיר אַנְפִּין נֶאֱמַר.


כשהסתכל וידע, מה כתוב "יְהֹוָה פָּעָלְךָ בְּקֶרֶב שָׁנִים חַיֵּיהוּ" - זה לְעַתִּיק יָמִים נֶאֱמַר. וּבְכָל מקום שנמצא "יְהֹוָה ׀ יְהֹוָה", בְּיוֹ״ד הֵ״א שְׁתֵּי פעמים, אוֹ בְּאָלֶ״ף דָּלֶ״ת וְיוּ״ד הֵ״א - אֶחָד לִזְעֵיר אַנְפִּין, וְאֶחָד לְעַתִּיק העתיקים. וְאַף עַל פִּי שכולם אֶחָד, ושם אֶחָד נקראו.


ולמדנו, מָתַי נקרא שם מלא? בִּזְמַן שכתוב "יְהֹוָה אֱלֹהִים". שֶׁזֶּה הוּא שם מלא, שעתיק מֵהַכֹּל, ומזעיר אַנְפִּין. וְהַכֹּל הוּא שם מלא נקרא. וּשְׁאָר לא נקרא שם מלא, כְּמוֹ שביארנו, "וַיִּטַּע יְהֹוָה אֱלֹהִים" (בראשית ב ח), שֵׁם מָלֵּא בִּנְטִיעוֹת הגן. וּבְכָל מקום "יְהֹוָה אֱלֹהִים" נקרא שם מלא? "יְהֹוָה ׀ יְהֹוָה", הַכֹּל הוּא בִּכְלָל. וההוא זְמַן מִתְעוֹרְרִים רַחֲמִים בכל.


"יְהֹוָה פָּעָלְךָ בְּקֶרֶב שָׁנִים חַיֵּיהוּ" - לְעַתִּיק יוֹמִין נֶאֱמַר. "מַאן פָּעָלְךָ" - זְעֵיר אַנְפִּין. "בְּקֶּרֶב שָׁנִים"- הֵן 'שָׁנִים קַדְמוֹנִיּוֹת', שנקראות 'יְמֵי קֶדֶם', וְלֹא נקראות 'שְׁנוֹת עוֹלָם'. "שָׁנִים קַדְמוֹנִיּוֹת" הֵן 'יְמֵי קֶדֶם'. 'שְׁנוֹת עוֹלָם' אֵלֶּה הֵן 'יְמֵי עוֹלָם'. וְכָאן, "בְּקֶרֶב שָׁנִים" - מֵהֶן "שָׁנִים"? 'שָׁנִים קַדְמוֹנִיּוֹת'. "חַיֵּיהוּ" לְמִי? "חַיֵּיהוּ" לִזְעֵיר אַנְפִּין. שכל הארה שֶׁלּוֹ מֵאוֹתָן "שָׁנִים קַדְמוֹנִיּות" התקיימו, וּמִשּׁוּם כך אמר "חַיֵּיהוּ". "בְּרוֹגֶז רַחֵם תִּזְכּור" - להוא חֶסֶד עֶלְיוֹן של עַתִּיק העתיקים, שבו מִתְעוֹרֵר רַחֲמִים לכל, לְמִי שֶׁרוֹצֶה לְרַחֵם, וּלְמִי שראוי לְרַחֵם.
