תִּקּוּנֵי דִּזְעֵיר אַפִּין


הַשְׁתָּא אִתְתָּקָנוּ, וְאִתְכָּוְונוּ דַּעְתָּא, לְמִתְקָן תִּקּוּנוֹי דִּזְעֵיר אַנְפִּין, הֵיךְ יִתְתָּקַּן, וְהֵיךְ יִתְלַבָּשׁ בְּתִקּוּנוֹי מִתִקּוּנֵי עַתִּיק יוֹמִין, קַדִּישָׁא דְּקַדִּישִין, טְמִירָא דִּטְמִירִין, טְמִירָא מִכֹּלָּא. דְּהַשְׁתָּא חוֹבָתָא עָלַיְיכוּ, לְמִגְזַר דִּינָא קוּשְטָאָה, יָאֶה וְשַׁפִּירָא, וּלְאַתְקְנָא כָּל תִּקּוּנִין עַל בּוּרְיֵיהּ.


תִּקּוּנֵי דִּזְעֵיר אַנְפִּין, מִתִּקוּנֵי דַּאֲרִיךְ אַנְפִּין אִתְתָּקָּנוּ. וְאִתְפָּשָׁטוּ תִּקּוּנוֹי מִכָּאן וּמִכָּאן, כְּחֵיזוּ בַּר נָשׁ, לְמִשְלְטָא וּמְשָׁלְפַא בֵּיהּ רוּחָא דִּטְמִירָא דְּכָל טְמִירִין. בְּגִין לְמֵיתַב עַל כּוּרְסְיָּיא, דִּכְתִּיב, "וְעַל דְּמוּת הַכִּסֵּא דְּמוּת כְּמַרְאֵה אָדָם עָלָיו מִלְמָעְלָה" (יחזקאל א כו). "כְּמַרְאֵה אָדָם" - דְּכָלִיל כָּל דִּיוּקְנִין; "כְּמַרְאֵה אָדָם" - דְּכָלִיל כָּל שְׁמָהָן; "כְּמַרְאֵה אָדָם" : דְּבֵיהּ סְתִימִין כָּל עָלְמִין, עִלָּאִין וְתַתָּאִין; "כְּמַרְאֵה אָדָם" - דְּכָלִיל כָּל רָזִין דְּאִתְאַמָּרוּ וְאִתָּתְקָנוּ עַד דְּלָא אִבְרֵי עָלְמָא, וְאַף עַל גַּב דְּלָא אִתְקָיָּימוּ.


תָּאנָא בִּצְנִיעוּתָא דְּסִפְרָא, עַתִּיקָא דְּעַתִּיקִין עַד לָא זַמִּין תִּקּוּנוֹי, בָּאנֵי מַלְכִין, גָּלִיף מַלְכִין, וּמְשַׁעֵר מַלְכִין, וְלָא הֲווֹ מִתְקַיְּימֵּי, עַד דְּדָחֵי לוֹן, וְאַצְנַע לוֹן לְבָתַר זִמְנָא, הֲדָא הוּא דִכְתִיב, "וְאֵלֶּה הַמְּלָכִים אֲשֶׁר מָלְכוּ בְּאֶרֶץ אֱדוֹם" (בראשית לו לא). "בְּאֶרֶץ אֱדוֹם" - בְּאֲתָר דְּכָל דִּינִין מִתְקַיְּימִין תַּמָּן, וְכֻלְּהוּ לָא אִתְקָיָּימוּ עַד דְּרֵישָׁא חִוָּורָא, עַתִּיקָא דְּעַתִּיקִין, אִתְתָּקַּן. כַּד אִתְתָּקַּן - תַּקִּין כָּל תִּקּוּנִין דִּלְתַתָּא, תַּקִּין כָּל תִּקּוּנִין דְּעִלָּאִין וְתַתָּאִין. מִכָּאן אוֹלִיפְנָא, כָּל רֵישָׁא דְּעַמָּא, דְּלָא אִתְתָּקַּן הוּא בְּקַדְמִיתָא - לֵית עַמָּא מִתְתַּקְנָא. וְאִי אִֽיהוּ מִתְתַּקָּן - כֻּלְּהוּ מִתְתַּקְּנָן. וְאִי אִֽיהוּ לָא מִתְתַּקֵּן בְּקַדְמִיתָא - לָא יַכְלִין עַמָּא לְאִתְתַּקְּנָא.


מְנָלָן? מֵעַתִּיק יוֹמִין. דְּעַד לָא אִתְתָּקַּן הוּא בְּתִקּוּנוֹי, לָא אִתְתְּקָּנוּ כָּל אִינּוּן דְּבָעוּ לְאִתְתַּקְּנָא, וְכֻלְּהוּ עָלְמִין אִתְחֲרָבוּ. הֲדָא הוּא דִכְתִיב, "וַיִּמְלוֹךְ בֶּאֱדוֹם בֶּלַע בֶּן־בְּעוֹר" (בראשית לו לב). "וַיִּמְלוֹךְ בֶּאֱדוֹם" - רָזָא יַקִירָא, הוּא אֲתָר דְּכָל דִּינִין מִתְקַטְּרִין תַּמָּן, וְתַּלְיָין מִתַּמָּן.


"בֶּלַע בֶּן־בְּעוֹר" - תָּאנָא, הוּא גִּזְרַת דִּינָא, תַּקִּיפָא דְתַקִּיפִין, דִּבְגִינֵיהּ מִתְקַטְּרָן אֶלֶף אַלְפִין מָארֵי דִּיבָבָא וִילָלָה. "וְשֵׁם עִירוֹ דִּנְהָבָה" (בראשית לו לב), מַאי "דִּנְהָבָה"? כְּלוֹמַר, דִּין הָבָה, כְּמָה דְאַתְּ אָמֵר, "לַעֲלוּקָה שְׁתֵּי בָּנוֹת הַב הַב" (משלי ל טו).


כֵּיוָן דְּסָלִיק לְאִתְיַשְּׁבָא בֵּיהּ, לָא קָאִים, וְלָא הֲוָה יָכִיל לְמֵיקָם, וְכֻלְּהוּ עָלְמִין אִתְחֲרָבוּ. מַאי טַעֲמָא? מִשּׁוּם דְּאָדָם לָא אִתְתָּקַּן, דְּתִקּוּנָא דְּאָדָם בְּדִיוּקְנֵיהּ, כָּלִיל כֹּלָּא, וְיָכִיל [דף קלה ב] כֹּלָּא לְאִתְיַשְּׁבָא בֵּיהּ.


וּבְגִין דְּתִקּוּנָא דָּא דְּאָדָם, לָא אִשְׁתְּכַח, לָא יָכִילוּ לְמֵיקָם וּלְאִתְיַשְּׁבָא, וְאִתְבְּטָלוּ. וְאִתְבְּטָלוּ, סַלְּקָא דַּעְתָּךְ, וְהָא כֻּלְּהוּ בְּאָדָם אִתְכְּלִילָן? אֶלָּא, אִתְבְּטָלוּ וְאִסְתַּלָּקוּ מֵהַהוּא תִּקּוּנָא, עַד דְּיֵיתֵי דִּיּוּקְנָא דְּאָדָם. וְכַד אָתָא הַאי דִּיּוּקְנָא, אִתְגֵּלָפוּ כֻּלְּהוּ, וְאִתְחֲזָּרוּ לְקִיּוּמָא אַחֲרָא. מִנְּהוֹן אִתְבְּסָמוּ, וּמִנְּהוֹן לָא אִתְבְּסָמוּ כְּלָל.


וְאִי תֵּימָא, וְהָא כְּתִיב "וַיָּמָת". "וַיָּמָת" - דְּאִתְבְּטָלוּ לְגַמְרֵי? לָאו הָכִי. אֶלָּא, כָּל מַאן דְּנָחִית מִדַּרְגָּא קַדְמָאָה דְּהֲוָה בֵּיהּ, קָארֵי בֵּיהּ 'מִיתָה', כְּמָה דְאַתְּ אָמֵר, "וַיָּמָת מֶלֶךְ מִצְרַיִם" (שמות ב כג), דְּנָחַת מִדַּרְגָּא קַדְמָאָה דְּהֲוָה קָם בֵּיהּ. וְכֵיוָן דְּאִתְתָּקַן אָדָם, אִתְקְרוּן בִּשְׁמָהָן אַחֲרָנִין, וְאִתְבְּסָמוּ בְּקִיּוּמָא בֵּיהּ, וְקַיְימִין בְּדוּכְתַּיְיהוּ.


וְכֻלְּהוּ אִתְקְרוּן בִּשְׁמָהָן אַחֲרָנִין מִן קַדְמָאִין, בַּר הַהוּא דִּכְתִּיב בֵּיהּ, "וְשֵׁם אִשְׁתּוֹ מְהֵיטַבְאֵל בַּת־מַטְרֵד בַּת מֵי זָהָב" (בראשית לו לט). מַאי טַעְמָא? מִשּׁוּם דְּהָנֵי לָא אִתְבְּטָלוּ כִּשְׁאַר אַחֲרָנִין, מִשּׁוּם דְּהֲוָה דְּכַר וְנוּקְבָּא, כְּהַאי תְּמָרָא, דְּלָא סַלְּקָא אֶלָּא דְּכַר וְנוּקְבָּא. וּבְגִין כָּךְ, הַשְׁתָּא דְּאִשְׁתְּכָחוּ דְּכַר וְנוּקְבָּא, לָא כְּתִיב בְּהוּ 'מִיתָה' כְּאַחֲרָנִין, וְאִתְקְיָימוּ. אֲבָל לָא אִתְיָשָׁבוּ, עַד דְּאִתְתְּקַן דִּיּוּקְנָא דְּאָדָם, וְכֵיוָן דְּאִתְתְּקַן דִּיּוּקְנָא דְּאָדָם, אִתְחֲזָרוּ וְאִתְקְיָימוּ בְּקִיּוּמָא אַחֲרָא, וְאִתְיְישָבוּ.
