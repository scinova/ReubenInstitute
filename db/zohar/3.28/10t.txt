הזקן של ארוך פָּנִים


פָּתַח רַבִּי שִׁמְעוֹן וְאָמַר:

- אוֹי [לְ]מִי שֶׁמּוֹשִׁיט יָדוֹ בַּזָּקָן הַיָּקָר הָעֶלְיוֹן, שֶׁל הַזָּקֵן הַקָּדוֹשׁ, חָבוּי וְסָתוּם מֵהַכֹּל! זָקָן שֶׁל הַתִּשְׁבַּחַת הַהוּא, זָקָן שֶׁסָּתוּם וְיָקָר מִכֹּל תִּיקּוּנָיו, זָקָן שֶׁלֹּא יוֹדְעִים עֶלְיוֹנִים וְתַחְתּוֹנִים, זָקָן שֶׁהוּא תִּתְשַׁבַּחַת כֹּל הַתִּשְׁבָּחוֹת, זָקָן שֶׁלֹּא הָיָה בֶּן־אָדָם, נָבִיא, אוֹ קִדּוּשׁ, שֶׁיִּתְקָרֵב לִרְאוֹת אוֹתוֹ, זָקֵן שֶׁהוּא תָּלוּי תָּלוּי בְּשַׂעֲרוֹתָיו עַד טַבּוּר הַלֵּב, לָבָן כַּשֶּׁלֶג, יָקָר הַיְּקָרִים, נִסְתָּר הַנִּסְתָּרִים, אֱמוּנָה שֶׁל אֱמוּנַת הַכֹּל.


לָמַדְנוּ, בִּצְנִיעוּת הַסֵּפֶר, שֶׁהַזֶּה זָקָן _הוּא_ אֱמוּנַת הַכֹּל, יוֹצֵא מֵאָוזְנָיו, וְיוֹרֵד סְבִיבוֹ שֶׁל הַפֶּה הַקָּדוֹשׁ, וְיוֹרֵד וְעוֹלֶה, וּמְחַפֶּה [בְּמִּנְחָה|בְּמִּנְחָה מְרוּבָּה] שֶׁל הַבּוֹשֶׂם הַטּוֹב, הַלָּבָן שֶׁל הַיָּקָר. וְיוֹרֵד בְּשָׁוֶוה, וּמְחַפֶּה עַד הַטַּבּוּר. הוּא הַזָּקָן הַיָּקָר, אֱמוּנָה שְׁלֵמָה, שֶׁזּוֹרְמִים בּוֹ שְׁלוֹשׁ עֶשְׂרֵה נְבִיעוֹת, מַבּוּעִים שֶׁל שֶׁמֶן טוֹב, בִּשְׁלוֹשָׁה עָשָׂר תִּיקּוּנִים מִתַּתְקָן.


הַתִּיקּוּן הָרִאשׁוֹן: מִתַּתְקָן הַשֵּׂיעָר מִלְּמַעְלָה, וּמַתְחִיל מֵהַהוּא תִּיקּוּן שֶׁל שֵׂיעָר רֹאשׁוֹ, שֶׁעוֹלֶה בְּתִיקּוּנָיו לְמַעְלָה מֵאָוזְנָיו, וְיוֹרֵד מִלִּפְנֵי פֶּתַח הָאָוזְנַיִים, בְּחוּט אֶחָד, בְּשִׁיקּוּל טוֹב, עַד רֹאשׁ הַפֶּה.


הַתִּיקּוּן הַשֵּׁנִי: מִתַּקֵּן הַשֵּׂיעָר מֵרֹאשׁ הַפֶּה, וְעוֹלֶה עַד הָרֹאשׁ הָאַחֵר שֶׁל הַפֶּה, בְּתִיקּוּן שָׁקוּל.


הַתִּיקּוּן הַשְּׁלִישִׁי: מהאמצע שמתחת לחוטם, מִתַּחַת שני נקבים, יוצאת דֶּרֶךְ אַחַת, והשיער נפסק בְּאוֹתָהּ הַדֶּרֶךְ, ומלא שֵׂיעָר מהצד הזה וּמֵהַצְּד הזה, בְּתִיקּוּן שָׁלֵם, מסביבה של הַדֶּרֶךְ הַהִיא.


הַתִּיקּוּן הָרְבִיעִי: מִתַּתְקָן הַשֵּׂיעָר וְיוֹרֵד תַּחַת הַפֶּה, מֵרֹאשׁ אֶחָד לראש אֶחָד, בְּתִיקּוּן שָׁלֵם.


הַתִּיקּוּן הַחֲמִישִׁי: תַּחַת הַפֶּה יוצאת דֶּרֶךְ אַחֶרֶת, בְּשִׁיקּוּל של הַדֶּרֶךְ הָעֶלְיוֹנָה, וְאֵלֶּה שְׁתֵּי הדרכים רְשׁוּמוֹת עך הַפֶּה, מִכָּאן וּמִכָּאן.


הַתִּיקּוּן השישי: מתתתקן הַשֵּׂיעָר, וְעוֹלֶה וְיוֹצֵא מִלְּמַטָּה לְמַעְלָה לראש הַפֶּה, וּמְכַסֶּה מִנְחַת הַבּוֹשֶׂם הטוב, עַד רֹאשׁ הַפֶּה שֶׁלְּמַעְלָה. וְיוֹרֵד הַשֵּׂיעָר לראש פתח  הַדֶּרֶךְ מִתַּחַת לפה.


הַתִּיקּוּן השביעי: נפסק הַשֵּׂיעָר, ונראים שני תַּפּוּחִים, במנחת הַבּוֹשֶׂם הטוב, יָפִים ונאים למראה, בגללם מתקיים העולם, זֶהוּ שכתוב, "בְּאוֹר פְּנֵי מֶלֶךְ חַיִּים" (משלי טז טו).


הַתִּיקּוּן השמיני: יוצא חוט אֶחָד של שְׂעָרוֹת שֶׁסובבות את הזקו, ותלויות בְּשָׁוֶוה עַד הַטַּבּוּר.


הַתִּיקּוּן התשיעי: מתעוררות ומתערבות שְׂעָרוֹת הזקן, עִם אוֹתָן שְׂעָרוֹת שֶׁתְּלוּיוֹת בְּשָׁוֶוה, וְלֹא יוֹצְאוֹת זו מִזּוֹ.


הַתִּיקּוּן העשירי: יורדות שְׂעָרוֹת תַּחַת הזקן, ומכסות הגרון שֶׁתַּחַת הזקן.


הַתִּיקּוּן האחד עשר: שלא יוצאים נים מנים, ומשוערים בשיעור שָׁלֵם.


הַתִּיקּוּן השניים עשר:  שלא תלויות שְׂעָרוֹת עַל הַפֶּה, והפה מתפנה מכל צדדיו, ונאות שְׂעָרוֹת סביב סביב לוֹ.


הַתִּיקּוּן הַשְּׁלוֹשָׁה־עָשָׂר: שֶׁתְּלוּיוֹת שְׂעָרוֹת מִתַּחַת לַזָּקָן, מִכָּאן וּמִכָּאן, בִּיקָר יָאֶה, בִּיקָר יָפֶה, מְחַפּוֹת עַד הַטַּבּוּר. לֹא נִרְאָה מִכָּל פְּנֵי מִנְחַת הַבּוֹשֶׂם, חוּץ מֵהַהֵם תַּפּוּחִים יָפִים _וְ_לְבָנִים, שֶׁמּוֹצִיאִים חַיִּים לָעוֹלָם, וּמַרְאִים שִׂמְחָה לִזְעֵיר־אַנְפִּין.


בִּשְׁלוֹשָׁה עשר תִּיקּוּנִים אֵלּוּ, נמשכים יוֹרְדִים שלושה עשר מַבּוּעִים של משחה, ונמשכים לכל אוֹתָם שֶׁלְּמַטָּה. וּמְאִירִים בְּאוֹתָהּ המשחה. ונמשחים מאותה המשחה, שבשלושה עשר תִּיקּוּנִים אֵלּוּ נרשם הזקן הַיָּקָר, הסתום מֵהַכֹּל, של עַתִּיק העתיקים. משני תַּפּוּחִים יָפִים של פָּנָיו, מְאִירִים פָּנָיו של זְעֵיר אַנְפִּין, וְכָל כפתור וּפָרַח שנמצאים לְמַטָּה, מְאִירִים ומתלהטים מֵאוֹתוֹ האור שֶׁלְּמַעְלָה.שלושה עשר תִּיקּוּנִים אֵלּוּ, נמצאו בַּזָּקָן, ובשלמות הזקן בְּתִיקּוּנָיו, נִקְרָאִים 'בן אדם נֶאֱמָן', שכל מי שרואה זקנו, תולה בו אמונתו.


לָמַדְנוּ בנסתרות הַסֵּפֶר, שלושה עשר ציקונים אֵלּוּ, שתלוים בַּזָּקָן הַיָּקָר, בשביעי נמצאים בעולם, ונפתחים בִּשְׁלוֹשָׁה עשר שַׁעֲרֵי רַחֲמִים. וּמִי שֶׁמּוֹשִׁיט יָדוֹ להישבע, כְּמִי שֶׁנִּשְׁבַּע בִּשְׁלוֹשָׁה עשר תיקוני הזקן. זֹאת באריף אַנְפִּין, בזעיר אַנְפִּין בְּכַמָּה? אמר לרבי יִצְחָק: קוּם בקיומך, וסלסל בסלסול תִּיקּוּן הַמֶּלֶךְ הקדוש, אֵיךְ יתתקנו!


קם רבי יִצְחָק, פתח וְאָמַר: "מִי־אֵל כָּמוֹךָ נוֹשֵׂא עָוֹן …" (מיכה ז יח), "יָשׁוּב יְרַחֲמֵנוּ …" (מיכה ז יט), "תִּתֵּן אֱמֶת לְיַעֲקֹב …" (מיכה ז כ). לָמַדְנוּ, ששלוש עֶשְׂרֵה מידות נראות כָּאן, וכולן יוֹצְאוֹת משלושה עשר מַבּוּעִים של שֶׁמֶן תיקוני הזקן הקדוש, עַתִּיק העתיקים, נִסְתָּר הַנִּסְתָּרִים. לָמַדְנוּ, שתיקון הזקן נִסְתָּר וְסָתוּם, נִסְתָּר וְלֹא נִסְתָּר, סתום וְלֹא סתום. בְּתִיקּוּנָיו נודע וְלֹא נודע.
