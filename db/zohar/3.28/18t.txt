תִּקּוּנָא תְּמִינָאָה


הַתִּיקּוּן השמיני: יוצא חוט אֶחָד של שְׂעָרוֹת, סְבִיבוֹ של הזקן, ותלויות בְּשָׁוֶוה עַד הַטַּבּוּר. קוּם אֶלְעָזָר בני, התקן תִּיקּוּן זה!


קָם רִבִּי אֶלְעָזָר, פָּתַח וְאָמַר: הַכֹּל תָּלוּי בְּמַזָּל, וַאֲפִילּוּ סֵפֶר תּוֹרָה בָּהֵיכָל. דבר זה ביארנו אוֹתוֹ בספר המוצנע, וְכָאן יֵשׁ לְהִסְתַּכֵּל, וְכִי הַכֹּל תָּלוּי בְּמַזָּל? ולמדנו, סֵפֶר תּוֹרָה קֹדֶשׁ, וְנַרְתֵּקוֹ קֹדֶשׁ, וְהַהֵיכָל קֹדֶשׁ. וכתוב "וְקָרָא זֶה אֶל־זֶה וְאָמַר, קָדשׁ קָדשׁ קָדשׁ" (ישעיה ו ג) - הֲרֵי שלושה הֵם, וְסֵפֶר תּוֹרָה כְּנֶגְדָּם, נרתירו קֹדֶשׁ, וְהַהֵיכָל קֹדֶשׁ, וְהוּא קָדשׁ. וְהַתּוֹרָה נִתְּנָה בְּשָׁלֹשׁ קְדוּשׁוֹת, בְּשָׁלֹשׁ מַעֲלוֹת, בְּיָּמִים שְׁלֹשָׁה. וּשְׁכִינָה בְּשָׁלֹשׁ: לוּחוֹת וַאֲרוֹן וְהֵיכַל. וְהַכֹּל בְּסֵפֶר תּוֹרָה תָּלוּי, והוא תָּלוּי בְּמַזָּל? וכתוב "וּמֵאוֹתוֹת הַשָּׁמַיִם אַל תֵּחָתּוּ" (ירמיה י ב). מי שהוא בִּקְדוּשׁוֹת הַלָּלוּ, יהיה תָּלוּי בְּמַזָּל?


אֶלָּא, כך ביארנו בספר המוצנע, זה החוט, הַיָּקָר הקדוש, שכל שְׂעָרוֹת הזקן תלויות בו, נקרא 'מַזָּל'.מה הַטַּעַם? מִשּׁוּם שכל קודשי הקודשים של הקודשים, בזה המזל תלויים. וספר תורה, אַף אל פִּי שהוא קדוש, לא חלים עָלָיו עֶשְׂרֵה קידושים טֶרֶם נכנס להיכל. כיוון שנכנס להיכל, נקרא 'קָדוֹשׁ בְּעֶשֶׂר קְדוּשׁוֹת'. באופן הזה, שלא נקרא 'הֵיכָל', אֶלָּא כשמתברות עֶשֶׂר קְדוּשׁוֹת. ולמדנו, שֶהַכֹּל תָּלוּי בְּמַזָּל, שהוא אוֹתוֹ החוט הַיָּקָר הקדוש, שכל השערות תלויות בו.


מַדּוּעַ נקרא 'מַזָּל'? מִשּׁוּם שממנו תלויים המזלות, ומתמזלים מִמֶּנּוּ עֶלְיוֹנִים וְתַחְתּוֹנִים. וּמִשּׁוּם כך, הוּא תָּלוּי, וּבוֹ תלויים כל דִּבְרֵי העולם, עֶלְיוֹנִים וְתַחְתּוֹנִים. וַאֲפִילּוּ סֵפֶר תּוֹרָה שֶׁבָּהֵיכָל, שמתעטר עֶשֶׂר קְדוּשׁוֹת, לא יוצא מהכלל עִם שְׁאָר הקדושות, וכולם תלויים בו. וּמִי שרואה את הַתִּיקּוּן הזה, נכבשים עוונותיו מִלְּפָנָיו ונכנעים, זֶהוּ שכתוב, "יִכְבּוֹש עֲוֹנוֹתֵינוּ" (מיכה ז יט). אמר לוֹ רבי שִׁמְעוֹן: בָּרוּךְ בני לקודש הקודשים, עַתִּיק מהכל
