תִּקּוּנָא תְּלִיתָאָה


קוּם רִבִּי חִיָיא! קָם רִבִּי חִיָּיא, פָּתַח וְאָמַר: "וָאוֹמַר אֲהָהּ יְהֹוָה אֱלֹהִים הִנֵּה לָא יָדַעְתִּי דַּבֵּר, כִּי נַעַר אָנֹכִי" (ירמיה א ו). וְכִי יִרְמִיָּה לָא הֲוָה יָדַע לְמַלְּלָא? וְהָא כַּמָה מִלוּלִין נַפְקֵי מִפּוּמוֹי, עַד לָא אָמַר דָּא, וְהוּא אָמַר מִלָּה כְּדִיבָא, דִּכְתִּיב, "הִנֵּה לֹא יָדַעְתִּי דַּבֵּר"? אֶלָּא, חַס וְשָׁלוֹם דְּאִֽיהוּ אָמַר עַל דָּא. אֶלָּא הָכִי תָּאנָא, מַה בֵּין דִּבּוּר לַאֲמִירָה? אֲמִירָה - הוּא דְּלָא בָּעֵי לְאַרְמָא קָלָא; דִּבּוּר - בָּעֵי לְאַרְמָא קָלָא, וּלְאַכְרְזָא מִלִּין.


דִּכְתִּיב, "וַיְדַבֵּר אֱלֹהִים אֵת כָּל־הַדְּבָרִים הָאֵלֶּה לֵאמֹר" (שמות כ א). וְתָאנָא, כָּל עָלְמָא שָׁמְעוּ הַהוּא דִּבּוּר, וְכָל עָלְמָא אִזְדַּעֲזָעוּ. וּבְגִין כַּךְ כְּתִיב "וַיְדַבֵּר", וְלָא כְּתִיב 'וַיֹּאמֶר'. אוּף הָכָא, כְּתִיב "הִנֵּה לֹא יָדַעְתִּי דַּבֵּר", לְאַכְרְזָא מִלָּה, וּלְאוֹכָחָא בְּרוּחַ קֻדְשָׁא לְעָלְמָא.


אִי הָכִי, הָא כְּתִיב "וַיְדַבֵּר יְהֹוָה אֶל־מֹשֶׁה לֵאמֹר" (שמות ו י). אֶלָּא, מַאן הוּא נְבִיאָה עִלָּאָה כְּמֹשֶׁה? דְּלָא זָכָה בַּר נָשׁ כְּוָותֵיהּ, דְּהוּא שָׁמַע דִּבּוּר בְּהַכְרָזָה, וְלָא דָּחִיל, וְלָא אִזְדַּעְזָע. וּשְׁאַר נְבִיאִים אִזְדַּעֲזָעוּ, אֲפִֽילּוּ בַּאֲמִירָה, וְדַחֲלִין בִּדְחִילוּ.


וְתָאנָא, תִּקּוּנָא קַדְמָאָה דְּדִיקְנָא, וְתִנְיָינָא, לְאָתָבָא לִתְלִיתָאָה, דִּכְתִּיב, "הֶן־כָּל־אֵלֶּה יִפְעַל־אֵל פַעֲמַיִם שָׁלֹשׁ עִם־גָּבֶר" (איוב לג כט).


תִּקּוּנָא תְּלִיתָאָה: מֵאֶמְצָעִיתָא דִּתְחוֹת חוֹטָמָא, מִתְּחוֹת תְּרֵין נוּקְבִין, נָפִיק חַד אָרְחָא, וְשַׂעֲרָא אִתְפְּסָק בְּהַהוּא אָרְחָא. אֲמַאי אִתְפְּסָק? מִשּׁוּם דְּהַאי אוֹרְחָא אִתְתָּקַּן לְאַעְבְּרָא בֵּיהּ. וּבְגִין כַּךְ, יְתִיב תְּחוֹת נוּקְבֵי חוֹטָמָא, הַאי אוֹרְחָא. וְשַׂעֲרָא לָא אִתְרַבֵּי בְּהַאי אוֹרְחָא, מִשּׁוּם דִּכְתִּיב "וְעוֹבֵר־עַל פֶּשַׁע" (מיכה ז יח), לְמֵיהַב אַעְבְּרָא עַד פּוּמָא קַדִּישָׁא, דְּיֵימָּא 'סָלַחְתִּי'. תָּאנָא, כַּמָה עַרְקִיסָאוֹת מְחַכָּאן לְהַהוּא פּוּמָא, וְלָא אִתְגְּלֵי לְחָד מִנַּיְיהוּ, דְּהָא אִסְתְּלַּק וְאִתְעַטָּר, יְדִיעַ וְלָא יְדִיעַ.


תָּאנָא, בִּצְנִיעוּתָא דְּסִפְרָא, מַהוּ דִּכְתִּיב "פֶּשַׁע". זָכוּ  - "עוֹבֵר", לֹא זָכוּ  - "פֶּשַׁע". הַאי בִּזְעֵיר אַנְפִּין.


מַאי בֵּין הַאי לְהַאי? בִּזְעֵיר אַנְפִּין, כַּד נָחִית הַהוּא אוֹרְחָא מִתְּחוֹת נוּקְבֵי חוֹטָמוֹי, כְּתִיב, "וַיִּחַר־אַף יְהֹוָה בָּם וַיֵּלַךְ" (במדבר יב ט). מַאי "וַיֵּלַךְ"? דְּנָפִיק רוּחָא דְּרוּגְזָא מֵאִינּוּן נוּקְבֵי, וּמַאן דְּאַשְׁכַּח קַמֵּיהּ, אָזִיל וְלָא אִשְׁתְּכַח, הֲדָא הוּא דִכְתִיב, "כִּי רוּחַ יְהֹוָה נָשְׁבָה בּוֹ" (ישעיה מ ז). בְּאֲרִיךְ אַנְפִּין כְּתִיב, "וְעוֹבֵר־עַל פֶּשַׁע" (מיכה ז יח), וּכְתִיב, "וְרוּחַ עָבְרָה וַתְּטַהֲרֵם" (איוב לז כא). וְתָאנָא, הָכָא כְּתִיב, "עוֹבֵר עַל פֶּשַׁע" - בְּהַהוּא אָרְחָא; הָתָם, "וְעָבַר יְהֹוָה לִנְגּוֹף אֶת־מִצְרַיִם" (שמות יב כב).


זַכָּאָה חוּלָקֵיהּ דְּמַאן דְּזָכֵי לְהַאי. וְדָא הוּא תִּקּוּנָא תְּלִיתָאָה דְּדִיקְנָא יַקִּירָא, קַדִּישָׁא עִלָּאָה, עַתִּיקָא דְּעַתִּיקֵי. אָמַר רַבִּי שִׁמְעוֹן: וַדַּאי קוּדְשָׁא בְּרִיךְ הוּא יַסְגֵּי לְאוֹטָבָא לָךְ, וְיֶחְדֵּי לְאַגָּנָא עֲלָךְ.


וְתָאנָא, מַאי דִּכְתִּיב, "שׂוֹשׂ אָשִׂישׂ בַּיְהֹוָה" (ישעיה סא י)? בְּעַתִּיק יוֹמִין אִתְּמַר, דְּהָא הוּא חֶדְוָותָא דְּכֹלָּא. תָּאנָא, בְּשַׁעֲתָא דְּאִתְגְּלֵי הַאי אוֹרְחָא דְּדִיקְנָא דְּעַתִּיק יוֹמִין, כֻּלְּהוּ מָארֵי דִּיבָבָא וִילָלָה, וּמָארֵיהוֹן דְּדִינָא סְתִימִין וּשְׁתִּיקִין, וְלֵית דְּיִפְתַּח פִּטְרָא לְאַבְאָשָׁא, מִשּׁוּם דְּהַאי אוֹרְחָא אִתְגַלְיָיא לְתַקְּנָא. וּמֵהַאי, מַאן דְּאָחִיד וְאַזְהַר לְשַׁתְקָאָה - לְהַאי אוֹרְחָא רְשִׁים, דְּהוּא סִימָנָא דְּעַתִּיקָא קַדִישָׁא.
