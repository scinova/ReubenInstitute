תִּקּוּנָא תְּלִיתָאָה


קוּם רִבִּי חִיָיא! קָם רִבִּי חִיָּיא, פָּתַח וְאָמַר: "וָאוֹמַר אֲהָהּ יְהֹוָה אֱלֹהִים הִנֵּה לָא יָדַעְתִּי דַּבֵּר, כִּי נַעַר אָנֹכִי" (ירמיה א ו). וְכִי ירמיה לא הָיָה יוֹדֵעַ לְדַבֵּר? וַהֲרֵי כַּמָּה דיבורים יצאו מִפִּיו, טֶרֶם אמר זֹאת, והוא אמר דבר כזב, שכתוב, "הִנֵּה לֹא יָדַעְתִּי דַּבֵּר"? אֶלָּא, חס ושלום שהוא אמר עַל זה. אֶלָּא כך לָמַדְנוּ, מַה בֵּין דִּבּוּר לַאֲמִירָה? אֲמִירָה - הוּא שלא רוֹצֶה להארים את הקול; דיבור - רוֹצֶה להרים את הקול, ולהכריז דְּבָרִים.


שכתוב, "וַיְדַבֵּר אֱלֹהִים אֵת כָּל־הַדְּבָרִים הָאֵלֶּה לֵאמֹר" (שמות כ א). ולמדנו, כל העולם שָׁמְעוּ את אוֹתוֹ הדיבור, וְכָל העולם הִזְדַּעְזְעוּ. וּמִשּׁוּם כך, כתוב "וַיְדַבֵּר", וְלֹא כתוב 'וַיֹּאמֶר'. גַּם כָּאן, כתוב "הִנֵּה לֹא יָדַעְתִּי דַּבֵּר", להכריז דבר, להוכיח ברוח קְדוֹשָׁה לעולם.


אִם כך, הֲרֵי כתוב "וַיְדַבֵּר יְהֹוָה אֶל־מֹשֶׁה לֵאמֹר" (שמות ו י). אֶלָּא, מי הוּא נָבִיא עֶלְיוֹן כמשה? שלא זָכָה אַף בן אדם כָּמוֹהוּ, שהוא שמע דיבור בהכרזה, וְלֹא פחד, וְלֹא הִזְדַּעֲזַע. וּשְׁאָר נביאים הִזְדַּעְזְעוּ, אֲפִילּוּ באמירה, ופחדו בפחד.


ולמדנו, שתיקון הזקן הָרִאשׁוֹן, והשני, הֵם ליישב את הַשְּׁלִישִׁי, שכתוב, "הֶן־כָּל־אֵלֶּה יִפְעַל־אֵל פַעֲמַיִם שָׁלֹשׁ עִם־גָּבֶר" (איוב לג כט).


הַתִּיקּוּן הַשְּׁלִישִׁי: מאמצע שמתחת לחוטם, מִתַּחַת שני נקבים, יוצאת דֶּרֶךְ אַחַת, והשיער נפסק בְּאוֹתָהּ הַדֶּרֶךְ. מַדּוּעַ נפסק? מִשּׁוּם שזאת הַדֶּרֶךְ נתתקנה לְהַעֲבִיר בה. וּמִשּׁוּם כך, יוֹשֶׁבֶת תַּחַת נקבי הַחוֹטֶם, זֹאת הַדֶּרֶךְ. הַשֵּׂיעָר לא מתרבה בזאת הדר, מִשּׁוּם שכתוב "וְעוֹבֵר עַל פֶּשַׁע" (מיכה ז יח), לָתֵת מעבר עַד הַפֶּה הקדוש, שיאמר 'סָלַחְתִּי'. לָמַדְנוּ, כַּמָּה ערכאות מְחַכּוֹת לְאוֹתוֹ הַפֶּה, וְלֹא מתגלה לאף אֶחָד מֵהֶם, שֶׁהֲרֵי הוּא מתעלה ומתעטר, יָדוּעַ וְלֹא יָדוּעַ.


לָמַדְנוּ, בנסתרות הַסֵּפֶר, מהו שכתוב "פֶּשַׁע". זָכוּ  - "עוֹבֵר", לֹא זָכוּ  - "פֶּשַׁע". זֹאת בזעיר אַנְפִּין.


מה בֵּין זה לזה? בזעיר אַנְפִּין, כשיורדת אותה הַדֶּרֶךְ מִתַּחַת נקבי חוטמו, כתוב, "וַיִּחַר־אַף יְהֹוָה בָּם וַיֵּלַךְ" (במדבר יב ט). מהו "וַיֵּלַךְ"? שֶׁיָּצָא ירוח הרוגז מֵאוֹתָם נקבים, וּמִי שנמצא לְפָנָיו, הוֹלֵךְ וְלֹא נִמְצָא,זֶהוּ שכתוב, "כִּי רוּחַ יְהֹוָה נָשְׁבָה בּוֹ" (ישעיה מ ז). באריך אַנְפִּין כתוב, "וְעוֹבֵר־עַל פֶּשַׁע" (מיכה ז יח), וכתוב, "וְרוּחַ עָבְרָה וַתְּטַהֲרֵם" (איוב לז כא). ולמדנו, שֶׁכָּאן כתוב, "עוֹבֵר עַל פֶּשַׁע" - בְּאוֹתָהּ הַדֶּרֶךְ;  שם, "וְעָבַר יְהֹוָה לִנְגּוֹף אֶת־מִצְרַיִם" (שמות יב כב).


אַשְׁרֵי חֶלְקוֹ של מי שֶׁזָּכָה לזה. וזהו הַתִּיקּוּן הַשְּׁלִישִׁי של הזקן הַיָּקָר, הקדוש העליון, עַתִּיק העתיקים. אמר רבי שִׁמְעוֹן: וַדַּאי הקדוש בָּרוּךְ הוּא ירבה להיטיב לך, וישמח להגן עָלַיִךְ.


ולמדנו, מהו שכתוב, "שׂוֹשׂ אָשִׂישׂ בַּיְהֹוָה" (ישעיה סא י)? ברוח עַתִּיק יָמִים נֶאֱמַר, שֶׁהֲרֵי הוּא הַשִׂמְחָה של הַכֹּל. לָמַדְנוּ, שֶׁבְּשָׁעָה שמתגלה זֹאת הַדֶּרֶךְ של הזקן של עַתִּיק יָמִים, כל בַּעֲלֵי היבבות ויללותף וּבַעֲלֵי הדין, סותמים ושותקים, וְאֵין מי שיפתח פה לְהַרע, מִשּׁוּם שזאת הַדֶּרֶךְ מתגלה כְּדֵי לְתַקֵּן. ומזה, מי שאוחז ומזהיר לשתוק - לזאת הַדֶּרֶךְ רָשׁוּם, שֶׁהִיא הסימן של הָעַתִּיק הקדוש
