חוֹטָמָא דְּאֲרִיךְ אַנְפִּין


וְתָא חֲזִי, כְּתִיב, "אֲנִי יְהֹוָה רִאשׁוֹן וְאֶת אַחֲרֹנִים אֲנִי הוּא" (יְשַׁעְיָה מא ד). כֹּלָּא הוּא, וְהוּא סְתִים מִכָּל סִטְרוֹהִי. בְּהַאי חֳטָמָא, תָּאנָא, בְּחֳטָמָא אִשְׁתְּמוֹדַע פַּרְצוּפָא.


וְתָא חֲזֵי, מַה {130b} בֵּין עַתִּיקָא, לִזְעֵיר אַנְפִּין? דָּא מָארֵיהּ דְּחוֹטָמָא, מֵחַד נוּקְבָּא חַיִּין, וּמֵחַד נוּקְבָּא חַיִּין דְּחַיִּין. הַאי חוֹטָמָא, הוּא פַּרְדַּשְׁקָא, דְּבֵיהּ נָשִׁיב רוּחָא דְּחַיֵּי לִזְעֵיר אַנְפִּין, וְקָרֵינָן לֵיהּ 'סְלִיחָה'. וְהוּא נַחַת רוּחַ, אִתְבַּסְּמוּתָא דְּרוּחָא.


דְּרוּחָא דְּנָפֵק מֵאִנּוּן נִקְבֵי, חַד רוּחָא נָפֵק לִזְעֵיר אַנְפִּין, לְאַתְּעָרָא לֵהּ בְּגִנְּתָא דְּעֵדֶן. וְחַד רוּחָא דְּחַיֵּי, דְּבֵהּ זַמִּין לְזַמָּנָא לִבְרֵהּ דְּדָוִד, לְמִנְדַּע חָכְמְתָא. וּמֵהַהוּא נִקְבָּא, אִתְּעַר וְנָפֵק רוּחָא מִמּוֹחָא סְתִימָא, וְזַמִּין לְאַשְׁרָאָה עַל מַלְכָּא מְשִׁיחָא, דִּכְתִּיב, "וְנָחָה עָלָיו רוּחַ יְהֹוָה, רוּחַ חָכְמָה וּבִינָה, רוּחַ עֵצָה וּגְבוּרָה, רוּחַ דַּעַת וְיִרְאַת יְהֹוָה" (יְשַׁעְיָה יא ב). הָא הָכָא אַרְבַּע רוּחִין, וְהָא רוּחָא חֲדָא אָמְרֵינַן. אַמַּאי תְּלָת? קוּם רִבִּי יוֹסֵי בְּקִיּוּמָךְ!


קָם רִבִּי יוֹסֵי וְאָמַר: בְּיוֹמוֹי דְּמַלְכָּא מְשִׁיחָא, לָא יֵימְרוּן חַד לְחַד 'אֲלִיף לִי חָכְמְתָא', דִּכְתִּיב, "וְלֹא יְלַמְּדוּ עוֹד אִישׁ אֶת־רֵעֵהוּ ... כִּי־כֻלָּם יֵדְעוּ אוֹתִי לְמִקְּטַנָּם וְעַד־גְּדוֹלָם" (ירמיה לא לג). וּבְהַהוּא זִמְנָא, יִתְּעַר עַתִּיק יוֹמִי רוּחָא דְּנָפִיק מִמּוֹחָא סְתִימָאָה דְּכֹלָּא, וְכַד יִשְׁלוֹף דָּא, כָּל רוּחִין דִּלְתַּתָּא יִתְּעֲרוּן עִמֵּיהּ. וּמַאן אִינּוּן? אִינּוּן כִּתְרִין קַדִּישִׁין דִּזְעֵיר אַנְפִּין. וְאִינּוּן שִׁיתָא רוּחִין, דְּאִינּוּן תְּלַת רוּחִין דִּכְלִילָן תְּלַת אַחֲרָנִין, דִּכְתִּיב, "רוּחַ חָכְמָה וּבִינָה, רוּחַ עֵצָה וּגְבוּרָה, רוּחַ דַּעַת וְיִרְאַת יְהֹוָה".


דְּתָנֵינַן, כְּתִיב, "וַיֵּשֶׁב שְׁלֹמֹה עַל כִּסֵּא יְהֹוָה" (דִּבְרֵי הַיָּמִים א כט כג), וּכְתִיב, "שֵׁשׁ מַעֲלוֹת לַכִּסֵּא" (מְלָכִים א י יט). וּמַלְכָּא מְשִׁיחָא זַמִּין לְמִתַּב בַּשַּׁבְעָא. שִׁתָּא אִינּוּן, וְרוּחָא דְּעַתִּיק יוֹמִין דַּעֲלַיְהוּ, הָא שַׁבְעָא, כְּמָה דְּאִתְּמַר. אֲמַר לֵהּ רַבִּי שִׁמְעוֹן: רוּחָךְ יָנוּחַ לְעָלְמָא דְאָתֵי.


תָּא חֲזֵי, כְּתִיב, "כֹּה אָמַר יְהֹוָה, מֵאַרְבַּע רוּחוֹת בֹּאִי הָרוּח" (יחזקאל לז ט). וְכִי אַרְבַּע רוּחֵי עָלְמָא, מַאי עַבְדֵּי הָכָא? אֶלָּא, אַרְבַּע רוּחֵי יִתְּעֲרוּן. תְּלָת אִינּוּן, וְרוּחָא דְּעַתִּיקָא סְתִימָא - אַרְבַּע. וְהָכִי הֲווֹ, דְּכַד יִפּוּק דָּא, נָפְקִין עִמֵּיהּ תְּלָתָא דִּכְלִילָן בְּגוֹ תְּלָתָא אַחֲרָנִין.


וְזַמִּין קֻדְשָׁא בְּרִיךְ הוּא לְאַפָּקָא חַד רוּחָא דְּכָלִיל מִכֻּלְּהוּ, דִּכְתִּיב, "מֵאַרְבַּע רוּחוֹת בֹּאִי הָרוּחַ". 'אַרְבַּע רוּחוֹת בֹּאִי' לָא כְּתִיב כָּאן, אֶלָּא "מֵאַרְבַּע רוּחוֹת בֹּאִי". וּבְיוֹמֵי דְּמַלְכָּא מְשִׁיחָא, לָא יִצְטַרְּכוּן לְמֵילַף חַד לְחַד, דְּהָא רוּחָא דִּלְהוֹן, דְּכָלִיל מִכָּל רוּחִין, יְדִיעַ כֹּלָּא: "חָכְמָה וּבִינָה, עֵצָה וּגְבוּרָה, דַּעַת וְיִרְאַת יְהֹוָה". מִשּׁוּם דְּאִיהוּ רוּחָא דִּכְלִילָא מִכָּל רוּחֵי. בְּגִין כָּךְ כְּתִיב, "מֵאַרְבַּע רוּחוֹת", דְּאִינּוּן אַרְבַּע דִּכְלִילַן בְּשַׁבְעָא דַּרְגִין עִלָּאִין דַּאֲמַרַן. וְתָאָנָא, דְּכֻלְּהוּ כְּלִילַן בְּהַאי רוּחָא דְּעַתִּיקָא דְּעַתִּיקִין, דְּנָפֵק מִמּוֹחָא סְתִימָא לַנִּקְבָּא דְּחָטַּמָא.


וְתָּא חֲזֵי, מַה בֵּין חוֹטָמָא לְחוֹטָמָא? חוֹטָמָא דְּעַתִּיק יוֹמִין - חַיִּין מִכָּל סִטְרוֹי. חוֹטָמָא דִּזְעֵיר אַנְפִּין - כְּתִיב, "עָלָה עָשָׁן בְּאַפּוֹ וְאֵשׁ מִפִּיו תֹּאכֵל …" (שמואל ב כב ט). "עָלָה עָשָׁן בְּאַפּוֹ", וּמֵהַהוּא עָשָׁן דָּלִיק נוּר, כַּד סָלִיק תְּנָנָא. לְבָתַר, "גֶּחָלִים בָּעֲרוּ מִמֶּנּוּ" - מַהוּ 'מִמֶּנּוּ'? מֵאוֹתוֹ עָשָׁן, מֵהַהוּא חוֹטָמָא, מֵהַהוּא אֶשָּׁא.


תָּאנָא, כַּד הֲוָה רַב הִמְנוּנָא סָבָא בָּעֵי לְצַלָּאָה צְלוֹתֵהּ, אֲמַר: 'לְבַעַל הַחֹטֶם אֲנִי מִתְפַּלֵּל, לְבַעַל הַחֹטֶם אֲנִי מִתְחַנֵּן'. וְהַיְנוּ דִּכְתִּיב, "וּתְהִלָּתִי אֶחֱטָם לָךְ" (יְשַׁעְיָה מח ט) - הַאי קְרָא לְעַתִּיק יוֹמִין אֲמָרוֹ.


תָּנָא, אוֹרְכָא דְּחוֹטָמָא, תְּלַת מְאָה וְשִׁבְעִין וְחָמֵשׁ עָלְמִין, אִתְמַלְּיָין מִן הַהוּא חוֹטָמָא. וְכֻלְּהוּ מִתְדַּבְּקִן בִּזְעֵיר אַנְפִּין. הַאי תּוּשְׁבַּחְתָּא דְּתִקּוּנָא דְּחוֹטָמָא הוּא. וְכָל תִּקּוּנֵי דְּעַתִּיק יוֹמִין, אִתְחָזוּן וְלָא אִתְחָזוּן - אִתְחָזוּן לְמָארֵי מִדִּין, וְלָא אִתְחָזוּן לְכֹלָּא.
