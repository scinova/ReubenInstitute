תִּקּוּנָא דִּתְלֵיסָר


הַתִּיקּוּן השלושה עשר: תלויות שְׂעָרוֹת שֶׁתַּחַת הזקן, מִכָּאן וּמִכָּאן, ביוקרה נאה, ביוקרה יָפֶה, ומכסות עַד הַטַּבּוּר, וְלֹא נראות מפני מִנְחַת הבשמים, חוּץ מֵאוֹתָם תַּפּוּחִים יָפִים ולבנים.


אמר רבי שִׁמְעוֹן: אַשְׁרֵי חֶלְקוֹ של מי שנמצא בזאת האסיפה הַקְּדוֹשָׁה והעליונה שאנחנו בה. אַשְׁרֵי חֶלְקוֹ בעולם הזה, וּבְעוֹלָם הבא. שֶׁאָנוּ יוֹשְׁבִים בקדושה עֶלְיוֹנָה, אֵשׁ עֶלְיוֹנָה סובבת לָנוּ, וַהֲרֵי כל התיקונים הָעֶלְיוֹנִים של הזקן הקדוש נתתקנו, והתעטרו והסתובבו לִמְקוֹמָם.


וְזֶה הַתִּיקּוּן השלושה עשר, שהוא תִּיקּוּן נאה, שבו נאחזים הַכֹּל. כולם משתוקקים לזקוף את הָרֹאשׁ כנגדו, מִמֶּנּוּ תלויים כל אֵלּוּ שבזעיר אַנְפִּין אוחזים, מִמֶּנּוּ תלויים עֶלְיוֹנִים וְתַחְתּוֹנִים, וְכָל האוצרות הָעֶלְיוֹנִים והתחתונים גְּנוּזִים בו, וּבוֹ כְּלוּלִים. והוא המזל שממנו מתמזלים מִמֶּנּוּ הַכֹּל. זֶהוּ תִּיקּוּן השלם, שֶׁמַּשְׁלִים לכל התיקונים, זה מַשְׁלִים לכל.


לָמַדְנוּ, שֶׁאֵלּוּ התיקונים נִקְרָאִים 'יְמֵי קֶדֶם' - יָמִים קודמים מקודמים. וְאוֹתָם אֵלּוּ שנמצאו בזעיר אַנְפִּין, נִקְרָאִים "יְמֵי עוֹלָם" (ישעיה סג יא). ולמדנו, שֶׁאֵלּוּ 'יְמֵי קֶדֶם', כולם מתתקנים בְּתִיקּוּן הזקן של עַתִּיק העתיקים, נִסְתָּר הַנִּסְתָּרִים. וְזֶה הַתִּיקּוּן השלושה עשר, כּוֹלֵל אוֹתָם, כְּמוֹ שנאמר. וְזֶה הַיּוֹם לא נכלל בו, אֶלָּא הוּא כּוֹלֵל הַכֹּל.


וּבְאוֹתָהּ הפעם שהתעורר עַתִּיק יָמִים בתיקונים שֶׁלְּמַעְלָה, אוֹתוֹ אֶחָד נקרא 'יוֹם אֶחָד', שבו עָתִיד להוקיר את זקנו, זֶהוּ שכתוב, "יוֹם אֶחָד הוּא יִוָּדַע לַיְהֹוָה" (זכריה יָד ז) - הוּא לְבַדּוֹ יוֹתֵר מֵהַכֹּל, הוּא שכולל הַכֹּל, הוּא שֶׁנִּקְרָא בשם הידוע.


שלמדנו, שבאתר שֶׁיֵּשׁ יוֹם, אֵין לַיְלָה, שֶׁאֵין יוֹם לְלֹא לַיְלָה. וּמִשּׁוּם שבאותו הַזְּמַן, זְמַן יהיה של יוקרת הזקן,  והוא לְבַדּוֹ יהיה נִמְצָא. לא ייקרא לא יוֹם וְלֹא לַיְלָה, שאיו יוֹם נקרא, אֶלָּא מהצד שֶׁלָּנוּ, וְאֵין לַיְלָה נקרא, אֶלָּא מהצד שֶׁלָּנוּ. וּמִשּׁוּם שֶׁזֶּה הַתִּיקּוּן כּוֹלֵל את הַכֹּל, לא נודע וְלֹא נראה מִמֶּנּוּ. וּמִמֶּנּוּ זורם שֶׁמֶן המשחה לשלוש עֶשְׂרֵה עברים של מבועות, לכל אוֹתָם שֶׁלְּמַטָּה, שמאירים האותה המשחה.


בִּשְׁלוֹשָׁה עשר תִּיקּוּנִים אֵלּוּ, מִתַּתְקָן הזקן הקדוש והעליון, וְאֵלּוּ תִּיקּוּנִים שבזה הזקן, מתתקנים וְיוֹרְדִים לכמה עיברים. וְלֹא נראים אֵיךְ מתפשטים ואיך יוצאים, מֵהַכֹּל נסתמו, מֵהַכֹּל נסתרו. אֵין מי שידע מקום זה הָעַתִּיק, בהתפשטות שֶׁלָּהֶם כולם כְּלוּלִים, כְּמוֹ שנאמר, נודע וְלֹא נודע, נִסְתָּר וְלֹא נִסְתָּר, עָלָיו נקרא, "אֲנִי יְהֹוָה הוּא שְׁמִי וּכְבוֹדִי לְאַחֵר לֹא אֶתֵּן" (ישעיה מב ח). ווכתוב, "הוּא עָשָׂנוּ וְלֹא אֲנַחְנוּ" (תהלים ק ג). וכתוב,  "וְעַתִּיק יוֹמִין יְתִיב (וְעַתִּיק יָמִים יוֹשֵׁב)" (דניאל ז ט), במקומו יוֹשֵׁב, וְאֵין מי שיודע אוֹתוֹ, יוֹשֵׁב וְלֹא נִמְצָא. וכתוב, "אוֹדְךָ עַל כִּי נוֹרָאוֹת נִפְלֵיתִי …" (תהלים קלט יָד)
