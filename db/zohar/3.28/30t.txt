חוֹטָמָא דִּזְעֵיר אַנְפִּין


הַחוֹטֶם, לָמַדְנוּ בנסתרות הַסֵּפֶר, הַחוֹטֶם של זְעֵיר אַנְפִּין, בחוטם נודע הפרצוף. בזה הַחוֹטֶם התפרש הדבר שכתוב, "עָלָה עָשָׁן בְּאַפּו" (שמואל ב כב ט). "עָלָה עָשָׁן בְּאַפּוֹ" - בזה העשן, נִכְלְלוּ הָאֵשׁ, וגחלי הָאֵשׁ. שֶׁאֵין עשן בְּלִי אֵשׁ, וְלֹא אֵשׁ בְּלִי עשן. וכולם נתעלו ויצאו מחוטמו.


ולמדנו, כשמתחברים שְׁלוֹשׁ אֵלּוּ, שכלולים בזה העשן, שֶׁיּוֹצֵא מהחוטם. מתקמט הַחוֹטֶם, ונושב וְיוֹצֵא עשן שחור ואדום, בִּשְׁנֵי גוונים. וקראנו לוֹ, 'אַף וְחֵימָה וּמַשְׁחִית'. וְאִם תֹּאמַר, 'אַף וְחֵימָה', שכתוב, "כִּי יָגוֹרְתִּי מִפְּנֵי הָאַף וְהַחֵימָה" (דְּבָרִים ט יט), שֶׁהֵם העשן השחור והאדום, 'מַשְׁחִית' מִנַּיִן לָנוּ? שכתוב, "לִפְנֵי שַׁחֵת יְהֹוָה אֶת־סְדֹם וְאֶת־עֲמֹרָה" (בראשית יג י). "שַׁחֵת" הַמַּשְׁחִית, בָּאֵשׁ השורפת את המוקד.


ולמדנו, שחמש גבורות הֵן, בזה הזעיר אַנְפִּין, והתעלו לאלף וְאַרְבַּע מאות גבורות. ומתפשטות בחוטמו, בפה, בזרועותיו, בידיים, באצבעות. וּמִשּׁוּם כך כתוב, "מִי יְמַלֵּל גְּבוּרוֹת יְהֹוָה" (תהלים קו ב). כתוב כָּאן "גְּבוּרוֹת", וכתוב שם "לְךָ יְהֹוָה הַגְּדוּלָה וְהַגְּבוּרָה" (דִּבְרֵי הַיָּמִים א כט יא)? אֶלָּא, כך לָמַדְנוּ, שכשמתחברות כל הגבורות כאחת, נקראת 'גְּבוּרָה' אַחַת.


וְכָל הגבורות, מתחילות לרדת מחוטמו. ומזה תלויות, אֶלֶף וְאַרְבַּע מאות רבבות, לכל אַחַת מֵהֶן. וּבָזֶה העשן שֶׁמּוֹצִיא מחוטמו, תלויות אֶלֶף וְאַרְבַּע מאות של צַד גְּבוּרָה זֹאת. וְכָל הגבורות תלויות מִזֶּה הַחוֹטֶם, שכתוב, "דּוֹר לְדוֹר יְשַׁבַּח מַעֲשֶׂיךָ" (תהלים קמה ד). וכששורה גְּבוּרָה זו, כל הגבורו מתלהטות ושטות, עַד שיורדות לְ"לַהַט הַחֶרֶב הַמִּתְהַפֶּכֶת" (בראשית ג כד).


כתוב, "כִּי־מַשְׁחִתִים אֲנַחְנוּ אֶת־הַמָּקוֹם הַזֶּה" (בראשית יט יג), וכתוב, "לִפְנֵי שַׁחֵת יְהֹוָה אֶת־סְדֹם֙ וְאֶת־עֲמֹרָ֔ה" (בראשית יג י), וכתוב, "וַיְהֹוָה הִמְטִיר עַל־סְדֹם וְעַל־עֲמֹרָה" (בראשית יט כד)? אֶלָּא, כך לָמַדְנוּ, 'לֹא דַּיָּין לָרְשָׁעִים … ', אֶלָּא שהופכים מִדַּת רַחֲמִים לְמִדַּת הַדִּין.


ואיך הוֹפְכִים? וַהֲרֵי כתוב, "אֲנִי יְהֹוָה לֹא שָׁנִיתִי" (מלאכי ג ו)? אֶלָּא, בכל זְמַן שעתיק העתיקים, הָרֹאשׁ הַלָּבָן, רָצוֹן הָרְצוֹנוֹת, מתגלה - רַחֲמִים מרובים נמצאו בכל. ובשעה שלא מתגלה - כל הדינים של זְעֵיר אַנְפִּין זמינים, וכביכול רַחֲמִים, עושה הדין, אוֹתוֹ הָעַתִּיק מֵהַכֹּל.


שלמדנו, שכשמתגלה עַתִּיק העתיקים, רָצוֹן הָרְצוֹנוֹת, כל המאורות שֶׁנִּקְרָאִים בשם זה - מְאִירִים, וְרַחֲמִים נמצאו בכל. ובשעה שלא מתגלה נִסְתָּר הַנִּסְתָּרִים, וְלֹא מְאִירִים אֵלּוּ המאורים - מִתְעוֹרְרִים דינים, ונעשה הדין. מי גּוֹרֵם לזה הדין? רָצוֹן הָרְצוֹנוֹת, שלא מתגלה.וּמִשּׁוּם כך, הוֹפְכִים הָחוטאים רַחֲמִים לדין. ומה שנאמר כָּאן, "מֵאֵת יְהֹוָה מִן־הַשָּׁמָיִם" (בראשית יט כד), בזעיר אַנְפִּין נֶאֱמַר. ומשמע שכתוב "מִן־הַשָּׁמָיִם" - אֵשׁ וּמַיִם, רַחֲמִים ודין, לְהוֹצִיא את מי שֶׁאֵין בו דין כלל.


לָמַדְנוּ, שֶׁזֶּה הַחוֹטֶם הוּא קָטָן, וכשמתחיל העשן לְהוֹצִיא, יוצא בחיפזון, ונעשה הדין. וּמִי מעכב לזה הַחוֹטֶם שלא יוציא עשן? הַחוֹטֶם של הָעַתִּיק הקדוש, שהוא נקרא 'אֶרֶךְ אַפַּיִם' מֵהַכֹּל.


וְהַיְינוּ, הַסּוֹד שלמדנו, "יְהֹוָה ׀ יְהֹוָה" (שמות לד ו) - מפסיק הַטַּעַם בֵּינֵיהֶם. כל מקום שהשם נִזְכָּר שְׁתֵּי פעמים, מפסיק הַטַּעַם בֵּינֵיהֶם, כְּגוֹן "אַבְרָהָם ׀ אַבְרָהָם" (בראשית כב יא), "יַעֲקֹב ׀ יַעֲקֹב (בראשית מו ב), "שְׁמוּאֵל ׀ שְׁמוּאֵל" (שמואל א ג י), בכולם מפסיק הַטַּעַם בֵּינֵיהֶם. חוּץ מִ"מֹּשֶׁה מֹשֶׁה" (שמות ג ד), שלא מפסיק הַטַּעַם בֵּינֵיהֶם. מה הַטַּעַם? "אַבְרָהָם ׀ אַבְרָהָם" - אחרון שָׁלֵם, רִאשׁוֹן לא שָׁלֵם, שעכשיו שָׁלֵם בעשרה נסיונות, וּמִשּׁוּם כך מפסיק הַטַּעַם בֵּינֵיהֶם, שעכשיו לא הָיָה הוּא כְּבָרִאשׁוֹנָה.


"יַעֲקֹב ׀ יַעֲקֹב" - אחרון שָׁלֵם, רִאשׁוֹן לא שָׁלֵם, שעכשיו הִתְבַּשֵּׂר ביוסף, ושרתה עָלָיו הַשְּׁכִינָה. וְעוֹד, שעכשיו השתלם בארץ, הָאִילָן הקשוש כְּעֵין שֶׁלְּמַעְלָה, בשנים עשר תחומים, בְּשִׁבְעִים ענפים, מה שלא הָיָה בָּרִאשׁוֹנָה. וּמִשּׁוּם כך, הָאַחֲרוֹן שָׁלֵם, הָרִאשׁוֹן לא שָׁלֵם, ומפסיק הַטַּעַם בֵּינֵיהֶם. "שְׁמוּאֵל ׀ שְׁמוּאֵל" - מפסיק הַטַּעַם בֵּינֵיהֶם. מה הַטַּעַם? הָאַחֲרוֹן שָׁלֵם, הָרִאשׁוֹן לא שָׁלֵם, שעכשיו הוּא נָבִיא, וקודם לכן לא הָיָה נָבִיא. אֲבָל "מֹשֶׁה מֹשֶׁה" - לא מפסיק הַטַּעַם בֵּינֵיהֶם, שמהיום שנולד, שָׁלֵם הָיָה, שכתוב, "וַתֵּרֶא אוֹתוֹ כִּי־טוֹב הוּא" (שמות ב ב).


אַף כָּאן, "יְהֹוָה ׀ יְהֹוָה" - מפסיק הַטַּעַם בֵּינֵיהֶם, הָרִאשׁוֹן שָׁלֵם, הָאַחֲרוֹן שָׁלֵם בכל. ומשה, במקום הדין אמר, להוריד לָהֶם מהעתיק הקדוש, רַחֲמִים לִזְעֵיר אַנְפִּין. שכך לָמַדְנוּ, שכמה הכוח של מֹשֶׁה, שהוריד מידות של רַחֲמִים לְמַטָּה. וכשמתגלה הָעַתִּיק בזעיק אַנְפִּין, הַכֹּל ברחמים נראים, והחוטם נשקט, והאש והעשן לא יוצא, כְּמוֹ שנאמר, "וּתְהִלָּתִי אֶחֱטָם־לָךְ" (ישעיה מח ט).


ולמדנו, שבשני הנקבים של הַחוֹטֶם - באחד הנקבים יוצא עשן, לוהט, ושוקע בנקב הטהום הגדול; ומאחד הנקבים, יוצאת אֵשׁ שיוקדת בשלהבתה, ומתלהטת באלף וְאַרְבַּע מאות עוֹלָמוֹת שבצד השמאל. וּמִי שֶׁגּוֹרֵם לקרב בזה, נקרא "אֵשׁ יְהֹוָה" (במדבר יא אׁׁׁ), אֵשׁ אוכלת ויוקדת כל שְׁאָר האשים. וזאת הָאֵשׁ לא מתבסמת, אֶלָּא בָּאֵשׁ הַמִּזְבֵּחַ. וְזֶה העשן שֶׁיּוֹצֵא מהנקב הָאַחֵר, לא מתבסם אֶלָּא בעשן של הקורבן.


וְהַכֹּל תלויים בחוטם, מִשּׁוּם כך כתוב, "וַיָּרַח יְהֹוָה אֶת־רֵיחַ הַנִּיחֹחַ" (בראשית ח כא). שהכל בחוטם תלויים, להריח זה הַחוֹטֶם, בעשן, והאש האדומה. וּמִשּׁוּם כך מתקבל ברצון, וְזֶה שכתוב, "וַיִּחַר־אַף יְהֹוָה" (במדבר יב ט), "וְחָרָה־אַף יְהֹוָה" (דְּבָרִים יא יז), "וְחָרָה אַפִּי" (שמות כב כג), "פֶּן־יֶחרֶה אַף־יְהֹוָה" (דְּבָרִים ו טו). הַכֹּל בזעיר אַנְפִּין נֶאֱמַר, וְלֹא בעתיק.
