לְשַׁבָּחָא לְדָוִד תּוּשְׁבְּחָא וְשִׁירָא:
יְקוּם אֱלָהָא יִתְבַּדְרוּן בַּעֲלֵי דְבָבוֹי וְיַעַרְקוּן סָנְאוֹי מִן קֳדָמוֹי:
הֵיכְמָא דִשְׁקִיף תְּנָנָא יִשְׁקְפוּן הֵיכְמָא דִי יִתְמְסֵי שַׁעֲוָא מִן קֳדָם אֶשָׁא יְהוֹבְדוּן רַשִׁיעַיָא מִן קֳדָם אֱלָהָא:
וְצַדִיקַיָא יֶחְדוּן וִידוּצוּן קֳדָם יְיָ וְיֶחְדוּן בְּחֶדְוָא:
שַׁבַּחוּ קֳדָם אֱלָהָא שַׁבַּחוּ שׁוּם יְקָרֵיהּ קַלְסוּן לְיָתֵב עַל כּוּרְסֵיהּ יְקָרֵיהּ בַּעֲרָבוֹת יָהּ שְׁמֵיהּ וּבִיעוּ קֳדָמוֹי:
אֲבוּהוֹן דְיַתְמֵי וְדַיָנֵהוֹן דְאַרְמְלָן אֱלָהָא בִּמְדוֹר בֵּית שְׁכִינַת קוּדְשֵׁיהּ:
אֱלָהָא דִי מְזַוֵג זוּגִים יְחִידָאִין לְבַר זוּגָא כַּחֲדָא דְאַפֵּיק בֵּית יִשְׂרָאֵל דַהֲווֹן אֲסִירִין בְּמִצְרַיִם עַל עוֹבָדֵיהוֹן דַאֲבָהָתְהוֹן כָּשְׁרִין בְּפוּמְבִּי בְּרַם פַּרְעֹה וּמַשִׁרְיָתֵיהּ דְסָרֵבוּ לְשַׁלָחוּתְהוֹן שְׁרוֹן צְחִיחִין:
אֱלָהָא כַּד נְפַקְתָּא בְעַמוּדָא דַעֲנָנָא וּבְעַמוּדָא דְאֶשְׁתָּא קֳדָם עַמָךְ כַּד טְיַלְתָּא בְּמַדְבְּרָא דִישִׁימוֹן לְעַלְמִין כַּד יְהַבְתָּא אוֹרַיְתָא לְעַמָךְ:
אַרְעָא אִתְרְגִישַׁת לְחוֹד שְׁמַיָא אַטִיפוּ טַלָא מִן קֳדָם יְיָ דֵין סִינַי סְלִיק תְּנָנֵיהּ כִּתְנָנָא דְאַתּוּנָא מִן קֳדָם דְאִתְגְלֵי עֲלוֹי יְיָ אֱלָהָא דְיִשְׂרָאֵל:
כַּד שְׁמַעוּ בֵּית יִשְׂרָאֵל יַת קָל גְבוּרְתָּךְ פְּרַחוּ נַפְשֵׁיהוֹן מִן יַד אַחֵתְתָּא עִלַוֵיהוֹן טַלִין דִתְחַיוּתָא וּמִטְרִין דְרַעֲוָא אֲרֵימְתָּא אֱלָהָא עַל אַחְסַנְתָּךְ וּכְנִשְׁתָּא דְאִשְׁתַּלְהַיַת אַתְּ אַתְקֵינְתָּא:
חַיָתָךְ אַתֵיבְתָּא בָהּ אִתְּקֵנְתָּא מַשִׁרְיַת מַלְאֲכַיָא לְאוֹטָבָא לַעֲנִיַיָא דֶאֱלָהָא:
יְיָ יְהַב פִּתְגָמֵי אוֹרַיְתָא לְעַמֵיהּ בְּרַם מֹשֶׁה וְאַהֲרֹן מְכַשְׁרִין מֵימַר אֱלָהָא לְחַלְוַת רַבָּא דְיִשְׂרָאֵל:
מַלְכְּוָתָא עִם חֵילֵיהוֹן אִטַלְטָלוּ מִן פַּלְטְרֵיהוֹן וְחַכִּימַיָא אִטַלְטְלוּ מִן מַדָעֵהוֹן וּכְנִשְׁתָּא דְיִשְׂרָאֵל מַפְלְגָא עַדָאָהּ מִן שְׁמַיָא:
אִין אַתּוּן רַשִׁיעַיָא שָׁכְבִין בֵּינֵי קִלְקְלָתָא כְּנִשְׁתָּא דְיִשְׂרָאֵל דְדַמְיָא לְיוֹנְתָא מְטַלְלָא בַּעֲנָנֵי יְקָרָא מַפְלְגָא בִּזַת מִצְרָאֵי סִימָא דִזְקִיק וְטַסְבַּרְיָהָא מַלְיָן אוֹבְרִיזִין סְנִין:
כַּד פַּרְסַת אַיְדָהָא עַל יַמָא בִצְלוֹ שַׁדַי אַמְאֵךְ מַלְכְּוָן וְאַמְטוּלְתָּהּ עֲנַן גֵיהִנֹם הֵיךְ תַּלְגָא פְּצָא יָתָהּ מִטוּלָא דְמוֹתָא:
טוּר מוֹרִיָה אֲתַר דִי פְלַחוּ אֲבָהָת עַלְמָא קֳדָם יְיָ אִתְבְּחַר לְבִנְיַן בֵּית מַקְדְשָׁא וְטוּר סִינַי לְמוֹהֲבִית אוֹרַיְתָא טוּר מַתְנָן טוּר תָּבוֹר וְכַרְמֵל אִתְפְּסִילוּ אִתְעֲבֵד לְהוֹן גִבְבָא הֵיךְ טוּר מַתְנָן:
אֲמַר אֱלָהָא לְמָה אַתּוּן טָפְזִין טוּרַיָא לֵית רַעֲוָתִי לְמִתַּן אוֹרַיְתָא עַל טוּרַיָא גֵיוְתָנִין מְבַסְרָנַיָא הָא טוּר סִינַי דְמַכִּיךְ רַגֵג מֵימְרָא דַייָ לְאַשְׁרָאָה עֲלוֹי שְׁכִינְתֵּיהּ לְחוֹד בִּשְׁמֵי שְׁמַיָא יְיָ יִשְׁרֵי לְעָלְמָא:
אַרְתִּיכִין דֶאֱלָהָא תְּרֵין רִבְבָן דְנוּר דְלִיק תְּרֵין אַלְפִין דְאַנְגְלַיָא מַדְבְּרִין לְהוֹן שְׁכִנְתָּא דַייָ שְׁרָת עֲלֵיהוֹן עַל טוּרָא דְסִינָי בִּקְדוּשָׁה:
סְלֵקְתָּא לִרְקִיעַ (מֹשֶׁה נְבִיָא) שְׁבִיתָא שִׁבְיְתָא (אַלֵפְתָּא פִּתְגָמֵי אוֹרַיְתָא) יְהַבְתָּא לְהוֹן מַתְּנָן לִבְנֵי נָשָׁא וּבְרַם סַרְבָּנַיָא דִי מִתְגַיְירִין וְתָיְבִין בִּתְתוּבָא שְׁרָת עֲלֵיהוֹן שְׁכִינַת יְקָרָא דַייָ אֱלֹהִים:
בְּרִיךְ יְיָ עַל כָּל יוֹמָא וְיוֹמָא טְעֵן לָנָא מוֹסִיף פִּקוּדַיָא עֲלַוֵי פִּקוּדַיָא תַּקִיפָא דִי הוּא פֻרְקָנָנָא וּסְעִידָנָא לְעָלְמִין:
אֱלָהָא לָנָא תְקוֹף וּפוּרְקָן וּמִן קֳדָם יְיָ מִתְגָרִין בְּרַשִׁיעַיָא מוֹתָא וּמַפְקָנוּת נִשְׁמָתָא בְּסַרְנוֹקָא:
בְּרַם אֱלָהָא יִתְּבַר רֵישׁ בַּעֲלֵי דְבָבוֹי יְנַתֵּר שְׂעַר גְבַר דִי מִתְהַלֵךְ בְּחוֹבוֹ:
צַדִיקַיָא דְמִיתוּ וְאִתְאֲכַלוּ מִן חֵיוַת בְּרָא אֲמַר יְיָ מִכּוּתְנָן אֲתֵיב אֲתֵיב צַדִיקַיָא דִי יִשְׁתְּנִיקוּ בִּמְצוּלְתֵּיהּ דְיַמָא:
מִן בִּגְלַל דְיֶחֱמוּן פּוּרְעֲנוּת רַשִׁיעָא יִטְמְשׁוּן רַגְלֵיהוֹן בְּדַם קַטִילִין לִישָׁן חֵיוָת בָּרָא מִן תַּרְבֵּיהוֹן יִדְהֲנוּן מִן בַּעֲלֵי דְבָבָא מִנְהוֹן יִסְבְּעוּן:
חֲמוֹן בֵּית יִשְׂרָאֵל הֲלִיכוֹת שְׁכִינְתָּךְ עַל יַמָא אֱלָהָא אָמְרִין הֲלִיכָת אֱלָהָא מְלַךְ כָּל עַלְמָא בִקְדוּשָׁא:
אַקְדִימוּ וַאֲמַרוּ שִׁירְתָא בָּתַר מֹשֶׁה וְאַהֲרֹן דִמְנַגְנִין קֳדָמֵיהוֹן בִּמְצַע נְשַׁיָא צַדִיקָתָא דִי עִם מִרְיָם מְתוֹפְפִין:
בֵּינֵי כְנִישַׁיָא בָּרִיכוּ אֱלָהָא רוֹמְמוּ יְיָ עוֹבְרַיָא בִּמְעֵי אִמְהוֹן זַרְעָא דְיִשְׂרָאֵל:
תַּמָן בִּנְיָמִין זְעֵיר בְּשִׁבְטַיָא דְמִן שֵׁירוּיָא נְחַת לְיַמָא מְטוּל הֵיכְנָא קַבֵּל מַלְכוּתָא מִן שֵׁירוּיָא וּבַתְרֵיהוֹן נְחַתוּ רַבְרְבֵי יְהוּדָה רְגַמוּ יַתְהוֹן שִׁבְטַיָא אַבְנִין וְקַבִּילוּ רַבָּנוּתָא בַּתְרֵיהוֹן רַבָּנֵי זְבוּלֻן הֲווֹן תַּגָרֵיהוֹן רַבָּנֵי נַפְתָּלִי הֲווֹן גִבָּרֵהוֹן:
פַּקֵד אֱלָהָא עוּשְׁנָךְ אִתְעַשַׁן אֱלָהָא שְׁרֵי בְּבֵית מַקְדְשָׁא דְנָן דַעֲבַדְתָּ לָנָא:
מִן הֵיכְלָךְ תְּקַבֵּל קוּרְבָּנַיָא עַל יְרוּשְׁלֶם שְׁכִנְתָּךְ שָׁרְיָא מִפַּנִקְטֵיהוֹן לָךְ יַיְתוּן מַלְכַיָא קוּרְבָּנַיָא:
נְזוֹף בְּמַשִׁרְיַת חַיָבִין תְּרַע יַתְהוֹן הֵיךְ קַנְיָא כְּנִישַׁת גִבָּרִין דְמִתְרַחֲצִין בְּעֶגְלַיָא טַעֲוַת עַמַיָא רְעוּתֵיהּ בְּעַמָא דִי מִתְעַסְקִין בִּרְעוּתָא בְּאוֹרַיְתָא דִזְקִיקָא מִן סִימָא בַּדַר עַמַיָא דִלְאַגָחָא קְרָבָא צָבִין:
יֵיתוּן בְּנוֹי דְחָם חוּסְמָנַיָא מִמִצְרַיִם לְאִתְגַיָרָא בְּנוֹי דְכוּשׁ יִרְהֲטוּן לְמִפְרַשׂ יְדֵיהוֹן בִּצְלוֹ קֳדָם אֳלָהָא:
מַלְכְּוָת אַרְעָא שַׁבַּחוּ קֳדָם יְיָ שַׁבַּחוּ יְיָ לְעָלְמִין:
לְיָתֵב עַל כּוּרְסֵיהּ בִּשְׁמֵי שְׁמַיָא מִן לְקַדְמִין הוּא בְּמֵימְרֵיהּ יְהַב בְּקָלֵיהּ קָל רוּחַ נְבוּאֲתָא לִנְבִיַיָא:
הֲבוּ יְקָר עוּשְׁנָא לֵאלָהָא דִי עַל יִשְׂרָאֵל גֵיוְתָנוּתֵיהּ וְעוּשְׁנֵיהּ בִּשְׁמַיָא:
דְחִיל אֱלָהָא מִן בֵּית מַקְדְשָׁךְ תַּקִיפָא דְיִשְׂרָאֵל הוּא יְהַב עוּשְׁנָא וְתוּקְפָא לְעַמֵיהּ בְּרִיךְ אֱלָהָא:
