שְׁבָחָא לְדָוִד יְיָ שְׁמַע צְלוֹתִי אֲצֵית לְתַחֲנוּנָי בְּקוּשְׁטָךְ עֲנֵי יָתִי בְּצִדְקָתָךְ:
וְלָא תֵעוּל לְבֵית דִינָא עִם עַבְדָךְ אֲרוּם לָא יִזְכֵי קֳדָמָךְ כָּל דְחָי:
אֲרוּם רָדֵף בְּעֵל דְבָבָא נַפְשִׁי דִכְדֵךְ לְאַרְעָא חַיָתִי אוֹתְבַנִי בַּחֲשׁוֹכַיָא הֵיךְ שְׁכִיבֵי עַלְמָא הָדֵין:
בְּאִשְׁתַּלְהֲיוֹת עֲלַי רוּחִי בְּגוּשְׁמִי יִשְׁתַּעֲמֵם לִבִּי:
אִדְכַּרִית יוֹמִין מִלְקַדְמִין רְנַנֵית בְּכוּלְהוֹן עוֹבָדָךְ בְּעוֹבָדֵי אַיְדָךְ אֲמַלֵל:
פְּרִיסֵית אַיְדִי בִצְלוֹ קֳדָמָךְ נַפְשִׁי הֵיךְ אַרְעָא דְצַחֲיָא לְמַיָא מוֹדִיקָא לָךְ לְעָלְמִין:
סַרְהַב עֲנֵי יָתִי יְיָ רְגִיגַת לָךְ רוּחִי לָא תְסַלֵק שְׁכִנְתָּךְ מִנִי וְאִתְמְתִילֵית עִם נָחֲתֵי גוֹב בֵית קְבוּרְתָּא:
אַשְׁמִיעַ יָתִי בִּצְפַר טוּבָךְ אֲרוּם בְּמֵימְרָךְ סִבְרֵית אוֹדַע יְתִי אוֹרַח דְנָן דַאֲהַלֵךְ אֲרוּם לְוָתָךְ זְקַפֵית נַפְשִׁי בִּצְלוֹ:
פְּצֵי יָתִי מִבַּעֲלֵי דְבָבַי יְיָ מֵימְרָךְ מַנֵיתִי לְפָרִיק:
אַלֵף יָתִי לְמֶעְבַּד רְעוּתָךְ אֲרוּם אַנְתְּ אֱלָהִי רוּחַ קוּדְשָׁךְ טָבָא תְּדַבֵּר יָתִי בְּאַרְעָא תְּרִיץ:
מִן בִּגְלַל שְׁמָךְ יְיָ תְּקַיְמִנַנִי בְּצִדְקָךְ אַפֵּק מִן עָקְתָא נַפְשִׁי:
וּבְחַסְדָךְ תְּמַגֵר בַּעֲלֵי דְבָבִי וּתְהוֹבַד כָּל מְעִיקֵי נַפְשִׁי אֲרוּם אֲנָא עַבְדָךְ:
