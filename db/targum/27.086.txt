צְלוֹתָא דְצַלֵי דָוִד אַצְלֵי יְיָ אוּדְנָךְ עֲנֵי יָתִי אֲרוּם עַנְיָא וַחֲשִׁיכָא אֲנָא:
נְטוֹר נַפְשִׁי אֲרוּם חֲסִידָא אֲנָא פְּרוֹק עַבְדָךְ אַתְּ אֱלָהִי דִי אֲנָא מִתְרְחֵץ עֲלָךְ:
חוּס עֲלַי יְיָ אֲרוּם קֳדָמָךְ אֲצַלֵי כָּל יוֹמָא:
אַחְדֵי נַפְשָׁא דְעַבְדָךְ אֲרוּם לְוָתָךְ יְיָ נַפְשִׁי אֶזְקוֹף בִּצְלוֹ:
אֲרוּם אַתְּ הוּא יְיָ טָב לְצַדִיקַיָא וּשְׁבֵק לִדְתָיְבִין לְאוֹרַיְתֵיהּ וּמַסְגֵי טִיבוּ לְכָל דִמְצַלְיָן קֳדָמָךְ:
אֲצֵית יְיָ צְלוֹתִי וְקַבֵּל לְקָל תַּחֲנוּנָי:
בְּיוֹמָא דְעָקְתִי אֶקְרֵי לְוָתָךְ אֲרוּם אַתְּ עָנֵי יָתִי:
לֵית בַּר מִינָךְ בְּאַנְגְלֵי מְרוֹמָא יְיָ וְלֵית כְּעוֹבָדָךְ:
כָּל עַמְמַיָא דִי עֲבַדְתָּ יֵיתוּן וְיִסְגְדוּן קֳדָמָךְ יְיָ וְיִתְּנוּן אַיְקַר לִשְׁמָךְ:
אֲרוּם רַבָּא אַתְּ אֱלָהָא וְעָבֵּד פְּרִישָׁן אַתְּ הוּא אֱלָהָא בִּלְחוֹדָךְ:
אַלֵף יָתִי יְיָ אוֹרְחָתָךְ אֶתְהַלֵךְ בְקוּשְׁטָךְ יַחֵד לִבְּבֵי לְמִדְחַל שְׁמָךְ:
אוֹדִי קֳדָמָךְ יְיָ אֱלֹהָי בְּכָל לִבְּבִי וְאַיְקַר שְׁמָךְ לַעֲלָם:
אֲרוּם טוּבָךְ רַב עָלַי וּפַצִיתָא נַפְשִׁי מִשְׁיוֹל אַרְעִיתָא:
אֱלָהָא זְדוֹנִין קָמוּ עֲלַי וְתַקִיפִין תְּבָעוּ נַפְשִׁי וְלָא שָׁוִיאוּ יָתָךְ לְקִבְלְהוֹן:
וְאַתְּ יְיָ אֱלָהָא רַחֲמָנָא וְחִנְנָא מְרַחֵק רְגַז וּמַסְגֵי לְמֶעְבַּד טַבְוָן וּקְשׁוֹט:
אִתְפְּנֵי לְוָתִי וְחוּס עֲלַי הַב עוֹשְׁנָךְ לְעַבְדָךְ וּפְרוֹק לְבַר אַמְתָךְ:
עֲבֵד עִמִי אָת לְטָבָא בִּזְמַן דְעָיֵל שְׁלֹמֹה בְּרִי יַת אֲרוֹנָא בְּבֵית מַקְדְשָׁא יִתְפַּתְּחָן תַּרְעַיָא אַמְטוּלְתִּי וְיֶחֱמוּ סַנְאַי אֲרוּם שְׁבַקְתָּא לִי וְיִבַּהֲתוּן וִידוּן אֲרוּם אַתְּ יְיָ סִיַעְתַּנִי וְנִחַמְתָּנִי:
