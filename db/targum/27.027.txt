לְדָוִד יְיָ נְהוֹרִי וּפוּרְקָנִי מִמַן אֶדְחַל יְיָ עוּשְׁנָא דְחַיַי מִמַן אֶדְחַל:
כַּד יִתְקַרְבוּן עֲלַי מַבְאֲשִׁין לְגָרְמָא יַת בִּסְרֵי מְעִיקַי וּבַעֲלֵי דְבָבַי הִנוּן אִתְקַלוּ וּנְפָלוּ:
אִין תִּשְׁרֵי עֲלַי מַשִׁרְיַת רַשִׁיעֵי לָא אִדְחַל לְבָבִי אִין תְּקוּם עֲלַי קְרָבָא בְּדָא אֲנָא סָבִיר:
חֲדָא בְּעִיתִי מִן קֳדָם יְיָ יָתָהּ אֶתְבַּע דְאֵתֵב בְּבֵית מַקְדְשָׁא דַייָ כָּל יוֹמֵי חַיַי לְמֶחֱמֵי בְּבַּסִימוּתָא דַייָ וּלְבַקָרָא בְּהֵיכְלֵיהּ:
אֲרוּם יִטְשִׁנַנִי בְּטַלְלֵיהּ בְּיוֹם בִּישְׁתָּא יְטַמְרִנַנִי בְּטִמוּר מַשְׁכְּנֵיהּ בִּכְרַךְ תַּקִיף יְרוֹרָם יָתִי:
וְהַשְׁתָּא יִתְרוֹרָם רֵישִׁי עַל בַּעֲלֵי דְבָבִי חֲזוֹר חֲזוֹר וְאֶכּוֹס בְמַשְׁכְּנֵיהּ נִכְסֵי רַעֲוָא אֲשַׁבַּח וְאֲבִּיעַ קֳדָם יְיָ:
קַבֵּל יְיָ צְלוֹתִי בְּמִקְרָאִי וְחוּס עֲלַי וּרְחֵם יָתִי:
לָךְ אֲמַר לִבִּי בְּעֵי אַפָּי יַת סְבַר אַפָּךְ יְיָ אֶתְבַּע:
לָא תְסַלֵק שְׁכִנְתָּךְ מִנִי לָא תַצְלֵי בְרוּגְזָא לְעַבְדָךְ סִיוּעִי הֲוֵיתָא לָא תְטַלְטֵל יָתִי וְלָא תִשְׁבְּקִנַנִי אֱלָהָא דְפוּרְקָנִי:
מְטוּל דְאַבָּא וְאִמָא יִשְׁבְּקוּ יָתִי יְיָ יִכְנוֹשׁ לִי:
אַלֵפְנִי יְיָ אוּרְחֲתָךְ וּדְבַרְנִי בְּהִלְכְתָא תְרִיצָא מְטוּל תּוּשְׁבְּחָתַי:
לָא תִמְסְרִנַנִי בִּרְעוּת מְעִיקַי אֲרוּם קָמוּ עָלַי סָהֲדֵי שִׁקְרָא וּמְמַלְלֵי חֲטוֹפָא:
אִלוּלֵי דִי הֵימְנִית לְמֶחְמֵי בְטוּבָא דַייָ בְּאַרְעָא דְחַיֵי עַלְמָא:
סְבוֹר עַל יְיָ תְּקוֹף וְעַלֵם לִבָּךְ וּסְבוֹר עַל יְיָ:
