אֱלָהָא מָרֵי פוּרְעֲנוּתָא יְיָ אֱלָהָא מָרֵי פּוּרְעֲנוּתָא הוֹפָע:
אִתְנַטַל דַיָן אַרְעָא אֲתִיב גוּמְלָא בִישָׁא עַל גֵיוְתָנַיָא:
עַד אֵימָתַי רַשִׁיעַיָא יְיָ עַד אֵימָתַי רַשִׁיעַיָא יֵתְבוּן בְּשַׁלְוָתָא:
יְבוּעוּן וִימַלְלוּן גִדוּפִין אָמְרִין מִלִין דִקְלָנָא כָּל עָבְדֵי שְׁקָר:
עַמָךְ יְיָ יְשׁוּפוּן וְאַחְסַנְתָּךְ יְמַסְכְּנוּן:
אַרְמַלְתָּא וְגִיוֹרָא יִקְטְלוּן וְיַתְמֵי יְרַצְחוּן:
וַאֲמַרוּ לָא יֶחֱמֵי יָהּ וְלָא יִתְבַּיֵן אֱלָהָא דְיַעֲקֹב:
הִתְבּוֹנְנוּ דְשָׁטְיָן בְּעַמָא וּכְסִילַיָא אֵימָתַי תַּשְׂכִּילוּן:
הַאֶפְשַׁר דִי נְצִיב אוּדְנָא הֲלָא שָׁמֵיעַ אוּלְפָנָא אִין אֶפְשַׁר דִי בְרָא עַיְנָא וְלָא אִסְתַּכֵּל בְּאוֹרַיְתָא:
הַאֶפְשַׁר דִיהַב אוֹרַיְתָא לְעַמֵיהּ וְכַד יְחוֹבוּן הֲלָא יִתְוַכְּחוּן הֲלָא יְיָ אַלִיף לְאָדָם קָדָמֵי מַנְדְעָא:
קֳדָם יְיָ יְדִיעָן מַחְשְׁבַת בְּנֵי נָשָׁא אֲרוּם הִנוּן לְמָא:
טָב לְגַבְרָא דִי תְכַסְנִנֵיהּ יָהּ וּמִן אוֹרַיְתָךְ תְּאַלְפִנֵיהּ:
לְשַׁדָכָא לֵיהּ מִן יוֹמֵי בִישָׁא עַד דְיִתְבְּרֵי לְרַשִׁיעַיָא שׁוּחָה:
אֲרוּם לָא יִנְטוֹשׁ יְיָ עַמֵיהּ וְאַחְסַנְתֵּיהּ לָא יִשְׁבּוֹק:
אֲרוּם עַד צִדְקָתָא יְתוּב דִינָא וּבַתְרוֹי יִתְפָּרְקוּן כָּל תְּרִיצֵי לִבָּא:
מַן יָקוּם לִי לְאַגָחָא קְרָבָא עִם מַבְאֲשִׁין מַן יִתְעַתַּד לִי לְמִנְצִי עִם עָבְדֵי שְׁקָר:
אִלוּלֵי יְיָ סָעִיד לִי כִּזְעֵר שְׁרָת בִּשְׁתִּיקוּתָא נַפְשִׁי:
אִין אֲמַרִית אִתְמוֹטַטָא רִגְלִי טוּבָךְ יְיָ יַסְעִיד יָתִי:
בְּסִגְעֵי מַחְשְׁבָתָא בְגַוַי נִיחוּמָתָךְ יְפַרְקְנוּן נַפְשִׁי:
הַאֶפְשַׁר דְיִתְחַבֵּר עִמָךְ כּוּרְסֵי שִׁקְרָא בִּרְיַת לְעוּתָא הֲיִתְעַתֵּד עֲלֵי קְיָם:
יִכְנְשׁוּן בִּישָׁן עַל נַפְשָׁא דְצַדִיקָא וְאַדְמָא זַכַּי יְחַיְבוּן דִין דִקְטוֹל:
וִיהֵי יְיָ לִי לְסָעִיד וֵאלָהַי לִתְקוֹף רוּחֲצָנִי:
וַאֲתִיב עֲלֵיהוֹן יַת שְׁקָרֵיהוֹן וּבְבִישָׁתְהוֹן יְגַמְרִינוּן יְגַמְרִינוּן יְיָ אֱלָהָנָא:
