וּנְטַל מִתַּמָן אַבְרָהָם לְאַרְעָא דְרוֹמָא וְיָתִיב בֵּינֵי רְקָם וּבֵינֵי חַגְרָא וְאִתְנְתֵיב בִּגְרָר
וַאֲמַר אַבְרָהָם עַל שָרָה אִנְתְּתֵיהּ אֲחוֹתִי הִיא וְשַׁדַר אֲבִימֶלֶךְ מַלְכָּא דִגְרָר וּדְבַר יַת שָרָה
וְאָתָא מֵימַר מִן קֳדָם יְיָ לְוַת אֲבִימֶלֶךְ בְּחֶלְמָא דְלֵילְיָא וַאֲמַר לֵיהּ הָא אַנְתְּ מַיִית עַל עֵיסַק אִנְתְּתָא דְאַנַסְתְּ וְהִיא מְבַעֲלָא לִגְבַר
וַאֲבִימֶלֶךְ לָא קָרִיב לְגַבָּה לִמְסַאֲבָהּ וְאָמַר יְיָ הֲבַר עַמְמִין דְלָא חַב אוּף חָמֵי לֵיהּ לְמִזְכֵּי בְּדִינָא אִתְקְטֵיל
הֲלָא הוּא אָמַר לִי דְאָחַת הִיא וְהִיא אוּף הִיא אָמְרַת אָחִי הוּא בְּקַשְׁטוּת לְבָבִי וּבְזַכְּאוּת יְדִי עֲבָדִית דָא
וַאֲמַר לֵיהּ מֵימְרָא דַיְיָ בְּחֵילְמָא אוּף קֳדָמַי גְלֵי אֲרוּם בְּקַשִׁיטוּת לְבָבֵךְ עָבַדְתְּ דָא וּמְנָעִית אוּף אֲנָא מִלְמֶחֱטֵי קֳדָמַי בְּגִין כֵּן לָא שְׁבַקְתִּיךְ לְמִקְרַב לְגַבָּהּ
וּכְעַן אָתִיב אִיתַּת גְבַר אֲרוּם נְבִיָא הוּא וְיַצְלִי עֲלָךְ וְתֵחֵי וְאִין לֵיתָךְ מֵתִיב דַע אֲרֵי מֵימְמַת תְּמוּת אַנְתְּ וְכָל דִילָךְ
וְאַקְדִים אֲבִימֶלֶךְ בְּצַפְרָא וּקְרָא לְכָל עַבְדוֹי וּמַלֵיל יַת כָּל פִּתְגָמַיָא הָאִילֵין קֳדָמֵיהוֹן וּדְחִילוּ גוּבְרַיָא לַחֲדָא
וּקְרָא אֲבִימֶלֶךְ לְאַבְרָהָם וְאָמַר מַה עָבַדְתְּ לָנָא וּמַה חָבִית לָךְ אֲרוּם אָתֵיתָא עָלַי וְעַל מַלְכוּתִי חוֹבָא רַבָּא עוּבְדִין דְלָא כַשְׁרִין לְאִתְעַבְדָא עָבַדְתְּ עִמִי
וַאֲמַר אֲבִימֶלֶךְ לְאַבְרָהָם מַה חָמִיתָא אֲרוּם עָבַדְתְּ יַת פִּתְגָמָא הָדֵין
וַאֲמַר אַבְרָהָם אֲרוּם אֲמָרִית בִּלְבָבִי לְחוֹד לֵית דַחַלְתָּא דַיְיָ בְּאַתְרָא הָדֵין וְיִקְטְלוּנַנִי עַל עֵיסַק אִנְתְּתִי
וּבְרַם בְּקוּשְׁטָא אַחְתִי בְּרַת אָחָא דְאִיבָא הִיא בְּרַם לָא מִגְנִיסַת אִימָא וַהֲוַת לִי לְאִנְתּוּ
וַהֲוָה כַּד בָּעוּ לְאַטְעָאָה יָתִי פִּלְחֵי טַעֲוָותָא וּנְפָקֵית מִבֵּית אַבָּא וַאֲמָרִית לָהּ דֵין טִיבוּתִיךְ דְתַעַבְדִין עִמִי לְכָל אַתְרָא דִנְהַךְ לְתַמָן אִמְרִי עָלַי דְאָחִי הוּא
וּנְסֵיב אֲבִימֶלֶךְ עָאן וְתוֹרִין וְעַבְדִין וְאַמְהִין וִיהַב לְאַבְרָהָם וְאָתֵיב לֵיהּ יַת שָרָה אִנְתְּתֵיהּ
וּלְוַת שָרָה אָמַר הָא יְהָבִית אֶלֶף סִלְעִין דִכְסַף לְאָחוּךְ הָא אִינוּן לָךְ תַּחְמְדָא דְעַיְינִין חֲלַף דְאִתְכַּסִית מִן בַּעֲלִיךְ לֵילְיָא חָדָא וְחָמִית יָתָךְ דְאִילוּ יְהָבִית יַת כָּל דַהֲוָה לִי לָא הֲוֵינָא כְמֵיסַת וְאִתְוַכְּחָן מִילַיָא וִידַע אַבְרָהָם אֲרוּם לָא קָרֵיב אֲבִימֶלֶךְ לְוַת שָרָה אִנְתְּתֵיהּ
וְצַלִי אַבְרָהָם קֳדָם יְיָ וְאָסֵי יְיָ יַת אֲבִימֶלֶךְ וְיַת אִנְתְּתֵיהּ וּלְחֵינְתוּ וְאִתְרַוָחוּ
אֲרוּם מֵיחַד אָחִיד מֵימְרָא דַיְיָ בְּאַנְפֵּי כָּל צִירֶיהָ בֵּית וַלְדָא לִנְשַׁיָא דִבְּבֵית אֲבִימֶלֶךְ עַל עֵיסַק שָרָה אִתַּת אַבְרָהָם
