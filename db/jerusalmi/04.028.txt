וּמַלֵיל יְיָ עִם משֶׁה לְמֵימָר
פַּקֵיד יַת בְּנֵי יִשְרָאֵל וְתֵימַר לְהוֹן יַת קוּרְבָּנִי לְחֵים סִדוּר פָּתוֹרֵי יֵיכְלוּן כַּהֲנַיָא וּמַה דְאַתּוּן מְקַרְבִין עַל גַב מַדְבְּחָא לֵית רְשׁוּ לִגְבַר דְיֵיכוּל מִנֵיהּ הֲלָא אֵישָׁתָא הִיא דְאַכְלָא יָתֵיהּ וְהוּא מִתְקַבֵּל קֳדָמֵי לְרֵיחַ רַעֲוָא. עַמִי בְּנֵי יִשְרָאֵל הֲווֹ זְהִירִין לִמְקַרְבָא יָתֵיהּ מֵאַפְרָשׁוּת לִשְׁכְּתָא קָרְבַּן קֳדָמַי בְּזִמְנֵיהּ
וְתֵימַר לְהוֹן דֵין סֵדֶר קָרְבָּנַיָא דִי תְקַרְבוּן קֳדָם יְיָ אִמְרִין בְּנֵי שָׁנָה שַׁלְמִין תְּרֵין לְיוֹמָא עֲלָתָא תְּדִירָא
יַת אִימַר חָד תַּעֲבֵיד בְּצַפְרָא לִמְכַפְּרָא עַל חוֹבֵי לֵילְיָא וְיַת אִימַר תִּנְיַין תַּעֲבֵיד בֵּינֵי שִׁמְשָׁתָא לִמְכַפְּרָא עַל חוֹבֵי יְמָמָא
וְחַד מִן עַשְרָא בִּתְלַת סָוִוין סְמִידָא דְחִינְטַיָא לְמִנְחָתָא פְּתִיכָא בִּמְשַׁח זֵיתָא כְּתִישָׁא רַבְעוּת הִינָא
עֲלַת תְּדִירָא הֵיכְמָא דַהֲוַת מִקְרְבָא עַל טַוְורָא דְסִינַי מְטוֹל לְאִתְקַבָּלָא בְּרַעֲוָא קוּרְבָּנָא קֳדָם יְיָ
וְנִסוּכֵיהּ רַבְעוּת הִינָא לְאִימַר חַד בְּמָאנֵי בֵּית קוּדְשָׁא יִתְנְסַךְ נִיסוּךְ חֲמַר עַתִּיק וְאִין לָא מַשְׁכַּח חֲמַר עַתִּיק מַיְיתֵי חֲמַר בַּר אַרְבְּעִין יוֹמִין לִמְנַסְכָא קֳדָם יְיָ
וְיַת אִימַר תִּנְיַין תַּעֲבֵיד בֵּינֵי שִׁמְשְׁתָא הֵי כְדוֹרוֹנָא דְצַפְרָא וְהֵי כִנְסוּכֵיהּ תַּעֲבֵיד קָרְבַּן דְמִתְקַבֵּל בְּרַעֲוָא קֳדָם יְיָ
וּבְיוֹמָא דְשַׁבַּתָּא תְּרֵין אִמְרִין בְּנֵי שָׁנָה שַׁלְמִין וּתְרֵין עֶשְרוֹנִין סְמִידָא לְמִנְחָתָא פְּתִיכָא בִּמְשַׁח זֵיתָא וְנִיסוּכָהּ
עוֹלַת שַׁבְּתָא תִּתְעֲבֵד בְּשַׁבַּתָּא וּמִיתּוֹסְפָא עַל עוֹלַת תְּדִירָא וְנִיסוּכָהּ
וּבְרֵישֵׁי יַרְחֵיכוֹן תִּקְרְבוּן עֲלָתָא קֳדָם יְיָ תּוֹרִין בְּנֵי תוֹרֵי דְלָא עֵירוּבִין תְּרֵין וּדְכַר חָד אִמְרִין בְּנֵי שְׁנָא שַׁבְעָא שַׁלְמִין
וּתְלָתָא עֶשְרוֹנִין סְמִידָא לְמִנְחָתָא פְּתִיכָא בִּמְשַׁח זֵיתָא לְתוֹרָא חָד וּתְרֵין עֶשְרוֹנִין סְמִידָא דְמִנְחָתָא פְּתִיכָא בִּמְשַׁח זֵיתָא לְדִכְרָא חָד
וְעֶשְרוֹנָא עֶשְרוֹנָא סְמִידָא לְמִנְחָתָא פְתִיכָא בִמְשַׁח זֵיתָא לְאִמְרָא חָד עֲלָתָא לְאִתְקַבָּלָא בְּרַעֲוָא קוּרְבָּנָא קֳדָם יְיָ
וְנִיסוּכֵיהוֹן דְמִתְקְרַב עִמְהוֹן פַּלְגוּת הִינָא יְהֵי לְתוֹרָא וְתַלְתּוּת הִינָא לְדִכְרָא וְרַבְעוּת הִינָא לְאִמְרָא חֲמַר עִינְבֵי דָא עֲלָתָא תֶהֱוֵי מִתְקַרְבָא בְּכָל רֵישׁ יֶרַח וְיֶרַח בִּזְמַן אִתְחַדְתוּת כָּל רֵישֵׁי יַרְחֵי שַׁתָּא
וּצְפִיר בַּר עִזֵי חָד לְחַטָאתָא קֳדָם יְיָ עַל חוּסְרַן סִיהֲרָא עַל עוֹלַת תְּדִירָא תֶּעֱבַד וְנִיסוּכֵיהּ
וּבְיַרְחָא דְנִיסָן בְּאַרְבֵּיסְרֵי יוֹמִין לְיַרְחָא נִכְסַת פִּסְחָא קֳדָם יְיָ
וּבַחֲמֵישַׁת עֲשַר יוֹמִין לְיַרְחָא הָדֵין חַגָא שׁוּבְעָא יוֹמִין פַּטִירֵי יִתְאָכֵל
בְּיוֹמָא קַמָאָה דְחַגָא מְאָרַע קַדִישׁ כָּל עִיבִידַת פּוּלְחָנָא לָא תַעַבְדוּן
וּתְקַרְבוּן קָרְבָּנָא עֲלָתָא קֳדָם יְיָ תּוֹרִין בְּנֵי תוֹרֵי תְּרֵין וּדְכַר חָד וְשׁוּבְעָא אִמְרִין בְּנֵי שְׁנָא שַׁלְמִין יְהוֹן לְכוֹן
וּמִנְחַתְהוֹן סְמִידָא דְחִינְטַיָא פְּתִיכָא בִּמְשַׁח זֵיתָא תְּלָתָא עֶשְרוֹנִין לְתוֹרָא וּתְרֵין עֶשְרוֹנִין לְדִכְרָא תַעַבְדוּן
וְעִשְרוֹנָא עִשְרוֹנָא תַּעֲבֵיד לְאִימַר חָד הֵיכְדֵין לְשׁוּבְעָא אִמְרִין
וּצְפִירָא דְחַטָאתָא חָד לְכַפָּרָא עֲלֵיכוֹן
בַּר מֵעֲלַת צַפְרָא דִי לְעָלָתָא תְּדִירָא תַּעַבְדוּן יַת אִלֵין קוּרְבָּנַיָא
כְּאִלְיֵן קוּרְבָּנֵי יוֹמָא קַמָאָה תַעַבְדוּן כָּל יוֹמָא שׁוּבְעָא יוֹמִין דְחַגָא לְחֵים קָרְבַּן דְמִתְקַבֵּל בְּרַעֲוָא קֳדָם יְיָ עַל עֲלָתָא תְּדִירָא יִתְעֲבֵיד וְנִיסוּכֵיהּ
וּבְיוֹמָא שְׁבִיעָאָה מְאָרַע קַדִישׁ יְהֵי לְכוֹן כָּל עִיבִידַת פּוּלְחָנָא לָא תַעַבְדוּן
וּבְיוֹמָא דְבִכּוּרַיָא בִּקְרוּבְכוֹן דוֹרָנָא מִן עֲלַלְתָּא חַדְתָּא קֳדָם יְיָ בַּעֲצַרְתְּכוֹן כַּד יִתְמְלוּן שִׁבְעָתֵי שְׁבוּעַיָא מְאָרַע קַדִישׁ יְהֵי לְכוֹן כָּל עִיבִידַת פּוּלְחָנָא לָא תַעַבְדוּן
וּתְקַרְבוּן עֲלָתָא לְאִתְקַבָּלָא בְּרַעֲוָא קֳדָם יְיָ תּוֹרִין בְּנֵי תּוֹרֵי תְּרֵין וּדְכַר חָד שׁוּבְעָא אִמְרִין בְּנֵי שְׁנָא
וּמִנְחַתְהוֹן סְמִידָא דְחִינְטַיָא פְּתִיכָא בִּמְשַׁח זֵיתָא תְּלָתָא עֶשְרוֹנִין לְתוֹרָא חָד תְּרֵין עֶשְרוֹנִין לְדִיכְרָא חָד
עִשְרוֹנָא עִשְרוֹנָא לְאִימַר חָד כְּדֵין לְשׁוּבְעָא אִימְרִין
צְפִיר בַּר עִזֵי חָד לְכַפָּרָא עֲלֵיכוֹן
בַּר מִן עָלַת תְּדִירָא וּמִנְחָתֵיהּ תַּעַבְדוּן שַׁלְמִין יֶהֱווֹן לְכוֹן וַחֲמַר נִיסוּכֵיהוֹן
