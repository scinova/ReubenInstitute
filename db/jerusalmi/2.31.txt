וּמַלֵּיל יְיָ עִם משֶׁה לְמֵימָר
חֲמֵי משֶׁה דְּקַרְיַית בְּשׁוּם טַב בְּצַלְאֵל בַּר אוּרִי בַּר חוּר לְשִׁבְטָא דִיהוּדָה
וְאַשְׁלֵמִית עִמֵּיהּ רוּחַ קוּדְשָׁא מִן קֳדָם יְיָ בְּחָכְמָתָא בְּסוּכְלְתָנוּ וּבְמַנְדְּעָא וּבְכָל עִיבִידְתָּא
לְמֵיחְשַׁב בְּרַעֲיוֹנֵיהוֹן הֵיךְ לְמֶעֱבַד בְּדַהֲבָא וּבְכַסְפָּא וּבִנְחָשָׁא
וּבְאַגְלָפוּת מַרְגְּלִיתָא לְאַשְׁלָמָא וּבְנַגָרוֹת קִיסָא לְמֶעֱבַד כָּל עִיבִידְתָּא
וַאֲנָא הָא מַנְיַית עִמֵּיהּ יַת אָהֳלִיאָב בַר אֲחִיסָמָךְ לְשִׁבְטָא דְדָן וּבְלִבָּא דְּכָל חַכִּימֵי לִיבָּא אוֹסִיפִית רוּחַ חוּכְמָתָא וְיַעַבְדוּן יַת כָּל מַה דְּאַפְקֵדִיתָךְ
יַת מַשְׁכַּן זִימְנָא וְיַת אֲרוֹנָא דְסַהֲדוּתָא וְיַת כַּפּוּרְתָּא דְעָלוֹי וְיַת כָּל מָנֵי מַשְׁכְּנָא
וְיַת פְּתוֹרָא וְיַת כָּל מָנוֹי וְיַת מְנַרְתָּא דָכִיתָא וְיַת כָּל מָנָהָא וְיַת מַדְבְּחָא דִקְטוֹרֶת בּוּסְמַיָא
וְיַת מַדְבְּחָא דְעָלָתָא וְיַת כָּל מָנוֹי וְיַת כִּיּוֹרָא וְיַת בְּסִיסֵיהּ
וְיַת לְבוּשֵׁי שִׁמּוּשָׁא וְיַת לְבוּשֵׁי קוּדְשָׁא לְאַהֲרן כַּהֲנָא וְיַת לְבוּשֵׁי בְנוֹי לְשַׁמָּשָׁא
וְיַת מְשַׁח רְבוּתָא וְיַת קְטוֹרֶת בּוּסְמַיָא לְקוּדְשָׁא הֵי כְּכָל מַה דְּפַקִידְתָּךְ יַעַבְדוּן
וַאֲמַר יְיָ לְמשֶׁה לְמֵימָר
וְאַנְתְּ תְּמַלֵּיל עִם בְּנֵי יִשְרָאֵל לְמֵימַר בְּרַם יַת יוֹמֵי שַׁבַּיָא דִי לִי תִנְטְרוּן אֲרוּם אָת הוּא בֵּין מֵימְרִי וּבֵינֵיכוֹן לְדָרֵיכוֹן לְמִידוֹעַ אֲרוּם אֲנָא הוּא יְיָ מְקַדִּישְׁכוֹן
וְתִנְטְרוּן יַת שַׁבְּתָא אֲרוּם קוּדְשָׁא הִיא לְכוֹן דְּיַפְסִינָהּ אִיתְקְטָלָא יִתְקְטֵיל אֲרוּם כָּל מַאן דְּיַעֲבֵיד בָּהּ עִיבִידְתָּא וְיִשְׁתֵּיצֵי בַּר נְשָׁא הַהוּא מֵעַמֵּיהּ
שִׁיתָּא יוֹמִין יִתְעֲבֵיד עִיבִידְתָּא וּבְיוֹמָא שְׁבִיעָאָה שַׁבָּת שַׁבְּתָא קוּדְשָׁא קֳדָם יְיָ כָּל מַאן דְּיַעֲבֵיד עִיבִידְתָּא בְּיוֹמָא דְשַׁבְּתָא אִתְקְטָלָא יִתְקְטֵיל בְּמַטְלוּת אַבְנִין
וְיִנְטְרוּן בְּנֵי יִשְרָאֵל יַת שַׁבְּתָא לְמֶעֱבַד תַּפְנוּקֵי שַׁבְּתָא לְדָרֵיהוֹן קְיַים עֲלָם
בֵּין מֵימְרִי וּבֵין בְּנֵי יִשְרָאֵל אָת הִיא לְעַלָם אֲרוּם בְּשִׁיתָּא יוֹמִין בְּרָא יְיָ וְשַׁכְלֵיל יַת שְׁמַיָא וְיַת אַרְעָא וּבְיוֹמָא שְׁבִיעָאָה שְׁבָת וְנָח
וִיהַב לְמשֶׁה כַּד פָּסַק לְמַלָּלָא עִמֵּיהּ בְּטַוְורָא דְסִינַי תְּרֵין לוּחֵי סַהֲדוּתָא לוּחֵי דְאֶבֶן סַפִּירִינוּן מִכּוּרְסֵי יְקָרָא מַתְקַלְהוֹן אַרְבְּעִין סְאִין כְּתִיבִין בְּאֶצְבְּעָא דַיְיָ
