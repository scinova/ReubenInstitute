וַאֲרוּם תְּאַרַע יַד עָרֵל וְתּוֹתַב דְעִמָךְ וְיִתְמַסְכֵּן אָחוּךְ עִמֵיהּ וְיִזְדַבֵּן לְעָרֵל תּוֹתָב דְעִמָךְ אוֹ לִשְׁרִישׁ פּוּלְחָנָא נוּכְרָאָה לִמְשַׁמְשָׁא לָהּ וּלְפַלְחָהָא דְמִזַרְעִית גִיוֹרָא
בָּתַר דְיִשְׁתְּמוֹדַע לְכוֹן דְיִזְדַבֵּן מִן יַד פּוּרְקָנָא יְהֵי לֵיהּ חַד מִן אָחוֹי יִפְרְקִינֵיהּ
אוֹ אַחְבוֹי אוֹ בַּר אַחְבוֹי יִפְרְקִינֵיהּ אוֹ מִקָרִיב בִּישְרֵיהּ מִזַרְעִיתֵיהּ יִפְרְקִינֵיהּ אוֹ תְּאַרַע יְדֵיהּ אוֹ יְדָא דְצִבּוּרָא וְיִתְפָּרִיק
וִידַיֵיק עִם עַרְלָאָה דְזַבְּנֵיהּ מִשַׁתָּא דְיִזְדַבֵּן לֵיהּ עַד שַׁתָּא דְיוֹבְלָא וִיהֵי כְסַף זַבִּינוֹ כְּמִנְיַן שְׁנַיָא הֵי כְיוֹמֵי אֲגִירָא יְהֵי עִמֵיהּ
אִין עַד כְּדוּן אִית סַגִי בִּשְׁנַיָא לְפוּם סְמוּכְהוֹן יָתִיב פּוּרְקָנֵיהּ מִכְּסַף זְבִינוֹי
וְאִם קָלִיל יִשְׁתַּיְירוּן בִּשְׁנַיָא עַד שַׁתָּא דְיוֹבְלָא וִיחַשֵׁב לֵיהּ כְּפוּם סְכוּם שְׁנוֹי יָתֵיב יַת פּוּרְקָנֵיהּ
הֵי כְּאָגִיר שְׁנָא בִּשְׁנָא יְהֵי עִמֵיהּ לָא יְשַׁעְבֵּיד בֵּיהּ בְּקַשְׁיוּ וְאַנְתְּ חָמֵי לֵיהּ
וְאִין לָא יִתְפְּרַק בְּאִילֵין שְׁנַיָא וְיִפּוֹק בַר חוֹרִין בְּשַׁתָּא דְיוֹבְלָא הוּא וּבְנוֹי עִמֵיהּ
אֲרוּם דִילִי הִינוּן בְּנֵי יִשְרָאֵל מִשְׁתַּעְבְּדוּן לְאוֹרַיְיתָא עַבְדַי הִינוּן דְאַפֵּיקִית יַתְהוֹן פְּרִיקִין מֵאַרְעָא דְמִצְרַיִם אֲנָא הוּא יְיָ אֱלָהָכוֹן
