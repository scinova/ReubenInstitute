מֵימְרָא דַיְיָ אֱלָהָכוֹן דִּמְדַבֵּר קֳדָמֵיכוֹן הוּא יַגִּיחַ לְכוֹן כְּכָל מַה דַּעֲבַד לְכוֹן בְּמִצְרַיִם לְמֵיחְמֵיכוֹן
וּבְמַדְבְּרָא דַחֲמֵיתָא חִוְויָא קָלָן מַלְיָין אֶרֶס דִּקְטוֹל וְסוֹבְרָךְ יְיָ אֱלָהָךְ בַּעֲנָנֵי אִיקַר שְׁכִינְתֵּיהּ הֵיכְמָא דִמְסוֹבַר גְּבַר יַת בְּרֵיהּ בְּכָל אָרְחָא דְהַלֵיכְתּוּן עַד זְמַן מֵיתֵיכוֹן עַד אַתְרָא הָדֵין
וּבְפִתְגָמָא הָדֵין לֵיתֵיכוֹן מְהֵימְנִין בְּמֵימְרָא דַיְיָ אֱלָהָכוֹן
דִּמְדַבֵּר קֳדָמֵיכוֹן בְּאָרְחָא לְאִיתַקְנָא לְכוֹן אֲתַר בֵּית מֵשְׁרוֹי לְאַשְׁרָיוּתְכוֹן בְּעַמּוּדָא דְאֵישָׁא בְּלֵילְיָא לְאַנְהָרוּתְכוֹן בְּאָרְחָא דְתֶהֱכוֹן בָּהּ וְעַמּוּדָא דַעֲנָנָא בִּימָמָא
וּשְׁמִיעַ קֳדָם יְיָ יַת קַל פִּתְגָמֵיכוֹן וּרְגַז וְקַיֵּים לְמֵימָר
אִין יֶחֱמוּן גְּבַר בְּגוּבְרַיָא הָאִילֵין דָּרָא בִישָׁא הָדֵין יַת אַרְעָא טַבְתָא דְקַיֵּימִית לְמִתַּן לְאַבְהַתְכוֹן
אֱלָהֵן כָּלֵב בַּר יְפוּנֶה הוּא יֶחֱמִינָהּ וְלֵיהּ אִיתֵּן יַת אַרְעָא טַבְתָא אַרְעָא דְחֶבְרוֹן דִּמְטַיֵּיל בָּהּ וְלִבְנוֹי חוֹלַף דְּאַשְׁלִים בָּתַר דַּחַלְתָּא דַיְיָ
אוּף עָלַי הֲוָה רְגַז מִן קֳדָם יְיָ מְטוּלְכוֹן לְמֵימָר אוּף אַנְתְּ לָא תֵיעוּל לְתַמָּן
יְהוֹשֻׁעַ בַּר נוּן דִּמְשַׁמֵּשׁ בְּבֵית אוּלְפָנָךְ הוּא יֵיעוֹל לְתַמָּן יָתֵיהּ תַּקִּיף אֲרוּם הוּא יַחְסְנִינָהּ יַת יִשְרָאֵל
וְטַפְלְכוֹן דַּאֲמַרְתּוּן לַעֲדִי יְהוֹן וּבְנֵיכוֹן דְּלָא יָדְעִין יוֹמָא דֵין בֵּין טַב לְבִישׁ הִינוּן יַעֲלוּן לְתַמָּן וּלְהוֹן אֶתְּנִינָהּ וְהִינוּן יֵירְתוּנָהּ
וְאַתּוּן אִתְפְּנִיוּ לְכוֹן וְטוּלוּ לְמַדְבְּרָא אוֹרַח יַמָא דְסוּף
וַאֲתֵיבְתּוּן וַאֲמַרְתּוּן לִי חַבְנָא קֳדָם יְיָ אֲנַחְנָא נִיסַק וְנָגִיחַ קְרָב כְּכָל דְּפַקְדָנָא יְיָ אֱלָהָנָא וַאֲסַרְתּוּן גְּבַר יַת מָאנֵי זַיְינֵיהּ וְשָׁרֵיתוּן לְמֵיסוֹק לְטַוְורָא
וַאֲמַר יְיָ לִי אֵימַר לְהוֹן לָא תִסְקוּן וְלָא תְסַדְּרוּן סִדְרֵי קְרָב אֲרוּם לֵית שְׁכִינְתִּי מְהַלְכָא בֵּינֵיכוֹן וְלָא תִיתְבְּרוּן קֳדָם בַּעֲלֵי דְבָבֵיכוֹן
וּמַלֵלִית לְכוֹן וְלָא קַבֵּילְתּוּן וְסָרֵיבְתּוּן עַל מֵימְרָא דַיְיָ וְאַרְשַׁעְתּוּן וּסְלִיקְתּוּן לְטַוְורָא
וּנְפַק אֱמוֹרָאָה דְיָתִיב בְּטַוְורָא הַהוּא לִקְדָמוּתְכוֹן וּרְדָפוּ יַתְכוֹן הֵיכְמָא דְחַדִדַן וְחַיְישַׁן אוּרְעֲיָיתָא וּמְחוֹ יַתְכוֹן בְּגַבְלָא עַד חָרְמָה
וְתֵיבְתּוּן וּבְכִיתוּן קֳדָם יְיָ וְלָא קַבֵּיל יְיָ צְלוּתְכוֹן וְלָא אָצִית לְמִילֵיכוֹן
וִיתֵיבְתּוּן בִּרְקַם יוֹמִין סַגִּיאִין כְּיוֹמַיָא דִיתֵיבְתּוּן
