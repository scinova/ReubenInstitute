וּמַלֵיל יְיָ עִם משֶׁה לְמֵימָר
מַלֵיל עִם בְּנֵי יִשְרָאֵל וְתֵימַר לְהוֹן גְבַר אֲרוּם יַפְרִישׁ פֵּירוּשׁ דְנִדְרָא בְּעִילוֹי נַפְשָׁתָא לִשְׁמָא דַיְיָ
וִיהֵי עִלוֹיֵיהּ דְבִיר דְכַר מִן בַּר עַשְרִין שְׁנִין וְעַד בַּר שִׁתִּין שְׁנִין וִיהֵי עֲלוֹיֵיהּ חַמְשִׁין סִילְעִין דִכְסַף בְּסִלְעֵי קוּדְשָׁא
וְאִין בְּרַת נוּקְבָא הִיא וִיהֵי עֲלוֹיֵיהּ תְּלָתִין סִלְעִין
וְאִין מִבַּר חֲמֵשׁ שְׁנִין וְעַד בַּר עַשְרִין שְׁנִין וִיהֵי עֲלוֹיֵיהּ דְבִיר דְכַר עַשְרִין סִלְעִין וּלְנוּקְבָא עֲשַר סִלְעִין
וְאִין מִבַּר יַרְחָא וְעַד בַּר חֲמֵשׁ שְׁנִין וִיהֵי עֲלוֹיֵיהּ דְבִיר דְכַר חֲמֵשׁ סִלְעִין דִכְסַף וְדִבְרַתָּא נוּקְבָא עִילוֹיֵיהּ תְּלָתָא סִלְעִין דִכְסַף
וְאִין מִבַּר שִׁתִּין שְׁנִין וּלְעֵילָא אִין בִּיר דְכַר וִיהֵי עֲלוֹיֵיהּ חֲמֵיסַר סִלְעִין וְלִבְרַתָּא נוּקְבָא עֲשַר סִלְעִין
וְאִין מִסְכֵּן הוּא מִמֵסַת עֲלוֹיֵיהּ וְיוֹקִמִינֵיהּ קֳדָם כַּהֲנָא וְיַעֲלִינֵיהּ כַּהֲנָא כְּמֵיסַת דְתַדְבֵּיק יַד נוֹדִירָא הֵיכְדֵין יַעֲלִינֵיהּ כַּהֲנָא
וְאִין בְּעִירָא דִיקַרְבוּן מִינָהּ קוּרְבָּנָא קֳדָם יְיָ כָּל דְיִתֵּן מִינֵיהּ קֳדָם יְיָ יְהֵי קוּדְשָׁא
לָא יְשַׁלְחְפִינֵיהּ וְלָא יִפְרַג יָתֵיהּ שְׁלִים בִּדְבֵיהּ מוּמָא וּדְבֵיהּ מוּמָא בִּשְׁלִים וְאִין מִיפְרוֹג יִפְרַג בְּעִירָא בִּבְעִירָא וִיהֵי הוּא וּפְרוֹגֵיהּ יְהֵי קוּדְשָׁא
וְאִין כָּל בְּעִירָא מְסָאֲבָא דְלָא יִקְרְבוּן מִינָהּ קוּרְבָּנָא קֳדָם יְיָ וִיקֵם יַת בְּעִירָא קֳדָם כַּהֲנָא
וְיַעֲלֵי כַהֲנָא יָתֵיהּ בֵּין טַב לְבֵין בִּישׁ הֵיכְמָא דְיַעֲלֵי כַּהֲנָא הֵיכְדֵין יְהֵי
וְאִם מִפְרוֹק יִפְרְקִינֵיהּ וְיוֹסֵף חוֹמֶשׁ דְמֵיהּ עַל עַלוֹיֵיהּ
וּגְבַר אַרוּם יִקְדַשׁ יַת בֵּיתֵיהּ קוּדְשָׁא קֳדָם יְיָ וְיַעֲלִינֵיהּ כַּהֲנָא בֵּין טַב לְבֵין בִּישׁ הֵיכְמָא דְמַעֲלֵי כַּהֲנָא הֵיכְדִין יְקוּם
וְאִם דֵין דְמַקְדִישׁ יִפְרוֹק יַת בֵּיתֵיהּ וְיוֹסֵף חוֹמֶשׁ כְּסַף עֲלוֹיֵיהּ עֲלוֹי וִיהֵי דִילֵיהּ
וְאִין מִן חֲקַל אַחֲסַנְתֵּיהּ יִקְדַשׁ גְבַר קֳדָם יְיָ וִיהֵי עֲלוֹיֵיהּ כְּמִיסַת זַרְעֵיהּ אֲתַר דְמִזְרַע בֵּית כּוֹר שְעוֹרִין בְּחַמְשִׁין סַלְעִין דִכְסַף
אִין מִשַׁתָּא דְיוֹבֵלָא יִקְדַשׁ חַקְלָא הֵי כְּעִילוֹיֵיהּ יְקוּם
וְאִין בָּתַר יוֹבֵלָא יַקְדִישׁ חַקְלֵיהּ וִידַיֵיק לֵיהּ כַּהֲנָא יַת סְכוּם כַּסְפָּא כְּמִיסַת שְׁנַיָיא דְמִשְׁתַּיְירָן עַד שַׁתָּא דְיוֹבֵלָא וּמְנַכֵּי לֵיהּ מִן עֲלוֹיֵיהּ
וְאִין מִפְרוֹק יִפְרוֹק יַת חַקְלָא דֵין דְאַקְדֵישׁ יָתֵיהּ וְיוֹסִיף חוֹמֶשׁ כְּסַף עֲלוֹיֵיהּ עֲלוֹי וִיקוּם לֵיהּ
וְאִין לָא יִפְרוֹק יַת חַקְלָא וְאִין זַבִּין יַת חַקְלָא לִגְבַר חוֹרַן לָא יִתְפְּרֵק תּוּב
וִיהֵי חַקְלָא בְּמִפְקֵיהּ בְּיוֹבְלָא קוּדְשָׁא קֳדָם יְיָ הֵי כַחֲקַל אַפְרָשָׁא לְכַהֲנָא תְּהֵי אַחְסַנְתֵּיהּ
וְאִין יַת חֲקִיל זְבִינוֹי דְלָא מִן חֲקִיל אַחְסַנְתֵּיהּ יַקְדִישׁ קֳדָם יְיָ
וִידַיֵיק לֵיהּ כַּהֲנָא יַת סְכוּם דְמֵי עֲלוֹיֵיהּ עַד שַׁתָּא דְיוֹבְלָא וְיִתֵּן יַת עֲלוֹיֵיהּ בְּיוֹמָא הַהוּא קוּדְשָׁא קֳדָם יְיָ
בְּשַׁתָּא דְיוֹבְלָא יְתוּב חַקְלָא לְמַן דְזַבְּנֵיהּ מִנֵיהּ לְמַן דִילֵיהּ אַחֲסָנַת אַרְעָא
וְכָל עִילוֹיֵיהּ יְהֵי בְּסִלְעֵי קוּדְשָׁא עַשְרִין מָעִין הַוְיָא סִלְעָא
בְּרַם בּוּכְרָא דְיִתְפְּרֵשׁ לִשְׁמָא דַייָ בִּבְעִירָא לֵית אֶפְשַׁר דְיַקְדִישׁ גְבַר יָתֵיהּ אִין תּוֹר אִין אִימַר לִשְׁמָא דַיְיָ דַיְיָ הוּא
וְאִין בִּבְעִירָא מְסָאֲבָא וְיִפְרוֹק בְּעִילוֹיֵיהּ וְיוֹסֵף חוֹמֶשׁ דְמוֹי עֲלוֹי וְאִין לָא מִתְפְּרֵק וְיִזְדַבֵּן בִּדְמֵי עֲלוֹיֵיהּ
בְּרַם כָּל אַפְרָשָׁא דִי יַפְרֵישׁ גְבַר קֳדָם יְיָ מִן כָּל דְאִית לֵיהּ מֵאֱנָשָׁא וּבְעִירָא וּמֵחֲקַל אַחֲסַנְתֵּיהּ לָא יִזְדַבֵּן וְלָא יִתְפְּרֵק כָּל אַפְרָשָׁא קוֹדֶשׁ קוּדְשִׁין הוּא קֳדָם יְיָ
כָּל אַפְרָשָׁא דִי יִתְפְּרַשׁ מִן אֵינָשָׁא לָא יִתְפְּרֵק בְּכַסְפָּא אֱלָהִין בְּעִלְוַון וּבְנִכְסַת קוּדְשִׁין וּבְמִבְעֵי רַחֲמִין קֳדָם יְיָ מְטוּל דְדֵין קְטוֹל מִתְחַיֵיב
וְכָל מַעַשְרָא דְאַרְעָא מִזַרְעָא דְאַרְעָא וּמִפֵּירֵי אִילָנָא דַיְיָ הוּא קוֹדֶשׁ קוּדְשִׁין הוּא קֳדָם יְיָ
וְאִין מִפְרוֹק יִפְרוֹק גְבַר מִמַעַשְרֵיהּ חוֹמֶשׁ דְמוֹי יוֹסֵף עֲלוֹי
וְכָל מַעַשְרָא דְתוֹרֵי וְעָנֵי כָּל דְחַלְפוּן תְּחוֹת שַׁרְבִיטָא עֲשִירָאָה יְהֵי קוּדְשָׁא קֳדָם יְיָ
לָא יְפַשְׁפֵשׁ בֵּין טַב לְבִישׁ וְלָא יְפַרְגִינֵיהּ וְאִין מִפְרוֹג יְפַרְגִינֵיהּ וִיהֵי הוּא וּפֵירוּגֵיהּ יְהֵי קוּדְשָׁא לָא יִתְפְּרוֹק
אִילִין פִּקוּדַיָא דְפַקֵיד יְיָ יַת משֶׁה וְלֵית אֶפְשַׁר לְחַדְתָּא בְּהוֹן מִדַעַם וּפַקֵדִינוּן מְטוֹל לְאַחְוָואוּתְהוֹן לְוַת בְּנֵי יִשְרָאֵל בְּטוּרָא דְסִינַי
