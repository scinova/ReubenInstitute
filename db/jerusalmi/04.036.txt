וּקְרִיבוּ לְבֵי דִינָא רֵישֵׁי אַבְהָתָא לִגְנִיסַת בְּנֵי גִלְעָד בַּר מָכִיר בַּר מְנַשֶׁה וְלִגְנִיסַת בְּנֵי גִלְעָד בַּר יוֹסֵף וּמַלִילוּ קֳדָם משֶׁה וְקָדָם רַבְרְבַיָא רֵישֵׁי אַבְהָתָא לִבְנֵי יִשְרָאֵל
וַאֲמָרוּ יַת רִבּוֹנִי פַּקֵיד יְיָ לְמִתַּן יַת אַרְעָא בְּאַחְסַנְתָּא בְּעַדְבָא לִבְנֵי יִשְרָאֵל וְרִבּוֹנִי אִתְפַּקֵיד מִן קֳדָם יְיָ לְמִתַּן יַת אַחְסָנַת צְלָפְחָד אָחוּנָא לִבְנָתֵיהּ
וַהֲוָאָה לְחַד מִבְּנֵי שִׁבְטַיָא דִבְנֵי יִשְרָאֵל לִנְשִׁין וְתִתְמְנַע אַחְסַנְתְּהוֹן מֵאַחְסָנַת אַבְהָתָנָא וְתִיתוֹסַף עַל אַחְסָנַת שִׁבְטַיָא דְיֶהֱוְיָן לְהוֹן וּמִן עֲדַב אַחְסַנְתָּנָא יִתְמְנַע
וְאִין יֶהֱוֵי יוּבְלָא לִבְנֵי יִשְרָאֵל וְתִיתוֹסַף אַחְסַנְתְּהוֹן עַל אַחְסָנַת שִׁבְטַיָא דְיֶהֱוְיָן לְהוֹן וּמֵאַחְסָנַת שִׁבְטָא דְאַבְהָתָנָא יִתְמְנַע אַחְסַנְתְּהוֹן
וּפַקֵיד משֶׁה יַת בְּנֵי יִשְרָאֵל עַל מֵימְרָא דַיְיָ לְמֵימָר יָאוּת שִׁבְטָא דִבְנֵי יוֹסֵף מְמַלְלִין
דֵין פִּתְגָמָא דְפַקֵיד יְיָ לָא לְדָרַיָא דַעֲתִידִין לְמֵיקוּם בָּתַר פִּילוּג אַרְעָא אֱלָהֵן לִבְנַת צְלָפְחָד לְמֵימָר לִדְתַקִין בְּעֵינֵיהוֹן תֶּהֱוְיָן לִנְשִׁין לְחוֹד לִגְנִיסַת שֵׁבֶט אֲבוּהֶן תֶּהֱוְיַן לִנְשִׁין
מְטוּל דְלָא תִיתְקֵף אַחְסָנָא לִבְנֵי יִשְרָאֵל מִשִׁבְטָא לְשִׁבְטָא חוֹרָנָא אֲרוּם גְבַר בְּאַחְסָנַת שִׁבְטָא דְאַבְהָתוֹי יִדְבְּקוּן בְּנֵי יִשְרָאֵל
וְכָל בְּרַתָּא יַרְתַת אַחְסָנָא מִשִׁבְטָא דִבְנֵי יִשְרָאֵל לְחַד מִזַרְעִית שִׁבְטָא דַאֲבוּהָא תְּהֵי לְאִנְתּוּ בְּדִיל דְיֵרְתוּן בְּנֵי יִשְרָאֵל גְבַר אַחְסָנַת אַבְהָתוֹהִי
וְלָא תַסְחַר אַחְסָנָא מִשִׁבְטָא לְשִׁבְטָא אוֹחֲרָנָא אֲרֵי גְבַר בְּאַחְסַנְתֵּיהּ יִדְבְּקוּן שִׁבְטַיָא דִבְנֵי יִשְרָאֵל
הֵיכְמָא דְפַקֵיד יְיָ יַת משֶׁה הֵיכְדֵין עֲבָדוּ בְּנַת צְלָפְחָד
וַהֲוָאָה מַחֲלָה תִרְצָה וְחָגְלָה וּמִלְכָּה וְנוֹעָה בְּנַת צְלָפְחָד לִבְנֵי חֲבִיבֵיהוֹן לִנְשִׁין
מִגְנִיסַת בְּנֵי מְנַשֶׁה בַּר יוֹסֵף הֲוָאָה לִנְשִׁין וַהֲוַת אַחְסַנְתְּהוֹן עַל שֵׁיבַט גְנִיסַת אֲבוּהֶן
אִלֵין פִּקוּדַיָא וְסִדְרֵי דִינַיָא דְפַקֵיד יְיָ בִּידָא דְמשֶׁה לְוַת בְּנֵי יִשְרָאֵל בְּמֵישְׁרַיָא דְמוֹאָב עַל יַרְדְנָא דִירֵיחוֹ
