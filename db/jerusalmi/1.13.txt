וּסְלֵיק אַבְרָם מִמִצְרַיִם הוּא וְאִנְתְּתֵיהּ וְכָל דִילֵיהּ וְלוֹט עִמֵיהּ לְמֵיעַל לְדָרוֹמָא
וְאַבְרָם תַּקִיף לַחֲדָא בְּגֵיתֵי בְּכַסְפָּא וּבְדַהֲבָא
וְאָזַל לְמַטְלוֹנוֹהִי מִדָרוֹמָא וְעַד בֵּית אֵל וְתַב עַד אַתְרָא דְפַרְסֵיהּ תַּמָן לְמַשְׁכְּנֵיהּ מִן אַוְולָא בֵּין בֵּית אֵל וּבֵין עָי
לַאֲתַר מַדְבְּחָא דְעָבַר תַּמָן בְּשֵׁירוּיָה וְצַלִי תַּמָן אַבְרָם בִּשְׁמָא דַיְיָ
וְאוּף לְלוֹט דַהֲוָה מִדְבַּר בִּזְכוּתֵיהּ דְאַבְרָם הֲווֹ עָאן וְתוֹרִין וּמַשְׁכְּנִין
וְלָא סוֹבָרַת יַתְהוֹם אַרְעָא לְמֵיתַב כַּחֲדָא אֲרוּם הֲוָה קִנְיָנֵהוֹן סַגִי וְלָא יָכִילוּ לְמֵיתַב כַּחֲדָא
וַהֲוָה דַיָינִין בֵּין רִעָאֵי גֵיתֵיהּ דְאַבְרָם וּבֵין רִעָאֵי גֵיתֵיהּ דְלוֹט דִרְעָאֵי דְאַבְרָם הֲווֹ מִיפַקְדִין מִנֵיהּ לָא תֶהְכוּן בִּכְנַעֲנָאֵי וּבִפְרִיזָאֵי דְעַד כְּדוֹן אֵית לְהוֹם רְשׁוּתָא בְּאַרְעָא וַהֲווֹ זְמָמִין בְּעִירֵיהוֹן דְלָא יֵיכְלוּן גְזֵלָה עַד דַהֲווּ אַתְיָן לַאֲתַר מַרְעֵיהוֹן וְרָעֵי דְלוֹט הֲווֹ מְבַקְרִין וְאַזְלִין וְאָכְלִין בְּחַקְלֵי כְּנַעֲנָאֵי וּפְרִיזָאֵי דְעַד כְּדוֹן יַתְבִין בְּאַרְעָא
וַאֲמַר אַבְרָם לְלוֹט לָא כְעַן תְּהֵי מַצוּתָא בֵּינָא וּבֵינָךְ וּבֵין רַעֲוָותִי וּבֵין רַעֲוָותָךְ אֲרוּם גוּבְרִין אָחִין אֲנַחְנָא
הֲלָא כָל אַרְעָא קֳדָמָךְ אִתְפָּרַשׁ כְּדוּן מִינִי אִם אַנְתְּ לְצִיפוּנָא וַאֲנָא לְדָרוֹמָא אִם אַנְתְּ לְדָרוֹמָא וַאֲנָא לְצִיפּוּנָא
וּזְקַף לוֹט יַת עֵינוֹי לִזְנוֹי וַחֲמָא יַת כָּל מֵישַׁר יַרְדְנָא אֲרוּם כּוּלָה בֵּית שַׁקְיָיא קֳדָם עַד לָא חַבֵּל יְיָ בְּרוּגְזֵיהּ יַת סְדוֹם וְיַת עֲמוֹרָה הֲוַת אַרְעָא הַהִיא מְשַׁבְּחָא בְּאִילָנַיָא כְּגִינוּנִיתָא דַיְיָ וּכְעִילָתָא כְאַרְעָא דְמִצְרַיִם מַעֲלָךְ לְצוֹעַר
וּבְחַר לֵיהּ לוֹט יַת כָּל מֵישַׁר יַרְדְנָא וּנְטַל לוֹט מִמַדִינְחָא וְאִיתְפַּרְשׁוּ אֵינַשׁ מֵעַל אָחוֹי
אַבְרָם יָתֵיב בְּאַרְעָא דִכְנָעַן וְלוֹט יָתֵיב בְּקִרְוֵוי מֵישְׁרָא וּפְרַס לְמַשְׁכְּנֵיהּ עַד סְדוֹם
וְאֵינָשִׁין דִסְדוֹם בִּישִׁין בְּמָמוֹנְהוֹן דֵין לְדֵין וְחַיָיבִין בְּגוּפְהוֹן בְּגִילוּי עֶרְיְיתָא וְשַׁדְיוּת אֲדַם זַכְּאַי וּפַלְחַן פּוּלְחָנָא נוּכְרָאָה וּמְרָדִין לִשְׁמָא דַיְיָ לַחֲדָא
וַיְיָ אָמַר לְאַבְרָם בָּתַר דְאִתְפְּרַשׁ לוֹט מִנֵיהּ זְקוֹף כְּדוֹן עֵינָךְ וְתֶחֱמֵי מִן אַתְרָא דְאַנְתְּ תַּמָן לְצִיפּוּנָא וּלְדָרוֹמָא וּלְמַדִינְחָא וּלְמַעַרְבָא
אֲרוּם יַת כָּל אַרְעָא דְאַנְתְּ חָמֵי לָךְ אֶתְּנִינָהּ וְלִבְנָךְ עַד עַלְמָא
וְאֵשַׁוֵי יַת בְּנָךְ סַגִיאִין כְּעַפְרָא דְאַרְעָא דְהֵיכְמָא דְאִית אֶפְשַר לִגְבַר לְמִימְנֵי יַת עַפְרָא דְאַרְעָא אַף כֵּן בְּנָךְ אֶפְשַׁר דְיִתְמְנוּן
קוּם טַיֵיל בְּאַרְעָא וַעֲבַד בָּהּ חֶזְקָתָא לְאָרְכָּא וּלְפַתְיָא אֲרוּם לָךְ אֶתְּנִינָהּ
וּפְרַס אַבְרָם לְמַשְׁכְּנֵיהּ תַיְירִין אִימוֹר (ס"א משׁכוני לתירי ואמרי) וְאָתָא וְיָתִיב בְּחֵיזוֹ מַמְרֵא דִי בְּחֶבְרוֹן וּבְנָא תַמָן מַדְבְּחָא קֳדָם יְיָ
