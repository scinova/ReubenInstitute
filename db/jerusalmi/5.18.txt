וְאַחֲסָנַת חֲקַל וּכְרַם לָא תְהֵי לֵיהּ בֵּינֵי אָחוֹי עֶשְרִין וְאַרְבַּע מוֹהֲבוּתָא דִכְהוּנְתָּא דִיהַב לֵיהּ יְיָ הִינוּן אַחְסַנְתֵּיהּ הֵיכְמָא דְמַלִּיל לֵיהּ
וְדֵין יְהֵי חוּלְקָא דְּחָמֵי לְכַהֲנַיָא מִן עַמָּא מִלְוַת דִּבְחֵי דִבְחִין אִין תּוֹר אִין אִימַר וְיִתְּנוּן לְכַהֲנָא אִדְרָעָא דְיַמִּינָא וְלוּחַיָא אַרְעֲיָא וְלוּחַיָא דִרְקִיתָא וְקֵיבְתָא
שֵׁירוּי עִיבּוּרֵיכוֹן חַמְרֵיכוֹן וּמִשְׁחֵיכוֹן וְשֵׁירוּי גִּיזַת עָנֵיכוֹן כְּמִסַת קָטוֹרָא תִּתְּנוּן לֵיהּ
מְטוּל דְּבֵיהּ אִתְרְעֵי יְיָ אֱלָהָכוֹן מִכָּל שִׁבְטֵיכוֹן לְמֵיקוּם לְשַׁמָּשָׁא בִּשְׁמָא דַיְיָ הוּא וּבְנוֹי כָּל יוֹמַיָא
וַאֲרוּם יֵיתֵי לֵיוָאָה מֵחֲדָא מִן קִרְוֵיכוֹן מִן כָּל יִשְרָאֵל דְּהוּא דַיָיר תַּמָּן וְיֵיתֵי בְּכָל כְּרוּךְ דְּרָעֲיָא נַפְשֵׁיהּ לְאַתְרָא דְיִתְרְעֵי יְיָ
וִישַׁמֵּשׁ בְּשׁוּם מֵימְרָא דַיְיָ אֱלָהֵיהּ הֵי כְּכָל אָחוֹי לֵיוָאֵי דִמְשַׁמְשִׁין תַּמָּן קֳדָם יְיָ
חוּלַק כָּל קֳבֵיל חוּלַק בְּשָׁוֶה יֵיכְלוּן בַּר מִמּוֹתְבֵי קוּרְבָּנַיָא דְיֵיכְלוּן כַּהֲנַיָא דְאוֹרִיתוּ לְהוֹן אֶלְעָזָר וְאִיתָמָר אֲבָהַתְכוֹן
אֲרוּם אַתּוּן עַלְלִין לְאַרְעָא דַיְיָ אֱלָהָכוֹן יָהֵיב לְכוֹן לָא תֵילְפוּן לְמֶעֱבַד כְּרִיחוֹקֵי עַמְמַיָא הָאִינוּן
לָא יִשְׁתְּכַח בְּכוֹן מְעַבְּרִין בְּנֵיהוֹן וּבְנָתֵיהוֹן בְּנוּרָא לָא קְסוֹמֵי קוּסְמִין וְלָא חֲרוּרֵי עַיְינִין וְלָא נַטוֹרֵי נְחָשִׁין וְלָא חַרְשִׁין
וּמְחַבְּרִין וְאַסְרִין חֵיוִין וְעַקְרַבִּין וְכָל מִינֵי רַחֲשִׁין וְשַׁאֲלִין אִיבָא טַמְיָא וּגְרַם יָדוּעַ וּתְבַע מִן מֵיתַיָא
אֲרוּם מְרַחֵק קֳדָם יְיָ כָּל עֲבֵיד אִלֵין וּמְטוֹל תּוֹעֵיבְתָּא הָאִילֵין יְיָ אֱלָהָכוֹן מְתָרֵיכֵיהוֹן מִן קֳדָמֵיכוֹן
שַׁלְמִין תְּהוֹן בְּדַחַלְתָּא דַיְיָ אֱלָהָכוֹן
אֲרוּם עַמְמַיָא הָאִלֵין דְּאַתּוּן עֲתִידִין לְמֵירוֹת יַתְהוֹן לְחַרוֹרֵי עֵינָא וּלְקִיסוּמֵי קוּסְמַיָא צַיְיתוּן וְאַתּוּן לָא כְוַותְהוֹן אֱלָהֵן כַּהֲנַיָא שַׁיְילוּן אוּרַיָא וְתוּמַיָא וְנָבִיָא תְרִיצַיָא יְהַב לְכוֹן יְיָ אֱלָהָכוֹן
נְבִיָא מִבֵּינֵיכוֹן מִן אֲחוּכוֹן דְּדָמֵי לִי בְּרוּחַ קוּדְשָׁא יְקִים לְכוֹן יְיָ אֱלָהָכוֹן מִנֵּיהּ תְקַבְּלוּן
כְּכל דְּשַׁיִּילְתּוּן מִן קֳדָם יְיָ אֱלָהָכוֹן בְּחוֹרֵב בְּיוֹמָא דְאִתְכַּנָּשׁוּ שִׁבְטַיָא לִמְקַבְּלָא אוֹרַיְיתָא לְמֵימָר לָא נוֹסִיף לְמִשְׁמַע יַת קַל דִּיבּוּרָא מִן קֳדָם יְיָ אֱלָהָנָא וְיַת אֵישָׁתָא רַבְּתָא הֲדָא לָא נֶחֱמֵי תּוּב דְּלָא נְמוּת
וַאֲמַר יְיָ לִי אַתְקִינָן מַה דְּמַלֵּילָן
נְבִיָא אָקִים לְהוֹן מִבֵּינֵי אֲחוּהוֹן דְּרוּחַ קוּדְשָׁא בֵּיהּ כְּוָותָךְ וְאֶתֵּן פִּתְגָמֵי נְבוּאָתִי בְּפוּמֵיהּ וִימַלֵּיל עִמְהוֹן יַת כָּל דְּאִיפְקְדִינֵיהּ
וִיהֵי גַבְרָא דְלָא יִתְקַבֵּל פִּתְגָמֵי נְבוּאָתִי דִימַלֵּיל בִּשְׁמִי מֵימְרִי יִפְרַע מִנֵּיהּ
בְּרַם נְבִי שִׁקְרָא דְיִרְשַׁע לְמַלָלָא פִּתְגָמָא בִּשְׁמִי יַת דְּלָא פַקִידְתֵּיהּ לְמַלָּלָא וְדִי יְמַלֵּל בְּשׁוּם טַעֲוַות עַמְמַיָא וְיִתְקְטֵיל נְבִיָא הַהוּא בְּסַיְיפָא
וַאֲרוּם תֵּימְרוּן בְּרַעֲיוֹנֵיכוֹן הֵיכְדֵין נֵידַע יַת פִּתְגָּמָא דְלָא מַלְלֵיהּ יְיָ
