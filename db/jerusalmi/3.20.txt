וּמַלֵיל יְיָ עִם משֶׁה לְמֵימָר
וְעִם בְּנֵי יִשְרָאֵל תְּמַלֵיל לְמֵימָר גְבַר טְלֵי אוֹ סִיב מִן גְנִיסַת בְּנֵי יִשְרָאֵל דְיַעֲבֵיר מִזַרְעֵיהּ לְמוֹלֶךְ לְמִיתוֹקְדָא בְּנוּרָא אִיתְקְטָלָא יִתְקְטֵיל עַמָא בֵּית יִשְרָאֵל יְחַיְיבוּן יָתֵיהּ בְּמַטְלוּת אַבְנִין
וַאֲנָא אֶתֵּן פְּנוּיָיתָא לְמֶעֱסוֹק בְּגַבְרָא הַהוּא וְאֵישֵׁיצֵי יָתֵיהּ מִגוֹ עַמֵיהּ אֲרוּם מִזַרְעֵיהּ יְהַב לְפּוּלְחָנָא אוֹחֲרָא מִן בִּגְלַל לְסַאֲבָא יַת מַקְדָשִׁי וּלְאַפְסָא יַת שְׁמָא דְקוּדְשִׁי
וְאִין מִכְבַּשׁ יִכְבְשׁוּן עַמָא בֵית יִשְרָאֵל יַת עֵינֵיהוֹן מִן גַבְרָא הַהוּא בִּדְיִתֵּן מִזַרְעֵיהּ לְפוּלְחָנָא אוֹחֲרָא מְטוֹל דְלָא לְמִקְטוֹל יָתֵיהּ
וְאֶשַׁוֵי אֲנָא פְנוּיַיְתָא לְמֶעֱסוֹק בְּגַבְרָא הַהוּא וּבִגְנִיסָתֵיהּ דִמְחַפְיַין עֲלוֹי לְאִתְרְדָאָה בְּיִסוּרֵי וְיָתֵיהּ אֵישֵׁיצֵי וְיַת כָּל דְטָעַן בַּתְרוֹי לְמִטְעֵי בָּתַר פּוּלְחָנָא אוֹחֲרָא מִגוֹ עַמֵיהּ
וּבַר נַשׁ דְיִסְטֵי בָּתַר שַׁאֲלֵי בִּידִין וּמַסְקֵי זְכוּרוּ וְתַבְעֵי גְרַם יְדוֹעַ לְמִטְעֵי בַּתְרֵיהוֹן וְאֶתֵּן פְּנִיוּתָא לְמֶעֱסוֹק בְּבַר נְשָׁא הַהוּא וְאֵישֵׁיצֵי יָתֵיהּ בְּמוֹתָנָא מִגוֹ עַמֵיהּ
וְתִתְקַדְשׁוּן וּתְהוֹן קַדִישִׁין בְּגוּפֵיכוֹן מְטוֹל דַאֲקַבֵּל בְּרַעֲוָא צְלוֹתְכוֹן אֲנָא הוּא יְיָ מְקַדִשְׁכוֹן
וְתִטְרוּן יַת קְיָמַי וְתַעַבְדוּן יַתְהוֹן אֲנָא הוּא יְיָ מְקַדִשְׁכוֹן
אֲרוּם גְבַר טְלֵי וּגְבַר סִיב דְיֵילוּט יַת אָבוֹי וְיַת אִמֵיהּ בִּשְׁמָא מְפָרֵשׁ אִתְקְטָלָא יִתְקְטֵיל בְּאַטְלוּת אַבְנִין מְטוֹל דְאָבוֹי וְאִמֵיהּ לָט קְטָלָא חַיָיב
וּגְבַר דִי יִגוֹר יַת אִתַּת גְבַר מֵאַרְסָא אוֹ דִי יִגוֹר יַת אִתַּת חַבְרֵיהּ מִבַּעְלָא אִיתְקְטָלָא יִתְקְטֵיל עַל מִיבְעֲלָא בְּשִׁינוּק סוּדְרָא אַקוֹשָׁא בְּגוֹ רְכִיכָא וְעַל מֵאַרְסָא בְּאַטְלוּת אַבְנִין גִיוֹרָא וְגִיוֹרְתָּא
וּגְבַר דִי יְשַׁמֵשׁ עִם אִתַּת אָבוֹי בֵּין דְהִיא אִמֵיהּ בֵּין דְהִיא חוֹרַנְיָיתָא עֲרְיָיתָא דְאָבוֹי בָּזֵי אִיתְקְטָלָא יִתְקַטְלוּן תְּרֵיהוֹן קְטָלָא חַיָיבִין בְּאַטְלוּת אַבְנִין
וּגְבַר דִי יְשַׁמֵשׁ עִם כַּלָתֵיהּ אִיתְקְטָלָא יִתְקַטְלוּן תְּרֵיהוֹן תִּבְלָא עָבָדוּ קְטָלָא חַיָיבִין בְּאַטְלוּת אַבְנִין
וּגְבַר דִי יְשַׁמֵשׁ עִם דְכוּרָא תַּשְׁמִישִׁין דְאִיתָא תּוֹעֵיבְתָּא עָבָדוּ תְּרֵיהוֹן אִיתְקְטָלָא יִתְקַטְלוּן בְּאַטְלוּת אַבְנִין
וּגְבַר דְיִסַב יַת אִיתָא וְיַת אִימָהּ זְנוּ הִיא בְּנוּרָא יוֹקְדוּן יָתֵיהּ וְיַתְהֶן בְּאַתְּכוּת אַבָּר לְפוּמְהוֹן וְלָא תְהֵי זְנוּ בֵּינֵיכוֹן
וּגְבַר דִי יִתֵּן תַּשְׁמִישֵׁיהּ בִּבְעִירָא אִיתְקְטָלָא יִתְקְטֵיל בְּאַטְלוּת אַבְנִין וְיַת בְּעִירָא תְּקַטְלוּן בְּקוֹלְפֵי
וְאִתְּתָא דִי תִקְרַב לְצֵיד כָּל בְּעִירָא לְמִתְהַנְיָא מִינָהּ וְתִקְטְלוּן יַת אִיתְּתָא בְּאַטְלוּת אַבְנִין וְיַת בְּעִירָא בְּקוֹלְפֵי אִתְקְטָלָא יִתְקַטְלוּן דִין קְטוֹל חַיָיבִין
וּגְבַר דִי יְשַׁמֵשׁ עִם אַחֲתֵיהּ בְּרַת אָבוֹי אוֹ בְּרַת אִימֵיהּ וְיִבְזֵי יַת עֶרְיְיתָהּ וְהִיא תִבְזֵי יַת עֶרְיְיתֵיהּ גְנַיי הוּא אֲרוּם חִסְדָא עֲבָדִית עִם קַדְמָאֵי מִן בִּגְלַל דְיִתְמְלֵי עַלְמָא מִנְהוֹן עַד לָא אִיתְיְהִיב קְיָימָא בְּעַלְמָא וּמִן דְאִתְיְהִיב קְיָימָא בְּעַלְמָא כָּל דְיֶעֱבַד כְּדֵין וְיִשְׁתֵּיצוּן בְּמוֹתָנָא וְחָמְיַין בְּבִישְׁתְּהוֹן בְּנֵי עַמְהוֹן מְטוֹל דְעֶרְיַית אַחְתֵיהּ בְּזֵי חוֹבֵיהּ יְקַבֵּל
וּגְבַר דִי יְשַׁמֵשׁ עִם אִתְּתָא דְוָותָא וּבְזֵי יַת עִרְיְיתָהּ וְיַת מַבּוּעַ אֲדַם סוּבְתָא בְּזֵי וְהִיא בָּזִית יַת מַבּוּעַ אַדְמָהָא וְיִשְׁתֵּיצוּן תַּרְוֵיהוֹן בְּמוֹתָנָא מִגוֹ עַמְהֶן
וְעֶרְיַת אֲחַת אִמָךְ וַאֲחַת אַבָּךְ לָא תִבְזֵי אֲרוּם יַת קְרֵיבֵיהּ בִּישְרֵיהּ בְּזֵי חוֹבֵיהוֹן יְקַבְּלוּן בְּמוֹתָנָא יְסוּפוּן
וּגְבַר דִי יְשַׁמֵשׁ עִם אִתַּת אַחְבוֹי עֶרְיַת אַחְבוֹי בְּזֵי חוֹבֵיהוֹן יְקַבְּלוּן בְּמוֹתָנָא יְסוּפוּן בְּלָא וְלַד יְמוּתוּן
וּגְבַר דִי יִסַב יַת אִתַּת אַחְבוֹי בְּחַיוֹי מְרַחְקָא הִיא עֶרְיְיתָא דַאֲחוֹי בְּזֵי בְּלָא וְלַד יְהוֹן
וְתִיטְרוּן אַתּוּן כְּנִישְׁתָּא דְיִשְרָאֵל יַת כָּל קְיָימַי וְיַת כָּל סִדְרֵי דִינַיי וְתַעַבְדוּן יַתְהוֹן וְלָא תִפְלוֹט יַתְכוֹן אַרְעָא דַאֲנָא מָעִיל יַתְכוֹן לְתַמָן לְמֵיתַב בָּהּ
וְלָא תְהָכוּן בְּנִימוּסֵי עַמְמַיָא דַאֲנָא מַגְלֵי מִן קֳדָמֵיכוֹן אֲרוּם יַת כָּל מְרַחֲקַיָיא הָאִילֵין עָבָדוּ וְרָחִיק מֵימְרִי יַתְהוֹן
וַאֲמָרִית לְכוֹן אַתּוּן תִּזְדַהֲרוּן מִן תּוֹעֵיבְתָּא אִילֵין מִן בִּגְלַל דְתֵירְתוּן יַת אַרְעֲהוֹן וַאֲנָא אִיתְּנִינָהּ לְכוֹן לְמֵירַת יָתָהּ אֲרַע עַבְדָא חֲלַב וּדְבַשׁ אֲנָא הוּא יְיָ אֱלָהָכוֹן דְאַפְרָשִׁית יַתְכוֹן מִן עַמְמַיָא
וְתִפְרְשׁוּן בֵּין בְּעִירָא דְמִכַּשְׁרָא לְמֵיכַל לִדְמִיפְסְלָא לְמֵיכַל וּבֵין עוֹפָא דְמִפְסוֹל לְמֵיכַל לִדְמַכְשַׁר לְמֵיכַל וְלָא תְשַׁקְצוּן יַת נַפְשָׁתֵיכוֹן בִּבְעִירָא דְרִיסַת חַיְיתָא וּבְעוֹפָא דָרִיס בַּר נִיצְצָא וּבְכָל מַן דְתַרְחֵישׁ אַרְעָא דְאַפְרָשִׁית לְכוֹן לְסוֹאֲבֵיהוֹן
וְתֶהֱווֹן קֳדָמַי קַדִישִׁין אֲרוּם קַדִישׁ אֲנָא יְיָ דִבְחָרִית בְּכוֹן וְאַפְרָשִׁית יַתְכוֹן מִן עַמְמַיָא מְטוֹל לְמֶהֱוֵי פַּלְחִין קֳדָמַי
וּגְבַר אוֹ אִתְּתָא אֲרוּם יְהֵי בְּהוֹן בִּידִין אוֹ זְכוּרוּ אִתְקְטָלָא יִתְקַטְלוּן בְּאַבְנָא יְאַטְלוּן יַתְהוֹן קְטָלָא חַיָיבִין
