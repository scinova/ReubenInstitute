וּבְרִיךְ יְיָ יַת נחַ וְיַת בְּנוֹי, וַאֲמַר לְהוֹם: פּוּשוּ וּסְגוּ, וּמְלוּ יַת אַרְעָא.
וְדַחֲלַתְכוֹן וְאֵימַתְכוֹן יְהֵי עַל כָּל חֵיוַת אַרְעָא, וְעַל כָּל עוֹפָא דִשְׁמַיָא; מִכָּל דְתַרְחֵשׁ אַרְעָא, וּבְכָל נוּנֵי יַמָא - בְּיֶדְכֶם _יִתְמַסְרוּן_.
כָּל רִיחֲשָׁא דְהוּא חַי - _דִ_לְכוֹן יְהֵי לְמֵיכַל; כִּירוֹק עִסְבָּא יְהָבִית לְכוֹן יַת כּוֹלָא.
בְּרַם בִּשְרָא +דְתָלִישׁ מִן חַיוּתָא חַיָא, בִּזְמַן דִ+בְנַפְשֵׁיהּ _בֵּיהּ_+, אוֹ דְתָלִישׁ מִן חַיוּתָא נְכִיסְתָּא, וְעַד דְלָא נַפְקָא כּוֹלָא נִשְׁמָתָא+ - לָא תֵיכְלוּן!
וּבְרַם יַת דִמְכוֹן לְנַפְשָׁתֵיכוֹן אֵתְבּוֹעַ, מִן יְדָא דְכָל חַיְתָא +דְקַטְלָא לְבַר נְשָׁא+ אִיתְבּוֹעִינֵיהּ +לְאִיתְקַטְלָא עֲלֵיהּ+; וּמִידָא דְאֵינָשָׁא, מִיַד גְבַר +דְיֵישׁוֹד יַת דְמָא דְ+אָחוֹי, אִתְבַּע יַת נַפְשָׁא דְאֵינָשָׁא.
דְיֵישׁוֹד דְמָא דְאֵינָשָׁא - +בְּסַהֲדִין דְיָינַיָא מְחַיְיבִין לֵיהּ קְטוֹל; וּדְיֵישׁוֹד בְּלָא סַהֲדִין - מָרֵי עָלְמָא עָתִיד לְאִתְפַּרְעָא מִנֵיהּ לְיוֹם דִינָא רַבָּא+; אֲרוּם בִּדְיוֹקְנָא דַיְיָ עֲבַד יַת אֵינָשָׁא.
וְאַתּוּן - פּוּשׁוּ וּסְגוּ! _אִיתְיְלִידוּ_ בְּאַרְעָא, וּסְגוּ בָהּ!
וַאֲמַר יְיָ לְנחַ, וְלִבְנוֹי עִמֵיהּ, לְמֵימַר:
אֲנָא - הָא אֲנָא מְקַיֵים קְיָמַי עִמְכוֹן, וְ_עִם בְּנֵיכוֹן_ בַּתְרֵיכוֹן.
וְעִם כָּל נַפְשָׁא חַיְיתָא דְעִמְכוֹן: בְּעוֹפָא, +וּ+בִבְעִירָא, וּבְכָל חַיַת אַרְעָא +דְ+עִמְכוֹן; מִכָּל נַפְקֵי תֵיבוּתָא, לְכָל חַיַת אַרְעָא.
וַאֲקַיֵים יַת קְיָימַי עִמְכוֹן, וְלָא _יִשְׁתֵּצֵי_ כָּל בִּשְרָא עוֹד מִמוֹי דְטוּבְעָנָא; וְלָא יְהִי עוֹד טוּבְעָנָא _לְחַבָּלָא_ אַרְעָא.
וַאֲמַר יְיָ: דָא אַת קַיָימָא, דַאֲנָא _מְקַיֵים_ בֵּין +מֵימְרִי+ וּבֵינֵיכוֹן, וּבֵין כָּל נַפְשַׁת חַיְתָא דְעִמְכוֹן - לְדָרֵי עַלְמָא.
יַת קַשְׁתִּי יְהָבִית בַּעֲנָנָא; וּתְהֵי לְסִימָן קְיַים, בֵּין +מֵימְרִי+ וּבֵינֵי אַרְעָא.
וִיהֵי +כַּד אֶפְרוֹס+ עֲנָנֵי +יְקָרָא+ עֲלוֹי אַרְעָא, וְתִתְחַמֵי קַשְׁתָּא +בִּימָמָא, עַד לָא יִטְמַע שִׁמְשָׁא+ בַּעֲנָנָא.
וּדְכִירְנָא יַת קְיָימִי, דְבֵין +מֵימְרִי+ וּבֵינֵיכוֹן, וּבֵין כָּל נַפְשַׁת חַיְתָא, בְּכָל בִּיסְרָא; וְלָא יְהִי תּוּב מַיָא לְטוּבְעָנָא, _לְחַבָּלָא_ כָּל בִּישְרָא.
וּתְהִי קַשְׁתָּא בַּעֲנָנָא; וְאַחְמִינָהּ, לְמִדְכַּר קְיַים עֲלַם, בֵּין +מֵימְרָא+ דַיְיָ וּבֵין כָּל נַפְשַׁת חַיְיתָא, בְּכָל בִּסְרָא דְעַל אַרְעָא.
וַאֲמַר יְיָ לְנחַ: "דָא אָת קְיַים, דְקַיְימִית בֵּין +מֵימְרִי+, וּבֵין _מֵימַר_ כָּל בִּסְרָא דְעַל אַרְעָא".
וַהֲווֹ בְנֵי נחַ דִנְפָקוּ מִן תֵּיבוּתָא שֵׁם וְחָם וְיָפֶת וְחָם הוּא אֲבוֹי דִכְנָעַן
תְּלָתָא אִלֵין בְּנוֹי דְנחַ וּמֵאִילֵין אִתְבְּדָרוּ לְמֵיתַב בְּכָל אַרְעָא
וּשְׁרֵי נחַ לְמֶהֱוֵי גְבַר פְּלַח בְּאַרְעָא וְאַשְׁכַּח גוּפְנָא דְמוֹשְׁכֵיהּ נַהֲרָא מִן גִינוּנִיתָא דְעֵדֶן וְנַצְבֵיהּ לְכַרְמָא וּבֵיהּ בְּיוֹמָא אֲנֵיצַת וּבְשִׁילַת עִנְבִין וְעַצְרִינוּן
וְשָׁתֵי מִן חַמְרָא וְרָוֵי וְאִתְעַרְטַל בְּגוֹ מַשְׁכְּנֵיהּ
וַחֲמָא חָם אֲבוֹי דִכְנָעַן יַת עֶרְיְיתָא דַאֲבוֹי וְחַוֵי לִתְרֵי אֲחוֹהִי בְּשׁוֹקָא
וּנְסֵב שֵׁם וְיֶפֶת יַת אִסְטְלָא וְשַׁוִיוּ עַל כְּתַף תַּרְוֵיהוֹן וְאָזְלוּ מְאַחְזַרְיָן וְכַסִיאוּ יַת עִירְיְיתָא דַאֲבוּהוֹן וְאַפֵּיהוֹם מֵאַחְזְרִין וְעֶרְיְתָא דַאֲבוּהוֹן לָא חָמוּן
וְאִתְעַר נחַ מִן חַמְרֵיהּ וִידַע בְּאִשְׁתָּעוֹת חֶלְמָא יַת דְעָבַד לֵיהּ חָם בְּרֵיהּ דְהוּא קָלִיל בִּזְכוּתָא דְגָרַם לֵיהּ דְלָא יוֹלִיד בַּר רְבִיעָיֵי
וַאֲמַר לִיט כְּנָעַן דְהוּא בְּרֵיהּ רְבִיעָיֵי עֲבִיד מְשַׁעֲבֵּד יְהִי לְאָחוֹי
וַאֲמַר בְּרִיךְ יְיָ אֱלָהָא דְשֵׁם דַעֲבִידְתֵּיהּ צַדִיקָא וּבְגִין יֶהֱוֵי כְּנָעַן עֲבִיד לֵיהּ
יַשְׁפַּר יְיָ תְּחוּמֵיהּ דְיֶפֶת וְיִתְגַיְירוּן בְּנוֹי וְיִשְׁרוּן בְּמֶדְרָשָׁא דְשֵׁם וִיהֵי כְנָעַן עֲבִיד לְהוֹן
וַחֲיָא נחַ בָּתַר טוּבְעָנָא תְּלַת מְאָה וְחַמְשִׁין שְׁנִין
