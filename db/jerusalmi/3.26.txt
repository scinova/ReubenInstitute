וּדְמִשְׁתַּיֵיר מִנְכוֹן יִתְיַמְסוּן בְּחוֹבֵיהוֹן בְּאַרְעֲתָא דְסַנְאֵיכוֹן וְאוּף בְּחוֹבֵי אַבְהַתְהוֹן בִּישַׁיָא דַאֲחִידִין בִּידֵיהוֹן עִמְהוֹן יִתְיַמְסוּן
וִיוַדוּן בְּשַׁעַת אַנִיקְהוֹן יַת חוֹבֵיהוֹן וְיַת חוֹבֵי אַבְהַתְהוֹן בְּשִׁקְרֵיהוֹן דְשַׁקְרוּ בְּמֵימְרִי וְאוּף דְהַלִיכוּ עִמִי בְּעַרְאַי
בְּרַם אֲנָא אִידַבֵּר יַתְהוֹן בְּעַרַאי בְּעַלְמָא וְאָעֵיל יַתְהוֹן בְּגָלוּתָא בְּאַרַע בַּעֲלֵי דְבָבֵיהוֹן הָא בְּכֵן יִתְבַּר לִבְּהוֹן זְדָנָא וּבְכֵן יַרְעוּן יַת חוֹבֵיהוֹן
וְאִדְכַּר בְּרַחֲמִין יַת קְיָימָא דְקַיֵימִית עִם יַעֲקב בְּבֵית אֵל וְאוּף יַת קְיָימָא דְקַיֵימִית עִם יִצְחָק בְּטַוָור מוֹרִיָה וְאוּף יַת קְיָימָא דְקַיֵימִית עִם אַבְרָהָם בֵּינֵי פִּיסוּגַיָא אֶדְכּוֹר וְאַרְעָא דְיִשְרָאֵל אֶדְכּוֹר בְּרַחֲמִין
וְאַרְעָא תִּתְרְטֵישׁ וְתִשְׁתְּבֵיק מִנְהוֹן וְתַרְעֵי יַת שׁוֹבֵי שְׁמִיטָהָא כָּל יוֹמִין דְהִיא צַדְיָא מִנְכוֹן וְהִינוּן יַרְעוּן יַת חוֹבֵיהוֹן לְוָוטִין חֲלַף בִּירְכָן יִמְטוּן עֲלֵיהוֹן מֵיכְלָא כָּל קֳבֵיל מֵיכְלָא הוּא מְטוּל דְהִינוּן בְּסִידְרֵי דִינַי קָצוּ וְיַת קְיָימֵי אוֹרַיְיתִי רְחִיקַת נַפְשֵׁיהוֹן
וְאוּף עַל כָּל דָא אֲרַחִים יַתְהוֹן בְּמֵימְרִי כַּד יֶהֶוְויָין גַלְיָין בְּאַרַע בַּעֲלֵי דְבָבֵיהוֹן לָא אֶמְאָסִינוּן בְּמַלְכוּתָא דְבָבֶל וְלָא יִרְחַק מֵימְרִי יַתְהוֹן בְּמַלְכוּתָא דְמָדַי לִמְשֵׁיצַיָא יַתְהוֹן בְּמַלְכוּתָא דְיָוָן לְמִפְסוֹק קְיָימִי עִמְהוֹן בְּמַלְכוּתָא דֶאֱדוֹם אֲרוּם אֲנָא הוּא יְיָ אֱלָהָכוֹן בְּיוֹמוֹי דְגוֹג
וּדְכִירְנָא לְהוֹן קְיַים דְקַיֵימִית עִם אֲבָהַתְהוֹן קֳדָמַי בִּזְמַן דִפְרִיקִית וְאַפֵּיקִית יַתְהוֹן פְּרִיקִין מֵאַרְעָא דְמִצְרַיִם וְחַמְיַין כָּל עַמְמַיָא יַת כָּל גְּבוּרְתָּא דַעֲבָדִית לְהוֹן מְטוֹל לְמֶהֱוֵי לְהוֹן לֶאֱלָהָא אֲנָא יְיָ
אִלֵין קְיָימַיָא וְסִדְרֵי דִינַיָיא וּגְזֵירַת אוֹרַיְיתָא דִיהַב יְיָ בֵּין מֵימְרֵיהּ וּבֵין בְּנֵי יִשְרָאֵל בְּטַוְורָא דְסִינַי עַל יְדָא דְמשֶׁה
