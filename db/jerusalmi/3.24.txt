וּמַלֵיל יְיָ עִם משֶׁה לְמֵימָר
פַּקֵיד יַת בְּנֵי יִשְרָאֵל וְיִסְבוּן מִן דִילָךְ מְשַׁח זֵיתָא זַכִּיךְ כָּתִישׁ לְאַנְהָרָא לְאַדְלָקָא בּוֹצִינַיָא תְּדִירָא בְּיוֹמָא דְשַׁבְּתָּא וּבְיוֹמָא דְעוּבְדָא
מִבָּרָא לְפַרְגוֹדָא דְסַהֲדוּתָא לְעַלְמָא דִשְׁכִינְתָּא שַׁרְיָא בְּיִשְרָאֵל בְּמַשְׁכַּן זִמְנָא יְסַדֵר יָתֵיהּ אַהֲרן מֵרַמְשָׁא וְעַד צַפְרָא קֳדָם יְיָ תְּדִירָא קְיַים עֲלָם לְדָרֵיכוֹן
עַל מְנַרְתָּא דָכִיתָא יְסַדֵר יַת בּוֹצִינַיָא קֳדָם יְיָ תְּדִירָא
וְתִיסַב סְמִידָא וְתֵיפֵי יָתָהּ תְּרֵיסְרֵי גְרִיצָן שָׁוִוין לִתְלֵיסַר שִׁבְטַיָא תְּרֵי עֶשְרוֹנִין תְּהֵי גְרִיצְתָּא חֲדָא
וּתְסַדֵר יַתְהוֹן תַּרְתֵּין סִדוּרִין שִׁית בְּסִדְרָא חֲדָא וְשִׁית בְּסִדְרָא חֲדָא עַל פְּתוֹרָא בִּטְהֲרֵיהּ דִמְסַדֵר קֳדָם יְיָ
וְתִתֵּן עַל סִידוּרַיָא צְרִיד לְבוֹנְתָּא בְּרִירְתָּא וּתְהֵי לְלַחֲמָא לְאַדְכָּרָא קוּרְבָּנָא קֳדָם יְיָ
בְּיוֹמָא דְשַׁבְּתָּא בְּיוֹמָא דְשַׁבְּתָּא יְסַדְרִינֵיהּ חֲדַת קֳדָם יְיָ תְּדִירָא מִן בְּנֵי יִשְרָאֵל קְיַים עֲלָם
וּתְהֵי לְאַהֲרן וְלִבְנוֹי וְיֵכְלוּנֵיהּ מִן בָּתַר אִסְתַּלְקוּתֵיהּ מֵעַל פָּתוֹרָא בַּאֲתַר קַדִישׁ אֲרוּם קוֹדֶשׁ קוּדְשִׁין הוּא לֵיהּ מִקוּרְבָּנַיָא דַיְיָ קְיַים עֲלָם
גַבְרָא חַיָיבָא מְרוֹד בֶּאֱלָהָא שְׁמַיָא נְפַק מִמִצְרַיִם בַּר גַבְרָא מִצְרָאָה דְקָטַל גַבְרָא בַּר יִשְרָאֵל בְּמִצְרַיִם וְעָאַל עַל אִנְתְּתֵיהּ וְאִתְעַבָּרַת וִילֵידַת בַּר בְּגוֹ בְנֵי יִשְרָאֵל וְכַד הֲווֹן יִשְרָאֵל שַׁרְיָין בְּמַדְבְּרָא בָּעָא לְמִפְרוֹס מַשְׁכְּנֵיהּ בְּגוֹ שִׁבְטָא דִבְנֵי דָן וְלָא שַׁבְקוּהָ מִן בִּגְלַל דְטִיכְסִין דְיִשְרָאֵל גְבַר עַל טִיכְסֵיהּ בְּאַתְוָון לְיֵיחוּס אַבָהַתְהוֹן שַׁרְיָין וְאִתְקוֹטְטוּ כַּחֲדָא בְּמַשְׁרִיתָא וַאֲזָלוּ לְבֵי דִינָא בַּר אִתְּתָא בַּת יִשְרָאֵל וְגַבְרָא בַּר יִשְרָאֵל דְמִן שִׁבְטָא דְדָן
וְכַד נָפַק מִבֵּי דִינָא כַּד מְחַיֵיב פָּרֵישׁ וְחָרִיף בַּר אִתְּתָא בַּת יִשְרָאֵל יַת שְׁמָא רַבָּא וְיַקִירָא דְמִתְפָּרֵשׁ דְשָׁמַע בְּסִינַי וְאָזִיד וְאַרְגִיז וְשׁוּם אִמֵיהּ שְׁלוֹמִית בַּת דִבְרִי לְשִׁבְטָא דְדָן
דֵין הוּא חַד מִן אַרְבַּעְתֵּי דִינִין דִי עֲלוּ קֳדָם משֶׁה נְבִיָיא וְדָן יַתְהוֹן עַל פּוּם מֵימְרָא דִלְעֵיל מִנְהוֹן דִינָא מָמוֹנָא וּמִנְהוֹן דִינָא נַפְשָׁתָא בְּדִינֵי מָמוֹנָא הֲוָה משֶׁה זָרִיז וּבְדִינֵי נַפְשָׁתָא הֲוָה מָתוּן וּבְאִילֵין אָמַר משֶׁה לָא שְׁמָעִית מִן בִּגְלַל לְמֵילְפָא רֵישֵׁי סַנְהֶדְרַיְיתָא דְיִשְרָאֵל דַעֲתִידִין לְמֵיקוּם בַּתְרֵיהּ דֶיֶהֱווֹן זְרִיזִין בְּדִינֵי מָמוֹנָא וּמְתוּנִין בְּדִינֵי נַפְשָׁתָא וְלָא יְבַהֲתוּן לְמִישְׁאֲלָא דִינָא דְמִתְקַשֵׁי לְהוֹן אֲרוּם משֶׁה רַבְּהוֹן דְיִשְרָאֵל צָרִיךְ דְיֵימַר לָא שְׁמָעִית בְּגִין כֵּן אַצְנָעוּ יָתֵיהּ בְּבֵית מַטְרָא עַד זְמַן דְיִתְפָּרֵשׁ לְהוֹן עַל גְזֵירַת מֵימְרָא דַיְיָ
וּמַלֵיל יְיָ עִם משֶׁה לְמֵימָר
הַנְפֵּק יַת מַרְגְזָא לְמִבָּרָא לְמַשְׁרִיתָא וְיִסְמְכוּן כָּל סַהֲדַיָא דְשָׁמְעוּ בְּאַרְגָזוּתֵיהּ וְדַיְינַיָא יַת יְדֵיהוֹן עַל רֵישֵׁיהּ וִיאַטְלוּן יָתֵיהּ בְּאַבְנִין כָּל כְּנִישְׁתָּא
וְעִם בְּנֵי יִשְרָאֵל תְּמַלֵיל לְמֵימָר גְבַר טְלֵי אוֹ גְבַר סִיב דְיִרְגֵז וְיֶחֱרַף שׁוּם כִּינוּי אֱלָהֵיהּ וִיקַבֵּל חוֹבֵיהּ
בְּרַם מַאן דִמְפָרֵשׁ וּמְחַרֵף שְׁמָא דַיְיָ אִתְקְטָלָא יִתְקְטֵיל אַטְלָא יְאַטְלוּן יָתֵיהּ אַבְנִין כָּל כְּנִישְׁתָּא כְּגִיוֹרָא כְּיַצִיבָא בִּזְמַן דְיֶחֱרַף שְׁמָא דִמְיַיחֵד יִתְקְטֵיל
וּגְבַר אֲרוּם יַקְטֵיל כָּל נַפְשָׁתֵי דְבַר נְשָׁא מִבְּנֵי יִשְרָאֵל אִתְקְטָלָא יִתְקְטֵיל בְּסַיְיפָא
וּדְיִקְטוֹל נְפַשׁ בְּעִירָא יְשַׁלְמִינָהּ נַפְשָׁא חֲלַף נַפְשָׁא
וּגְבַר אֲרוּם יִתֵּן מוּמָא בְּחַבְרֵיהּ הֵיכְמָא דְעָבַד הֵיכְדֵין יִתְעֲבֵיד לֵיהּ
דְמֵי תַבְרָא חוֹלַף תַּבְרָא דְמֵי עֵינָא חוֹלַף עֵינָא דְמֵי שִׁינָא חוֹלַף שִׁינָא הֵיכְדֵין דְיִתֵּן מוּמָא בְּאֵינָשָׁא הֵיכְדֵין יִתְיְהֵב בֵּיהּ
וּמַאן דְיִקְטוֹל בְּעִירָא יְשַׁלְמִינָהּ וּמַאן דְיִקְטוֹל אֵינָשָׁא יִתְקְטֵיל
דִינָא חֲדָא יְהֵי לְכוֹן כְּגִיוֹרָא כְּיַצִיבָא יְהֵי אֲרוּם אֲנָא יְיָ אֱלָהָכוֹן
וּמַלֵיל משֶׁה עִם בְּנֵי יִשְרָאֵל וְאַפִּיקוּ יַת מַרְגְזָא לְמִבָּרָא לְמַשְׁרִיתָא וְאַטְלוּ יָתֵיהּ אַבְנִין וּבְנֵי יִשְרָאֵל עָבָדוּ לְמִסְמַךְ יְדֵיהוֹן וּלְמִדְחַף וּלְמִצְלַב וּלְמִקְבּוֹר הֵיכְמָא דְפַקֵיד יְיָ יַת משֶׁה
