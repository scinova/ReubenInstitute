וְנָטַר עֵשָו שְנָא בְּלִבֵּיהּ עַל יַעֲקב אָחוּי עַל סֵדֶר בִּרְכָתָא דִי בָּרְכֵיהּ אָבוֹי וַאֲמַר עֵשָו בְּלִבֵּיהּ לֵית אֲנָא עָבִיד הֵיכְמָא דְעָבַד קַיִן דְקָטַל יַת הֶבֶל בְּחַיֵי אָבוּי וְהָדָר אָבוּי וְאוֹלִיד יַת שֵׁת בְּרַם מִתְעַכֵּב אֲנָא עַד זְמַן דְיִמְטוּן יוֹמֵי אֶבְלָא דְמִיתַת אַבָּא וּבְכֵן אֲנָא קָטִיל יַת יַעֲקב אָחִי וַאֲנָא מִשְׁתַּכַּח קָטִיל וְיָרִית
וְאִתְחַוָה לְרִבְקָה בְּרוּחַ קוֹדֶש יַת פִּתְגָמֵי עֵשָו בְּרָהּ רַבָּה דְחָשִׁיב בְּלִבֵּיהּ לְמִקְטוֹל לְיַעֲקב וְשָׁדְרַת וְקָרְאַת לְיַעֲקב בְּרָהּ זְעֵירָא וַאֲמַרַת לֵיהּ הָא עֵשָו אָחוּךְ כָּמִין לָךְ כָּמָן וּמִתְיַעֵט עֲלָךְ לְמִקְטְלָךְ
וּכְדוּן בְּרִי קַבֵּיל מִינִי קוּם עֲרוֹק לָךְ לְנַפְשָׁךְ וְאִיזֵיל לְוַת לָבָן אָחִי לְחָרָן
וְתֵיתִיב עִמֵיהּ יוֹמִין קַלִילִין עַד דִי תִשְׁדוֹךְ רִיתְחָא דְאָחוּךְ
עַד דְנוּחַ רוּגְזָא דְאָחוּךְ מִינָךְ וְיִתְנְשֵׁי יַת מַה דְעָבַדְתָּ לֵיהּ וַאֲשַׁדֵר וְאִידְבְּרִינָךְ מִתַּמָן לָמָה אֶתְכַּל אוּף תַּרְוֵיכוֹן יוֹמָא חָד דְאַנְתְּ מִתְקְטֵיל וְהוּא מִטְרַד הֵיכְמָא דְאִתְכְּלַת חַוָה מִן הֶבֶל דְקַטְלֵיהּ קַיִן וְאִטְרִדָן תַּרְוֵיהוֹן מִן אַפֵּי אָדָם וְחַוָה כָּל יוֹמֵי חַיֵיהוֹם דְאָדָם וְחַוָה
וַאֲמָרַת רִבְקָה לְיִצְחָק אִיתְעָקִית בְּחַיַי מִן קֳדָם רְגוּז בְּנַת חֵת אִין נְסִיב יַעֲקב אִיתָא רְשִׁיעֲתָא מִבְּנַת חֵת כְּאִלֵין מִבְּנַתְהוֹן דְעַמָא דְאַרְעָא לָמָה לִי חַיִין
