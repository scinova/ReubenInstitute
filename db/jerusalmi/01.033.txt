וּזְקֵיף יַעֲקב יַת עֵינוֹי וַחֲמָא וְהָא עֵשָו אָתֵי וְעִמֵיהּ אַרְבַּע מְאָה גוּבְרִין פּוּלְמוּרְכִין וּפַלֵיג יַת בְּנַיָא עַל לֵאָה וְעַל רָחֵל וְעַל תַּרְתֵּין לְחֵינָתָא
וְשַׁוִי יַת לְחֵינָתָא הִינָן וּבְנֵיהֶן בְּקַדְמֵיתָא אֲרוּם אָמַר אִלוּ אָתֵי עֵשָו לְחַבְּלָא בְּרֻבַיָא לְמֶעֱבַד זְנוּ בִּנְשַׁיָא יַעֲבֵּיד בְּאִילֵן וּבְגוֹ פִּתְגָמָא הָדֵין נְקוּם וְנַגַח עִמֵיהּ קְרָבָא וְיַת לֵאָה וְרִיבָהָא בַּתְרָאִין וְיַת רָחֵל וְיַת יוֹסֵף בַּתְרָאִין
וְאִיהוּ עָבַר לְקָדָמֵיהוֹן מַצְלֵי וּבְעֵי רַחֲמִין מִן קֳדָם יְיָ וְגָחַן עַל אַרְעָא שְׁבַע זִמְנִין עַד מִיקְרְבֵיהּ עַד אָחוֹי
וּרְהַט עֵשָו לִקְדָמוּתֵיהּ וְגַפֵיף לֵיהּ וְאִתְרְכִין עַל צַוְורֵיהּ וּנְשֵׁיק לֵיהּ וּבְכוּן עֵשָו בָּכָא עַל צַעֲרָא דְשִׁינוֹי דְאִיתְמִזְמִיזוּ וְיַעֲקב בָּכָא עַל צַעֲרָא דְצַוָורֵיהּ
וּזְקַף יַת עֵינוֹי וַחֲמָא יַת נְשַׁיָא וְיַת רַבְיָא וְאָמַר מַאן אִלֵין לָךְ וְאָמַר בְּנַי הִינוּן דְאִתְיְהִיבוּ לִי בְּמֵיחוּס מִן קֳדָם יְיָ עַל עַבְדָךְ
וּקְרִיבַן לְחֵינָתָא הִינוּן וּבְנֵיהֶן וּגְחִינָן
וּקְרִיבַת אוּף לֵאָה וּבְנָהָא וּגְחִינָן וּבָתַר כֵּן קָרִיב יוֹסֵף וְאִתְעַתַּד קָמֵי רָחֵל וְכַסְיָיהּ בְּקוֹמָתֵהּ וּגְחִינָן
וַאֲמַר מַן לָךְ כָּל מַשְׁרֵי הָדֵין דְאַרְעִית וַאֲמַר דוֹרוֹן הוּא דְשַׁדְרִית לְאַשְׁכָּחָא רַחֲמִין בְּעֵינֵי רִבּוֹנִי
וַאֲמַר עֵשָו אִית לִי נִכְסִין סַגִיאִין אָחִי יִתְקַיֵים לָךְ מַאן דְאִית לָךְ
וַאֲמַר יַעֲקב לָא תֵימָא כֵּן בְּבָעוּ אִין כְּדוּן אַשְׁכָּחִית רַחֲמִין בְּעֵינָךְ וּתְקַבֵּל דוֹרוֹנִי מִן יְדִי אֲרוּם בְּגִין כֵּן חָמִית סְבַר אַפָּךְ וּדְמֵי לִי הֵי כְּמִיחְמֵי אַפֵּי מַלְאָכָא דִידָךְ דְהָא אִתְרְעִיתָא לִי
קַבֵּיל כְּדוֹן יַת דוֹרוֹן דְאִתַּתְיָא לָךְ אֲרוּם אִתְיְהֵב לִי בְּמֵיחוּס מִן קֳדָם יְיָ וַאֲרוּם אִית לִי נִכְסִין סַגִיאִין וּדְחַק בֵּיהּ וְקַבֵּיל
וַאֲמַר נְטַיֵיל וּנְהַךְ וְאָזִיל לְקִבְלָךְ עַד דְתִמְטֵי לְבֵי מְשׁוּרַיִיךְ
וַאֲמַר לֵיהּ רִבּוֹנִי יְדַע אֲרוּם טַלְיָיא חַטְיָן וְעָנָא וְתוֹרֵי דְמֵיַינְקַן עָלַי וְאִין דְחִיקְנָא לְהוֹם יוֹמָא חַד וִימוּתוּן כָּל עָנָא
יְעִיבַּר בְּבָעוּ רִבּוֹנִי וִיטַיֵיל קֳדָם עַבְדֵיהּ וַאֲנָא אִידְבַּר בִּנְיָיחָא לְבִלְחוֹדוֹי לְרֶגֶל עִיבִדְתָּא דְאִית קֳדָמָי וּלְרֶגֶל אוּלְפַן טַלְיָא עַד זְמַן דְאֵיתֵי לְוַת רִיבּוֹנִי לְגַבְלָא
וַאֲמַר עֵשָו אֶשְׁבּוֹק כְּדוּן עִמָךְ מִן פּוֹלְמוּסִין דְעִמִי וַאֲמַר לְמָא דְנַן אַשְׁכַּח רַחֲמִין קֳדָם רִבּוֹנִי
וְאִתְעֲבֵיד נִיסָא לְיַעֲקב וְתַב בְּיוֹמָא הַהוּא עֵשָו לְאָרְחֵיהּ לְגַבְלָא
וְיַעֲקב נָטַל לְסוּכּוֹת וְאִיתְעַכֵּב תַּמָן תְּרֵיסַר יַרְחֵי שַׁתָּא וּבְנָא לֵיהּ בֵּי מֶדְרָשָׁא וּלְגֵיתוֹי עָבַד מַטְלָן בְּגִין כֵּן קָרָא שְׁמָא דְאַתְרָא סוּכּוֹת
וְאָתָא יַעֲקב שָׁלֵם בְּכָל דִילֵיהּ לְקַרְתָּא דִשְׁכֶם דִי בְאַרְעָא דִכְנָעַן בְּמֵיתֵיהּ מִפַּדַן אֲרָם וּשְׁרָא לְקָבֵיל קַרְתָּא
וּזְבַן יַת אַחֲסָנַת חֲקַל דִפְרַס תַּמָן מַשְׁכְּנֵיהּ מִן יַד בְּנֵי חֲמוֹר אָבוּי דִשְׁכֶם בְּמֵאָה מַרְגַלְיָין
וְאָקִים תַּמָן מַדְבְּחָא וְתַמָן יְהַב מַעְשְרַיָא דְאַפְרֵישׁ מִן כָּל דִילֵיהּ קֳדָם אֵל אֱלָהָא דְיִשְרָאֵל
