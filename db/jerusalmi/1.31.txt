אִילוּלֵי פּוֹן אֱלָהֵיהּ דְאַבָּא אֱלָהֵיהּ דְאַבְרָהָם וּדְדָחִיל לֵיהּ יִצְחָק הֲוָה בְסַעֲדִי אֲרוּם כְּדוֹן רֵיקָם שְׁלַחְתַּנִי יַת סִיגוּפִי וְיַת לֵיאוֹת יְדַיי גְלֵי קֳדָם יְיָ בְּגִין כֵּן אוֹכַח בְּרַמְשָׁא
וְאָתֵיב לָבָן וַאֲמַר לְיַעֲקב בָּנָתָא דִי נְסֵיבַת לִנְשִׁין בְּרַתַּיַי הִינִין וּבְנַיָא דִילִידַן הֵי כִּבְנַי חֲשִׁיבִין וְעָנָא מִן עָנַיי הֲווֹ וְכָל דְאַנְתְּ חָמֵי מִן דִילִי הוּא וְלִבְרַתַּיַי מָה אֵיכוּל לְמֶעֱבַד לְאִילֵין יוֹמָא דֵין אוֹ לִבְנֵיהֶן דִילִידָן
וּכְדוֹן אִיתָא וְנִגְזַר קְיַם אֲנָא וְאַנְתְּ וִיהֵי לְסָהִיד בֵּינָא וּבֵינָךְ
וּנְסֵיב יַעֲקב אַבְנָא וְזָקְפָא לְקָמָא
וַאֲמַר יַעֲקב לִבְנוֹי דַהֲוָה קָרֵי לְהוֹם אָחוּי לְקוּטוּ אַבְנִין וְלַקִטוּ אַבְנִין וַעֲבָדוּ אוֹגַר וַאֲכָלוּ תַמָן עַל אוֹגְרָא
וּקְרָא לֵיהּ לָבָן אוֹגַר סָהִיד וְיַעֲקב קָרָא לֵיהּ בְּלִישַׁן בֵּית קוּדְשָׁא גַלְעֵד
וַאֲמַר לָבָן דָגוֹרָא הָדֵין סָהִיד בֵּינָא וּבֵנָךְ יוֹמָא דֵין עַל כֵּן קְרָא שְׁמֵיהּ גַלְעֵד
וּסְכוּתָא אִתְקְרִיאַת דִי אֲמַר יִסְתְּכִי יְיָ בֵּינָא וּבֵינָךְ אֲרוּם נִטְמַר גְבַר מִן חַבְרֵיהּ
אִין תַּסְגִיף יַת בְּרַתַּי לְמֶעֱבַד לָהֵן עוּלְבָּנָא וְאִם תִּיסַב עַל בְּרַתַּיַי לֵית אֵינַשׁ לְמִידַן יָתָן חָמֵי מֵימְרָא דַיְיָ סָהִיד בֵּינָא וּבֵינָךְ
וַאֲמַר לָבָן לְיַעֲקב הָא אוֹגַר הָדֵין וְהָא קָמָא דְאַקֵימִית בֵּינָא וּבֵינָךְ
סְהִיד אוֹגַר הָדֵין וּסְהִידָא קָמָא אִין אֲנָא לָא אֲעִיבַר לְוָתָךְ יַת אוֹגַר הָדֵין וְאִין אַנְתְּ לָא תְעִיבַר לְוָתִי יַת אוֹגַר הָדֵין וְיַת קָמָא הָדָא לְאַבְאָשָׁא
אֱלָהֵיהּ דְאַבְרָהָם וֶאֱלָהֵיהּ דְנָחוֹר יְדִינוּן בֵּינָנָא אֱלָהֵיהּ דַאֲבוּהוֹן וְקַיֵים יַעֲקב בֶּאֱלָהָא דְדָחֵיל לֵיהּ אָבוֹי יִצְחָק
וּנְכַס יַעֲקב נִכְסְתָא בְּטַוְורָא וְזַמֵן לְקָרִיבוֹי דְאָתוּ עִם לָבָן לְמִסְעוֹד לַחֲמָא וּסְעָדוּ וּבָתוּ בְּטַוְורָא
