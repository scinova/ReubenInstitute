וְטַלְיָא אַפְרֵישׁ יַעֲקב וִיהַב בְּרֵישׁ עָנָא מַשְׁכּוֹבִיתָא כָּל יַת דִרְגוֹל וְכָל דִלְחוּשׁ בְּעָנָא דְלָבָן וְשַׁוֵי לֵיהּ עֶדְרִין בִּלְחוֹדוֹי וְלָא עַרְבֵּיבִנוּן עִם עָנָא דְלָבָן
וַהֲוָה בְּכָל עִידַן דְמִתְיַחֲמָן עָנָא מְבַכְּרָתָא וּמְשַׁוִי יַעֲקב יַת חוּטְרַיָא לְעֵינֵי עָנָא בְּמוּרְכְיָיתָא לְיַחֲמוּתְהֶן קְבֵל חוּטְרַיָיא
וּבְלִקוּשֵׁי עָנָא לָא מְשַׁוִי וַהֲווֹן לַקִישַׁיָא לְלָבָן וּבַכִּירַיָיא לְיַעֲקב
וּתְקֵיף גַבְרָא לַחֲדָא לַחֲדָא וַהֲווֹ לֵיהּ עָאן סַגִיאִין וְאַמְהָן וְעַבְדִין וּגְמַלִין וְחַמְרִין
