ויהיו חיי שׁרה וַהֲווֹ חַיֵי שָרָה מְאָה וְעֶשְרִין וּשְׁבַע שְׁנִין שְׁנֵי חַיָיהָא דְשָרָה
וּמִיתַת שָרָה בְּקִרְיַית אַרְבַּע הוּא חֶבְרוֹן בְּאַרְעָא דִכְנָעַן וְאָתָא אַבְרָהָם מִן טַוָור פּוּלְחָנָא וְאַשְׁכְּחָהּ דְמִיתַת וְיָתִיב לְמִסְפַּד לְשָרָה וּלְמִבְכְּיָיהּ
וְקָם אַבְרָהָם מִן מַחֲמֵי אַפִּין עַל מֵיתֵיהּ וּמַלֵיל עִם בְּנֵי חִתָּאָה לְמֵימַר
דַיָיר וְתוֹתָב אֲנָא עִמְכוֹן בְּבָעוּ זַבּוּנִי לִי אַחֲסָנַת קְבוּרְתָּא עִמְכוֹן וְאִיקְבַּר יַת מֵיתִי תַּמָן
וְאָתִיבוּ בְּנֵי חִתָּאָה יַת אַבְרָהָם לְמֵימַר לֵיהּ
קַבֵּל מִנָנָא רוֹבָּנָנָא רַב קֳדָם יְיָ אַתְּ בֵּינָנָא בִּשְׁפַר קִיבְרָנָא קְבַר יַת מִיתָךְ אִינַשׁ מִינָנָא יַת קְבוּרְתֵּיהּ לָא יִמְנַע מִינָךְ בְּגִין דְלָא לְמִיקְבַּר מִיתָךְ
וְקָם אַבְרָהָם וְגָחַן לְעַמָא דְאַרְעָא לִבְנֵי חִתָּאָה
וּמַלֵיל עִמְהוֹן לְמֵימַר אִין אִית רְעוּ עִם נַפְשֵׁיכוֹן לְמִיקְבַּר יַת מֵיתִי מִן מֶחֱמֵי אַפַּיַי קַבְּלוּ מִינִי וּבְעוּ עָלַי קֳדָם עֶפְרוֹן בַּר צוֹחַר
וְיִזְבּוּן לִי יַת מְעָרַת כָּפֵילְתָּא דִילֵיהּ דְמִתְבַּנְיָא בִּסְטַר חַקְלֵיהּ בִּכְסַף שְׁלִים יִתְּנִינָהּ לִי בֵּינֵיכוֹן לְאַחְסָנַת קְבוּרְתָּא
וְעֶפְרוֹן יָתֵיב בְּגוֹ בְּנֵי חִיתָּאָה וְאָתֵיב עֶפְרוֹן חִיתָּאָה יַת אַבְרָהָם בְּאַנְפֵּי בְּנֵי חִיתָּאָה לְכָל עָלֵי תְרַע קַרְתֵּיהּ לְמֵימַר
בְּבָעוּ רִבּוֹנִי קַבֵּיל מִינִי חַקְלָא יְהָבִית לָךְ וּמְעַרְתָּא דְבֵיהּ לָךְ יְהַבְתָּהּ לְמַתְּנָא בְּאַנְפֵּי בְּנֵי עַמִי יְהַבְתָּהּ לָךְ אִיזֵל קְבַר מִיתָךְ
וְגָחַן לֵיהּ אַבְרָהָם בְּאַנְפֵּי בְּנֵי חִיתָּאָה
וּמַלֵיל עִם עֶפְרוֹן בְּאַנְפֵּי עַמָא דְאַרְעָא לְמֵימַר בְּרַם אִם אַנְתְּ צָבֵי לְמֶעֱבַד לִי טִיבוּ קַבֵּל מִינִי אֶתֵּן כַּסְפָּא דְמֵי חַקְלָא סַב מִנִי וְאֶקְבּוֹר יַת מִיתִי תַּמָן
וְאָתִיב עֶפְרוֹן יַת אַבְרָהָם לְמֵימַר לֵיהּ
רִבּוֹנִי קַבֵּל מִנִי אֲרַע דְטִימִין דִילֵהּ אַרְבַּע מְאָה סַלְעִין דִכְסַף בֵּינָא וּבֵינָךְ מַה הוּא וְיַת מִיתָךְ קְבַר
וְקַבֵּל אַבְרָהָם מִן עֶפְרוֹן וּתְקַל אַבְרָהָם לְעֶפְרוֹן יַת כַּסְפָּא דְמַלֵיל בְּאַנְפֵּיהּ בְּנֵי חִתָּאָה אַרְבַּע מְאָה סִלְעִין דִכְסַף טַב עָבְרִין בְּכָל פְּתוֹר וּמִתְקַבְּלִין בְּכָל פְּרַקְמַטְיָא
וְקָם זַבִּין חֲקַל עֶפְרוֹן דִי בְכָפֶלְתָּא דִקְדָם מַמְרֵא חַקְלָא וּמְעַרְתָּא דְבֵיהּ וְכָל אִילָנָא דִבְחַקְלָא דִבְכָל תְּחוּמֵיהּ חֲזוֹר חֲזוֹר
לְאַבְרָהָם לִזְבִינוֹי בְּאַנְפֵּי בְּנֵי חִתָּאָה לְכָל עָלֵי תְּרַע קַרְתֵּיהּ
וּמִן בָּתַר כְּדֵין קְבַר אַבְרָהָם יַת שָרָה אִנְתְּתֵיהּ לִמְעָרַת חֲקֵיל כָּפֶלְתָּא דְעַל אַנְפֵּי מַמְרֵא הוּא חֶבְרוֹן בְּאַרְעָא דִכְנָעַן
וְקָם חַקְלָא וּמְעָרְתָּא דְבֵיהּ לְאַבְרָהָם לְאַחֲסָנַת קְבוּרְתָּא מִן בְּנֵי חִיתָּאָה
