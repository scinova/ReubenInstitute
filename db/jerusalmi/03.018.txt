וּמַלֵיל יְיָ עִם משֶׁה לְמֵימָר
מַלֵיל עִם בְּנֵי יִשְרָאֵל וְתֵימַר לְהוֹן אֲנָא הוּא יְיָ אֱלָהָכוֹן
כְּעוֹבְדִין בִּישִׁין דְעַמָא אַרְעָא דְמִצְרַיִם דִי תֵיתְבוּן בָּהּ לָא תַעַבְדוּן וְהֵי כְּעוֹבְדִין בִּישִׁין דְעַמָא דְאַרְעָא דִכְנָעַן דַאֲנָא מָעִיל יַתְכוֹן לְתַמָן לָא תַעַבְדוּן וּבְנִימוּסֵיהוֹן לָא תְהָכוּן
יַת סִדְרֵי דִינַי תַעַבְדוּן וְיַת קְיָימַי תִיטְרוּן לְהַלָכָא בְּהוֹן אֲנָא יְיָ אֱלָהָכוֹן
וְתִטְרוּן יַת קְיָמַי וְיַת סִדְרֵי דִינַי דְאִין יַעֲבֵיד יַתְהוֹן אֵינָשָׁא וְיֵיחֵי בְּהוֹן בְּחַיֵי עַלְמָא וְחוּלְקֵיהּ עִם צַדִיקַיָא אֲנָא יְיָ
גְבַר טְלֵי וּגְבַר סִיב לְכָל קְרִיבַת בִּישְרֵיהּ לָא תְקַרְבוּן לְבַזָאָה עִרְיָיתָא בְּתַשְׁמִישְׁתָּא וּבְפִרְסוּמֵי עִרְיָיא אֲנָא יְיָ
עֶרְיַית אָבוּךְ וְעֶרְיַית אִמָךְ לָא תְבַזֵי אִיתָא לָא תְשַׁמֵשׁ עִם אַבְהָא וּגְבַר לָא יְשַׁמֵשׁ עִם אִמֵיהּ אִמָךְ הִיא לָא תְגַלֵי עִרְיָיתָהּ
עֶרְיַית אִתַּת אָבוּךְ לָא תִבְזֵי מְטוּל דְעִרְיְיתָא דְאָבוּךְ הִיא
עֶרְיַית אֲחָתָךְ בַּת אָבוּךְ אוֹ בַּת אִמָךְ מַה דְיוֹלִיד אַבָּךְ מִן אִיתָא אוֹחְרִי אוֹ מִן אִמָךְ אוֹ מַה דִילֵידַת אִמָךְ מִן אַבָּךְ אוֹ מִן גְבַר חוֹרַן לָא תִבְזֵי עֶרְיַתְהֶן
עֶרְיַית בְּרַת בְּנָךְ אוֹ בְּרַת בְּרַתָּךְ לָא תִבְזֵי עֶרְיַיתְהֶן אֲרוּם הֵי כְעִרְיָיתָךְ הִינוּן
עֶרְיַית בַּת אִתַּת אָבוּךְ דִילֵידַת מִן אַבָּךְ אַחְתָךְ הִיא לָא תִבְזֵי עִירְיְיתָהּ
עֶרְיַית אֲחַת אַבָּךְ לָא תִבְזֵי קְרֵיבַת בְּשַר אַבָּךְ הִיא
עֶרְיַית אֲחַת אִמָךְ לָא תִבְזֵי אֲרוּם קְרֵיבַת בְּשַר אִמָךְ הִיא
עֶרְיַית אַחְבוּךְ לָא תִבְזֵי וּלְוַת אִנְתְּתֵיהּ לָא תִקְרֵב בְּתַשְׁמִישׁ עַרְסָא אִיתַּת אַחְבוּךְ הִיא
עֶרְיַית כַּלָתָךְ לָא תִבְזֵי אִתַּת בְּרָךְ הִיא לָא תִבְזֵי עִירְיְיתָהּ
עֶרְיַית אִתַּת אָחוּךְ לָא תִבְזֵי בְּחַיֵי אָחוּךְ וּבָתַר מוֹתֵיהּ אִין אִית לֵיהּ בְּנִין עִירְיְיתָא דְאָחוּךְ הוּא
עֶרְיַית אִתָּא וּבְרַתָּהּ לָא תִבְזֵי יַת בְּרַת בְּרָהּ וְיַת בְּרַת בְּרַתָּהּ לָא תִיסַב לְבַזָאָה עִרְיְיתָהּ קְרִיבַת בִּישְרָא הִינוּן זְנוּ הוּא
וְאִתְּתָא בְּחַיֵי אָחָתָא לָא תִיסַב לְאַעֲקָא לָהּ לְבַזָאָה עֶרְיְיתָהּ עֲלָהּ כָּל יוֹמֵי חַיָיהָא
וּלְצֵיד אִתְּתָא בִּזְמַן רִיחוּק סְאוּבְתָּהּ לָא תִקְרַב לְבַזָּאָה עֶרְיְיתָהּ
וּלְצֵיד אִתַּת חַבְרָךְ לָא תִתֵּן תַּשְׁמִישָׁךְ לְזַרְעָא לְאִסְתָּאֲבָא בָהּ
וּמִן זַרְעָךְ לָא תִתֵּן בְּתַשְׁמִישְׁתָּה לְצֵיד בַּת עַמְמִין לְמָעְבְרָא לְפוּלְחָנָא אוֹחְרָא וְלָא תָפִיס יַת שְׁמָא דֶאֱלָהָךְ אֲנָא יְיָ
וְעִם דְכוּרֵי לָא תִשְׁכּוּב בְּתַשְׁמִישַׁיָא דְאִתְּתָא מְרַחְקָא הִיא
וּבְכָל בְּעִירָא לָא תִתֵּן תַּשְׁמִישָׁךְ לְאִסְתָּאֲבָא בָּהּ וְאִתְּתָא לָא תִקְרַב קֳדָם בְּעִירָא לְאִתְהַנְיָא מִינֵיהּ תִּבְלָא הוּא
לָא תִסְתַּאֲבוּן בַּחֲדָא מִכָּל אִילֵין אֲרוּם בְּכָל אִילֵין אִיסְתַּאֲבוּ עַמְמַיָא דַאֲנָא מַגְלֵי מִן קָדָמֵיכוֹן
וְאִסְתַּאֲבָא אַרְעָא וְאַסְעָרִית חוֹבָאָה עֲלָהּ וּפְלָטַת אַרְעָא יַת דַיְירָהָא
וְתִיטְרוּן אַתּוּן כְנִישְׁתָּא דְיִשְרָאֵל יַת קְיָימַי וְיַת סִדְרֵי דִינַי וְלָא תַעַבְדוּן חֲדָא מִכָּל תּוֹעֶבְיָא הָאִילֵין יַצִיבַיָא וְגִיוֹרַיָא דִי יִתְגַיְירוּן בֵּינֵיכוֹן
אֲרוּם יַת תּוֹעֵיבְתָּא הָאִילֵין עָבָדוּ אִינְשֵׁי אַרְעָא דִקְדָמֵיכוֹן וְאִסְתְּאָבַת אַרְעָא
וְלָא תִפְלוֹט אַרְעָא יַתְכוֹן בְּסָאוֹבְכוּן יָתָהּ הֵיכְמָא דִפְלָטַת יַת עַמָא דִקְדָמֵיכוֹן
אֲרוּם כָּל מַאן דְיֶעֱבַד חֲדָא מִכָּל תּוֹעֵיבְתָּא הָאִילֵין וְיִשְׁתֵּיצְיַין נַפְשָׁתָא דִיעַבְדוּן הֵכְדֵין מִגוֹ עַמְמֵיהֶן
וְתִטְרוּן יַת מַטְרַת מֵימְרִי לְאִזְדַהֲרָא מְטוֹל לְמֶעֱבַד מִנִימוּסֵי תּוֹעֵבְתָּא דְאִיתְעַבִידוּ בְּאַרְעָא קֳדָמֵיכוֹן וְלָא תִסְתַּאֲבוּן בְּהוֹן אֲנָא הוּא יְיָ אֱלָהָכוֹן
