וְתַעֲבֵיד קַרְנוֹהִי עַל אַרְבַּע זִיוְיָתֵיהּ מִנֵיהּ יֶהֶוְיַן קַרְנוֹי זְקִיפִין לְעֵיל וְתַחְפֵי יָתֵיהּ נְחָשָׁא
וְתַעֲבֵיד דּוּדְוָתֵיהּ לְמֵירְדָדָא יָתֵיהּ וּמִגְרִפַתֵּיהּ וּמִזִירִקוֹי וּמְשִׁילְיָיתֵיהּ וּמַחְתְּיָיתֵיהּ כָּל מָנוֹי תַּעֲבֵיד נְחָשָׁא
וְתַעֲבֵיד לֵיהּ קַנְקַל עוֹבַד מְצַרְתָּא דְנַחָשָׁא וְתַעֲבֵיד עַל מְצַרְתָּא אַרְבַּע עִזְקַן דִּנְחָשָׁא עַל אַרְבַּע צִיטְרוֹי
וְתִיתֵּן יָתָהּ תְּחוֹת סוֹבְבֵי מַדְבְּחָא מִלְרַע וּתְהֵי מְצַרְתָּא עַד פַּלְגוּת מַדְבְּחָא וְאִין נָפִיל גּוֹמָא אוֹ גוּמְרָא דְאֵשָׁא מֵעִילַוֵי מַדְבְּחָא נָפִיל עִילַוֵי קַנְקַל וְלָא יִתְמְטֵי לְאַרְעָא וְנַסְבִין יָתֵיהּ כַהֲנַיָא מֵעִילַוֵי קַנְקַל וּמְהַדְרִין יָתֵיהּ עַל מַדְבְּחָא
וְתַעֲבֵיד אֲרִיחִין דְּקֵיסֵי שִׁיטָא וְתַחְפֵּי יַתְהוֹן נְחָשָׁא

וְלִתְרַע דַּרְתָּא פְרָסָא עַשְרִין אַמִּין דְּתִיכְלָא וְאַרְגְּוָונָא וּצְבַע זְהוֹרֵי וּבוּץ שְׁזִיר עוֹבַד צִיּוּר מַחְטָא עַמּוּדֵיהוֹן אַרְבָּעָה וְחוּמְרֵיהוֹן אַרְבָּעָה
