אַנְתְּ אִיתְחָמֵיתָא יַת פְּרִישָׁתָא הָאִלֵּין מְטוּל לְמִנְדַע אֲרוּם יְיָ הוּא אֱלָהָא לֵית תּוּב בַּר מִינֵיהּ
מִשְׁמֵי מְרוֹמָא אַשְׁמָעִינְכוֹן יַת קַל מֵימְרֵיהּ לִמְכַסְנָא יַתְכוֹן בְּאוּלְפָנֵיהּ וְעִילַוֵי אַרְעָא אַחֲמִינְכוֹן יַת אֵישָׁתֵיהּ רַבְּתָא וּפִתְגָמוֹי שְׁמַעְתּוּן מִגּוֹ אֵישָׁתָא
וְחוֹלַף דִּרֵחֵים יַת אֲבָהַתְכוֹן אַבְרָהָם וְיִצְחָק וְאִתְרְעֵי בִּבְנוֹי דְיַעֲקב בַּתְרוֹי וְאַפְקִינְכוֹן בְּאַפֵּי רְעוּתֵיהּ בְּחֵילֵיהּ רַב מִמִּצְרָיִם
לְתַרְכָא עַמְמִין רַבְרְבִין וְתַקִּיפִין מִנְכוֹן מִן קֳדָמֵיכוֹן לְאַעֲלוּתְכוֹן לְמִתַּן לְכוֹן יַת אַרְעֲהוֹן אַחְסָנָא כְּיוֹמָא הָדֵין
וְתִנְדְּעוּן יוֹמָא דֵין וּתְתוּבוּן עַל לִבְּכוֹן אֲרוּם יְיָ אֱלהִים דִּשְׁכִינְתֵּיהּ שַׁרְיָא בִּשְׁמַיָא מִלְעֵילָא וְשַׁלִּיט עַל אַרְעָא מִלְרַע לֵית חוֹרָן בַּר מִנֵיהּ
וְתִנְטְרוּן יַת קְיָימוֹי וְיַת פִּקּוּדוֹי דַּאֲנָא מְפַקֵיד לְכוֹן יוֹמָא דֵין דְּיוֹטִיב לְכוֹן וְלִבְנֵיכוֹן בַּתְרֵיכוֹן וּמִן בִּגְלַל דְּתִינְגְּדוּן יוֹמָא דֵין עַל אַרְעָא דַיְיָ אֱלָהָכוֹן יָהֵיב לְכוֹן כָּל יוֹמַיָא
הָא בְּכֵן אַפְרֵשׁ משֶׁה תְּלַת קִרְוִין בְּעִבְרָא דְיוֹרְדְּנָא מַדְנַח שִׁמְשָׁא
לְמֵעֲרוֹק לְתַמָּן קָטוֹלָא דְיִקְטוֹל יַת חַבְרֵיהּ וְלָא אִתְכְּוַון וְהוּא לָא שָנֵי לֵיהּ מֵאִיתְמְלֵי וּמִדְקָמוֹהִי וְיֵעִירוֹק לַחֲדָא מִקִּרְוַויָא הָאִלֵין וְיִתְקַיָּים
יַת כְּוָתִירִין בְּמַדְבְּרָא בְּאַרַע מִישְׁרָא לְשֵׁבֶט רְאוּבֵן וְיַת רָמָתָא בְגִלְעָד לְשֵׁבֶט גָּד וְיַת דַּבְרָא בְּמַתְנָן לְשֵׁבֶט מְנַשֶּׁה
וְדָא אַחְוָיַת אוֹרַיְיתָא דִי סַדֵּר משֶׁה קֳדָם בְּנֵי יִשְרָאֵל
אִלֵין סָהִידְוָותָא וּקְיָמַיָא וְדִינַיָא דְמַלֵיל משֶׁה עִם בְּנֵי יִשְרָאֵל בִּזְמַן מִפַּקְהוֹן מִמִּצְרָיִם
וּתְנָנוּן משֶׁה בְּעִבְרָא דְיוֹרְדְּנָא בְּחֵילְתָא לְקָבֵיל בֵּית פְּעוֹר בְּאַרַע סִיחוֹן מַלְכָּא דֶאֱמוֹרָאֵי דְיָתֵיב בְּחוּשְׁבְּנָא דִמְחָא משֶׁה וּבְנֵי יִשְרָאֵל בְּמִפַּקְהוֹן מִמִּצְרָיִם
וְיָרִיתוּ יַת אַרְעֵיהּ וְיַת אַרְעָא דְעוֹג מַלְכָּא דְמַתְנָן תְּרֵין מַלְכֵי אֱמוֹרָאָה דִבְעִיבְרָא דְיוֹרְדְּנָא מַדְנַח שִׁמְשָׁא
מֵעֲרוֹעֵר דְּעַל גֵּיף נַחֲלֵי אַרְנוֹן וְעַד טַוְורָא דְסִיאוֹן הוּא טַוָור תַּלְגָא
וְכָל מֵישְׁרָא עִיבְרָא דְיוֹרְדְּנָא מַדִּינְחָא וְעַד יַמָא דְמֵישְׁרָא תְּחוֹת מַשְׁפַּךְ מְרָמָתָא
