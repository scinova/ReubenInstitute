וַאֲמַר פַּרְעה לְיוֹסֵף אֲנָא פַרְעה מַלְכָּא וְאַנְתְּ אַלְקַפְטָא וּבַר מִמֵימְרָךְ לָא יְרִים גְבַר יְדֵיהּ לְמִיסוֹר זַיְינֵיהּ וְיַת רִיגְלֵיהּ לְמִירְכְּוּב עַל סוּסְיָא בְּכָל אַרְעָא דְמִצְרַיִם
וּקְרָא פַרְעה שְׁמֵיהּ דְיוֹסֵף גַבְרָא דִטְמִירָן מְפַרְסֵם וִיהַב לֵיהּ יַת אָסְנַת דִילֵידַת דִינָה לִשְׁכֶם וּרְבֵיתָה אִיתַּת פּוֹטִיפֶרַע רַבָּא דְטָנִיס לְאִינְתּוּ וּנְפַק יוֹסֵף שַׁלִיט עַל אַרְעָא דְמִצְרַיִם
וְיוֹסֵף בַּר תְּלָתִין שְׁנִין כַּד קָם קֳדָם פַּרְעה מַלְכָּא דְמִצְרַיִם וּנְפַק יוֹסֵף מִלְוַת פַּרְעה וְעָבַר רַב וְשַׁלִיט בְּכָל אַרְעָא דְמִצְרַיִם
וְאַצְלָחַת אַרְעָא דְכָל תּוּבָלָא עָבַד מְלֵי תְּרֵין קוּמְצִין בְּשֶׁבַע שְׁנֵי שׁוּבְעָא עַד דְיִמְלוּן כָּל אוֹצְרַיָא
וּכְנַשׁ יַת כָּל עִיבּוּר שְׁבַע שְׁנֵי שׁוּבְעָא דַהֲווֹ בְּאַרְעָא דְמִצְרַיִם וִיהַב עִיבּוּרָא בְּקִרְוַיָא עִבּוּר חַקְלֵי קַרְתָּא דִבְחוֹזְרָנָהָא כְּנַשׁ בְּגַוָוהּ

וּלְיוֹסֵף אִתְיְלִידוּ תְּרֵין בְּנִין עַד לָא עָלַת שַׁתָּא דְכַפְנָא דִילִידַת לֵיהּ אָסְנַת דְרִיבַת בְּבֵית פּוֹטִיפֶרַע רַבָּא דְטָנִיס
וּקְרָא יוֹסֵף שׁוּם בּוּכְרָא מְנַשֶׁה אֲרוּם אַנְשְׁיַינִי יְיָ יַת כָּל לֵיאוּתִי וְיַת כָּל בֵּית אַבָּא
וְיַת שׁוּם תִּנְיַין קְרָא אֶפְרַיִם אֲרוּם אָמַר תַּקִיף יָתִי יְיָ בְּאַרַע סִיגוּפִי וְהֵיכְדֵין עָתִיד לְמִתְקוֹף בֵּית אַבָּא הָכָא בְּסִיגוּפֵיהוֹם
וּשְׁלִימָן שְׁבַע שְׁנֵי שׁוּבְעָא דְהַוָאָה בְּאַרְעָא דְמִצְרָיִם
וּשְׁרִיאָן שְׁבַע שְׁנֵי כַפְנָא לְמֵיתֵי הֵיכְמָא דַאֲמַר יוֹסֵף וַהֲוָה כַפְנָא בְּכָל אַרְעָתָא וּבְכָל אַרְעָא דְמִצְרַיִם הֲוָה לַחְמָא
וּכְפֵינַת כָּל אַרְעָא דְמִצְרַיִם דְלָא הֲוַת מַפְקָא בַּר זַרְעָא וּצְוַח עַמָא קֳדָם פַּרְעה בְּגִין לַחְמָא וַאֲמַר פַּרְעה לְכָל מִצְרָאֵי אִיזִילוּ לְוַת יוֹסֵף דְיֵימַר לְכוֹן תַּעֲבְדוּן
וְכַפְנָא הֲוַת עַל כָּל אַנְפֵּי אַרְעָא וּפְתַח יוֹסֵף יַת כָּל אוֹצְרִין דִבְהוֹן עִיבּוּרָא וְזַבִּין לְמִצְרָאֵי וּתְקֵף כַּפְנָא בְּאַרְעָא דְמִצְרָיִם
וְכָל דַיְירֵי אַרְעָא עָלוּ לְמִצְרַיִם לְמִיזְבַּן עִיבּוּרָא מִן יוֹסֵף אֲרוּם תַּקִיף כַּפְנָא בְּכָל אַרְעָא
