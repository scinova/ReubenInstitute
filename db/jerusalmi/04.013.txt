וידבר וּמלֵיל יְיָ עִם משֶׁה לְמֵימָר:
שְׁלַח לָךְ גוּבְרִין +חַרִיפִין+, וִיאַלְלוּן יַת אַרְעָא דִכְנַעַן דַאֲנָא יָהֵיב לִבְנֵי יִשְרָאֵל, גַבְרָא חָד גַבְרָא חָד לְשִׁבְטָא דְאַבְהָתוֹי תְּשַׁלְחוּן +מִן קֳדָם+ כָּל +אַמַרְכּוֹל+ דִבְהוֹן.
וְשַׁדַר יַתְהוֹן משֶׁה מִמַדְבְּרָא דְפָארָן עַל פּוּם מֵימְרָא דַיְיָ כֻּלְהוֹן גוּבְרִין חֲרִיפִין דְמִתְמְנָן רֵישֵׁי עַל בְּנֵי יִשְרָאֵל הִינוּן
וְאִלֵין שְׁמָהָתָא דִתְרֵיסַר גוּבְרַיָא מְאַלְלַיָא לְשִׁבְטָא דִרְאוּבֵן עַזְגָד שַׁמוּעַ בַּר זַכּוּר
לְשִׁבְטָא דְשִׁמְעוֹן עַזְגָד שָׁפָט בַּר חוֹרִי
לְשִׁבְטָא דִיהוּדָה עַזְגָד כָּלֵב בַּר יְפֻנֶה
לְשִׁבְטָא דְיִשָשכָר עַזְגָד יִגְאָל בַּר יוֹסֵף
לְשִׁבְטָא דְאֶפְרַיִם עַזְגָד הוֹשֵׁעַ בַּר נוּן
לְשִׁבְטָא דְבִנְיָמִין עַזְגָד פַּלְטִי בַּר רָפוּא
לְשִׁבְטָא דִזְבוּלֻן עַזְגָד גַדִיאֵל בַּר סוֹדִי
לְשִׁבְטָא דְיוֹסֵף לְשִׁבְטָא דִמְנַשֶׁה עַזְגָד גַדִי בַּר סוּסִי
לְשִׁבְטָא דְדָן עַזְגָד עַמִיאֵל בַּר גְמַלִי
לְשִׁבְטָא דְאָשֵׁר עַזְגָד סְתוּר בַּר מִיכָאֵל
לְשִׁבְטָא דְנַפְתָּלִי עַזְגָד נַחְבִּי בַּר וָפְסִי
לְשִׁבְטָא דְגָד עַזְגָד גְאוּאֵל בַּר מָכִי
אִלֵין שְׁמָהָן גוּבְרַיָא דְשָׁדַר משֶׁה לְאַלְלָא יַת אַרְעָא וּכְדִי חֲמָא משֶׁה עִנְוָותָנוּתֵיהּ קָרָא משֶׁה לְהוֹשֵׁעַ בַּר נוּן יְהוֹשֻׁעַ
וְשָׁדַר יַתְהוֹן משֶׁה לְאַלְלָא יַת אַרְעָא דִכְנָעַן וַאֲמַר לְהוֹן סָקוּ בַּהֲדֵין צִטְרָא בִּדְרוֹמָא וְתִסְקוּן לְטוּרָא
וְתֵיחֲמוּן יַת אַרְעָא מָה הִיא וְיַת עַמָא דְיָתֵיב עֲלָהּ הֲתַקִיף הוּא אִין חֲלָשׁ הַזְעֵיר הוּא אִין סַגִי
וּמָה אַרְעָא דְהוּא יָתֵיב בָּהּ הֲטַבְתָּא הִיא אִין בִּישְׁתָּא וּמָה קִרְוַיָא דְהוּא יָתֵיב בְּהוֹן הַבִכְּרַכֵּי פַּצִיחֵי חַנְיָן אִין בְּחַקְרִין
וּמָה שְׁבַח אַרְעָא הַשְׁמִינִין אִינוּן פִּרְיָהּ אִין פַּתְרָנִין הַאִית בָּהּ אִילָנִין דְמֵיכַל אִין לָא וְתַעַבְדוּן חֶזְקָתָא וְתִסְבּוּן מֵאִבָּא דְאַרְעָא וְיוֹמַיָא דִי אֲזָלוּ בְּעֶשְרִין וְתִשְׁעָה לְיַרְחָא דְסִיוָן יוֹמֵי דְמִן בִּכּוּרֵי עִנְבִין
וּסְלִיקוּ וְאַלִילוּ יַת אַרְעָא מִן מַדְבְּרָא דְצִין עַד פַּלְטִיוֹת מֵעַלָךְ לְאַנְטוּכְיָא
וּסְלִיקוּ מִן צְטַר דָרוֹמָא וְאָתוֹ עַד חֶבְרוֹן וְתַמָן אֲחִימָן שֵׁשַׁי וְתַלְמַי מַרְבְּיָנֵי דַעֲנָק גִבָּרָא וְחֶבְרוֹן שְׁבַע שְׁנִין אִתְבַּנְיַת קֳדָם עַד לָא תִתְבְּנֵי טָאנֵיס דְמִצְרַיִם
וְאָתוּ עַד נַחְלָא דְאַתְכְּלָא וְקָצוּ מִתַּמָן עוֹבַרְתָּא וְאִתְכַּל דְעַנְבִין חַד וְסוֹבְרוֹהִי בְּאַסְלָא בִּכְּתֵיף תְּרֵין מִנְהוֹן וְכֵן מִן רוּמָנַיָא וְכֵן מִן תֵּינַיָא
לְאַתְרָא הַהוּא קָרוּ נַחְלָא דְאַתְכְּלָא עַל עֵיסַק עוֹבַרְתָּא דְקָצִיצוּ מִתַּמָן בְנֵי יִשְרָאֵל וַהֲוָה חַמְרָא נָטֵיף מִנֵיהּ כְּנַחְלָא
וְתָבוּ מִלְאַלְלָא יַת אַרְעָא בִתְמַנְיָא יוֹמִין בְּיַרְחָא דְאָב מִסוֹף אַרְבְּעִין יוֹמִין
וַאֲזָלוּ וְאָתוּ לְוַת משֶׁה וּלְוַת אַהֲרן וּלְוַת כָּל כְּנִישְׁתָּא דְיִשְרָאֵל לְמַדְבְּרָא דְפָּארָן לִרְקַם וְאָתִיבוּ לְהוֹן פִּתְגָמָא וּלְוַת כָּל כְּנִישְׁתָּא וְחָמִיאוֹנוּן יַת אִבָּא דְאַרְעָא
וְאִשְׁתָּעִיוּ לֵיהּ וְאָמָרוּ אָתָאנָא לְאַרְעָא דְשָׁדַרְתָּנָא וְאוּף עַבְדָא חֲלַב וּדְבָשׁ הִיא וְדֵין אִבָּהּ
לְחוֹד אֲרוּם תַּקִיף עַמָא דְיָתֵיב בְּאַרְעָא וְקִירְוַיָא כְּרִיכִין חַנְיָין רַבְרְבָן לַחֲדָא וְאוּף מַרְבְּיָנֵי דַעֲנָק גִבָּרָא חֲמֵינָא תַמָן
עֲמַלְקָאֵי יַתְבִין בְּאַרַע דָרוֹמָא וְחִתָּאֵי וִיבוּסָאֵי וֶאֱמוֹרָאֵי יַתְבִין בְּטַוְורָא וּכְנַעֲנָאֵי יַתְבִין עַל יַמָא וְעַל תְּחוּם יוֹרְדְנָא
וּשְׁתִיק כָּלֵב יַת עַמָא וְאֵיצֵתִינוּן לְוַת משֶׁה וַאֲמַר מֵיסַק נֵיסַק וְנִירַת יָתָהּ אֲרוּם מֵיכַל נֵיכוֹל לָהּ
וְגוּבְרַיָא דְסַלִיקוּ עִמֵיהּ אָמְרוּ לָא נֵיכוֹל לְמֵיסוֹק לְוַת עַמָא אֲרוּם תַּקִיף הוּא מִנָנָא
וְאַפִּיקוּ טִיב בִּישׁ עַל אַרְעָא דְאַלִילוּ יָתָהּ לְוַת בְּנֵי יִשְרָאֵל לְמֵימָר אַרְעָא דְעָבַרְנוּ בָהּ לְאַלְלָא יָתָהּ אַרְעָא מְקַטְלָא יַתְבָהָא הִיא בְמַרְעִין וְכָל עַמָא דְגַוָהּ גַבְרַיָא מָרֵי מִיכְלַן בִּישִׁין
וְתַמָן חָמֵינָא יַת גַבְרַיָא בְנֵי עֲנָק מִגְנִיסַת גוּבְרַיָא וַהֲוֵינָא דַמְיַין בְּאַנְפֵּי נַפְשָׁתָנָא הֵי כְקַמְצָן וְהֵיכְדֵין הֲוֵינָא דַמְיַין בְּאַנְפֵּי נַפְשֵׁיהוֹן
