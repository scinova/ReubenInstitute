_מִן אַוְולָא_ בְּרָא יְיָ, יַת שְׁמַיָא וְיַת אַרְעָא.
וְאַרְעָא הֲוַת _תַּהֲיָיא וּבַהֲיָיא_, +צַדְיָא מִבְּנֵי נַשׁ, וְרֵיקַנְיָא מִן כָּל בְּעִיר,+ וַחֲשׁוֹכָא עַל אַפֵּי תְהוֹמָא, וְרוּחַ +רַחֲמִין מִן קֳדָם+ יְיָ _מְנַתְבָא_ עַל אַנְפֵּי מַיָא.
וַאֲמַר יְיָ: "יְהִי נְהוֹרָא, [לְאַנְהָרָא עָלְמָא|לְאַנְהָרָא עִלָּאָה]!" - וּ+מִן יַד +הֲוָה נְהוֹרָא.
וַחֲזָא יְיָ יַת נְהוֹרָא אֲרוּם טַב, וְ_אַפְרֵישׁ_ יְיָ בֵּין נְהוֹרָא וּבֵין חֲשׁוֹכָא.
וּקְרָא יְיָ לִנְהוֹרָא '_יְמָמָא_', +וְעַבְדֵהּ לְמִפְלַח בֵּיהּ דְּיְירֵי עַלְמָא,+ וְלַחֲשׁוֹכָא קָרָא 'לֵילְיָא'+, וְעַבְדֵיהּ לְמֵינַח בֵּיהּ בִּרְיָּיתָא+. וַהֲוָה רְמַשׁ, וַהֲוָה צְפַר - יוֹם חָד.
וַאֲמַר יְיָ: "יְהִי רְקִיעָא בִּ_מְצִיעוּת_ מַיָא! וִיהִי _מַפְרִישׁ_ בֵּינֵי מוֹי +עִלָאֵי,+ לְ+בֵינֵי+ מוֹי +תַּתָּאֵי+".
וַעֲבַד יְיָ יַת רְקִיעָא+, סוּמְכֵיהּ תְּלַת אֶצְבָּעָתָא - בֵינֵי סִטְרֵי שְׁמַיָא לְמוֹי דְאוֹקְיָינוֹס+; וְ_אַפְרֵשׁ_ בֵּינֵי מַיָא דְמִלְרַע לִרְקִיעָא, וּבֵינֵי מַיָא דִלְעֵיל +בְּקוּבְתָא+ דִרְקִיעָא - וַהֲוָה כֵן.
וּקְרָא יְיָ לִרְקִיעָא 'שְׁמַיָא'. וַהֲוָה רְמַשׁ, וַהֲוָה צְפַר - יוֹם תִּנְיָין.
וַאֲמַר יְיָ: "_יִתְכַּנְשׁוּן_ מַיָין +תַּתָּאִין דְאִשְׁתַּאֲרוּ+ מִן לְרַע לִשְׁמַיָא לְדוּכְתָּא חַד, +וְתִתְנַגַב אַרְעָא דְ+תִתְחַמֵי יַבֶּשְׁתָּא!" - וַהֲוָה כֵן.
וּקְרָא יְיָ לִנְגִיבְתָּא 'אַרְעָא', וּלְ_בֵית כְנִישַת_ מַיָין קָרָא 'יַמְמֵי'. וַחֲמָא יְיָ אֲרוּם טָב.
וַאֲמַר יְיָ: "_תַּרְבֵּי_ אַרְעָא דִיתְאֵי, עִישְבָּא _דְבִיזְרֵיהּ מִזְדְרָע_, וְאִילֵן פֵּירֵי עֲבִיד פֵּירֵי לִזְנֵיהּ, דְבִזְרֵיהּ בֵּיהּ עַל אַרְעָא!" - וַהֲוָה כֵן.
וְהַנְפָקַת אַרְעָא דִיתְאִין, עִיסְבָא _דְבִזְרֵיהּ מִזְדְרַע_, וְאִילָן עַבִיד פֵּירֵי, … לִזְנֵיהּ; וַחֲמָא יְיָ אֲרוּם טָב.
וַהֲוָה רְמַשׁ, וַהֲוָה צְפַר - יוֹם תְּלִיתָאֵי.
וַאֲמַר יְיָ: "יְהוֹן נְהוֹרִין בִּרְקִיעָא דִשְׁמַיָא, _לְמַפְרְשָׁא_ בֵּינֵי _יְמָמָא_ וּבֵינֵי לֵילְיָא. וִיהוֹן לְ_סִימָנִין_ וּלְ+זִמְנֵי+ מוֹעֲדִין, וּ+לְמִימְנֵי בְּהוֹן חוּשְׁבַּן+ יוֹמִין+, וּלְמַקְדְשָׁא רֵישֵׁי יַרְחִין+ וְ+רֵישֵׁי+ שְׁנִין+, עִבּוּרֵי יַרְחִין וְעִבּוּרֵי שְׁנִין, וּתְקוּפַת שִׁמְשָׁא וּמוֹלַד סִיהֲרָא, וּמַחֲזוֹרִין+,
וִיהוֹן לִנְהוֹרִין בִּרְקִיעָא דִשְׁמַיָא, לִמְנַהֲרָא עֲלוֹי אַרְעָא” - וַהֲוָה כֵן.
וַעֲבַד יְיָ יַת תְּרֵין נְהוֹרַיָא רַבְרְבַיָא+, וַהֲווֹן שָׁוִין בְּאִיקְרֵהוֹן עֶשְרִין וְחַד שָׁעִין, בָּצִיר מִנְהוֹן שִׁית מְאָה וּתְרֵין שׁוּבְעִין חוֹלְקֵי שַׁעֲתָּא, וּמִן בָּתַר כֵּן אִישְׁתָּעִית סִיהֲרָא עִילוֹי שִׁמְשָׁא לִישָׁן תְּלִיתָאֵי - וְאִזְדְעָרַת. וּמַנֵי+ יַת +שִׁמְשָׁא, +[דְּהוּא|דַהֲוָה] נְהוֹרָא רַבָּא - _לְמִישְׁלַט בִּימָמָא_, וְיַת +סִיהֲרָא,+ [דְּהוּא|דַהֲוָה] נְהוֹרָא זְעֵירָא - _לְמִשְׁלַט_ בְּלֵילְיָא, וְיַת כּוֹכְבַיָא.
וְ_סָדַר_ יַתְהוֹן יְיָ +עַל שִׁטְרֵיהוֹן+ בִּרְקִיעָא דִשְׁמַיָא, לִמְנַהֲרָא עַל אַרְעָא,
וּ_לְמִשַׁמְשָׁא_ בִּ_ימָמָא_ וּבְלֵילְיָא, וּ_לְמַפְרָשָׁא_ בֵּין נְהוֹרָא +דִימָמָא+ וּבֵין חֲשׁוֹךְ +לֵילְיָא+. וַחֲמָא יְיָ אֲרוּם טָב.
וַהֲוָה רְמַשׁ, וַהֲוָה צְפַר - יוֹם רְבִיעָאֵי.
וַאֲמַר יְיָ: "_יְרַחֲשׁוּן_ +רְקָקֵי+ מוֹי _רְחֵישׁ_, נַפְשָׁא חַיְיתָא! וְעוֹפָא _דְטַיֵיס_ +וּשַׁרְכַּפְתֵּיהּ+ עַל אַרְעָא, +וּשְבִיל טַיְיסֵיהּ+ עַל _אַוִיר_ רָקִיעַ שְׁמַיָא!"
וּבְרָא יְיָ יַת תַּנִינַיָא רַבְרְבַיָא+ - יַת לִוְיָתָן וּבַר זוּגֵיהּ, דְמִתְעַתְּדִין לְיוֹם נֶחֱמָתָא+, וְיַת כָּל נַפְשָׁתָא חַיְתָא דְ_רִיחֲשָׁא_ דְ_אַרְחִישׁוּ_ מַיָא +צְלִילְתָּא+ לִזְנֵיהוֹן+ - זְנֵי דַכְיָין, וּזְנֵי דְלָא דַכְיָין,+ וְיַת כָּל עוֹף +דְטַיֵיס בְּ+גַדְפִין לִזְנוֹהִי+ - זְנֵי דַכְיָין, וּזְנֵי דְלָא דַכְיָין+. וַחֲמָא יְיָ אֲרוּם טָב.
(חָסֵר תַּרְגּוּם לְפָסוּק זֶה)
וַהֲוָה רְמַשׁ, וַהֲוָה צְפַר - יוֹם חֲמִישָׁאֵי.
וַאֲמַר יְיָ: "תְּהַנְפֵּק +גַרְגִישְׁתָּא+ דְאַרְעָא נַפְשַׁת _בִּרְיָיתָא_ לִזְנֵיהּ+ - זְנֵי דַכְיָין, וּזְנֵי דְלָא דַכְיָין+, בְּעִירֵי וְרִחֲשֵׁי, _וּבַרְיַית_ אַרְעָא לִזְנֵיהּ!" - וַהֲוָה כֵן.
וַעַבַד יְיָ יַת חֵיוַת אַרְעָא לִזְנָהּ+ - זְנֵי דַכְיָין, וּזְנֵי דְלָא דַכְיָין+, וְיַת בְּעִירָא לִזְנָהּ, וְיַת כָּל רִחִיש אַרְעָא לִזְנָהּ. … .
וַאֲמַר יְיָ +לְמַלְאָכַיָא דִמְשַׁמְשִׁין קוֹמוֹי, דְאִיתְבְּרִי בְּיוֹם תִּנְיָן לִבְרִיַית עַלְמָא+: "נַעֲבֵיד אָדָם בְּצַלְמָנָא _בִּ_דְיוֹקְנָנָא! וְ_יִשְׁלְטוּן_ בְּכַוְורֵי יַמָא, וּבְעוֹפָא דְ+אַוִיר+ שְׁמַיָא, וּבִבְעִירָא וּבְכָל אַרְעָא, וּבְכָל _רִיחֲשָׁא_ _דִרְחֵישׁ_ עֲלוֹי אַרְעָא".
וּבְרָא יְיָ יַת אָדָם בִּ_דְיוּקְנֵיהּ_, בְּצַלְמָא יְיָ בְּרָא יָתֵיהּ+, בְּמָאתָן וְאַרְבְּעִין וּתְמַנֵי אֵיבָרִין, בִּתְלַת מְאָה וְשִׁיתִּין וְחַמְשָׁא גִידִין, וּקְרַם עֲלוֹי מוֹשְׁכָא, וּמְלֵי יָתֵיהּ בִּסְרָא וְאִידְמָא+; דְכַר וְנוּקְבָא +בִּגְוַנְהוֹן+ בְּרָא יַתְהוֹן.
וּבְרִיךְ יַתְהוֹן יְיָ, וַאֲמַר לְהוֹן יְיָ: "פּוּשׁוּ וּסְגוּ, וּמְלוּ יַת אַרְעָא +בְּנִין וּבְנָן+, וּתְקוּפוּ עֲלָהּ +בְּנִכְסִין+!" וּשְׁלוּטוּ בְּכַוְורֵי יַמָא, וּבְעוֹפָא דִשְׁמַיָא, וּבְכָל רִיחֲשָׁא חַיְיתָא דְרַחֲשָׁא עֲלוֹי אַרְעָא!"
וַאֲמַר יְיָ: “הָא יְהָבִית לְכוֹן יַת כָּל עִיסְבָּא דְבִיזְרֵיהּ מִזְדְרַע דְעִילוֹי אַנְפֵּי כָּל אַרְעָא, וְיַת כָּל אִילָנֵי +סְרָקָא, לִצְרוֹךְ בִּנְיָינָא וּלְאַסָקוּתָא, וּ+דְבֵיהּ פֵּירֵי אִלָנָא דְבִזְרֵיהּ מִזְדְרַע - לְכוֹן יְהֵי לְמֵיכָל,
וּלְכָל חֵיוַת אַרְעָא, וּלְכָל עוֹפָא דִשְׁמַיָא, וּלְכָל דְרָחִישׁ עַל אַרְעָא, דְבֵיהּ נַפְשָׁא חַיְיתָא, יַת כָּל יְרוֹקֵי עִסְבִּין, … ." - וַהֲוָה כֵן.
וַחֲזָא יְיָ יַת כָּל דְעָבַד, וַהֲוָה טַב לַחֲדָא. וַהֲוָה רְמַשׁ, וַהֲוָה צְפַר - יוֹם שְׁתִיתָאֵי.
