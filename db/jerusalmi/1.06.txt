וַהֲוָה אֲרֵי שְׁרִיאוּ +בְּנֵי+ _נְשָׁא_ לְמִסְגֵי עַל אַנְפֵּי אַרְעָא, וּבְנָתָא +שַׁפִּירְתָּא+ אִיתְיְלִידוּ לְהוֹן. 
וְחָמוֹן בְּנֵי _רַבְרְבַיָא_ יַת בְּנַת אֱנָשָׁא, אֲרוּם שַׁפִּירָן הִינוּן, +וְכַחֲלָן וּפַקְסָן, וּמְהַלְכִין בְּגִילוּי בִּשְׁרָא, וְהִרְהִירוּ לִזְנוּ,+ וּנְסִיבוּ לְהוֹן נְשִׁין, מִכָּל דְ_אִתְרְעִיוּ_.
וַאֲמַר יְיָ +בְּמֵימְרֵיהּ+: "לָא יִתְדְנוּן _כָּל דָרַיָא בִּשַׁיָא_+, דַעֲתִידִין לְמֵיקָם בְּסֵדֶר דַיָנַיָא דְדָרָא דְמַבּוּלָא, לְמוֹבְדָא וּלְמִשְׁתֵּיצֵי מִגוֹ עַלְמָא. הֲלָא יְהָבִית רוּחַ קָדְשִׁי בְּהוֹם, מִן בִּגְלַל דִי יַעֲבְדוּן עוֹבָדִין טָבִין, וְהָא אַבְאִישׁוּ עוֹבָדֵיהוֹן+. הָא _יְהָבִית לְהוֹן אַרְכָּא_ - מְאָה וְעַשְרִין שְׁנִין+, מִן בִּגְלַל דְיַעֲבְדוּן תְּתוּבָא, וְלָא עָבָדוּ+".
+שַׁמְחַזָאֵי וְעוּזִיאֵל הִינוּן+ נְפִילִין +מִן שְׁמַיָא וַ+הֲווֹ בְּאַרְעָא בְּיוֹמַיָא הָאִינוּן, וְאוּף בָּתַר כֵּן, דַעֲלוּן בְּנֵי _רַבְרְבַיָא_ לְוַת בְּנַת אֱנָשָׁא, וִילִידַן לְהוֹן; וְהִינוּן +מִתְקְרִין+ גַבְרִין דִמְעַלְמָא אִינְשֵׁי שְׁמָהָן.
וַחֲמָא יְיָ, אֲרוּם סְגִיאַת בִּישַׁת אֱנָשָׁא בְּאַרְעָא; וְכָל יִצְרָא דִמְחַשְׁבַת לִבֵּיהּ - לְחוֹד בִּישׁ, כָּל יוֹמָא.
וְ_תַב_ יְיָ +בְּמֵימְרֵיהּ+, אֲרוּם עֲבַד יַת אֶנָשָׁא בְּאַרְעָא; וְאֵידְיָין +עֲלֵיהוֹן בְּמֵימְרֵיהּ+.
וַאֲמַר יְיָ: "_אַבְטִיל_ אֱנָשָׁא +בְּמֵימְרִי+ דִבְרֵיתִי מֵעַל אַנְפֵּי אַרְעָא - מֵאֱנָשָׁא עַד בְּעִירָא, עַד רִיחֲשָׁא וְעַד עוֹפָא דִשְׁמַיָא; אַרוּם _תָּבִית בְּמֵימְרִי_ אֲרוּם עֲבַדְתִּינוּן".
וְנח+, דַהֲוָה צַדִיק+ - אַשְׁכַּח חִינָא קֳדָם יְיָ.
אִלֵין _יְחוּסִין_ דִגְנִיסַת נחַ: נחַ +הֲוָה+ גְבַר זַכְּאַי _שְׁלִים_ +בְּעוֹבְדִין טָבִין+ הֲוָה בְּדָרוֹהִי; _בְּדַחַלְתָּא דַיְיָ_ הֲלִיךְ נחַ.
וְאוֹלִיד נחַ תְּלָתָא בְּנִין: יַת שֵׁם, יַת חָם, וְיַת יָפֶת.
_וְאִתְחַבָּלַת_ אַרְעָא +בְּגִין דַיְירָהָא, דְסָטוּ מִן אָרְחָן דְתַקְנָן+ קֳדָם יְיָ, וְאִיתְמְלִיאַת אַרְעָא _חֲטוּפִין_.
וַחֲמָא יְיָ יַת אַרְעָא, וְהָא _אִתְחַבָּלַת_, אֲרוּם _חַבִּילוּ_ כָּל בִּשְרָא+, כָּל חַד וְחַד,+ יַת אוֹרְחֵיהּ עַל אַרְעָא.
וַאֲמַר יְיָ לְנחַ: _סוֹפָא_ דְכָל בִּשְרָא _מָטָא_ מִן קֳדָמַי, אֲרוּם אִתְמְלִיאַת אַרְעָא _חֲטוּפִין_ מִן +עוֹבְדֵיהוֹן בִּישַׁיָא+, וְהָא אֲנָא _מְחַבֵּלְהוֹן_ _עִם_ אַרְעָא.
עִיבַד לָךְ תֵיבוֹתָא דְקִיסִין קַדְרוּנִין, +מְאָה וְחַמְשִׁין+ _קוֹלִין_ תַּעֲבֵיד לְתֵיבוּתָא +בִּשְמָאלָהּ, וּתְלָתִין וְשִׁית בְּפוּתְיָהּ, וַעֲשָרָה בָּתִּין בִּמְצִיעָא, לְאַצְנְעָא בְּהוֹן מְזוֹנָא, וַחֲמֵשׁ אַפוּטְנִיוּתָא בִּימִינָא, וַחֲמֵשׁ בִּשְמָאלָהּ+, וְ_תֵשׁוּעַ_ יָתָהּ _מִן גֵיו_ וּמִבָּרָא בְּ_חֵימְרָא_.
(חסר תרגום לפסוק זה)
+אִיזֵל לְפִישׁוֹן, וְסַב מִתַּמָן+ יוֹהֲרָא+, וְ+_תִשְׁוִנֵיהּ_ בְּתֵיבוּתָא +לְאַנְהָרָא לְכוֹן+, וּלְגַרְמִידָא תְשֵׁצִינֵיהּ מִלְעֵיל וְתַרְעָא דְתֵיבוּתָא בְּצִטְרָא תְשַׁוֵי; +מְדוֹרִין+ תַּתָּאִין, תִּנְיָנִין וּתְלִיתָאִין תַּעַבְדִינֵהּ.
וַאֲנָא הָאֲנָא מַיְתֵי יַת טוּבְעָנָא מַיָא עַל אַרְעָא, _לְמִכְלָא_ כָּל בִּשְרָא דְבֵיהּ רוּחָא דְחַיֵי, מִן תְּחוֹת שְׁמַיָא; כָּל דִבְאַרְעָא _יִתְנְגַד_.
וְאָקִים יַת קְיָימִי עִמָךְ, וְ_תֵיעוֹל_ לְתֵיבוּתָא אַנְתְּ, וּבְנָךְ, וְאִנְתְּתָךְ, וּנְשֵׁי בְנָךְ עִמָּךְ.
וּמִן כָּל דְחַי, מִכָּל בִּשְר - תְּרֵין מִכּוֹלָא _תֵּיעוּל_ לְתֵיבוּתָא, לְקַיְימָא עִמָךְ, דְכַר וְנוּקְבָא יְהוֹן.
מֵעוֹפָא לִזְנֵיהּ, וּמִבְּעִירָה לִזְנֵיהּ, וּמִכָּל רִיחֲשָׁא דְאַרְעָא לִזְנֵיהּ - תְּרֵין מִכּוֹלָא _יַעֲלוּן_ לְוָותָךְ +עַל יַד מַלְאָכָא דְאָחַד, וּמֵיעַל יַתְהוֹן+ לָךְ לְקַיְימָא.
וְאַתְּ - סַב לָךְ מִכָּל מֵיכַל דְמִיתְאָכִיל, וִיהֵי לָךְ וּלְהוֹן לְמֵיכָל.
וַעֲבַד נחַ כְּכָל דְפַקְדֵיהּ יְיָ.
