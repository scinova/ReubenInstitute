וידבר וּמַלֵיל יְיָ עִם משֶׁה לְמֵימָר
מַלֵיל עִם אַהֲרן וְתֵימַר לֵיהּ בִּזְמַן אַדְלָקוּתָךְ יַת בּוֹצִינַיָא כָּל קֳבֵיל אַפֵּי מְנַרְתָּא יְהוֹן מְנַהֲרִין שִׁבְעָתֵי בּוֹצִינַיָא תְּלַת לִקֳבֵיל רוּחַ מַעַרְבָא וּתְלַת לִקֳבֵל רוּחַ מַדִינְחָא וּשְׁבִיעָא בִּמְצִיעָאָה
וַעֲבַד כֵּן אַהֲרן כָּל קֳבֵיל אַפֵּי מְנַרְתָּא אַדְלֵיק בּוֹצִינָהָא הֵיכְמָא דְפַקֵיד יְיָ יַת משֶׁה
וְדֵין עוֹבַד מְנַרְתָּא מִינָא קַשְׁיָא דִדְהַב עַד בָּסִיס דִידָהּ וְעַד שׁוֹשַׁנְיָיהָא עוֹבַד אוּמַן בְּקוּרְנְסָא הִיא מִתְעַבְדָא הֵי כְחֵיזוּ דְאַחֲזֵי יְיָ יַת משֶׁה הֵיכְדֵין עָבַד בְּצַלְאֵל יַת מְנַרְתָּא
וּמַלֵיל יְיָ עִם משֶׁה לְמֵימָר
קָרֵיב יַת לֵיוָאֵי מִגוֹ בְּנֵי יִשְרָאֵל וּתְדַכֵּי יַתְהוֹן
וּכְדֵין תַּעֲבֵיד לְהוֹן לְדַכּוֹאֵיהוֹן אֲדֵי עֲלֵיהוֹן מַיָא דְחַטָאתָא וְיַעַבְרוּן גְלַב עַל כָּל בִּשְרֵיהוֹן וִיחַוְורוּן לְבוּשֵׁיהוֹן וְיִדְכּוּן בְּאַרְבְּעִין סָוִוין דְמַיָא
וְיִסְבּוּן תּוֹר בַּר תּוֹרֵי וּמִנְחָתֵיהּ סוּלְתָּא פְּתִיכָא בִּמְשַׁח זֵיתָא וְתוֹר תִּנְיַין בַּר תּוֹרֵי תִּיסַב לְחַטָאתָא
וְתַקְרֵיב יַת לֵיוָאֵי קֳדָם מַשְׁכַּן זִמְנָא וְתִכְנוֹשׁ יַת כָּל כְּנִישְׁתָּא דִבְנֵי יִשְרָאֵל
וְתַקְרֵיב יַת לֵיוָאֵי קֳדָם יְיָ וְיִסְמְכוּן בְּנֵי יִשְרָאֵל יַת יְדֵיהוֹן עַל לֵיוָאֵי
וְיָרִים אַהֲרן יַת לֵיוָאֵי אֲרָמָא קֳדָם יְיָ מִן בְּנֵי יִשְרָאֵל וִיהוֹן לְמִפְלַח יַת פּוּלְחָנָא דַיְיָ
וְלֵיוָאֵי יִסְמְכוּן יַת יְדֵיהוֹן עַל רֵישׁ תּוֹרֵי וְעִיבַד יַת חַד חַטָאתָא וְיַת חַד עֲלָתָא קֳדָם יְיָ לְכַפָּרָא עַל לֵיוָאֵי
וּתְקֵים יַת לֵיוָאֵי קֳדָם אַהֲרן וְקָדָם בְּנוֹי וּתְרֵים יַתְהוֹן אֲרָמָא קֳדָם יְיָ
וְתַפְרֵישׁ יַת לֵיוָאֵי מִגוֹ בְּנֵי יִשְרָאֵל וִיהוֹן מְשַׁמְשִׁין קֳדָמַי לֵיוָאֵי
וּמִן בָּתַר כְּדֵין יֵיעָלוּן לֵיוָאֵי לְמִפְלַח פּוּלְחַן מַשְׁכַּן זִמְנָא וּתְדַכֵּי יַתְהוֹן וּתְרֵים יַתְהוֹן אֲרָמָא
אֲרוּם מַפְרְשָׁא מְפַרְשִׁין הִינוּן קֳדָמַי מִגוֹ בְּנֵי יִשְרָאֵל חוֹלַף פְּתַח כָּל וַלְדָא בּוּכְרַיָא כֻּלְהוֹן דְמִבְּנֵי יִשְרָאֵל קָרֵיבִית יַתְהוֹן קֳדָמַי
אֲרוּם דִילִי כָּל בּוּכְרָא בִּבְנֵי יִשְרָאֵל בְּאֵינָשָׁא וּבִבְעִירָא בְּיוֹמָא דִקְטָלִית כָּל בּוּכְרָא בְּאַרְעָא דְמִצְרַיִם אַקְדֵשִׁית יַתְהוֹן קֳדָמָי
וּקְרִיבַת יַת לֵיוָאֵי חוֹלַף כָּל בּוּכְרָא בִּבְנֵי יִשְרָאֵל
וִיהָבִית יַת לֵיוָאֵי יְהִיבִין לְאַהֲרן וְלִבְנוֹי מִגוֹ בְּנֵי יִשְרָאֵל לְמִפְלַח יַת פּוּלְחַן בְּנֵי יִשְרָאֵל בְּמַשְׁכַּן זִמְנָא וּלְכַפָּרָא עַל בְּנֵי יִשְרָאֵל וְלָא יְהֵי בִּבְנֵי יִשְרָאֵל מוֹתָא בִּזְמַן מִקְרַב בְּנֵי יִשְרָאֵל לְקוּדְשָׁא
וַעֲבַד משֶׁה וְאַהֲרן וְכָל כְּנִישְׁתָּא דִבְנֵי יִשְרָאֵל לְלֵיוָאֵי הֵיכְמָא דְפַקֵיד יְיָ יַת משֶׁה עַל לֵיוָאֵי הֵיכְדֵין עָבָדוּ לְהוֹן בְּנֵי יִשְרָאֵל
וְאִדְכִּיאוּ לֵיוָאֵי וְחַוָרוּ לְבוּשֵׁיהוֹן וַאֲרֵים אַהֲרן יַתְהוֹן אֲרָמָא קֳדָם יְיָ וְכַפֵּר עֲלֵיהוֹן אַהֲרן לְדַכּוֹאֵיהוֹן
וּמִבָּתַר כְּדֵין אָעֲלוּ לֵיוָאֵי לְמִפְלַח יַת פּוּלְחַנְהוֹן בְּמַשְׁכַּן זִמְנָא לִקֳדָם אַהֲרן וְלִקֳדָם בְּנוֹי הֵיכְמָא דְפַקֵיד יְיָ יַת משֶׁה עַל לֵיוָאֵי הֵיכְדֵין עָבָדוּ לְהוֹן
וּמַלֵיל יְיָ עִם משֶׁה לְמֵימָר
דָא אַחְוַויְיתָא דִי לְלֵיוָאֵי דְלָא מוּפְסָלִין בְּמוּמְהוֹן בְּרַם מִבַּר עֶשְרִין וְחָמֵשׁ שְׁנִין וּלְעֵילָא יֵיתֵי לְחַיְילָא חֵילָא בְּפוּלְחַן מַשְׁכַּן זִמְנָא
וּמִבַּר חַמְשִׁין שְׁנִין יֵיתוּב מֵחֵיל פּוּלְחָנָא וְלָא יִפְלַח תּוּב
וִישַׁמֵשׁ עִם אָחוֹי בְּמַשְׁכַּן זִמְנָא לְמִטַר מַטְרָא וּפוּלְחָנָא לָא יִפְלַח הֵיכְדֵין תַּעֲבֵיד לְלֵיוָאֵי בְּמַטְרַתְהוֹן
