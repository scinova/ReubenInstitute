וּמַלֵיל יְיָ עִם משֶׁה לְמֵימָר
עִיבַד לָךְ מִדִילָךְ תַּרְתֵּין חֲצוֹצְרַן דִכְסַף מִמִינָא קַשְׁיָא עוֹבַד אוּמַן תַּעֲבֵד יַתְהוֹן וִיהוֹן לָךְ לְעַרְעָא כְנִשְׁתָּא וּלְאַטְלָא יַת מַשִׁרְיָתָא
וְיִתְקְעוּן בְּהוֹן וְיִזְדַמְנוּן לְוָתָךְ כָּל כְּנִישְׁתָּא לִתְרַע מַשְׁכַּן זִמְנָא
וְאִין בְּחָדָא יִתְקְעוּן וְיִזְדַמְנוּן לְוָתָךְ רַבְרְבַיָא רֵישֵׁי אַלְפַיָא דְיִשְרָאֵל
וְתִתְקְעוּן יַבֶּבְתָּא וְיִטְלוּן מַשִׁרְיָתָא דְשָׁרַן קִידוּמָא
וְתִתְקְעוּן יַבֶּבְתָּא תִּנְיָינוּת וְיִטְלוּן מַשִׁרְיָתָא דְשָׁרַן דָרוֹמְתָּא יַבֶּבְתָּא יִתְקְעוּן לְמַטְלָנֵיהוֹן
וּבִזְמַן מַכְנַשׁ יַת קְהָלָא תִּתְקְעוּן וְלָא תְיַבְּבוּן
וּבְנֵי אַהֲרן כַּהֲנַיָא שְׁלֵימַיָא יִתְקְעוּן בַּחֲצוֹצְרָתָא וִיהוֹן לְכוֹן לִקְיַם עֲלָם לְדָרֵיכוֹן
וַאֲרוּם תְּעוּלוּן לִמְסַדְרָא סִדְרֵי קְרָבָא בְּאַרְעֲכוֹן עַל מְעִיקֵי דִמְעִיקִין לְכוֹן וּתְיַבְּבוּן בַּחֲצוֹצְרָתָא וְיֵיעוּל דוּכְרַנְכוֹן לְטָבָא קֳדָם יְיָ אֱלָהָכוֹן וְתִתְפָּרְקוּן מִבַּעֲלֵי דְבָבֵיכוֹן
וּבְיוֹם חֶדְוַותְכוֹן וּמוֹעֲדֵיכוֹן וּבְרֵישֵׁי יַרְחֵיכוֹן וְתִתְקְעוּן בַּחֲצוֹצְרָתָא עַל עֲלָוַותְכוֹן וְעַל נִכְסַת קוּדְשֵׁיכוֹן וִיהוֹן לְכוֹן לְדוּכְרָנָא טָבָא קֳדָם אֱלָהָכוֹן בְּרַם סַטָנָא מִתְעַרְבֵב לְקַל יַבְּבוּתְכוֹן אֲנָא הוּא יְיָ אֱלָהָכוֹן
וַהֲוָה בְּשַׁתָּא תִנְיֵיתָא בְּיַרְחָא תִּנְיֵיתָא הוּא יַרְחָא דְאִיָיר בְּעֶשְרִין לְיַרְחָא אִסְתַּלֵק עֲנַן יְקָרָא מֵעִילַוֵי מַשְׁכְּנָא דְסַהֲדוּתָא
וּנְטָלוּ בְּנֵי יִשְרָאֵל לְמַטְלָנֵיהוֹן מִמַדְבְּרָא דְסִינַי וּשְׁרָא עֲנַן יְקָרָא בְּמַדְבְּרָא דְפָּארָן
וּנְטָלוּ בְּקַדְמֵיתָא עַל פּוּם מֵימְרָא דַיְיָ בִּידָא דְמשֶׁה
וּנְטָלוּ טֵיקַס מַשִׁרְיַית בְּנֵי יְהוּדָה לְחֵילֵיהוֹן וְרַבָּא דְהַוָה מְמַנֵי עַל חֵילְוַות שִׁבְטָא דִבְנֵי יְהוּדָה נַחְשׁוֹן בַּר עַמִינָדָב
וְרַבָּא דְהַוָה מְמַנֵי עַל חֵילְוַות שִׁבְטָא דִבְנֵי יִשָשכָר נְתַנְאֵל בַּר צוּעָר
וְרַבָּא דְהַוָה מְמַנֵי עַל חֵילְוַות שִׁבְטָא דִבְנֵי זְבוּלֻן אֱלִיאָב בַּר חֵלוֹן
וּמִתְפְּרַק מַשְׁכְּנָא וְנַטְלִין בְּנֵי גֵרְשׁוֹן וּבְנֵי מְרָרִי נַטְלֵי מַשְׁכְּנָא
וּנְטַל טֵיקַס מַשִׁירְיַת רְאוּבֵן לְחֵילֵיהוֹן וְרַבָּא דְהַוָה מְמַנֵי עַל חֵילְוַות שִׁבְטָא דִבְנֵי רְאוּבֵן אֱלִיצוּר בַּר שְׁדֵיאוּר
וְרַבָּא דְהַוָה מְמַנֵי עַל חֵילְוַות שִׁבְטָא דִבְנֵי שִׁמְעוֹן שְׁלֻמִיאֵל בַּר צוּרִי שַׁדָי
וְרַבָּא דְהַוָה מְמַנֵי עַל חֵילְוַות שִׁבְטָא דִבְנֵי גָד אֶלְיָסָף בַּר דְעוּאֵל
וְנַטְלִין גְנִיסַת קְהָת נַטְלֵי מוּקְדְשָׁא וּמְקִימִין יַת מַשְׁכְּנָא עַד מֵיתֵיהוֹן
וּנְטַל טֵיקַס מַשִׁירְיַית בְּנֵי אֶפְרַיִם לְחֵילֵיהוֹן וְרַבָּא דְהַוָה מְמַנֵי עַל חֵילְוַות שִׁבְטָא דִבְנֵי אֶפְרַיִם אֱלִישָׁמָע בַּר עַמִיהוּד
וְרַבָּא דְהַוָה מְמַנֵי עַל חֵילְוַות שִׁבְטָא דִבְנֵי מְנַשֶׁה גַמְלִיאֵל בַּר פְּדָהצוּר
וְרַבָּא דְהַוָה מְמַנֵי עַל חֵילְוַות שִׁבְטָא דִבְנֵי בִּנְיָמִין אֲבִידָן בַּר גִדְעוֹנִי
וּנְטַל טֵיקַס מַשִׁרְיַית בְּנֵי דָן מַכְנִישׁ לְכָל מַשִׁרְיָיתָא לְחֵילֵיהוֹן וְרַבָּא דְהַוָה מְמַנֵי עַל חֵילְוַות שִׁבְטֵיהּ דִבְנֵי דָן אֲחִיעֶזֶר בַּר עַמִישַׁדָי
וְרַבָּא דְהַוָה מְמַנֵי עַל חֵילְוַות שִׁבְטָא דִבְנֵי אָשֵׁר פַּגְעִיאֵל בַּר עָכְרָן
וְרַבָּא דְהַוָה מְמַנֵי עַל חֵילְוַות שִׁבְטָא דִבְנֵי נַפְתָּלִי אֲחִירַע בַּר עֵינָן
אִלֵין מַטְלָנֵי בְּנֵי יִשְרָאֵל לְחֵילֵיהוֹן וְאִסְתַּלֵק עֲנַן יְקָרָא מֵעִילַוֵי מַשְׁכְּנָא וּנְטָלוּ
וַאֲמַר משֶׁה לְחוֹבָב בַּר רְעוּאֵל מִדְיָנָאָה חֲמוֹי דְמשֶׁה נַטְלִין אֲנַחְנָא מִיכָּא לְאַתְרָא דַאֲמַר יְיָ יָתֵיהּ אֶתֵּן לְכוֹן אִיתָא עִמָנָא וְנוֹטִיב לָךְ אֲרוּם יְיָ מַלֵיל לְאוֹטָבָא לְגִיוֹרָא עַל יִשְרָאֵל
וַאֲמַר לֵיהּ לָא אִיזֵיל אֱלָהֵין לְאַרְעִי וּלְיַלְדוּתִי אִיזֵיל
וַאֲמַר לָא כְדוּן תִּשְׁבּוֹק יָתָנָא אֲרוּם בְּגִין כֵּן דִידַעְתְּ כַּד הֲוֵינָן שְׁרַן בְּמַדְבְּרָא לְמִידַן וְאַלִיפַת לָנָא עֵיסַק דִינָא וַהֲוָיַת חָבִיב עֲלָן כְּבָבַת עֵינָנָא
וִיהֵי אֲרוּם תֵזִיל עִמָנָא וִיהֵי טָבָא הַהוּא דְיוֹטֵב יְיָ עִמָנָא וְנוֹטִיב לָךְ בְּפִלוּג אַרְעָא
וּנְטָלוּ מִטוּרָא דְאִתְגְלֵי עֲלוֹי אִיקַר שְׁכִינְתָּא דַיְיָ מַהֲלַךְ תְּלָתָא יוֹמִין וַאֲרוֹן קְיָימָא דַיְיָ מְטַיֵיל קֳדָמֵיהוֹן תְּלָתִין וְשִׁית מִלִין הֲלִיךְ בְּיוֹמָא הַהוּא וְהוּא הֲוָה מִידַבֵּר קֳדָם מַשִׁרְיָיתָא דְיִשְרָאֵל מַהֲלַךְ תְּלָתָא יוֹמִין לְאַתְקָנָא לְהוֹן אֲתַר בֵּית מֵשְׁרוֹי
וַעֲנַן אִיקַר שְׁכִינְתָּא דַיְיָ מְטַלֵיל עִילַוֵיהוֹן בִּימָמָא בְּמֵטַלְהוֹן מִן מַשְׁרִיתָא
וַהֲוָה כַּד הֲוָה בָּעֵי לְמֵיטַל אֲרוֹנָא הֲוָה עֲנָנָא מְקַפֵּל וְקָאִים וְלָא הֲוָה נָטִיל עַד דְמשֶׁה הֲוָה קָאֵי בִּצְלוֹ מַצְלֵי וּבָעֵי רַחֲמִין מִן קֳדָם יְיָ וְכֵן אָמַר אִתְגְלֵי כְּדוֹן מֵימְרָא דַיְיָ בִּתְקוֹף רוּגְזָךְ וְיִתְבַּדְרוּן בַּעֲלֵי דְבָבֵיהוֹן דְעַמָךְ וְלָא יֶהֱוֵי לְסַנְאֵיהוֹן רֵיגַל לְמֵיקוּם קֳדָמָךְ
וְכַד הֲוָה בָּעֵי לְמִשְׁרֵי אֲרוֹנָא הֲוָה עֲנָנָא מְקַפֵּל וְקָאִים וְלָא הֲוָה פָּרִיס עַד דְהַוָה משֶׁה קָאִים בִּצְלוֹ וּמַצְלֵי וּבָעֵי רַחֲמִין מִן קֳדָם יְיָ וְכֵן אָמַר תּוּב כְּדוֹן מֵימְרָא דַיְיָ בְּרַחֲמָךְ טַבְיָא וּדְבַר יַת עַמָךְ יִשְרָאֵל וְאַשְׁרֵי אִיקַר שְׁכִינָתָךְ בֵּינֵיהוֹן וְרַחֵם יַת רִבְוָותָא דְבֵית יַעֲקב מִנְיַן אַלְפַיָא דְיִשְרָאֵל
