לְדָוִד לְוָתָךְ יְיָ אֲנָא אֶקְרֵי תַּקִיפֵי לָא תִשְׁתּוֹק מִנִי דִלְמָא תִשְׁתּוֹק מִנִי וְאִתְמְתֵלִית עִם נָחֲתֵי גוּבָא:
קַבֵּל בְּקָל בָּעוּתִי בְּמִצַלְיִי לְוָתָךְ כַּד אֶפְרוֹשׁ יְדַי בִּצְלוֹ קֳדָם הֵיכַל קֻדְשָׁךְ:
לָא תִנְגְדִינַנִי עִם רַשִׁיעַיָא וְעִם עָבְדֵי עַוְלָא דִי מְמַלְלִין שְׁלָמָא עִם חַבְרֵיהוֹן וּבִישְׁתָא בְּלִבְּהוֹן:
הַב לְהוֹן כְּעוֹבָדֵיהוֹן וּכְבִישׁוּת עוֹבָדֵיהוֹן כְּעוֹבָדֵי יְדֵיהוֹן פְּרַע לְהוֹן אֲתִיב פּוּרְעֲנוּתְהוֹן לְהוֹן:
מְטוּל דְלָא מִתְבַּיְנִין לְאוֹרַיְיתָא דַייָ וּלְעוֹבָדֵי אַיְדוֹי יְפַגְרִנוּן וְלָא יִבְנִנוּן:
בְּרִיךְ יְיָ אֲרוּם קַבֵּל קַל צְלוֹתִי:
יְיָ עוּשְׁנִי וּתְרֵיסִי בֵּיהּ סַבָּר לִבִּי וְסַעֲדַנִי וְדָאֵיץ לִבִּי וּמִתּוּשְׁבַּחְתִּי אוֹדֶה קֳדָמוֹי:
יְיָ תְּקוֹף לְהוֹן וְעוּשְׁנָא פּוּרְקָנוּת מְשִׁיחֵיהּ הוּא:
פְּרוֹק יַת עַמָךְ וּבָרֵךְ יַת אַחְסַנְתָּךְ זוּן לְהוֹן וְסוֹבֵר יַתְהוֹן עַד עַלְמָא:
