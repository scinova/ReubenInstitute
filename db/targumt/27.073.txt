תּוּשְׁבַּחְתָּא עַל יְדֵי דְאָסָף בְּרַם טָב לְיִשְׂרָאֵל אֱלָהָא לִבְרִירֵי לִבָּא:
וַאֲנָא כִּזְעֵיר אִתְמוֹטַטָא רִגְלֵי כִּלְמָא דְאִזְדַעְזְעָא אִסְתַּוְרָי:
אֲרוּם טְנַנֵית בִּמְתַלְעַבַיָא בְּעִדָן דִי שְׁלָם רַשִׁיעַיָא אֶחֱמֵי:
אֲרוּם לֵיתֵיהוֹן תָּוְהִין וּמִצְטַעֲרִין מְטוּל יוֹם מִיתַתְהוֹן אִתְבַּעֲיָן מַחְשְׁבָתְהוֹן וּפַטִים וְאַלִים לִבְּהוֹן:
בְּלֵיעוּת גַּבְרִין דְמִתְעֲסְקִין בְּאוֹרַיְתָא לֵיתֵיהוֹן לָעִין וְעִם בְּנֵי נָשָׁא צַדִיקֵי דְסָכְלִין יְסוּרֵי לָא מִתְכַּתְּשִׁין:
לְכַךְ כֵּן עֲטַרְתִּנוּן גֵיוְתָנוּתָא כְּלִילָא דִמְשַׁוְיָן בְּרֵישֵׁיהוֹן הִיא מִן חֲטוֹף דִילְהוֹן:
שְׁנָא מִן שׁוּמְנָא פַּרְצוּפֵיהוֹן עֲבַרוּ חוֹטְבֵיהוֹן חֲפוּרַת לִבָּא:
יִתְמַקְמְקוּן מִן פִּטְמָא וִימַלְלוּן לְאַבְאָשָׁא וְלִטְלוּמָא מִן רוּם לִבְּהוֹן יְמַלְלוּן:
שַׁוִיאוּ בְקַדִישֵׁי שְׁמַיָא פּוּמְהוֹן וְלִישָׁנְהוֹן מְצַלְהֲבָה בְּקַדִישֵׁי אַרְעָא:
בְּכֵן תָּבוּ עַל עַמָיָא דַייָ לְאַלְקָיוּתְהוֹן וְיִמְחוֹנוּן בְּמַרְזוֹפִין וְדִמְעִין סַגִיעִין מָחֲתִין לְהוֹן:
וְיֵימְרוּן הֵכְדֵין יְדַע אֱלָהָא וְאִית מַנְדְעָא בְּעִלָאָה:
הָא אִלֵין רַשִׁיעַיָא דְיָתְבִין בְּשַׁלְוָתָא בְּעַלְמָא הָדֵין אַדְבִּיקוּ נִכְסִין קְנוּ חֵילָא:
בְּרַם לִסְרִיקוּתָא זַכְוַת לִבְבִי וְשֵׁזְבֵית בִּבְרִירוּ יְדָי:
וַהֲוֵיתִי כַתִּישׁ כָּל יוֹמָא וּמַכְסְנוּתִי לְצַפְרִירֵי:
אִין אֲמַרֵית אִשְׁתָּעִי דִכְמָתְהוֹן הָא עַל דָר בְּנָיִךְ אַרְשֵׁעִית:
וַחֲשִׁיבֵית לְמִדַע יַת דָא לְעוּת הוּא בְעֵינָי:
עַד זְמַן פּוּרְקָנָא דְאֵיתֵי לְוַת מַקְדְשֵׁי אֱלָהָא אֶתְבְּיַן לְסוֹפְהוֹן:
בְּרַם בְּקִבְלַיָא שַׁוִיתָא לְהוֹן דְמִיתִנוּן לְשַׁהְוָתָא:
הֵכְדֵין הֲווֹ לְצָדוֹ כְשַׁעְתָּא כָּלוּ אִשְׁתֵּיצוּן מִן אִתְרַגַשְׁיָן:
הֵיךְ חֶלְמָא מִן גְבַר דְמִתְּעָר יְיָ לְיוֹם דִינָא רַבָּא בְּאִתְּעַר יַתְהוֹן מִן בֵּית קְבוּרְתְּהוֹן בִּרְגַז דְמוּתְהוֹן תִּבְסַר:
אֲרוּם יְחוּשׁ לִבִּי וְכוּלְיָתִי בָּעֲרָן כְּאֶשָׁא:
וַאֲנָא שַׁטְיָא לָא אֵידָע הֵיךְ בְּעִירֵי אִתְחַשְׁבֵית גַבָּךְ:
וַאֲנָא תְּדִירָא עִמָךְ אוֹחַדְתָּא בְיַד יְמִינִי:
בְּמִלְכְּתָךְ תַּדְבְּרִנַנִי וּבָתַר דִשְׁלִים אַיְקָרָא דַאֲמַרְתָּא עֲלָי לְאַיְתָאָה תִּסְבִּנַנִי:
מַן כְּוָתָךְ דִילִי בִּשְׁמַיָא אֶלָא אַתְּ וְעִמָךְ חַבְרָא לָא צְבִיתִי בְּאַרְעָא:
אִשְׁתֵּיצָא גוּשְׁמִי וְלִבְּבִי תַקִיפָא דְבָחֵין לִבְבִי וְחוּלָקִי אֱלָהָא לְעַלְמָא:
אֲרוּם הָא רַשִׁיעַיָא דִי מִתְרַחֲקִין מִנָךְ יֹאבְדוּן שֵׁיצֵיתָא כָּל דְטָעֵי מִן בָּתַר דְחַלְתָּךְ:
וַאֲנָא לְמִקְרַב בַּיָי לִי טָב שַׁוְיֵת בַּיָי אֱלֹהִים רוּחְצָנִי לְאִשְׁתָּעַיָא לְצַדִיקַיָא כּוּלְהוֹן פִּקוּדֵי שְׁלִיחוּתָךְ:
