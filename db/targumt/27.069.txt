לְשַׁבָּחָא עַל גָלוּת סַנְהֶדְרִין עַל יַד דָוִד:
פְּרוֹק יָתִי אֱלָהָא אֲרוּם מָטוּ מַשְׁרְיָת חַיָבִין עַד דִי אֲעֵיקִין לִי הֵיךְ מַיָא דִמְטוֹ עַד נַפְשָׁא:
טְמַעֵית בְּגַלוּתָא הֵיךְ בְּמַיָא דִמְצוּלְתָּא וְלֵית אֲתַר לְמֵקוּם מְטֵית בְּעָמְקֵי תְּקוֹף סִיעַת רַשִׁיעַיָא וּנְבוּזַרְאֲדָן אַגְלֵי יָתִי:
אִשְׁתַּלְהֲיַת בְּמִקְרִי אִתְחַרְחַר עֻנְקִי כָּלוּ עֵינַי מִלְאוֹרְכָא לֵאלָהִי:
סַגִיעִין מִן סַעֲרָן דְרֵישִׁי סָנְאַי מַגָן תְּקַפוּ מַקְנִיטַי בַּעֲלֵי דְבָבַי סָהֲדֵי שִׁקְרָא דְלָא אַנְסֵית עַל סַהֲדוּתְכוֹן הַבְּכֵן אֲשַׁלֵם:
אֱלָהָא אַתְּ יְדַעְתָּא לְטַפְּשׁוּתִי וְסוּרְחָנוּתִי מִן קֳדָמָךְ לָא אִתְכַּסִיאוּ:
לָא יִבְהֲתוּן בִּי דְסָבְרִין עֲלָךְ לָא יִכְסְפוּן בִּי דְתָבְעִין אוּלְפָן מִנָךְ אֱלָהָא דְיִשְׂרָאֵל:
אֲרוּם אַמְטוּלְתָּךְ סוֹבָרִית חִסוּדָא חָפַת כִּסוּפָא אַפֵּי:
נוּכְרִי אִתְחַשִׁיבֵית לְאֶחָי וְהֵיךְ בַּר עַמִין לִבְנֵי אִמִי:
אֲרוּם טְנָנָא דְבֵית מַקְדְשָׁא גְמַרֵת יָתִי וְחִסוּדַי רַשִׁיעַיָא דִמְחַסְדִין לָךְ בְּמִשְׁתַּפְּהוֹן טַעֲוַתְהוֹן לְאַיקָרָךְ נְפַלוּ עֲלָי:
וּבְכֵית בְּצוֹמָא דְנַפְשִׁי וַהֲוָת חִסוּדְתִּי לִקְלָנָא לִי:
וִיהֲבֵית תְּחוֹת כְסוּתִי סַקָא וַהֲוֵית לְהוֹן לְמַתְלָא:
יְמַלְלוּן עֲלַי יָתְבֵי תַרְעָא בֵית קַרְנָתָא וְזִמְרֵיהוֹן דְאָזְלִין לְמִשְׁתֵּי מַרְוָת בְּבֵית קַרְקַסְוָן:
וַאֲנָא צְלוֹתִי קֳדָמָךְ יְיָ בְּעִדָן רַעֲוָת אֱלָהָא בִּסְגִיעֵי טוּבָךְ עֲנֵי יָתִי בִּקְשׁוֹט פּוּרְקָנָךְ:
פְּצֵי יָתִי מִן טִינָא וְלָא אֶטְמוֹעַ אִתְפְּצֵי מִן סָנְאַי דְדָמִין לְעוּמְקֵי מוֹי:
לָא יִגְלֵי יָתִי עַם תַּקִיף וְלָא יִסַלְעֵם לְמֶחְפֵּי עֲלַי מְצוּלְתָּא חָסִין וְלָא תִפְתַּח עֲלַי גֵהִנָם פּוּמָהּ:
עֲנֵי יָתִי יְיָ אֲרוּם טָב חַסְדָךְ הֵיךְ כִּסְגִיעוּת רַחֲמָךְ תִּסְתַּכֵּל לְוָתִי:
וְאַל תִּסְתַּלַק שְׁכִנְתָּךְ מֵעַבְדָךְ אֲרוּם עַיִק לִי סַרְהֵב עֲנֵי יָתִי:
קְרֵב לְוַת נַפְשִׁי פְּרוֹק יָתָהּ מְטוּל דְלָא אִתְרַבְרְבוּ עֲלַי בַּעֲלֵי דְבָבִי פְּרוֹק יָתִי:
אַתְּ יְדַעְתָּא קְלָנִי וּבַהֲתִי וּכְסוּפִי לְקִבְלָךְ קָיְימִין כָּל מְעִיקַי:
קְלָנָא תְּבַרַת לִבִּי וְהָא הִיא תַקִיפָא וְאַמְתֵּנִית לְמָרֵי מַסְפְּדַיָא וְלֵיתֵיהוֹן וּלְמָרֵי תַנְחוּמִין וְלָא אַשְׁכְּחֵית:
וְיַהֲבוּ בִסְעוּדְתִּי מְרֵירַת רֵישֵׁי חוּרְמָנִי וּלְצָחוּתִי אַשְׁקִיאוּ יָתִי חָלָא:
יְהֵי פָתוֹרְהוֹן דְסַדַרוּ קֳדָמֵי בְּאָכְלֵי קֳדָמֵיהוֹן לְקוּלָא וְנִכְסַתְהוֹן לְתַקָלָא:
יִתְחַשְׁכַן עֵינֵיהוֹן מִן לְמֶחֱמֵי וְחִרְצֵיהוֹן תְּדִירָא תְזַעְזַע:
שְׁדֵי עִלַוֵיהוֹן רוּגְזָךְ וּתְקוֹף רוּגְזָךְ יַדְבְּקִנוּן:
תְּהֵי מַשְׁכָּנוּתְהוֹן צַדְיָא בְּמַשְׁכְּנֵיהוֹן לָא יְהֵי יָתֵיב:
אֲרוּם דִי אַנְתְּ מְחִיתָא רְדַפוּ וּמִן דִכְאִיב עַל קְטוֹלָךְ יִשְׁתַּעְיָן:
הַב עַוְיָת עַל עַוְיַתְהוֹן וְלָא יִדְכּוּן לְמֵעַל בִּכְנִישַׁת צַדִיקָיִךְ:
יִתְמְחוּן מִסְפַּר דָכְרָן חַיַיָא וְעִם צַדִיקַיָא לָא יִתְכַּתְבוּן:
וַאֲנָא עַנְיָא וּכְאֵבָא פוּרְקָנָךְ אֱלָהָא תְשֵׁזְבִנָנִי:
אֲשַׁבַּח שׁוּם אֱלָהָא בְּשִׁירָא וַאֲרַבִּנֵיהּ בְּתוֹדָתָא:
וְתִשְׁפַּר צְלוֹתִי קֳדָם יְיָ מִן תּוֹר פַּטִים וּבְחִיר דְקָרֵיב אָדָם קֳדָמֵי דְקַדִימוּ קַרְנוֹי לְטִלְפוֹהִי:
חֲמוּן עִנְוְתָנַיָא וְיֶחְדוּן תָּבְעֵי אוּלְפָן מִן קֳדָם אֱלָהָא וִיחִי לִבְּכוֹן:
אֲרוּם מְקַבֵּל צְלוֹתְהוֹן דַחֲשִׁיכֵי יְיָ וְלַאֲסִירוֹי לָא בְסַר:
יְשַׁבְּחוּנֵיהּ מַלְאֲכֵי שְׁמַיָא וְיָתְבֵי אַרְעָא יַמַיָא וְכָל דְרָחֲשִׁין בְּהוֹן:
אֲרוּם אֱלָהָא יִפְרוֹק צִיוֹן וִישַׁכְלֵל קִרְוִין דִיהוּדָה וִיתוּבוּן תָּמָן וְיֵרְתוּנָהּ:
וּבְנֵי עַבְדוֹי יַחְסְנוּן יָתָהּ וְרָחֳמֵי שְׁמֵיהּ יִשְׁרוֹן בְּגַוָהּ:
