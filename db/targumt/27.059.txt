לְשַׁבָּחָא עַל עָקְתָא בִּזְמַן דִי אֲמַר דָוִד לָא תְחַבֵּל עַל יַד דָוִד מַכִּיךְ וּשְׁלִים כַּד שְׁדַר שָׁאוּל וּנְטַרוּ יַת בֵּיתָא לְמִקְטְלֵיהּ:
פְּצֵי יָתִי מִבַּעֲלֵי דְבָבִי אֱלָהִי מִן קָיְמִין עֲלַי תְּשֵׁזְבִנַנִי:
פְּצֵי יְתִי מֵעָבְדֵי שְׁקַר מִן גוּבְרִין קֲטוֹלִין פְּרוֹק יָתִי:
אֲרוּם הָא כְּמַנוּ לְנַפְשִׁי מִתְכַּנְשִׁין עֲלַי עֲשִׁינִין לָא עַל סוּרְחָנִי וְלָא עַל חוֹבִי יְיָ:
עַל לָא עַוְיָן רָהֲטִין וּמְסַדְרִין קְרָבָא אִתְגַבֵּר לְקַדְמוּתָא וַחֲמֵי:
וְאַנְתְּ יְיָ אֱלֹהִים צְבָאוֹת אֱלָהָא דְיִשְׂרָאֵל אִתְעָר לְמִסְעָר עַל כָּל עַמְמַיָא לָא תְחוֹס כָּל שַׁלִיטֵי שְׁקַר לְעָלְמִין:
יְתוּבוּן לְרַמְשָׁא מִתְרַגְשִׁין הֵיךְ כִּכְלַב וְיַחְזְרוּן קַרְתָּא:
הָא יַבִּיעוּן בְּפוּמְהוֹן מִלַיָא דִשְׁנִינָן הֵיךְ סַיְפָא בְּסִפְוָתְהוֹן אָמְרִין נִתְגַבֵּר אֲרוּם מַן הוּא דִי שְׁמַע וְיִתְפְּרַע:
וְאַנְתְּ יְיָ תְּגַחֵךְ לְהוֹן תִּתְלָעֵב לְכָל עַמְמַיָא:
עוּשְׁנוֹי לָךְ אֶנְטוֹר אֲרוּם אֱלָהָא שֵׁזְבוּתִי:
אֱלָהָא טוּבִי יְקַדְמִנַנִי אֱלָהָנָא יַחְמִנַנִי נִקְמָתָא בִמְעִיקָי:
לָא תִקְטְלִינוּן מִן יַד דִלְמָא יִתְנְשׁוּן עַמִי טַלְטֵל יַתְהוֹן מִן בָּתֵּיהוֹן בְּתוּקְפָךְ וּתְמַסְכֵּן יַתְהוֹן מִן מָמוֹנֵיהוֹן תְּרִיסָנָא יְיָ:
מְטוּל חוֹבַת פּוּמְהוֹן וּמַמְלַל סִפְוָתְהוֹן יִתְאַחְדָן בְּגֵיוְתָנוּתְהוֹן אֲרוּם מִן מוֹמָתָא וּמִן שִׁקְרָא יִשְׁתַּעֲיָן:
שֵׁצֵי יַתְהוֹן בְּכָלוּ שֵׁצֵי יַתְהוֹן עַד דִי לֵיתִנוּן וְיִנְדְעוּן אֲרוּם אֱלָהָא שַׁלִיט בְּיַעֲקֹב לִסְיָפֵי אַרְעָא לְעָלְמָא:
וִיתוּבוּן לְרַמְשָׁא יִתְרַגְשׁוּן הֵיךְ כִּכְלַב וְיַחְזְרוּן קַרְתָּא:
הִנוּן יְטַלְטְלוּן לְמִבַּז בִּזְתָא לְמֵיכָל וְלָא יְנִיחוּן עַד דִי יִסְבְּעוּן וִיבִיתוּן:
וַאֲנָא אַשַׁבַּח עוּשְׁנָךְ וַאֲבוּעַ לְעִדַן צַפְרָא טוּבָךְ אֲרוּם הֲוֵיתָא מְשֵׁזִיב לִי וְרוֹחֲצָנִי בְּיוֹמָא דְעַיִק לִי:
עוּשְׁנִי לְוָתָךְ אֲשַׁבַּח אֲרוּם אֱלָהָא שֵׁזְבוּתִי אֱלָהָא טוּבִי:
