לְשַׁבָּחָא עַל עַתִּיק סַהֲדוּתָא דִי בְנֵי יַעֲקֹב וְלָבָן פַּרְשְׁגֶן עַל יַד דָוִד לְאַלָפָא:
כַּד כְּנַשׁ דָוִד מַשִׁרְיָן וַעֲבַר עַל אַיְגַר סַהֲדוּתָא וַאֲגִיחַ עִם אֲרַם דִי עִם צוֹבָא וּמִן בָּתַר כֵּן תָּב יוֹאָב וּמְחָא יַת אֱדוֹמָאֵי בְּמִישׁוֹר מִלְחָא וּנְפַלוּ מִן חֵילֵיהוֹן דְדָוִד וְיוֹאָב תְּרֵיסַר אַלְפִין:
אֲמַר דָוִד אֱלָהָא שְׁבַקְתָּנָא תְּקֵפְתָּא עֲלָנָא תְּקוֹף רוּגְזָא תּוּב בִּיקָרָךְ לָנָא:
אַרְגֵישְׁתָּא אַרְעָא דְיִשְׂרָאֵל זַעֲזַעְתָּא וּסְדֵקְתָּא יָתָהּ אַסֵי תַבְרְיָהָא אֲרוּם אִתְמוֹטַטַת:
אַחְמֵיתָא עַמָךְ קַשְׁיָא אַשְׁקֵיתָא לָנָא חֲמַר דִלְוָט:
יַהַבְתָּא לְדָחֲלָיִךְ נִסָא לְאִתְנַסָאָהּ בֵּיהּ מִן בִּגְלַל קוּשְׁטֵיהּ דְאַבְרָהָם לְעָלְמִין:
מִן בִּגְלַל זְכוּתֵיהּ דְיִצְחָק יִתְפַּצְיָן רְחִימָיִךְ פְּרוֹק מְטוּל חֲסִידוּתָא דְיַעֲקֹב וְקַבֵּל צְלוֹתִי:
אֱלָהָא מַמְלֵל בְּבֵית מַקְדְשֵׁיהּ אַבּוּעַ אֲרוּם יִתְגַבְּרוּן דְבֵית יִשְׂרָאֵל אֲפַלֵיג בִּזְתָא עִם בְּנֵי יוֹסֵף דְיָתְבִין בִּשְׁכֶם וּבְמֵישַׁר סוּכּוֹת אֶמְשַׁח מִישְׁחֲתָא וַאֲפַלֵג עֲדָאָה:
עִמִי הֲווּן דְבֵית גִלְעָד וְעִמִי הֲווּן דְבֵית מְנַשֶׁה וְגִיבָּרִין דְבֵית אֶפְרַיִם עוֹשְׁנָא דְרֵישִׁי וּמִדְבֵּית יְהוּדָה סִפְרַיָא דְבֵית אוּלְפָנִי:
בְּעֶטְיַת מוֹאֲבָאֵי אִטַמָשׁוּ רִגְלַי בַּאֲרַם גִּבָּרֵיהוֹן הֵיךְ דוּד שְׁזוּנִי עַל פִּרְקַת קְדַל גִבָּרֵי אֱדוֹם טִלְקֵית מְסָנַי עִלַוֵי פְלִשְׁתָּאֵי יַבִּיבוּ כְּנִשְׁתָּא דְיִשְׂרָאֵל:
מַן הוּא דְיוֹבִיל יָתִי לְקַרְתָּא חֲרוֹבָא דְצוֹר מַן הוּא דְדַבְּרַנִי עַד אֱדוֹם:
הֲלָא אַתְּ הוּא יְיָ שְׁבַקְתָּנָא וְלָא תִפּוֹק אֱלָהָא עִם חֵילְוָתָנָא:
הַב לָן סַעְדָא מִמְעִיקָא וּמַגָן הוּא פּוּרְקָנָא דְבַר נָשָׁא:
בְּמֵימְרָא דַייָ נַעְבֵּד חֵילָא וְהוּא יִכְבּוֹשׁ מְעִיקָנָא:
