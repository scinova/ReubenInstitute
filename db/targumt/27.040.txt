לְשַׁבָּחָא לְדָוִד תּוּשְׁבַּחְתָּא:
מְסַבָּרָא סַבְּרֵית יְיָ וּצְלָא לְוָתִי וְקַבֵּיל בָּעוּתִי:
וְאַסַקְנִי מִגוֹב אִתְרְגוּשְׁתָּא מִסְיֵין טִשְׁטוּשָׁא וַאֲקֵים עַל כֵּיפָא רִגְלַי תַּקִין אַתוּרָי:
וְסַדַר בְּפוּמִי שְׁבַח חֲדָתָא תְּהֵי תּוּשְׁבַּחְתָּא קֳדָם אֱלָהָנָא יֶחֱמוּן סַגִיאִין וְיִדְחֲלוּן וְיִסְבְּרוּן בְּמֵימְרָא דַיְיָ:
טוּבוֹי דְגַבְרָא דְשַׁוֵי יְיָ רוּחְצָנֵיהּ וְלָא אִסְתְּכֵי לְוַת סוּרְבָנַיָא וּמְמַלְלֵי כַדִיבוּתָא:
סַגִיאֵי נִסַיָא דַעֲבַדְתָּא אַתְּ יְיָ אֱלָהִי פְרִישְׁוָתָךְ וּרְעוּתָךְ לְוָתָנָא לֵית אֶפְשַׁר לְסַדָרָא לְוָתָךְ תּוּשְׁבַּחְתָּךְ אַתְנֵי וַאֲמַלֵל תַּקִיפוּ מִן לְחַוָאָה:
נִכְסָא וְדוֹרוֹנָא לָא צְבִיתָא אוּדְנִין לְאַצָתָא פוּרְקָנָךְ כְּרֵיתָא לִי עַלְתָא וְקָרְבַּן חַטָאתִי לָא שְׁיֵלְתָּא:
הֵידֵין אֲמַרֵית הָא עֲלֵית לְחַיֵי עַלְמָא כַּד אֶעֱסוֹק בִּמְגִלַת סִפְרָא דְאוֹרַיְיתָא דְאִכְּתִיב אַמְטֻלְתִּי:
לְמֶעֱבַד רְעוּתָךְ אֱלָהִי רְעֵיתִי וְאוֹרַיְתָךְ כְּלִילָא בִמְצַע מַעְיְנָי:
בַּשְׂרֵית צִדְקָא בִּכְנִישְׁתָּא רַבָּא הָא סִפְוָתַי לָא אֶמְנַע יְיָ אֱלָהִי אַתְּ חַכֵּימְתָּ:
צִדְקָתָךְ לָא טַמְרֵית בְּגוֹ לִבִּי קוּשְׁטָךְ וּפוּרְקָנָךְ אֲמַרֵית לָא כַסֵיתִי טִיבוּתָךְ וְהֵימָנוּתָךְ בִּקְהָלָא רַבָּא:
בִּגְלַל כֵּן אַתְּ יְיָ לָא תִמְנַע רַחֲמָךְ מִנִי טוּבָךְ וְקוּשְׁטָךְ תְּדִירָא יִנְטְרוּנַנִי:
אֲרוּם תַּקִיפִין עֲלַי בִּישִׁין עַד דְלֵית מִנְיָן אַדְבִּיקוּ יָתִי חוֹבַי וְלָא יְכֵלִית לְמֶחֱמֵי תְּקַפוּ מִן בִּינְתֵי רֵישֵׁי וְרַעְיוֹנַי שַׁבְקוּנִי:
אִתְרְעֵי יְיָ לְשֵׁזָבוּתַנִי יְיָ דְסִיוּעִי זְרִיז:
יִבְהַתוּן וְיִתְחַסְדוּן כַּחֲדָא תָּבְעֵי נַפְשִׁי לְמִגְמְרָהּ יְזוּרוּן לַאֲחוֹרָא וְיִכְסְפוּן דְצָבְיָן בִישְׁתִּי:
יִתְבַּיְרוּן מְטוּל בַּהֲתֵיהוֹן דְאָמְרִין לִי חֲדֵינָא עַל תַּבְרֵיהּ חֶדְוָא עַל סִגוּפֵיהּ:
יֶחְדוּן וִידוּצוּן בְּמֵימְרָךְ כָּל בַּעֲיָךְ וְיֵמְרוּן תְּדִירָא יִסְגֵי חֵילָא דַייָ דִי רָחֲמִין פּוּרְקָנָךְ:
וַאֲנָא עַנְיָא וַחֲשׁוֹכָא יְיָ יַחֲשֵׁב טָב לִי עֲלַי סִיוּעִי וְשֵׁזְבוּתִי אַתְּ אֱלָהִי לָא תוֹחַר:
