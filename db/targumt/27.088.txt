שִׁירָא וְתוּשְׁבַּחְתָּא עַל יְדֵיהוֹן דִבְנֵי קֹרַח עַל צְלוֹתָא לְשַׁבָּחָא שִׂכְלָא טָבָא עַל יְדוֹי דְהֵימָן יַצִיבָא:
יְיָ אֱלָהָא פוּרְקָנֵיהּ יֵימָם קַבְּלֵית בְּלֵילְיָא צְלוֹתִי לְקִבְלָךְ:
תֵּעוֹל קֳדָמָךְ צְלוֹתִי אַצְלֵי אוּדְנָךְ לְבָעוּתִי: (ת"א יֵיתֵי קֳדָמָךְ צְלוֹתִי עַל עַמָךְ בֵּית יִשְׂרָאֵל וְהַט אוּדְנָךְ לְתוּשְׁבַּחְתִּי דְשַׁבְּחֵת לִיקָרָךְ):
אֲרוּם שְׂבַעַת בְּבִישָׁן נַפְשִׁי וְחַיַי לְשֵׁיוֹל מְטוּ:
אִתְחַשְׁבֵית עִם נַחֲתֵי בֵית אֲסִירֵי הֲוֵיתִי הֵיךְ בַּר נַשׁ דְלֵית לֵיהּ חֵילָא:
הֵיךְ רַשִׁיעַיָא דְמִיתוּ וְלָא תָבוּ יִתְעֲבַדוּ בְנֵי חוֹרִין מִן מַצוּתָא הֵיךְ קְטִילִין דְחַרְבָּא שָׁכְבֵי בֵית קְבוּרְתָּא דְלָא דְכַרְתִּנוּן תוּב וְהִנוּן מֵאַפֵּי שְׁכִנְתָּךְ אִתְפַּלָגוּן:
שַׁוִיתָא יָתִי בְּגָלוּתָא דִמְתִילָא לְגוּבָא אַרְעִיתָא בַּחֲשׁוֹכַיָא בִּמְצוּלְתָּא:
עֲלַי סַמְכַת חֵמְתָךְ וְכָל גְזֵירָן בִישָׁן תַבַּרוּ יָתִי סַגֵפְתָּא לְעָלְמִין:
אַרְחֵקְתָּא מְיוּדָעַי מִנִי שַׁוִיתָא יָתִי רִיחוּקִין לְהוֹן כָּלוּא בְּבֵית אֲסִירֵי וְלָא אֶפּוֹק:
עֵינִי זַלְגַת דִמְעִין מִן עֲנִיתָא קְרִיתָךְ יְיָ בְּכָל יוֹמָא פַּרְסֵית לְוָתָךְ בִּצְלוֹ יְדָי:
הַאֶפְשַׁר דִי לְמֵיתַיָא תַעְבֵּיד פְּרִישָׁן אִין גוּשְׁמַיָא דְאִתְמְסִיאוּ בְּעַפְרָא יְקוּמוּן יוֹדוּן קֳדָמָךְ לְעָלְמִין:
הַאֶפְשַׁר דִי יִתְחוֹי בְּבֵית קְבוּרְתָא טוּבָךְ קוּשְׁטָא בְּבֵית אָבְדָנָא:
הַאֶפְשַׁר דִי תְוַדְעוּ בַּחֲשָׁךְ גֵהִנָם פְּרִישְׁוָתָךְ וְצִדְקָתָךְ דְצַחוּתָא וְצַדְיָא:
וַאֲנָא קֳדָמָךְ יְיָ צַלֵיתִי וּבְצַפְרָא צְלוֹתִי תְקַדְמִינָךְ:
לָמָה יְיָ שְׁבַקְתָּא נַפְשִׁי תַּעֲלִים אַפָּךְ מִנִי דְלָא אֶחְמֵי בִנְהוֹרָךְ נְהוֹרָא:
עַנְיָא אֲנָא וּנְגִיד מִן טַלְיָא סוֹבַרֵית רְחִימְתָּךְ טְעִינָא עֲלַי:
עֲלַי עֲבַרוּ רוּגְזָיִךְ בַּעֲתָיִךְ שֵׁיצִיאוּ יָתִי:
אַחְזַרוּ יָתִי הֵיךְ מוֹי כָּל יוֹמָא אַקִיפוּן עֲלַי כַּחֲדָא:
אַרְחֵקְתָּא מִנִי רְחִימָא וְחַבְרָא וּלְיָדְעֵי חֲשִׁיךְ אֲנָא בְּפוּמְהוֹן:
