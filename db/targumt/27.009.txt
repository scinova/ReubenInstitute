לְשַׁבָּחָא עַל מִיתוֹתָא דְגַבְרָא דִי נְפַק מִבֵּנֵי מַשִׁרְיָתָא תּוּשְׁבַּחְתָּא לְדָוִד:
אֲשַׁבַּח קֳדָם יְיָ בְּכָל לִבִּי אֶשְׁתָּעֵי כּוּלְהוֹן פְּרִישְׁוָתָךְ:
אֶחְדֵי וְאֶבְדָח בְּמֵימְרָךְ אֲשַׁבַּח שְׁמָךְ עִלָאָה:
בִּדְיְתוּב בְּעֵל דְבָבֵי לַאֲחוֹרָא יִתְקְלוּן וִיהוֹבְדוּן מִן קֳדָמָךְ:
מְטוּל דַעֲבַדְתָּא פּוּרְעֲנוּתִי וְדִינִי יְתֵבְתָּא עַל כּוּרְסֵי דַיָנָא זַכָּאָה:
נְזַפְתָּא עַמְמֵי דִפְלִשְׁתָּאֵי הוֹבַדְתָּא גָלְיַת רַשִׁיעָא שׁוּמְהוֹן מְחִיתָא לְעַלְמֵי עָלְמִין:
וְכַד נְפַל בְּעֵל דְבָבָא אִשְׁתֵּצָיוּ חֶלְוָתֵיהּ וּכְרַכֵּיהוֹן אִצְתַּדְיוּ לְעַלְמָא וְקִרְוֵיהוֹן צְדִיתָא הוֹבַדְתָּא דָכְרָנְהוֹן מִנְהוֹן:
וּמֵימְרָא דַייָ לְעַלְמִין מוֹתְבֵיהּ בִּשְׁמֵי מְרוֹמָא תַקִין לְדִינָא כּוּרְסֵיהּ:
וְהוּא יְדֵין עַמָא דְאַרְעָא בִּזְכוּתָא יְדִין עַמְמַיָא בִּתְרִיצוּתָא:
וִיהֵי מֵימְרָא דַייָ תְּקוֹף לְמַסְכְּנָא תְּקוֹף בְּעִדָנֵי עַקְתָא:
וְיִסְתַּכְּלוּן בְּסַבָּרוּתָךְ יָדְעֵי שְׁמָךְ מְטוּל דְלָא שְׁבַקְתָּא תָּבְעָיִךְ יְיָ:
שַׁבַּחוּ קֳדָם יְיָ דְאַשְׁרֵי שְׁכִנְתֵּיהּ בְּצִיוֹן תַּנִיאוּ בְּעַמְמַיָא עוֹבָדוֹי:
מְטוּל דְתָבֵיעַ דְמָא דְזַכָּאָה יַת צַדִיקַיָא דְכִיר וְלָא שְׁלִי קְבֵלַת עִנְוְתָנִין:
חוּס עֲלַי יְיָ חֲמֵי סִגוּפִי מִבַּעֲלֵי דְבָבִי מְרוֹמֵם לִי מִמַעֲלָנֵי מוֹתָא:
מִן בִּגְלַל דְאִשְׁתָּעֵי לְכוּלְהוֹן שִׁבְחָתָךְ בְּמַעֲלָנֵי תַרְעֵי כְנִישַׁת צִיוֹן אַדוּץ בְּפוּרְקָנָךְ:
טְבַעוּ עַמְמַיָא בְּשׁוּחֲתָא דַעֲבָדוּ בִּמְצַדְתָּא דְנָא כְּמַנוּ אִתְאֲחַדַת רִגְלֵיהוֹן:
גְלֵי קֳדָם יְיָ דִין דַעֲבַד בְּעוֹבָדֵי אַיְדוֹי אִתְּקַל רַשִׁיעָא יְרַנְנוּן צַדִיקַיָא לְעָלְמִין:
יְתוּבוּן רַשִׁיעֵי לִשְׁיוֹל כּוּלְהוֹן עַמְמַיָא דְאַשְׁלִיוּ דְחַלְתָּא דַיְיָ:
מִטוּל דְלָא לְעָלְמַיָא מִשְׁתְּלֵי חֲשׁוֹכָא סוֹבָרָא דְעִנְוְתָנִי לָא תְהוֹבֵד לְעָלְמָא:
קוּם יְיָ לָא יִתְעַשַׁן בַּר נָשָׁא רַשִׁיעָא יִתְדָנוּן עַמְמַיָא קֳדָמָךְ:
שַׁוִי יְיָ דְחַלְתָּא לְהוֹן יִנְדְעוּן עַמְמַיָא דְבַר נָשׁ הִינוּן לְעָלְמִין:
