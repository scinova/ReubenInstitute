לְשַׁבָּחָא עַל תְּקוֹף קוּרְבַּן תְּדִירָא דִקְרִצְתָּא תּוּשְׁבַּחְתָּא לְדָוִד:
אֵלִי אֵלִי מְטוּל מַה שְׁבַקְתַּנִי רָחִיק מִן פּוּרְקָנִי מִלֵי אַכְלְיוּתִי:
אֱלָהִי אֲנָא קָרֵי בִימָמָא וְלָא תְקַבֵּל וּבְלֵילְיָא לָא שְׁתִיקוּתָא לִי:
וְאַנְתְּ קַדִישׁ דִמְיַתֵּב עַלְמָא עַל תּוּשְׁבְּחָן יִשְׂרָאֵל:
עֲלָךְ סַבַּרוּ אַבְהָתָנָא סַבַּרוּ בְּמֵימְרָךְ וְשֵׁזֶבְתִּנוּן:
קֳדָמָךְ צַלוּ וְאִשְׁתְּזִיבוּ וַעֲלָךְ אִתְרְחִיצוּ וְלָא בְהִיתוּ:
וַאֲנָא זְחִיל חֲלָשׁ וְלָא גִבָּר סָכָל חִסוּדֵי בְּנֵי נָשָׁא וּמִתְרַמְרְמוּתָא דְעַמְמֵי:
כָּל דְחָמְיָן לִי יִדְהֲכוּן עֲלָי מִתְגָרְדִין בְּסִפְוָתְהוֹן יְטַלְטְלוּן בְּרֵישֵׁיהוֹן:
שַׁבַּח קֳדָם יְיָ וְשֵׁזְבֵיהּ פְּצָא יָתֵיהּ מְטוּל דְאִתְרְעֵי בֵיהּ:
מְטוּל דְאַתְּ אַפֵקְתַּנִי מִכְּרֵיסָא אַסְבַּרְתַּנִי עַל תַדְיֵי אִמִי:
בְּסַעְדָךְ אִטַלְקֵית מִמַעַיְנָא מִכְּרֵיסַת אִמִי אֵלִי אָתְּ:
לָא תִתְרָחֵיק מִנִי מְטוּל דְעַקְתָא קְרֵיבָא אֲרוּם לֵית פָּרִיק:
אַחֲזַרוּ עֲלַי פַּלְחֵי כּוֹכְבַיָא דִמְתִילִין לְתוֹרִין סַגִיאִין רַבְרְבָנֵי מַתְנָן אַקְפוּנִי:
פְּתַחוּ עֲלַי פּוּמְהוֹן הֵיךְ כְּאַרְיָא מַכְלֵי וְתָבַר:
הֵיךְ כְּמַיָא אִשְׁתְּדֵת וְאִתְפַּרְפְּרָן כּוּלְהוֹן גַרְמַי הֲוָה לִבִּי מִתְמְסֵי הֵיךְ שַׁעֲוָה בְּגוֹ מֵעָי:
יָבֵשׁ הֵיךְ פְּחַר חֵילִי וְלִישָׁנִי אִדְבֵּיק לְמוֹרִיגִי וּלְבֵית קְבוּרְתָּא אַמְטֵיתַנִי:
מְטוּל דְאַחֲזַרוּ עֲלָי רַשִׁיעֵי דִמְתִילִין לְכַלְבַיָא סַגִיעִין כְּנִישַׁת מַבְאִישִׁין אַקְפוּנִי נָכְתִין הֵיךְ כְּאַרְיָא אַיְדַי וְרִגְלָי:
אֲחַוֵי כָּל הַלְבְּשׁוּשַׁי גַרְמַי אִנוּן מִסְתַּכְּלִין מְבַזָן לִי:
מְפַלְגִין לְבוּשַׁי וְעִלַוֵי פְתָנֵי יִרְמוֹן עַדְבִין:
וְאַנְתְּ יְיָ לָא תִתְרָחִיק תּוּקְפִי לְסִיוֹעִי אוֹחִי:
פְּצֵי מִקָטְלִין בְּחַרְבָּא נַפְשִׁי מִן אַיְדָא דְכַלְבָּא רוּחָא דְגוּשְׁמִי:
פְּרוֹקְנִי מִפּוּם אַרְיָא וּמִגַבְרַיָא דְתַקִיפִין וְרָמִין כְּרֵימָנָא קַבֵּלְתָּא צְלוֹתִי:
אֲחַוֵי גְבוּרַת שְׁמָךְ לְאֶחָי בִּמְצַע כְּנִשְׁתָּא אֲשַׁבְּחִנָךְ:
דַחֲלַיָא דַייָ שַׁבַּחוּ קֳדָמוֹי כָּל זַרְעָא דְיַעֲקֹב הָבוּ לֵיהּ יְקָר וְאִזְדַעְזִעוּ מִנֵיהּ כָּל זַרְעָא דְיִשְׂרָאֵל:
מְטוּל דְלָא בַסֵר וְלָא שָׁאַט צְלוֹתְהוֹן דַעֲנִיֵי וְלָא סַלִיק שְׁכִינְתֵּיהּ מִבֵּינֵיהוֹן וּבְצַלְאֵיהוֹן קֳדָמוֹהִי מְקַבֵּל:
מִנָךְ תּוּשְׁבַּחְתִּי בִּכְנִישַׁת סְגִיעַת עַמָא נִדְרִי אֲשַׁלֵם קֲבֵיל דַחֲלוֹי:
יֵיכְלוּן עֲנְוְתָנֵי וְיִשְׂבְּעוּן יְשַׁבַּחוּ קֳדָם יְיָ דְתָבְעִין לֵיהּ יִשְׁרֵי רוּחַ נְבוּאָה בְרַעְיוֹנֵי לִבְּכוֹן לְעָלְמִין:
יִדְכְּרוּן נְסוֹי וִיתוּבוּן קֳדָם יְיָ כָּל סְיָפֵי אַרְעָא וְיִסְגְדוּ קֳדָמָךְ כָּל גְנִיסַת עַמְמַיָא:
מְטוּל דְמִן קֳדָם יְיָ מַלְכוּתָא וְשַׁלִיט בְּעַמְמַיָא:
סְעוּדוּ וּסְגִידוּ כָּל דְהִינֵי אַרְעָא קֳדָמוֹי יְגַחְנוּן כָּל נָחְתִין בֵּי קְבוּרְתָּא וּנְפַשׁ רַשִׁיעָא לָא יְחַיֵי:
זַרְעֵיהּ דְאַבְרָהָם יִפְלְחוּן קֳדָמוֹי וִיחַווּן כֹּחַ גְבוּרְתָּא דַייָ לְדָרָא בַּתְרָאָה:
יְתוּבוּן בְּנֵיהוֹן וִיתַנוּן צְדַקְתֵּיהּ לְעַמֵיהּ דַעֲתִיד לְמֵילַד פְּרִישָׁן דַעֲבַד:
