וְאָמַר ה׳ לְנֹחַ: “_כַּנֵּס_ אַתָּה וְכֹל +אַנְשֵׁי+ בֵּיתְךָ לַתֵּיבָה, כִּי אוֹתְךָ רָאִיתִי צַדִּיק לְפָנַי בַּדּוֹר הַזֶּה.
מִכֹּל הַבְּהֵמָה הַזַּכָּה(~הַטְּהוֹרָה), תִּקַּח לְךָ שִׁבְעָה-שִׁבְעָה, _זָכָר וּנְקֵבָה_; וּמִן הַבְּהֵמָה _שֶׁלֹּא־יֶשְׁנָהּ_(=שֶׁאֵינֶנָּה) זַכָּה הִיא - שְׁנַיִים, _זָכָר וּנְקֵבָה_.
אַף מֵעוֹף הַשָּׁמַיִים, שִׁבְעָה-שִׁבְעָה, זָכָר וּנְקֵבָה, [כְּדֵי] _לְקַיֵּים_ [אֶת] הַזֶּרַע עַל פְּנֵי כׇל הָאָרֶץ”.
“כִּי לִ+זְמַן+ יָמִים עוֹד שִׁבְעָה, אֲנִי _מוֹרִיד גֶּשֶׁם_ עַל הָאָרֶץ, אַרְבָּעִים _יְמָמוֹת_, וְאַרְבָּעִים לֵילוֹת; וְאֶמְחֶה אֶת כׇּל הַיְקוּם אֲשֶׁר עָשִׂיתִי, מֵעַל פְּנֵי הָאָרֶץ”.
וְעָשָׂה נֹחַ, כְּכׇל אֲשֶׁר צִוָּוהוּ ה׳.
וְנֹחַ בֶּן שֵׁשׁ מֵאוֹת שָׁנִים; וְהַמַּבּוּל הָיָה, מַיִם עַל הָאָרֶץ.
_וְנִכְנָס_ נֹחַ, וּבָנָיו וְאִשְׁתּוֹ, וּנְשׁוֹת בָּנָיו, עִמּוֹ לַתֵּיבָה, מִן פְּנֵי מֵי _הַהֲצָפָה_.
מִן הַבְּהֵמָה הַזַּכָּה, וּמִן הַבְּהֵמָה אֲשֶׁר אֵינֶנָּה זַכָּה, וּמִן הָעוֹף, וְכֹל אֲשֶׁר רוֹחֵשׁ עַל הָאָרֶץ -
שְׁנַיִים-שְׁנַיִים _נִכְנְסוּ_ אֶל נֹחַ לַתֵּיבָה, זָכָר וּנְקֵבָה, כְּמָה־אֲשֶׁר(=כְּמוֹ שֶׁ) צִיוָּוה ה׳ אֶת נֹחַ.
וְהָיָה לִ+זְמַן+ שִׁבְעָה יָמִים, וּמֵי הַהֲצָפָה הָיוּ עַל הָאָרֶץ.
בִּשְׁנַת שֵׁשׁ מֵאוֹת שָׁנִים לְחַיֵּי נֹחַ, בְּיֶרַח(~בְּחוֹדֶשׁ) הַשֵּׁנִי, בְּשִׁבְעָה עָשָׂר יוֹם לַיֶּרַח (~י"ז בְּחֶשְׁוָן) - בַּיּוֹם הַזֶּה, נִבְקְעוּ כֹּל מַבּוּעֵי(~מַעְיָינוֹת) הַתְּהוֹם הַגָּדוֹל, _וְחַלּוֹנוֹת_ הַשָּׁמַיִים נִפְתְּחוּ.
וְהָיָה הַגֶּשֶׁם +יוֹרֵד+ עַל הָאָרֶץ, אַרְבָּעִים _יְמָמוֹת_, וְאַרְבָּעִים לֵילוֹת.
_בְּמַרְאָה_(~לְאוֹר) הַיּוֹם הַזֶּה _נִכְנָס_ נֹחַ, וְשֵׁם וְחָם וָיֶפֶת בְּנֵי נֹחַ, וְאֵשֶׁת נֹחַ, וְשָׁלוֹשׁ נְשׁוֹת בָּנָיו, עִמָּם לַתֵּיבָה.
הֵם, וְכֹל הַחַיָּה לְמִינָהּ, וְכֹל הַבְּהֵמָה לְמִינָהּ, וְכֹל רוֹחֵשׁ, שֶׁרוֹחֵשׁ עַל הָאָרֶץ לְמִינֵהוּ, וְכֹל הָעוֹף לְמִינֵהוּ, כֹּל צִיפּוֹר, כֹּל _שֶׁפּוֹרֵחַ_.
_וְנִכְנְסוּ_ _עִם_ נֹחַ לַתֵּיבָה, שְׁנַיִים-שְׁנַיִים מִכׇּל הַבָּשָׂר, אֲשֶׁר בּוֹ רוּחַ שֶׁל־חַיִּים.
וְ_הַנִּכְנָסִים_ - זָכָר וּנְקֵבָה - מִכֹּל הַבָּשָׂר _נִכְנְסוּ_, כְּמוֹ אֲשֶׁר צִיוָּוה אוֹתוֹ ה׳, וְ_הֵגֵן_ +(בְּמַאֲמָרוֹ)+ עָלָיו.
וְהָיְיתָה _הַהֲצָפָה_, אַרְבָּעִים _יְמָמוֹת_ עַל הָאָרֶץ; וְרַבּוּ הַמַּיִם, וְנָשְׂאוּ אֶת הַתֵּיבָה, וְנֵרוֹמָה(=נִתְרוֹמְמָה) מֵעַל הָאָרֶץ.
_וְחָזְקוּ_ הַמַּיִם, וְרַבּוֹ מְאוֹד עַל הָאָרֶץ; וְהִלְּכָה הַתֵּיבָה עַל פְּנֵי הַמַּיִם.
וְהַמַּיִם _חָזְקוּ_ מְאוֹד-מְאוֹד עַל הָאָרֶץ, וְ_נִתְחַפּוּ_(~הִתְכַּסּוּ) כֹּל הֶהָרִים הָרַמִּים(~הַגְּבוֹהִים), אֲשֶׁר תַּחַת כֹּל הַשָּׁמָיִים.
חֲמֵשׁ עֶשְׂרֵה אַמּוֹת מִלְּמַעְלָה, _חָזְקוּ_ הַמַּיִם, _וְנִתְחַפּוֹ_ הֶהָרִים.
_וּמֵת_ כׇּל הַבָּשָׂר שֶׁרוֹחֵשׁ עַל הָאָרֶץ: בָּעוֹף, וּבַבְּהֵמָה, וּבַחַיָּה, וּבְכָל הָרוֹחֵשׁ שֶׁרוֹחֵשׁ עַל הָאָרֶץ, וְכֹל הָאָדָם.
כֹּל אֲשֶׁר נִשְׁמַת רוּחַ שֶׁל־חַיִּים בְּפָנָיו, מִכֹּל אֲשֶׁר _בַּיַּבָּשָׁה_ - מֵתוּ.
וּמָחָה אֶת כׇּל הַיְקוּם אֲשֶׁר עַל פְּנֵי הָאָרֶץ, מֵאָדָם עַד בְּהֵמָה, עַד רוֹחֵשׁ, וְעַד עוֹף הַשָּׁמַיִים, וְנִמְחוּ מִן הָאָרֶץ; וְנִשְׁאַר אַךְ נֹחַ, וַאֲשֶׁר עִמּוֹ בְּתֵיבָה.
_וְחָזְקוּ_ הַמַּיִם עַל הָאָרֶץ, מֵאָה וַחֲמִישִּׁים יָמִים.
