וְהַנָּחָשׁ הָיָה _חָכָם_+, לָבוּשׁ+ מִכֹּל חַיּוֹת _הַחוּץ_, שֶׁעָשָׂה ה׳ אֱלֹהִים; וְאָמַר לָאִישָּׁה: "הַ[אִם ]אֱמֶת שֶׁאָמַר +ה׳+ אֱלֹהִים 'לֹא תֹּאכְלוּ מִכֹּל אִילַן הַגִּינָּה'?!".
וְאָמְרָה הָאִישָּׁה לַנָּחָשׁ: “מִ+שְּׁאָר+ פֵּירוֹת אִילַן הַגִּינָּה [כֵּן] +יֵשׁ לָנוּ רְשׁוּת+ _לֶאֱכוֹל_,
וְ[רַק ]מִפֵּירוֹת הָאִילָן שֶׁבְּ_אֶמְצַע_ הַגִּינָּה, אָמַר ה׳ 'לֹא תֹּאכְלוּ מִמֶּנּוּ, וְלֹא _תִּקָּרְבוּ_ בּוֹ - שֶׁמָּא תָּמוּתוּ!'”.
+בָּהּ הִיא הַשָּׁעָה+(=בְּאוֹתָה הַשָּׁעָה)+, +אָמַר הַנָּחָשׁ +רְכִילוּת+(~לְשׁוֹן הָרַע)+ עַל בּוֹרְאוֹ, וְאָמַר+ לָאִישָּׁה: "לֹא מוֹת תָּמוּתוּ! +אֲבָל כֹּל אוֹמָן - שׂוֹנֵא +[אֶת]+ בֶּן אֻמָּנוּתוֹ+.
כִּי _גָּלוּי_ +לִפְנַי+ ה׳, כִּי בַּיּוֹם שֶׁתֹּאכְלוּ מִמֶּנּוּ, … …; וְתִהְיוּ כְּ_מַלְאָכִים גְּדוֹלִים־גְּדוֹלִים_ שֶׁ_חֲכָמִים לָדַעַת_ +בֵּין+ טוֹב לְרַע.
וְרָאֲתָה הָאִישָּׁה +אֶת סַמָּאֵל, מַלְאַךְ הַמָּוֶות, וּפָחֲדָה, וְיָדְעָה+ כִּי טוֹב הָאִילָן לְמַאֲכָל, וְכִי _רִיפּוּי_ הוּא לְ+אוֹר הָ+עֵינַיִים, וּמְחַמֵּד הָאִילָן _לְהִסְתַּכֵּל בּוֹ_; וְלָקְחָה מִפִּרְיוֹ - וְאָכְלָה. וְנָתְנָה אַף לְ_בַעֲלָהּ_ עִמָּהּ - וְאָכַל.
וְ_נָאוֹרוּ_ עֵינַי שְׁנֵיהֶם, וְיָדְעוּ, כִּי _עַרְטִילָאִים_ הֵם, +שֶׁנִּתְעַרְטְלוּ+(~הִתְפַּשְּׁטוּ)+ מִן לְבוּשׁ הַצִּיפּוֹרֶן שֶׁנִּבְרְאוּ בּוֹ, וְהָיוּ רוֹאִים +[אֶת]+ בְּעָתָתָם+(=הַבְּעָתָה שֶׁלָּהֶם)ּ+,+ וְ_חִייְּטוּ_(~מִלָּשׁוֹן 'לְחַיָּט', לְחַבֵּר בְּחוּט) +לָהֶם+ מֵעֲלֵי תְּאֵנִים, וְעָשׂוּ לָהֶם _קִימּוּרִים_(~חֲגוֹרוֹת).
וְשָׁמְעוּ אֶת קוֹל +מַאֲמַר+ ה׳ _מְטַיֵּיל_ בַּגִּינָּה, לִ_מְנוֹחַ_ הַיּוֹם; וְנֶחְבָּא אָדָם וְאִישְׁתּוֹ מִן +פְנֵי+ ה׳ אֱלֹהִים בְּ_אֶמְצַע_ אִילָנֵי הַגִּינָּה.
וְקָרָא ה׳ אֱלֹהִים לָאָדָם, וְאָמַר לוֹ: "… …? +הֲלֹא כֹּל הָעוֹלָם שֶׁבְּרִיתִי גָּלוּי לְפָנַי, חוֹשֶׁךְ כָּאוֹר; וְאֵיךְ אַתָּה חוֹשֵׁב בְּלִיבְּךָ לְהִתְחַבֵּא מִן פָּנַי? הֲלֹא +[אֶת הַ]+מָּקוֹם שֶׁאַתָּה נֶחְבָּא בּוֹ - אֲנִי רוֹאֶה. וְהֵיכָן הֵן הַמִּצְווֹת שֶׁפְּקַדְתִּיךָ?+"
וְאָמַר: "אֶת קוֹל +מַאֲמָרְךָ+ שָׁמַעְתִּי בַּגִּינָּה; וּפָחַדְתִּי כִּי _מְעֻורְטָל_ אָנִי+, וְהַמִּצְוָוה שֶׁפְּקַדְתַּנִי - הֶעֱבַרְתִּי מִמֶּנִּי+ - וְנֶחְבֵּאתִי +מִן הַבּוּשָׁה+".
וְאָמַר: "מִי הִגִּיד לְךָ, כִּי _מְעֻורְטָל_ אַתָּה? +שֶׁמָּא+ מִן פֵּירוֹת הָאִילָן, שֶׁצִּיוִּויתִיךָ שֶׁלֹּא לֶאֱכוֹל מִמֶּנּוּ - אָכַלְתְּ?!"
וְאָמַר אָדָם: "הָאִישָּׁה שֶׁנָּתַתָּ אֶצְלִי - הִיא נָתְנָה לִי מִן +פֵּירוֹת+ הָאִילָן, וְאָכַלְתִּי".
וְאָמַר ה׳ אֱלֹהִים לְאִישָּׁה: "מַה זֹּאת עָשִׂית??" וְאָמְרָה הָאִישָּׁה "הַנָּחָשׁ הִשִּׁיאַנִי(=הִשִּׂיא אוֹתִי) +בְּחוֹכְמָתוֹ, וְהִטְעַנִי בְּרִישְׁעוּתוֹ+ - וְאָכַלְתִּי".
+וְהֵבִיא ה׳ אֱלֹהִים +[אֶת]+ שְׁלָשְׁתָּם לְדִין+, וְאָמַר לַנָּחָשׁ: "כִּי עָשִׂיתָ זֹאת - אָרוּר אַתָּה מִכֹּל הַבְּהֵמָה, וּמִכֹּל חַיּוֹת _הַחוּץ_!! עַל _מֵעֶיךָ_ _תִּהְיֶה מְטַיֵּיל_, +וְרַגְלֶיךָ יִקָּצְצוּ, וְעוֹרְךָ יִהְיֶה מַפְשִׁיט אַחַת לְשֶׁבַע שָׁנִים, וְאֶרֶס הַמָּוֶות בְּפִיךָ+, וְעָפָר תֹּאכַל כֹּל יְמֵי חַיֶּיךָ.
וְשִׂנְאָה אָשִׂים בֵּינְךָ וּבֵין הָאִשָּׁה, בֵּין מִשְׁפְּחוֹת בִּנְךָ וּבֵין מִשְׁפְּחוֹת בָּנֶיהָ; +וִיהִי כַּאֲשֶׁר יִהְיוּ בָּנֶיהָ שֶׁל־הָאִשָּׁה שׁוֹמְרִים +[אֶת]+ מִצְוַת הַתּוֹרָה - יִהְיוּ מְכַוְּנִים וּמַכִּים אוֹתְךָ עַל רֹאשְׁךָ, וְכַאֲשֶׁר עוֹזְבִים +[אֶת]+ מִצְוַת הַתּוֹרָה - תִּהְיֶה מִתְכַּוֵּן וּמַקִּישׁ אוֹתָם בְּעִקְבֵיהֶם. אֲבָל לָהֶם יִהְיֶה רִפּוּי, וּלְךָ לֹא יִהְיֶה רִפּוּי, וַעֲתִידִים הֵם לַעֲשׂוֹת שְׁפִיּוּת בָּעִקְבָה בִּימֵי מֶלֶךְ הַמָּשִׁיחַ+.
לָאִשָּׁה אָמַר: "רִבּוּי אַרְבֶּה [אֶת] סִגּוּפֵךְ +בְּאָדָם בְּתוּלִים+ וְעִדּוּיֶיךָ, בְּצַעַר תֵּלְדִי בָּנִים; וְאַל בַּעֲלֵךְ תִּהְיֶה תְּשׁוּקָתֵךְ, וְהוּא יִהְיֶה שׁוֹלֵט בָּךְ+,לְהִזְדַּכּוֹת(~לְהִטָּהֵר) וְלָחֲטֹא".
וִלַאָדָם אָמַר: " כִּי _קִיבַּלְתָּ_(~שָׁמַעְתָּ) לְמַאֲמַר אִשְׁתְּךָ, וְאָכַלְתָּ מִן הָאִילָן אֲשֶׁר צִיוִּויתִיךָ לֵאמֹר: 'לֹא תֹּאכַל מִמֶּנּוּ' - אֲרוּרָה הָאֲדָמָה +בגלל שלא הגידה לך חטאך+; בְּעָמָל תַּאֲכִילֶנָּה, כֹּל יְמֵי חַיֶּיךָ.
וְכוּבִים וַאֲטָדִים תַּצְמִיחַ +וְתַרְבֶּה בִּשְׁבִילְךָ+, וְתֹאכַל אֶת עֵשֶׂב +שֶׁעַל פְּנֵי+ הַחוּץ. +עָנָה הָאָדָם וְאָמַר: "בְּבַקָּשַׁת רַחֲמִים מִן פָּנֶיךָ ה׳, שֶׁלֹּא נִתְחַשֵּׁב כַּבְּהֵמָה, שֶׁנֹּאכַל עֵשֶׂב פְּנֵי הַחוּץ. נַעֲמוֹד כָּעֵת, וְנַעֲמוֹל בַּעֲמַל יָדַיי, וְנֹאכַל מָזוֹן מִן מְזוֹן הָאָרֶץ, וּבְכֵן יִפָּרֵד כָּעֵת לְפָנֶיךָ, בֵּין בְּנֵי אֱנוֹשׁ וּבֵין בְּנֵי בְּהֵמָה+".
בְּעָמָל _כַּף יָדְךָ_ תֹּאכַל _מָזוֹן_, עַד שֶׁתַּחְזוֹר לֶעָפָר, שֶׁמִּמֶּנּוּ _נִבְרֵאתָ_; כִּי עָפָר אַתָּה, וּלְעָפָר תָּשׁוּב! +שֶׁמִּן עָפָר אַתָּה עָתִיד לַעֲמוֹד, לָתֵת דִּין וְחֶשְׁבּוֹן, עַל כֹּל מָה שֶׁעָשִׂיתָ, בַּיּוֹם הַדִּין הַגָּדוֹל+".
וְקָרָא אָדָם [אֶת] שֵׁם אִשְׁתּוֹ 'חַוָּה' - כִּי הִיא הָיְיתָה אִמָּא שֶׁל־כֹּל _בְּנֵי הָאֱנוֹשׁ_.
וְעָשָׂה ה אֱלֹהִים לָאָדָם וּלְאִשְׁתּוֹ, _לְבוּשִׁים שֶׁל־כָּבוֹד_ +מִן עוֹר הַנָּחָשׁ שֶׁהִשְׁלִיךְ מִמֶּנּוּ+, עַל עוֹר בִּשְׂרֵיהֶם, +תְּמוּרַת שׁוֹפְרוֹתֵיהֶם שֶׁנִּשְׁלְחוּ,+ וְהִלְבִּישָׁם.
וְאָמַר ה' אֱלֹהִים +לַמַּלְאָכִים אֲשֶׁר מְשַׁמְּשִׁים לְפָנָיו+: "הִנֵּה אָדָם הָיָה יְחִידִי +בָּאָרֶץ - אֵיךְ־מָה+(=כְּמוֹ)+ שֶׁאֲנִי יְחִידִי בִּשְׁמֵי הַמָּרוֹם - וַעֲתִידִים לַעֲמֹד+ מִמֶּנּוּ שֶׁיָּדְעוּ +לְהַפְרִיד בֵּין+ טוֹב לְרַע.+ אִלּוּ שָׁמַר +[עַל]+ הַמִּצְווֹת שֶׁצִּוִּיתִיו, יֵשׁ־הוּא+(=הִנּוֹ)+ חַי וְקַיָּם בְּעֵץ הַחַיִּים, עַד לָעוֹלָמִין!+ וְעַתָּה, +עַד שֶׁלֹּא שָׁמַר +[עַל]+ מָה שֶׁצִּוִּיתִיו - נִגְזֹר עָלָיו וְנִגְרְשׁוּ מִן גַּן הָעֵדֶן, קֹדֶם+ עַד לֹא יִפְשֹׁט יָדוֹ וְיִקַּח [...] מִן +פֵּרוֹת+ אִילַן הַחַיִּים, +שֶׁהֲרֵי אִם+ אֹכֵל +הוּא מִמֶּנּוּ+ - הוֹוֶה חַי +וְקַיָּם עַד+ לְעוֹלָמִים!"
וְ_גֵרְשָׁהוּ_ ה׳ אֱלֹהִים מִגִּינַּת הָעֵדֶן, +וְהָלַךְ וְיָשַׁב בְּהַר מוֹרִיָּה,+ [כְּדֵי] לַעֲבוֹד אֶת הָאֲדָמָה שֶׁ_נִּבְרָא_ מִשָּׁם.
וְגֵירֵשׁ אֶת אָדָם; +מִן+ _שֶׁהֵשִׂים_ [אֶת] +כְּבוֹד שְׁכִינָתוֹ+ מִן לַקּוֹדְמִים _בֵּין שְׁנֵי_ הַכְּרוּבִים+; קוֹדֶם שֶׁעַד־לֹא+(=עֲדַיִין לֹא)+ בָּרָא +[אֶת]+ הָעוֹלָם, בָּרָא +[אֶת]+ הַתּוֹרָה. וְהִתְקִין +[אֶת]+ גִּינַּת הָעֵדֶן לַצַּדִּיקִים, שֶׁיּוּכְלוּ וְיִתְפַּנְּקוּ מִן פֵּירוֹת הָאִילָן, עַל אֲשֶׁר עָבְדוּ בְּחַיֵּיהֶם בְּלִימּוּד הַתּוֹרָה בָּעוֹלָם הַזֶּה, וְקִיְּימוּ +[אֶת]+ הַמִּצְוֹת. הִתְקִין גְּהֵינָם לָרְשָׁעִים, שֶׁמִּשּׁוֹלֶה +לַחֶרֶב _הַשְּׁנוּנָה_+, אָכְלָה מִשְּׁנֵי צְדָדִים. הִתְקִין בְּתוֹכָהּ זִיקּוּקִים שֶׁל־אוֹר, וְגוֹמְרִים** שֶׁל־אֵשׁ, לָדוּן בָּהֶם לָרְשָׁעִים שֶׁמָּרְדוּ בְּחַיֵּיהֶם בְּלִימּוּד הַתּוֹרָה הַטּוֹבָה - הִיא הַתּוֹרָה לְעַבְּדָהּ+(=לַעֲבוֹד אוֹתָהּ)+ מִן פֵּירוֹת אִילָן הַחַיִּים, שֶׁנִּתְתַּקְּנָה בְּמַאֲמַר ה׳, +לְשׁוֹמְרָהּ+, שֶׁיִּהְיֶה עוֹמֵד וּמְטַיֵּיל בִּשְׁבִילֵי +דֶּרֶךְ [...] הַחַיִּים+, לָעוֹלָם הַבָּא+.
