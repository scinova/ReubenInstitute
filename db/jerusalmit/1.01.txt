_מִן הַתְחָלָה_ בָּרָא ה׳, אֶת הַשְׁמִים וְאֶת הָאָרֶץ.
וְהָאָרֶץ הָיְיתָה _תּוֹהָה וּבוֹהָה_, +שׁוֹמְמָה מִבְּנֵי אֱנוֹשׁ, וְרֵיקָנִית מִן כֹּל בְּהֵמָה+, וְחוֹשֶׁךְ עַל פְּנֵי הַתְּהוֹם, וְרוּחַ +רַחֲמִים מִן לִפְנֵי+ ה' _מְנַשֶּׁבֶת_ עַל פְּנֵי הַמַּיִם.
וַאֲמַר ה׳: "יְהִי אוֹר, [לְהָאִיר הָעוֹלָם|לְהָאִיר הָעֶלְיוֹן]!" - וּ+מִן יָד +הָיָה אוֹר.
וְרָאָה ה׳ אֶת הָאוֹר כִּי טוֹב, וְ_הִפְרִיד_ ה׳ בֵּין הָאוֹר וּבֵין הַחוֹשֶׁךְ.
וְקָרָא ה׳ לָאוֹר '_יְמָמָה_', +וְעָשָׂהוּ+ (=עָשָׂה אוֹתוֹ) [בִּשְׁבִיל] +לַעֲבוֹד+ (=שֶׁיַּעַבְדוּ) +בּוֹ דַּיָּירֵי הָעוֹלָם,+ וְלַחוֹשֶׁךְ קָרָא 'לַיְלָה', +וְעָשָׂהוּ+ [בִּשְׁבִיל] +לָנוּחַ+ (=שֶׁתָּנוּחַ) +בּוֹ הַבְּרִיאָה+. וְהָיָה עֶרֶב, וְהָיָה בּוֹקֶר - יוֹם אֶחָד.
וְאָמַר ה׳: "יְהִי רָקִיעַ בְּ_אֶמְצַע_ הַמַּיִם! וִיהִי _מַפְרִיד_ בֵּינֵי מֵימָיו +הָעֶלְיוֹנִים,+ לְ+בֵינֵי+ מֵימָיו +הַתַּחְתּוֹנִים+".
וְעָשָׂה ה׳ אֶת הָרָקִיעַ+, עָובְיוֹ שָׁלֹשׁ אֶצְבָּע+[וֹת]+ - בֵּינֵי צִידֵּי הַשָּׁמַיִים לְמֵימָיו שֶׁל־הָאוֹקְיָינוֹס+; וְ_הִפְרִיד_ בֵּינִי הַמַּיִם שֶׁמִּלְּמַטָּה לָרָקִיעַ, וּבֵינֵי הַמַּיִם שֶׁלְּעָל +בְּאוֹהֶל+ הָרָקִיעַ - וְהָיָה כֵּן.
וְקָרָא ה׳ לָרָקִיעַ 'שָׁמַיִם'. וְהָיָה עֶרֶב, וְהָיָה בּוֹקֶר - יוֹם שֵׁנִי.
וְאָמַר ה׳: "_יִתְכַּנְּסוּ_ מַיִם +תַּחְתּוֹנִים שֶׁנִּשְׁאֲרוּ+ מִן לְמַטָּה לַשָּׁמַיִם לְמָקוֹם אֶחָד, +וְתִתְיַיבֵּשׁ הָאָרֶץ+ [כְּדֵי] +שֶׁ+תֵּרָאֶה הַיַּבָּשָׁה!" - וְהָיָה כֵּן.
וְקָרָא ה׳ לַיַּבָּשָׁה 'אֶרֶץ', וּלְ_בֵית כִּינּוּס_ הַמַּיִם קָרָא 'יַמִּים'. וְרָאָה ה׳ כִּי טוֹב.
וְאָמַר ה׳: "_תְּגַדֵּל_ הָאָרֶץ דְּשָׁאִים, עֵשֶׂב _שֶׁזַּרְעוֹ מִזְדָּרֵעַ_, וְאִילַן פֵּירוֹת [שֶׁ]עוֹשֶׂה פֵּירוֹת לְמִינוֹ, שֶׁזַּרְעוֹ בוֹ עַל הָאָרֶץ!" - וְהָיָה כֵּן.
וְהוֹצִיאָה הָאָרֶץ דְּשָׁאִים, עֵשֶׂב _שֶׁזַּרְעוֹ מִזְדָּרֵעַ_, וְאִילָן [שֶׁ]עוֹשֶׂה פֵּירוֹת, ... לְמִינוֹ; וְרָאָה ה׳ כִּי טוֹב.
וְהָיָה עֶרֶב, וְהָיָה בּוֹקֶר - יוֹם שְׁלִישִׁי.
וְאָמַר ה׳: "יִהְיוּ אוֹרוֹת בִּרְקִיעַ הַשָּׁמַיִים, _לְהַפְרִיד_ בֵּינֵי יְמָמָה וּבֵינֵי לַיְלָה. וְיִהְיוּ לְ_סִימָנִים_ וְלִ+זְמַנֵּי+ מוֹעֲדִים, וְלִ+מְנוֹת בַּהֵם +[אֶת]+ חֶשְׁבּוֹן+ [הַ]יָּמִים+, וּלְקַדֵּשׁ רָאשֵׁי חֳודָשִׁים+ וְ+רָאשֵׁי+ שָׁנִים+, עִיבּוּרֵי חֳודָשִׁים +(=חֳודָשִׁים מְעֻובָּרִים)+ וְעִיבּוּרֵי שָׁנִים +(=שָׁנִים מְעֻובָּרוֹת)+, וּתְקוּפַת הַשֶּׁמֶשׁ וּמוֹלַד הַיָּרֵחַ, וּמַחְזוֹרִים+,
וְיִהְיוּ לְאוֹרוֹת בִּרְקִיעַ הַשָּׁמַיִם, לְהָאִיר עֳלֶיהָ הָאָרֶץ” - וְהָיָה כֵּן.
ועשה ה׳ את שני האורות הגדולים+, והיו שוים בכבודם עשרים ואחת שעות, פחות מהם שש מאות שבעים ושניים חלקי השעה, ומן אחר כן +(=ואחר מכן)+, סיפרה הלבנה עליה השמש לשון שלישית +(~לשון הרע)+, ונתמעטה +(~נהיתה יותר קטנה)+, ומינתה+ את +השמש,+ [שהוא|שהיה] האור הגדול - _לשלוט_ ביממה, ואת +לבנה,+ [שהוא|שהיה] האור הקטן - _לשלוט_ בלילה, ואת הכוכבים.
ו_סידר_ אותם ה׳ +על מסלוליהם+ ברקיע השמים, להאיר על הארץ,
ו_לשמש_ (~לשרת) ב_יממה_ ובלילה, ו_להפריד_ בין אור +היממה+ ובין חושך +הלילה+. וראה ה׳ כי טוב.
והיה ערב, והיה בוקר - יום רביעי.
ואמר ה׳: "ירחשו +ביצות+ המים רחש, נפש חיה! ועוף _שטס_ +וקינו+ על הארץ, +ושביל טיסתו+ על _אויר_ רקיע השמים!"
וברא ה׳ את התנינים הגדולים+ - את לויתן ובן זוגו, שמתעתדים +(~שמתכוננים)+ ליום הנחמה+, ואת כל נפש חיה ש_רוחשת_ ש_רחשו_ המים +הצלולים+ למיניהם+ - מינים זכים+ (~נקיים, טהורים)+, ומינים שלא זכים,+ ואת כל עוף +שטס ב+כנפיים למינו+ - מינים זכים, ומינים שלא זכים+. וראה ה׳ כי טוב.
(חסר תרגום לפסוק זה)
והיה ערב, והיה בוקר - יום חמישי.
ואמר ה: "תוציא +גרגרת+ (~גרגרי עפר, אבק) הארץ נפשות _הבריאה_ למינה+ - מינים זכים +(~נקיים, טהורים)+, ומינים שלא זכים+, בהמות ורחשים, _ובריית_ הארץ למינה!" - והיה כן.
ועשה ה׳ את חיות הארץ למינה+ - מינים זכים, ומינים שלא זכים+, ואת הבהמה למינה, ואת כל רחש הארץ למיניהו. ... .
ואמר ה׳ +למלאכים שמשמשים לפניו, שנבראו ביום שני לבריאת העולם+: "נעשה אדם בצלמנו _ב_דמותנו! ו_ישלטו_ בדגי הים, ובעוף +אויר+ השמים, ובבהמה ובכל הארץ, ובכל _רחש_ _שרוחש_ עליה הארץ".
וברא ה׳ את האדם בדמותו, בצלם ה׳ ברא אותו+, במאתיים וארבעים ושמונה איברים +(=רמ״ח איברים)+, בשלוש מאות ושישים וחמישה גידים +(=שס״ה גידים)+, וקרם עליו עור, ומילא אותו בשר ודם+; זכר ונקבה +בצורתם+ (=בדמותם) ברא אותם.
ובירך אותם ה׳, ואמר להם ה׳: "פושו ורבו, ומלאו את הארץ +בנים ובנות+, וחזקו עליה +בנכסים!+ ושלטו בדגי הים, ובעוף השמים, ובכל רחש חיה הרוחשת על הארץ!"
ואמר ה׳: “הנה הבאתי לכם את כל העשב שזרעו מזדרע שעל פני כל הארץ, ואת כל אילני +הסרק, לצורך בנין ולהסקה, ו+שבו פירות אילן שזרעו מזדרע - לכם יהיה למאכל,
ולכל חיות הארץ, ולכל עוף השמים, ולכל הרוחש על הארץ, שבו נפש חיה, את כל ירוקי עשבים, … ." - והיה כן.
וראה ה את כל שעשה, והנה טוב מאוד. והיה ערב, והיה בוקר - יום שישי.
