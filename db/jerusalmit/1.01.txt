_מִן הַתְחָלָה_ בָּרָא ה׳, אֶת הַשְׁמִים וְאֶת הָאָרֶץ.
וְהָאָרֶץ הָיְיתָה _תּוֹהָה וּבוֹהָה_, +שׁוֹמְמָה מִבְּנֵי אֱנוֹשׁ, וְרֵיקָנִית מִן כֹּל בְּהֵמָה+, וְחוֹשֶׁךְ עַל פְּנֵי הַתְּהוֹם, וְרוּחַ +רַחֲמִים מִן פְּנֵי+ ה׳ _מְנַשֶּׁבֶת_ עַל פְּנֵי הַמַּיִם.
וַאֲמַר ה׳: "יְהִי אוֹר, [לְהָאִיר הָעוֹלָם|לְהָאִיר הָעֶלְיוֹן]!" - וּ+מִן יָד +הָיָה אוֹר.
וְרָאָה ה׳ אֶת הָאוֹר כִּי טוֹב, וְ_הִפְרִיד_ ה׳ בֵּין הָאוֹר וּבֵין הַחוֹשֶׁךְ.
וְקָרָא ה׳ לָאוֹר '_יְמָמָה_', +וְעָשָׂהוּ+(=עָשָׂה אוֹתוֹ) [כְּדֵי שֶׁיּוּכְלוּ] +לַעֲבוֹד בּוֹ דַּיָּירֵי הָעוֹלָם,+ וְלַחוֹשֶׁךְ קָרָא 'לַיְלָה', +וְעָשָׂהוּ+ [כְּדֵי שֶׁתּוּכַל] +לָנוּחַ בּוֹ הַבְּרִיאָה+. וְהָיָה עֶרֶב, וְהָיָה בּוֹקֶר - יוֹם אֶחָד.
וְאָמַר ה׳: "יְהִי רָקִיעַ בְּ_אֶמְצַע_ הַמַּיִם! וִיהִי _מַפְרִיד_ בֵּינֵי מֵימָיו +הָעֶלְיוֹנִים,+ לְ+בֵינֵי+ מֵימָיו +הַתַּחְתּוֹנִים+".
וְעָשָׂה ה׳ אֶת הָרָקִיעַ+, עָובְיוֹ שָׁלֹשׁ אֶצְבָּע+[וֹת]+ - בֵּינֵי צִידֵּי הַשָּׁמַיִים לְמֵימָיו שֶׁל־הָאוֹקְיָינוֹס+; וְ_הִפְרִיד_ בֵּינִי הַמַּיִם שֶׁמִּלְּמַטָּה לָרָקִיעַ, וּבֵינֵי הַמַּיִם שֶׁלְּעָל +בְּאוֹהֶל+ הָרָקִיעַ - וְהָיָה כֵן.
וְקָרָא ה׳ לָרָקִיעַ 'שָׁמַיִם'. וְהָיָה עֶרֶב, וְהָיָה בּוֹקֶר - יוֹם שֵׁנִי.
וְאָמַר ה׳: "_יִתְכַּנְּסוּ_ מַיִם +תַּחְתּוֹנִים שֶׁנִּשְׁאֲרוּ+ מִן לְמַטָּה לַשָּׁמַיִם לְמָקוֹם אֶחָד, +וְתִתְיַיבֵּשׁ הָאָרֶץ+ [כְּדֵי] _שֶׁ_תֵּרָאֶה הַיַּבָּשָׁה!" - וְהָיָה כֵּן.
וְקָרָא ה׳ לַיַּבָּשָׁה 'אֶרֶץ', וּלְ_בֵית כִּינּוּס_ הַמַּיִם קָרָא 'יַמִּים'. וְרָאָה ה׳ כִּי טוֹב.
וְאָמַר ה׳: "_תְּגַדֵּל_ הָאָרֶץ דְּשָׁאִים, עֵשֶׂב _שֶׁזַּרְעוֹ מִזְדָּרֵעַ_, וְאִילַן פֵּירוֹת [שֶׁ]עוֹשֶׂה פֵּירוֹת לְמִינוֹ, שֶׁזַּרְעוֹ בוֹ עַל הָאָרֶץ!" - וְהָיָה כֵן.
וְהוֹצִיאָה הָאָרֶץ דְּשָׁאִים, עֵשֶׂב _שֶׁזַּרְעוֹ מִזְדָּרֵעַ_, וְאִילָן [שֶׁ]עוֹשֶׂה פֵּירוֹת, … , לְמִינוֹ; וְרָאָה ה׳ כִּי טוֹב.
וְהָיָה עֶרֶב, וְהָיָה בּוֹקֶר - יוֹם שְׁלִישִׁי.
וְאָמַר ה׳: "יִהְיוּ אוֹרוֹת בִּרְקִיעַ הַשָּׁמַיִים, _לְהַפְרִיד_ בֵּינֵי _הַיְּמָמָה_ וּבֵינֵי הַלַּיְלָה. וְיִהְיוּ לְ_סִימָנִים_ וְלִ+זְמַנֵּי+ מוֹעֲדִים, וְ+לִמְנוֹת בַּהֵם +[אֶת]+ חֶשְׁבּוֹן+ [הַ]יָּמִים+, וּלְקַדֵּשׁ רָאשֵׁי יְרָחִים+(~חוֹדָשִׁים) וְ+רָאשֵׁי+ שָׁנִים+, עִיבּוּרֵי יְרָחִים +(=חוֹדָשִׁים מְעֻובָּרִים)+ וְעִיבּוּרֵי שָׁנִים +(=שָׁנִים מְעֻובָּרוֹת)+, וּתְקוּפַת הַשֶּׁמֶשׁ וּמוֹלַד הַסַּהַר+(=הַיָּרֵחַ)+, וּמַחְזוֹרִים+,
וְיִהְיוּ לְאוֹרוֹת בִּרְקִיעַ הַשָּׁמַיִם, לְהָאִיר עֳלֶיהָ הָאָרֶץ” - וְהָיָה כֵן.
וְעָשָׂה ה׳ אֶת שְׁנֵי הָאוֹרוֹת הַגְּדוֹלִים+, וְהָיוּ שָׁוִים בִּכְבוֹדָם עֶשְׂרִים וְאַחַת שָׁעוֹת, פָּחוֹת מֵהֶם שֵׁשׁ מֵאוֹת שִׁבְעִים וּשְׁנַיִים חֶלְקֵי הַשָּׁעָה. וּמִן אַחַר כֵּן +(=וְאַחַר מִכֵּן)+, סִיפֵּר הַסַּהַר עֲלִיָּה הַשֶּׁמֶשׁ לָשׁוֹן שְׁלִישִׁית +(~לְשׁוֹן הָרַע)+ - וְהִתְקַטֵּן+(~נִהְיָה קָטָן)+. וּמִינָּה+ אֶת +הַשֶּׁמֶשׁ,+ [שֶׁהוּא|שֶׁהָיָה] הָאוֹר הַגָּדוֹל - _לִשְׁלוֹט בַּיְּמָמָה_, וְאֶת +הַסַּהַר,+ [שֶׁהוּא|שֶׁהָיָה] הָאוֹר הַקָּטָן - _לִשְׁלוֹט_ בְּלִילָה, וְאֶת הַכּוֹכָבִים.
וְ_סִדֵּר_ אוֹתָם ה׳ +עַל מַסְלוּלֵיהֶם+ בִּרְקִיעַ הַשָּׁמַיִם, לְהָאִיר עַל הָאָרֶץ,
וּ_לְשַׁמֵּשׁ_(~לְשָׁרֵת) בַּ_יְּמָמָה_ וּבַלַּיְלָה, וּ_לְהַפְרִיד_ בֵּין אוֹר +הַיְּמָמָה+ וּבֵין חוֹשֶׁךְ +הַלַּיְלָה+. וְרָאָה ה׳ כִּי טוֹב.
וְהָיָה עֶרֶב, וְהָיָה בּוֹקֶר - יוֹם רְבִיעִי.
וְאָמַר ה׳: "_יִרְחֲשׁוּ_ +בִּצּוֹת+ הַמַּיִם _רוֹחֵשׁ_, נֶפֶשׁ חַיָּה! וְעוֹף _שֶׁטָּס_ +וְ+[שֶׁ]+קִינּוֹ+(=הַקֵּן שֶׁלּוֹ) עַל הָאָרֶץ, +וּשְׁבִיל טִיסָתוֹ+ עַל _אֲוִיר_ רְקִיעַ הַשָּׂמִים!"
וּבָרָא ה׳ אֶת הַתַּנִּינִים הַגְּדוֹלִים+ - אֶת לִוְיָתָן וּבֶן זוּגוֹ, שֶׁמִּתְעַתְּדִים+(~שֶׁמִּתְכּוֹנְנִים)+ לְיוֹם הַנֶּחָמָה+; וְאֶת כֹּל נֶפֶשׁ חַיָּה שֶׁ_רוֹחֶשֶׁת_ שֶׁ_רִחֲשׁוּ_ הַמַּיִם +הַצְּלוּלִים+, לְמִינֵיהֶם+ - מִינִים זַכִּים+(~נְקִיִּים, טְהוֹרִים)+, וּמִינִים שֶׁלֹּא זַכִּים,+ וְאֶת כֹּל עוֹף +שֶׁטַּס בִּ+כְנָפִיִּים לְמִינוֹ+ - מִינִים זַכִּים, וּמִינִים שֶׁלֹּא זַכִּים+. וְרָאָה ה׳ כִּי טוֹב.
[חָסֵר תַּרְגּוּם לְפָסוּק זֶה]
וְהָיָה עֶרֶב, וְהָיָה בּוֹקֶר - יוֹם חֲמִישִׁי.
וְאָמַר ה׳: "יוֹצִיא +חֵימָר+ הָאָרֶץ נַפְשׁוֹת _הַבְּרִיאָה_ לְמִינָהּ+ - מִינִים זַכִּים+(~נְקִיִּים, טְהוֹרִים)+, וּמִינִים שֶׁלֹּא זַכִּים+, בְּהֵמוֹת וְ_רוֹחֲשִׁים_, _וּבִרְיַית_ הָאָרֶץ לְמִינָהּ!" - וְהָיָה כֵּן.
וְעָשָׂה ה׳ אֶת חַיּוֹת הָאָרֶץ לְמִינָהּ+ - מִינִים זַכִּים, וּמִינִים שֶׁלֹּא זַכִּים+, וְאֶת הַבְּהֵמָה לְמִינָהּ, וְאֶת כֹּל _רוֹחֵשׁ_ הָאָרֶץ לְמִינֵיהוּ. … .
וְאָמַר ה׳ +לַמַּלְאָכִים שֶׁמְּשַׁמְּשִׁים לְפָנָיו, שֶׁנִּבְרְאוּ בְּיוֹם שֵׁנִי לִבְרִיאַת הָעוֹלָם+: "נַעֲשָׂה אָדָם בְּצַלְמֵנוּ _בִּ_דְמוּתֵנוּ! וְ_יִשְׁלְטוּ_ בִּדְגֵי הַיָּם, וּבְעוֹף +אֲוִיר+ הַשָּׁמַיִם, וּבַבְּהֵמָה וּבְכֹל הָאָרֶץ, וּבְכֹל _רוֹחֵשׁ_ _שֶׁרוֹחֵשׁ_ עָלֶיהָ הָאָרֶץ".
וּבָרָא ה׳ אֶת הָאָדָם בִּ_דְיוֹקָנוֹ_(=בִּדְמוּתוֹ), בְּצֶלֶם ה׳ בָּרָא אוֹתוֹ+, בְּמָאתַיִים וְאַרְבָּעִים וּשְׁמוֹנָה אֵיבָרִים +(=רְמַ"ח אֵיבָרִים)+, בִּשְׁלוֹשׁ מֵאוֹת וְשִׁישִּׁים וַחֲמִישָּׁה גִּידִים +(=שְׁסָ"ה גִּידִים)+, וְקָרַם עָלָיו עוֹר, וּמִילֵּא אוֹתוֹ בָּשָׂר וָדָם+; זָכָר וּנְקֵבָה +בְּצוּרָתָם+ בָּרָא אוֹתָם.
וּבֵרֵךְ אוֹתָם ה׳, וְאָמַר לָהֶם ה׳: "_פְּשׁוּ_ וּרְבוּ, וּמִלְאוּ אֶת הָאָרֶץ +בָּנִים וּבָנוֹת+, וְ_חָזְקוּ_ עָלֶיהָ +בִּנְכָסִים!+ וְשִׁלְּטוּ בִּדְגֵי הַיָּם, וּבְעוֹף הַשָּׁמַיִם, וּבַכֹּל רַחַשׁ חַיָּה הָ_רוֹחֶשֶׁת_ עַל הָאָרֶץ!"
וְאָמַר ה׳: "הִנֵּה הֵבֵאתִי לָכֶם אֶת כֹּל הָעֵשֶׂב שֶׁזַּרְעוֹ מִזְדָּרֵעַ שֶׁעַל פְּנֵי כֹּל הָאָרֶץ, וְאֶת כֹּל אִילָנֵי +הַסְּרָק, לְצוֹרֶךְ בִּנְיָן וּלְהַסָּקָה, וְ+שֶׁבּוֹ פֵּירוֹת אִילָן שֶׁזַּרְעוֹ מִזְדָּרֵעַ - לָכֶם יִהְיֶה לְמַאֲכָל,
וּלְכֹל חַיּוֹת הָאָרֶץ, וּלְכֹל עוֹף הַשָּׁמַיִם, וּלְכֹל שֶׁרוֹחֵשׁ עַל הָאָרֶץ, שֶׁבּוֹ נֶפֶשׁ חַיָּה, אֶת כָּל יְרוֹקֵי עֲשָׂבִים, … ." - וְהָיָה כֵן.
וְרָאָה ה׳ אֶת כֹּל שֶׁעָשָׂה, וְ_הָיָה_ טוֹב מְאוֹד. וְהָיָה עֶרֶב, וְהָיָה בּוֹקֶר - יוֹם שִׁישִּׁי.
