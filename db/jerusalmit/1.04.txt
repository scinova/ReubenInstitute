וְאָדָם יָדַע אֶת חַוָּה אִשְׁתּוֹ+, שֶׁהָיְיתָה חֲמוּדָה לַמַּלְאָךְ+ - וְהִתְעַבְּרָה וְיָלְדָה אֶת קַיִן, וְאָמְרָה: 'קָנִיתִי +לְ+אִישׁ אֶת מַלְאַךְ ה׳'.
וְהוֹסִיפָה לָלֶדֶת +מִן בַּעְלָהּ אָדָם אֶת תֵּאוּמוֹ, וְ+אֶת הֶבֶל, וְהָיָה הֶבֶל רוֹעֶה צֹאן, וְקַיִן הָיָה +גֶּבֶר+ עוֹבֵד בָּאָרֶץ.
וְהָיָה מִסּוֹף הַיָּמִים+, וּבְאַרְבָּעָה עָשָׂר בְּנִיסָן+; וְהֵבִיא קַיִן מִפְּרִי הָאָרֶץ+, מִזְדָּרֵעַ כּוּתְנָה+, קוֹרְבָּן +מִבִּיכּוּרִים לִפְנֵי+ ה׳.
וְהֶבֶל הֵבִיא אַף הוּא מִבְּכוֹרוֹת צֹאנוֹ וּמֵחֶלְבֵיהֶם, וְ_הָיָה רָצוֹן_ +לִפְנַי+ ה׳	 +וְסֵבֶר פָּנִים+ בְּהֶבֶל וּ_בְקוֹרְבָּנוֹ_.
וּלְקַיִן וּלְ_קוֹרְבָּנוֹ_ לָא _סָבַר פָּנִים_ - וְ_חָזָק_ לְקַיִן מְאוֹד, וְהֶחֱמִיצוּ +הַדְּמוּיוֹת שֶׁל־+פָּנָיו.
וְאָמַר ה׳ לְקַיִן: "לָמָּה חָזָק לְךָ, וְלָמָה _נִכְבְּשׁוּ_ +הַדְּמוּיוֹת שֶׁל־+פָּנֶיךָ?
הֲלֹא אִם לֹא תֵּיטִיב [אֶת] _מַעֲשֶׂיךָ_ +- יֵעָזֵב לָךְ חֶטְאֲךָ+, וְאִם לֹא תֵּיטִיב [אֶת] _מַעֲשֶׂיךָ_ +בָּעוֹלָם הַזֶּה לְיוֹם הַדִּין הַגָּדוֹל+ - +חֶטְאֲךָ שׁוֹמֵר,+ וְעַל שַׁעֲרֵי +לִבְּךָ+ חֶטְאֲךָ רְבִיעַ, +וּבְיָדְךָ מָסַרְתִּי +[אֶת]+ רְשׁוּתוֹ שֶׁל יֵצֶר הָרַע,+ וְאֵלֶיךָ +תִּהְיֶה+ תַּאֲוָותוֹ, וְאַתָּה תִּהְיֶה שׁוֹלֵט בּוֹ+, בֵּין לְזַכּוֹת+(~לְטַהֵר)+, בֵּין לַחֲטוֹא+.
וְאָמַר קַיִן אֶל הֶבֶל אָחִיו+: "אָבוֹא וְנֵלֵךְ שְׁנֵינוּ לַחוּץ!"+, וְהָיָה כַּאֲשֶׁר _יָצְאוּ שְׁנֵיהֶם_ לַחוּץ, +עָנָה קַיִן וְאָמַר לְהֶבֶל: "מִסְתַּכֵּל אֲנִי, שֶׁבְּרַחֲמִים נִבְרָא הָעוֹלָם, אֲבָל לֹא בְּפֵירוֹת מַעֲשִׂים טוֹבִים הוּא מְדַבֵּר וּמֵשִׁיב פָּנִים. יֵשׁ בַּדִּין מִן בִּגְלַל מָה נִתְקַבֵּל קוֹרְבָּנְךָ, וְקוֹרְבָּנִי מִמֶּנִּי לֹא נִתְקַבֵּל בְּרָצוֹן?" עָנָה הֶבֶל וְאָמַר לְקַיִן: "בְּרַחֲמִים נִבְרָא הָעוֹלָם, וּבְפֵירוֹת מַעֲשִׂים טוֹבִים הוּא מַדְבָּר וְלוֹקֵחַ פָּנִים. אֵין בַּדִּין, וְעַל שֶׁהָיוּ פֵּירוֹת מַעֲשַׂי טוֹבִים מִשֶּׁלְּךָ, וְקוֹדְמִים לְשֶׁלְּךָ, נִתְקַבֵּל בְּרָצוֹן קוֹרְבָּנַי". עָנָה קַיִן, וְאָמַר לְהֶבֶל: "אֵין דִּין, וְאֵין דַּיָּין, וְאֵין עוֹלָם אַחֵר, וְאֵין לָתֵת שָׂכָר טוֹב לַצַּדִּיקִים, וְאֵין לִפְרוֹעַ מִן הָרְשָׁעִים". עָנָה הֶבֶל, וְאָמַר לְקַיִן: "יֵשׁ דִּין, וְיֵשׁ דַּיָּין, וְיֵשׁ עוֹלָם אַחֵר, וְיֵשׁ לָתֵת שָׂכָר טוֹב לַצַּדִּיקִים, וְיֵשׁ לִפְרוֹעַ מִן הָרְשָׁעִים. וְעַל עִיסֵּק הַמִּשְׁפָּטִים הָאֵלֶּה הָיוּ מִתְנַצִּים עַל פְּנֵי הַחוּץ,+ וְקָם קַיִן עַל הֶבֶל אָחִיו, +וְטָבַע אֶבֶן בְּמִצְחוֹ,+ וְהָרְגוּ.
ואמר ה לקין: "היכן הבל אחיך?" ואמר: "לא יודעני, שמא שומר אחי אני?!"
ואמר: "מה עשית?? קול דמי הריגת אחיך שנתבלעו בעפר צועקים לפני מן הארץ!
וכעת, בעבור שהרגתו, ארור אתה, מן הארץ שפתחה את פיה, וקיבלה את דמי אחיך מן ידך.
כי תעבוד בארץ, לא תוסיף לתת כוח פירותיה לך. מטלטל וגולה תהיה בארץ".
וְאָמַר קַיִן _לִפְנֵי_ ה׳: "רַב+, חָזָק+ מְרוּדִי מִלִּסְבּוֹל+, אֲבָל יוּכַל קְדׇמְךָ לִסְבּוֹל אוֹתוֹ+",
הֵן גֵּירַשְׁתָּ אוֹתִי הַיּוֹם הַזֶּה מֵעַל פְּנֵי הָאָרֶץ, מִן פָּנֶיךָ _הַאֶפְשָׁר שֶׁנֶּחְבָּא_?! וְאִם אֶהְיֶה מִטַּלְטֵל וְגוּלֵּה בָּאָרֶץ, כֹּל +זַךְ+(~נָקִי,טָהוֹר) שֶׁיִּמְצָאֵנִי - יַהַרְגֵנִי.
וְאָמַר לוֹ ה׳: "הֵן בְּכֵן כֹּל שֶׁהוֹרֵג קַיִן - לְשִׁבְעָה +דּוֹרוֹת+ _יִפָּרַע_ +מִמֶּנּוּ+!" וְרָשַׁם ה׳ עַל הַפָּנִים שֶׁל־קַיִן אוֹת +מִן הַשֵּׁם הַגָּדוֹל וְהַמְּכוּבָּד+, בְּגִין שֶׁלֹּא לַהֲרוֹג אוֹתוֹ כֹּל שֶׁיִּמְצָאֵהוּ +בְּהִסְתַּכְּלוּתוֹ בּוֹ+.
וְיָצָא קַיִן מִלִּפְנֵי ה׳, וְיָשַׁב בְּאֶרֶץ _טִלְטוּל גָּלוּתוֹ_, +שֶׁהָיְיתָה עֲשׂוּיָה עָלָיו+ מִלִּקְדָמִים +כַּגִּינָּה+ שֶׁל־עֵדֶן.
וְיָדַע קַיִן אֶת אִשְׁתּוֹ, וְהִתְעַבְּרָה, וְיָלְדָה אֶת חֲנוֹךְ; וְהָיָה בּוֹנֶה קִרְיָיה, וְקָרָא שֵׁם הַקִּרְיָיה כְּשֵׁם בְּנוֹ חֲנוֹךְ.
וְנוֹלַד לַחֲנוֹךְ אֶת עִירָד, וְעִירָד הוֹלִיד אֶת מְחוּיָאֵל, וּמְחוּיָאֵל הוֹלִיד אֶת מְתוּשָׁאֵל, וּמְתוּשָׁאֵל הוֹלִיד אֶת לָמֶךְ.
וְלָקַח לוֹ לֶמֶךְ שְׁתֵּי נָשִׁים: שֶׁם הָאַחַת 'עָדָה', וָשֶׁם הַשְּׁנִיָּה 'צִלָּהּ'.
וְיַלְדָּה עָדָה אֶת יָבָל - הוּא הָיָה _רַב בָּהֶם_ שֶׁל־כֹּל יוֹשְׁבֵי מִשְׁכָּנִים ו_בַּעֲלֵי בְּהֵמָה_.
וְשֶׁם אָחִיו 'יוֹבֵל' - הוּא הָיָה _רַב בַּהֵם_ שֶׁל־כֹּל +שֶׁמְּמוּנִּים+ _לְזַמֵּר_ בַּכִּנּוֹר וּבֶחָלִיל.
וְצִלָּה אַף הִיא, יָלְדָה אֶת תּוּבַל קַיִן - +רַב לְכֹל+ _אוֹמָן שֶׁיָּדַע בַּעֲשִׂייַּת_ נְחֹשֶׁת וּבַרְזֶל; וַאֲחוֹתוֹ שֶׁל־תּוּבַל קַיִן - נַעֲמָה+ - הִיא הָיְתָה בַּעֲלַת קִינוֹת וּזְמָרִים+.
וְאָמַר לֶמֶךְ לִנְשׁוֹתָיו, עָדָה וְצִלָּה: “_קַבֵּלְנָה_ [אֶת] קוֹלִי, נְשׁוֹת לֶמֶךְ, הַקְשֵׁבְנָה לְמַאֲמָרִי! כִּי +לֹא+ אִישׁ קָטַלְתִּי, +שֶׁנִּקְטַל תַּחְתָּיו, אַף לֹא+ עָלַם חָבַלְתִּי+, שֶׁבִּגְלָלוֹ יֹאבְדוּ זְרָעַי+.
כִּי קַיִן+, שֶׁחָטָא, וְשָׁב בְּתַאֲוָותוֹ, עַד+ שִׁבְעָה +דּוֹרוֹת הִיתְּלָה לוֹ+, וְלֶמֶךְ +בֶּן בְּנוֹ, שֶׁלֹּא חָטָא +-+ הַדִּין הוּא שֶׁיִּתְלֶה לוֹ עַד+ שִׁבְעִים וְשִׁבְעָה.
וְיָדַע אָדָם _שׁוּב_ אֶת אִשְׁתּוֹ+, לְסוֹף מֵאָה וּשְׁלוֹשִׁים שְׁנַיִם שֶׁנֶּהֱרַג הֶבֶל+, וְיָלְדָה בֵּן, וְקָרְאָה אֶת שְׁמוֹ 'שָׁת', הֵן +אָמְרָה:+ "_נָתַן_ לִי ה' _בֶּן_ אַחֵר, _חָלַף_(~תְּמוּרַת,בִּמְקוֹם) הֶבֶל, _שֶׁ_הָרְגוּ קַיִן".
וּלְשֵׁת אַף הוּא נוֹלָד בֵּן, וְקָרָא אֶת שְׁמוֹ 'אֱנוֹשׁ', +הוּא הַדּוֹר שֶׁבְּיָמָיו הֵחֵלּוּ לִטְעוֹת, וְעָשׂוּ לָהֶם אֱלִילִים, וּמְכַנִּים לֶאֱלִילֵיהֶם+ בְּשֵׁם +מַאֲמַר+ ה׳.
